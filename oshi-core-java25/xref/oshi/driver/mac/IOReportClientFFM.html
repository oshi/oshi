20  /**
21   * FFM equivalent of {@link IOReportClient}: manages a single IOReport subscription for GPU Stats and Energy Model
22   * channels, providing per-instance sampling of GPU active ticks, utilization, and power draw.
23   *
24   * <p>
25   * Returns sentinel values ({@code (0,0)} / {@code -1.0}) when IOReport is unavailable.
26   *
27   * <p>
28   * Call {@link #close()} when done to release all CoreFoundation references.
29   */
30  public final class IOReportClientFFM {
31  
32      private static final String GROUP_GPU_STATS = "GPU Stats";
33      private static final String GROUP_ENERGY = "Energy Model";
34      private static final String CHANNEL_GPU_ENERGY = "GPU Energy";
35      private static final String SUBGROUP_GPU_PERF_STATES = "GPU Performance States";
36      private static final String STATE_OFF = "OFF";
37      private static final String KEY_CHANNELS = "IOReportChannels";
38  
39      private final MemorySegment subscription;
40      private final MemorySegment subscribedChannels;
41  
42      private MemorySegment prevSampleUtil;
43      private MemorySegment prevSamplePower;
44      private long prevSamplePowerNanos;
45  
46      private boolean closed;
47  
48      private IOReportClientFFM(MemorySegment subscription, MemorySegment subscribedChannels) {
49          this.subscription = subscription;
50          this.subscribedChannels = subscribedChannels;
51      }
52  
53      /**
54       * Creates a new {@code IOReportClientFFM} subscribed to GPU Stats and Energy Model channels.
55       *
56       * @return a new client, or {@code null} if IOReport is unavailable or subscription fails
57       */
58      public static IOReportClientFFM create() {
59          if (!IOReportFunctions.isAvailable()) {
60              return null;
61          }
62          CFStringRef gpuGroup = CFStringRef.createCFString(GROUP_GPU_STATS);
63          CFStringRef energyGroup = CFStringRef.createCFString(GROUP_ENERGY);
64          MemorySegment gpuChannels = MemorySegment.NULL;
65          MemorySegment energyChannels = MemorySegment.NULL;
66          try {
67              gpuChannels = IOReportFunctions.IOReportCopyChannelsInGroup(gpuGroup.segment(), MemorySegment.NULL, 0, 0,
68                      0);
69              if (gpuChannels.equals(MemorySegment.NULL)) {
70                  return null;
71              }
72              energyChannels = IOReportFunctions.IOReportCopyChannelsInGroup(energyGroup.segment(), MemorySegment.NULL, 0,
73                      0, 0);
74              if (!energyChannels.equals(MemorySegment.NULL)) {
75                  IOReportFunctions.IOReportMergeChannels(gpuChannels, energyChannels, MemorySegment.NULL);
76              }
77              try (Arena arena = Arena.ofConfined()) {
78                  MemorySegment subRefOut = arena.allocate(ValueLayout.ADDRESS);
79                  MemorySegment sub = IOReportFunctions.IOReportCreateSubscription(MemorySegment.NULL, gpuChannels,
80                          subRefOut, 0, MemorySegment.NULL);
81                  if (sub.equals(MemorySegment.NULL)) {
82                      return null;
83                  }
84                  MemorySegment subPtr = subRefOut.get(ValueLayout.ADDRESS, 0);
85                  if (subPtr.equals(MemorySegment.NULL)) {
86                      cfRelease(sub);
87                      return null;
88                  }
89                  return new IOReportClientFFM(sub, subPtr);
90              }
91          } catch (Throwable e) {
92              return null;
93          } finally {
94              gpuGroup.release();
95              energyGroup.release();
96              if (!gpuChannels.equals(MemorySegment.NULL)) {
97                  cfRelease(gpuChannels);
98              }
99              if (!energyChannels.equals(MemorySegment.NULL)) {
100                 cfRelease(energyChannels);
101             }
102         }
103     }

105     /**
106      * Returns a {@link GpuTicks} snapshot of cumulative GPU active and idle ticks.
107      *
108      * @return GpuTicks snapshot; never null
109      */
110     public synchronized GpuTicks sampleGpuTicks() {
111         if (closed) {
112             return new GpuTicks(0L, 0L);
113         }
114         MemorySegment sample = MemorySegment.NULL;
115         try {
116             sample = IOReportFunctions.IOReportCreateSamples(subscription, subscribedChannels, MemorySegment.NULL);
117             if (sample.equals(MemorySegment.NULL)) {
118                 return new GpuTicks(0L, 0L);
119             }
120             Map<String, Long> states = extractChannelStates(sample, GROUP_GPU_STATS, SUBGROUP_GPU_PERF_STATES);
121             if (states.isEmpty()) {
122                 return new GpuTicks(0L, 0L);
123             }
124             long idle = states.getOrDefault(STATE_OFF, 0L);
125             long total = states.values().stream().mapToLong(Long::longValue).sum();
126             return new GpuTicks(total - idle, idle);
127         } catch (Throwable e) {
128             return new GpuTicks(0L, 0L);
129         } finally {
130             if (!sample.equals(MemorySegment.NULL)) {
131                 cfRelease(sample);
132             }
133         }
134     }

136     /**
137      * Returns instantaneous GPU utilization as a percentage (0–100), or {@code -1.0} if unavailable or closed.
138      *
139      * @return GPU utilization percentage, or -1.0
140      */
141     public synchronized double sampleGpuUtilization() {
142         if (closed) {
143             return -1d;
144         }
145         MemorySegment sample = MemorySegment.NULL;
146         try {
147             sample = IOReportFunctions.IOReportCreateSamples(subscription, subscribedChannels, MemorySegment.NULL);
148             if (sample.equals(MemorySegment.NULL)) {
149                 return -1d;
150             }
151             if (prevSampleUtil == null) {
152                 prevSampleUtil = sample;
153                 sample = MemorySegment.NULL;
154                 return -1d;
155             }
156             MemorySegment delta = IOReportFunctions.IOReportCreateSamplesDelta(prevSampleUtil, sample,
157                     MemorySegment.NULL);
158             cfRelease(prevSampleUtil);
159             prevSampleUtil = sample;
160             sample = MemorySegment.NULL;
161             if (delta.equals(MemorySegment.NULL)) {
162                 return -1d;
163             }
164             try {
165                 Map<String, Long> states = extractChannelStates(delta, GROUP_GPU_STATS, SUBGROUP_GPU_PERF_STATES);
166                 if (states.isEmpty()) {
167                     return -1d;
168                 }
169                 long off = states.getOrDefault(STATE_OFF, 0L);
170                 long total = states.values().stream().mapToLong(Long::longValue).sum();
171                 return total > 0 ? (total - off) * 100.0 / total : -1d;
172             } finally {
173                 cfRelease(delta);
174             }
175         } catch (Throwable e) {
176             return -1d;
177         } finally {
178             if (sample != null && !sample.equals(MemorySegment.NULL)) {
179                 cfRelease(sample);
180             }
181         }
182     }

184     /**
185      * Returns instantaneous GPU power in watts, or {@code -1.0} if unavailable or closed.
186      *
187      * @return GPU power in watts, or -1.0
188      */
189     public synchronized double samplePowerWatts() {
190         if (closed) {
191             return -1d;
192         }
193         long beforeNanos = prevSamplePowerNanos;
194         MemorySegment sample = MemorySegment.NULL;
195         try {
196             sample = IOReportFunctions.IOReportCreateSamples(subscription, subscribedChannels, MemorySegment.NULL);
197             if (sample.equals(MemorySegment.NULL)) {
198                 return -1d;
199             }
200             if (prevSamplePower == null) {
201                 prevSamplePower = sample;
202                 prevSamplePowerNanos = System.nanoTime();
203                 sample = MemorySegment.NULL;
204                 return -1d;
205             }
206             long nowNanos = System.nanoTime();
207             MemorySegment delta = IOReportFunctions.IOReportCreateSamplesDelta(prevSamplePower, sample,
208                     MemorySegment.NULL);
209             cfRelease(prevSamplePower);
210             prevSamplePower = sample;
211             prevSamplePowerNanos = nowNanos;
212             sample = MemorySegment.NULL;
213             if (delta.equals(MemorySegment.NULL)) {
214                 return -1d;
215             }
216             try {
217                 long dtNanos = nowNanos - beforeNanos;
218                 if (dtNanos <= 0) {
219                     return -1d;
220                 }
221                 long energyUj = extractGpuEnergyMicrojoules(delta);
222                 if (energyUj < 0) {
223                     return -1d;
224                 }
225                 return energyUj * 1000.0 / dtNanos;
226             } finally {
227                 cfRelease(delta);
228             }
229         } catch (Throwable e) {
230             return -1d;
231         } finally {
232             if (sample != null && !sample.equals(MemorySegment.NULL)) {
233                 cfRelease(sample);
234             }
235         }
236     }

238     /**
239      * Releases all CoreFoundation references held by this client. Idempotent.
240      */
241     public synchronized void close() {
242         if (closed) {
243             return;
244         }
245         closed = true;
246         if (prevSampleUtil != null) {
247             cfRelease(prevSampleUtil);
248             prevSampleUtil = null;
249         }
250         if (prevSamplePower != null) {
251             cfRelease(prevSamplePower);
252             prevSamplePower = null;
253         }
254         cfRelease(subscribedChannels);
255         cfRelease(subscription);
256     }

258     private long extractGpuEnergyMicrojoules(MemorySegment delta) throws Throwable {
259         CFStringRef channelsKey = CFStringRef.createCFString(KEY_CHANNELS);
260         try {
261             MemorySegment arrSeg = new CFDictionaryRef(delta).getValue(channelsKey);
262             if (arrSeg.equals(MemorySegment.NULL)) {
263                 return -1L;
264             }
265             CFArrayRef arr = new CFArrayRef(arrSeg);
266             int count = arr.getCount();
267             for (int i = 0; i < count; i++) {
268                 MemorySegment entrySeg = arr.getValueAtIndex(i);
269                 if (entrySeg.equals(MemorySegment.NULL)) {
270                     continue;
271                 }
272                 MemorySegment groupSeg = IOReportFunctions.IOReportChannelGetGroup(entrySeg);
273                 if (groupSeg.equals(MemorySegment.NULL)
274                         || !GROUP_ENERGY.equals(new CFStringRef(groupSeg).stringValue())) {
275                     continue;
276                 }
277                 MemorySegment nameSeg = IOReportFunctions.IOReportChannelGetChannelName(entrySeg);
278                 if (nameSeg.equals(MemorySegment.NULL)
279                         || !CHANNEL_GPU_ENERGY.equals(new CFStringRef(nameSeg).stringValue())) {
280                     continue;
281                 }
282                 return IOReportFunctions.IOReportSimpleGetIntegerValue(entrySeg, 0);
283             }
284         } finally {
285             channelsKey.release();
286         }
287         return -1L;
288     }

290     private Map<String, Long> extractChannelStates(MemorySegment dict, String group, String subgroup) throws Throwable {
291         CFStringRef channelsKey = CFStringRef.createCFString(KEY_CHANNELS);
292         try {
293             MemorySegment arrSeg = new CFDictionaryRef(dict).getValue(channelsKey);
294             if (arrSeg.equals(MemorySegment.NULL)) {
295                 return Collections.emptyMap();
296             }
297             CFArrayRef arr = new CFArrayRef(arrSeg);
298             int count = arr.getCount();
299             Map<String, Long> result = new HashMap<>();
300             for (int i = 0; i < count; i++) {
301                 MemorySegment entrySeg = arr.getValueAtIndex(i);
302                 if (entrySeg.equals(MemorySegment.NULL)) {
303                     continue;
304                 }
305                 MemorySegment groupSeg = IOReportFunctions.IOReportChannelGetGroup(entrySeg);
306                 if (groupSeg.equals(MemorySegment.NULL) || !group.equals(new CFStringRef(groupSeg).stringValue())) {
307                     continue;
308                 }
309                 if (subgroup != null) {
310                     MemorySegment subSeg = IOReportFunctions.IOReportChannelGetSubGroup(entrySeg);
311                     if (subSeg.equals(MemorySegment.NULL) || !subgroup.equals(new CFStringRef(subSeg).stringValue())) {
312                         continue;
313                     }
314                 }
315                 int stateCount = IOReportFunctions.IOReportStateGetCount(entrySeg);
316                 for (int s = 0; s < stateCount; s++) {
317                     MemorySegment nameSeg = IOReportFunctions.IOReportStateGetNameForIndex(entrySeg, s);
318                     if (nameSeg.equals(MemorySegment.NULL)) {
319                         continue;
320                     }
321                     String stateName = new CFStringRef(nameSeg).stringValue();
322                     long ticks = IOReportFunctions.IOReportStateGetResidency(entrySeg, s);
323                     if (!stateName.isEmpty()) {
324                         result.merge(stateName, ticks, Long::sum);
325                     }
326                 }
327             }
328             return result;
329         } finally {
330             channelsKey.release();
331         }
332     }

334     private static void cfRelease(MemorySegment seg) {
335         if (seg != null && !seg.equals(MemorySegment.NULL)) {
336             try {
337                 oshi.ffm.mac.CoreFoundationFunctions.CFRelease(seg);
338             } catch (Throwable ignored) {
339                 // CFRelease declares throws Throwable; swallow in cleanup path
340             }
341         }
342     }
