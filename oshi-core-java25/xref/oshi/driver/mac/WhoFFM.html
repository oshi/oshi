35  /**
36   * Utility to query logged-in users via {@code getutxent}, using FFM (no JNA dependency).
37   */
38  @ThreadSafe
39  public final class WhoFFM {
40  
41      private WhoFFM() {
42      }
43  
44      /**
45       * Query {@code getutxent} to get logged-in users.
46       *
47       * @return A list of logged-in user sessions
48       */
49      public static synchronized List<OSSession> queryUtxent() {
50          List<OSSession> whoList = new ArrayList<>();
51          try (Arena arena = Arena.ofConfined()) {
52              setutxent();
53              try {
54                  MemorySegment utPtr;
55                  while ((utPtr = getutxent()) != null) {
56                      MemorySegment ut = utPtr.reinterpret(UTMPX.byteSize(), arena, null);
57                      short type = ut.get(JAVA_SHORT, UTMPX.byteOffset(UT_TYPE));
58                      if (type == USER_PROCESS || type == LOGIN_PROCESS) {
59                          String user = ut.asSlice(UTMPX.byteOffset(UT_USER), UTX_USERSIZE).getString(0);
60                          String device = ut.asSlice(UTMPX.byteOffset(UT_LINE), UTX_LINESIZE).getString(0);
61                          String host = ut.asSlice(UTMPX.byteOffset(UT_HOST), UTX_HOSTSIZE).getString(0);
62                          long tvSec = ut.get(JAVA_LONG, UTMPX.byteOffset(UT_TV_SEC));
63                          int tvUsec = ut.get(JAVA_INT, UTMPX.byteOffset(UT_TV_USEC));
64                          long loginTime = tvSec * 1000L + tvUsec / 1000L;
65                          if (!isSessionValid(user, device, loginTime)) {
66                              return oshi.driver.unix.Who.queryWho();
67                          }
68                          whoList.add(new OSSession(user, device, loginTime, host));
69                      }
70                  }
71              } finally {
72                  endutxent();
73              }
74          } catch (Throwable e) {
75              return oshi.driver.unix.Who.queryWho();
76          }
77          return whoList;
78      }
79  }
