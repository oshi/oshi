5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.driver.common.windows.wmi.Win32Bios;
11  import oshi.driver.common.windows.wmi.Win32Bios.BiosProperty;
12  import oshi.driver.common.windows.wmi.Win32Bios.BiosSerialProperty;
13  import oshi.util.platform.windows.WbemcliUtilFFM.WmiQuery;
14  import oshi.util.platform.windows.WbemcliUtilFFM.WmiResult;
15  import oshi.util.platform.windows.WmiQueryHandlerFFM;
16  
17  /**
18   * Utility to query WMI class {@code Win32_BIOS} using FFM.
19   */
20  @ThreadSafe
21  public final class Win32BiosFFM extends Win32Bios {
22  
23      private Win32BiosFFM() {
24      }
25  
26      /**
27       * Queries the BIOS serial number.
28       *
29       * @return Assigned serial number of the software element.
30       */
31      public static WmiResult<BiosSerialProperty> querySerialNumber() {
32          WmiQuery<BiosSerialProperty> serialNumQuery = new WmiQuery<>(WIN32_BIOS_WHERE_PRIMARY_BIOS_TRUE,
33                  BiosSerialProperty.class);
34          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
35                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(serialNumQuery);
36      }
37  
38      /**
39       * Queries the BIOS description.
40       *
41       * @return BIOS name, description, and related fields.
42       */
43      public static WmiResult<BiosProperty> queryBiosInfo() {
44          WmiQuery<BiosProperty> biosQuery = new WmiQuery<>(WIN32_BIOS_WHERE_PRIMARY_BIOS_TRUE, BiosProperty.class);
45          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
46                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(biosQuery);
47      }
48  }
