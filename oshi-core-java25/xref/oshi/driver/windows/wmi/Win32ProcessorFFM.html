5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.driver.common.windows.wmi.Win32Processor;
11  import oshi.driver.common.windows.wmi.Win32Processor.BitnessProperty;
12  import oshi.driver.common.windows.wmi.Win32Processor.ProcessorIdProperty;
13  import oshi.driver.common.windows.wmi.Win32Processor.VoltProperty;
14  import oshi.util.platform.windows.WbemcliUtilFFM.WmiQuery;
15  import oshi.util.platform.windows.WbemcliUtilFFM.WmiResult;
16  import oshi.util.platform.windows.WmiQueryHandlerFFM;
17  
18  /**
19   * Utility to query WMI class {@code Win32_Processor} using FFM.
20   */
21  @ThreadSafe
22  public final class Win32ProcessorFFM extends Win32Processor {
23  
24      private Win32ProcessorFFM() {
25      }
26  
27      /**
28       * Returns processor voltage.
29       *
30       * @return Current voltage of the processor.
31       */
32      public static WmiResult<VoltProperty> queryVoltage() {
33          WmiQuery<VoltProperty> voltQuery = new WmiQuery<>(WIN32_PROCESSOR, VoltProperty.class);
34          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
35                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(voltQuery);
36      }
37  
38      /**
39       * Returns processor ID.
40       *
41       * @return Processor information that describes the processor features.
42       */
43      public static WmiResult<ProcessorIdProperty> queryProcessorId() {
44          WmiQuery<ProcessorIdProperty> idQuery = new WmiQuery<>(WIN32_PROCESSOR, ProcessorIdProperty.class);
45          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
46                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(idQuery);
47      }
48  
49      /**
50       * Returns address width.
51       *
52       * @return On a 32-bit operating system, the value is 32 and on a 64-bit operating system it is 64.
53       */
54      public static WmiResult<BitnessProperty> queryBitness() {
55          WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<>(WIN32_PROCESSOR, BitnessProperty.class);
56          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
57                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(bitnessQuery);
58      }
59  }
