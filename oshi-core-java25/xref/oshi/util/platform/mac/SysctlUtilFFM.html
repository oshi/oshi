23  /**
24   * Provides access to sysctl calls on macOS
25   */
26  @ThreadSafe
27  public final class SysctlUtilFFM {
28  
29      private static final Logger LOG = LoggerFactory.getLogger(SysctlUtilFFM.class);
30  
31      private static final String SYSCTL_FAIL = "Failed sysctl call: {}, Error code: {}";
32  
33      private SysctlUtilFFM() {
34      }
35  
36      /**
37       * Executes a sysctl call with an int result
38       *
39       * @param name name of the sysctl
40       * @param def  default int value
41       * @return The int result of the call if successful; the default otherwise
42       */
43      public static int sysctl(String name, int def) {
44          return sysctl(name, def, true);
45      }
46  
47      /**
48       * Executes a sysctl call with an int result
49       *
50       * @param name       name of the sysctl
51       * @param def        default int value
52       * @param logWarning whether to log the warning if not available
53       * @return The int result of the call if successful; the default otherwise
54       */
55      public static int sysctl(String name, int def, boolean logWarning) {
56          try (Arena arena = Arena.ofConfined()) {
57              MemorySegment nameSeg = arena.allocateFrom(name);
58              MemorySegment valueSeg = arena.allocate(JAVA_INT);
59              MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, JAVA_INT.byteSize());
60              if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg, logWarning)) {
61                  return def;
62              }
63              return valueSeg.get(JAVA_INT, 0);
64          } catch (Throwable e) {
65              if (logWarning) {
66                  LOG.warn("Failed to get sysctl value for {}", name, e);
67              }
68              return def;
69          }
70      }

105     /**
106      * Executes a sysctl call with a String result
107      *
108      * @param name       name of the sysctl
109      * @param def        default String value
110      * @param logWarning whether to log the warning if not available
111      * @return The String result of the call if successful; the default otherwise
112      */
113     public static String sysctl(String name, String def, boolean logWarning) {
114         try (Arena arena = Arena.ofConfined()) {
115             MemorySegment nameSeg = arena.allocateFrom(name);
116             MemorySegment sizeSeg = arena.allocate(SIZE_T);
117             if (!sysctlbyname(arena, nameSeg, MemorySegment.NULL, sizeSeg, logWarning)) {
118                 return def;
119             }
120             long size = sizeSeg.get(SIZE_T, 0);
121             MemorySegment valueSeg = arena.allocate(size + 1); // +1 for null terminator
122             if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg, logWarning)) {
123                 return def;
124             }
125             return valueSeg.getString(0);
126         } catch (Throwable e) {
127             if (logWarning) {
128                 LOG.warn("Failed to get sysctl value for {}", name, e);
129             }
130             return def;
131         }
132     }

134     /**
135      * Executes a sysctl call with a Structure result
136      *
137      * @param name   name of the sysctl
138      * @param struct structure for the result
139      * @return True if structure is successfuly populated, false otherwise
140      */
141     public static boolean sysctl(String name, MemorySegment struct) {
142         try (Arena arena = Arena.ofConfined()) {
143             MemorySegment nameSeg = arena.allocateFrom(name);
144             MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, struct.byteSize());
145             return sysctlbyname(arena, nameSeg, struct, sizeSeg, true);
146         } catch (Throwable e) {
147             LOG.warn("Failed to get sysctl value for {}", name, e);
148             return false;
149         }
150     }
151 
152     /**
153      * Executes a sysctl call with a Pointer result
154      *
155      * @param name name of the sysctl
156      * @return An allocated memory buffer containing the result on success, null otherwise. Its value on failure is
157      *         undefined.
158      */
159     public static MemorySegment sysctl(String name) {
160         try (Arena arena = Arena.ofConfined()) {
161             MemorySegment nameSeg = arena.allocateFrom(name);
162             MemorySegment sizeSeg = arena.allocate(SIZE_T);
163             if (!sysctlbyname(arena, nameSeg, MemorySegment.NULL, sizeSeg, true)) {
164                 return null;
165             }
166             long size = sizeSeg.get(SIZE_T, 0);
167             MemorySegment valueSeg = arena.allocate(size);
168             if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg, true)) {
169                 return null;
170             }
171             // Need to copy to a segment that will be released on GC
172             MemorySegment returnSeg = Arena.ofAuto().allocate(size);
173             returnSeg.copyFrom(valueSeg);
174             return returnSeg;
175         } catch (Throwable e) {
176             LOG.warn("Failed to get sysctl value for {}", name, e);
177             return null;
178         }
179     }
180 
181     /**
182      * Executes a sysctl call with a Memory Segment result
183      *
184      * @param mib    definition of the sysctl
185      * @param buffer buffer to hold the result. Must be allocated
186      * @return The size of data written to the buffer, or -1 if the call failed
187      */
188     public static long sysctl(int[] mib, MemorySegment buffer) {
189         try (Arena arena = Arena.ofConfined()) {
190             MemorySegment mibSeg = arena.allocateFrom(JAVA_INT, mib);
191             MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, buffer.byteSize());
192             MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
193             int result = MacSystemFunctions.sysctl(callState, mibSeg, mib.length, buffer, sizeSeg, MemorySegment.NULL,
194                     0L);
195 
196             if (result != 0) {
197                 LOG.warn(SYSCTL_FAIL, Arrays.toString(mib), getErrno(callState));
198                 return -1;
199             }
200             return sizeSeg.get(SIZE_T, 0);
201         } catch (Throwable e) {
202             LOG.warn("Failed to get sysctl value for {}", Arrays.toString(mib), e);
203             return -1;
204         }
205     }
206 
207     private static boolean sysctlbyname(Arena arena, MemorySegment name, MemorySegment oldp, MemorySegment oldlenp,
208             boolean logWarning) throws Throwable {
209         MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
210         var result = MacSystemFunctions.sysctlbyname(callState, name, oldp, oldlenp, MemorySegment.NULL, 0L);
211         if (result != 0) {
212             if (logWarning) {
213                 LOG.warn(SYSCTL_FAIL, name, getErrno(callState));
214             }
215             return false;
216         }
217         return true;
218     }
