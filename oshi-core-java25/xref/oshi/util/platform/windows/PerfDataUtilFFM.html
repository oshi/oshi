48  /**
49   * Helper class to centralize the boilerplate portions of PDH counter setup using the FFM API.
50   * <p>
51   * This utility collects counter data with a single call to {@code PdhCollectQueryData}, which yields valid results for
52   * raw counters. OSHI queries raw counter values and computes rates in Java rather than using PDH formatted rate
53   * counters, which would require two samples separated by an interval.
54   */
55  public final class PerfDataUtilFFM {

58      }

65      /**
66       * Query multiple PDH counter values in a single query, one per enum constant.
67       *
68       * @param <T>          An enum implementing {@link PdhCounterProperty}
69       * @param propertyEnum The enum class whose constants define the counters to query
70       * @param perfObject   The PDH object name (e.g., "Process")
71       * @return An {@link EnumMap} of values indexed by enum constant. May contain a subset of constants if individual
72       *         counters fail to add or read. Returns an empty map if the query itself fails to open or collect.
73       */
74      public static <T extends Enum<T> & PdhCounterProperty> Map<T, Long> queryCounters(Class<T> propertyEnum,
75              String perfObject) {
76          T[] props = propertyEnum.getEnumConstants();
77          EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
78  
79          try (Arena arena = Arena.ofConfined()) {
80              MemorySegment queryPtr = arena.allocate(ADDRESS);
81              checkSuccess(PdhOpenQuery(MemorySegment.NULL, MemorySegment.NULL, queryPtr));
82              MemorySegment query = null;
83  
84              try {
85                  query = queryPtr.get(ADDRESS, 0);
86  
87                  // Add all counters to the single query, skipping any that fail
88                  EnumMap<T, MemorySegment> counterHandles = new EnumMap<>(propertyEnum);
89                  EnumMap<T, Boolean> baseFlags = new EnumMap<>(propertyEnum);
90                  for (T prop : props) {
91                      String counterName = prop.getCounter();
92                      boolean isBase = PerfCounter.isBase(counterName);
93                      if (isBase) {
94                          counterName = PerfCounter.stripBaseSuffix(counterName);
95                      }
96                      String path = counterPath(perfObject, prop.getInstance(), counterName);
97                      try {
98                          counterHandles.put(prop, addEnglishCounter(arena, query, path));
99                          baseFlags.put(prop, isBase);
100                     } catch (Throwable t) {
101                         LOG.debug("Failed to add counter {}: {}", path, t.getMessage());
102                     }
103                 }
104                 if (counterHandles.isEmpty()) {
105                     return valueMap;
106                 }
107 
108                 // Collect once
109                 checkSuccess(PdhCollectQueryData(query));
110 
111                 // Read all values, skipping any that fail
112                 for (var entry : counterHandles.entrySet()) {
113                     try {
114                         valueMap.put(entry.getKey(),
115                                 readRawCounterValue(arena, entry.getValue(), baseFlags.get(entry.getKey())));
116                     } catch (Throwable t) {
117                         LOG.debug("Failed to read counter for {}: {}", entry.getKey(), t.getMessage());
118                     }
119                 }
120 
121             } finally {
122                 if (query != null) {
123                     PdhCloseQuery(query);
124                 }
125             }
126 
127         } catch (Throwable t) {
128             LOG.debug("PDH queryCounters failed for {}: {}", perfObject, t.getMessage(), t);
129             return new EnumMap<>(propertyEnum);
130         }
131         return valueMap;
132     }

134     private static String counterPath(String object, String instance, String counter) {
135         StringBuilder path = new StringBuilder();
136         path.append('\\').append(object);
137         if (instance != null) {
138             path.append('(').append(instance).append(')');
139         }
140         path.append('\\').append(counter);
141         return path.toString();
142     }

144     private static MemorySegment addEnglishCounter(Arena arena, MemorySegment query, String path) throws Throwable {
145         MemorySegment counterPtr = arena.allocate(ADDRESS);
146         checkSuccess(PdhAddEnglishCounter(query, toWideString(arena, path), MemorySegment.NULL, counterPtr));
147         return counterPtr.get(ADDRESS, 0);
148     }

150     private static final long RAW_CSTATUS_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("CStatus"));
151     private static final long RAW_FIRST_VALUE_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("FirstValue"));
152     private static final long RAW_SECOND_VALUE_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("SecondValue"));
153 
154     private static long readRawCounterValue(Arena arena, MemorySegment counterHandle, boolean secondValue)
155             throws Throwable {
156         MemorySegment value = arena.allocate(PDH_RAW_COUNTER_LAYOUT);
157         checkSuccess(PdhGetRawCounterValue(counterHandle, MemorySegment.NULL, value));
158         int cStatus = value.get(JAVA_INT, RAW_CSTATUS_OFFSET);
159         if (cStatus != PDH_CSTATUS_VALID_DATA && cStatus != PDH_CSTATUS_NEW_DATA) {
160             throw new IllegalStateException(
161                     String.format(Locale.ROOT, "PDH raw counter CStatus invalid: 0x%08X", cStatus));
162         }
163         return value.get(JAVA_LONG, secondValue ? RAW_SECOND_VALUE_OFFSET : RAW_FIRST_VALUE_OFFSET);
164     }

166     /**
167      * Enumerate instances of a PDH performance object, filtering by the wildcard pattern from the first enum constant.
168      *
169      * @param perfObject     The PDH object name (e.g., "Process")
170      * @param instanceFilter The wildcard filter pattern (lowercase)
171      * @return A list of matching instance names, or an empty list on failure
172      */
173     static List<String> enumInstances(String perfObject, String instanceFilter) {
174         // PdhEnumObjectItemsW requires the localized object name
175         String localizedObject = localizeObjectName(perfObject);
176         try (Arena arena = Arena.ofConfined()) {
177             MemorySegment objectName = toWideString(arena, localizedObject);
178             MemorySegment counterLen = arena.allocate(JAVA_INT);
179             MemorySegment instanceLen = arena.allocate(JAVA_INT);
180 
181             // First call: get required buffer sizes
182             counterLen.set(JAVA_INT, 0, 0);
183             instanceLen.set(JAVA_INT, 0, 0);
184             int result = PdhEnumObjectItemsW(MemorySegment.NULL, MemorySegment.NULL, objectName, MemorySegment.NULL,
185                     counterLen, MemorySegment.NULL, instanceLen, 100, 0);
186             if (result != ERROR_SUCCESS && result != PDH_MORE_DATA) {
187                 LOG.warn("Failed to enumerate instances for {}: 0x{}", perfObject, Integer.toHexString(result));
188                 return Collections.emptyList();
189             }
190 
191             // Retry loop for race conditions (e.g., process/thread list changes)
192             MemorySegment counterBuf;
193             MemorySegment instanceBuf;
194             int retries = 0;
195             do {
196                 int cLen = counterLen.get(JAVA_INT, 0);
197                 int iLen = instanceLen.get(JAVA_INT, 0);
198                 counterBuf = cLen > 0 ? arena.allocate((long) cLen * 2) : MemorySegment.NULL;
199                 instanceBuf = iLen > 0 ? arena.allocate((long) iLen * 2) : MemorySegment.NULL;
200 
201                 result = PdhEnumObjectItemsW(MemorySegment.NULL, MemorySegment.NULL, objectName, counterBuf, counterLen,
202                         instanceBuf, instanceLen, 100, 0);
203                 // On PDH_MORE_DATA, counterLen/instanceLen are updated with required sizes;
204                 // add headroom for race conditions where instances change between calls
205                 if (result == PDH_MORE_DATA) {
206                     counterLen.set(JAVA_INT, 0, counterLen.get(JAVA_INT, 0) + 1024);
207                     instanceLen.set(JAVA_INT, 0, instanceLen.get(JAVA_INT, 0) + 1024);
208                 }
209             } while (result == PDH_MORE_DATA && ++retries < 5);
210 
211             if (result != ERROR_SUCCESS) {
212                 LOG.warn("Failed to enumerate instances for {}: 0x{}", localizedObject, Integer.toHexString(result));
213                 return Collections.emptyList();
214             }
215 
216             // Parse multi-sz instance list and filter
217             List<String> instances = parseMultiSz(instanceBuf);
218             instances.removeIf(i -> !Util.wildcardMatch(i.toLowerCase(Locale.ROOT), instanceFilter));
219             return instances;
220         } catch (Throwable t) {
221             LOG.warn("Failed to enumerate instances for {}: {}", localizedObject, t.getMessage());
222             return Collections.emptyList();
223         }
224     }

226     /**
227      * Parse a null-terminated double-null-terminated wide string list (multi-sz) into a list of strings.
228      *
229      * @param buf the memory segment containing the multi-sz wide string data
230      * @return a list of parsed strings
231      */
232     private static List<String> parseMultiSz(MemorySegment buf) {
233         List<String> result = new ArrayList<>();
234         if (buf == null || buf.equals(MemorySegment.NULL)) {
235             return result;
236         }
237         long offset = 0;
238         long size = buf.byteSize();
239         while (offset < size) {
240             StringBuilder sb = new StringBuilder();
241             while (offset < size) {
242                 char c = buf.get(JAVA_CHAR, offset);
243                 offset += 2;
244                 if (c == '\0') {
245                     break;
246                 }
247                 sb.append(c);
248             }
249             if (sb.isEmpty()) {
250                 break; // double null = end of list
251             }
252             result.add(sb.toString());
253         }
254         return result;
255     }

258      * Query wildcard PDH counter values for all instances matching the filter defined by the first enum constant.
259      *
260      * @param <T>          An enum implementing {@link PdhCounterWildcardProperty}
261      * @param propertyEnum The enum class whose constants define the counters to query. The first constant defines the
262      *                     instance filter; remaining constants define counter names.
263      * @param perfObject   The PDH object name (e.g., "Process")
264      * @return A pair of (instances, valueMap) where valueMap is indexed by enum constant (excluding the first). Returns
265      *         empty list and empty map on failure.
266      */
267     public static <T extends Enum<T> & PdhCounterWildcardProperty> Pair<List<String>, Map<T, List<Long>>> queryWildcardCounters(
268             Class<T> propertyEnum, String perfObject) {
269         return queryWildcardCounters(propertyEnum, perfObject, null);
270     }

272     /**
273      * Query wildcard PDH counter values for all instances matching the given filter or the default from the first enum
274      * constant.
275      *
276      * @param <T>          An enum implementing {@link PdhCounterWildcardProperty}
277      * @param propertyEnum The enum class whose constants define the counters to query. The first constant defines the
278      *                     default instance filter; remaining constants define counter names.
279      * @param perfObject   The PDH object name (e.g., "Process")
280      * @param customFilter A custom instance filter to use. If null, uses the first element of the property enum.
281      * @return A pair of (instances, valueMap) where valueMap is indexed by enum constant (excluding the first). Returns
282      *         empty list and empty map on failure.
283      */
284     public static <T extends Enum<T> & PdhCounterWildcardProperty> Pair<List<String>, Map<T, List<Long>>> queryWildcardCounters(
285             Class<T> propertyEnum, String perfObject, String customFilter) {
286 
287         T[] props = propertyEnum.getEnumConstants();
288         if (props.length < 2) {
289             throw new IllegalArgumentException("Enum " + propertyEnum.getName()
290                     + " must have at least two elements, an instance filter and a counter.");
291         }
292         String instanceFilter = Util.isBlank(customFilter) ? props[0].getCounter().toLowerCase(Locale.ROOT)
293                 : customFilter.toLowerCase(Locale.ROOT);
294 
295         List<String> instances = enumInstances(perfObject, instanceFilter);
296         EnumMap<T, List<Long>> valuesMap = new EnumMap<>(propertyEnum);
297         if (instances.isEmpty()) {
298             return new Pair<>(Collections.emptyList(), valuesMap);
299         }
300 
301         try (Arena arena = Arena.ofConfined()) {
302             MemorySegment queryPtr = arena.allocate(ADDRESS);
303             checkSuccess(PdhOpenQuery(MemorySegment.NULL, MemorySegment.NULL, queryPtr));
304             MemorySegment query = null;
305 
306             try {
307                 query = queryPtr.get(ADDRESS, 0);
308 
309                 // Add counters for each instance × property (skip first which is the filter)
310                 // counterHandles[propIndex - 1][instanceIndex]
311                 // Detect _Base suffix: strip it from the counter path and read SecondValue instead
312                 boolean[] isBase = new boolean[props.length - 1];
313                 @SuppressWarnings("unchecked")
314                 List<MemorySegment>[] counterHandles = new List[props.length - 1];
315                 for (int p = 1; p < props.length; p++) {
316                     String counterName = props[p].getCounter();
317                     if (PerfCounter.isBase(counterName)) {
318                         counterName = PerfCounter.stripBaseSuffix(counterName);
319                         isBase[p - 1] = true;
320                     }
321                     counterHandles[p - 1] = new ArrayList<>(instances.size());
322                     for (String instance : instances) {
323                         String path = counterPath(perfObject, instance, counterName);
324                         try {
325                             counterHandles[p - 1].add(addEnglishCounter(arena, query, path));
326                         } catch (Throwable t) {
327                             LOG.warn("Failed to add wildcard counter {}: {}", path, t.getMessage());
328                             return new Pair<>(Collections.emptyList(), new EnumMap<>(propertyEnum));
329                         }
330                     }
331                 }
332 
333                 // Collect once
334                 checkSuccess(PdhCollectQueryData(query));
335 
336                 // Read values; use SecondValue for _Base counters
337                 for (int p = 1; p < props.length; p++) {
338                     List<Long> values = new ArrayList<>(instances.size());
339                     for (MemorySegment handle : counterHandles[p - 1]) {
340                         try {
341                             values.add(readRawCounterValue(arena, handle, isBase[p - 1]));
342                         } catch (Throwable t) {
343                             LOG.debug("Failed to read counter for {}: {}", props[p], t.getMessage());
344                             values.add(0L);
345                         }
346                     }
347                     valuesMap.put(props[p], values);
348                 }
349             } finally {
350                 if (query != null) {
351                     PdhCloseQuery(query);
352                 }
353             }
354         } catch (Throwable t) {
355             LOG.debug("PDH queryWildcardCounters failed for {}: {}", perfObject, t.getMessage(), t);
356             return new Pair<>(Collections.emptyList(), new EnumMap<>(propertyEnum));
357         }
358         return new Pair<>(instances, valuesMap);
359     }

361     /**
362      * Localize a PDH object name. {@code PdhEnumObjectItemsW} requires the localized name unlike
363      * {@code PdhAddEnglishCounterW} which handles English names natively.
364      *
365      * @param perfObject The English object name
366      * @return The localized name, or the original if localization fails
367      */
368     private static String localizeObjectName(String perfObject) {
369         return LOCALIZE_CACHE.computeIfAbsent(perfObject, PerfDataUtilFFM::localizeUsingPerfIndex);
370     }
371 
372     private static String localizeUsingPerfIndex(String perfObject) {
373         int index = lookupPerfIndexByEnglishName(perfObject);
374         if (index == 0) {
375             LOG.debug("Unable to find English counter index for {}", perfObject);
376             return perfObject;
377         }
378         String localized = lookupPerfNameByIndex(index);
379         if (localized.isEmpty()) {
380             return perfObject;
381         }
382         LOG.debug("Localized {} to {}", perfObject, localized);
383         return localized;
384     }
385 
386     private static int lookupPerfIndexByEnglishName(String name) {
387         return HkeyPerformanceDataUtilFFM.getCounterIndex(name);
388     }

390     private static String lookupPerfNameByIndex(int index) {
391         try (Arena arena = Arena.ofConfined()) {
392             MemorySegment bufSize = arena.allocate(JAVA_INT);
393             bufSize.set(JAVA_INT, 0, 0);
394             int result = PdhLookupPerfNameByIndexW(MemorySegment.NULL, index, MemorySegment.NULL, bufSize);
395             if (result != PDH_MORE_DATA && result != ERROR_SUCCESS) {
396                 return "";
397             }
398             int size = bufSize.get(JAVA_INT, 0);
399             if (size <= 0) {
400                 return "";
401             }
402             MemorySegment nameBuf = arena.allocate((long) size * 2);
403             result = PdhLookupPerfNameByIndexW(MemorySegment.NULL, index, nameBuf, bufSize);
404             if (result != ERROR_SUCCESS) {
405                 return "";
406             }
407             return readWideString(nameBuf);
408         } catch (Throwable t) {
409             LOG.debug("Failed to lookup perf name by index {}: {}", index, t.getMessage());
410             return "";
411         }
