5   package oshi;
6   
7   import oshi.hardware.HardwareAbstractionLayer;
8   import oshi.hardware.platform.linux.LinuxHardwareAbstractionLayer;
9   import oshi.hardware.platform.mac.MacHardwareAbstractionLayerFFM;
10  import oshi.hardware.platform.windows.WindowsHardwareAbstractionLayer;
11  import oshi.software.os.OperatingSystem;
12  import oshi.software.os.linux.LinuxOperatingSystem;
13  import oshi.software.os.mac.MacOperatingSystemFFM;
14  import oshi.software.os.windows.WindowsOperatingSystemFFM;
15  
16  import java.util.function.Supplier;
17  
18  import static oshi.util.Memoizer.memoize;
19  
20  public class SystemInfoFFM {
21  
22      private static final PlatformEnumFFM CURRENT_PLATFORM;
23  
24      static {
25          String osName = System.getProperty("os.name");
26          CURRENT_PLATFORM = switch (osName) {
27              case String name when name.startsWith("Linux") -> PlatformEnumFFM.LINUX;
28              case String name when name.startsWith("Mac") || name.startsWith("Darwin") -> PlatformEnumFFM.MACOS;
29              case String name when name.startsWith("Windows") -> PlatformEnumFFM.WINDOWS;
30              default -> PlatformEnumFFM.UNSUPPORTED;
31          };
32      }
33  
34      private static final String NOT_SUPPORTED = "Unsupported platform: " + CURRENT_PLATFORM;
35  
36      private final Supplier<OperatingSystem> os = memoize(SystemInfoFFM::createOperatingSystem);
37      private final Supplier<HardwareAbstractionLayer> hardware = memoize(SystemInfoFFM::createHardware);
38  
39      /**
40       * Constructs a new SystemInfo instance.
41       */
42      public SystemInfoFFM() {
43          // Empty constructor for API consistency
44      }
45  
46      private static OperatingSystem createOperatingSystem() {
47          return switch (CURRENT_PLATFORM) {
48              case LINUX -> new LinuxOperatingSystem();
49              case MACOS -> new MacOperatingSystemFFM();
50              case WINDOWS -> new WindowsOperatingSystemFFM();
51              default -> throw new UnsupportedOperationException(NOT_SUPPORTED);
52          };
53      }
54  
55      private static HardwareAbstractionLayer createHardware() {
56          return switch (CURRENT_PLATFORM) {
57              case LINUX -> new LinuxHardwareAbstractionLayer();
58              case MACOS -> new MacHardwareAbstractionLayerFFM();
59              case WINDOWS -> new WindowsHardwareAbstractionLayer();
60              default -> throw new UnsupportedOperationException(NOT_SUPPORTED);
61          };
62      }
63  
64      /**
65       * Gets the {@link PlatformEnumFFM} value representing this system.
66       *
67       * @return Returns the current platform
68       */
69      public static PlatformEnumFFM getCurrentPlatform() {
70          return CURRENT_PLATFORM;
71      }
72  
73      /**
74       * Creates a new instance of the appropriate platform-specific {@link OperatingSystem}.
75       *
76       * @return A new platform-specific instance implementing {@link OperatingSystem}.
77       */
78      public OperatingSystem getOperatingSystem() {
79          return os.get();
80      }
81  
82      /**
83       * Creates a new instance of the appropriate platform-specific {@link HardwareAbstractionLayer}.
84       *
85       * @return A new platform-specific instance implementing {@link HardwareAbstractionLayer}.
86       */
87      public HardwareAbstractionLayer getHardware() {
88          return hardware.get();
89      }
90  }
