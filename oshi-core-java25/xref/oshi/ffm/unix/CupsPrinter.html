27  /**
28   * CUPS-based printer implementation using the Java FFM API. Uses libcups if available, otherwise falls back to the
29   * {@code lpstat} command.
30   */
31  @Immutable
32  public final class CupsPrinter extends AbstractPrinter {

34      private static final Logger LOG = LoggerFactory.getLogger(CupsPrinter.class);

36      // Local URI schemes for directly-attached or local printers
37      private static final List<String> LOCAL_URI_PREFIXES = List.of("usb:", "parallel:", "serial:", "file:", "direct:",
38              "hp:", "lpd://127.", "lpd://localhost", "socket://127.", "socket://localhost");

40      CupsPrinter(String name, String driverName, String description, PrinterStatus status, String statusReason,
41              boolean isDefault, boolean isLocal, String portName) {
42          super(name, driverName, description, status, statusReason, isDefault, isLocal, portName);
43      }

45      /**
46       * Gets printers using CUPS. Uses libcups via FFM if available, otherwise falls back to the {@code lpstat} command.
47       *
48       * @return a list of printers
49       */
50      public static List<Printer> getPrinters() {
51          if (CupsFunctions.isAvailable()) {
52              return getPrintersFromLibCups();
53          }
54          LOG.debug("libcups not available, falling back to lpstat.");
55          return getPrintersFromLpstat();
56      }
57  
58      private static List<Printer> getPrintersFromLibCups() {
59          List<Printer> printers = new ArrayList<>();
60          try (Arena arena = Arena.ofConfined()) {
61              // cupsGetDests(cups_dest_t **dests) — allocate a pointer slot to receive the array pointer
62              MemorySegment ptrSlot = arena.allocate(ADDRESS);
63              int numDests = CupsFunctions.cupsGetDests(ptrSlot);
64              if (numDests <= 0) {
65                  return printers;
66              }
67              MemorySegment destsPtr = ptrSlot.get(ADDRESS, 0);
68              if (destsPtr == null || destsPtr.equals(MemorySegment.NULL)) {
69                  return printers;
70              }
71              try {
72                  MemorySegment destsArray = destsPtr.reinterpret(CupsFunctions.CUPS_DEST_T.byteSize() * numDests, arena,
73                          null);
74                  for (int i = 0; i < numDests; i++) {
75                      long offset = i * CupsFunctions.CUPS_DEST_T.byteSize();
76                      MemorySegment dest = destsArray.asSlice(offset, CupsFunctions.CUPS_DEST_T.byteSize());
77  
78                      MemorySegment namePtr = (MemorySegment) CupsFunctions.CUPS_DEST_NAME.get(dest, 0L);
79                      if (namePtr == null || namePtr.equals(MemorySegment.NULL)) {
80                          continue;
81                      }
82                      String name = ForeignFunctions.getStringFromNativePointer(namePtr, arena);
83                      int isDefaultInt = (int) CupsFunctions.CUPS_DEST_IS_DEFAULT.get(dest, 0L);
84                      boolean isDefault = isDefaultInt != 0;
85                      int numOptions = (int) CupsFunctions.CUPS_DEST_NUM_OPTIONS.get(dest, 0L);
86                      MemorySegment options = (MemorySegment) CupsFunctions.CUPS_DEST_OPTIONS.get(dest, 0L);
87  
88                      String deviceUri = CupsFunctions.getOption("device-uri", numOptions, options, arena);
89                      String printerInfo = CupsFunctions.getOption("printer-info", numOptions, options, arena);
90                      String printerMakeModel = CupsFunctions.getOption("printer-make-and-model", numOptions, options,
91                              arena);
92                      String printerState = CupsFunctions.getOption("printer-state", numOptions, options, arena);
93                      String stateReasons = CupsFunctions.getOption("printer-state-reasons", numOptions, options, arena);
94                      String printerTypeStr = CupsFunctions.getOption("printer-type", numOptions, options, arena);
95  
96                      PrinterStatus status = parseStateFromCups(printerState, stateReasons);
97                      String statusReason = "none".equals(stateReasons) ? "" : stateReasons;
98                      int printerType = ParseUtil.parseIntOrDefault(printerTypeStr, 0);
99                      boolean isLocal = (printerType & CupsFunctions.CUPS_PRINTER_REMOTE) == 0;
100 
101                     printers.add(new CupsPrinter(name, printerMakeModel, printerInfo, status, statusReason, isDefault,
102                             isLocal, deviceUri));
103                 }
104             } finally {
105                 CupsFunctions.cupsFreeDests(numDests, destsPtr);
106             }
107 
108         } catch (Throwable e) {
109             LOG.warn("Failed to query printers from libcups: {}", e.getMessage(), e);
110         }
111         return printers;
112     }

114     private static PrinterStatus parseStateFromCups(String state, String stateReasons) {
115         if (!stateReasons.isEmpty() && !"none".equals(stateReasons)) {
116             String lower = stateReasons.toLowerCase(Locale.ROOT);
117             if (lower.contains("error") || lower.contains("fault")) {
118                 return PrinterStatus.ERROR;
119             }
120         }
121         if (state.isEmpty()) {
122             return PrinterStatus.UNKNOWN;
123         }
124         switch (ParseUtil.parseIntOrDefault(state, -1)) {
125             case CupsFunctions.IPP_PRINTER_IDLE:
126                 return PrinterStatus.IDLE;
127             case CupsFunctions.IPP_PRINTER_PROCESSING:
128                 return PrinterStatus.PRINTING;
129             case CupsFunctions.IPP_PRINTER_STOPPED:
130                 return PrinterStatus.OFFLINE;
131             default:
132                 return PrinterStatus.UNKNOWN;
133         }
134     }

136     private static List<Printer> getPrintersFromLpstat() {
137         List<Printer> printers = new ArrayList<>();
138         String defaultPrinter = getDefaultPrinter();
139         Map<String, String> portMap = parsePortMap();
140         Map<String, String> descriptionMap = parseDescriptionMap();
141 
142         for (String line : ExecutingCommand.runNative(new String[] { "lpstat", "-p" })) {
143             if (line.startsWith("printer ")) {
144                 String[] parts = line.split("\\s+");
145                 if (parts.length >= 3) {
146                     String name = parts[1];
147                     PrinterStatus status = parseStatusFromLpstat(line);
148                     boolean isDefault = name.equals(defaultPrinter);
149                     String portName = portMap.getOrDefault(name, "");
150                     boolean isLocal = isLocalUri(portName);
151                     String driverName = getDriverForPrinter(name);
152                     String description = descriptionMap.getOrDefault(name, "");
153                     String statusReason = getStatusReasonFromLpstat(line);
154                     printers.add(new CupsPrinter(name, driverName, description, status, statusReason, isDefault,
155                             isLocal, portName));
156                 }
157             }
158         }
159         return printers;
160     }

162     private static Map<String, String> parsePortMap() {
163         Map<String, String> map = new HashMap<>();
164         for (String line : ExecutingCommand.runNative(new String[] { "lpstat", "-v" })) {
165             if (line.contains("device for")) {
166                 int forIdx = line.indexOf("device for ") + 11;
167                 int colonIdx = line.indexOf(':', forIdx);
168                 if (colonIdx > forIdx) {
169                     map.put(line.substring(forIdx, colonIdx).trim(), line.substring(colonIdx + 1).trim());
170                 }
171             }
172         }
173         return map;
174     }
175 
176     private static Map<String, String> parseDescriptionMap() {
177         Map<String, String> map = new HashMap<>();
178         String currentPrinter = null;
179         for (String line : ExecutingCommand.runNative(new String[] { "lpstat", "-l", "-p" })) {
180             if (line.startsWith("printer ")) {
181                 String[] parts = line.split("\\s+");
182                 if (parts.length >= 2) {
183                     currentPrinter = parts[1];
184                 }
185             } else if (currentPrinter != null && line.trim().startsWith("Description:")) {
186                 map.put(currentPrinter, line.substring(line.indexOf(':') + 1).trim());
187             }
188         }
189         return map;
190     }

192     private static String getDriverForPrinter(String printerName) {
193         for (String line : ExecutingCommand.runNative(new String[] { "lpoptions", "-p", printerName })) {
194             int idx = line.indexOf("printer-make-and-model='");
195             if (idx >= 0) {
196                 int start = idx + 24;
197                 int end = line.indexOf('\'', start);
198                 if (end > start) {
199                     return line.substring(start, end);
200                 }
201             }
202         }
203         return "";
204     }

206     private static String getDefaultPrinter() {
207         for (String line : ExecutingCommand.runNative(new String[] { "lpstat", "-d" })) {
208             if (line.contains("default destination:")) {
209                 String[] parts = line.split(":", 2);
210                 if (parts.length >= 2) {
211                     return parts[1].trim();
212                 }
213             }
214         }
215         return "";
216     }

218     private static PrinterStatus parseStatusFromLpstat(String line) {
219         String lower = line.toLowerCase(Locale.ROOT);
220         if (lower.contains("disabled") || lower.contains("not accepting")) {
221             return PrinterStatus.OFFLINE;
222         } else if (lower.contains("printing")) {
223             return PrinterStatus.PRINTING;
224         } else if (lower.contains("idle")) {
225             return PrinterStatus.IDLE;
226         } else if (lower.contains("error") || lower.contains("fault")) {
227             return PrinterStatus.ERROR;
228         }
229         return PrinterStatus.UNKNOWN;
230     }
231 
232     private static String getStatusReasonFromLpstat(String line) {
233         int dashIdx = line.indexOf(" - ");
234         return dashIdx > 0 ? line.substring(dashIdx + 3).trim() : "";
235     }
236 
237     private static boolean isLocalUri(String uri) {
238         if (uri.startsWith("/dev")) {
239             return true;
240         }
241         for (String prefix : LOCAL_URI_PREFIXES) {
242             if (uri.startsWith(prefix)) {
243                 return true;
244             }
245         }
246         return false;
247     }
