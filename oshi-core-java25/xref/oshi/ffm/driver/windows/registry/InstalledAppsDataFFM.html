36  public final class InstalledAppsDataFFM {
37  
38      private static final Logger LOG = LoggerFactory.getLogger(InstalledAppsDataFFM.class);
39  
40      private static final long THIRTY_YEARS_IN_SECS = 30L * 365 * 24 * 60 * 60;
41  
42      private static final Map<MemorySegment, List<String>> REGISTRY_PATHS = new LinkedHashMap<>();
43      private static final int[] ACCESS_FLAGS = { KEY_WOW64_64KEY, KEY_WOW64_32KEY };
44  
45      static {
46          REGISTRY_PATHS.put(MemorySegment.ofAddress(HKEY_LOCAL_MACHINE),
47                  Arrays.asList("SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Uninstall",
48                          "SOFTWARE\\WOW6432Node\\Microsoft\\Windows\\CurrentVersion\\Uninstall"));
49          REGISTRY_PATHS.put(MemorySegment.ofAddress(HKEY_CURRENT_USER),
50                  Arrays.asList("SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Uninstall"));
51      }
52  
53      private InstalledAppsDataFFM() {
54      }
55  
56      public static List<ApplicationInfo> queryInstalledApps() {
57          Set<ApplicationInfo> appInfoSet = new LinkedHashSet<>();
58  
59          for (Map.Entry<MemorySegment, List<String>> entry : REGISTRY_PATHS.entrySet()) {
60              MemorySegment rootKey = entry.getKey();
61              List<String> uninstallPaths = entry.getValue();
62  
63              for (String registryPath : uninstallPaths) {
64                  for (int accessFlag : ACCESS_FLAGS) {
65                      try {
66                          String[] keys = Advapi32UtilFFM.registryGetKeys(rootKey, registryPath, accessFlag);
67                          for (String key : keys) {
68                              String fullPath = registryPath + "\\" + key;
69                              try {
70                                  String name = registryValueToString(
71                                          getRegistryValueOrNull(rootKey, fullPath, "DisplayName", accessFlag));
72                                  if (name == null) {
73                                      continue;
74                                  }
75                                  String version = registryValueToString(
76                                          getRegistryValueOrNull(rootKey, fullPath, "DisplayVersion", accessFlag));
77                                  String publisher = registryValueToString(
78                                          getRegistryValueOrNull(rootKey, fullPath, "Publisher", accessFlag));
79                                  long installDate = registryValueToLong(
80                                          getRegistryValueOrNull(rootKey, fullPath, "InstallDate", accessFlag));
81                                  String installLocation = registryValueToString(
82                                          getRegistryValueOrNull(rootKey, fullPath, "InstallLocation", accessFlag));
83                                  String installSource = registryValueToString(
84                                          getRegistryValueOrNull(rootKey, fullPath, "InstallSource", accessFlag));
85  
86                                  Map<String, String> additionalInfo = new LinkedHashMap<>();
87                                  additionalInfo.put("installLocation", installLocation);
88                                  additionalInfo.put("installSource", installSource);
89  
90                                  ApplicationInfo app = new ApplicationInfo(name, version, publisher, installDate,
91                                          additionalInfo);
92                                  appInfoSet.add(app);
93  
94                              } catch (Throwable e) {
95                                  LOG.trace("Skipping key {}: {}", fullPath, e.getMessage());
96                              }
97                          }
98                      } catch (Throwable e) {
99                          LOG.trace("Skipping path {}: {}", registryPath, e.getMessage());
100                     }
101                 }
102             }
103         }
104 
105         return new ArrayList<>(appInfoSet);
106     }
107 
108     private static String registryValueToString(Object registryValueOrNull) {
109         if (registryValueOrNull instanceof Integer i) {
110             return Integer.toString(i);
111         }
112         return (String) registryValueOrNull;
113     }
114 
115     private static long registryValueToLong(Object registryValueOrNull) {
116         if (registryValueOrNull == null) {
117             return 0L;
118         }
119         long currentTimeSecs = System.currentTimeMillis() / 1000L;
120         long minSaneTimestamp = currentTimeSecs - THIRTY_YEARS_IN_SECS;
121 
122         if (registryValueOrNull instanceof Integer i) {
123             if (i > minSaneTimestamp && i < currentTimeSecs) {
124                 return i * 1000L;
125             }
126             return i;
127         } else if (registryValueOrNull instanceof String s) {
128             String dateStr = s.trim();
129             long epoch = ParseUtil.parseDateToEpoch(dateStr, "yyyyMMdd");
130             if (epoch == 0) {
131                 epoch = ParseUtil.parseDateToEpoch(dateStr, "MM/dd/yyyy");
132             }
133             return epoch;
134         }
135         return 0L;
136     }
137 
138     private static Object getRegistryValueOrNull(MemorySegment rootKey, String path, String key, int accessFlag)
139             throws Throwable {
140         try (Arena arena = Arena.ofConfined()) {
141             MemorySegment phkKey = arena.allocate(ADDRESS);
142             int rc = RegOpenKeyEx(rootKey, toWideString(arena, path), 0, KEY_READ | accessFlag, phkKey);
143             checkSuccess(rc);
144 
145             MemorySegment hKey = phkKey.get(ADDRESS, 0);
146             try {
147                 return registryGetValue(hKey, key);
148             } finally {
149                 rc = RegCloseKey(hKey);
150                 checkSuccess(rc);
151             }
152         } catch (Win32Exception e) {
153             LOG.trace("Unable to access " + path + " with flag " + accessFlag + ": " + e.getMessage());
154             return null;
155         }
156     }
