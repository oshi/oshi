5   package oshi.ffm.util;
6   
7   import java.nio.ByteBuffer;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.ffm.ForeignFunctions;
11  import oshi.util.FileUtil;
12  
13  /**
14   * FFM-specific extensions to {@link FileUtil} for reading native types from buffers.
15   */
16  @ThreadSafe
17  public final class FileUtilFFM {
18  
19      private FileUtilFFM() {
20      }
21  
22      /**
23       * Reads a native {@code long} value from a ByteBuffer, using the platform's native long size.
24       *
25       * @param buff The bytebuffer to read from
26       * @return The next value as a Java long
27       */
28      public static long readNativeLongFromBuffer(ByteBuffer buff) {
29          return ForeignFunctions.NATIVE_LONG_SIZE == 4 ? FileUtil.readIntFromBuffer(buff)
30                  : FileUtil.readLongFromBuffer(buff);
31      }
32  
33      /**
34       * Reads a native {@code size_t} value from a ByteBuffer, using the platform's native size_t size.
35       *
36       * @param buff The bytebuffer to read from
37       * @return The next value as a Java long
38       */
39      public static long readSizeTFromBuffer(ByteBuffer buff) {
40          return ForeignFunctions.NATIVE_SIZE_T_SIZE == 4 ? Integer.toUnsignedLong(FileUtil.readIntFromBuffer(buff))
41                  : FileUtil.readLongFromBuffer(buff);
42      }
43  
44      /**
45       * Reads a native pointer value from a ByteBuffer, using the platform's native pointer size.
46       *
47       * @param buff The bytebuffer to read from
48       * @return The next value as a Java long representing the pointer address, or 0 if insufficient data
49       */
50      public static long readPointerFromBuffer(ByteBuffer buff) {
51          if (buff.position() <= buff.limit() - ForeignFunctions.NATIVE_POINTER_SIZE) {
52              return ForeignFunctions.NATIVE_POINTER_SIZE == 4 ? Integer.toUnsignedLong(buff.getInt()) : buff.getLong();
53          }
54          return 0L;
55      }
56  }
