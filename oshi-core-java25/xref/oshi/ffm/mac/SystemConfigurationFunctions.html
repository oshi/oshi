14  public final class SystemConfigurationFunctions extends MacForeignFunctions {
15  
16      private SystemConfigurationFunctions() {
17      }
18  
19      private static final SymbolLookup SC_LIBRARY = frameworkLookup("SystemConfiguration");
20  
21      // CFArrayRef SCNetworkInterfaceCopyAll(void);
22  
23      private static final MethodHandle SCNetworkInterfaceCopyAll = LINKER
24              .downcallHandle(SC_LIBRARY.findOrThrow("SCNetworkInterfaceCopyAll"), FunctionDescriptor.of(ADDRESS));
25  
26      public static MemorySegment SCNetworkInterfaceCopyAll() throws Throwable {
27          return (MemorySegment) SCNetworkInterfaceCopyAll.invokeExact();
28      }
29  
30      // CFStringRef SCNetworkInterfaceGetBSDName(SCNetworkInterfaceRef interface);
31  
32      private static final MethodHandle SCNetworkInterfaceGetBSDName = LINKER.downcallHandle(
33              SC_LIBRARY.findOrThrow("SCNetworkInterfaceGetBSDName"), FunctionDescriptor.of(ADDRESS, ADDRESS));
34  
35      public static MemorySegment SCNetworkInterfaceGetBSDName(MemorySegment netif) throws Throwable {
36          return (MemorySegment) SCNetworkInterfaceGetBSDName.invokeExact(netif);
37      }
