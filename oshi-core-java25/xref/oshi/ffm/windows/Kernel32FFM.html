440     /**
441      * Retrieves the full name of the executable image for the specified process.
442      *
443      * @param hProcess Handle to the process
444      * @param dwFlags  Flags (0 for Win32 path format)
445      * @param arena    Arena for memory allocation
446      * @return The full process image name, or empty if failed
447      */
448     public static Optional<String> QueryFullProcessImageName(MemorySegment hProcess, int dwFlags, Arena arena) {
449         int maxPath = 260;
450         // Retry with growing buffer if needed (up to 32K which is max path on Windows)
451         while (maxPath <= 32768) {
452             MemorySegment buffer = arena.allocate(JAVA_CHAR, maxPath);
453             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
454             sizeSegment.set(JAVA_INT, 0, maxPath);
455 
456             if (QueryFullProcessImageName(hProcess, dwFlags, buffer, sizeSegment)) {
457                 return Optional.of(readWideString(buffer));
458             }
459             // Check if buffer was too small (returned size >= maxPath)
460             int returnedSize = sizeSegment.get(JAVA_INT, 0);
461             if (returnedSize < maxPath) {
462                 // Failure wasn't due to buffer size
463                 break;
464             }
465             // Double the buffer size and retry
466             maxPath *= 2;
467         }
468         return Optional.empty();
469     }
