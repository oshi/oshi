30      /**
31       * INVALID_HANDLE_VALUE constant (-1) used by Windows API to indicate failure.
32       */
33      private static final long INVALID_HANDLE_VALUE = -1L;
34  
35      /**
36       * Checks if a handle represents INVALID_HANDLE_VALUE.
37       *
38       * @param handle the handle to check
39       * @return true if the handle is null or equals INVALID_HANDLE_VALUE
40       */
41      private static boolean isInvalidHandle(MemorySegment handle) {
42          return handle == null || handle.address() == INVALID_HANDLE_VALUE;
43      }
44  
45      private static final MethodHandle CloseHandle = downcall(K32, "CloseHandle", JAVA_INT, ADDRESS);
46  
47      public static OptionalInt CloseHandle(MemorySegment handle) {
48          try {
49              return OptionalInt.of((int) CloseHandle.invokeExact(handle));
50          } catch (Throwable t) {
51              LOG.debug("Kernel32FFM.closeHandle failed", t);
52              return OptionalInt.empty();
53          }
54      }

56      private static final MethodHandle FindFirstVolume = downcall(K32, "FindFirstVolumeW", ADDRESS, ADDRESS, JAVA_INT);
57  
58      public static Optional<MemorySegment> FindFirstVolume(MemorySegment lpszVolumeName, int cchBufferLength) {
59          try {
60              MemorySegment handle = (MemorySegment) FindFirstVolume.invokeExact(lpszVolumeName, cchBufferLength);
61              if (isInvalidHandle(handle)) {
62                  int error = (int) GetLastError.invokeExact();
63                  LOG.error("FindFirstVolume failed with error: {}", error);
64                  return Optional.empty();
65              }
66              return Optional.of(handle);
67          } catch (Throwable t) {
68              LOG.debug("Kernel32FFM.FindFirstVolume failed", t);
69              return Optional.empty();
70          }
71      }

99      private static final MethodHandle GetComputerName = downcall(K32, "GetComputerNameW", JAVA_INT, ADDRESS, ADDRESS);
100 
101     public static Optional<String> GetComputerName() {
102         try (Arena arena = Arena.ofConfined()) {
103             int maxLen = 32;
104 
105             MemorySegment buffer = arena.allocate(JAVA_CHAR, maxLen);
106             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
107             sizeSegment.set(ValueLayout.JAVA_INT, 0, maxLen);
108 
109             boolean success = isSuccess((int) GetComputerName.invokeExact(buffer, sizeSegment));
110             if (!success) {
111                 int errorCode = (int) GetLastError.invokeExact();
112                 LOG.error("Failed to get computer name name. Error code: {}", errorCode);
113                 return Optional.empty();
114             }
115 
116             return Optional.of(readWideString(buffer));
117         } catch (Throwable t) {
118             LOG.debug("Kernel32FFM.GetComputerName failed: {}", t.getMessage());
119             return Optional.empty();
120         }
121     }

126     public static Optional<String> GetComputerNameEx() {
127         try (Arena arena = Arena.ofConfined()) {
128             int maxLen = 256;
129 
130             MemorySegment buffer = arena.allocate(JAVA_CHAR, maxLen);
131             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
132             sizeSegment.set(ValueLayout.JAVA_INT, 0, maxLen);
133 
134             final int COMPUTER_NAME_DNS_DOMAIN_FULLY_QUALIFIED = 3;
135 
136             boolean success = isSuccess(
137                     (int) GetComputerNameEx.invokeExact(COMPUTER_NAME_DNS_DOMAIN_FULLY_QUALIFIED, buffer, sizeSegment));
138             if (!success) {
139                 int errorCode = (int) GetLastError.invokeExact();
140                 LOG.error("Failed to get DNS domain name. Error code: {}", errorCode);
141                 return Optional.empty();
142             }
143 
144             return Optional.of(readWideString(buffer));
145         } catch (Throwable t) {
146             LOG.debug("Kernel32FFM.GetComputerNameEx failed: {}", t.getMessage());
147             return Optional.empty();
148         }
