30      private static final MethodHandle CloseHandle = downcall(K32, "CloseHandle", JAVA_INT, ADDRESS);
31  
32      public static OptionalInt CloseHandle(MemorySegment handle) {
33          try {
34              return OptionalInt.of((int) CloseHandle.invokeExact(handle));
35          } catch (Throwable t) {
36              LOG.debug("Kernel32FFM.closeHandle failed", t);
37              return OptionalInt.empty();
38          }
39      }

41      private static final MethodHandle GetComputerName = downcall(K32, "GetComputerNameW", JAVA_INT, ADDRESS, ADDRESS);
42  
43      public static Optional<String> GetComputerName() {
44          try (Arena arena = Arena.ofConfined()) {
45              int maxLen = 32;
46  
47              MemorySegment buffer = arena.allocate(JAVA_CHAR, maxLen);
48              MemorySegment sizeSegment = arena.allocate(JAVA_INT);
49              sizeSegment.set(ValueLayout.JAVA_INT, 0, maxLen);
50  
51              boolean success = isSuccess((int) GetComputerName.invokeExact(buffer, sizeSegment));
52              if (!success) {
53                  int errorCode = (int) GetLastError.invokeExact();
54                  LOG.error("Failed to get computer name name. Error code: {}", errorCode);
55                  return Optional.empty();
56              }
57  
58              return Optional.of(readWideString(buffer));
59          } catch (Throwable t) {
60              LOG.debug("Kernel32FFM.GetComputerName failed: {}", t.getMessage());
61              return Optional.empty();
62          }
63      }
64  
65      private static final MethodHandle GetComputerNameEx = downcall(K32, "GetComputerNameExW", JAVA_INT, JAVA_INT,
66              ADDRESS, ADDRESS);
67  
68      public static Optional<String> GetComputerNameEx() {
69          try (Arena arena = Arena.ofConfined()) {
70              int maxLen = 256;
71  
72              MemorySegment buffer = arena.allocate(JAVA_CHAR, maxLen);
73              MemorySegment sizeSegment = arena.allocate(JAVA_INT);
74              sizeSegment.set(ValueLayout.JAVA_INT, 0, maxLen);
75  
76              final int COMPUTER_NAME_DNS_DOMAIN_FULLY_QUALIFIED = 3;
77  
78              boolean success = isSuccess(
79                      (int) GetComputerNameEx.invokeExact(COMPUTER_NAME_DNS_DOMAIN_FULLY_QUALIFIED, buffer, sizeSegment));
80              if (!success) {
81                  int errorCode = (int) GetLastError.invokeExact();
82                  LOG.error("Failed to get DNS domain name. Error code: {}", errorCode);
83                  return Optional.empty();
84              }
85  
86              return Optional.of(readWideString(buffer));
87          } catch (Throwable t) {
88              LOG.debug("Kernel32FFM.GetComputerNameEx failed: {}", t.getMessage());
89              return Optional.empty();
90          }
91      }
