5   package oshi.ffm.windows;
6   
7   import java.util.Locale;
8   
9   public class Win32Exception extends RuntimeException {
10  
11      private final int errorCode;
12  
13      public Win32Exception(int errorCode) {
14          super(formatMessage(errorCode));
15          this.errorCode = errorCode;
16      }
17  
18      public int getErrorCode() {
19          return errorCode;
20      }
21  
22      private static String formatMessage(int errorCode) {
23          return String.format(Locale.ROOT, "Win32 API call failed with error code 0x%08X (%d)", errorCode, errorCode);
24      }
25  }
