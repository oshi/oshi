5   package oshi.ffm.windows.com;
6   
7   import java.lang.foreign.Arena;
8   import java.lang.foreign.MemoryLayout;
9   import java.lang.foreign.MemorySegment;
10  import java.lang.foreign.StructLayout;
11  
12  import static java.lang.foreign.ValueLayout.JAVA_BYTE;
13  import static java.lang.foreign.ValueLayout.JAVA_INT;
14  import static java.lang.foreign.ValueLayout.JAVA_SHORT;
15  
16  /**
17   * FFM representation of the Windows GUID structure.
18   * <p>
19   * A GUID (Globally Unique Identifier) is a 128-bit value used to identify COM interfaces and classes.
20   * </p>
21   */
22  public final class GuidFFM {
23  
24      /**
25       * The memory layout of a GUID structure (16 bytes).
26       */
27      public static final StructLayout LAYOUT = MemoryLayout.structLayout(JAVA_INT.withName("Data1"),
28              JAVA_SHORT.withName("Data2"), JAVA_SHORT.withName("Data3"),
29              MemoryLayout.sequenceLayout(8, JAVA_BYTE).withName("Data4"));
30  
31      /**
32       * Size of a GUID in bytes.
33       */
34      public static final long SIZE = LAYOUT.byteSize();
35  
36      private GuidFFM() {
37      }
38  
39      /**
40       * Creates a GUID memory segment from its components.
41       *
42       * @param arena the arena for memory allocation
43       * @param data1 the first 4 bytes (big-endian)
44       * @param data2 the next 2 bytes (big-endian)
45       * @param data3 the next 2 bytes (big-endian)
46       * @param data4 the remaining 8 bytes
47       * @return a memory segment containing the GUID
48       */
49      public static MemorySegment create(Arena arena, int data1, short data2, short data3, byte... data4) {
50          if (data4.length != 8) {
51              throw new IllegalArgumentException("Data4 must be exactly 8 bytes");
52          }
53          MemorySegment guid = arena.allocate(LAYOUT);
54          guid.set(JAVA_INT, 0, data1);
55          guid.set(JAVA_SHORT, 4, data2);
56          guid.set(JAVA_SHORT, 6, data3);
57          for (int i = 0; i < 8; i++) {
58              guid.set(JAVA_BYTE, 8 + i, data4[i]);
59          }
60          return guid;
61      }
62  
63      /**
64       * Creates a GUID memory segment from its components using integer values for convenience.
65       *
66       * @param arena the arena for memory allocation
67       * @param data1 the first 4 bytes
68       * @param data2 the next 2 bytes
69       * @param data3 the next 2 bytes
70       * @param d4_0  data4[0]
71       * @param d4_1  data4[1]
72       * @param d4_2  data4[2]
73       * @param d4_3  data4[3]
74       * @param d4_4  data4[4]
75       * @param d4_5  data4[5]
76       * @param d4_6  data4[6]
77       * @param d4_7  data4[7]
78       * @return a memory segment containing the GUID
79       */
80      public static MemorySegment create(Arena arena, int data1, int data2, int data3, int d4_0, int d4_1, int d4_2,
81              int d4_3, int d4_4, int d4_5, int d4_6, int d4_7) {
82          byte[] data4 = new byte[] { (byte) d4_0, (byte) d4_1, (byte) d4_2, (byte) d4_3, (byte) d4_4, (byte) d4_5,
83                  (byte) d4_6, (byte) d4_7 };
84          return create(arena, data1, (short) data2, (short) data3, data4);
85      }
86  
87      // Well-known GUIDs for WMI
88  
89      /**
90       * CLSID_WbemLocator: {4590F811-1D3A-11D0-891F-00AA004B2E24}
91       *
92       * @param arena the arena for memory allocation
93       * @return a memory segment containing the CLSID_WbemLocator GUID
94       */
95      public static MemorySegment CLSID_WbemLocator(Arena arena) {
96          return create(arena, 0x4590f811, 0x1d3a, 0x11d0, 0x89, 0x1f, 0x00, 0xaa, 0x00, 0x4b, 0x2e, 0x24);
97      }
98  
99      /**
100      * IID_IWbemLocator: {DC12A687-737F-11CF-884D-00AA004B2E24}
101      *
102      * @param arena the arena for memory allocation
103      * @return a memory segment containing the IID_IWbemLocator GUID
104      */
105     public static MemorySegment IID_IWbemLocator(Arena arena) {
106         return create(arena, 0xdc12a687, 0x737f, 0x11cf, 0x88, 0x4d, 0x00, 0xaa, 0x00, 0x4b, 0x2e, 0x24);
107     }
108 
109     /**
110      * IID_IUnknown: {00000000-0000-0000-C000-000000000046}
111      *
112      * @param arena the arena for memory allocation
113      * @return a memory segment containing the IID_IUnknown GUID
114      */
115     public static MemorySegment IID_IUnknown(Arena arena) {
116         return create(arena, 0x00000000, 0x0000, 0x0000, 0xc0, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x46);
117     }
