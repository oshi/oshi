5   package oshi.ffm.windows.com;
6   
7   import java.lang.foreign.Arena;
8   import java.lang.foreign.FunctionDescriptor;
9   import java.lang.foreign.Linker;
10  import java.lang.foreign.MemorySegment;
11  import java.lang.invoke.MethodHandle;
12  
13  import static java.lang.foreign.MemorySegment.NULL;
14  import static java.lang.foreign.ValueLayout.ADDRESS;
15  
16  /**
17   * Base class for COM object operations via FFM.
18   * <p>
19   * Provides utilities for vtable access. IUnknown interface methods (AddRef, Release, safeRelease) are in
20   * {@link IUnknownFFM}.
21   * </p>
22   */
23  public class ComObjectFFM extends IUnknownFFM {
24  
25      /**
26       * Size of a pointer in bytes (8 on 64-bit systems).
27       */
28      protected static final long PTR_SIZE = ADDRESS.byteSize();
29  
30      protected ComObjectFFM() {
31      }
32  
33      /**
34       * Gets the vtable pointer from a COM object.
35       *
36       * @param pObject the COM object pointer
37       * @param arena   the arena for memory reinterpretation
38       * @return the vtable memory segment
39       */
40      protected static MemorySegment getVtable(MemorySegment pObject, Arena arena) {
41          if (pObject == null || pObject.equals(NULL)) {
42              return NULL;
43          }
44          // COM object layout: first pointer is the vtable pointer
45          MemorySegment vtablePtr = pObject.reinterpret(PTR_SIZE, arena, null).get(ADDRESS, 0);
46          // Reinterpret with enough slots for IWbemLocator (7 methods) and IWbemServices (30 methods);
47          // 32 slots covers all COM interfaces used by this implementation
48          return vtablePtr.reinterpret(PTR_SIZE * 32, arena, null);
49      }
50  
51      /**
52       * Gets a function pointer from a vtable at the specified index.
53       *
54       * @param vtable the vtable memory segment
55       * @param index  the function index (0-based)
56       * @return the function pointer
57       */
58      protected static MemorySegment getVtableFunction(MemorySegment vtable, int index) {
59          return vtable.get(ADDRESS, (long) index * PTR_SIZE);
60      }
61  
62      /**
63       * Creates a downcall handle for a vtable function.
64       *
65       * @param fnPtr      the function pointer
66       * @param descriptor the function descriptor
67       * @return the method handle
68       */
69      protected static MethodHandle createDowncall(MemorySegment fnPtr, FunctionDescriptor descriptor) {
70          return Linker.nativeLinker().downcallHandle(fnPtr, descriptor);
71      }
72  }
