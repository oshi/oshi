51      private static final MethodHandle OpenEventLog = downcall(ADV, "OpenEventLogW", ADDRESS, ADDRESS, ADDRESS);
52  
53      public static Optional<MemorySegment> OpenEventLog(MemorySegment serverName, MemorySegment sourceName)
54              throws Throwable {
55          MemorySegment handle = (MemorySegment) OpenEventLog.invokeExact(serverName, sourceName);
56          if (handle == null || handle.equals(MemorySegment.NULL)) {
57              return Optional.empty();
58          }
59          return Optional.of(handle);
60      }
61  
62      public static Optional<MemorySegment> OpenEventLog(Arena arena, String source) {
63          try {
64              MemorySegment lpSource = WindowsForeignFunctions.toWideString(arena, source);
65              MemorySegment handle = (MemorySegment) OpenEventLog.invokeExact(MemorySegment.NULL, lpSource);
66              if (handle.address() == 0) {
67                  return Optional.empty();
68              }
69              return Optional.of(handle);
70          } catch (Throwable t) {
71              return Optional.empty();
72          }
73      }
