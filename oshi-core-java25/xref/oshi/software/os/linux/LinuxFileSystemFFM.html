5   package oshi.software.os.linux;
6   
7   import java.lang.foreign.Arena;
8   import java.lang.foreign.MemorySegment;
9   
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import oshi.annotation.concurrent.ThreadSafe;
14  import oshi.ffm.linux.LinuxLibcFunctions;
15  import oshi.software.common.os.linux.LinuxFileSystem;
16  
17  /**
18   * FFM-based Linux file system implementation. Implements {@code statvfs} via FFM.
19   */
20  @ThreadSafe
21  final class LinuxFileSystemFFM extends LinuxFileSystem {
22  
23      private static final Logger LOG = LoggerFactory.getLogger(LinuxFileSystemFFM.class);
24  
25      @Override
26      protected long[] queryStatvfs(String path) {
27          try (Arena arena = Arena.ofConfined()) {
28              MemorySegment pathSeg = arena.allocateFrom(path);
29              MemorySegment buf = arena.allocate(LinuxLibcFunctions.STATVFS_LAYOUT);
30              if (0 == LinuxLibcFunctions.statvfs(pathSeg, buf)) {
31                  long frsize = LinuxLibcFunctions.statvfsFrsize(buf);
32                  return new long[] { LinuxLibcFunctions.statvfsFiles(buf), LinuxLibcFunctions.statvfsFfree(buf),
33                          LinuxLibcFunctions.statvfsBlocks(buf) * frsize, LinuxLibcFunctions.statvfsBavail(buf) * frsize,
34                          LinuxLibcFunctions.statvfsBfree(buf) * frsize };
35              }
36              LOG.debug("statvfs failed for path: {}", path);
37          } catch (Throwable e) {
38              LOG.debug("FFM statvfs error for path {}: {}", path, e.toString());
39          }
40          return null;
41      }
42  }
