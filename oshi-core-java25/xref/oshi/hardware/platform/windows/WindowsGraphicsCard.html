44  /**
45   * Graphics Card obtained from the Windows registry, with VRAM sourced from DXGI.
46   *
47   * <p>
48   * VRAM detection priority:
49   * <ol>
50   * <li>{@code DXGI_ADAPTER_DESC.DedicatedVideoMemory}: the authoritative Windows API value, not subject to the 2 GiB cap
51   * that affects the 32-bit registry field.</li>
52   * <li>{@code HardwareInformation.qwMemorySize} (64-bit registry value): used when DXGI enumeration is unavailable or no
53   * adapter match is found.</li>
54   * </ol>
55   *
56   * <p>
57   * {@code HardwareInformation.MemorySize} (32-bit) is intentionally not used: Windows writes the sentinel value
58   * {@code 0x7FFFF000} (~2 GiB) into this field for GPUs with more than 2 GiB of dedicated VRAM, making it unreliable for
59   * modern discrete GPUs.
60   *
61   * <p>
62   * When DXGI is available, ghost adapters (stale registry entries from hardware no longer present) are excluded because
63   * {@code IDXGIFactory::EnumAdapters} only enumerates physically present adapters. The returned list is ordered to match
64   * DXGI enumeration order, which places the primary desktop adapter first.
65   *
66   * <p>
67   * Dynamic metrics (utilization, VRAM used, shared memory, ticks) are sourced from PDH GPU Engine / GPU Adapter Memory
68   * counters (Windows 10 1709+) and optionally from LibreHardwareMonitor WMI sensors when LHM is running.
69   */
70  @ThreadSafe
71  final class WindowsGraphicsCard extends AbstractGraphicsCard {

129     /**
130      * public method used by {@code AbstractHardwareAbstractionLayer} to access the graphics cards.
131      *
132      * <p>
133      * When DXGI is available, ghost adapters are excluded and the list is ordered with the primary desktop adapter
134      * first. On systems without DXGI, all registry entries are returned in registry key order.
135      *
136      * @return List of {@link WindowsGraphicsCard} objects.
137      */
138     public static List<GraphicsCard> getGraphicsCards() {
139         // Query DXGI once. Fails gracefully to empty list if unavailable.
140         List<DxgiAdapterInfo> dxgiAdapters = WindowsDxgi.queryAdapters();
141         boolean dxgiAvailable = !dxgiAdapters.isEmpty();
142         // Mutable copy for match-and-consume (prevents same adapter matching two registry entries).
143         List<DxgiAdapterInfo> remainingDxgi = new ArrayList<>(dxgiAdapters);
144 
145         // Build LHM parent map: GPU name (normalized) -> LHM identifier
146         Map<String, String> lhmParentMap = buildLhmParentMap();
147 
148         // When DXGI is available, collect cards keyed by their DXGI enumeration index so the
149         // final list is ordered primary-adapter-first (DXGI guarantees adapter 0 is the primary
150         // desktop adapter). Registry entries with no DXGI match are ghost adapters and excluded.
151         // When DXGI is unavailable, fall back to simple insertion order.
152         TreeMap<Integer, GraphicsCard> dxgiOrdered = new TreeMap<>();
153         List<GraphicsCard> cardList = new ArrayList<>();
154 
155         int index = 1;
156         String[] keys = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, DISPLAY_DEVICES_REGISTRY_PATH);
157         for (String key : keys) {
158             if (!key.startsWith("0")) {
159                 continue;
160             }
161 
162             try {
163                 String fullKey = DISPLAY_DEVICES_REGISTRY_PATH + key;
164                 if (!Advapi32Util.registryValueExists(WinReg.HKEY_LOCAL_MACHINE, fullKey, ADAPTER_STRING)) {
165                     continue;
166                 }
167 
168                 String name = RegistryUtil.getStringValue(WinReg.HKEY_LOCAL_MACHINE, fullKey, DRIVER_DESC);
169                 String deviceId = "VideoController" + index++;
170                 String vendor = RegistryUtil.getStringValue(WinReg.HKEY_LOCAL_MACHINE, fullKey, VENDOR);
171                 String versionInfo = RegistryUtil.getStringValue(WinReg.HKEY_LOCAL_MACHINE, fullKey, DRIVER_VERSION);
172 
173                 // Parse PCI vendor/device IDs from MatchingDeviceId (e.g. "pci\ven_8086&dev_56a0&...")
174                 String matchingDeviceId = RegistryUtil.getStringValue(WinReg.HKEY_LOCAL_MACHINE, fullKey,
175                         MATCHING_DEVICE_ID);
176                 Pair<Integer, Integer> pciIds = ParseUtil.parseDeviceIdToVendorProductIds(matchingDeviceId);
177                 int pciVendorId = pciIds == null ? 0 : pciIds.getA();
178                 int pciDeviceId = pciIds == null ? 0 : pciIds.getB();
179 
180                 // Primary: DXGI DedicatedVideoMemory.
181                 // Track whether a DXGI match was found separately from the vram value, so that a
182                 // legitimate DedicatedVideoMemory == 0 (e.g. a software/render-only adapter) is
183                 // preserved and does not trigger the registry fallback.
184                 long vram = -1L;
185                 int dxgiIndex = -1;
186                 String luidPrefix = "";
187                 int pciBusNumber = -1;
188                 String pciBusId = "";
189                 String locationInfo = RegistryUtil.getStringValue(WinReg.HKEY_LOCAL_MACHINE, fullKey,
190                         LOCATION_INFORMATION);
191                 DxgiAdapterInfo dxgiMatch = WindowsDxgi.findMatch(remainingDxgi, pciVendorId, pciDeviceId, name);
192                 if (dxgiMatch != null) {
193                     vram = dxgiMatch.getDedicatedVideoMemory();
194                     dxgiIndex = dxgiAdapters.indexOf(dxgiMatch);
195                     luidPrefix = buildLuidPrefix(dxgiMatch);
196                     pciBusNumber = parsePciBusNumber(locationInfo);
197                     pciBusId = buildPciBusId(locationInfo);
198                 } else if (dxgiAvailable) {
199                     // DXGI is available but this registry entry has no matching adapter:
200                     // it is a ghost device (stale driver from hardware no longer present). Skip it.
201                     continue;
202                 }
203 
204                 // Fallback: 64-bit registry value qwMemorySize, only when DXGI had no match.
205                 if (vram < 0 && Advapi32Util.registryValueExists(WinReg.HKEY_LOCAL_MACHINE, fullKey, QW_MEMORY_SIZE)) {
206                     Object regValue = Advapi32Util.registryGetValue(WinReg.HKEY_LOCAL_MACHINE, fullKey, QW_MEMORY_SIZE);
207                     vram = registryValueToVram(regValue);
208                 }
209 
210                 // Normalise sentinel: if still unresolved report 0.
211                 if (vram < 0) {
212                     vram = 0L;
213                 }
214 
215                 // HardwareInformation.MemorySize (32-bit) is intentionally omitted: Windows caps
216                 // it at 0x7FFFF000 (~2 GiB) for GPUs with more VRAM, making it unreliable.
217 
218                 String lhmParent = lhmParentMap.getOrDefault(WindowsDxgi.normalizeName(Util.isBlank(name) ? "" : name),
219                         "");
220 
221                 GraphicsCard card = new WindowsGraphicsCard(Util.isBlank(name) ? Constants.UNKNOWN : name,
222                         Util.isBlank(deviceId) ? Constants.UNKNOWN : deviceId,
223                         Util.isBlank(vendor) ? Constants.UNKNOWN : vendor,
224                         Util.isBlank(versionInfo) ? Constants.UNKNOWN : versionInfo, vram, luidPrefix, lhmParent,
225                         pciBusNumber, pciBusId);
226                 // Remove dxgiMatch from remainingDxgi only after the card is successfully
227                 // constructed. This ensures that if earlier registry reads in this try block
228                 // throw a Win32Exception, dxgiMatch remains in remainingDxgi and is still
229                 // available for subsequent iterations or WMI fallback processing.
230                 if (dxgiMatch != null) {
231                     remainingDxgi.remove(dxgiMatch);
232                 }
233                 if (dxgiIndex >= 0) {
234                     dxgiOrdered.put(dxgiIndex, card);
235                 } else {
236                     cardList.add(card);
237                 }
238             } catch (Win32Exception e) {
239                 if (e.getErrorCode() != WinError.ERROR_ACCESS_DENIED) {
240                     // Ignore access denied errors, re-throw others
241                     throw e;
242                 }
243             }
244         }
245 
246         // Merge: DXGI-ordered cards first (primary adapter at index 0), then any non-DXGI cards.
247         List<GraphicsCard> result = new ArrayList<>(dxgiOrdered.values());
248         result.addAll(cardList);
249 
250         if (result.isEmpty()) {
251             return getGraphicsCardsFromWmi(remainingDxgi, lhmParentMap);
252         }
253         return result;
254     }

267     // fall back if something went wrong
268     private static List<GraphicsCard> getGraphicsCardsFromWmi(List<DxgiAdapterInfo> dxgiAdapters,
269             Map<String, String> lhmParentMap) {
270         List<GraphicsCard> cardList = new ArrayList<>();
271         if (IS_VISTA_OR_GREATER) {
272             boolean dxgiAvailable = !dxgiAdapters.isEmpty();
273             // dxgiAdapters is not mutated; remainingDxgi is the working copy consumed during matching.
274             // dxgiAdapters is retained as the stable reference for indexOf ordering lookups.
275             List<DxgiAdapterInfo> remainingDxgi = new ArrayList<>(dxgiAdapters);
276             TreeMap<Integer, GraphicsCard> dxgiOrdered = new TreeMap<>();
277 
278             WmiResult<VideoControllerProperty> cards = Win32VideoController.queryVideoController();
279             for (int index = 0; index < cards.getResultCount(); index++) {
280                 // ConfigManagerErrorCode 0 = working properly; non-zero = disabled/error (ghost device).
281                 // When DXGI is unavailable, keep all entries for maximum compatibility.
282                 if (dxgiAvailable
283                         && WmiUtil.getUint32(cards, VideoControllerProperty.CONFIGMANAGERERRORCODE, index) != 0) {
284                     continue;
285                 }
286                 String name = WmiUtil.getString(cards, VideoControllerProperty.NAME, index);
287                 Triplet<String, String, String> idPair = ParseUtil.parseDeviceIdToVendorProductSerial(
288                         WmiUtil.getString(cards, VideoControllerProperty.PNPDEVICEID, index));
289                 String deviceId = idPair == null ? Constants.UNKNOWN : idPair.getB();
290                 String vendor = WmiUtil.getString(cards, VideoControllerProperty.ADAPTERCOMPATIBILITY, index);
291                 if (idPair != null) {
292                     if (Util.isBlank(vendor)) {
293                         deviceId = idPair.getA();
294                     } else {
295                         vendor = vendor + " (" + idPair.getA() + ")";
296                     }
297                 }
298                 String versionInfo = WmiUtil.getString(cards, VideoControllerProperty.DRIVERVERSION, index);
299                 if (!Util.isBlank(versionInfo)) {
300                     versionInfo = "DriverVersion=" + versionInfo;
301                 } else {
302                     versionInfo = Constants.UNKNOWN;
303                 }
304                 // Prefer DXGI DedicatedVideoMemory when a match can be found via the PCI IDs
305                 // extracted from PNPDEVICEID. Fall back to WMI AdapterRAM (32-bit capped) only
306                 // when no DXGI match is available.
307                 Pair<Integer, Integer> pciIds = ParseUtil.parseDeviceIdToVendorProductIds(
308                         WmiUtil.getString(cards, VideoControllerProperty.PNPDEVICEID, index));
309                 int pciVendorId = pciIds == null ? 0 : pciIds.getA();
310                 int pciDeviceId = pciIds == null ? 0 : pciIds.getB();
311                 DxgiAdapterInfo dxgiMatch = WindowsDxgi.findMatch(remainingDxgi, pciVendorId, pciDeviceId, name);
312                 long vram;
313                 int dxgiIndex = -1;
314                 String luidPrefix = "";
315                 int pciBusNumber = -1;
316                 String pciBusId = "";
317                 if (dxgiMatch != null) {
318                     vram = dxgiMatch.getDedicatedVideoMemory();
319                     dxgiIndex = dxgiAdapters.indexOf(dxgiMatch);
320                     luidPrefix = buildLuidPrefix(dxgiMatch);
321                     // pciBusNumber and pciBusId are not available via WMI; leave as -1 / empty.
322                     // ADL and NVML correlation will be skipped for cards enumerated via this path.
323                 } else {
324                     vram = WmiUtil.getUint32asLong(cards, VideoControllerProperty.ADAPTERRAM, index);
325                 }
326                 String lhmParent = lhmParentMap.getOrDefault(WindowsDxgi.normalizeName(Util.isBlank(name) ? "" : name),
327                         "");
328                 GraphicsCard card = new WindowsGraphicsCard(Util.isBlank(name) ? Constants.UNKNOWN : name, deviceId,
329                         Util.isBlank(vendor) ? Constants.UNKNOWN : vendor, versionInfo, vram, luidPrefix, lhmParent,
330                         pciBusNumber, pciBusId);
331                 // Remove dxgiMatch from remainingDxgi only after the card is successfully
332                 // constructed, matching the registry path's defensive pattern so that a
333                 // failure during construction leaves the match available for subsequent entries.
334                 if (dxgiMatch != null) {
335                     remainingDxgi.remove(dxgiMatch);
336                 }
337                 if (dxgiIndex >= 0) {
338                     dxgiOrdered.put(dxgiIndex, card);
339                 } else {
340                     cardList.add(card);
341                 }
342             }
343             List<GraphicsCard> result = new ArrayList<>(dxgiOrdered.values());
344             result.addAll(cardList);
345             return result;
346         }
347         return cardList;
348     }

350     /**
351      * Queries LHM WMI for GPU hardware entries and returns a map from normalized GPU name to LHM parent identifier.
352      * Returns an empty map if LHM is not running. If two GPU entries normalize to the same name the key is mapped to an
353      * empty string so callers skip LHM for that ambiguous name rather than returning the wrong adapter's metrics.
354      *
355      * @return map of normalized GPU name to LHM hardware identifier
356      */
357     private static Map<String, String> buildLhmParentMap() {
358         Map<String, String> map = new HashMap<>();
359         try {
360             WmiResult<LhmHardwareProperty> hw = LhmSensor.queryGpuHardware();
361             for (int i = 0; i < hw.getResultCount(); i++) {
362                 String identifier = WmiUtil.getString(hw, LhmHardwareProperty.IDENTIFIER, i);
363                 String hwName = WmiUtil.getString(hw, LhmHardwareProperty.NAME, i);
364                 if (!identifier.isEmpty() && !hwName.isEmpty()) {
365                     String norm = WindowsDxgi.normalizeName(hwName);
366                     if (map.containsKey(norm)) {
367                         // Two adapters with the same normalized name: mark ambiguous so neither
368                         // is used (empty string is the "skip LHM" sentinel for callers).
369                         map.put(norm, "");
370                     } else {
371                         map.put(norm, identifier);
372                     }
373                 }
374             }
375         } catch (Exception e) {
376             LOG.debug("LHM GPU hardware query failed (LHM may not be running): {}", e.getMessage());
377         }
378         return map;
379     }

381     /**
382      * Parses the PCI bus number from a Windows registry {@code LocationInformation} string of the form
383      * {@code "PCI bus N, device N, function N"}.
384      *
385      * @param locationInfo the LocationInformation registry value
386      * @return PCI bus number, or -1 if not parseable
387      */
388     static int parsePciBusNumber(String locationInfo) {
389         if (locationInfo == null || locationInfo.isEmpty()) {
390             return -1;
391         }
392         // Format: "PCI bus N, device N, function N" (case-insensitive)
393         String lower = locationInfo.toLowerCase(Locale.ROOT);
394         int busIdx = lower.indexOf("pci bus ");
395         if (busIdx < 0) {
396             return -1;
397         }
398         int start = busIdx + 8;
399         int end = lower.indexOf(',', start);
400         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
401         return ParseUtil.parseIntOrDefault(numStr, -1);
402     }

404     /**
405      * Parses the PCI device number from a Windows registry {@code LocationInformation} string.
406      *
407      * @param locationInfo the LocationInformation registry value
408      * @return PCI device number, or -1 if not parseable
409      */
410     static int parsePciDevice(String locationInfo) {
411         if (locationInfo == null || locationInfo.isEmpty()) {
412             return -1;
413         }
414         String lower = locationInfo.toLowerCase(Locale.ROOT);
415         int devIdx = lower.indexOf("device ");
416         if (devIdx < 0) {
417             return -1;
418         }
419         int start = devIdx + 7;
420         int end = lower.indexOf(',', start);
421         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
422         return ParseUtil.parseIntOrDefault(numStr, -1);
423     }

425     /**
426      * Parses the PCI function number from a Windows registry {@code LocationInformation} string.
427      *
428      * @param locationInfo the LocationInformation registry value
429      * @return PCI function number, or -1 if not parseable
430      */
431     static int parsePciFunction(String locationInfo) {
432         if (locationInfo == null || locationInfo.isEmpty()) {
433             return -1;
434         }
435         String lower = locationInfo.toLowerCase(Locale.ROOT);
436         int fnIdx = lower.indexOf("function ");
437         if (fnIdx < 0) {
438             return -1;
439         }
440         int start = fnIdx + 9;
441         int end = lower.indexOf(',', start);
442         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
443         return ParseUtil.parseIntOrDefault(numStr, -1);
444     }

446     /**
447      * Builds a PCI bus ID string in {@code "0000:BB:DD.F"} format from a Windows registry {@code LocationInformation}
448      * string. Returns an empty string if any component cannot be parsed.
449      *
450      * @param locationInfo the LocationInformation registry value
451      * @return PCI bus ID string, or empty string if not parseable
452      */
453     static String buildPciBusId(String locationInfo) {
454         int bus = parsePciBusNumber(locationInfo);
455         int device = parsePciDevice(locationInfo);
456         int function = parsePciFunction(locationInfo);
457         if (bus < 0 || device < 0 || function < 0) {
458             return "";
459         }
460         return String.format(Locale.ROOT, "0000:%02x:%02x.%x", bus, device, function);
461     }

463     /**
464      * Builds the PDH LUID instance prefix for the given DXGI adapter. The prefix has the form
465      * {@code luid_0xHHHHHHHH_0xLLLLLLLL_phys_0} matching the Windows PDH GPU Engine and GPU Adapter Memory counter
466      * instance names.
467      *
468      * <p>
469      * The LUID is read directly from {@code DXGI_ADAPTER_DESC.AdapterLuid}, so this method works correctly on multi-GPU
470      * systems. A zero LUID (both parts zero) indicates the adapter did not supply a valid LUID; in that case an empty
471      * string is returned and PDH metrics will report {@code -1}.
472      *
473      * @param adapter the DXGI adapter info containing the LUID
474      * @return PDH LUID instance prefix string, or empty string if the LUID is zero
475      */
476     private static String buildLuidPrefix(DxgiAdapterInfo adapter) {
477         int low = adapter.getLuidLowPart();
478         int high = adapter.getLuidHighPart();
479         if (low == 0 && high == 0) {
480             // Zero LUID is invalid; fall back to PDH enumeration for single-GPU case.
481             return buildLuidPrefixFromPdh();
482         }
483         return String.format(Locale.ROOT, "luid_0x%08x_0x%08x_phys_0", high, low);
484     }

486     /**
487      * Fallback LUID prefix discovery by enumerating PDH GPU Adapter Memory instances. Used when the DXGI adapter
488      * reports a zero LUID.
489      *
490      * <p>
491      * Only reliable on single-GPU systems: when multiple GPU Adapter Memory instances are present, the correct
492      * per-adapter mapping cannot be determined without a valid LUID, so an empty string is returned and PDH metrics
493      * will report {@code -1}.
494      *
495      * @return PDH LUID instance prefix string, or empty string if not determinable
496      */
497     private static String buildLuidPrefixFromPdh() {
498         Pair<List<String>, Map<GpuAdapterMemoryProperty, List<Long>>> adapterData = GpuInformation
499                 .queryGpuAdapterMemoryCounters();
500         List<String> instances = adapterData.getA();
501         if (instances.isEmpty()) {
502             return "";
503         }
504         // GPU Adapter Memory instance names have the form: luid_0xHHHH_0xLLLL_phys_0
505         // There is one instance per physical adapter. We return the full instance name as the prefix.
506         // If there is only one adapter, use it directly.
507         if (instances.size() == 1) {
508             return instances.get(0);
509         }
510         // Multiple adapters: we cannot reliably match by name here since GPU Adapter Memory
511         // instances do not carry a name. Return empty; callers will get -1 for PDH metrics.
512         // A future improvement could correlate via DXGI LUID enumeration.
513         LOG.debug("Multiple GPU Adapter Memory instances found ({}); LUID matching not yet implemented for multi-GPU",
514                 instances.size());
515         return "";
516     }
