36      private static final String IOUSB = "IOUSB";
37      private static final String IOSERVICE = "IOService";
38      private static final Logger LOG = LoggerFactory.getLogger(MacUsbDeviceFFM.class);

40      public MacUsbDeviceFFM(String name, String vendor, String vendorId, String productId, String serialNumber,
41              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
42          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);

45      public static List<UsbDevice> getUsbDevices(boolean tree) {
46          List<UsbDevice> devices = getUsbDevices();
47          if (tree) {
48              return devices;
49          }
50          List<UsbDevice> deviceList = new ArrayList<>();
51          for (UsbDevice device : devices) {
52              addDevicesToList(deviceList, device.getConnectedDevices());
53          }
54          return deviceList;
55      }

57      private static List<UsbDevice> getUsbDevices() {
58          Map<Long, String> nameMap = new HashMap<>();
59          Map<Long, String> vendorMap = new HashMap<>();
60          Map<Long, String> vendorIdMap = new HashMap<>();
61          Map<Long, String> productIdMap = new HashMap<>();
62          Map<Long, String> serialMap = new HashMap<>();
63          Map<Long, List<Long>> hubMap = new HashMap<>();
64  
65          List<Long> usbControllers = new ArrayList<>();
66          IORegistryEntry root = IOKitUtilFFM.getRoot();
67          if (root == null) {
68              return Collections.emptyList();
69          }
70          IOIterator iter = root.getChildIterator(IOUSB);
71          if (iter != null) {
72              CFStringRef locationIDKey = CFStringRef.createCFString("locationID");
73              CFStringRef ioPropertyMatchKey = CFStringRef.createCFString("IOPropertyMatch");
74              try {
75                  IORegistryEntry device = iter.next();
76                  while (device != null) {
77                      long id = 0L;
78                      IORegistryEntry controller = device.getParentEntry(IOSERVICE);
79                      if (controller != null) {
80                          id = controller.getRegistryEntryID();
81                          nameMap.put(id, controller.getName());
82                          MemorySegment ref = controller.createCFProperty(locationIDKey.segment());
83                          if (ref != null && !ref.equals(MemorySegment.NULL)) {
84                              CFTypeRef locationId = new CFTypeRef(ref);
85                              getControllerIdByLocation(id, locationId, locationIDKey, ioPropertyMatchKey, vendorIdMap,
86                                      productIdMap);
87                              locationId.release();
88                          }
89                          controller.release();
90                      }
91                      usbControllers.add(id);
92                      addDeviceAndChildrenToMaps(device, id, nameMap, vendorMap, vendorIdMap, productIdMap, serialMap,
93                              hubMap);
94                      device.release();
95                      device = iter.next();
96                  }
97              } finally {
98                  locationIDKey.release();
99                  ioPropertyMatchKey.release();
100                 iter.release();
101             }
102         }
103         root.release();
104 
105         List<UsbDevice> controllerDevices = new ArrayList<>();
106         for (Long controller : usbControllers) {
107             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
108                     productIdMap, serialMap, hubMap));
109         }
110         return controllerDevices;
111     }

113     private static void addDeviceAndChildrenToMaps(IORegistryEntry device, long parentId, Map<Long, String> nameMap,
114             Map<Long, String> vendorMap, Map<Long, String> vendorIdMap, Map<Long, String> productIdMap,
115             Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
116         long id = device.getRegistryEntryID();
117         hubMap.computeIfAbsent(parentId, x -> new ArrayList<>()).add(id);
118         nameMap.put(id, device.getName().trim());
119         String vendor = device.getStringProperty("USB Vendor Name");
120         if (vendor != null) {
121             vendorMap.put(id, vendor.trim());
122         }
123         Long vendorId = device.getLongProperty("idVendor");
124         if (vendorId != null) {
125             vendorIdMap.put(id, String.format(Locale.ROOT, "%04x", 0xffff & vendorId));
126         }
127         Long productId = device.getLongProperty("idProduct");
128         if (productId != null) {
129             productIdMap.put(id, String.format(Locale.ROOT, "%04x", 0xffff & productId));
130         }
131         String serial = device.getStringProperty("USB Serial Number");
132         if (serial != null) {
133             serialMap.put(id, serial.trim());
134         }
135 
136         IOIterator childIter = device.getChildIterator(IOUSB);
137         if (childIter != null) {
138             try {
139                 IORegistryEntry childDevice = childIter.next();
140                 while (childDevice != null) {
141                     addDeviceAndChildrenToMaps(childDevice, id, nameMap, vendorMap, vendorIdMap, productIdMap,
142                             serialMap, hubMap);
143                     childDevice.release();
144                     childDevice = childIter.next();
145                 }
146             } finally {
147                 childIter.release();
148             }
149         }
150     }

152     private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
153         for (UsbDevice device : list) {
154             deviceList.add(new MacUsbDeviceFFM(device.getName(), device.getVendor(), device.getVendorId(),
155                     device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(),
156                     Collections.emptyList()));
157             addDevicesToList(deviceList, device.getConnectedDevices());
158         }
159     }

161     private static void getControllerIdByLocation(long id, CFTypeRef locationId, CFStringRef locationIDKey,
162             CFStringRef ioPropertyMatchKey, Map<Long, String> vendorIdMap, Map<Long, String> productIdMap) {
163         // Build matching dict: { IOPropertyMatch: { locationID: <locationId> } }
164         try {
165             CFAllocatorRef alloc = new CFAllocatorRef(CoreFoundationFunctions.CFAllocatorGetDefault());
166             CFMutableDictionaryRef propertyDict = new CFMutableDictionaryRef(CoreFoundationFunctions
167                     .CFDictionaryCreateMutable(alloc.segment(), 0, MemorySegment.NULL, MemorySegment.NULL));
168             propertyDict.setValue(locationIDKey, locationId);
169             CFMutableDictionaryRef matchingDict = new CFMutableDictionaryRef(CoreFoundationFunctions
170                     .CFDictionaryCreateMutable(alloc.segment(), 0, MemorySegment.NULL, MemorySegment.NULL));
171             matchingDict.setValue(ioPropertyMatchKey, propertyDict);
172 
173             IOIterator serviceIterator = IOKitUtilFFM.getMatchingServices(matchingDict.segment());
174             propertyDict.release();
175 
176             boolean found = false;
177             if (serviceIterator != null) {
178                 try {
179                     IORegistryEntry matchingService = serviceIterator.next();
180                     while (matchingService != null && !found) {
181                         IORegistryEntry parent = matchingService.getParentEntry(IOSERVICE);
182                         if (parent != null) {
183                             byte[] vid = parent.getByteArrayProperty("vendor-id");
184                             if (vid != null && vid.length >= 2) {
185                                 vendorIdMap.put(id, String.format(Locale.ROOT, "%02x%02x", vid[1], vid[0]));
186                                 found = true;
187                             }
188                             byte[] pid = parent.getByteArrayProperty("device-id");
189                             if (pid != null && pid.length >= 2) {
190                                 productIdMap.put(id, String.format(Locale.ROOT, "%02x%02x", pid[1], pid[0]));
191                                 found = true;
192                             }
193                             parent.release();
194                         }
195                         matchingService.release();
196                         matchingService = serviceIterator.next();
197                     }
198                 } finally {
199                     serviceIterator.release();
200                 }
201             }
202         } catch (Throwable e) {
203             LOG.debug("Failed to retrieve controller vendor/product IDs for id {}", id, e);
204         }
205     }

207     private static MacUsbDeviceFFM getDeviceAndChildren(Long registryEntryId, String vid, String pid,
208             Map<Long, String> nameMap, Map<Long, String> vendorMap, Map<Long, String> vendorIdMap,
209             Map<Long, String> productIdMap, Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
210         String vendorId = vendorIdMap.getOrDefault(registryEntryId, vid);
211         String productId = productIdMap.getOrDefault(registryEntryId, pid);
212         List<Long> childIds = hubMap.getOrDefault(registryEntryId, new ArrayList<>());
213         List<UsbDevice> usbDevices = new ArrayList<>();
214         for (Long childId : childIds) {
215             usbDevices.add(getDeviceAndChildren(childId, vendorId, productId, nameMap, vendorMap, vendorIdMap,
216                     productIdMap, serialMap, hubMap));
217         }
218         Collections.sort(usbDevices);
219         return new MacUsbDeviceFFM(nameMap.getOrDefault(registryEntryId, vendorId + ":" + productId),
220                 vendorMap.getOrDefault(registryEntryId, ""), vendorId, productId,
221                 serialMap.getOrDefault(registryEntryId, ""), "0x" + Long.toHexString(registryEntryId), usbDevices);
222     }
