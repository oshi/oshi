5   package oshi.hardware.platform.mac;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.ffm.mac.MacSystem;
11  import oshi.util.ParseUtil;
12  import oshi.util.platform.mac.SysctlUtilFFM;
13  import oshi.util.tuples.Pair;
14  
15  import java.lang.foreign.Arena;
16  import java.lang.foreign.MemorySegment;
17  
18  import static java.lang.foreign.ValueLayout.JAVA_INT;
19  import static java.lang.foreign.ValueLayout.JAVA_LONG;
20  import static oshi.ffm.mac.MacSystem.VM_STATISTICS;
21  import static oshi.ffm.mac.MacSystem.XSW_USAGE_TOTAL;
22  import static oshi.ffm.mac.MacSystem.XSW_USAGE_USED;
23  
24  @ThreadSafe
25  final class MacVirtualMemoryFFM extends MacVirtualMemory {
26      private static final Logger LOG = LoggerFactory.getLogger(MacVirtualMemoryFFM.class);
27  
28      MacVirtualMemoryFFM(MacGlobalMemoryFFM macGlobalMemory) {
29          super(macGlobalMemory);
30      }
31  
32      @Override
33      protected Pair<Long, Long> querySwapUsage() {
34          long swapUsed = 0L;
35          long swapTotal = 0L;
36          try (Arena arena = Arena.ofConfined()) {
37              var xswUsage = arena.allocate(MacSystem.XSW_USAGE);
38              if (SysctlUtilFFM.sysctl("vm.swapusage", xswUsage)) {
39                  swapUsed = xswUsage.get(JAVA_LONG, MacSystem.XSW_USAGE.byteOffset(XSW_USAGE_USED));
40                  swapTotal = xswUsage.get(JAVA_LONG, MacSystem.XSW_USAGE.byteOffset(XSW_USAGE_TOTAL));
41              }
42          }
43          return new Pair<>(swapUsed, swapTotal);
44      }
45  
46      @Override
47      protected Pair<Long, Long> queryVmStat() {
48          try (Arena arena = Arena.ofConfined()) {
49              // Allocate memory for VM statistics structure and count
50              MemorySegment vmStats = arena.allocate(VM_STATISTICS);
51              if (MacMemoryUtilFFM.callVmStat(arena, vmStats)) {
52                  long swapPagesIn = ParseUtil.unsignedIntToLong(
53                          vmStats.get(JAVA_INT, MacSystem.VM_STATISTICS.byteOffset(MacSystem.VM_PAGEINS)));
54                  long swapPagesOut = ParseUtil.unsignedIntToLong(
55                          vmStats.get(JAVA_INT, MacSystem.VM_STATISTICS.byteOffset(MacSystem.VM_PAGEOUTS)));
56                  return new Pair<>(swapPagesIn, swapPagesOut);
57              }
58          } catch (Throwable e) {
59              // Ignored
60          }
61          return new Pair<>(0L, 0L);
62      }
63  }
