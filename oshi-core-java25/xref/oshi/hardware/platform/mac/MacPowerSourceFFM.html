50      public static List<PowerSource> getPowerSources() {
51          String psDeviceName = Constants.UNKNOWN;
52          double psTimeRemainingInstant = 0d;
53          double psPowerUsageRate = 0d;
54          double psVoltage = -1d;
55          double psAmperage = 0d;
56          boolean psPowerOnLine = false;
57          boolean psCharging = false;
58          boolean psDischarging = false;
59          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
60          int psCurrentCapacity = 0;
61          int psMaxCapacity = 1;
62          int psDesignCapacity = -1;
63          int psCycleCount = -1;
64          String psChemistry = Constants.UNKNOWN;
65          LocalDate psManufactureDate = null;
66          String psManufacturer = Constants.UNKNOWN;
67          String psSerialNumber = Constants.UNKNOWN;
68          double psTemperature = 0d;
69  
70          IOService smartBattery = IOKitUtilFFM.getMatchingService("AppleSmartBattery");
71          if (smartBattery != null) {
72              try {
73                  String s = smartBattery.getStringProperty("DeviceName");
74                  if (s != null) {
75                      psDeviceName = s;
76                  }
77                  s = smartBattery.getStringProperty("Manufacturer");
78                  if (s != null) {
79                      psManufacturer = s;
80                  }
81                  s = smartBattery.getStringProperty("BatterySerialNumber");
82                  if (s != null) {
83                      psSerialNumber = s;
84                  }
85  
86                  Integer temp = smartBattery.getIntegerProperty("ManufactureDate");
87                  if (temp != null) {
88                      int day = temp & 0x1f;
89                      int month = (temp >> 5) & 0xf;
90                      int year80 = (temp >> 9) & 0x7f;
91                      try {
92                          psManufactureDate = LocalDate.of(1980 + year80, month, day);
93                      } catch (DateTimeException e) {
94                          // Corrupt bitfield — leave psManufactureDate as null
95                      }
96                  }
97  
98                  temp = smartBattery.getIntegerProperty("DesignCapacity");
99                  if (temp != null) {
100                     psDesignCapacity = temp;
101                 }
102                 temp = smartBattery.getIntegerProperty("MaxCapacity");
103                 if (temp != null) {
104                     psMaxCapacity = temp;
105                 }
106                 temp = smartBattery.getIntegerProperty("CurrentCapacity");
107                 if (temp != null) {
108                     psCurrentCapacity = temp;
109                 }
110                 psCapacityUnits = CapacityUnits.MAH;
111 
112                 temp = smartBattery.getIntegerProperty("TimeRemaining");
113                 if (temp != null) {
114                     psTimeRemainingInstant = temp * 60d;
115                 }
116                 temp = smartBattery.getIntegerProperty("CycleCount");
117                 if (temp != null) {
118                     psCycleCount = temp;
119                 }
120                 temp = smartBattery.getIntegerProperty("Temperature");
121                 if (temp != null) {
122                     psTemperature = temp / 100d;
123                 }
124                 temp = smartBattery.getIntegerProperty("Voltage");
125                 if (temp != null) {
126                     psVoltage = temp / 1000d;
127                 }
128                 temp = smartBattery.getIntegerProperty("Amperage");
129                 if (temp != null) {
130                     psAmperage = temp;
131                 }
132                 psPowerUsageRate = psVoltage * psAmperage;
133 
134                 Boolean bool = smartBattery.getBooleanProperty("ExternalConnected");
135                 if (bool != null) {
136                     psPowerOnLine = bool;
137                 }
138                 bool = smartBattery.getBooleanProperty("IsCharging");
139                 if (bool != null) {
140                     psCharging = bool;
141                 }
142                 psDischarging = !psCharging && !psPowerOnLine;
143             } finally {
144                 smartBattery.release();
145             }
146         }
147 
148         // Capture final copies for use in lambda/inner scope
149         final String fDeviceName = psDeviceName;
150         final double fTimeRemainingInstant = psTimeRemainingInstant;
151         final double fPowerUsageRate = psPowerUsageRate;
152         final double fVoltage = psVoltage;
153         final double fAmperage = psAmperage;
154         final boolean fPowerOnLine = psPowerOnLine;
155         final boolean fCharging = psCharging;
156         final boolean fDischarging = psDischarging;
157         final CapacityUnits fCapacityUnits = psCapacityUnits;
158         final int fCurrentCapacity = psCurrentCapacity;
159         final int fMaxCapacity = psMaxCapacity;
160         final int fDesignCapacity = psDesignCapacity;
161         final int fCycleCount = psCycleCount;
162         final String fChemistry = psChemistry;
163         final LocalDate fManufactureDate = psManufactureDate;
164         final String fManufacturer = psManufacturer;
165         final String fSerialNumber = psSerialNumber;
166         final double fTemperature = psTemperature;
167 
168         MemorySegment powerSourcesInfo = MemorySegment.NULL;
169         CFArrayRef cfList = null;
170         CFStringRef nameKey = null;
171         CFStringRef isPresentKey = null;
172         CFStringRef currentCapacityKey = null;
173         CFStringRef maxCapacityKey = null;
174         try {
175             powerSourcesInfo = IOPSCopyPowerSourcesInfo();
176             cfList = new CFArrayRef(IOPSCopyPowerSourcesList(powerSourcesInfo));
177             double psTimeRemainingEstimated = IOPSGetTimeRemainingEstimate();
178             int powerSourcesCount = cfList.getCount();
179 
180             nameKey = CFStringRef.createCFString("Name");
181             isPresentKey = CFStringRef.createCFString("Is Present");
182             currentCapacityKey = CFStringRef.createCFString("Current Capacity");
183             maxCapacityKey = CFStringRef.createCFString("Max Capacity");
184 
185             List<PowerSource> psList = new ArrayList<>(powerSourcesCount);
186             for (int ps = 0; ps < powerSourcesCount; ps++) {
187                 MemorySegment pwrSrcPtr = cfList.getValueAtIndex(ps);
188                 MemorySegment dictionary = IOPSGetPowerSourceDescription(powerSourcesInfo, pwrSrcPtr);
189                 CFDictionaryRef dict = new CFDictionaryRef(dictionary);
190 
191                 MemorySegment result = dict.getValue(isPresentKey);
192                 if (!result.equals(MemorySegment.NULL)) {
193                     CFBooleanRef isPresentRef = new CFBooleanRef(result);
194                     if (isPresentRef.booleanValue()) {
195                         result = dict.getValue(nameKey);
196                         String psName = CFUtilFFM.cfPointerToString(result);
197 
198                         double currentCapacity = 0d;
199                         if (dict.getValueIfPresent(currentCapacityKey, MemorySegment.NULL)) {
200                             result = dict.getValue(currentCapacityKey);
201                             currentCapacity = new CFNumberRef(result).intValue();
202                         }
203                         double maxCapacity = 1d;
204                         if (dict.getValueIfPresent(maxCapacityKey, MemorySegment.NULL)) {
205                             result = dict.getValue(maxCapacityKey);
206                             maxCapacity = new CFNumberRef(result).intValue();
207                         }
208                         double psRemainingCapacityPercent = maxCapacity <= 0 ? 0d
209                                 : Math.min(1d, currentCapacity / maxCapacity);
210 
211                         psList.add(new MacPowerSourceFFM(psName, fDeviceName, psRemainingCapacityPercent,
212                                 psTimeRemainingEstimated, fTimeRemainingInstant, fPowerUsageRate, fVoltage, fAmperage,
213                                 fPowerOnLine, fCharging, fDischarging, fCapacityUnits, fCurrentCapacity, fMaxCapacity,
214                                 fDesignCapacity, fCycleCount, fChemistry, fManufactureDate, fManufacturer,
215                                 fSerialNumber, fTemperature));
216                     }
217                 }
218             }
219             return psList;
220         } catch (Throwable e) {
221             return new ArrayList<>();
222         } finally {
223             if (isPresentKey != null) {
224                 isPresentKey.release();
225             }
226             if (nameKey != null) {
227                 nameKey.release();
228             }
229             if (currentCapacityKey != null) {
230                 currentCapacityKey.release();
231             }
232             if (maxCapacityKey != null) {
233                 maxCapacityKey.release();
234             }
235             if (cfList != null) {
236                 cfList.release();
237             }
238             if (!powerSourcesInfo.equals(MemorySegment.NULL)) {
239                 // powerSourcesInfo owns powerSourcesList; release info last
240                 try {
241                     CoreFoundationFunctions.CFRelease(powerSourcesInfo);
242                 } catch (Throwable ignored) {
243                     // CFRelease declares throws Throwable; swallow to keep finally clean
244                 }
245             }
246         }
247     }
