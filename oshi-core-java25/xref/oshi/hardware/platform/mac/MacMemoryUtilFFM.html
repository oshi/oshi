5   package oshi.hardware.platform.mac;

7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   import oshi.ffm.mac.MacSystem;
10  
11  import java.lang.foreign.Arena;
12  import java.lang.foreign.MemorySegment;
13  
14  import static java.lang.foreign.ValueLayout.JAVA_INT;
15  import static oshi.ffm.ForeignFunctions.CAPTURED_STATE_LAYOUT;
16  import static oshi.ffm.ForeignFunctions.getErrno;
17  import static oshi.ffm.mac.MacSystem.VM_STATISTICS;
18  import static oshi.ffm.mac.MacSystemFunctions.host_statistics;
19  import static oshi.ffm.mac.MacSystemFunctions.mach_host_self;
20  
21  final class MacMemoryUtilFFM {
22  
23      private static final Logger LOG = LoggerFactory.getLogger(MacMemoryUtilFFM.class);
24  
25      private MacMemoryUtilFFM() {
26      }
27  
28      /**
29       * Call host_statistics and put the result in vmStats.
30       *
31       * @param arena   the memory arena
32       * @param vmStats the VM statistics memory segment
33       * @return {@code true} when the call succeeded, {@code false} otherwise.
34       */
35      static boolean callVmStat(Arena arena, MemorySegment vmStats) throws Throwable {
36          MemorySegment count = arena.allocate(JAVA_INT);
37          MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
38          // Set the count to the size of the VM statistics structure in integers
39          count.set(JAVA_INT, 0, (int) (VM_STATISTICS.byteSize() / JAVA_INT.byteSize()));
40  
41          int result = host_statistics(callState, mach_host_self(), MacSystem.HOST_VM_INFO, vmStats, count);
42          if (result != 0) {
43              LOG.error("Failed to get host VM info. Error code: {}", getErrno(callState));
44              return false;
45          }
46          return true;
47      }

49  }
