21  /**
22   * Firmware data obtained from ioreg.
23   */
24  @Immutable
25  final class MacFirmwareFFM extends AbstractFirmware {
26  
27      private final Supplier<Quintet<String, String, String, String, String>> manufNameDescVersRelease = memoize(
28              MacFirmwareFFM::queryEfi);
29  
30      @Override
31      public String getManufacturer() {
32          return manufNameDescVersRelease.get().getA();
33      }
34  
35      @Override
36      public String getName() {
37          return manufNameDescVersRelease.get().getB();
38      }
39  
40      @Override
41      public String getDescription() {
42          return manufNameDescVersRelease.get().getC();
43      }
44  
45      @Override
46      public String getVersion() {
47          return manufNameDescVersRelease.get().getD();
48      }
49  
50      @Override
51      public String getReleaseDate() {
52          return manufNameDescVersRelease.get().getE();
53      }
54  
55      private static Quintet<String, String, String, String, String> queryEfi() {
56          String manufacturer = null;
57          String name = null;
58          String description = null;
59          String version = null;
60          String releaseDate = null;
61  
62          IORegistryEntry platformExpert = IOKitUtilFFM.getMatchingService("IOPlatformExpertDevice");
63          if (platformExpert != null) {
64              try {
65                  IOIterator iter = platformExpert.getChildIterator("IODeviceTree");
66                  if (iter != null) {
67                      try {
68                          IORegistryEntry entry = iter.next();
69                          while (entry != null) {
70                              try {
71                                  String entryName = entry.getName();
72                                  if (entryName != null) {
73                                      switch (entryName) {
74                                          case "rom":
75                                              byte[] data = entry.getByteArrayProperty("vendor");
76                                              if (data != null) {
77                                                  manufacturer = toUtf8(data);
78                                              }
79                                              data = entry.getByteArrayProperty("version");
80                                              if (data != null) {
81                                                  version = toUtf8(data);
82                                              }
83                                              data = entry.getByteArrayProperty("release-date");
84                                              if (data != null) {
85                                                  releaseDate = toUtf8(data);
86                                              }
87                                              break;
88                                          case "chosen":
89                                              data = entry.getByteArrayProperty("booter-name");
90                                              if (data != null) {
91                                                  name = toUtf8(data);
92                                              }
93                                              break;
94                                          case "efi":
95                                              data = entry.getByteArrayProperty("firmware-abi");
96                                              if (data != null) {
97                                                  description = toUtf8(data);
98                                              }
99                                              break;
100                                         default:
101                                             if (Util.isBlank(name)) {
102                                                 name = entry.getStringProperty("IONameMatch");
103                                             }
104                                             break;
105                                     }
106                                 }
107                             } finally {
108                                 entry.release();
109                             }
110                             entry = iter.next();
111                         }
112                     } finally {
113                         iter.release();
114                     }
115                 }
116                 if (Util.isBlank(manufacturer)) {
117                     byte[] data = platformExpert.getByteArrayProperty("manufacturer");
118                     if (data != null) {
119                         manufacturer = toUtf8(data);
120                     }
121                 }
122                 if (Util.isBlank(version)) {
123                     byte[] data = platformExpert.getByteArrayProperty("target-type");
124                     if (data != null) {
125                         version = toUtf8(data);
126                     }
127                 }
128                 if (Util.isBlank(name)) {
129                     byte[] data = platformExpert.getByteArrayProperty("device_type");
130                     if (data != null) {
131                         name = toUtf8(data);
132                     }
133                 }
134             } finally {
135                 platformExpert.release();
136             }
137         }
138         return new Quintet<>(Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer,
139                 Util.isBlank(name) ? Constants.UNKNOWN : name,
140                 Util.isBlank(description) ? Constants.UNKNOWN : description,
141                 Util.isBlank(version) ? Constants.UNKNOWN : version,
142                 Util.isBlank(releaseDate) ? Constants.UNKNOWN : releaseDate);
143     }
144 
145     private static String toUtf8(byte[] data) {
146         return new String(data, StandardCharsets.UTF_8).trim().replace("\0", "");
147     }
148 }
