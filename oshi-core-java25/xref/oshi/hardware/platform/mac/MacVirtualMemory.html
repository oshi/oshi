5   package oshi.hardware.platform.mac;
6   
7   import static oshi.util.Memoizer.defaultExpiration;
8   import static oshi.util.Memoizer.memoize;
9   
10  import java.util.function.Supplier;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.hardware.common.AbstractVirtualMemory;
14  import oshi.util.tuples.Pair;
15  
16  /**
17   * Memory obtained by host_statistics (vm_stat) and sysctl.
18   */
19  @ThreadSafe
20  abstract class MacVirtualMemory extends AbstractVirtualMemory {
21  
22      private final MacGlobalMemory global;
23  
24      private final Supplier<Pair<Long, Long>> usedTotal = memoize(this::querySwapUsage, defaultExpiration());
25  
26      private final Supplier<Pair<Long, Long>> inOut = memoize(this::queryVmStat, defaultExpiration());
27  
28      /**
29       * Constructor for MacVirtualMemory.
30       *
31       * @param macGlobalMemory The parent global memory class instantiating this
32       */
33      MacVirtualMemory(MacGlobalMemory macGlobalMemory) {
34          this.global = macGlobalMemory;
35      }
36  
37      @Override
38      public long getSwapUsed() {
39          return usedTotal.get().getA();
40      }
41  
42      @Override
43      public long getSwapTotal() {
44          return usedTotal.get().getB();
45      }
46  
47      @Override
48      public long getVirtualMax() {
49          return this.global.getTotal() + getSwapTotal();
50      }
51  
52      @Override
53      public long getVirtualInUse() {
54          return this.global.getTotal() - this.global.getAvailable() + getSwapUsed();
55      }
56  
57      @Override
58      public long getSwapPagesIn() {
59          return inOut.get().getA();
60      }
61  
62      @Override
63      public long getSwapPagesOut() {
64          return inOut.get().getB();
65      }
66  
67      protected abstract Pair<Long, Long> querySwapUsage();
68  
69      protected abstract Pair<Long, Long> queryVmStat();
70  }
