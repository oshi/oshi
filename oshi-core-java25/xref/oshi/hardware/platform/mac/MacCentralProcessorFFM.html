60      private static final int ARM_CPUTYPE = 0x0100000C;
61      private static final int M1_CPUFAMILY = 0x1b588bb3;
62      private static final int M2_CPUFAMILY = 0xda33d83d;
63      private static final int M3_CPUFAMILY = 0x8765edea;
64      private static final long DEFAULT_FREQUENCY = 2_400_000_000L;
65      private static final Pattern CPU_N = Pattern.compile("^cpu(\\d+)");

75      @Override
76      protected ProcessorIdentifier queryProcessorId() {
77          String cpuName = SysctlUtilFFM.sysctl("machdep.cpu.brand_string", "");
78          String cpuVendor;
79          String cpuStepping;
80          String cpuModel;
81          String cpuFamily;
82          String processorID;
83          // Initial M1 chips said "Apple Processor". Later branding includes M1, M1 Pro,
84          // M1 Max, M2, etc. So if it starts with Apple it's M-something.
85          if (cpuName.startsWith("Apple")) {
86              // Processing an M1 chip
87              cpuVendor = vendor.get();
88              cpuStepping = "0"; // No correlation yet
89              cpuModel = "0"; // No correlation yet
90              int type;
91              int family;
92              if (isArmCpu) {
93                  type = ARM_CPUTYPE;
94                  int mSeries = ParseUtil.getFirstIntValue(cpuName);
95                  switch (mSeries) {
96                      case 2:
97                          family = M2_CPUFAMILY;
98                          break;
99                      case 3:
100                         family = M3_CPUFAMILY;
101                         break;
102                     default:
103                         // Some M1 did not brand as such
104                         family = M1_CPUFAMILY;
105                 }
106             } else {
107                 type = SysctlUtilFFM.sysctl("hw.cputype", 0);
108                 family = SysctlUtilFFM.sysctl("hw.cpufamily", 0);
109             }
110             // Translate to output
111             cpuFamily = String.format(Locale.ROOT, "0x%08x", family);
112             // Processor ID is an intel concept but CPU type + family conveys same info
113             processorID = String.format(Locale.ROOT, "%08x%08x", type, family);
114         } else {
115             // Processing an Intel chip
116             cpuVendor = SysctlUtilFFM.sysctl("machdep.cpu.vendor", "");
117             int i = SysctlUtilFFM.sysctl("machdep.cpu.stepping", -1);
118             cpuStepping = i < 0 ? "" : Integer.toString(i);
119             i = SysctlUtilFFM.sysctl("machdep.cpu.model", -1);
120             cpuModel = i < 0 ? "" : Integer.toString(i);
121             i = SysctlUtilFFM.sysctl("machdep.cpu.family", -1);
122             cpuFamily = i < 0 ? "" : Integer.toString(i);
123             long processorIdBits = 0L;
124             processorIdBits |= SysctlUtilFFM.sysctl("machdep.cpu.signature", 0);
125             processorIdBits |= (SysctlUtilFFM.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
126             processorID = String.format(Locale.ROOT, "%016x", processorIdBits);
127         }
128         if (isArmCpu) {
129             calculateNominalFrequencies();
130         }
131         long cpuFreq = isArmCpu ? performanceCoreFrequency : SysctlUtilFFM.sysctl("hw.cpufrequency", 0L);
132         boolean cpu64bit = SysctlUtilFFM.sysctl("hw.cpu64bit_capable", 0) != 0;
133 
134         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
135                 cpuFreq);
136     }

138     @Override
139     protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
140         int logicalProcessorCount = SysctlUtilFFM.sysctl("hw.logicalcpu", 1);
141         int physicalProcessorCount = SysctlUtilFFM.sysctl("hw.physicalcpu", 1);
142         int physicalPackageCount = SysctlUtilFFM.sysctl("hw.packages", 1);
143         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
144         Set<Integer> pkgCoreKeys = new HashSet<>();
145         for (int i = 0; i < logicalProcessorCount; i++) {
146             int coreId = i * physicalProcessorCount / logicalProcessorCount;
147             int pkgId = i * physicalPackageCount / logicalProcessorCount;
148             logProcs.add(new LogicalProcessor(i, coreId, pkgId));
149             pkgCoreKeys.add((pkgId << 16) + coreId);
150         }
151         Map<Integer, String> compatMap = queryCompatibleStrings();
152         int perflevels = SysctlUtilFFM.sysctl("hw.nperflevels", 1, false);
153         List<PhysicalProcessor> physProcs = pkgCoreKeys.stream().sorted().map(k -> {
154             String compat = compatMap.getOrDefault(k, "").toLowerCase(Locale.ROOT);
155             // This is brittle. A better long term solution is to use sysctls
156             // hw.perflevel1.physicalcpu: 2
157             // hw.perflevel0.physicalcpu: 8
158             // Note the 1 and 0 values are reversed from OSHI API definition
159             int efficiency = ARM_P_CORES.contains(compat) ? 1 : 0;
160             return new PhysicalProcessor(k >> 16, k & 0xffff, efficiency, compat);
161         }).collect(Collectors.toList());
162         List<ProcessorCache> caches = orderedProcCaches(getCacheValues(perflevels));
163         List<String> featureFlags = getFeatureFlagsFromSysctl();
164         return new Quartet<>(logProcs, physProcs, caches, featureFlags);
165     }

167     private Set<ProcessorCache> getCacheValues(int perflevels) {
168         int linesize = (int) SysctlUtilFFM.sysctl("hw.cachelinesize", 0L);
169         int l1associativity = SysctlUtilFFM.sysctl("machdep.cpu.cache.L1_associativity", 0, false);
170         int l2associativity = SysctlUtilFFM.sysctl("machdep.cpu.cache.L2_associativity", 0, false);
171         Set<ProcessorCache> caches = new HashSet<>();
172         for (int i = 0; i < perflevels; i++) {
173             int size = SysctlUtilFFM.sysctl("hw.perflevel" + i + ".l1icachesize", 0, false);
174             if (size > 0) {
175                 caches.add(new ProcessorCache(1, l1associativity, linesize, size, Type.INSTRUCTION));
176             }
177             size = SysctlUtilFFM.sysctl("hw.perflevel" + i + ".l1dcachesize", 0, false);
178             if (size > 0) {
179                 caches.add(new ProcessorCache(1, l1associativity, linesize, size, Type.DATA));
180             }
181             size = SysctlUtilFFM.sysctl("hw.perflevel" + i + ".l2cachesize", 0, false);
182             if (size > 0) {
183                 caches.add(new ProcessorCache(2, l2associativity, linesize, size, Type.UNIFIED));
184             }
185             size = SysctlUtilFFM.sysctl("hw.perflevel" + i + ".l3cachesize", 0, false);
186             if (size > 0) {
187                 caches.add(new ProcessorCache(3, 0, linesize, size, Type.UNIFIED));
188             }
189         }
190         return caches;
191     }

193     private List<String> getFeatureFlagsFromSysctl() {
194         List<String> x86Features = Stream.of("features", "extfeatures", "leaf7_features").map(f -> {
195             String key = "machdep.cpu." + f;
196             String features = SysctlUtilFFM.sysctl(key, "", false);
197             return Util.isBlank(features) ? null : (key + ": " + features);
198         }).filter(Objects::nonNull).collect(Collectors.toList());
199         return x86Features.isEmpty() ? ExecutingCommand.runNative("sysctl -a hw.optional") : x86Features;
200     }

202     @Override
203     public long[] querySystemCpuLoadTicks() {
204         long[] ticks = new long[TickType.values().length];
205         try (Arena arena = Arena.ofConfined()) {
206             MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
207             MemorySegment cpuLoadInfo = arena.allocate(MacSystem.HOST_CPU_LOAD_INFO_DATA);
208             MemorySegment count = arena.allocateFrom(ValueLayout.JAVA_INT,
209                     (int) (MacSystem.HOST_CPU_LOAD_INFO_DATA.byteSize() / MacSystem.INT_SIZE));
210             int machPort = MacSystemFunctions.mach_host_self();
211             if (0 != MacSystemFunctions.host_statistics(callState, machPort, MacSystem.HOST_CPU_LOAD_INFO, cpuLoadInfo,
212                     count)) {
213                 LOG.error("Failed to get System CPU ticks. Error code: {} ", getErrno(callState));
214                 return ticks;
215             }
216             var cpuTicksHandle = MacSystem.HOST_CPU_LOAD_INFO_DATA.varHandle(MacSystem.CPU_TICKS,
217                     MemoryLayout.PathElement.sequenceElement());
218             ticks[TickType.USER.getIndex()] = Integer
219                     .toUnsignedLong((int) cpuTicksHandle.get(cpuLoadInfo, 0L, (long) MacSystem.CPU_STATE_USER));
220             ticks[TickType.NICE.getIndex()] = Integer
221                     .toUnsignedLong((int) cpuTicksHandle.get(cpuLoadInfo, 0L, (long) MacSystem.CPU_STATE_NICE));
222             ticks[TickType.SYSTEM.getIndex()] = Integer
223                     .toUnsignedLong((int) cpuTicksHandle.get(cpuLoadInfo, 0L, (long) MacSystem.CPU_STATE_SYSTEM));
224             ticks[TickType.IDLE.getIndex()] = Integer
225                     .toUnsignedLong((int) cpuTicksHandle.get(cpuLoadInfo, 0L, (long) MacSystem.CPU_STATE_IDLE));
226         } catch (Throwable e) {
227             LOG.error("Failed to get System CPU ticks", e);
228         }
229         // Leave IOWait and IRQ values as 0
230         return ticks;
231     }

233     @Override
234     public long[] queryCurrentFreq() {
235         if (isArmCpu) {
236             Map<Integer, Long> physFreqMap = getPhysicalProcessors().stream()
237                     .collect(Collectors.toMap(PhysicalProcessor::getPhysicalProcessorNumber,
238                             p -> p.getEfficiency() > 0 ? performanceCoreFrequency : efficiencyCoreFrequency));
239             return getLogicalProcessors().stream().map(LogicalProcessor::getPhysicalProcessorNumber)
240                     .map(p -> physFreqMap.getOrDefault(p, performanceCoreFrequency)).mapToLong(f -> f).toArray();
241         }
242         return new long[] { getProcessorIdentifier().getVendorFreq() };
243     }

247         if (isArmCpu) {

250         return SysctlUtilFFM.sysctl("hw.cpufrequency_max", getProcessorIdentifier().getVendorFreq());

253     @Override
254     public double[] getSystemLoadAverage(int nelem) {
255         if (nelem < 1 || nelem > 3) {
256             throw new IllegalArgumentException("Must include from one to three elements.");
257         }
258         double[] average = new double[nelem];
259         try (Arena arena = Arena.ofConfined()) {
260             MemorySegment loadavgSeg = arena.allocate(ValueLayout.JAVA_DOUBLE, nelem);
261             int retval = MacSystemFunctions.getloadavg(loadavgSeg, nelem);
262             if (retval < nelem) {
263                 Arrays.fill(average, -1d);
264             } else {
265                 for (int i = 0; i < nelem; i++) {
266                     average[i] = loadavgSeg.getAtIndex(ValueLayout.JAVA_DOUBLE, i);
267                 }
268             }
269         } catch (Throwable e) {
270             Arrays.fill(average, -1d);
271         }
272         return average;
273     }

275     @Override
276     public long[][] queryProcessorCpuLoadTicks() {
277         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
278         try (Arena arena = Arena.ofConfined()) {
279             MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
280             MemorySegment procCountSeg = arena.allocate(ValueLayout.JAVA_INT);
281             MemorySegment procInfoPtrSeg = arena.allocate(ValueLayout.ADDRESS);
282             MemorySegment procInfoCountSeg = arena.allocate(ValueLayout.JAVA_INT);
283             int machPort = MacSystemFunctions.mach_host_self();
284             if (0 != MacSystemFunctions.host_processor_info(callState, machPort, MacSystem.PROCESSOR_CPU_LOAD_INFO,
285                     procCountSeg, procInfoPtrSeg, procInfoCountSeg)) {
286                 LOG.error("Failed to update CPU Load. Error code: {}", getErrno(callState));
287                 return ticks;
288             }
289             int procCount = procCountSeg.get(ValueLayout.JAVA_INT, 0);
290             int procInfoCount = procInfoCountSeg.get(ValueLayout.JAVA_INT, 0);
291             MemorySegment rawProcInfoPtr = procInfoPtrSeg.get(ValueLayout.ADDRESS, 0);
292             MemorySegment procInfoPtr = rawProcInfoPtr.reinterpret((long) procInfoCount * MacSystem.INT_SIZE);
293             try {
294                 if (procCount != ticks.length) {
295                     LOG.warn("host_processor_info returned {} CPUs but expected {}; capping iteration", procCount,
296                             ticks.length);
297                 }
298                 int cpuLimit = Math.min(procCount, ticks.length);
299                 for (int cpu = 0; cpu < cpuLimit; cpu++) {
300                     int offset = cpu * MacSystem.CPU_STATE_MAX;
301                     ticks[cpu][TickType.USER.getIndex()] = FormatUtil.getUnsignedInt(
302                             procInfoPtr.getAtIndex(ValueLayout.JAVA_INT, offset + MacSystem.CPU_STATE_USER));
303                     ticks[cpu][TickType.NICE.getIndex()] = FormatUtil.getUnsignedInt(
304                             procInfoPtr.getAtIndex(ValueLayout.JAVA_INT, offset + MacSystem.CPU_STATE_NICE));
305                     ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil.getUnsignedInt(
306                             procInfoPtr.getAtIndex(ValueLayout.JAVA_INT, offset + MacSystem.CPU_STATE_SYSTEM));
307                     ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil.getUnsignedInt(
308                             procInfoPtr.getAtIndex(ValueLayout.JAVA_INT, offset + MacSystem.CPU_STATE_IDLE));
309                 }
310             } finally {
311                 try {
312                     MacSystemFunctions.vm_deallocate(MacSystemFunctions.mach_task_self(), rawProcInfoPtr.address(),
313                             (long) procInfoCount * MacSystem.INT_SIZE);
314                 } catch (Throwable e) {
315                     LOG.warn("Failed to vm_deallocate processor info buffer", e);
316                 }
317             }
318         } catch (Throwable e) {
319             LOG.error("Failed to update CPU Load", e);
320         }
321         return ticks;
322     }

340     private static String platformExpert() {
341         String manufacturer = null;
342         IORegistryEntry platformExpert = IOKitUtilFFM.getMatchingService("IOPlatformExpertDevice");
343         if (platformExpert != null) {
344             try {
345                 byte[] data = platformExpert.getByteArrayProperty("manufacturer");
346                 if (data != null) {
347                     manufacturer = new String(data, StandardCharsets.UTF_8).replace("\0", "").trim();
348                 }
349             } finally {
350                 platformExpert.release();
351             }
352         }
353         return Util.isBlank(manufacturer) ? "Apple Inc." : manufacturer;
354     }

356     // Called by initProcessorCount in the constructor
357     // These populate the physical processor id strings
358     private static Map<Integer, String> queryCompatibleStrings() {
359         Map<Integer, String> compatibleStrMap = new HashMap<>();
360         // All CPUs are an IOPlatformDevice
361         // Iterate each CPU and save frequency and "compatible" strings
362         IOIterator iter = IOKitUtilFFM.getMatchingServices("IOPlatformDevice");
363         if (iter != null) {
364             try {
365                 IORegistryEntry cpu = iter.next();
366                 while (cpu != null) {
367                     try {
368                         String name = cpu.getName();
369                         if (name != null) {
370                             Matcher m = CPU_N.matcher(name.toLowerCase(Locale.ROOT));
371                             if (m.matches()) {
372                                 int procId = ParseUtil.parseIntOrDefault(m.group(1), 0);
373                                 // Compatible key is null-delimited C string array in byte array
374                                 byte[] data = cpu.getByteArrayProperty("compatible");
375                                 if (data != null) {
376                                     // Byte array is null delimited
377                                     // Example value for M2: "apple,blizzard", "ARM,v8"
378                                     compatibleStrMap.put(procId,
379                                             new String(data, StandardCharsets.UTF_8).replace('\0', ' ').trim());
380                                 }
381                             }
382                         }
383                     } finally {
384                         cpu.release();
385                     }
386                     cpu = iter.next();
387                 }
388             } finally {
389                 iter.release();
390             }
391         }
392         return compatibleStrMap;
393     }

395     // Called when initiating instance variables which occurs after constructor has
396     // populated physical processors
397     private boolean isArmCpu() {
398         return getPhysicalProcessors().stream().map(PhysicalProcessor::getIdString).anyMatch(id -> id.contains("arm"));
399     }

401     private void calculateNominalFrequencies() {
402         IOIterator iter = IOKitUtilFFM.getMatchingServices("AppleARMIODevice");
403         if (iter != null) {
404             try {
405                 IORegistryEntry device = iter.next();
406                 try {
407                     while (device != null) {
408                         if ("pmgr".equalsIgnoreCase(device.getName())) {
409                             performanceCoreFrequency = getMaxFreqFromByteArray(
410                                     device.getByteArrayProperty("voltage-states5-sram"));
411                             efficiencyCoreFrequency = getMaxFreqFromByteArray(
412                                     device.getByteArrayProperty("voltage-states1-sram"));
413                             return;
414                         }
415                         device.release();
416                         device = iter.next();
417                     }
418                 } finally {
419                     if (device != null) {
420                         device.release();
421                     }
422                 }
423             } finally {
424                 iter.release();
425             }
426         }
427     }

429     private long getMaxFreqFromByteArray(byte[] data) {
430         // Max freq is 8 bytes from the end of the array
431         if (data != null && data.length >= 8) {
432             byte[] freqData = Arrays.copyOfRange(data, data.length - 8, data.length - 4);
433             return ParseUtil.byteArrayToLong(freqData, 4, false);
434         }
435         return DEFAULT_FREQUENCY;
436     }
