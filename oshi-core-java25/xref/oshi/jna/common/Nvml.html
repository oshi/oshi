5   package oshi.jna.common;
6   
7   import com.sun.jna.Library;
8   import com.sun.jna.Pointer;
9   import com.sun.jna.Structure;
10  import com.sun.jna.Structure.FieldOrder;
11  import com.sun.jna.ptr.IntByReference;
12  import com.sun.jna.ptr.PointerByReference;
13  
14  /**
15   * JNA bindings for the NVIDIA Management Library (NVML). This class should be considered non-API as it may be removed
16   * if/when its code is incorporated into the JNA project.
17   */
18  public interface Nvml {
19  
20      int NVML_SUCCESS = 0;
21      int NVML_TEMPERATURE_GPU = 0;
22      int NVML_CLOCK_GRAPHICS = 0;
23      int NVML_CLOCK_MEM = 2;
24      int NVML_DEVICE_NAME_BUFFER_SIZE = 96;
25  
26      interface NvmlLibrary extends Library {
27          int nvmlInit_v2();
28  
29          int nvmlShutdown();
30  
31          int nvmlDeviceGetCount_v2(IntByReference deviceCount);
32  
33          int nvmlDeviceGetHandleByIndex_v2(int index, PointerByReference device);
34  
35          int nvmlDeviceGetName(Pointer device, byte[] name, int length);
36  
37          int nvmlDeviceGetPciInfo_v3(Pointer device, NvmlPciInfo pci);
38  
39          int nvmlDeviceGetUtilizationRates(Pointer device, NvmlUtilization utilization);
40  
41          int nvmlDeviceGetMemoryInfo(Pointer device, NvmlMemory memory);
42  
43          int nvmlDeviceGetTemperature(Pointer device, int sensorType, IntByReference temp);
44  
45          int nvmlDeviceGetPowerUsage(Pointer device, IntByReference power);
46  
47          int nvmlDeviceGetClockInfo(Pointer device, int clockType, IntByReference clock);
48  
49          int nvmlDeviceGetFanSpeed(Pointer device, IntByReference speed);
50      }
51  
52      @FieldOrder({ "gpu", "memory" })
53      class NvmlUtilization extends Structure {
54          public int gpu;
55          public int memory;
56      }
57  
58      @FieldOrder({ "total", "free", "used" })
59      class NvmlMemory extends Structure {
60          public long total;
61          public long free;
62          public long used;
63      }
64  
65      @FieldOrder({ "busIdLegacy", "domain", "bus", "device", "pciDeviceId", "pciSubSystemId", "busId" })
66      class NvmlPciInfo extends Structure {
67          public byte[] busIdLegacy = new byte[16];
68          public int domain;
69          public int bus;
70          public int device;
71          public int pciDeviceId;
72          public int pciSubSystemId;
73          public byte[] busId = new byte[32];
74      }
75  }
