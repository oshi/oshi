5   package oshi.jna.platform.mac;
6   
7   import com.sun.jna.Library;
8   import com.sun.jna.Native;
9   import com.sun.jna.Pointer;
10  import com.sun.jna.ptr.PointerByReference;
11  import com.sun.jna.platform.mac.CoreFoundation.CFDictionaryRef;
12  import com.sun.jna.platform.mac.CoreFoundation.CFStringRef;
13  import com.sun.jna.platform.mac.CoreFoundation.CFTypeRef;
14  
15  /**
16   * IOReport is a private Apple framework that provides access to hardware performance counters, including GPU residency
17   * and energy metrics on Apple Silicon.
18   *
19   * <p>
20   * These are non-public APIs. Mappings are provided for OSHI internal use only and should not be considered stable.
21   */
22  public interface IOReport extends Library {
23  
24      IOReport INSTANCE = Native.load("IOReport", IOReport.class);
25  
26      /**
27       * Opaque handle returned by {@link #IOReportCreateSubscription}.
28       */
29      class IOReportSubscriptionRef extends CFTypeRef {
30          public IOReportSubscriptionRef() {
31              super();
32          }
33  
34          public IOReportSubscriptionRef(Pointer p) {
35              super(p);
36          }
37      }
38  
39      /**
40       * Returns a mutable dictionary describing all channels in the given group (and optional subgroup).
41       *
42       * @param group    channel group name (e.g. {@code "GPU Stats"})
43       * @param subgroup subgroup name, or {@code null} for all subgroups
44       * @param a        reserved, pass 0
45       * @param b        reserved, pass 0
46       * @param c        reserved, pass 0
47       * @return a CFDictionaryRef channel descriptor (caller must release), or {@code null} on failure
48       */
49      CFDictionaryRef IOReportCopyChannelsInGroup(CFStringRef group, CFStringRef subgroup, long a, long b, long c);
50  
51      /**
52       * Merges the channel descriptors from {@code b} into {@code a} in place.
53       *
54       * @param a     destination channel descriptor (mutable)
55       * @param b     source channel descriptor
56       * @param null3 reserved, pass {@code null}
57       */
58      void IOReportMergeChannels(CFDictionaryRef a, CFDictionaryRef b, CFTypeRef null3);
59  
60      /**
61       * Creates a subscription for the channels described by {@code channels}. The framework writes the actually
62       * subscribed channel descriptor into {@code subscribedChannels}; callers must use that value (not the original
63       * {@code channels}) when calling {@link #IOReportCreateSamples}.
64       *
65       * @param a                  reserved, pass {@code null}
66       * @param channels           channel descriptor from {@link #IOReportCopyChannelsInGroup}
67       * @param subscribedChannels receives the subscribed channel descriptor (caller must release)
68       * @param b                  reserved, pass 0
69       * @param c                  reserved, pass {@code null}
70       * @return subscription handle (caller must release), or {@code null} on failure
71       */
72      IOReportSubscriptionRef IOReportCreateSubscription(Pointer a, CFDictionaryRef channels,
73              PointerByReference subscribedChannels, long b, CFTypeRef c);
74  
75      /**
76       * Takes a sample of all subscribed channels.
77       *
78       * @param subscription       the subscription handle
79       * @param subscribedChannels the subscribed channel descriptor from {@link #IOReportCreateSubscription}
80       * @param reserved           reserved, pass {@code null}
81       * @return a CFDictionaryRef sample (caller must release), or {@code null} on failure
82       */
83      CFDictionaryRef IOReportCreateSamples(IOReportSubscriptionRef subscription, CFDictionaryRef subscribedChannels,
84              CFTypeRef reserved);
85  
86      /**
87       * Computes the delta between two samples taken from the same subscription.
88       *
89       * @param a        earlier sample
90       * @param b        later sample
91       * @param reserved reserved, pass {@code null}
92       * @return a CFDictionaryRef delta (caller must release), or {@code null} on failure
93       */
94      CFDictionaryRef IOReportCreateSamplesDelta(CFDictionaryRef a, CFDictionaryRef b, CFTypeRef reserved);
95  
96      /**
97       * Extracts the integer value from a single channel entry in a sample or delta dictionary.
98       *
99       * @param channel  a channel entry dictionary
100      * @param reserved reserved, pass 0
101      * @return the channel's integer value
102      */
103     long IOReportSimpleGetIntegerValue(CFDictionaryRef channel, int reserved);
104 
105     /**
106      * Returns the group name of a channel entry.
107      *
108      * @param channel a channel entry dictionary
109      * @return the group name as a CFStringRef
110      */
111     CFStringRef IOReportChannelGetGroup(CFDictionaryRef channel);
112 
113     /**
114      * Returns the subgroup name of a channel entry.
115      *
116      * @param channel a channel entry dictionary
117      * @return the subgroup name as a CFStringRef
118      */
119     CFStringRef IOReportChannelGetSubGroup(CFDictionaryRef channel);
120 
121     /**
122      * Returns the channel name of a channel entry.
123      *
124      * @param channel a channel entry dictionary
125      * @return the channel name as a CFStringRef
126      */
127     CFStringRef IOReportChannelGetChannelName(CFDictionaryRef channel);
128 
129     /**
130      * Returns the number of states in a channel entry.
131      *
132      * @param channel a channel entry dictionary
133      * @return the number of states
134      */
135     int IOReportStateGetCount(CFDictionaryRef channel);
136 
137     /**
138      * Returns the name of the state at the given index.
139      *
140      * @param channel a channel entry dictionary
141      * @param index   zero-based state index
142      * @return the state name as a CFStringRef
143      */
144     CFStringRef IOReportStateGetNameForIndex(CFDictionaryRef channel, int index);
145 
146     /**
147      * Returns the residency tick count for the state at the given index.
148      *
149      * @param channel a channel entry dictionary
150      * @param index   zero-based state index
151      * @return residency ticks
152      */
153     long IOReportStateGetResidency(CFDictionaryRef channel, int index);
154 }
