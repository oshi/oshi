27  /**
28   * Minimal JNA binding to {@code dxgi.dll} for enumerating display adapters and reading
29   * {@code DXGI_ADAPTER_DESC.DedicatedVideoMemory}.
30   *
31   * <p>
32   * DXGI is the authoritative Windows API for dedicated GPU memory. It is not subject to the 2 GiB cap that affects the
33   * 32-bit registry value {@code HardwareInformation.MemorySize}.
34   *
35   * <p>
36   * COM vtable layout used here:
37   *
38   * <pre>
39   * IUnknown   (vtable slots 0-2):  QueryInterface, AddRef, Release
40   * IDXGIObject (slots 3-6):        SetPrivateData, SetPrivateDataInterface, GetPrivateData, GetParent
41   * IDXGIFactory (slots 7-13):      EnumAdapters, MakeWindowAssociation, GetWindowAssociation,
42   *                                  CreateSwapChain, CreateSoftwareAdapter, (IDXGIFactory1 adds 2 more)
43   * IDXGIAdapter (slots 3-5 on adapter): EnumOutputs, GetDesc, CheckInterfaceSupport
44   * </pre>
45   *
46   * <p>
47   * This class should be considered non-API as it may be removed if/when its code is incorporated into the JNA project.
48   */
49  public final class WindowsDxgi {
50  
51      private static final Logger LOG = LoggerFactory.getLogger(WindowsDxgi.class);

54      private static final com.sun.jna.platform.win32.Guid.IID IID_IDXGI_FACTORY = new com.sun.jna.platform.win32.Guid.IID(
55              "{7B7166EC-21C7-44AE-B21A-C9AE321AE369}");

68       * Minimal binding to {@code dxgi.dll} just to call {@code CreateDXGIFactory}.
69       */
70      private interface DxgiLib extends StdCallLibrary {
71          DxgiLib INSTANCE = Native.load("dxgi", DxgiLib.class);
72  
73          /**
74           * Creates a DXGI factory object.
75           *
76           * @param riid      IID of the factory interface to create (IDXGIFactory)
77           * @param ppFactory receives the factory pointer
78           * @return HRESULT
79           */
80          int CreateDXGIFactory(REFIID riid, PointerByReference ppFactory);
81      }

83      private static final boolean DXGI_AVAILABLE;
84      static {
85          boolean available = false;
86          try {
87              @SuppressWarnings("unused")
88              DxgiLib lib = DxgiLib.INSTANCE;
89              available = true;
90          } catch (UnsatisfiedLinkError e) {
91              LOG.debug("dxgi.dll not available: {}", e.getMessage());
92          } catch (NoClassDefFoundError e) {
93              LOG.debug("JNA DxgiLib class failed to load: {}", e.getMessage());
94          }
95          DXGI_AVAILABLE = available;
96      }

102     /**
103      * Maps to the Windows {@code DXGI_ADAPTER_DESC} structure.
104      *
105      * <p>
106      * Layout (x64):
107      * <ul>
108      * <li>Description: WCHAR[128] = 256 bytes</li>
109      * <li>VendorId: UINT = 4 bytes</li>
110      * <li>DeviceId: UINT = 4 bytes</li>
111      * <li>SubSysId: UINT = 4 bytes</li>
112      * <li>Revision: UINT = 4 bytes</li>
113      * <li>DedicatedVideoMemory: SIZE_T = 8 bytes (x64)</li>
114      * <li>DedicatedSystemMemory: SIZE_T = 8 bytes</li>
115      * <li>SharedSystemMemory: SIZE_T = 8 bytes</li>
116      * <li>AdapterLuid: LUID = 8 bytes</li>
117      * </ul>
118      */
119     @FieldOrder({ "Description", "VendorId", "DeviceId", "SubSysId", "Revision", "DedicatedVideoMemory",
120             "DedicatedSystemMemory", "SharedSystemMemory", "AdapterLuidLowPart", "AdapterLuidHighPart" })
121     public static class DXGI_ADAPTER_DESC extends Structure {
122         public char[] Description = new char[128];
123         public int VendorId;
124         public int DeviceId;
125         public int SubSysId;
126         public int Revision;
127         // SIZE_T is pointer-sized; use long (8 bytes on x64, 4 bytes on x86)
128         public com.sun.jna.platform.win32.BaseTSD.SIZE_T DedicatedVideoMemory;
129         public com.sun.jna.platform.win32.BaseTSD.SIZE_T DedicatedSystemMemory;
130         public com.sun.jna.platform.win32.BaseTSD.SIZE_T SharedSystemMemory;
131         public int AdapterLuidLowPart;
132         public int AdapterLuidHighPart;
133     }

139     /**
140      * Wraps an {@code IDXGIAdapter} COM pointer for vtable-based invocation.
141      *
142      * <p>
143      * IDXGIAdapter vtable (inherits IUnknown + IDXGIObject):
144      *
145      * <pre>
146      * slot 0  QueryInterface  (IUnknown)
147      * slot 1  AddRef          (IUnknown)
148      * slot 2  Release         (IUnknown)
149      * slot 3  SetPrivateData  (IDXGIObject)
150      * slot 4  SetPrivateDataInterface (IDXGIObject)
151      * slot 5  GetPrivateData  (IDXGIObject)
152      * slot 6  GetParent       (IDXGIObject)
153      * slot 7  EnumOutputs     (IDXGIAdapter)
154      * slot 8  GetDesc         (IDXGIAdapter)
155      * slot 9  CheckInterfaceSupport (IDXGIAdapter)
156      * </pre>
157      */
158     private static final class DxgiAdapter extends COMInvoker {
159 
160         DxgiAdapter(Pointer p) {
161             setPointer(p);
162         }
163 
164         HRESULT GetDesc(DXGI_ADAPTER_DESC desc) {
165             return (HRESULT) _invokeNativeObject(8, new Object[] { getPointer(), desc.getPointer() }, HRESULT.class);
166         }
167 
168         int Release() {
169             return _invokeNativeInt(2, new Object[] { getPointer() });
170         }
171     }

177     /**
178      * Wraps an {@code IDXGIFactory} COM pointer for vtable-based invocation.
179      *
180      * <p>
181      * IDXGIFactory vtable (inherits IUnknown + IDXGIObject):
182      *
183      * <pre>
184      * slot 0  QueryInterface  (IUnknown)
185      * slot 1  AddRef          (IUnknown)
186      * slot 2  Release         (IUnknown)
187      * slot 3  SetPrivateData  (IDXGIObject)
188      * slot 4  SetPrivateDataInterface (IDXGIObject)
189      * slot 5  GetPrivateData  (IDXGIObject)
190      * slot 6  GetParent       (IDXGIObject)
191      * slot 7  EnumAdapters    (IDXGIFactory)
192      * </pre>
193      */
194     private static final class DxgiFactory extends COMInvoker {
195 
196         DxgiFactory(Pointer p) {
197             setPointer(p);
198         }
199 
200         HRESULT EnumAdapters(int index, PointerByReference ppAdapter) {
201             return (HRESULT) _invokeNativeObject(7, new Object[] { getPointer(), index, ppAdapter.getPointer() },
202                     HRESULT.class);
203         }
204 
205         int Release() {
206             return _invokeNativeInt(2, new Object[] { getPointer() });
207         }
208     }

214     /**
215      * Enumerates all DXGI display adapters and returns their identity and dedicated video memory.
216      *
217      * <p>
218      * Fails gracefully: if {@code dxgi.dll} is unavailable or any COM call fails, returns an empty list so callers can
219      * fall back to registry-based detection.
220      *
221      * @return list of {@link DxgiAdapterInfo}, one per adapter; empty if DXGI is unavailable
222      */
223     public static List<DxgiAdapterInfo> queryAdapters() {
224         if (!DXGI_AVAILABLE) {
225             return Collections.emptyList();
226         }
227         PointerByReference ppFactory = new PointerByReference();
228         REFIID riid = new REFIID(IID_IDXGI_FACTORY);
229 
230         int hr = DxgiLib.INSTANCE.CreateDXGIFactory(riid, ppFactory);
231 
232         if (COMUtils.FAILED(new HRESULT(hr))) {
233             LOG.debug("CreateDXGIFactory failed: 0x{}", Integer.toHexString(hr));
234             return Collections.emptyList();
235         }
236 
237         DxgiFactory factory = new DxgiFactory(ppFactory.getValue());
238         List<DxgiAdapterInfo> result = new ArrayList<>();
239         try {
240             for (int i = 0;; i++) {
241                 PointerByReference ppAdapter = new PointerByReference();
242                 // EnumAdapters takes a PointerByReference but the vtable slot expects the raw
243                 // pointer-to-pointer; pass the Pointer directly via the PointerByReference value.
244                 HRESULT enumHr = factory.EnumAdapters(i, ppAdapter);
245                 if (enumHr.intValue() == DXGI_ERROR_NOT_FOUND) {
246                     break;
247                 }
248                 if (COMUtils.FAILED(enumHr)) {
249                     LOG.debug("IDXGIFactory::EnumAdapters({}) failed: 0x{}", i, Integer.toHexString(enumHr.intValue()));
250                     break;
251                 }
252 
253                 DxgiAdapter adapter = new DxgiAdapter(ppAdapter.getValue());
254                 try {
255                     DXGI_ADAPTER_DESC desc = new DXGI_ADAPTER_DESC();
256                     HRESULT descHr = adapter.GetDesc(desc);
257                     if (COMUtils.SUCCEEDED(descHr)) {
258                         desc.read();
259                         String name = Native.toString(desc.Description);
260                         long vram = desc.DedicatedVideoMemory.longValue();
261                         result.add(new DxgiAdapterInfo(name, desc.VendorId, desc.DeviceId, vram,
262                                 desc.AdapterLuidLowPart, desc.AdapterLuidHighPart));
263                         LOG.debug("DXGI adapter {}: '{}' vendor=0x{} device=0x{} vram={} luid=0x{}_0x{}", i, name,
264                                 Integer.toHexString(desc.VendorId), Integer.toHexString(desc.DeviceId), vram,
265                                 Integer.toHexString(desc.AdapterLuidHighPart),
266                                 Integer.toHexString(desc.AdapterLuidLowPart));
267                     } else {
268                         LOG.debug("IDXGIAdapter::GetDesc({}) failed: 0x{}", i, Integer.toHexString(descHr.intValue()));
269                     }
270                 } finally {
271                     adapter.Release();
272                 }
273             }
274         } finally {
275             factory.Release();
276         }
277         return Collections.unmodifiableList(result);
278     }

280     /**
281      * Finds the best-matching DXGI adapter for a given vendor ID, device ID, and adapter name.
282      *
283      * <p>
284      * Matching priority:
285      * <ol>
286      * <li>Vendor ID + Device ID (exact, both non-zero)</li>
287      * <li>Normalized name match (case-insensitive, ignoring {@code (R)}, {@code (TM)}, extra spaces)</li>
288      * </ol>
289      *
290      * <p>
291      * If multiple adapters share the same vendor+device ID (e.g. multi-GPU), the first one is returned. If no confident
292      * match is found, returns {@code null}.
293      *
294      * @param adapters    list from {@link #queryAdapters()}
295      * @param vendorId    PCI vendor ID parsed from the registry key (0 if unknown)
296      * @param deviceId    PCI device ID parsed from the registry key (0 if unknown)
297      * @param adapterName adapter name from the registry {@code DriverDesc} value
298      * @return best-matching adapter, or {@code null}
299      */
300     public static DxgiAdapterInfo findMatch(List<DxgiAdapterInfo> adapters, int vendorId, int deviceId,
301             String adapterName) {
302         // Priority 1: vendor + device ID
303         if (vendorId != 0 && deviceId != 0) {
304             for (DxgiAdapterInfo a : adapters) {
305                 if (a.getVendorId() == vendorId && a.getDeviceId() == deviceId) {
306                     return a;
307                 }
308             }
309         }
310         // Priority 2: normalized name
311         if (adapterName != null && !adapterName.isEmpty()) {
312             String norm = normalizeName(adapterName);
313             for (DxgiAdapterInfo a : adapters) {
314                 if (normalizeName(a.getDescription()).equals(norm)) {
315                     return a;
316                 }
317             }
318         }
319         return null;
320     }

322     /**
323      * Converts a registry value (REG_QWORD as Long, REG_DWORD as Integer, or REG_BINARY as byte[]) to a VRAM size in
324      * bytes. REG_BINARY is interpreted as little-endian.
325      *
326      * <p>
327      * Exposed as public for testing.
328      *
329      * @param value the registry value object
330      * @return the VRAM size in bytes, or 0 if the value type is unrecognised
331      */
332     public static long registryValueToVram(Object value) {
333         if (value instanceof Long) {
334             return (long) value;
335         } else if (value instanceof Integer) {
336             return Integer.toUnsignedLong((int) value);
337         } else if (value instanceof byte[]) {
338             byte[] bytes = (byte[]) value;
339             // Import is not available here; inline the little-endian conversion
340             long total = 0L;
341             int size = Math.min(bytes.length, 8);
342             for (int i = 0; i < size; i++) {
343                 total = total << 8 | bytes[size - i - 1] & 0xff;
344             }
345             return total;
346         }
347         return 0L;
348     }

350     /**
351      * Normalizes an adapter name for fuzzy matching: lower-case, strips {@code (R)}/{@code (TM)}, collapses whitespace.
352      *
353      * @param name the raw adapter name, may be {@code null}
354      * @return normalized name, never {@code null}
355      */
356     public static String normalizeName(String name) {
357         if (name == null) {
358             return "";
359         }
360         return name.toLowerCase(java.util.Locale.ROOT).replace("(r)", "").replace("(tm)", "").replaceAll("\\s+", " ")
361                 .trim();
362     }
