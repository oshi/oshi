29      @Test
30      void testFindMatchByVendorAndDeviceId() {
31          DxgiAdapterInfo arc = new DxgiAdapterInfo("Intel(R) Arc(TM) A770 Graphics", 0x8086, 0x56A0,
32                  16L * 1024 * 1024 * 1024, 0, 0);
33          DxgiAdapterInfo igpu = new DxgiAdapterInfo("Intel(R) UHD Graphics 770", 0x8086, 0x4680, 128L * 1024 * 1024, 0,
34                  0);
35          List<DxgiAdapterInfo> adapters = Arrays.asList(arc, igpu);
36  
37          DxgiAdapterInfo match = WindowsDxgi.findMatch(adapters, 0x8086, 0x56A0, "Intel Arc A770");
38          assertThat("Should match Arc A770 by vendor+device ID", match, is(notNullValue()));
39          assertThat(match.getDedicatedVideoMemory(), is(16L * 1024 * 1024 * 1024));
40      }
41  
42      @Test
43      void testFindMatchByVendorAndDeviceIdPicksFirst() {
44          // Two adapters with same IDs (multi-GPU) — first one wins
45          DxgiAdapterInfo first = new DxgiAdapterInfo("NVIDIA GeForce RTX 4090", 0x10DE, 0x2684, 24L * 1024 * 1024 * 1024,
46                  0, 0);
47          DxgiAdapterInfo second = new DxgiAdapterInfo("NVIDIA GeForce RTX 4090", 0x10DE, 0x2684,
48                  24L * 1024 * 1024 * 1024, 0, 0);
49          List<DxgiAdapterInfo> adapters = Arrays.asList(first, second);
50  
51          DxgiAdapterInfo match = WindowsDxgi.findMatch(adapters, 0x10DE, 0x2684, "NVIDIA GeForce RTX 4090");
52          assertThat(match, is(first));
53      }
54  
55      @Test
56      void testFindMatchByNormalizedName() {
57          // No vendor/device IDs available — fall back to name matching.
58          // The stored DXGI description has (R) and (TM) markers; the registry DriverDesc
59          // typically omits them. Verify that normalization bridges the difference.
60          DxgiAdapterInfo adapter = new DxgiAdapterInfo("Intel(R) Arc(TM) A770 Graphics", 0x8086, 0x56A0,
61                  16L * 1024 * 1024 * 1024, 0, 0);
62          List<DxgiAdapterInfo> adapters = Collections.singletonList(adapter);
63  
64          DxgiAdapterInfo match = WindowsDxgi.findMatch(adapters, 0, 0, "Intel Arc A770 Graphics");
65          assertThat("Should match by normalized name after stripping (R)/(TM)", match, is(notNullValue()));
66          assertThat(match.getDedicatedVideoMemory(), is(16L * 1024 * 1024 * 1024));
67      }
68  
69      @Test
70      void testFindMatchNoMatchReturnsNull() {
71          DxgiAdapterInfo adapter = new DxgiAdapterInfo("AMD Radeon RX 7900 XTX", 0x1002, 0x744C,
72                  24L * 1024 * 1024 * 1024, 0, 0);
73          List<DxgiAdapterInfo> adapters = Collections.singletonList(adapter);
74  
75          DxgiAdapterInfo match = WindowsDxgi.findMatch(adapters, 0x8086, 0x56A0, "Intel Arc A770");
76          assertThat("Different vendor+device should not match", match, is(nullValue()));
77      }

79      @Test
80      void testFindMatchEmptyListReturnsNull() {
81          DxgiAdapterInfo match = WindowsDxgi.findMatch(Collections.emptyList(), 0x8086, 0x56A0, "Intel Arc A770");
82          assertThat(match, is(nullValue()));
83      }

85      @Test
86      void testFindMatchZeroIdsUsesNameFallback() {
87          DxgiAdapterInfo adapter = new DxgiAdapterInfo("AMD Radeon RX 7900 XTX", 0x1002, 0x744C,
88                  24L * 1024 * 1024 * 1024, 0, 0);
89          List<DxgiAdapterInfo> adapters = Collections.singletonList(adapter);
90  
91          // vendorId=0 and deviceId=0 should skip ID matching and try name
92          DxgiAdapterInfo match = WindowsDxgi.findMatch(adapters, 0, 0, "AMD Radeon RX 7900 XTX");
93          assertThat("Should fall through to name match when IDs are zero", match, is(notNullValue()));
94      }

125     @Test
126     void testRegistryValueToVramInteger() {
127         // 0x80000000 would overflow signed int; test unsigned widening
128         assertThat(WindowsDxgi.registryValueToVram(0x80000000), is(0x80000000L));
129     }

131     @Test
132     void testRegistryValueToVramByteArrayLittleEndian() {
133         // 16 GB = 0x0000000400000000 in little-endian bytes
134         byte[] bytes = { 0x00, 0x00, 0x00, 0x00, 0x04, 0x00, 0x00, 0x00 };
135         assertThat(WindowsDxgi.registryValueToVram(bytes), is(0x0000000400000000L));
136     }

143     // -------------------------------------------------------------------------
144     // Windows-only integration test
145     // -------------------------------------------------------------------------

147     @Test
148     @EnabledOnOs(OS.WINDOWS)
149     void testQueryAdaptersOnWindows() {
150         List<DxgiAdapterInfo> adapters = WindowsDxgi.queryAdapters();
151         // An empty list is acceptable in headless or virtual environments where DXGI
152         // is unavailable; only assert per-adapter invariants when adapters are present.
153         for (DxgiAdapterInfo a : adapters) {
154             assertThat("Description should not be null", a.getDescription(), is(notNullValue()));
155             assertThat("VendorId should be non-negative", a.getVendorId(), is(greaterThanOrEqualTo(0)));
156             assertThat("DedicatedVideoMemory should be non-negative", a.getDedicatedVideoMemory(),
157                     is(greaterThanOrEqualTo(0L)));
158         }
