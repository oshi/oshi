176     /**
177      * Asserts that a raw get() call returns the expected CIM and VARIANT types.
178      *
179      * @param wql         the WQL query
180      * @param property    the property name
181      * @param expectedCim the expected CIM type
182      * @param expectedVt  the expected VARIANT type
183      */
184     private static void assertRawGet(String wql, String property, int expectedCim, int expectedVt) {
185         withFirstObject(wql, (pObject, arena) -> {
186             var result = IWbemClassObjectFFM.get(pObject, property, arena);
187             try {
188                 assertThat(result.succeeded(), is(true));
189                 assertThat(result.cimType(), is(expectedCim));
190                 assertThat(VariantFFM.getVt(result.variant()), is(expectedVt));
191             } finally {
192                 VariantFFM.clear(result.variant());
193             }
194         });
195     }

209     // -------------------------------------------------------------------------
210     // Helper: execute WQL, get first object, run assertion, clean up
211     // -------------------------------------------------------------------------
212 
213     @FunctionalInterface
214     interface WmiObjectConsumer {
215         void accept(MemorySegment pObject, Arena arena) throws Throwable;
216     }
217 
218     private static void withFirstObject(String wql, WmiObjectConsumer consumer) {
219         var hr = Ole32FFM.CoInitializeEx(Ole32FFM.COINIT_MULTITHREADED);
220         boolean comInitialized = hr.isPresent() && Ole32FFM.succeeded(hr.getAsInt());
221         try (var arena = Arena.ofConfined()) {
222             Optional<MemorySegment> locator = Optional.empty();
223             Optional<MemorySegment> services = Optional.empty();
224             Optional<MemorySegment> enumerator = Optional.empty();
225             MemorySegment object = NULL;
226             try {
227                 locator = IWbemLocatorFFM.create(arena);
228                 assertThat("Failed to create IWbemLocator", locator.isPresent(), is(true));
229 
230                 services = IWbemLocatorFFM.connectServer(locator.get(), WbemcliFFM.DEFAULT_NAMESPACE, arena);
231                 assertThat("Failed to connect to WMI", services.isPresent(), is(true));
232 
233                 Ole32FFM.CoSetProxyBlanket(services.get(), -1, -1, Ole32FFM.RPC_C_AUTHN_LEVEL_CALL,
234                         Ole32FFM.RPC_C_IMP_LEVEL_IMPERSONATE, Ole32FFM.EOAC_NONE);
235 
236                 enumerator = IWbemServicesFFM.execQuery(services.get(), wql, arena);
237                 assertThat("WQL query failed: " + wql, enumerator.isPresent(), is(true));
238 
239                 var next = IEnumWbemClassObjectFFM.next(enumerator.get(), arena);
240                 assertThat("No WMI object returned for: " + wql, next.hasObject(), is(true));
241                 object = next.pObject();
242 
243                 consumer.accept(object, arena);
244             } catch (Throwable t) {
245                 throw new AssertionError("WMI consumer failed", t);
246             } finally {
247                 if (!NULL.equals(object)) {
248                     IUnknownFFM.safeRelease(object, arena);
249                 }
250                 enumerator.ifPresent(e -> IUnknownFFM.safeRelease(e, arena));
251                 services.ifPresent(s -> IUnknownFFM.safeRelease(s, arena));
252                 locator.ifPresent(l -> IUnknownFFM.safeRelease(l, arena));
253             }
254         } finally {
255             if (comInitialized) {
256                 Ole32FFM.CoUninitialize();
257             }
258         }
259     }
