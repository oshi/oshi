5   package oshi.util;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.math.BigInteger;
11  
12  import org.junit.jupiter.api.Test;
13  
14  import oshi.util.tuples.Pair;
15  import oshi.util.tuples.Quartet;
16  import oshi.util.tuples.Quintet;
17  import oshi.util.tuples.Triplet;
18  
19  /**
20   * Test object pair.
21   */
22  class TupleTest {
23  
24      @Test
25      void testTuples() {
26          Pair<String, Integer> pair = new Pair<>("A", 1);
27          Triplet<String, Integer, Long> triplet = new Triplet<>("B", 2, Long.MAX_VALUE);
28          Quartet<String, Integer, Long, Character> quartet = new Quartet<>("C", 3, Long.MIN_VALUE, 'c');
29          Quintet<String, Integer, Long, Character, BigInteger> quintet = new Quintet<>("D", 4, Long.valueOf("0"), 'd',
30                  BigInteger.ZERO);
31  
32          assertThat("pair.getA() should be A", pair.getA(), is("A"));
33          assertThat("triplet.getA() should be B", triplet.getA(), is("B"));
34          assertThat("quartet.getA() should be C", quartet.getA(), is("C"));
35          assertThat("quintet.getA() should be D", quintet.getA(), is("D"));
36  
37          assertThat("pair.getB().intValue() should be 1", pair.getB().intValue(), is(1));
38          assertThat("triplet.getB().intValue() should be 2", triplet.getB().intValue(), is(2));
39          assertThat("quartet.getB().intValue() should be 3", quartet.getB().intValue(), is(3));
40          assertThat("quintet.getB().intValue() should be 4", quintet.getB().intValue(), is(4));
41  
42          assertThat("triplet.getC().longValue() should be Long.MAX_VALUE", triplet.getC().longValue(),
43                  is(Long.MAX_VALUE));
44          assertThat("quartet.getC().longValue() should be Long.MIN_VALUE", quartet.getC().longValue(),
45                  is(Long.MIN_VALUE));
46          assertThat("quintet.getC().longValue() should be 0L", quintet.getC().longValue(), is(0L));
47  
48          assertThat("quartet.getD().charValue() should be c", quartet.getD().charValue(), is('c'));
49          assertThat("quintet.getD().charValue() should be d", quintet.getD().charValue(), is('d'));
50  
51          assertThat("quintet.getE() should be BigInteger.ZERO", quintet.getE(), is(BigInteger.ZERO));
52      }
53  }
