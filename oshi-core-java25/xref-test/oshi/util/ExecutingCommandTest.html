5   package oshi.util;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.emptyString;
10  import static org.hamcrest.Matchers.hasSize;
11  import static org.hamcrest.Matchers.is;
12  import static org.junit.jupiter.api.parallel.ExecutionMode.SAME_THREAD;
13  
14  import java.util.List;
15  
16  import org.junit.jupiter.api.BeforeEach;
17  import org.junit.jupiter.api.Test;
18  import org.junit.jupiter.api.condition.DisabledIf;
19  import org.junit.jupiter.api.condition.EnabledOnOs;
20  import org.junit.jupiter.api.condition.OS;
21  import org.junit.jupiter.api.parallel.Execution;
22  
23  import oshi.PlatformEnum;
24  
25  /**
26   * Test command line and returning the result of execution.
27   */
28  @Execution(SAME_THREAD)
29  class ExecutingCommandTest {
30  
31      private static final String ECHO = PlatformEnum.getCurrentPlatform().equals(PlatformEnum.WINDOWS)
32              ? "cmd.exe /C echo Test"
33              : "echo Test";
34      private static final String BAD_COMMAND = "noOSshouldHaveACommandNamedThis";
35  
36      static boolean isElevated() {
37          return UserGroupInfo.isElevated();
38      }
39  
40      @BeforeEach
41      void setUp() {
42          GlobalConfig.clear();
43      }
44  
45      @Test
46      void testRunNative() {
47          List<String> test = ExecutingCommand.runNative(ECHO);
48          assertThat("echo output", test, hasSize(1));
49          assertThat("echo output", test.get(0), is("Test"));
50          assertThat("echo first answer", ExecutingCommand.getFirstAnswer(ECHO), is("Test"));
51  
52          assertThat("bad command", ExecutingCommand.runNative(BAD_COMMAND), is(empty()));
53          assertThat("bad command first answer", ExecutingCommand.getFirstAnswer(BAD_COMMAND), is(emptyString()));
54      }
55  
56      @Test
57      @DisabledIf("isElevated")
58      void testRunPrivilegedNativeNoConfig() {
59          // Without config, runPrivilegedNative runs command directly
60          List<String> result = ExecutingCommand.runPrivilegedNative(ECHO);
61          assertThat("no config should run directly", result, hasSize(1));
62      }
63  
64      @Test
65      @DisabledIf("isElevated")
66      @EnabledOnOs(OS.LINUX)
67      void testRunPrivilegedNativeNotInAllowlist() {
68          // Configure prefix but don't include command in allowlist
69          // Command should still run without prefix
70          GlobalConfig.set(GlobalConfig.OSHI_OS_LINUX_PRIVILEGED_PREFIX, "sudo -n");
71          GlobalConfig.set(GlobalConfig.OSHI_OS_LINUX_PRIVILEGED_ALLOWLIST, "lshw");
72  
73          // echo is not in allowlist, should run without prefix
74          List<String> result = ExecutingCommand.runPrivilegedNative(ECHO);
75          assertThat("command not in allowlist should run without prefix", result, hasSize(1));
76      }
77  
78      @Test
79      @DisabledIf("isElevated")
80      void testRunPrivilegedNativeRunsCommand() {
81          // Without allowlist config, command runs directly without prefix
82          // Test that runPrivilegedNative actually executes the command
83          List<String> result = ExecutingCommand.runPrivilegedNative(ECHO);
84          assertThat("command should execute", result, hasSize(1));
85          assertThat("output should be Test", result.get(0), is("Test"));
86      }
87  
88      @Test
89      @DisabledIf("isElevated")
90      void testGetFirstPrivilegedAnswer() {
91          // Test getFirstPrivilegedAnswer returns first line
92          String answer = ExecutingCommand.getFirstPrivilegedAnswer(ECHO);
93          assertThat("first privileged answer", answer, is("Test"));
94      }
95  }
