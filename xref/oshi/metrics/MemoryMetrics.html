5   package oshi.metrics;
6   
7   import oshi.hardware.GlobalMemory;
8   
9   import io.micrometer.core.instrument.Gauge;
10  import io.micrometer.core.instrument.MeterRegistry;
11  import io.micrometer.core.instrument.Tag;
12  import io.micrometer.core.instrument.Tags;
13  import io.micrometer.core.instrument.binder.MeterBinder;
14  
15  /**
16   * {@link MeterBinder} for system memory metrics following
17   * <a href="https://opentelemetry.io/docs/specs/semconv/system/system-metrics/#memory-metrics">OpenTelemetry semantic
18   * conventions</a>.
19   *
20   * <p>
21   * Registers:
22   * <ul>
23   * <li>{@code system.memory.usage} — memory in use by state ({@code used}, {@code free}), in bytes</li>
24   * <li>{@code system.memory.limit} — total memory available in the system, in bytes</li>
25   * <li>{@code system.memory.utilization} — fraction of memory in use by state (0.0–1.0)</li>
26   * </ul>
27   */
28  public class MemoryMetrics implements MeterBinder {
29  
30      private static final String MEMORY_USAGE = "system.memory.usage";
31      private static final String MEMORY_LIMIT = "system.memory.limit";
32      private static final String MEMORY_UTILIZATION = "system.memory.utilization";
33      private static final Tag STATE_USED = Tag.of("system.memory.state", "used");
34      private static final Tag STATE_FREE = Tag.of("system.memory.state", "free");
35  
36      private final GlobalMemory memory;
37  
38      /**
39       * Creates a new {@code MemoryMetrics} binder.
40       *
41       * @param memory the {@link GlobalMemory} instance to read from
42       */
43      public MemoryMetrics(GlobalMemory memory) {
44          this.memory = memory;
45      }
46  
47      @Override
48      public void bindTo(MeterRegistry registry) {
49          Gauge.builder(MEMORY_USAGE, memory, mem -> mem.getTotal() - mem.getAvailable()).tags(Tags.of(STATE_USED))
50                  .description("Memory used in bytes").baseUnit("By").strongReference(true).register(registry);
51          Gauge.builder(MEMORY_USAGE, memory, GlobalMemory::getAvailable).tags(Tags.of(STATE_FREE))
52                  .description("Memory available in bytes").baseUnit("By").strongReference(true).register(registry);
53          Gauge.builder(MEMORY_LIMIT, memory, GlobalMemory::getTotal).description("Total memory available in the system")
54                  .baseUnit("By").strongReference(true).register(registry);
55          Gauge.builder(MEMORY_UTILIZATION, memory,
56                  mem -> mem.getTotal() == 0 ? 0d : (double) (mem.getTotal() - mem.getAvailable()) / mem.getTotal())
57                  .tags(Tags.of(STATE_USED)).description("Fraction of memory used").strongReference(true)
58                  .register(registry);
59          Gauge.builder(MEMORY_UTILIZATION, memory,
60                  mem -> mem.getTotal() == 0 ? 0d : (double) mem.getAvailable() / mem.getTotal())
61                  .tags(Tags.of(STATE_FREE)).description("Fraction of memory free").strongReference(true)
62                  .register(registry);
63      }
64  }
