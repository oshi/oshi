5   package oshi.metrics;
6   
7   import java.util.List;
8   import java.util.function.Supplier;
9   
10  import oshi.software.os.OSFileStore;
11  
12  import io.micrometer.core.instrument.Gauge;
13  import io.micrometer.core.instrument.MeterRegistry;
14  import io.micrometer.core.instrument.binder.MeterBinder;
15  
16  /**
17   * {@link MeterBinder} for system filesystem metrics following
18   * <a href="https://opentelemetry.io/docs/specs/semconv/system/system-metrics/#filesystem-metrics">OpenTelemetry
19   * semantic conventions</a>.
20   *
21   * <p>
22   * Registers per filesystem:
23   * <ul>
24   * <li>{@code system.filesystem.usage} — filesystem space usage by state ({@code used}, {@code free}), in bytes</li>
25   * <li>{@code system.filesystem.utilization} — fraction of filesystem space in use by state (0.0–1.0)</li>
26   * <li>{@code system.filesystem.limit} — total capacity of the filesystem, in bytes</li>
27   * </ul>
28   */
29  public class FileSystemMetrics implements MeterBinder {
30  
31      private static final String FS_USAGE = "system.filesystem.usage";
32      private static final String FS_UTILIZATION = "system.filesystem.utilization";
33      private static final String FS_LIMIT = "system.filesystem.limit";
34      private static final String DEVICE_KEY = "system.device";
35      private static final String MOUNTPOINT_KEY = "system.filesystem.mountpoint";
36      private static final String TYPE_KEY = "system.filesystem.type";
37      private static final String MODE_KEY = "system.filesystem.mode";
38      private static final String STATE_KEY = "system.filesystem.state";
39  
40      private final Supplier<List<OSFileStore>> fileStoreSupplier;
41  
42      /**
43       * Creates a new {@code FileSystemMetrics} binder.
44       *
45       * <p>
46       * Note: {@link #bindTo(MeterRegistry)} calls the supplier once to capture a snapshot of filesystems. Filesystems
47       * mounted after binding will not be tracked; unmounted filesystems may leave stale meters. To refresh, create and
48       * bind a new instance.
49       *
50       * @param fileStoreSupplier supplier that returns the current list of {@link OSFileStore} instances
51       */
52      public FileSystemMetrics(Supplier<List<OSFileStore>> fileStoreSupplier) {
53          this.fileStoreSupplier = fileStoreSupplier;
54      }
55  
56      @Override
57      public void bindTo(MeterRegistry registry) {
58          for (OSFileStore fs : fileStoreSupplier.get()) {
59              String device = fs.getVolume();
60              String mount = fs.getMount();
61              String type = fs.getType();
62              String opts = fs.getOptions();
63              String mode = opts != null && java.util.Arrays.asList(opts.split(",")).contains("rw") ? "rw" : "ro";
64  
65              // system.filesystem.usage — UpDownCounter (Gauge), unit "By", attr: state, device, mount, type, mode
66              Gauge.builder(FS_USAGE, fs, f -> {
67                  f.updateAttributes();
68                  return Math.max(0L, f.getTotalSpace() - f.getUsableSpace());
69              }).tag(STATE_KEY, "used").tag(DEVICE_KEY, device).tag(MOUNTPOINT_KEY, mount).tag(TYPE_KEY, type)
70                      .tag(MODE_KEY, mode).description("Filesystem space usage").baseUnit("By").strongReference(true)
71                      .register(registry);
72              Gauge.builder(FS_USAGE, fs, f -> {
73                  f.updateAttributes();
74                  return (double) f.getUsableSpace();
75              }).tag(STATE_KEY, "free").tag(DEVICE_KEY, device).tag(MOUNTPOINT_KEY, mount).tag(TYPE_KEY, type)
76                      .tag(MODE_KEY, mode).description("Filesystem space usage").baseUnit("By").strongReference(true)
77                      .register(registry);
78  
79              // system.filesystem.utilization — Gauge, unit "1", attr: state, device, mount, type, mode
80              Gauge.builder(FS_UTILIZATION, fs, f -> {
81                  f.updateAttributes();
82                  return f.getTotalSpace() == 0 ? 0d
83                          : Math.max(0d, (double) (f.getTotalSpace() - f.getUsableSpace()) / f.getTotalSpace());
84              }).tag(STATE_KEY, "used").tag(DEVICE_KEY, device).tag(MOUNTPOINT_KEY, mount).tag(TYPE_KEY, type)
85                      .tag(MODE_KEY, mode).description("Filesystem utilization").strongReference(true).register(registry);
86              Gauge.builder(FS_UTILIZATION, fs, f -> {
87                  f.updateAttributes();
88                  return f.getTotalSpace() == 0 ? 0d : (double) f.getUsableSpace() / f.getTotalSpace();
89              }).tag(STATE_KEY, "free").tag(DEVICE_KEY, device).tag(MOUNTPOINT_KEY, mount).tag(TYPE_KEY, type)
90                      .tag(MODE_KEY, mode).description("Filesystem utilization").strongReference(true).register(registry);
91  
92              // system.filesystem.limit — UpDownCounter (Gauge), unit "By", attr: device, mount, type, mode
93              Gauge.builder(FS_LIMIT, fs, f -> {
94                  f.updateAttributes();
95                  return (double) f.getTotalSpace();
96              }).tag(DEVICE_KEY, device).tag(MOUNTPOINT_KEY, mount).tag(TYPE_KEY, type).tag(MODE_KEY, mode)
97                      .description("Total capacity of the filesystem").baseUnit("By").strongReference(true)
98                      .register(registry);
99          }
100     }
101 }
