5   package oshi.metrics;
6   
7   import oshi.hardware.VirtualMemory;
8   
9   import io.micrometer.core.instrument.FunctionCounter;
10  import io.micrometer.core.instrument.Gauge;
11  import io.micrometer.core.instrument.MeterRegistry;
12  import io.micrometer.core.instrument.Tag;
13  import io.micrometer.core.instrument.Tags;
14  import io.micrometer.core.instrument.binder.MeterBinder;
15  
16  /**
17   * {@link MeterBinder} for system paging/swap metrics following
18   * <a href="https://opentelemetry.io/docs/specs/semconv/system/system-metrics/#pagingswap-metrics">OpenTelemetry
19   * semantic conventions</a>.
20   *
21   * <p>
22   * Registers:
23   * <ul>
24   * <li>{@code system.paging.usage} — swap/pagefile usage by state ({@code used}, {@code free}), in bytes</li>
25   * <li>{@code system.paging.utilization} — fraction of swap/pagefile in use by state (0.0–1.0)</li>
26   * <li>{@code system.paging.operations} — cumulative paging operations by direction ({@code in}, {@code out})</li>
27   * </ul>
28   *
29   * <p>
30   * Note: {@code system.paging.faults} is not implemented because OSHI does not expose system-level major/minor page
31   * fault counters.
32   */
33  public class PagingMetrics implements MeterBinder {
34  
35      private static final String PAGING_USAGE = "system.paging.usage";
36      private static final String PAGING_UTILIZATION = "system.paging.utilization";
37      private static final String PAGING_OPERATIONS = "system.paging.operations";
38      private static final Tag STATE_USED = Tag.of("system.paging.state", "used");
39      private static final Tag STATE_FREE = Tag.of("system.paging.state", "free");
40      private static final String DIRECTION_KEY = "system.paging.direction";
41  
42      private final VirtualMemory vm;
43  
44      /**
45       * Creates a new {@code PagingMetrics} binder.
46       *
47       * @param vm the {@link VirtualMemory} instance to read from
48       */
49      public PagingMetrics(VirtualMemory vm) {
50          this.vm = vm;
51      }
52  
53      @Override
54      public void bindTo(MeterRegistry registry) {
55          // system.paging.usage — UpDownCounter (Gauge), unit "By", attr system.paging.state
56          Gauge.builder(PAGING_USAGE, vm, VirtualMemory::getSwapUsed).tags(Tags.of(STATE_USED))
57                  .description("Unix swap or Windows pagefile usage").baseUnit("By").register(registry);
58          Gauge.builder(PAGING_USAGE, vm, v -> v.getSwapTotal() - v.getSwapUsed()).tags(Tags.of(STATE_FREE))
59                  .description("Unix swap or Windows pagefile usage").baseUnit("By").register(registry);
60  
61          // system.paging.utilization — Gauge, unit "1", attr system.paging.state
62          Gauge.builder(PAGING_UTILIZATION, vm,
63                  v -> v.getSwapTotal() == 0 ? 0d : (double) v.getSwapUsed() / v.getSwapTotal()).tags(Tags.of(STATE_USED))
64                  .description("Fraction of swap/pagefile used").register(registry);
65          Gauge.builder(PAGING_UTILIZATION, vm,
66                  v -> v.getSwapTotal() == 0 ? 0d : (double) (v.getSwapTotal() - v.getSwapUsed()) / v.getSwapTotal())
67                  .tags(Tags.of(STATE_FREE)).description("Fraction of swap/pagefile free").register(registry);
68  
69          // system.paging.operations — Counter, unit "{operation}", attr system.paging.direction
70          FunctionCounter.builder(PAGING_OPERATIONS, vm, VirtualMemory::getSwapPagesIn).tag(DIRECTION_KEY, "in")
71                  .description("Paging operations").baseUnit("{operation}").register(registry);
72          FunctionCounter.builder(PAGING_OPERATIONS, vm, VirtualMemory::getSwapPagesOut).tag(DIRECTION_KEY, "out")
73                  .description("Paging operations").baseUnit("{operation}").register(registry);
74      }
75  }
