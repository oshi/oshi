5   package oshi.metrics;
6   
7   import java.util.function.Supplier;
8   
9   import oshi.software.os.OSProcess;
10  
11  import io.micrometer.core.instrument.FunctionCounter;
12  import io.micrometer.core.instrument.Gauge;
13  import io.micrometer.core.instrument.MeterRegistry;
14  import io.micrometer.core.instrument.binder.MeterBinder;
15  
16  /**
17   * {@link MeterBinder} for process-level metrics following
18   * <a href="https://opentelemetry.io/docs/specs/semconv/system/process-metrics/">OpenTelemetry semantic conventions</a>.
19   *
20   * <p>
21   * Typically used to monitor the current JVM process via {@code os.getCurrentProcess()}.
22   *
23   * <p>
24   * Registers:
25   * <ul>
26   * <li>{@code process.cpu.time} — CPU seconds by mode (user, system)</li>
27   * <li>{@code process.memory.usage} — physical memory (RSS) in bytes</li>
28   * <li>{@code process.memory.virtual} — virtual memory in bytes</li>
29   * <li>{@code process.disk.io} — disk bytes by direction (read, write)</li>
30   * <li>{@code process.thread.count} — thread count</li>
31   * <li>{@code process.open_file_descriptor.count} — open file descriptors</li>
32   * <li>{@code process.paging.faults} — page faults by type (major, minor)</li>
33   * <li>{@code process.context_switches} — context switches by type (voluntary/involuntary on POSIX, or total on
34   * Windows)</li>
35   * <li>{@code process.uptime} — process uptime in seconds</li>
36   * </ul>
37   *
38   * <p>
39   * Not implemented:
40   * <ul>
41   * <li>{@code process.network.io} — OSHI does not expose per-process network I/O</li>
42   * </ul>
43   */
44  public class ProcessMetrics implements MeterBinder {
45  
46      private static final String CPU_TIME = "process.cpu.time";
47      private static final String MEMORY_USAGE = "process.memory.usage";
48      private static final String MEMORY_VIRTUAL = "process.memory.virtual";
49      private static final String DISK_IO = "process.disk.io";
50      private static final String THREAD_COUNT = "process.thread.count";
51      private static final String OPEN_FD = "process.open_file_descriptor.count";
52      private static final String PAGING_FAULTS = "process.paging.faults";
53      private static final String CONTEXT_SWITCHES = "process.context_switches";
54      private static final String UPTIME = "process.uptime";
55      private static final double MS_PER_SECOND = 1000.0;
56  
57      private final Supplier<OSProcess> processSupplier;
58  
59      /**
60       * Creates a new {@code ProcessMetrics} binder.
61       *
62       * @param processSupplier supplier that returns a fresh {@link OSProcess} snapshot (e.g.,
63       *                        {@code os::getCurrentProcess})
64       */
65      public ProcessMetrics(Supplier<OSProcess> processSupplier) {
66          this.processSupplier = processSupplier;
67      }
68  
69      @Override
70      public void bindTo(MeterRegistry registry) {
71          // process.cpu.time — Counter, unit "s", attr cpu.mode (Required)
72          FunctionCounter.builder(CPU_TIME, processSupplier, s -> s.get().getUserTime() / MS_PER_SECOND)
73                  .tag("cpu.mode", "user").description("Total CPU seconds broken down by different CPU modes")
74                  .baseUnit("s").register(registry);
75          FunctionCounter.builder(CPU_TIME, processSupplier, s -> s.get().getKernelTime() / MS_PER_SECOND)
76                  .tag("cpu.mode", "system").description("Total CPU seconds broken down by different CPU modes")
77                  .baseUnit("s").register(registry);
78  
79          // process.memory.usage — UpDownCounter (Gauge), unit "By"
80          Gauge.builder(MEMORY_USAGE, processSupplier, s -> s.get().getResidentMemory())
81                  .description("The amount of physical memory in use").baseUnit("By").strongReference(true)
82                  .register(registry);
83  
84          // process.memory.virtual — UpDownCounter (Gauge), unit "By"
85          Gauge.builder(MEMORY_VIRTUAL, processSupplier, s -> s.get().getVirtualSize())
86                  .description("The amount of committed virtual memory").baseUnit("By").strongReference(true)
87                  .register(registry);
88  
89          // process.disk.io — Counter, unit "By", attr disk.io.direction (Required)
90          FunctionCounter.builder(DISK_IO, processSupplier, s -> s.get().getBytesRead()).tag("disk.io.direction", "read")
91                  .description("Disk bytes transferred").baseUnit("By").register(registry);
92          FunctionCounter.builder(DISK_IO, processSupplier, s -> s.get().getBytesWritten())
93                  .tag("disk.io.direction", "write").description("Disk bytes transferred").baseUnit("By")
94                  .register(registry);
95  
96          // process.thread.count — UpDownCounter (Gauge), unit "{thread}"
97          Gauge.builder(THREAD_COUNT, processSupplier, s -> s.get().getThreadCount()).description("Process threads count")
98                  .baseUnit("{thread}").strongReference(true).register(registry);
99  
100         // process.open_file_descriptor.count — UpDownCounter (Gauge), unit "{file_descriptor}"
101         Gauge.builder(OPEN_FD, processSupplier, s -> s.get().getOpenFiles())
102                 .description("Number of file descriptors in use by the process").baseUnit("{file_descriptor}")
103                 .strongReference(true).register(registry);
104 
105         // process.paging.faults — Counter, unit "{fault}", attr system.paging.fault.type (Recommended)
106         FunctionCounter.builder(PAGING_FAULTS, processSupplier, s -> s.get().getMinorFaults())
107                 .tag("system.paging.fault.type", "minor").description("Number of page faults the process has made")
108                 .baseUnit("{fault}").register(registry);
109         FunctionCounter.builder(PAGING_FAULTS, processSupplier, s -> s.get().getMajorFaults())
110                 .tag("system.paging.fault.type", "major").description("Number of page faults the process has made")
111                 .baseUnit("{fault}").register(registry);
112 
113         // process.context_switches — Counter, unit "{context_switch}", attr process.context_switch.type (Required)
114         // Only emit voluntary/involuntary if the platform provides the split; otherwise emit total only
115         OSProcess probe = processSupplier.get();
116         long vol = probe.getVoluntaryContextSwitches();
117         long invol = probe.getInvoluntaryContextSwitches();
118         long total = probe.getContextSwitches();
119         if (vol + invol == total && total > 0) {
120             FunctionCounter.builder(CONTEXT_SWITCHES, processSupplier, s -> s.get().getVoluntaryContextSwitches())
121                     .tag("process.context_switch.type", "voluntary")
122                     .description("Number of times the process has been context switched").baseUnit("{context_switch}")
123                     .register(registry);
124             FunctionCounter.builder(CONTEXT_SWITCHES, processSupplier, s -> s.get().getInvoluntaryContextSwitches())
125                     .tag("process.context_switch.type", "involuntary")
126                     .description("Number of times the process has been context switched").baseUnit("{context_switch}")
127                     .register(registry);
128         } else if (total > 0) {
129             FunctionCounter.builder(CONTEXT_SWITCHES, processSupplier, s -> s.get().getContextSwitches())
130                     .tag("process.context_switch.type", "total")
131                     .description("Number of times the process has been context switched").baseUnit("{context_switch}")
132                     .register(registry);
133         }
134 
135         // process.uptime — Gauge, unit "s"
136         Gauge.builder(UPTIME, processSupplier, s -> s.get().getUpTime() / MS_PER_SECOND)
137                 .description("The time the process has been running").baseUnit("s").strongReference(true)
138                 .register(registry);
139     }
140 }
