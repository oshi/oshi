5   package oshi.spi;
6   
7   import oshi.annotation.PublicApi;
8   import oshi.hardware.HardwareAbstractionLayer;
9   import oshi.software.os.OperatingSystem;
10  
11  /**
12   * Service Provider Interface for OSHI's {@code SystemInfo} entry points.
13   * <p>
14   * Implementations are discovered via {@link java.util.ServiceLoader}. When multiple providers are available (e.g., both
15   * {@code oshi-core} and {@code oshi-core-ffm} on the classpath), {@link SystemInfoFactory} selects the best available
16   * provider based on {@link #getPriority()} and {@link #isAvailable()}.
17   *
18   * @see SystemInfoFactory
19   */
20  @PublicApi
21  public interface SystemInfoProvider {
22  
23      /**
24       * Creates a new instance of the appropriate platform-specific {@link OperatingSystem}.
25       *
26       * @return A new instance of {@link OperatingSystem}.
27       */
28      OperatingSystem getOperatingSystem();
29  
30      /**
31       * Creates a new instance of the appropriate platform-specific {@link HardwareAbstractionLayer}.
32       *
33       * @return A new instance of {@link HardwareAbstractionLayer}.
34       */
35      HardwareAbstractionLayer getHardware();
36  
37      /**
38       * Returns the priority of this provider. Higher values indicate higher priority. When multiple providers are
39       * available, the one with the highest priority is selected.
40       * <p>
41       * Priority 0 is reserved for a potential future no-native-access fallback. The JNA-based provider
42       * ({@code oshi-core}) returns 10. The FFM-based provider ({@code oshi-core-ffm}) returns 20 (preferred when
43       * available).
44       *
45       * @return the priority of this provider
46       */
47      int getPriority();
48  
49      /**
50       * Returns whether this provider is available in the current runtime environment.
51       * <p>
52       * A provider may be unavailable if the current platform is not supported or if required runtime features (such as
53       * the FFM API on JDK 25+) are not present.
54       *
55       * @return {@code true} if this provider can be used in the current environment
56       */
57      boolean isAvailable();
58  }
