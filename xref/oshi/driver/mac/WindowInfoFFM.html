36  /**
37   * Utility to query desktop windows using FFM (no JNA dependency).
38   */
39  @ThreadSafe
40  public final class WindowInfoFFM {
41  
42      // CGWindowListOption constants (CGWindow.h)
43      private static final int kCGNullWindowID = 0;
44      private static final int kCGWindowListOptionAll = 0;
45      private static final int kCGWindowListOptionOnScreenOnly = 1 << 0;
46      private static final int kCGWindowListExcludeDesktopElements = 1 << 4;
47  
48      private WindowInfoFFM() {
49      }
50  
51      /**
52       * Gets windows on the operating system's GUI desktop.
53       *
54       * @param visibleOnly Whether to restrict the list to only windows visible to the user.
55       * @return A list of {@link OSDesktopWindow} objects representing the desktop windows.
56       */
57      public static List<OSDesktopWindow> queryDesktopWindows(boolean visibleOnly) {
58          List<OSDesktopWindow> windowList = new ArrayList<>();
59          try (Arena arena = Arena.ofConfined()) {
60              int option = visibleOnly ? kCGWindowListOptionOnScreenOnly | kCGWindowListExcludeDesktopElements
61                      : kCGWindowListOptionAll;
62              MemorySegment rawArray = CGWindowListCopyWindowInfo(option, kCGNullWindowID);
63              if (rawArray.equals(MemorySegment.NULL)) {
64                  return windowList;
65              }
66              CFArrayRef windowInfo = new CFArrayRef(rawArray);
67              try {
68                  int numWindows = windowInfo.getCount();
69  
70                  CFStringRef kCGWindowIsOnscreen = CFStringRef.createCFString("kCGWindowIsOnscreen");
71                  CFStringRef kCGWindowNumber = CFStringRef.createCFString("kCGWindowNumber");
72                  CFStringRef kCGWindowOwnerPID = CFStringRef.createCFString("kCGWindowOwnerPID");
73                  CFStringRef kCGWindowLayer = CFStringRef.createCFString("kCGWindowLayer");
74                  CFStringRef kCGWindowBounds = CFStringRef.createCFString("kCGWindowBounds");
75                  CFStringRef kCGWindowName = CFStringRef.createCFString("kCGWindowName");
76                  CFStringRef kCGWindowOwnerName = CFStringRef.createCFString("kCGWindowOwnerName");
77                  try {
78                      for (int i = 0; i < numWindows; i++) {
79                          MemorySegment dictSeg = windowInfo.getValueAtIndex(i);
80                          if (dictSeg.equals(MemorySegment.NULL)) {
81                              continue;
82                          }
83                          CFDictionaryRef windowRef = new CFDictionaryRef(dictSeg);
84  
85                          // kCGWindowIsOnscreen is optional — absent means visible
86                          MemorySegment onscreenSeg = windowRef.getValue(kCGWindowIsOnscreen);
87                          boolean visible = onscreenSeg.equals(MemorySegment.NULL)
88                                  || new CFBooleanRef(onscreenSeg).booleanValue();
89                          if (visibleOnly && !visible) {
90                              continue;
91                          }
92  
93                          long windowNumber = new CFNumberRef(windowRef.getValue(kCGWindowNumber)).longValue();
94                          long ownerPID = new CFNumberRef(windowRef.getValue(kCGWindowOwnerPID)).longValue();
95                          int windowLayer = new CFNumberRef(windowRef.getValue(kCGWindowLayer)).intValue();
96  
97                          // Parse CGRect from the bounds dictionary
98                          MemorySegment boundsSeg = windowRef.getValue(kCGWindowBounds);
99                          MemorySegment rectBuf = arena.allocate(CG_RECT);
100                         Rectangle windowBounds = new Rectangle();
101                         if (!boundsSeg.equals(MemorySegment.NULL)
102                                 && CGRectMakeWithDictionaryRepresentation(boundsSeg, rectBuf)) {
103                             MemorySegment origin = rectBuf.asSlice(CG_RECT.byteOffset(CG_ORIGIN), CG_POINT.byteSize());
104                             MemorySegment size = rectBuf.asSlice(CG_RECT.byteOffset(CG_SIZE_ELEM), CG_SIZE.byteSize());
105                             double x = origin.get(JAVA_DOUBLE, CG_POINT.byteOffset(CG_X));
106                             double y = origin.get(JAVA_DOUBLE, CG_POINT.byteOffset(CG_Y));
107                             double w = size.get(JAVA_DOUBLE, CG_SIZE.byteOffset(CG_WIDTH));
108                             double h = size.get(JAVA_DOUBLE, CG_SIZE.byteOffset(CG_HEIGHT));
109                             windowBounds = new Rectangle(FormatUtil.roundToInt(x), FormatUtil.roundToInt(y),
110                                     FormatUtil.roundToInt(w), FormatUtil.roundToInt(h));
111                         }
112 
113                         String windowName = CFUtilFFM.cfPointerToString(windowRef.getValue(kCGWindowName), false);
114                         String ownerName = CFUtilFFM.cfPointerToString(windowRef.getValue(kCGWindowOwnerName), false);
115                         if (windowName.isEmpty()) {
116                             windowName = ownerName;
117                         } else {
118                             windowName = windowName + "(" + ownerName + ")";
119                         }
120 
121                         windowList.add(new OSDesktopWindow(windowNumber, windowName, ownerName, windowBounds, ownerPID,
122                                 windowLayer, visible));
123                     }
124                 } finally {
125                     kCGWindowIsOnscreen.release();
126                     kCGWindowNumber.release();
127                     kCGWindowOwnerPID.release();
128                     kCGWindowLayer.release();
129                     kCGWindowBounds.release();
130                     kCGWindowName.release();
131                     kCGWindowOwnerName.release();
132                 }
133             } finally {
134                 windowInfo.release();
135             }
136         } catch (Throwable e) {
137             return windowList;
138         }
139         return windowList;
140     }
