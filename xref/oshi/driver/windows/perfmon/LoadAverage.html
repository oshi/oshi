24  package oshi.driver.windows.perfmon;
25  
26  import java.util.Arrays;
27  import java.util.List;
28  import java.util.Map;
29  
30  import oshi.annotation.concurrent.ThreadSafe;
31  import oshi.driver.windows.perfmon.ProcessInformation.IdleProcessorTimeProperty;
32  import oshi.driver.windows.perfmon.SystemInformation.ProcessorQueueLengthProperty;
33  import oshi.util.tuples.Pair;
34  
35  /**
36   * Utility to calculate a load average equivalent metric on Windows. Starts a
37   * daemon thread to collect the necessary counters and averages in 5-second
38   * intervals.
39   */
40  @ThreadSafe
41  public final class LoadAverage {
42  
43      // Daemon thread for Load Average
44      private static Thread loadAvgThread = null;
45  
46      private static double[] loadAverages = new double[] { -1d, -1d, -1d };
47      private static final double[] EXP_WEIGHT = new double[] {
48              // 1-, 5-, and 15-minute exponential smoothing weight
49              Math.exp(-5d / 60d), Math.exp(-5d / 300d), Math.exp(-5d / 900d) };
50  
51      private LoadAverage() {
52      }
53  
54      public static double[] queryLoadAverage(int nelem) {
55          synchronized (loadAverages) {
56              return Arrays.copyOf(loadAverages, nelem);
57          }
58      }
59  
60      public static synchronized void stopDaemon() {
61          if (loadAvgThread != null) {
62              loadAvgThread.interrupt();
63              loadAvgThread = null;
64          }
65      }
66  
67      public static synchronized void startDaemon() {
68          if (loadAvgThread != null) {
69              return;
70          }
71          loadAvgThread = new Thread("OSHI Load Average daemon") {
72              @Override
73              public void run() {
74                  // Initialize tick counters
75                  Pair<Long, Long> nonIdlePair = LoadAverage.queryNonIdleTicks();
76                  long nonIdleTicks0 = nonIdlePair.getA();
77                  long nonIdleBase0 = nonIdlePair.getB();
78                  long nonIdleTicks;
79                  long nonIdleBase;
80  
81                  // Use nanoTime to synchronize queries at 5 seconds
82                  long initNanos = System.nanoTime();
83                  long delay;
84  
85                  // The two components of load average
86                  double runningProcesses;
87                  long queueLength;
88  
89                  try {
90                      Thread.sleep(2500L);
91                  } catch (InterruptedException e) {
92                      Thread.currentThread().interrupt();
93                  }
94                  while (!Thread.currentThread().isInterrupted()) {
95                      // get non-idle ticks, proxy for average processes running
96                      nonIdlePair = LoadAverage.queryNonIdleTicks();
97                      nonIdleTicks = nonIdlePair.getA() - nonIdleTicks0;
98                      nonIdleBase = nonIdlePair.getB() - nonIdleBase0;
99                      if (nonIdleBase > 0 && nonIdleTicks > 0) {
100                         runningProcesses = (double) nonIdleTicks / nonIdleBase;
101                     } else {
102                         runningProcesses = 0d;
103                     }
104                     nonIdleTicks0 = nonIdlePair.getA();
105                     nonIdleBase0 = nonIdlePair.getB();
106                     // get processes waiting
107                     queueLength = SystemInformation.queryProcessorQueueLength()
108                             .getOrDefault(ProcessorQueueLengthProperty.PROCESSORQUEUELENGTH, 0L);
109 
110                     synchronized (loadAverages) {
111                         // Init to running procs the first time
112                         if (loadAverages[0] < 0d) {
113                             Arrays.fill(loadAverages, runningProcesses);
114                         }
115                         // Use exponential smoothing to update values
116                         for (int i = 0; i < loadAverages.length; i++) {
117                             loadAverages[i] *= EXP_WEIGHT[i];
118                             loadAverages[i] += (runningProcesses + queueLength) * (1d - EXP_WEIGHT[i]);
119                         }
120                     }
121 
122                     delay = 5000L - (System.nanoTime() - initNanos) % 5_000_000_000L / 1_000_000;
123                     if (delay < 500L) {
124                         delay += 5000L;
125                     }
126                     try {
127                         Thread.sleep(delay);
128                     } catch (InterruptedException e) {
129                         Thread.currentThread().interrupt();
130                     }
131                 }
132             }
133         };
134         loadAvgThread.setDaemon(true);
135         loadAvgThread.start();
136     }
137 
138     private static Pair<Long, Long> queryNonIdleTicks() {
139         Pair<List<String>, Map<IdleProcessorTimeProperty, List<Long>>> idleValues = ProcessInformation
140                 .queryIdleProcessCounters();
141         List<String> instances = idleValues.getA();
142         Map<IdleProcessorTimeProperty, List<Long>> valueMap = idleValues.getB();
143         List<Long> proctimeTicks = valueMap.get(IdleProcessorTimeProperty.PERCENTPROCESSORTIME);
144         List<Long> proctimeBase = valueMap.get(IdleProcessorTimeProperty.ELAPSEDTIME);
145         long nonIdleTicks = 0L;
146         long nonIdleBase = 0L;
147         for (int i = 0; i < instances.size(); i++) {
148             if ("_Total".equals(instances.get(i))) {
149                 nonIdleTicks += proctimeTicks.get(i);
150                 nonIdleBase += proctimeBase.get(i);
151             } else if ("Idle".equals(instances.get(i))) {
152                 nonIdleTicks -= proctimeTicks.get(i);
153             }
154         }
155         return new Pair<>(nonIdleTicks, nonIdleBase);
156     }
