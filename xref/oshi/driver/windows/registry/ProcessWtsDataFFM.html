45  /**
46   * Utility to read process data from WTS native calls with backup from WMI.
47   */
48  @ThreadSafe
49  public final class ProcessWtsDataFFM {
50  
51      private static final Logger LOG = LoggerFactory.getLogger(ProcessWtsDataFFM.class);
52  
53      private static final boolean IS_WINDOWS7_OR_GREATER = VersionHelpersFFM.IsWindows7OrGreater();
54  
55      private ProcessWtsDataFFM() {
56      }
57  
58      /**
59       * Query process information using native WTS calls (Win7+) or WMI as fallback.
60       *
61       * @param pids An optional collection of process IDs to filter the list to. May be null for no filtering.
62       * @return A map with Process ID as the key and a {@link WtsInfo} object populated with data.
63       */
64      public static Map<Integer, WtsInfo> queryProcessWtsMap(Collection<Integer> pids) {
65          if (IS_WINDOWS7_OR_GREATER) {
66              return queryProcessWtsMapFromWTS(pids);
67          }
68          return queryProcessWtsMapFromWMI(pids);
69      }
70  
71      private static Map<Integer, WtsInfo> queryProcessWtsMapFromWTS(Collection<Integer> pids) {
72          Set<Integer> pidSet = pids != null ? new HashSet<>(pids) : null;
73          Map<Integer, WtsInfo> wtsMap = new HashMap<>();
74          try (Arena arena = Arena.ofConfined()) {
75              MemorySegment pLevel = arena.allocate(JAVA_INT);
76              pLevel.set(JAVA_INT, 0, WTS_PROCESS_INFO_LEVEL_1);
77              MemorySegment ppProcessInfo = arena.allocate(ADDRESS);
78              MemorySegment pCount = arena.allocate(JAVA_INT);
79  
80              if (!Wtsapi32FFM.WTSEnumerateProcessesEx(WTS_CURRENT_SERVER_HANDLE, pLevel, WTS_ANY_SESSION, ppProcessInfo,
81                      pCount)) {
82                  LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32FFM.GetLastError().orElse(-1));
83                  return wtsMap;
84              }
85  
86              int count = pCount.get(JAVA_INT, 0);
87              MemorySegment pProcessInfo = ppProcessInfo.get(ADDRESS, 0).reinterpret(PROCESS_INFO_EX_SIZE * count);
88              try {
89                  for (int i = 0; i < count; i++) {
90                      long base = i * PROCESS_INFO_EX_SIZE;
91                      int pid = pProcessInfo.get(JAVA_INT, base + PROCESS_INFO_EX_PROCESS_ID);
92                      if (pidSet != null && !pidSet.contains(pid)) {
93                          continue;
94                      }
95                      MemorySegment pName = pProcessInfo.get(ADDRESS, base + PROCESS_INFO_EX_PROCESS_NAME);
96                      String name = pName.equals(MemorySegment.NULL) ? "" : readWideString(pName.reinterpret(512));
97                      int threads = pProcessInfo.get(JAVA_INT, base + PROCESS_INFO_EX_NUMBER_OF_THREADS);
98                      long pagefileUsage = pProcessInfo.get(JAVA_INT, base + PROCESS_INFO_EX_PAGEFILE_USAGE)
99                              & 0xffff_ffffL;
100                     long kernelTime = pProcessInfo.get(JAVA_LONG, base + PROCESS_INFO_EX_KERNEL_TIME) / 10_000L;
101                     long userTime = pProcessInfo.get(JAVA_LONG, base + PROCESS_INFO_EX_USER_TIME) / 10_000L;
102                     int handles = pProcessInfo.get(JAVA_INT, base + PROCESS_INFO_EX_HANDLE_COUNT);
103                     wtsMap.put(pid, new WtsInfo(name, "", threads, pagefileUsage, kernelTime, userTime, handles));
104                 }
105             } finally {
106                 if (!Wtsapi32FFM.WTSFreeMemoryEx(WTS_TYPE_PROCESS_INFO_LEVEL_1, pProcessInfo, count)) {
107                     LOG.warn("Failed to Free Memory for Processes. Error code: {}",
108                             Kernel32FFM.GetLastError().orElse(-1));
109                 }
110             }
111         } catch (Throwable t) {
112             LOG.error("Failed to enumerate Processes via WTS", t);
113         }
114         return wtsMap;
115     }
116 
117     static Map<Integer, WtsInfo> queryProcessWtsMapFromWMI(Collection<Integer> pids) {
118         Map<Integer, WtsInfo> wtsMap = new HashMap<>();
119         WmiResult<ProcessXPProperty> processWmiResult = Win32ProcessFFM.queryProcesses(pids);
120         for (int i = 0; i < processWmiResult.getResultCount(); i++) {
121             wtsMap.put(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i), new WtsInfo(
122                     WmiUtil.getString(processWmiResult, ProcessXPProperty.NAME, i),
123                     WmiUtil.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i),
124                     WmiUtil.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i),
125                     // WMI Pagefile usage is in KB
126                     1024 * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffff_ffffL),
127                     WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10_000L,
128                     WmiUtil.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10_000L,
129                     WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i)));
130         }
131         return wtsMap;
132     }
