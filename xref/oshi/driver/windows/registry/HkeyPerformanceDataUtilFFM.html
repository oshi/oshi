52   * Utility to read HKEY_PERFORMANCE_DATA information using the FFM API.

59      /*
60       * Do a one-time lookup of the HKEY_PERFORMANCE_TEXT counter indices and store in a map for efficient lookups
61       * on-demand.
62       */
63      private static final Map<String, Integer> COUNTER_INDEX_MAP = mapCounterIndicesFromRegistry();
64  
65      private static int maxPerfBufferSize = 16384;
66  
67      private HkeyPerformanceDataUtilFFM() {
68      }

70      /**
71       * Looks up the performance counter index for the given English counter name.
72       *
73       * @param name The English counter name
74       * @return The counter index, or 0 if not found
75       */
76      public static int getCounterIndex(String name) {
77          return COUNTER_INDEX_MAP.getOrDefault(name, 0);
78      }

80      /**
81       * Reads and parses a block of performance data from the registry.
82       *
83       * @param <T>         PDH Counters use an Enum to identify the fields to query in either the counter or WMI backup,
84       *                    and use the enum values as keys to retrieve the results.
85       * @param objectName  The counter object for which to fetch data
86       * @param counterEnum Which counters to return data for
87       * @return A triplet containing the results. The first element maps the input enum to the counter values where the
88       *         first enum will contain the instance name as a {@link String}, and the remaining values will either be
89       *         {@link Long}, {@link Integer}, or {@code null} depending on whether the specified enum counter was
90       *         present and the size of the counter value. The second element is a timestamp in 100nSec increments
91       *         (Windows 1601 Epoch) while the third element is a timestamp in milliseconds since the 1970 Epoch.
92       */
93      public static <T extends Enum<T> & PdhCounterWildcardProperty> Triplet<List<Map<T, Object>>, Long, Long> readPerfDataFromRegistry(
94              String objectName, Class<T> counterEnum) {
95          // Load indices
96          // e.g., call with "Process" and ProcessPerformanceProperty.class
97          Pair<Integer, EnumMap<T, Integer>> indices = getCounterIndices(objectName, counterEnum, COUNTER_INDEX_MAP);
98          if (indices == null) {
99              return null;
100         }
101         int objectIndex = indices.getA();
102         EnumMap<T, Integer> enumIndexMap = indices.getB();
103         // The above test checks validity of objectName as an index but it could still
104         // fail to read
105         MemorySegment pPerfData = readPerfDataBuffer(objectName);
106         if (pPerfData == null) {
107             return null;
108         }
109         // Buffer is now successfully populated.
110         // See format at
111         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
112 
113         // Start with a data header (PERF_DATA_BLOCK)
114         // Then iterate one or more objects
115         // Each object contains
116         // [ ] Object Type header (PERF_OBJECT_TYPE)
117         // [ ][ ][ ] Multiple counter definitions (PERF_COUNTER_DEFINITION)
118         // Then after object(s), multiple:
119         // [ ] Instance Definition
120         // [ ] Instance name
121         // [ ] Counter Block
122         // [ ][ ][ ] Counter data for each definition above
123 
124         // Store timestamp
125         long perfTime100nSec = pPerfData.get(JAVA_LONG, PERF_DATA_BLOCK_PerfTime100nSec); // 1601
126         long now = ParseUtil.filetimeToUtcMs(perfTime100nSec, false); // 1970
127 
128         // Iterate object types.
129         long perfObjectOffset = pPerfData.get(JAVA_INT, PERF_DATA_BLOCK_HeaderLength);
130         int numObjectTypes = pPerfData.get(JAVA_INT, PERF_DATA_BLOCK_NumObjectTypes);
131         for (int obj = 0; obj < numObjectTypes; obj++) {
132             // Some counters will require multiple objects so we iterate until we find the
133             // right one. e.g. Process (230) is by itself but Thread (232) has Process
134             // object first
135             int objectNameTitleIndex = pPerfData.get(JAVA_INT,
136                     perfObjectOffset + PERF_OBJECT_TYPE_ObjectNameTitleIndex);
137             if (objectNameTitleIndex == objectIndex) {
138                 // We found a matching object.
139 
140                 // Counter definitions start after the object header
141                 long perfCounterOffset = perfObjectOffset
142                         + pPerfData.get(JAVA_INT, perfObjectOffset + PERF_OBJECT_TYPE_HeaderLength);
143                 int numCounters = pPerfData.get(JAVA_INT, perfObjectOffset + PERF_OBJECT_TYPE_NumCounters);
144 
145                 // Iterate counter definitions and fill maps with counter offsets and sizes
146                 Map<Integer, Integer> counterOffsetMap = new HashMap<>();
147                 Map<Integer, Integer> counterSizeMap = new HashMap<>();
148                 for (int counter = 0; counter < numCounters; counter++) {
149                     int counterNameTitleIndex = pPerfData.get(JAVA_INT,
150                             perfCounterOffset + PERF_COUNTER_DEF_CounterNameTitleIndex);
151                     int counterOffset = pPerfData.get(JAVA_INT, perfCounterOffset + PERF_COUNTER_DEF_CounterOffset);
152                     int counterSize = pPerfData.get(JAVA_INT, perfCounterOffset + PERF_COUNTER_DEF_CounterSize);
153                     counterOffsetMap.put(counterNameTitleIndex, counterOffset);
154                     counterSizeMap.put(counterNameTitleIndex, counterSize);
155                     // Increment for next Counter
156                     perfCounterOffset += pPerfData.get(JAVA_INT, perfCounterOffset + PERF_COUNTER_DEF_ByteLength);
157                 }
158 
159                 // Instances start after all the object definitions. The DefinitionLength
160                 // includes both the header and all the definitions.
161                 long perfInstanceOffset = perfObjectOffset
162                         + pPerfData.get(JAVA_INT, perfObjectOffset + PERF_OBJECT_TYPE_DefinitionLength);
163                 int numInstances = pPerfData.get(JAVA_INT, perfObjectOffset + PERF_OBJECT_TYPE_NumInstances);
164 
165                 // Iterate instances and fill map
166                 T[] counterKeys = counterEnum.getEnumConstants();
167                 List<Map<T, Object>> counterMaps = new ArrayList<>(numInstances);
168                 for (int inst = 0; inst < numInstances; inst++) {
169                     int instanceByteLength = pPerfData.get(JAVA_INT, perfInstanceOffset + PERF_INSTANCE_DEF_ByteLength);
170                     int nameOffset = pPerfData.get(JAVA_INT, perfInstanceOffset + PERF_INSTANCE_DEF_NameOffset);
171                     long perfCounterBlockOffset = perfInstanceOffset + instanceByteLength;
172 
173                     // Populate the enumMap
174                     Map<T, Object> counterMap = new EnumMap<>(counterEnum);
175                     // First enum index is the name, ignore the counter text which is used for other
176                     // purposes
177                     counterMap.put(counterKeys[0], readWideString(pPerfData.asSlice(perfInstanceOffset + nameOffset)));
178                     for (int i = 1; i < counterKeys.length; i++) {
179                         T key = counterKeys[i];
180                         int keyIndex = enumIndexMap.get(key);
181                         // All entries in size map have corresponding entry in offset map
182                         int size = counterSizeMap.getOrDefault(keyIndex, 0);
183                         // Currently, only DWORDs (4 bytes) and ULONGLONGs (8 bytes) are used to provide
184                         // counter values.
185                         if (size == 4) {
186                             counterMap.put(key,
187                                     pPerfData.get(JAVA_INT, perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
188                         } else if (size == 8) {
189                             counterMap.put(key,
190                                     pPerfData.get(JAVA_LONG, perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
191                         } else {
192                             // If counter defined in enum isn't in registry, fail
193                             return null;
194                         }
195                     }
196                     counterMaps.add(counterMap);
197 
198                     // counters at perfCounterBlockOffset + appropriate offset per enum
199                     // use pPerfData.getInt or getLong as determined by counter size
200                     // Currently, only DWORDs (4 bytes) and ULONGLONGs (8 bytes) are used to provide
201                     // counter values.
202 
203                     // Increment to next instance
204                     perfInstanceOffset = perfCounterBlockOffset
205                             + pPerfData.get(JAVA_INT, perfCounterBlockOffset + PERF_COUNTER_BLOCK_ByteLength);
206                 }
207                 // We've found the necessary object and are done, no need to look at any other
208                 // objects (shouldn't be any). Return results
209                 return new Triplet<>(counterMaps, perfTime100nSec, now);
210             }
211             // Increment for next object
212             perfObjectOffset += pPerfData.get(JAVA_INT, perfObjectOffset + PERF_OBJECT_TYPE_TotalByteLength);
213         }
214         // Failed, return null
215         return null;
216     }

218     /**
219      * Read the performance data for a counter object from the registry.
220      *
221      * @param objectName The counter object for which to fetch data. It is the user's responsibility to ensure this key
222      *                   exists in {@link #COUNTER_INDEX_MAP}.
223      * @return A buffer containing the data if successful, null otherwise.
224      */
225     private static synchronized MemorySegment readPerfDataBuffer(String objectName) {
226         // Need this index as a string
227         String objectIndexStr = Integer.toString(COUNTER_INDEX_MAP.get(objectName));
228 
229         // Now load the data from the registry.
230         // Use a global arena so the returned segment outlives this method
231         Arena arena = Arena.ofAuto();
232         try {
233             MemorySegment lpValueName = toWideString(arena, objectIndexStr);
234             MemorySegment lpcbData = arena.allocate(JAVA_INT);
235             lpcbData.set(JAVA_INT, 0, maxPerfBufferSize);
236             MemorySegment pPerfData = arena.allocate(maxPerfBufferSize);
237 
238             int ret = RegQueryValueEx(MemorySegment.ofAddress(WinRegFFM.HKEY_PERFORMANCE_DATA), lpValueName, 0,
239                     MemorySegment.NULL, pPerfData, lpcbData);
240             if (ret != ERROR_SUCCESS && ret != ERROR_MORE_DATA) {
241                 LOG.error("Error reading performance data from registry for {}.", objectName);
242                 return null;
243             }
244             // Grow buffer as needed to fit the data
245             while (ret == ERROR_MORE_DATA) {
246                 maxPerfBufferSize += 8192;
247                 lpcbData.set(JAVA_INT, 0, maxPerfBufferSize);
248                 pPerfData = arena.allocate(maxPerfBufferSize);
249                 ret = RegQueryValueEx(MemorySegment.ofAddress(WinRegFFM.HKEY_PERFORMANCE_DATA), lpValueName, 0,
250                         MemorySegment.NULL, pPerfData, lpcbData);
251             }
252             if (ret != ERROR_SUCCESS) {
253                 LOG.error("Error reading performance data from registry for {} (ret={}).", objectName, ret);
254                 return null;
255             }
256             return pPerfData;
257         } catch (Throwable t) {
258             LOG.error("Error reading performance data from registry for {}.", objectName, t);
259             return null;
260         }
261     }

263     /*
264      * Registry entries subordinate to HKEY_PERFORMANCE_TEXT key reference the text strings that describe counters in US
265      * English. Not supported in Windows 2000.
266      *
267      * With the "Counter" value, the resulting array contains alternating index/name pairs "1", "1847", "2", "System",
268      * "4", "Memory", ...
269      *
270      * These pairs are translated to a map for later lookup.
271      *
272      * @return An unmodifiable map containing counter name strings as keys and indices as integer values if the key is
273      * read successfully; an empty map otherwise.
274      */
275     private static Map<String, Integer> mapCounterIndicesFromRegistry() {
276         try {
277             String[] counterText = Advapi32UtilFFM.registryGetStringArray(
278                     MemorySegment.ofAddress(WinRegFFM.HKEY_LOCAL_MACHINE), HKEY_PERFORMANCE_TEXT, COUNTER);
279             return buildCounterIndexMap(counterText);
280         } catch (Exception e) {
281             LOG.error(
282                     "Unable to locate English counter names in registry Perflib 009. Counters may need to be rebuilt: ",
283                     e);
284         }
285         return buildCounterIndexMap(null);
286     }
