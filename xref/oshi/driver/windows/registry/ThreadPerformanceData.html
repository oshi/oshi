40  /**
41   * Utility to read thread data from HKEY_PERFORMANCE_DATA information with
42   * backup from Performance Counters or WMI
43   */
44  @ThreadSafe
45  public final class ThreadPerformanceData {
46  
47      private static final String THREAD = "Thread";
48  
49      private ThreadPerformanceData() {
50      }
51  
52      /**
53       * Query the registry for thread performance counters
54       *
55       * @param pids
56       *            An optional collection of thread IDs to filter the list to. May be
57       *            null for no filtering.
58       * @return A map with Thread ID as the key and a {@link PerfCounterBlock} object
59       *         populated with performance counter information if successful, or null
60       *         otherwise.
61       */
62      public static Map<Integer, PerfCounterBlock> buildThreadMapFromRegistry(Collection<Integer> pids) {
63          // Grab the data from the registry.
64          Triplet<List<Map<ThreadPerformanceProperty, Object>>, Long, Long> threadData = HkeyPerformanceDataUtil
65                  .readPerfDataFromRegistry(THREAD, ThreadPerformanceProperty.class);
66          if (threadData == null) {
67              return null;
68          }
69          List<Map<ThreadPerformanceProperty, Object>> threadInstanceMaps = threadData.getA();
70          long perfTime100nSec = threadData.getB(); // 1601
71          long now = threadData.getC(); // 1970 epoch
72  
73          // Create a map and fill it
74          Map<Integer, PerfCounterBlock> threadMap = new HashMap<>();
75          // Iterate instances.
76          for (Map<ThreadPerformanceProperty, Object> threadInstanceMap : threadInstanceMaps) {
77              int pid = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.IDPROCESS)).intValue();
78              if ((pids == null || pids.contains(pid)) && pid > 0) {
79                  int tid = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.IDTHREAD)).intValue();
80                  String name = (String) threadInstanceMap.get(ThreadPerformanceProperty.NAME);
81                  long upTime = (perfTime100nSec - (Long) threadInstanceMap.get(ThreadPerformanceProperty.ELAPSEDTIME))
82                          / 10_000L;
83                  if (upTime < 1) {
84                      upTime = 1;
85                  }
86                  long user = ((Long) threadInstanceMap.get(ThreadPerformanceProperty.PERCENTUSERTIME)).longValue()
87                          / 10_000L;
88                  long kernel = ((Long) threadInstanceMap.get(ThreadPerformanceProperty.PERCENTPRIVILEGEDTIME))
89                          .longValue() / 10_000L;
90                  int priority = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.PRIORITYCURRENT)).intValue();
91                  int threadState = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.THREADSTATE)).intValue();
92                  int threadWaitReason = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.THREADWAITREASON))
93                          .intValue();
94                  // Start address is pointer sized when fetched from registry, so this could be
95                  // either Integer (uint32) or Long depending on OS bitness
96                  Object addr = threadInstanceMap.get(ThreadPerformanceProperty.STARTADDRESS);
97                  long startAddr = addr.getClass().equals(Long.class) ? (Long) addr
98                          : Integer.toUnsignedLong((Integer) addr);
99                  int contextSwitches = ((Integer) threadInstanceMap.get(ThreadPerformanceProperty.CONTEXTSWITCHESPERSEC))
100                         .intValue();
101                 threadMap.put(tid, new PerfCounterBlock(name, tid, pid, now - upTime, user, kernel, priority,
102                         threadState, threadWaitReason, startAddr, contextSwitches));
103             }
104         }
105         return threadMap;
106     }

108     /**
109      * Query PerfMon for thread performance counters
110      *
111      * @param pids
112      *            An optional collection of process IDs to filter the list to. May
113      *            be null for no filtering.
114      * @return A map with Thread ID as the key and a {@link PerfCounterBlock} object
115      *         populated with performance counter information.
116      */
117     public static Map<Integer, PerfCounterBlock> buildThreadMapFromPerfCounters(Collection<Integer> pids) {
118         Map<Integer, PerfCounterBlock> threadMap = new HashMap<>();
119         Pair<List<String>, Map<ThreadPerformanceProperty, List<Long>>> instanceValues = ThreadInformation
120                 .queryThreadCounters();
121         long now = System.currentTimeMillis(); // 1970 epoch
122         List<String> instances = instanceValues.getA();
123         Map<ThreadPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
124         List<Long> tidList = valueMap.get(ThreadPerformanceProperty.IDTHREAD);
125         List<Long> pidList = valueMap.get(ThreadPerformanceProperty.IDPROCESS);
126         List<Long> userList = valueMap.get(ThreadPerformanceProperty.PERCENTUSERTIME); // 100-nsec
127         List<Long> kernelList = valueMap.get(ThreadPerformanceProperty.PERCENTPRIVILEGEDTIME); // 100-nsec
128         List<Long> startTimeList = valueMap.get(ThreadPerformanceProperty.ELAPSEDTIME); // filetime
129         List<Long> priorityList = valueMap.get(ThreadPerformanceProperty.PRIORITYCURRENT);
130         List<Long> stateList = valueMap.get(ThreadPerformanceProperty.THREADSTATE);
131         List<Long> waitReasonList = valueMap.get(ThreadPerformanceProperty.THREADWAITREASON);
132         List<Long> startAddrList = valueMap.get(ThreadPerformanceProperty.STARTADDRESS);
133         List<Long> contextSwitchesList = valueMap.get(ThreadPerformanceProperty.CONTEXTSWITCHESPERSEC);
134 
135         int nameIndex = 0;
136         for (int inst = 0; inst < instances.size(); inst++) {
137             int pid = pidList.get(inst).intValue();
138             if (pids == null || pids.contains(pid)) {
139                 int tid = tidList.get(inst).intValue();
140                 String name = Integer.toString(nameIndex++);
141                 long startTime = startTimeList.get(inst);
142                 startTime = FILETIME.filetimeToDate((int) (startTime >> 32), (int) (startTime & 0xffffffffL)).getTime();
143                 if (startTime > now) {
144                     startTime = now - 1;
145                 }
146                 long user = userList.get(inst) / 10_000L;
147                 long kernel = kernelList.get(inst) / 10_000L;
148                 int priority = priorityList.get(inst).intValue();
149                 int threadState = stateList.get(inst).intValue();
150                 int threadWaitReason = waitReasonList.get(inst).intValue();
151                 long startAddr = startAddrList.get(inst).longValue();
152                 int contextSwitches = contextSwitchesList.get(inst).intValue();
153 
154                 // if creation time value is less than current millis, it's in 1970 epoch,
155                 // otherwise it's 1601 epoch and we must convert
156                 threadMap.put(tid, new PerfCounterBlock(name, tid, pid, startTime, user, kernel, priority, threadState,
157                         threadWaitReason, startAddr, contextSwitches));
158             }
159         }
160         return threadMap;
161     }

163     /**
164      * Class to encapsulate data from the registry performance counter block
165      */
166     @Immutable
167     public static class PerfCounterBlock {
168         private final String name;
169         private final int threadID;
170         private final int owningProcessID;
171         private final long startTime;
172         private final long userTime;
173         private final long kernelTime;
174         private final int priority;
175         private final int threadState;
176         private final int threadWaitReason;
177         private final long startAddress;
178         private final int contextSwitches;
179 
180         public PerfCounterBlock(String name, int threadID, int owningProcessID, long startTime, long userTime,
181                 long kernelTime, int priority, int threadState, int threadWaitReason, long startAddress,
182                 int contextSwitches) {
183             this.name = name;
184             this.threadID = threadID;
185             this.owningProcessID = owningProcessID;
186             this.startTime = startTime;
187             this.userTime = userTime;
188             this.kernelTime = kernelTime;
189             this.priority = priority;
190             this.threadState = threadState;
191             this.threadWaitReason = threadWaitReason;
192             this.startAddress = startAddress;
193             this.contextSwitches = contextSwitches;
194         }
195 
196         /**
197          * @return the name
198          */
199         public String getName() {
200             return name;
201         }
202 
203         /**
204          * @return the threadID
205          */
206         public int getThreadID() {
207             return threadID;
208         }
209 
210         /**
211          * @return the owningProcessID
212          */
213         public int getOwningProcessID() {
214             return owningProcessID;
215         }
216 
217         /**
218          * @return the startTime
219          */
220         public long getStartTime() {
221             return startTime;
222         }
223 
224         /**
225          * @return the userTime
226          */
227         public long getUserTime() {
228             return userTime;
229         }
230 
231         /**
232          * @return the kernelTime
233          */
234         public long getKernelTime() {
235             return kernelTime;
236         }
237 
238         /**
239          * @return the priority
240          */
241         public int getPriority() {
242             return priority;
243         }
244 
245         /**
246          * @return the threadState
247          */
248         public int getThreadState() {
249             return threadState;
250         }
251 
252         /**
253          * @return the threadWaitReason
254          */
255         public int getThreadWaitReason() {
256             return threadWaitReason;
257         }
258 
259         /**
260          * @return the startMemoryAddress
261          */
262         public long getStartAddress() {
263             return startAddress;
264         }
265 
266         /**
267          * @return the contextSwitches
268          */
269         public int getContextSwitches() {
270             return contextSwitches;
271         }
272     }
