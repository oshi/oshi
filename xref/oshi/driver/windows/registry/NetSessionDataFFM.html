5   package oshi.driver.windows.registry;
6   
7   import static java.lang.foreign.MemorySegment.NULL;
8   import static java.lang.foreign.ValueLayout.ADDRESS;
9   import static java.lang.foreign.ValueLayout.JAVA_INT;
10  import static oshi.ffm.windows.WindowsForeignFunctions.readWideString;
11  
12  import java.lang.foreign.Arena;
13  import java.lang.foreign.MemorySegment;
14  import java.util.ArrayList;
15  import java.util.List;
16  
17  import oshi.annotation.concurrent.ThreadSafe;
18  import oshi.ffm.windows.Netapi32FFM;
19  import oshi.software.os.OSSession;
20  
21  /**
22   * Utility to read network session data via FFM.
23   */
24  @ThreadSafe
25  public final class NetSessionDataFFM {
26  
27      // SESSION_INFO_10 on 64-bit: LPWSTR cname(8) + LPWSTR username(8) + DWORD time(4) + DWORD idle_time(4) = 24
28      private static final long SESSION_INFO_10_SIZE = 24;
29      private static final long OFFSET_CNAME = 0;
30      private static final long OFFSET_USERNAME = 8;
31      private static final long OFFSET_TIME = 16;
32  
33      private NetSessionDataFFM() {
34      }
35  
36      public static List<OSSession> queryUserSessions() {
37          List<OSSession> sessions = new ArrayList<>();
38          try (Arena arena = Arena.ofConfined()) {
39              MemorySegment bufptr = arena.allocate(ADDRESS);
40              MemorySegment entriesRead = arena.allocate(JAVA_INT);
41              MemorySegment totalEntries = arena.allocate(JAVA_INT);
42  
43              int status = Netapi32FFM.NetSessionEnum(NULL, NULL, NULL, 10, bufptr, Netapi32FFM.MAX_PREFERRED_LENGTH,
44                      entriesRead, totalEntries, NULL);
45              if (status == 0) {
46                  MemorySegment buf = bufptr.get(ADDRESS, 0);
47                  try {
48                      int count = entriesRead.get(JAVA_INT, 0);
49                      if (count > 0) {
50                          buf = buf.reinterpret(SESSION_INFO_10_SIZE * count);
51                          for (int i = 0; i < count; i++) {
52                              long base = i * SESSION_INFO_10_SIZE;
53                              MemorySegment pCname = buf.get(ADDRESS, base + OFFSET_CNAME).reinterpret(512);
54                              MemorySegment pUsername = buf.get(ADDRESS, base + OFFSET_USERNAME).reinterpret(512);
55                              int time = buf.get(JAVA_INT, base + OFFSET_TIME);
56                              String cname = readWideString(pCname);
57                              String username = readWideString(pUsername);
58                              long logonTime = System.currentTimeMillis() - (1000L * time);
59                              sessions.add(new OSSession(username, "Network session", logonTime, cname));
60                          }
61                      }
62                  } finally {
63                      Netapi32FFM.NetApiBufferFree(buf);
64                  }
65              }
66          } catch (Throwable t) {
67              // Silently return what we have
68          }
69          return sessions;
70      }
71  }
