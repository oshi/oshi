29  /**
30   * Utility to read WTS session data via FFM.
31   */
32  @ThreadSafe
33  public final class SessionWtsDataFFM {
34  
35      private static final int WTS_ACTIVE = 0;
36      private static final int WTS_CLIENTADDRESS = 14;
37      private static final int WTS_SESSIONINFO = 24;
38      private static final int WTS_CLIENTPROTOCOLTYPE = 16;
39  
40      private static final int AF_INET = 2;
41      private static final int AF_INET6 = 23;
42  
43      // WTS_SESSION_INFOW on 64-bit: DWORD(4) + pad(4) + LPWSTR(8) + DWORD(4) + pad(4) = 24
44      private static final long SESSION_INFO_SIZE = 24;
45      private static final long SESSION_INFO_SESSIONID_OFFSET = 0;
46      private static final long SESSION_INFO_WINSTATION_OFFSET = 8;
47      private static final long SESSION_INFO_STATE_OFFSET = 16;
48  
49      // WTSINFOW offsets (64-bit):
50      // State(4)+SessionId(4)+counters(6*4=24)+WinStationName(32*2)+Domain(17*2)+UserName(21*2)
51      // then padding to 8-byte align, then 5 LARGE_INTEGERs
52      // State: 0, SessionId: 4, IncomingBytes: 8, ..., OutgoingCompressedBytes: 28
53      // WinStationName: 32, Domain: 32+64=96, UserName: 96+34=130
54      // After UserName (130+42=172), pad to 176, then ConnectTime(8), DisconnectTime(8), LastInputTime(8), LogonTime(8)
55      // LogonTime offset = 176 + 24 = 200
56      private static final int WINSTATIONNAME_LENGTH = 32;
57      private static final int DOMAIN_LENGTH = 17;
58      private static final int USERNAME_LENGTH = 20;
59      private static final long WTSINFO_USERNAME_OFFSET = 32 + WINSTATIONNAME_LENGTH * 2L + DOMAIN_LENGTH * 2L;
60      // After fixed-size char arrays, align to 8 for LARGE_INTEGERs
61      private static final long WTSINFO_AFTER_STRINGS = 32 + WINSTATIONNAME_LENGTH * 2L + DOMAIN_LENGTH * 2L
62              + (USERNAME_LENGTH + 1) * 2L;
63      private static final long WTSINFO_LOGONTIME_OFFSET;
64  
65      static {
66          // Align to 8 bytes, then skip ConnectTime(8) + DisconnectTime(8) + LastInputTime(8)
67          long aligned = (WTSINFO_AFTER_STRINGS + 7) & ~7L;
68          WTSINFO_LOGONTIME_OFFSET = aligned + 24; // 4th LARGE_INTEGER
69      }
70  
71      // WTS_CLIENT_ADDRESS: DWORD AddressFamily(4) + byte[20] Address
72      private static final long CLIENT_ADDR_FAMILY_OFFSET = 0;
73      private static final long CLIENT_ADDR_ADDRESS_OFFSET = 4;
74  
75      private SessionWtsDataFFM() {
76      }
77  
78      public static List<OSSession> queryUserSessions() {
79          List<OSSession> sessions = new ArrayList<>();
80          try (Arena arena = Arena.ofConfined()) {
81              MemorySegment ppSessionInfo = arena.allocate(ADDRESS);
82              MemorySegment pCount = arena.allocate(JAVA_INT);
83  
84              if (!Wtsapi32FFM.WTSEnumerateSessions(Wtsapi32FFM.WTS_CURRENT_SERVER_HANDLE, 0, 1, ppSessionInfo, pCount)) {
85                  return sessions;
86              }
87  
88              MemorySegment pSessionInfo = ppSessionInfo.get(ADDRESS, 0);
89              try {
90                  int count = pCount.get(JAVA_INT, 0);
91  
92                  if (count > 0) {
93                      // Reinterpret to cover all session entries
94                      pSessionInfo = pSessionInfo.reinterpret(SESSION_INFO_SIZE * count);
95  
96                      MemorySegment ppBuffer = arena.allocate(ADDRESS);
97                      MemorySegment pBytes = arena.allocate(JAVA_INT);
98  
99                      for (int i = 0; i < count; i++) {
100                         long base = i * SESSION_INFO_SIZE;
101                         int state = pSessionInfo.get(JAVA_INT, base + SESSION_INFO_STATE_OFFSET);
102                         if (state != WTS_ACTIVE) {
103                             continue;
104                         }
105                         int sessionId = pSessionInfo.get(JAVA_INT, base + SESSION_INFO_SESSIONID_OFFSET);
106                         MemorySegment pWinStation = pSessionInfo.get(ADDRESS, base + SESSION_INFO_WINSTATION_OFFSET);
107 
108                         // Query protocol type
109                         if (!Wtsapi32FFM.WTSQuerySessionInformation(Wtsapi32FFM.WTS_CURRENT_SERVER_HANDLE, sessionId,
110                                 WTS_CLIENTPROTOCOLTYPE, ppBuffer, pBytes)) {
111                             continue;
112                         }
113                         MemorySegment pBuffer = ppBuffer.get(ADDRESS, 0).reinterpret(2);
114                         short protocolType;
115                         try {
116                             protocolType = pBuffer.get(JAVA_SHORT, 0);
117                         } finally {
118                             Wtsapi32FFM.WTSFreeMemory(pBuffer);
119                         }
120 
121                         // Only non-console sessions; console sessions come from registry
122                         if (protocolType <= 0) {
123                             continue;
124                         }
125                         String device = readWideString(pWinStation.reinterpret(256));
126 
127                         // Query WTSINFO for username and logon time
128                         if (!Wtsapi32FFM.WTSQuerySessionInformation(Wtsapi32FFM.WTS_CURRENT_SERVER_HANDLE, sessionId,
129                                 WTS_SESSIONINFO, ppBuffer, pBytes)) {
130                             continue;
131                         }
132                         pBuffer = ppBuffer.get(ADDRESS, 0).reinterpret(WTSINFO_LOGONTIME_OFFSET + 8);
133                         String userName;
134                         long logonTime;
135                         try {
136                             userName = readWideString(
137                                     pBuffer.asSlice(WTSINFO_USERNAME_OFFSET, (USERNAME_LENGTH + 1) * 2L));
138                             logonTime = ParseUtil.filetimeToUtcMs(pBuffer.get(JAVA_LONG, WTSINFO_LOGONTIME_OFFSET),
139                                     false);
140                         } finally {
141                             Wtsapi32FFM.WTSFreeMemory(pBuffer);
142                         }
143 
144                         // Query client address
145                         if (!Wtsapi32FFM.WTSQuerySessionInformation(Wtsapi32FFM.WTS_CURRENT_SERVER_HANDLE, sessionId,
146                                 WTS_CLIENTADDRESS, ppBuffer, pBytes)) {
147                             sessions.add(new OSSession(userName, device, logonTime, "::"));
148                             continue;
149                         }
150                         pBuffer = ppBuffer.get(ADDRESS, 0).reinterpret(24);
151                         int addrFamily;
152                         byte[] address;
153                         try {
154                             addrFamily = pBuffer.get(JAVA_INT, CLIENT_ADDR_FAMILY_OFFSET);
155                             address = pBuffer.asSlice(CLIENT_ADDR_ADDRESS_OFFSET, 20)
156                                     .toArray(java.lang.foreign.ValueLayout.JAVA_BYTE);
157                         } finally {
158                             Wtsapi32FFM.WTSFreeMemory(pBuffer);
159                         }
160 
161                         String host = "::";
162                         if (addrFamily == AF_INET) {
163                             try {
164                                 host = InetAddress.getByAddress(Arrays.copyOfRange(address, 2, 6)).getHostAddress();
165                             } catch (UnknownHostException _) {
166                                 host = "Illegal length IP Array";
167                             }
168                         } else if (addrFamily == AF_INET6) {
169                             int[] ipArray = convertBytesToInts(address);
170                             host = ParseUtil.parseUtAddrV6toIP(ipArray);
171                         }
172                         sessions.add(new OSSession(userName, device, logonTime, host));
173                     }
174                 }
175             } finally {
176                 Wtsapi32FFM.WTSFreeMemory(pSessionInfo);
177             }
178         } catch (Throwable _) {
179             // Silently return what we have
180         }
181         return sessions;
182     }
183 
184     private static int[] convertBytesToInts(byte[] address) {
185         IntBuffer intBuf = ByteBuffer.wrap(Arrays.copyOfRange(address, 2, 18)).order(ByteOrder.BIG_ENDIAN)
186                 .asIntBuffer();
187         int[] array = new int[intBuf.remaining()];
188         intBuf.get(array);
189         return array;
190     }
