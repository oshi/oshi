24  package oshi.driver.windows.registry;
25  
26  import java.util.Collection;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  import oshi.driver.windows.perfmon.ProcessInformation.ProcessPerformanceProperty;
32  import oshi.software.os.OSProcess;
33  import oshi.software.os.OperatingSystem;
34  import oshi.util.tuples.Triplet;
35  
36  public final class ProcessPerformanceData {
37  
38      private static final String PROCESS = "Process";
39  
40      private ProcessPerformanceData() {
41      }
42  
43      /**
44       * Query the registry for process performance counters
45       *
46       * @param os
47       *            The OperatingSystem object
48       * @param pids
49       *            An optional collection of process IDs to filter the list to. May
50       *            be null for no filtering.
51       * @return A map with Process ID as the key and an OSProcess object populated
52       *         with performance counter information if successful, or null
53       *         otherwise.
54       */
55      public static Map<Integer, OSProcess> buildProcessMapFromRegistry(OperatingSystem os, Collection<Integer> pids) {
56          // Grab the data from the registry.
57          Triplet<List<Map<ProcessPerformanceProperty, Object>>, Long, Long> processData = HkeyPerformanceDataUtil
58                  .readPerfDataFromRegistry(PROCESS, ProcessPerformanceProperty.class);
59          if (processData == null) {
60              return null;
61          }
62          List<Map<ProcessPerformanceProperty, Object>> processInstanceMaps = processData.getA();
63          long perfTime100nSec = processData.getB(); // 1601
64          long now = processData.getC(); // 1970 epoch
65  
66          // Create a map and fill it
67          Map<Integer, OSProcess> processMap = new HashMap<>();
68          // Iterate instances.
69          for (Map<ProcessPerformanceProperty, Object> processInstanceMap : processInstanceMaps) {
70              int pid = ((Integer) processInstanceMap.get(ProcessPerformanceProperty.PROCESSID)).intValue();
71              String name = (String) processInstanceMap.get(ProcessPerformanceProperty.NAME);
72              if ((pids == null || pids.contains(pid)) && !"_Total".equals(name)) {
73                  OSProcessml#OSProcess">OSProcess proc = new OSProcess(os);
74                  proc.setProcessID(pid);
75                  proc.setName(name);
76                  long upTime = (perfTime100nSec - (Long) processInstanceMap.get(ProcessPerformanceProperty.CREATIONDATE))
77                          / 10_000L;
78                  proc.setUpTime(upTime < 1L ? 1L : upTime);
79                  proc.setStartTime(now - upTime);
80                  proc.setBytesRead((Long) processInstanceMap.get(ProcessPerformanceProperty.READTRANSFERCOUNT));
81                  proc.setBytesWritten((Long) processInstanceMap.get(ProcessPerformanceProperty.WRITETRANSFERCOUNT));
82                  proc.setResidentSetSize((Long) processInstanceMap.get(ProcessPerformanceProperty.PRIVATEPAGECOUNT));
83                  proc.setParentProcessID((Integer) processInstanceMap.get(ProcessPerformanceProperty.PARENTPROCESSID));
84                  proc.setPriority((Integer) processInstanceMap.get(ProcessPerformanceProperty.PRIORITY));
85  
86                  processMap.put(pid, proc);
87              }
88          }
89          return processMap;
90      }
91  }
