5   package oshi.driver.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.driver.common.windows.wmi.OhmHardware;
9   import oshi.ffm.util.platform.windows.WbemcliUtilFFM.WmiQuery;
10  import oshi.ffm.util.platform.windows.WbemcliUtilFFM.WmiResult;
11  import oshi.ffm.util.platform.windows.WmiQueryHandlerFFM;
12  
13  /**
14   * Utility to query Open Hardware Monitor WMI data for Hardware using FFM.
15   */
16  @ThreadSafe
17  public final class OhmHardwareFFM extends OhmHardware {
18  
19      private OhmHardwareFFM() {
20      }
21  
22      /**
23       * Queries the hardware identifiers for a monitored type.
24       *
25       * @param h           An instantiated {@link WmiQueryHandlerFFM}. User should have already initialized COM.
26       * @param typeToQuery which type to filter based on
27       * @param typeName    the name of the type
28       * @return WmiResult of hardware identifier properties.
29       */
30      public static WmiResult<IdentifierProperty> queryHwIdentifier(WmiQueryHandlerFFM h, String typeToQuery,
31              String typeName) {
32          WmiQuery<IdentifierProperty> hwIdentifierQuery = new WmiQuery<>(OHM_NAMESPACE,
33                  buildHardwareWmiClassNameWithWhere(typeToQuery, typeName), IdentifierProperty.class);
34          return h.queryWMI(hwIdentifierQuery, false);
35      }
36  }
