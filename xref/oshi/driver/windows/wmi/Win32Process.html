24  package oshi.driver.windows.wmi;
25  
26  import java.util.Collection;
27  import java.util.Objects;
28  import java.util.Set;
29  import java.util.stream.Collectors;
30  
31  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery; //NOSONAR squid:S1191
32  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
33  
34  import oshi.annotation.concurrent.ThreadSafe;
35  import oshi.util.platform.windows.WmiQueryHandler;
36  
37  /**
38   * Utility to query WMI class {@code Win32_Process}
39   */
40  @ThreadSafe
41  public final class Win32Process {
42  
43      private static final String WIN32_PROCESS = "Win32_Process";
44  
45      /**
46       * Process command lines.
47       */
48      public enum CommandLineProperty {
49          PROCESSID, COMMANDLINE;
50      }
51  
52      /**
53       * Process properties accessible from WTSEnumerateProcesses in Vista+
54       */
55      public enum ProcessXPProperty {
56          PROCESSID, NAME, KERNELMODETIME, USERMODETIME, THREADCOUNT, PAGEFILEUSAGE, HANDLECOUNT, EXECUTABLEPATH;
57      }
58  
59      private Win32Process() {
60      }
61  
62      /**
63       * Returns process command lines
64       *
65       * @param pidsToQuery
66       *            Process IDs to query for command lines. Pass {@code null} to query
67       *            all processes.
68       * @return A {@link WmiResult} containing process IDs and command lines used to
69       *         start the provided processes.
70       */
71      public static WmiResult<CommandLineProperty> queryCommandLines(Set<Integer> pidsToQuery) {
72          String sb = WIN32_PROCESS;
73          if (pidsToQuery != null) {
74              sb += " WHERE ProcessID="
75                      + pidsToQuery.stream().map(String::valueOf).collect(Collectors.joining(" OR PROCESSID="));
76          }
77          WmiQuery<CommandLineProperty> commandLineQuery = new WmiQuery<>(sb, CommandLineProperty.class);
78          return Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(commandLineQuery);
79      }
80  
81      /**
82       * Returns process info
83       *
84       * @param pids
85       *            Process IDs to query.
86       * @return Information on the provided processes.
87       */
88      public static WmiResult<ProcessXPProperty> queryProcesses(Collection<Integer> pids) {
89          String sb = WIN32_PROCESS;
90          if (pids != null) {
91              sb += " WHERE ProcessID="
92                      + pids.stream().map(String::valueOf).collect(Collectors.joining(" OR PROCESSID="));
93          }
94          WmiQuery<ProcessXPProperty> processQueryXP = new WmiQuery<>(sb, ProcessXPProperty.class);
95          return Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(processQueryXP);
96      }
97  }
