24  package oshi.driver.windows.wmi;
25  
26  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery; //NOSONAR squid:S1191
27  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.util.platform.windows.WmiQueryHandler;
31  
32  /**
33   * Utility to query WMI class {@code MSAcpi_ThermalZoneTemperature}
34   */
35  @ThreadSafe
36  public final class MSAcpiThermalZoneTemperature {
37  
38      public static final String WMI_NAMESPACE = "ROOT\\WMI";
39      private static final String MS_ACPI_THERMAL_ZONE_TEMPERATURE = "MSAcpi_ThermalZoneTemperature";
40  
41      /**
42       * Current temperature property.
43       */
44      public enum TemperatureProperty {
45          CURRENTTEMPERATURE;
46      }
47  
48      private MSAcpiThermalZoneTemperature() {
49      }
50  
51      /**
52       * Queries the current temperature
53       *
54       * @return Temperature at thermal zone in tenths of degrees Kelvin.
55       */
56      public static WmiResult<TemperatureProperty> queryCurrentTemperature() {
57          WmiQuery<TemperatureProperty> curTempQuery = new WmiQuery<>(WMI_NAMESPACE, MS_ACPI_THERMAL_ZONE_TEMPERATURE,
58                  TemperatureProperty.class);
59          return WmiQueryHandler.createInstance().queryWMI(curTempQuery);
60      }
61  }
