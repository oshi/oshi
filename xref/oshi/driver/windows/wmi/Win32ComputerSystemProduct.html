24  package oshi.driver.windows.wmi;
25  
26  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery; //NOSONAR squid:S1191
27  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.util.platform.windows.WmiQueryHandler;
31  
32  /**
33   * Utility to query WMI class {@code Win32_ComputerSystemProduct}
34   */
35  @ThreadSafe
36  public final class Win32ComputerSystemProduct {
37  
38      private static final String WIN32_COMPUTER_SYSTEM_PRODUCT = "Win32_ComputerSystemProduct";
39  
40      /**
41       * Computer System ID number
42       */
43      public enum ComputerSystemProductProperty {
44          IDENTIFYINGNUMBER, UUID;
45      }
46  
47      private Win32ComputerSystemProduct() {
48      }
49  
50      /**
51       * Queries the Computer System Product.
52       *
53       * @return Assigned serial number and UUID.
54       */
55      public static WmiResult<ComputerSystemProductProperty> queryIdentifyingNumberUUID() {
56          WmiQuery<ComputerSystemProductProperty> identifyingNumberQuery = new WmiQuery<>(WIN32_COMPUTER_SYSTEM_PRODUCT,
57                  ComputerSystemProductProperty.class);
58          return WmiQueryHandler.createInstance().queryWMI(identifyingNumberQuery);
59      }
60  }
