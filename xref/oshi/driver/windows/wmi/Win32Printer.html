5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery;
10  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.util.platform.windows.WmiQueryHandler;
14  
15  /**
16   * Utility to query WMI class {@code Win32_Printer}
17   */
18  @ThreadSafe
19  public final class Win32Printer {
20  
21      private static final String WIN32_PRINTER = "Win32_Printer";
22  
23      /**
24       * Printer properties from WMI
25       */
26      public enum PrinterProperty {
27          NAME, DRIVERNAME, PRINTERSTATUS, DETECTEDERRORSTATE, DEFAULT, LOCAL, PORTNAME, DESCRIPTION;
28      }
29  
30      private Win32Printer() {
31      }
32  
33      /**
34       * Queries printer information.
35       *
36       * @return Information regarding printers
37       */
38      public static WmiResult<PrinterProperty> queryPrinters() {
39          WmiQuery<PrinterProperty> printerQuery = new WmiQuery<>(WIN32_PRINTER, PrinterProperty.class);
40          return Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(printerQuery);
41      }
42  }
