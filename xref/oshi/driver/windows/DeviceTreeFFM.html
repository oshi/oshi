6   
7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   
9   import java.lang.foreign.Arena;
10  import java.lang.foreign.MemorySegment;
11  import java.util.ArrayDeque;
12  import java.util.HashMap;
13  import java.util.Map;
14  import java.util.Optional;
15  import java.util.Queue;
16  import java.util.Set;
17  import java.util.stream.Collectors;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import oshi.annotation.concurrent.ThreadSafe;
23  import oshi.ffm.platform.windows.Cfgmgr32FFM;
24  import oshi.ffm.platform.windows.SetupApiFFM;
25  import oshi.util.tuples.Quintet;
26  
27  /**
28   * FFM-based utility to query device interfaces via Configuration Manager Device Tree functions.
29   */
30  @ThreadSafe
31  public final class DeviceTreeFFM {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(DeviceTreeFFM.class);
34  
35      private static final int MAX_PATH = 260;
36  
37      private DeviceTreeFFM() {
38      }
39  
40      /**
41       * Queries devices matching the specified device interface GUID and returns maps representing device tree
42       * relationships, name, device ID, and manufacturer.
43       *
44       * @param guidBytes 16-byte GUID of the device interface class
45       * @return A {@link Quintet} of: top-level device set, parent map, name map, device ID map, manufacturer map
46       */
47      public static Quintet<Set<Integer>, Map<Integer, Integer>, Map<Integer, String>, Map<Integer, String>, Map<Integer, String>> queryDeviceTree(
48              byte[] guidBytes) {
49          if (guidBytes == null || guidBytes.length != 16) {
50              throw new IllegalArgumentException("guidBytes must be exactly 16 bytes");
51          }
52          Map<Integer, Integer> parentMap = new HashMap<>();
53          Map<Integer, String> nameMap = new HashMap<>();
54          Map<Integer, String> deviceIdMap = new HashMap<>();
55          Map<Integer, String> mfgMap = new HashMap<>();
56  
57          try (Arena arena = Arena.ofConfined()) {
58              MemorySegment guidSeg = arena.allocate(16);
59              guidSeg.copyFrom(MemorySegment.ofArray(guidBytes));
60  
61              Optional<MemorySegment> hDevInfoOpt = SetupApiFFM.SetupDiGetClassDevs(guidSeg,
62                      SetupApiFFM.DIGCF_PRESENT | SetupApiFFM.DIGCF_DEVICEINTERFACE);
63              if (hDevInfoOpt.isPresent()) {
64                  MemorySegment hDevInfo = hDevInfoOpt.get();
65                  try {
66                      MemorySegment devInfoData = arena.allocate(SetupApiFFM.SP_DEVINFO_DATA_SIZE);
67                      MemorySegment childSeg = arena.allocate(JAVA_INT);
68                      MemorySegment siblingSeg = arena.allocate(JAVA_INT);
69                      MemorySegment buf = arena.allocate(MAX_PATH * 2L);
70                      MemorySegment sizeSeg = arena.allocate(JAVA_INT);
71  
72                      Queue<Integer> deviceTree = new ArrayDeque<>();
73                      for (int i = 0;; i++) {
74                          devInfoData.fill((byte) 0);
75                          devInfoData.set(JAVA_INT, 0, SetupApiFFM.SP_DEVINFO_DATA_SIZE);
76                          if (!SetupApiFFM.SetupDiEnumDeviceInfo(hDevInfo, i, devInfoData)) {
77                              break;
78                          }
79                          int devInst = devInfoData.get(JAVA_INT, SetupApiFFM.SP_DEVINFO_DATA_DEVINST_OFFSET);
80                          deviceTree.add(devInst);
81  
82                          while (!deviceTree.isEmpty()) {
83                              int node = deviceTree.poll();
84  
85                              deviceIdMap.put(node, Cfgmgr32FFM.getDeviceId(node, arena));
86  
87                              String name = Cfgmgr32FFM.getDevNodeProperty(node, Cfgmgr32FFM.CM_DRP_FRIENDLYNAME, buf,
88                                      sizeSeg);
89                              if (name.isEmpty()) {
90                                  name = Cfgmgr32FFM.getDevNodeProperty(node, Cfgmgr32FFM.CM_DRP_DEVICEDESC, buf,
91                                          sizeSeg);
92                              }
93                              if (name.isEmpty()) {
94                                  name = Cfgmgr32FFM.getDevNodeProperty(node, Cfgmgr32FFM.CM_DRP_CLASS, buf, sizeSeg);
95                                  String svc = Cfgmgr32FFM.getDevNodeProperty(node, Cfgmgr32FFM.CM_DRP_SERVICE, buf,
96                                          sizeSeg);
97                                  if (!svc.isEmpty()) {
98                                      name = name + " (" + svc + ")";
99                                  }
100                             }
101                             nameMap.put(node, name);
102                             mfgMap.put(node,
103                                     Cfgmgr32FFM.getDevNodeProperty(node, Cfgmgr32FFM.CM_DRP_MFG, buf, sizeSeg));
104 
105                             // Traverse children
106                             try {
107                                 if (Cfgmgr32FFM.CM_Get_Child(childSeg, node, 0) == Cfgmgr32FFM.CR_SUCCESS) {
108                                     int child = childSeg.get(JAVA_INT, 0);
109                                     parentMap.put(child, node);
110                                     deviceTree.add(child);
111                                     while (Cfgmgr32FFM.CM_Get_Sibling(siblingSeg, child, 0) == Cfgmgr32FFM.CR_SUCCESS) {
112                                         int sibling = siblingSeg.get(JAVA_INT, 0);
113                                         parentMap.put(sibling, node);
114                                         deviceTree.add(sibling);
115                                         child = sibling;
116                                     }
117                                 }
118                             } catch (Throwable t) {
119                                 LOG.debug("CM_Get_Child/Sibling threw for node {}", node, t);
120                             }
121                         }
122                     }
123                 } finally {
124                     SetupApiFFM.SetupDiDestroyDeviceInfoList(hDevInfo);
125                 }
126             }
127         }
128 
129         Set<Integer> controllerDevices = deviceIdMap.keySet().stream().filter(k -> !parentMap.containsKey(k))
130                 .collect(Collectors.toSet());
131         return new Quintet<>(controllerDevices, parentMap, nameMap, deviceIdMap, mfgMap);
132     }
133 }
