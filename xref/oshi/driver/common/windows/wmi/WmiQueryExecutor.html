5   package oshi.driver.common.windows.wmi;
6   
7   /**
8    * Common interface for executing WMI queries, abstracting JNA and FFM implementations.
9    * <p>
10   * JNA users who wish to customize COM initialization (e.g., to avoid per-query COM init/uninit overhead) should extend
11   * {@code oshi.util.platform.windows.WmiQueryHandler} in the {@code oshi-core} module. See the {@code oshi-demo} module
12   * for examples.
13   */
14  public interface WmiQueryExecutor {
15  
16      /**
17       * Query WMI for values. Initializes and uninitializes COM for each query.
18       *
19       * @param <T>   the enum type for the query properties
20       * @param query the WMI query
21       * @return the query results
22       */
23      <T extends Enum<T>> WmiResult<T> queryWMI(WmiQuery<T> query);
24  
25      /**
26       * Query WMI for values.
27       *
28       * @param <T>     the enum type for the query properties
29       * @param query   the WMI query
30       * @param initCom whether to initialize COM for this query. If {@code false}, assumes COM is already initialized.
31       * @return the query results
32       */
33      <T extends Enum<T>> WmiResult<T> queryWMI(WmiQuery<T> query, boolean initCom);
34  
35      /**
36       * Creates a new instance of the platform-appropriate WMI query executor.
37       * <p>
38       * On JNA, this returns an instance of {@code WmiQueryHandler} (or a user-configured subclass). On FFM, this returns
39       * an instance of {@code WmiQueryHandlerFFM}.
40       *
41       * @return a new executor instance, or {@code null} if creation fails
42       */
43      static WmiQueryExecutor createInstance() {
44          throw new UnsupportedOperationException("Use platform-specific createInstance()");
45      }
46  }
