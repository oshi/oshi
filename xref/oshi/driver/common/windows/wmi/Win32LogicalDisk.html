5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants and property enum for WMI class {@code Win32_LogicalDisk}.
11   */
12  @ThreadSafe
13  public class Win32LogicalDisk {
14  
15      /**
16       * The WMI class name.
17       */
18      public static final String WIN32_LOGICAL_DISK = "Win32_LogicalDisk";
19  
20      /**
21       * Logical disk properties.
22       */
23      public enum LogicalDiskProperty {
24          /** ACCESS property. */
25          ACCESS,
26          /** DESCRIPTION property. */
27          DESCRIPTION,
28          /** DRIVETYPE property. */
29          DRIVETYPE,
30          /** FILESYSTEM property. */
31          FILESYSTEM,
32          /** FREESPACE property. */
33          FREESPACE,
34          /** NAME property. */
35          NAME,
36          /** PROVIDERNAME property. */
37          PROVIDERNAME,
38          /** SIZE property. */
39          SIZE,
40          /** VOLUMENAME property. */
41          VOLUMENAME;
42      }
43  
44      /**
45       * Constructor.
46       */
47      protected Win32LogicalDisk() {
48      }
49  
50      /**
51       * Builds the WMI class name with optional WHERE clause for logical disk queries.
52       *
53       * @param nameToMatch an optional string to filter match, null otherwise
54       * @param localOnly   whether to only search local drives
55       * @return the WMI class name with WHERE clause appended
56       */
57      public static String buildWmiClassNameWithWhere(String nameToMatch, boolean localOnly) {
58          StringBuilder wmiClassName = new StringBuilder(WIN32_LOGICAL_DISK);
59          boolean where = false;
60          if (localOnly) {
61              wmiClassName.append(" WHERE (DriveType = 2 OR DriveType = 3 OR DriveType = 6)");
62              where = true;
63          }
64          if (nameToMatch != null) {
65              wmiClassName.append(where ? " AND" : " WHERE").append(" Name=\"").append(nameToMatch.replace("\"", "\\\""))
66                      .append('"');
67          }
68          return wmiClassName.toString();
69      }
70  
71      /**
72       * Queries logical disk information.
73       *
74       * @param h           An instantiated {@link WmiQueryExecutor}.
75       * @param nameToMatch an optional string to filter match, null otherwise
76       * @param localOnly   Whether to only search local drives
77       * @return Logical Disk Information
78       */
79      public static WmiResult<LogicalDiskProperty> queryLogicalDisk(WmiQueryExecutor h, String nameToMatch,
80              boolean localOnly) {
81          WmiQuery<LogicalDiskProperty> logicalDiskQuery = new WmiQuery<>(
82                  buildWmiClassNameWithWhere(nameToMatch, localOnly), LogicalDiskProperty.class);
83          return h.queryWMI(logicalDiskQuery);
84      }
85  }
