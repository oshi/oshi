5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants, property enums, and WHERE clause builders for LibreHardwareMonitor WMI sensor data.
11   *
12   * <p>
13   * LHM publishes sensor data to {@code ROOT\LibreHardwareMonitor} when it is running. This class provides the shared
14   * constants and query builders for the {@code Sensor} and {@code Hardware} tables.
15   */
16  @ThreadSafe
17  public class LhmSensor {
18  
19      /**
20       * The WMI namespace for LibreHardwareMonitor.
21       */
22      public static final String LHM_NAMESPACE = "ROOT\\LibreHardwareMonitor";
23  
24      /**
25       * The WMI class name for sensors.
26       */
27      public static final String SENSOR = "Sensor";
28  
29      /**
30       * The WMI class name for hardware.
31       */
32      public static final String HARDWARE = "Hardware";
33  
34      /**
35       * Sensor properties returned by LHM WMI queries.
36       */
37      public enum LhmSensorProperty {
38          /** Sensor name. */
39          NAME,
40          /** Sensor value. */
41          VALUE,
42          /** Parent hardware identifier. */
43          PARENT;
44      }
45  
46      /**
47       * LHM Hardware properties.
48       */
49      public enum LhmHardwareProperty {
50          /** Hardware identifier. */
51          IDENTIFIER,
52          /** Hardware name. */
53          NAME;
54      }
55  
56      /**
57       * Default constructor.
58       */
59      protected LhmSensor() {
60      }
61  
62      /**
63       * Builds the WMI class name with WHERE clause for sensor queries.
64       *
65       * @param parent     the LHM hardware identifier (e.g. {@code /gpu-nvidia/0})
66       * @param sensorType the sensor type string (e.g. {@code "Load"}, {@code "SmallData"})
67       * @return the WMI class name with WHERE clause
68       */
69      public static String buildSensorWmiClassNameWithWhere(String parent, String sensorType) {
70          StringBuilder sb = new StringBuilder(SENSOR);
71          sb.append(" WHERE Parent=\"").append(parent);
72          sb.append("\" AND SensorType=\"").append(sensorType).append('"');
73          return sb.toString();
74      }
75  
76      /**
77       * Builds the WMI class name with WHERE clause for GPU hardware queries.
78       *
79       * @return the WMI class name with WHERE clause filtering GPU hardware types
80       */
81      public static String buildGpuHardwareWmiClassName() {
82          return HARDWARE + " WHERE HardwareType=\"GpuNvidia\" OR HardwareType=\"GpuAmd\" OR HardwareType=\"GpuIntel\"";
83      }
84  
85      /**
86       * Queries all sensors of a given type belonging to a specific hardware parent.
87       *
88       * @param h          An instantiated {@link WmiQueryExecutor}.
89       * @param parent     the LHM hardware identifier (e.g. {@code /gpu-nvidia/0})
90       * @param sensorType the sensor type string (e.g. {@code "Load"}, {@code "SmallData"})
91       * @return WMI result containing NAME, VALUE, and PARENT columns
92       */
93      public static WmiResult<LhmSensorProperty> querySensors(WmiQueryExecutor h, String parent, String sensorType) {
94          WmiQuery<LhmSensorProperty> query = new WmiQuery<>(LHM_NAMESPACE,
95                  buildSensorWmiClassNameWithWhere(parent, sensorType), LhmSensorProperty.class);
96          return h.queryWMI(query);
97      }
98  
99      /**
100      * Queries all GPU hardware entries from LHM to discover parent identifiers.
101      *
102      * @param h An instantiated {@link WmiQueryExecutor}.
103      * @return WMI result with IDENTIFIER and NAME columns for all GPU hardware entries
104      */
105     public static WmiResult<LhmHardwareProperty> queryGpuHardware(WmiQueryExecutor h) {
106         WmiQuery<LhmHardwareProperty> query = new WmiQuery<>(LHM_NAMESPACE, buildGpuHardwareWmiClassName(),
107                 LhmHardwareProperty.class);
108         return h.queryWMI(query);
109     }
