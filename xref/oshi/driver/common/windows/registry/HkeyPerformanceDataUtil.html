12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  import oshi.annotation.concurrent.ThreadSafe;
16  import oshi.driver.common.windows.perfmon.PdhCounterWildcardProperty;
17  import oshi.util.ParseUtil;
18  import oshi.util.tuples.Pair;

20  /**
21   * Base class for HKEY_PERFORMANCE_DATA utilities. Contains the common logic for building the counter name/index map and
22   * resolving counter indices from enum definitions. Subclasses provide the platform-specific native call to read the
23   * registry string array.
24   */
25  @ThreadSafe
26  public abstract class HkeyPerformanceDataUtil {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(HkeyPerformanceDataUtil.class);
29  
30      /**
31       * Subclass-only constructor.
32       */
33      protected HkeyPerformanceDataUtil() {
34      }
35  
36      /**
37       * Registry key containing English counter name/index pairs.
38       */
39      public static final String HKEY_PERFORMANCE_TEXT = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Perflib\\009";
40  
41      /**
42       * Registry value name for the counter text.
43       */
44      public static final String COUNTER = "Counter";
45  
46      /**
47       * Builds the counter index map from a string array of alternating index/name pairs.
48       * <p>
49       * The array format is pairs of {@code "<index>", "<name>"}. The first pair is a length/count entry (e.g.,
50       * {@code "1", "1847"}) followed by actual counter definitions such as {@code "2", "System", "4", "Memory", "6",
51       * "% Processor Time", ...}. Even positions (0, 2, 4, ...) hold numeric index strings and odd positions (1, 3, 5,
52       * ...) hold the corresponding counter names.
53       *
54       * @param counterText The string array from the registry, or null if the read failed
55       * @return An unmodifiable map of counter name to index
56       */
57      protected static Map<String, Integer> buildCounterIndexMap(String[] counterText) {
58          HashMap<String, Integer> indexMap = new HashMap<>();
59          if (counterText != null && counterText.length > 1) {
60              for (int i = 1; i < counterText.length; i += 2) {
61                  int idx = ParseUtil.parseIntOrDefault(counterText[i - 1], 0);
62                  if (idx > 0) {
63                      indexMap.putIfAbsent(counterText[i], idx);
64                  }
65              }
66          }
67          return Collections.unmodifiableMap(indexMap);
68      }
69  
70      /**
71       * Looks up the counter index values for the given counter object and the enum of counter names.
72       *
73       * @param <T>             An enum containing the counters, whose class is passed as {@code counterEnum}
74       * @param objectName      The counter object to look up the index for
75       * @param counterEnum     The {@link Enum} containing counters to look up the indices for. The first Enum value will
76       *                        be ignored.
77       * @param counterIndexMap The map of counter names to their indices
78       * @return A {@link Pair} containing the index of the counter object as the first element, and an {@link EnumMap}
79       *         mapping counter enum values to their index as the second element, if the lookup is successful; null
80       *         otherwise.
81       */
82      protected static <T extends Enum<T> & PdhCounterWildcardProperty> Pair<Integer, EnumMap<T, Integer>> getCounterIndices(
83              String objectName, Class<T> counterEnum, Map<String, Integer> counterIndexMap) {
84          Integer counterIndex = counterIndexMap.get(objectName);
85          if (counterIndex == null) {
86              LOG.debug("Couldn't find counter index of {}.", objectName);
87              return null;
88          }
89          T[] enumConstants = counterEnum.getEnumConstants();
90          EnumMap<T, Integer> indexMap = new EnumMap<>(counterEnum);
91          // Start iterating at 1 because first Enum value defines the name/instance and
92          // is not a counter name
93          for (int i = 1; i < enumConstants.length; i++) {
94              T key = enumConstants[i];
95              Integer idx = counterIndexMap.get(key.getCounter());
96              if (idx == null) {
97                  LOG.debug("Couldn't find counter index of {}.", key.getCounter());
98                  return null;
99              }
100             indexMap.put(key, idx);
101         }
102         // We have all the pieces! Return them.
103         return new Pair<>(counterIndex, indexMap);
104     }
105 }
