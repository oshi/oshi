195     /**
196      * Returns processor performance counters.
197      *
198      * @param executor the performance counter query executor
199      * @return Performance Counters for processors, or empty if OS counters are disabled.
200      */
201     public static Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> queryProcessorCounters(
202             PerfCounterQueryExecutor executor) {
203         if (executor.isPerfOsDisabled()) {
204             return new Pair<>(Collections.emptyList(), Collections.emptyMap());
205         }
206         return executor.isWin7OrGreater()
207                 ? executor.queryInstancesAndValues(ProcessorTickCountProperty.class, PROCESSOR_INFORMATION,
208                         WIN32_PERF_RAW_DATA_COUNTERS_PROCESSOR_INFORMATION_WHERE_NOT_NAME_LIKE_TOTAL)
209                 : executor.queryInstancesAndValues(ProcessorTickCountProperty.class, PROCESSOR,
210                         WIN32_PERF_RAW_DATA_PERF_OS_PROCESSOR_WHERE_NAME_NOT_TOTAL);
211     }
212 
213     /**
214      * Returns system performance counters.
215      *
216      * @param executor the performance counter query executor
217      * @return Performance Counters for the total of all processors, or empty if OS counters are disabled.
218      */
219     public static Map<SystemTickCountProperty, Long> querySystemCounters(PerfCounterQueryExecutor executor) {
220         if (executor.isPerfOsDisabled()) {
221             return Collections.emptyMap();
222         }
223         return executor.queryValues(SystemTickCountProperty.class, PROCESSOR,
224                 WIN32_PERF_RAW_DATA_PERF_OS_PROCESSOR_WHERE_NAME_TOTAL);
225     }
