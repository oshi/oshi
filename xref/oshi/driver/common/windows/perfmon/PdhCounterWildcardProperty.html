7   /**

9    * enum element specifies an instance filter and remaining elements define counter names.
10   */
11  public interface PdhCounterWildcardProperty {
12      /**
13       * @return Returns the counter. The first element of the enum will return the instance filter rather than a counter.
14       */
15      String getCounter();
16  }
