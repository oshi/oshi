5   package oshi.driver.common.windows.perfmon;
6   
7   import java.util.Arrays;
8   import java.util.List;
9   import java.util.Map;
10  
11  import oshi.annotation.concurrent.ThreadSafe;
12  import oshi.driver.common.windows.perfmon.ProcessInformation.IdleProcessorTimeProperty;
13  import oshi.driver.common.windows.perfmon.SystemInformation.ProcessorQueueLengthProperty;
14  import oshi.util.tuples.Pair;
15  
16  /**
17   * Utility to calculate a load average equivalent metric on Windows. Starts a daemon thread to collect the necessary
18   * counters and averages in 5-second intervals. Subclasses provide the platform-specific perfmon queries.
19   */
20  @ThreadSafe
21  public abstract class LoadAverage {
22  
23      /**
24       * Default constructor.
25       */
26      protected LoadAverage() {
27      }
28  
29      // Daemon thread for Load Average
30      private Thread loadAvgThread = null;
31  
32      private final double[] loadAverages = new double[] { -1d, -1d, -1d };
33      private static final double[] EXP_WEIGHT = new double[] {
34              // 1-, 5-, and 15-minute exponential smoothing weight
35              Math.exp(-5d / 60d), Math.exp(-5d / 300d), Math.exp(-5d / 900d) };
36  
37      /**
38       * Queries the raw idle-process performance counters. Implemented per-backend (JNA/FFM); the aggregation into
39       * non-idle ticks is shared in {@link #queryNonIdleTicks()}.
40       *
41       * @return a pair of (instance names, counter values keyed by property)
42       */
43      protected abstract Pair<List<String>, Map<IdleProcessorTimeProperty, List<Long>>> queryIdleProcessCounters();
44  
45      /**
46       * Queries the raw processor-queue-length performance counter. Implemented per-backend (JNA/FFM).
47       *
48       * @return the counter values keyed by property
49       */
50      protected abstract Map<ProcessorQueueLengthProperty, Long> queryProcessorQueueLength();
51  
52      /**
53       * Computes the non-idle ticks from the idle-process performance counters by summing the {@code _Total} instance and
54       * subtracting the {@code Idle} instance.
55       *
56       * @return A pair of (nonIdleTicks, nonIdleBase) values
57       */
58      protected Pair<Long, Long> queryNonIdleTicks() {
59          Pair<List<String>, Map<IdleProcessorTimeProperty, List<Long>>> idleValues = queryIdleProcessCounters();
60          List<String> instances = idleValues.getA();
61          Map<IdleProcessorTimeProperty, List<Long>> valueMap = idleValues.getB();
62          List<Long> proctimeTicks = valueMap.get(IdleProcessorTimeProperty.PERCENTPROCESSORTIME);
63          List<Long> proctimeBase = valueMap.get(IdleProcessorTimeProperty.ELAPSEDTIME);
64          if (proctimeTicks == null || proctimeBase == null) {
65              return new Pair<>(0L, 0L);
66          }
67          long nonIdleTicks = 0L;
68          long nonIdleBase = 0L;
69          for (int i = 0; i < instances.size(); i++) {
70              if (i >= proctimeTicks.size() || i >= proctimeBase.size()) {
71                  break;
72              }
73              if ("_Total".equals(instances.get(i))) {
74                  nonIdleTicks += proctimeTicks.get(i);
75                  nonIdleBase += proctimeBase.get(i);
76              } else if ("Idle".equals(instances.get(i))) {
77                  nonIdleTicks -= proctimeTicks.get(i);
78              }
79          }
80          return new Pair<>(nonIdleTicks, nonIdleBase);
81      }
82  
83      /**
84       * Returns the processor queue length from the raw performance counter.
85       *
86       * @return The processor queue length
87       */
88      protected long queryQueueLength() {
89          return queryProcessorQueueLength().getOrDefault(ProcessorQueueLengthProperty.PROCESSORQUEUELENGTH, 0L);
90      }
91  
92      /**
93       * Queries the load average values.
94       *
95       * @param nelem number of elements to return (1, 2, or 3)
96       * @return array of load averages
97       */
98      public double[] queryLoadAverage(int nelem) {
99          synchronized (loadAverages) {
100             return Arrays.copyOf(loadAverages, nelem);
101         }
102     }
103 
104     /**
105      * Stops the load average daemon thread.
106      */
107     public synchronized void stopDaemon() {
108         if (loadAvgThread != null) {
109             loadAvgThread.interrupt();
110             loadAvgThread = null;
111         }
112     }
113 
114     /**
115      * Starts the load average daemon thread.
116      */
117     public synchronized void startDaemon() {
118         if (loadAvgThread != null) {
119             return;
120         }
121         loadAvgThread = new Thread("OSHI Load Average daemon") {
122             @Override
123             public void run() {
124                 // Initialize tick counters
125                 Pair<Long, Long> nonIdlePair = queryNonIdleTicks();
126                 long nonIdleTicks0 = nonIdlePair.getA();
127                 long nonIdleBase0 = nonIdlePair.getB();
128                 long nonIdleTicks;
129                 long nonIdleBase;
130 
131                 // Use nanoTime to synchronize queries at 5 seconds
132                 long initNanos = System.nanoTime();
133                 long delay;
134 
135                 // The two components of load average
136                 double runningProcesses;
137                 long queueLength;
138 
139                 try {
140                     Thread.sleep(2500L);
141                 } catch (InterruptedException e) {
142                     Thread.currentThread().interrupt();
143                 }
144                 while (!Thread.currentThread().isInterrupted()) {
145                     // get non-idle ticks, proxy for average processes running
146                     nonIdlePair = queryNonIdleTicks();
147                     nonIdleTicks = nonIdlePair.getA() - nonIdleTicks0;
148                     nonIdleBase = nonIdlePair.getB() - nonIdleBase0;
149                     if (nonIdleBase > 0 && nonIdleTicks > 0) {
150                         runningProcesses = (double) nonIdleTicks / nonIdleBase;
151                     } else {
152                         runningProcesses = 0d;
153                     }
154                     nonIdleTicks0 = nonIdlePair.getA();
155                     nonIdleBase0 = nonIdlePair.getB();
156                     // get processes waiting
157                     queueLength = queryQueueLength();
158 
159                     synchronized (loadAverages) {
160                         // Init to running procs the first time
161                         if (loadAverages[0] < 0d) {
162                             Arrays.fill(loadAverages, runningProcesses);
163                         }
164                         // Use exponential smoothing to update values
165                         for (int i = 0; i < loadAverages.length; i++) {
166                             loadAverages[i] *= EXP_WEIGHT[i];
167                             loadAverages[i] += (runningProcesses + queueLength) * (1d - EXP_WEIGHT[i]);
168                         }
169                     }
170 
171                     delay = 5000L - (System.nanoTime() - initNanos) % 5_000_000_000L / 1_000_000;
172                     if (delay < 500L) {
173                         delay += 5000L;
174                     }
175                     try {
176                         Thread.sleep(delay);
177                     } catch (InterruptedException e) {
178                         Thread.currentThread().interrupt();
179                     }
180                 }
181             }
182         };
183         loadAvgThread.setDaemon(true);
184         loadAvgThread.start();
185     }
186 }
