5   package oshi.driver.common.unix.aix;
6   
7   import java.nio.ByteBuffer;
8   
9   import oshi.util.FileUtil;
10  
11  /**
12   * Pure-Java mirror of the AIX {@code lwpsinfo_t} struct populated from {@code /proc/<pid>/lwp/<tid>/lwpsinfo}.
13   * <p>
14   * Fields and ordering match {@code <sys/procfs.h>}.
15   */
16  public class AixLwpsInfo {
17  
18      /** Length of {@code pr_clname} (printable character representing pr_policy). */
19      public static final int PRCLSZ = 8;
20  
21      public long pr_lwpid; // thread id
22      public long pr_addr; // internal address of thread
23      public long pr_wchan; // wait addr for sleeping thread
24      public int pr_flag; // thread flags
25      public byte pr_wtype; // type of thread wait
26      public byte pr_state; // numeric scheduling state
27      public byte pr_sname; // printable character representing pr_state
28      public byte pr_nice; // nice for cpu usage
29      public int pr_pri; // priority, high value = high priority
30      public int pr_policy; // scheduling policy
31      public byte[] pr_clname = new byte[PRCLSZ]; // printable character representing pr_policy
32      public int pr_onpro; // processor on which thread last ran
33      public int pr_bindpro; // processor to which thread is bound
34  
35      public AixLwpsInfo(ByteBuffer buff) {
36          this.pr_lwpid = FileUtil.readLongFromBuffer(buff);
37          this.pr_addr = FileUtil.readLongFromBuffer(buff);
38          this.pr_wchan = FileUtil.readLongFromBuffer(buff);
39          this.pr_flag = FileUtil.readIntFromBuffer(buff);
40          this.pr_wtype = FileUtil.readByteFromBuffer(buff);
41          this.pr_state = FileUtil.readByteFromBuffer(buff);
42          this.pr_sname = FileUtil.readByteFromBuffer(buff);
43          this.pr_nice = FileUtil.readByteFromBuffer(buff);
44          this.pr_pri = FileUtil.readIntFromBuffer(buff);
45          this.pr_policy = FileUtil.readIntFromBuffer(buff);
46          FileUtil.readByteArrayFromBuffer(buff, this.pr_clname);
47          this.pr_onpro = FileUtil.readIntFromBuffer(buff);
48          this.pr_bindpro = FileUtil.readIntFromBuffer(buff);
49      }
50  }
