19  /**
20   * Utility class parsing partition information from the BSD {@code disklabel} command. Shared by OpenBSD and NetBSD,
21   * which produce the same disklabel output format.
22   */
23  @ThreadSafe
24  public final class Disklabel {
25  
26      private static final String TOTAL_MARKER = "total sectors:";
27      private static final String BPS_MARKER = "bytes/sector:";
28      private static final String LABEL_MARKER = "label:";
29      private static final String DUID_MARKER = "duid:";
30  
31      private Disklabel() {
32      }
33  
34      /**
35       * Gets disk and partition information by running {@code disklabel} and parsing its output. Falls back to {@code df}
36       * if no partitions are returned (e.g., when not running as root).
37       *
38       * @param diskName The disk to fetch partition information from
39       * @return A quartet containing the disk's name/label, DUID, size, and a list of partitions
40       */
41      public static Quartet<String, String, Long, List<HWPartition>> getDiskParams(String diskName) {
42          Quartet<String, String, Long, List<HWPartition>> result = parseDiskParams(diskName,
43                  ExecutingCommand.runNative("disklabel -n " + diskName), Disklabel::getMajorMinor);
44          if (!result.getD().isEmpty()) {
45              return result;
46          }
47          // No partitions came back from disklabel. Fall back to df for the partition list, but keep whatever
48          // header metadata (label / DUID / size) disklabel already produced — replacing only the empty partition
49          // list rather than the entire Quartet.
50          Quartet<String, String, Long, List<HWPartition>> fallback = parseDfFallback(diskName,
51                  ExecutingCommand.runNative("df"), Disklabel::getMajorMinor);
52          return new Quartet<>(result.getA(), result.getB(), result.getC(), fallback.getD());
53      }
54  
55      /**
56       * Parses {@code disklabel -n} output to extract disk metadata and partitions.
57       *
58       * @param diskName         the disk being labeled
59       * @param disklabelLines   raw output from {@code disklabel -n <diskName>}
60       * @param majorMinorLookup function that resolves a (disk, partition-letter) pair to a (major, minor) device number
61       *                         pair; tests may pass a stub that returns (0,0) without invoking {@code stat}.
62       * @return A quartet of label, DUID, size in bytes, and partitions. An empty partition list signals the caller
63       *         should fall back to {@link #parseDfFallback}.
64       */
65      public static Quartet<String, String, Long, List<HWPartition>> parseDiskParams(String diskName,
66              List<String> disklabelLines, BiFunction<String, String, Pair<Integer, Integer>> majorMinorLookup) {
67          // disklabel (requires root) supports 15 configurable partitions, `a' through `p', excluding `c'.
68          // The `c' partition describes the entire physical disk. By convention `a' of the boot disk is root,
69          // `b' is swap, and `i' is usually the boot record.
70          List<HWPartition> partitions = new ArrayList<>();
71          long totalSectors = 1L;
72          int bytesPerSector = 1;
73          String label = "";
74          String duid = "";
75          for (String line : disklabelLines) {
76              if (line.contains(TOTAL_MARKER)) {
77                  // Parse as long to avoid int overflow for disks larger than ~2 TiB at 512-byte sectors.
78                  totalSectors = ParseUtil.parseLongOrDefault(line.split(TOTAL_MARKER)[1].trim(), 1L);
79              } else if (line.contains(BPS_MARKER)) {
80                  bytesPerSector = ParseUtil.getFirstIntValue(line);
81              } else if (line.contains(LABEL_MARKER)) {
82                  label = line.split(LABEL_MARKER)[1].trim();
83              } else if (line.contains(DUID_MARKER)) {
84                  duid = line.split(DUID_MARKER)[1].trim();
85              }
86              if (line.trim().indexOf(':') == 1) {
87                  // Partition rows have a single letter followed by a colon:
88                  // a: 2097152 1024 4.2BSD 2048 16384 12958 # /
89                  String[] split = ParseUtil.whitespaces.split(line.trim(), 9);
90                  if (split.length > 4) {
91                      String name = split[0].substring(0, 1);
92                      Pair<Integer, Integer> majorMinor = majorMinorLookup.apply(diskName, name);
93                      partitions.add(new HWPartition(diskName + name, name, split[3], duid + "." + name,
94                              ParseUtil.parseLongOrDefault(split[1], 0L) * bytesPerSector, majorMinor.getA(),
95                              majorMinor.getB(), split.length > 5 ? split[split.length - 1] : ""));
96                  }
97              }
98          }
99          return new Quartet<>(label, duid, totalSectors * bytesPerSector, partitions);
100     }
101 
102     /**
103      * Parses {@code df} output as a fallback when {@code disklabel} returns no partitions (typically because the caller
104      * lacks root privileges).
105      *
106      * @param diskName         the disk whose partitions are being enumerated
107      * @param dfLines          raw output from {@code df}
108      * @param majorMinorLookup function that resolves a (disk, partition-letter) pair to a (major, minor) device number
109      *                         pair
110      * @return A quartet of UNKNOWN label/DUID, 0 size, and partitions discovered from {@code df}. Callers that already
111      *         have disklabel header metadata should combine it with the partition list from this result.
112      */
113     public static Quartet<String, String, Long, List<HWPartition>> parseDfFallback(String diskName,
114             List<String> dfLines, BiFunction<String, String, Pair<Integer, Integer>> majorMinorLookup) {
115         List<HWPartition> partitions = new ArrayList<>();
116         for (String line : dfLines) {
117             if (line.startsWith("/dev/" + diskName)) {
118                 String[] split = ParseUtil.whitespaces.split(line);
119                 if (split.length > 5) {
120                     String name = split[0].substring(5 + diskName.length());
121                     Pair<Integer, Integer> majorMinor = majorMinorLookup.apply(diskName, name);
122                     long partSize = ParseUtil.parseLongOrDefault(split[1], 1L) * 512L;
123                     partitions.add(new HWPartition(split[0], split[0].substring(5), Constants.UNKNOWN,
124                             Constants.UNKNOWN, partSize, majorMinor.getA(), majorMinor.getB(), split[5]));
125                 }
126             }
127         }
128         return new Quartet<>(Constants.UNKNOWN, Constants.UNKNOWN, 0L, partitions);
129     }
130 
131     private static Pair<Integer, Integer> getMajorMinor(String diskName, String name) {
132         int major = 0;
133         int minor = 0;
134         String majorMinor = ExecutingCommand.getFirstAnswer("stat -f %Hr,%Lr /dev/" + diskName + name);
135         int comma = majorMinor.indexOf(',');
136         if (comma > 0 && comma < majorMinor.length()) {
137             major = ParseUtil.parseIntOrDefault(majorMinor.substring(0, comma), 0);
138             minor = ParseUtil.parseIntOrDefault(majorMinor.substring(comma + 1), 0);
139         }
140         return new Pair<>(major, minor);
141     }
