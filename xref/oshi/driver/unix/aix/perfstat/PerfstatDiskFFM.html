29  /**
30   * FFM-backed driver for {@code perfstat_disk}, mirroring {@code oshi.driver.unix.aix.perfstat.PerfstatDiskJNA}.
31   */
32  @ThreadSafe
33  public final class PerfstatDiskFFM {
34  
35      private static final Logger LOG = LoggerFactory.getLogger(PerfstatDiskFFM.class);
36  
37      private PerfstatDiskFFM() {
38      }
39  
40      /** POJO mirror of the {@code perfstat_disk_t} fields OSHI consumes. */
41      public static final class Disk {
42          public String name = "";
43          public String description = "";
44          public long size;
45          public long bsize;
46          public long xfers;
47          public long wblks;
48          public long rblks;
49          public long qdepth;
50          public long time;
51      }
52  
53      /**
54       * Queries {@code perfstat_disk} for all disks.
55       *
56       * @return one {@link Disk} per disk, or an empty array on error
57       */
58      public static Disk[] queryDiskStats() {
59          return ForeignFunctions.callInArenaOrDefault(arena -> {
60              int count = perfstat_disk(MemorySegment.NULL, MemorySegment.NULL, PERFSTAT_DISK_T_SIZE, 0);
61              if (count <= 0) {
62                  return new Disk[0];
63              }
64              MemorySegment buf = arena.allocate((long) PERFSTAT_DISK_T_SIZE * count);
65              MemorySegment firstName = arena.allocate(PERFSTAT_ID_T_SIZE);
66              int ret = perfstat_disk(firstName, buf, PERFSTAT_DISK_T_SIZE, count);
67              if (ret <= 0) {
68                  return new Disk[0];
69              }
70              Disk[] result = new Disk[ret];
71              for (int i = 0; i < ret; i++) {
72                  long off = (long) i * PERFSTAT_DISK_T_SIZE;
73                  Disk d = new Disk();
74                  d.name = diskName(buf, off);
75                  d.description = diskDescription(buf, off);
76                  d.size = diskSize(buf, off);
77                  d.bsize = diskBsize(buf, off);
78                  d.xfers = diskXfers(buf, off);
79                  d.wblks = diskWblks(buf, off);
80                  d.rblks = diskRblks(buf, off);
81                  d.qdepth = diskQdepth(buf, off);
82                  d.time = diskTime(buf, off);
83                  result[i] = d;
84              }
85              return result;
86          }, LOG, LogLevel.TRACE, "Failed to query disk statistics", new Disk[0]);
87      }
88  }
