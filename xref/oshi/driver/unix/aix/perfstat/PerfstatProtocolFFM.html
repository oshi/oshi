32  /**
33   * FFM-backed driver for {@code perfstat_protocol}, mirroring {@code oshi.driver.unix.aix.perfstat.PerfstatProtocolJNA}.
34   * Only the TCP and UDP variants of the union are read — OSHI doesn't consume the other protocols.
35   */
36  @ThreadSafe
37  public final class PerfstatProtocolFFM {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(PerfstatProtocolFFM.class);
40  
41      private PerfstatProtocolFFM() {
42      }
43  
44      /** POJO mirror of one protocol entry with TCP+UDP variants populated when applicable. */
45      public static final class Protocol {
46          public String name = "";
47          // TCP stats (populated only when name equals "tcp")
48          public long tcpIpackets;
49          public long tcpIerrors;
50          public long tcpOpackets;
51          public long tcpInitiated;
52          public long tcpAccepted;
53          public long tcpEstablished;
54          public long tcpDropped;
55          // UDP stats (populated only when name equals "udp")
56          public long udpIpackets;
57          public long udpIerrors;
58          public long udpOpackets;
59          public long udpNoSocket;
60      }
61  
62      /**
63       * Queries {@code perfstat_protocol} for all protocols.
64       *
65       * @return one {@link Protocol} per entry, or an empty array on error
66       */
67      public static Protocol[] queryProtocols() {
68          return ForeignFunctions.callInArenaOrDefault(arena -> {
69              int count = perfstat_protocol(MemorySegment.NULL, MemorySegment.NULL, PERFSTAT_PROTOCOL_T_SIZE, 0);
70              if (count <= 0) {
71                  return new Protocol[0];
72              }
73              MemorySegment buf = arena.allocate((long) PERFSTAT_PROTOCOL_T_SIZE * count);
74              MemorySegment firstName = arena.allocate(PERFSTAT_ID_T_SIZE);
75              int ret = perfstat_protocol(firstName, buf, PERFSTAT_PROTOCOL_T_SIZE, count);
76              if (ret <= 0) {
77                  return new Protocol[0];
78              }
79              Protocol[] result = new Protocol[ret];
80              for (int i = 0; i < ret; i++) {
81                  long off = (long) i * PERFSTAT_PROTOCOL_T_SIZE;
82                  Protocol p = new Protocol();
83                  p.name = protoName(buf, off);
84                  if ("tcp".equals(p.name)) {
85                      p.tcpIpackets = protoTcpIpackets(buf, off);
86                      p.tcpIerrors = protoTcpIerrors(buf, off);
87                      p.tcpOpackets = protoTcpOpackets(buf, off);
88                      p.tcpInitiated = protoTcpInitiated(buf, off);
89                      p.tcpAccepted = protoTcpAccepted(buf, off);
90                      p.tcpEstablished = protoTcpEstablished(buf, off);
91                      p.tcpDropped = protoTcpDropped(buf, off);
92                  } else if ("udp".equals(p.name)) {
93                      p.udpIpackets = protoUdpIpackets(buf, off);
94                      p.udpIerrors = protoUdpIerrors(buf, off);
95                      p.udpOpackets = protoUdpOpackets(buf, off);
96                      p.udpNoSocket = protoUdpNoSocket(buf, off);
97                  }
98                  result[i] = p;
99              }
100             return result;
101         }, LOG, Level.TRACE, "Failed to query protocol statistics", new Protocol[0]);
102     }
103 }
