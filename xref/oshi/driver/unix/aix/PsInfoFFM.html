32  /**
33   * FFM-backed driver for the {@code queryArgsEnv} address-space read on AIX. The pure-Java parsing of
34   * {@code /proc/<pid>/psinfo} and {@code /proc/<pid>/lwp/<tid>/lwpsinfo} lives in {@link PsInfo} (oshi-common).
35   */
36  @ThreadSafe
37  public final class PsInfoFFM {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(PsInfoFFM.class);
40  
41      private static final long PAGE_SIZE = 4096L;
42  
43      private PsInfoFFM() {
44      }
45  
46      /**
47       * Reads the argument and environment strings from process address space using FFM-bound libc calls.
48       *
49       * @param pid    the process id
50       * @param psinfo a populated {@link AixPsInfo} containing the offset pointers
51       * @return a pair of (argv list, env map)
52       */
53      public static Pair<List<String>, Map<String, String>> queryArgsEnv(int pid, AixPsInfo psinfo) {
54          List<String> args = new ArrayList<>();
55          Map<String, String> env = new LinkedHashMap<>();
56  
57          Triplet<Integer, Long, Long> addrs = PsInfo.queryArgsEnvAddrs(pid, psinfo);
58          if (addrs == null) {
59              return new Pair<>(args, env);
60          }
61  
62          try (Arena arena = Arena.ofConfined()) {
63              MemorySegment procasPath = arena.allocateFrom("/proc/" + pid + "/as");
64              int fd = AixLibcFunctions.open(procasPath, 0);
65              if (fd < 0) {
66                  LOG.trace("No permission to read file: /proc/{}/as", pid);
67                  return new Pair<>(args, env);
68              }
69              try {
70                  int argc = addrs.getA();
71                  long argv = addrs.getB();
72                  long envp = addrs.getC();
73  
74                  // Data model: 1 = 32-bit, anything else (typically 64-bit) gives 8-byte pointers.
75                  long increment;
76                  Path p = Paths.get("/proc/" + pid + "/status");
77                  try {
78                      byte[] status = Files.readAllBytes(p);
79                      increment = status[17] == 1 ? 8 : 4;
80                  } catch (IOException _) {
81                      return new Pair<>(args, env);
82                  }
83  
84                  MemorySegment buffer = arena.allocate(PAGE_SIZE * 2);
85  
86                  // Read the pointers to the arg strings
87                  long bufStart = conditionallyReadPage(fd, buffer, 0, argv);
88                  long[] argPtr = new long[argc];
89                  long argp = bufStart == 0 ? 0 : readPointerFromBuffer(buffer, argv - bufStart, increment);
90                  if (argp > 0) {
91                      for (int i = 0; i < argc; i++) {
92                          long offset = argp + (long) i * increment;
93                          bufStart = conditionallyReadPage(fd, buffer, bufStart, offset);
94                          argPtr[i] = bufStart == 0 ? 0 : readPointerFromBuffer(buffer, offset - bufStart, increment);
95                      }
96                  }
97  
98                  // Also read the pointers to the env strings; stop at first null.
99                  bufStart = conditionallyReadPage(fd, buffer, bufStart, envp);
100                 List<Long> envPtrList = new ArrayList<>();
101                 long addr = bufStart == 0 ? 0 : readPointerFromBuffer(buffer, envp - bufStart, increment);
102                 int limit = 500;
103                 long offset = addr;
104                 while (addr != 0 && --limit > 0) {
105                     bufStart = conditionallyReadPage(fd, buffer, bufStart, offset);
106                     long envPtr = bufStart == 0 ? 0 : readPointerFromBuffer(buffer, offset - bufStart, increment);
107                     if (envPtr != 0) {
108                         envPtrList.add(envPtr);
109                     } else {
110                         break;
111                     }
112                     offset += increment;
113                 }
114 
115                 // Now read the arg strings from the buffer
116                 for (int i = 0; i < argPtr.length && argPtr[i] != 0; i++) {
117                     bufStart = conditionallyReadPage(fd, buffer, bufStart, argPtr[i]);
118                     if (bufStart != 0) {
119                         String argStr = readStringFromBuffer(buffer, argPtr[i] - bufStart);
120                         if (!argStr.isEmpty()) {
121                             args.add(argStr);
122                         }
123                     }
124                 }
125 
126                 // And now read the env strings from the buffer
127                 for (Long envPtr : envPtrList) {
128                     bufStart = conditionallyReadPage(fd, buffer, bufStart, envPtr);
129                     if (bufStart != 0) {
130                         String envStr = readStringFromBuffer(buffer, envPtr - bufStart);
131                         int idx = envStr.indexOf('=');
132                         if (idx > 0) {
133                             env.put(envStr.substring(0, idx), envStr.substring(idx + 1));
134                         }
135                     }
136                 }
137             } finally {
138                 AixLibcFunctions.close(fd);
139             }
140         } catch (Throwable t) {
141             LOG.debug("FFM queryArgsEnv failed for pid {}: {}", pid, t.getMessage());
142         }
143         return new Pair<>(args, env);
144     }

146     /**
147      * Reads the page containing {@code addr} into {@code buffer} unless the buffer already covers it (as indicated by
148      * {@code bufStart}). Returns the new buffer start, or 0 on read failure.
149      */
150     private static long conditionallyReadPage(int fd, MemorySegment buffer, long bufStart, long addr) throws Throwable {
151         if (addr < bufStart || addr - bufStart > PAGE_SIZE) {
152             long newStart = Math.floorDiv(addr, PAGE_SIZE) * PAGE_SIZE;
153             long result = AixLibcFunctions.pread(fd, buffer, buffer.byteSize(), newStart);
154             if (result < PAGE_SIZE) {
155                 LOG.debug("Failed to read page from address space: {} bytes read", result);
156                 return 0;
157             }
158             return newStart;
159         }
160         return bufStart;
161     }
162 
163     private static long readPointerFromBuffer(MemorySegment buffer, long offset, long increment) {
164         return increment == 8 ? buffer.get(JAVA_LONG, offset) : Integer.toUnsignedLong(buffer.get(JAVA_INT, offset));
165     }
166 
167     /**
168      * Reads a NUL-terminated string starting at {@code offset} in the buffer. Falls back to the buffer end if no NUL is
169      * found in the buffered window.
170      */
171     private static String readStringFromBuffer(MemorySegment buffer, long offset) {
172         long max = buffer.byteSize();
173         if (offset < 0 || offset >= max) {
174             return "";
175         }
176         StringBuilder sb = new StringBuilder();
177         for (long i = offset; i < max; i++) {
178             byte b = buffer.get(JAVA_BYTE, i);
179             if (b == 0) {
180                 break;
181             }
182             sb.append((char) (b & 0xFF));
183         }
184         return sb.toString();
185     }
