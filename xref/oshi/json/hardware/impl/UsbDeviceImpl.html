19  package oshi.json.hardware.impl;
20  
21  import java.util.Properties;
22  
23  import javax.json.Json;
24  import javax.json.JsonArrayBuilder;
25  import javax.json.JsonBuilderFactory;
26  import javax.json.JsonObject;
27  import javax.json.JsonObjectBuilder;
28  
29  import oshi.json.hardware.UsbDevice;
30  import oshi.json.json.AbstractOshiJsonObject;
31  import oshi.json.json.NullAwareJsonObjectBuilder;
32  import oshi.json.util.PropertiesUtil;
33  
34  /**
35   * Wrapper class to implement USBDevice interface with platform-specific objects
36   */
37  public class UsbDeviceImpl extends AbstractOshiJsonObject implements UsbDevice {
38  
39      private static final long serialVersionUID = 1L;
40  
41      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
42  
43      private oshi.hardware.UsbDevice usbDevice;
44  
45      /**
46       * Creates a new platform-specific USBDevice object wrapping the provided
47       * argument
48       * 
49       * @param usbDevice
50       *            a platform-specific USBDevice object
51       */
52      public UsbDeviceImpl(oshi.hardware.UsbDevice usbDevice) {
53          this.usbDevice = usbDevice;
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public String getName() {
61          return this.usbDevice.getName();
62      }
63  
64      /**
65       * {@inheritDoc}
66       */
67      @Override
68      public String getVendor() {
69          return this.usbDevice.getVendor();
70      }
71  
72      /**
73       * {@inheritDoc}
74       */
75      @Override
76      public String getVendorId() {
77          return this.usbDevice.getVendorId();
78      }
79  
80      /**
81       * {@inheritDoc}
82       */
83      @Override
84      public String getProductId() {
85          return this.usbDevice.getProductId();
86      }
87  
88      /**
89       * {@inheritDoc}
90       */
91      @Override
92      public String getSerialNumber() {
93          return this.usbDevice.getSerialNumber();
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      @Override
100     public UsbDevice[] getConnectedDevices() {
101         oshi.hardware.UsbDevice[] usbs = this.usbDevice.getConnectedDevices();
102         UsbDevice[] usbDevices = new UsbDevice[usbs.length];
103         for (int i = 0; i < usbs.length; i++) {
104             usbDevices[i] = new UsbDeviceImpl(usbs[i]);
105         }
106         return usbDevices;
107     }
108 
109     /**
110      * {@inheritDoc}
111      */
112     @Override
113     public JsonObject toJSON(Properties properties) {
114         JsonObjectBuilder json = NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder());
115         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.name")) {
116             json.add("name", getName());
117         }
118         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.vendor")) {
119             json.add("vendor", getVendor());
120         }
121         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.vendorId")) {
122             json.add("vendorId", getVendorId());
123         }
124         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.productId")) {
125             json.add("productId", getProductId());
126         }
127         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.serialNumber")) {
128             json.add("serialNumber", getSerialNumber());
129         }
130         if (PropertiesUtil.getBoolean(properties, "hardware.usbDevices.connectedDevices")
131                 && PropertiesUtil.getBoolean(properties, "hardware.usbDevices.tree")) {
132             JsonArrayBuilder usbDeviceArrayBuilder = jsonFactory.createArrayBuilder();
133             for (UsbDevice usbDevice : getConnectedDevices()) {
134                 usbDeviceArrayBuilder.add(usbDevice.toJSON());
135             }
136             json.add("connectedDevices", usbDeviceArrayBuilder.build());
137         }
138         return json.build();
139     }
140 
141     @Override
142     public String toString() {
143         return this.usbDevice.toString();
144     }
145 }
