128     /**
129      * Parse hertz from a string, eg. "2.00MHz" is 2000000L.
130      *
131      * @param hertz
132      *            Hertz size.
133      * @return {@link java.lang.Long} Hertz value or -1 if not parseable.
134      */
135     public static long parseHertz(String hertz) {
136         Matcher matcher = HERTZ_PATTERN.matcher(hertz.trim());
137         if (matcher.find() && matcher.groupCount() == 3) {
138             // Regexp enforces #(.#) format so no test for NFE required
139             double value = Double.valueOf(matcher.group(1)) * multipliers.getOrDefault(matcher.group(3), -1L);
140             if (value >= 0d) {
141                 return (long) value;
142             }
143         }
144         return -1L;
145     }

227     /**
228      * Parse a byte aray into a string of hexadecimal digits including leading zeros
229      *
230      * @param bytes
231      *            The byte array to represent
232      * @return A string of hex characters corresponding to the bytes. The string is
233      *         upper case.
234      */
235     public static String byteArrayToHexString(byte[] bytes) {
236         // Solution copied from https://stackoverflow.com/questions/9655181
237         char[] hexChars = new char[bytes.length * 2];
238         for (int j = 0; j < bytes.length; j++) {
239             int v = bytes[j] & 0xFF;
240             hexChars[j * 2] = HEX_ARRAY[v >>> 4];
241             hexChars[j * 2 + 1] = HEX_ARRAY[v & 0x0F];
242         }
243         return new String(hexChars);
244     }

246     /**
247      * Parse a string of hexadecimal digits into a byte array
248      *
249      * @param digits
250      *            The string to be parsed
251      * @return a byte array with each pair of characters converted to a byte, or
252      *         empty array if the string is not valid hex
253      */
254     public static byte[] hexStringToByteArray(String digits) {
255         int len = digits.length();
256         // Check if string is valid hex
257         if (!VALID_HEX.matcher(digits).matches() || (len & 0x1) != 0) {
258             LOG.warn("Invalid hexadecimal string: {}", digits);
259             return new byte[0];
260         }
261         byte[] data = new byte[len / 2];
262         for (int i = 0; i < len; i += 2) {
263             data[i / 2] = (byte) (Character.digit(digits.charAt(i), 16) << 4
264                     | Character.digit(digits.charAt(i + 1), 16));
265         }
266         return data;
267     }

285     /**
286      * Convert a long value to a byte array using Big Endian, truncating or padding
287      * with zeros (if necessary) so the array has the specified length.
288      *
289      * @param value
290      *            The value to be converted
291      * @param valueSize
292      *            Number of bytes representing the value
293      * @param length
294      *            Number of bytes to return
295      * @return A byte array of specified length representing the long in the first
296      *         valueSize bytes
297      */
298     public static byte[] longToByteArray(long value, int valueSize, int length) {
299         long val = value;
300         // Convert the long to 8-byte BE representation
301         byte[] b = new byte[8];
302         for (int i = 7; i >= 0 && val != 0L; i--) {
303             b[i] = (byte) val;
304             val >>>= 8;
305         }
306         // Then copy the rightmost valueSize bytes
307         // e.g., for an integer we want rightmost 4 bytes
308         return Arrays.copyOfRange(b, 8 - valueSize, 8 + length - valueSize);
309     }

325     /**
326      * Convert a byte array to its integer representation.
327      *
328      * @param bytes
329      *            An array of bytes no smaller than the size to be converted
330      * @param size
331      *            Number of bytes to convert to the long. May not exceed 8.
332      * @return An integer representing the byte array as a 64-bit number
333      */
334     public static long byteArrayToLong(byte[] bytes, int size) {
335         if (size > 8) {
336             throw new IllegalArgumentException("Can't convert more than 8 bytes.");
337         }
338         if (size > bytes.length) {
339             throw new IllegalArgumentException("Size can't be larger than array length.");
340         }
341         long total = 0L;
342         for (int i = 0; i < size; i++) {
343             total = total << 8 | bytes[i] & 0xff;
344         }
345         return total;
346     }

365     /**
366      * Convert an unsigned integer to a long value. The method assumes that all bits
367      * in the specified integer value are 'data' bits, including the
368      * most-significant bit which Java normally considers a sign bit. The method
369      * must be used only when it is certain that the integer value represents an
370      * unsigned integer, for example when the integer is returned by JNA library in
371      * a structure which holds unsigned integers.
372      *
373      * @param unsignedValue
374      *            The unsigned integer value to convert.
375      * @return The unsigned integer value widened to a long.
376      */
377     public static long unsignedIntToLong(int unsignedValue) {
378         // use standard Java widening conversion to long which does
379         // sign-extension,
380         // then drop any copies of the sign bit, to prevent the value being
381         // considered a negative one by Java if it is set
382         long longValue = unsignedValue;
383         return longValue & 0xffffffffL;
384     }

399     /**
400      * Parses a string of hex digits to a string where each pair of hex digits
401      * represents an ASCII character
402      *
403      * @param hexString
404      *            A sequence of hex digits
405      * @return The corresponding string if valid hex; otherwise the original
406      *         hexString
407      */
408     public static String hexStringToString(String hexString) {
409         // Odd length strings won't parse, return
410         if (hexString.length() % 2 > 0) {
411             return hexString;
412         }
413         int charAsInt;
414         StringBuilder sb = new StringBuilder();
415         try {
416             for (int pos = 0; pos < hexString.length(); pos += 2) {
417                 charAsInt = Integer.parseInt(hexString.substring(pos, pos + 2), 16);
418                 if (charAsInt < 32 || charAsInt > 127) {
419                     return hexString;
420                 }
421                 sb.append((char) charAsInt);
422             }
423         } catch (NumberFormatException e) {
424             LOG.trace(DEFAULT_LOG_MSG, hexString, e);
425             // Hex failed to parse, just return the existing string
426             return hexString;
427         }
428         return sb.toString();
429     }

505     /**
506      * Attempts to parse a string of the form [DD-[hh:]]mm:ss[.ddd] to a number of
507      * milliseconds. If it fails, returns the default.
508      *
509      * @param s
510      *            The string to parse
511      * @param defaultLong
512      *            The value to return if parsing fails
513      * @return The parsed number of seconds, or the default if parsing fails
514      */
515     public static long parseDHMSOrDefault(String s, long defaultLong) {
516         Matcher m = DHMS.matcher(s);
517         if (m.matches()) {
518             long milliseconds = 0L;
519             if (m.group(1) != null) {
520                 milliseconds += parseLongOrDefault(m.group(1), 0L) * 86400000L;
521             }
522             if (m.group(2) != null) {
523                 milliseconds += parseLongOrDefault(m.group(2), 0L) * 3600000L;
524             }
525             if (m.group(3) != null) {
526                 milliseconds += parseLongOrDefault(m.group(3), 0L) * 60000L;
527             }
528             milliseconds += parseLongOrDefault(m.group(4), 0L) * 1000L;
529             milliseconds += (long) (1000 * parseDoubleOrDefault("0." + m.group(5), 0d));
530             return milliseconds;
531         }
532         return defaultLong;
533     }

594         return line.substring(firstOcc + 1, line.lastIndexOf(c)).trim();
595     }

628     /**
629      * Removes all matching sub strings from the string. More efficient than regexp.
630      *
631      * @param original
632      *            source String to remove from
633      * @param toRemove
634      *            the sub string to be removed
635      * @return The string with all matching substrings removed
636      */
637     public static String removeMatchingString(final String original, final String toRemove) {
638         if (original == null || original.isEmpty() || toRemove == null || toRemove.isEmpty()) {
639             return original;
640         }
641 
642         int matchIndex = original.indexOf(toRemove, 0);
643         if (matchIndex == -1) {
644             return original;
645         }
646 
647         StringBuilder buffer = new StringBuilder(original.length() - toRemove.length());
648         int currIndex = 0;
649         do {
650             buffer.append(original.substring(currIndex, matchIndex));
651             currIndex = matchIndex + toRemove.length();
652             matchIndex = original.indexOf(toRemove, currIndex);
653         } while (matchIndex != -1);
654 
655         buffer.append(original.substring(currIndex));
656         return buffer.toString();
657     }

659     /**
660      * Parses a delimited string to an array of longs. Optimized for processing
661      * predictable-length arrays such as outputs of reliably formatted Linux proc or
662      * sys filesystem, minimizing new object creation. Users should perform other
663      * sanity checks of data.
664      *
665      * As a special case, non-numeric fields (such as UUIDs in OpenVZ) at the end of
666      * the list are ignored. Values greater than the max long value return the max
667      * long value.
668      *
669      * The indices parameters are referenced assuming the length as specified, and
670      * leading characters are ignored. For example, if the string is "foo 12 34 5"
671      * and the length is 3, then index 0 is 12, index 1 is 34, and index 2 is 5.
672      *
673      * @param s
674      *            The string to parse
675      * @param indices
676      *            An array indicating which indexes should be populated in the final
677      *            array; other values will be skipped. This idex is zero-referenced
678      *            assuming the rightmost delimited fields of the string contain the
679      *            array.
680      * @param length
681      *            The total number of elements in the string array. It is
682      *            permissible for the string to have more elements than this;
683      *            leading elements will be ignored. This should be calculated once
684      *            per text format by {@link #countStringToLongArray}.
685      * @param delimiter
686      *            The character to delimit by.
687      * @return If successful, an array of parsed longs. If parsing errors occurred,
688      *         will be an array of zeros.
689      */
690     public static long[] parseStringToLongArray(String s, int[] indices, int length, char delimiter) {
691         long[] parsed = new long[indices.length];
692         // Iterate from right-to-left of String
693         // Fill right to left of result array using index array
694         int charIndex = s.length();
695         int parsedIndex = indices.length - 1;
696         int stringIndex = length - 1;
697 
698         int power = 0;
699         int c;
700         boolean delimCurrent = false;
701         boolean numeric = true;
702         boolean numberFound = false; // ignore nonnumeric at end
703         boolean dashSeen = false; // to flag uuids as nonnumeric
704         while (--charIndex > 0 && parsedIndex >= 0) {
705             c = s.charAt(charIndex);
706             if (c == delimiter) {
707                 // first parseable number?
708                 if (!numberFound && numeric) {
709                     numberFound = true;
710                 }
711                 if (!delimCurrent) {
712                     if (numberFound && indices[parsedIndex] == stringIndex--) {
713                         parsedIndex--;
714                     }
715                     delimCurrent = true;
716                     power = 0;
717                     dashSeen = false;
718                     numeric = true;
719                 }
720             } else if (indices[parsedIndex] != stringIndex || c == '+' || !numeric) {
721                 // Doesn't impact parsing, ignore
722                 delimCurrent = false;
723             } else if (c >= '0' && c <= '9' && !dashSeen) {
724                 if (power > 18 || power == 17 && c == '9' && parsed[parsedIndex] > 223372036854775807L) {
725                     parsed[parsedIndex] = Long.MAX_VALUE;
726                 } else {
727                     parsed[parsedIndex] += (c - '0') * ParseUtil.POWERS_OF_TEN[power++];
728                 }
729                 delimCurrent = false;
730             } else if (c == '-') {
731                 parsed[parsedIndex] *= -1L;
732                 delimCurrent = false;
733                 dashSeen = true;
734             } else {
735                 // Flag as nonnumeric and continue unless we've seen a numeric
736                 // error on everything else
737                 if (numberFound) {
738                     LOG.error("Illegal character parsing string '{}' to long array: {}", s, s.charAt(charIndex));
739                     return new long[indices.length];
740                 }
741                 parsed[parsedIndex] = 0;
742                 numeric = false;
743             }
744         }
745         if (parsedIndex > 0) {
746             LOG.error("Not enough fields in string '{}' parsing to long array: {}", s, indices.length - parsedIndex);
747             return new long[indices.length];
748         }
749         return parsed;
750     }

752     /**
753      * Parses a delimited string to count elements of an array of longs. Intended to
754      * be called once to calculate the {@code length} field for
755      * {@link #parseStringToLongArray}.
756      *
757      * As a special case, non-numeric fields (such as UUIDs in OpenVZ) at the end of
758      * the list are ignored.
759      *
760      * @param s
761      *            The string to parse
762      * @param delimiter
763      *            The character to delimit by
764      * @return The number of parsable long values which follow the last unparsable
765      *         value.
766      */
767     public static int countStringToLongArray(String s, char delimiter) {
768         // Iterate from right-to-left of String
769         // Fill right to left of result array using index array
770         int charIndex = s.length();
771         int numbers = 0;
772 
773         int c;
774         boolean delimCurrent = false;
775         boolean numeric = true;
776         boolean dashSeen = false; // to flag uuids as nonnumeric
777         while (--charIndex > 0) {
778             c = s.charAt(charIndex);
779             if (c == delimiter) {
780                 if (!delimCurrent) {
781                     if (numeric) {
782                         numbers++;
783                     }
784                     delimCurrent = true;
785                     dashSeen = false;
786                     numeric = true;
787                 }
788             } else if (c == '+' || !numeric) {
789                 // Doesn't impact parsing, ignore
790                 delimCurrent = false;
791             } else if (c >= '0' && c <= '9' && !dashSeen) {
792                 delimCurrent = false;
793             } else if (c == '-') {
794                 delimCurrent = false;
795                 dashSeen = true;
796             } else {
797                 // we found non-digit or delimiter. If not last field, exit
798                 if (numbers > 0) {
799                     return numbers;
800                 }
801                 // Else flag as nonnumeric and continue
802                 numeric = false;
803             }
804         }
805         // We got to beginning of string with only numbers, count start as a delimiter
806         // and exit
807         return numbers + 1;
808     }

811      * Get a String in a line of text between two marker strings
812      *
813      * @param text
814      *            Text to search for match
815      * @param before
816      *            Start matching after this text
817      * @param after
818      *            End matching before this text
819      * @return Text between the strings before and after, or empty string if either
820      *         marker does not exist
821      */
822     public static String getTextBetweenStrings(String text, String before, String after) {
823 
824         String result = "";
825 
826         if (text.indexOf(before) >= 0 && text.indexOf(after) >= 0) {
827             result = text.substring(text.indexOf(before) + before.length(), text.length());
828             result = result.substring(0, result.indexOf(after));
829         }
830         return result;
831     }

866     /**
867      * Converts a string in CIM Date Format, as returned by WMI for DateTime types,
868      * into a {@link java.time.OffsetDateTime}.
869      *
870      * @param cimDateTime
871      *            A non-null DateTime String in CIM date format, e.g.,
872      *            <code>20160513072950.782000-420</code>
873      * @return The parsed {@link java.time.OffsetDateTime} if the string is
874      *         parsable, otherwise {@link oshi.util.Constants#UNIX_EPOCH}.
875      */
876     public static OffsetDateTime parseCimDateTimeToOffset(String cimDateTime) {
877         // Keep first 22 characters: digits, decimal, and + or - sign
878         // But alter last 3 characters from a minute offset to hh:mm
879         try {
880             // From WMI as 20160513072950.782000-420,
881             int tzInMinutes = Integer.parseInt(cimDateTime.substring(22));
882             // modified to 20160513072950.782000-07:00 which can be parsed
883             LocalTime offsetAsLocalTime = LocalTime.MIDNIGHT.plusMinutes(tzInMinutes);
884             return OffsetDateTime.parse(
885                     cimDateTime.substring(0, 22) + offsetAsLocalTime.format(DateTimeFormatter.ISO_LOCAL_TIME),
886                     ParseUtil.CIM_FORMAT);
887         } catch (IndexOutOfBoundsException // if cimDate not 22+ chars
888                 | NumberFormatException // if TZ minutes doesn't parse
889                 | DateTimeParseException e) {
890             LOG.trace("Unable to parse {} to CIM DateTime.", cimDateTime);
891             return Constants.UNIX_EPOCH;
892         }
893     }
