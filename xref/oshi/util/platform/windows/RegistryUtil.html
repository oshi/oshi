8   import com.sun.jna.platform.win32.Advapi32Util;
9   import com.sun.jna.platform.win32.Win32Exception;
10  import com.sun.jna.platform.win32.WinReg;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import oshi.util.ParseUtil;

15  import com.sun.jna.platform.win32.WinReg.HKEY;
16  
17  import java.util.Objects;
18  
19  import static com.sun.jna.platform.win32.WinError.ERROR_SUCCESS;
20  import static com.sun.jna.platform.win32.WinNT.KEY_READ;
21  import static oshi.util.ParseUtil.decodeBinaryToString;
22  
23  public final class RegistryUtil {
24  
25      private static final Logger LOG = LoggerFactory.getLogger(RegistryUtil.class);
26  
27      private static final long THIRTY_YEARS_IN_SECS = 30L * 365 * 24 * 60 * 60;
28  
29      private static final Advapi32 ADV = Advapi32.INSTANCE;
30  
31      private RegistryUtil() {
32      }
33  
34      /**
35       * Returns a registry value as a Long. (without access flag) Currently supports String and Integer
36       *
37       * @param root the registry root
38       * @param path the registry path
39       * @param key  the registry key
40       * @return the registry value as a long
41       */
42      public static long getLongValue(HKEY root, String path, String key) {
43          try {
44              Object val = Advapi32Util.registryGetValue(root, path, key);
45              return registryValueToLong(val);
46          } catch (Win32Exception e) {
47              LOG.trace("Unable to access " + path + ": " + e.getMessage());
48          }
49          return 0L;
50      }
51  
52      /**
53       * Returns a registry value as a Long. (with access flag) Currently supports String and Integer
54       *
55       * @param root       the registry root
56       * @param path       the registry path
57       * @param key        the registry key
58       * @param accessFlag the access flag
59       * @return the registry value as a long
60       */
61      public static long getLongValue(HKEY root, String path, String key, int accessFlag) {
62          Object val = getRegistryValueOrNull(root, path, key, accessFlag);
63          return registryValueToLong(val);
64      }
65  
66      /**
67       * Gets a registry value or null if not found
68       *
69       * @param root       the registry root
70       * @param path       the registry path
71       * @param key        the registry key
72       * @param accessFlag the access flag
73       * @return the registry value or null
74       */
75      public static Object getRegistryValueOrNull(HKEY root, String path, String key, int accessFlag) {
76          HKEY hKey = null;
77          try {
78              hKey = Advapi32Util.registryGetKey(root, path, KEY_READ | accessFlag).getValue();
79              Object value = Advapi32Util.registryGetValue(root, path, key);
80              return Objects.isNull(value) ? null : value;
81          } catch (Win32Exception e) {
82              LOG.trace("Unable to access " + path + " with flag " + accessFlag + ": " + e.getMessage());
83          } finally {
84              if (hKey != null) {
85                  int rc = ADV.RegCloseKey(hKey);
86                  if (rc != ERROR_SUCCESS) {
87                      throw new Win32Exception(rc);
88                  }
89              }
90          }
91          return null;
92      }
93  
94      /**
95       * Returns a registry value as a long. Supports Integer and String dates. Converts Unix timestamps (seconds) into
96       * milliseconds.
97       *
98       * @param val the registry value
99       * @return the value as a long
100      */
101     private static long registryValueToLong(Object val) {
102         if (val == null) {
103             return 0L;
104         }
105 
106         // Calculate reasonable timestamp bounds (current time to 30 years ago)
107         long currentTimeSecs = System.currentTimeMillis() / 1000L;
108         long minSaneTimestamp = currentTimeSecs - THIRTY_YEARS_IN_SECS;
109         if (val instanceof Integer) {
110             int value = (Integer) val;
111             if (value > minSaneTimestamp && value < currentTimeSecs) {
112                 return value * 1000L;
113             }
114             return value;
115         } else if (val instanceof String) {
116             String dateStr = ((String) val).trim();
117             // Try yyyyMMdd first
118             long epoch = ParseUtil.parseDateToEpoch(dateStr, "yyyyMMdd");
119             if (epoch == 0) {
120                 // If that fails, try MM/dd/yyyy
121                 epoch = ParseUtil.parseDateToEpoch(dateStr, "MM/dd/yyyy");
122             }
123             return epoch;
124         }
125         return 0L;
126     }
127 
128     /**
129      * Returns a registry value as a String. (without access flag) Currently supports String and Binary
130      *
131      * @param root the registry root
132      * @param path the registry path
133      * @param key  the registry key
134      * @return the registry value as a string
135      */
136     public static String getStringValue(WinReg.HKEY root, String path, String key) {
137         try {
138             Object val = Advapi32Util.registryGetValue(root, path, key);
139             return registryValueToString(val);
140         } catch (Win32Exception e) {
141             LOG.trace("Unable to access " + path + ": " + e.getMessage());
142             return null;
143         }
144     }
145 
146     /**
147      * Returns a registry value as a String. (with access flag) Currently supports String and Binary
148      *
149      * @param root       the registry root
150      * @param path       the registry path
151      * @param key        the registry key
152      * @param accessFlag the access flag
153      * @return the registry value as a string
154      */
155     public static String getStringValue(WinReg.HKEY root, String path, String key, int accessFlag) {
156         Object val = getRegistryValueOrNull(root, path, key, accessFlag);
157         return registryValueToString(val);
158     }
159 
160     /**
161      * Decodes registry value to String using multiple fallback encodings.
162      *
163      * @param val the registry value
164      * @return the decoded string
165      */
166     private static String registryValueToString(Object val) {
167         if (val == null) {
168             return null;
169         }
170 
171         // Already a string (REG_SZ or REG_EXPAND_SZ)
172         if (val instanceof String) {
173             return ((String) val).trim();
174         }
175 
176         // handle binary (REG_BINARY)
177         if (val instanceof byte[]) {
178             return decodeBinaryToString((byte[]) val);
179         }
180 
181         return null;
182     }
