55      public static boolean isCurrentProcessElevated() {
56          try (Arena arena = Arena.ofConfined()) {
57  
58              Optional<MemorySegment> hProcessOpt = Kernel32FFM.GetCurrentProcess();
59              if (hProcessOpt.isEmpty()) {
60                  return false;
61              }
62              MemorySegment hProcess = hProcessOpt.get();
63  
64              MemorySegment hTokenPtr = arena.allocate(ADDRESS);
65  
66              if (!OpenProcessToken(hProcess, WinNTFFM.TOKEN_QUERY, hTokenPtr)) {
67                  return false;
68              }
69  
70              MemorySegment hToken = hTokenPtr.get(ADDRESS, 0);
71  
72              try {
73                  MemorySegment elevation = arena.allocate(WinNTFFM.TOKEN_ELEVATION);
74                  MemorySegment returnLength = arena.allocate(JAVA_INT);
75  
76                  boolean success = GetTokenInformation(hToken, WinNTFFM.TokenElevation, elevation,
77                          (int) WinNTFFM.TOKEN_ELEVATION.byteSize(), returnLength);
78  
79                  if (!success) {
80                      return false;
81                  }
82  
83                  int tokenIsElevated = elevation.get(JAVA_INT,
84                          WinNTFFM.TOKEN_ELEVATION.byteOffset(MemoryLayout.PathElement.groupElement("TokenIsElevated")));
85  
86                  return tokenIsElevated > 0;
87              } finally {
88                  Kernel32FFM.CloseHandle(hToken);
89              }
90          } catch (Throwable t) {
91              LOG.debug("Advapi32FFM.isCurrentProcessElevated failed", t);
92              return false;
93          }
94      }

96      public static String[] registryGetKeys(MemorySegment hKey) throws Throwable {
97          try (Arena arena = Arena.ofConfined()) {
98              MemorySegment lpcSubKeys = arena.allocate(JAVA_INT);
99              MemorySegment lpcMaxSubKeyLen = arena.allocate(JAVA_INT);
100 
101             int rc = RegQueryInfoKey(hKey, NULL, NULL, NULL, lpcSubKeys, lpcMaxSubKeyLen, NULL, NULL, NULL, NULL, NULL,
102                     NULL);
103 
104             if (rc != ERROR_SUCCESS) {
105                 throw new Win32Exception(rc);
106             }
107 
108             int subKeyCount = lpcSubKeys.get(JAVA_INT, 0);
109             int maxNameLen = lpcMaxSubKeyLen.get(JAVA_INT, 0);
110 
111             List<String> keys = new ArrayList<>(subKeyCount);
112             for (int i = 0; i < subKeyCount; i++) {
113                 MemorySegment nameBuf = arena.allocate((maxNameLen + 1) * 2);
114                 MemorySegment nameLen = arena.allocate(JAVA_INT);
115                 nameLen.set(JAVA_INT, 0, maxNameLen + 1);
116 
117                 rc = RegEnumKeyEx(hKey, i, nameBuf, nameLen, NULL, NULL, NULL, NULL);
118                 if (rc != ERROR_SUCCESS) {
119                     throw new Win32Exception(rc);
120                 }
121 
122                 keys.add(readWideString(nameBuf));
123             }
124 
125             return keys.toArray(String[]::new);
126         }
127     }

129     public static String[] registryGetKeys(MemorySegment rootKey, String keyPath, int samDesiredExtra)
130             throws Throwable {
131         try (Arena arena = Arena.ofConfined()) {
132             MemorySegment phkResult = arena.allocate(ADDRESS);
133             int rc = RegOpenKeyEx(rootKey, toWideString(arena, keyPath), 0, KEY_READ | samDesiredExtra, phkResult);
134             if (rc != ERROR_SUCCESS) {
135                 throw new Win32Exception(rc);
136             }
137 
138             MemorySegment hKey = phkResult.get(ADDRESS, 0);
139             try {
140                 return registryGetKeys(hKey);
141             } finally {
142                 rc = RegCloseKey(hKey);
143                 if (rc != ERROR_SUCCESS) {
144                     throw new Win32Exception(rc);
145                 }
146             }
147         }
148     }

150     public static int registryGetDword(MemorySegment hKey, String valueName) throws Throwable {
151         try (Arena arena = Arena.ofConfined()) {
152             MemorySegment pData = arena.allocate(JAVA_INT);
153             MemorySegment lpType = arena.allocate(JAVA_INT);
154             MemorySegment lpcbData = arena.allocate(JAVA_INT);
155             lpcbData.set(JAVA_INT, 0, Integer.BYTES);
156 
157             int rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, pData, lpcbData);
158             checkSuccess(rc, ERROR_INSUFFICIENT_BUFFER);
159             return pData.get(JAVA_INT, 0);
160         }
161     }

163     public static String registryGetString(MemorySegment hKey, String valueName, int size) throws Throwable {
164         try (Arena arena = Arena.ofConfined()) {
165             MemorySegment data = arena.allocate(size + 2);
166             MemorySegment lpType = arena.allocate(JAVA_INT);
167             MemorySegment lpcbData = arena.allocate(JAVA_INT);
168             lpcbData.set(JAVA_INT, 0, size);
169 
170             int rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, data, lpcbData);
171             checkSuccess(rc, ERROR_INSUFFICIENT_BUFFER);
172             return readWideString(data);

176     public static Object registryGetValue(MemorySegment hKey, String valueName) throws Throwable {
177         try (Arena arena = Arena.ofConfined()) {
178             MemorySegment lpType = arena.allocate(JAVA_INT);
179             MemorySegment lpcbData = arena.allocate(JAVA_INT);
180 
181             int rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, MemorySegment.NULL, lpcbData);
182             checkSuccess(rc, ERROR_INSUFFICIENT_BUFFER);
183 
184             int type = lpType.get(JAVA_INT, 0);
185             int size = lpcbData.get(JAVA_INT, 0);
186 
187             return switch (type) {
188                 case REG_SZ, REG_EXPAND_SZ -> registryGetString(hKey, valueName, size);
189                 case REG_DWORD -> registryGetDword(hKey, valueName);
190                 default -> {
191                     LOG.warn("Unsupported registry data type " + type + " for " + valueName);
192                     yield null;
193                 }
194             };
195         }
196     }

198     public static long querySystemBootTime() {
199         String eventLog = systemLog.get();
200         try (Arena arena = Arena.ofConfined()) {
201             Optional<MemorySegment> hEventLogOpt = Advapi32FFM.OpenEventLog(arena, eventLog);
202             if (hEventLogOpt.isEmpty()) {
203                 LOG.warn("Unable to open system Event Log. Falling back to uptime.");
204                 return System.currentTimeMillis() / 1000L - Kernel32UtilFFM.querySystemUptime();
205             }
206 
207             MemorySegment hEventLog = hEventLogOpt.get();
208 
209             int bufSize = 64 * 1024;
210             MemorySegment buffer = arena.allocate(bufSize);
211             MemorySegment bytesRead = arena.allocate(JAVA_INT);
212             MemorySegment minBytesNeeded = arena.allocate(JAVA_INT);
213 
214             long event6005Time = 0L;
215 
216             long OFFSET_EVENTID = WinNTFFM.OFFSET_EVENTID;
217             long OFFSET_TIME_GENERATED = WinNTFFM.OFFSET_TIME_GENERATED;
218             long OFFSET_LENGTH = WinNTFFM.OFFSET_LENGTH;
219 
220             while (Advapi32FFM.ReadEventLog(hEventLog,
221                     WinNTFFM.EVENTLOG_BACKWARDS_READ | WinNTFFM.EVENTLOG_SEQUENTIAL_READ, buffer, bufSize, bytesRead,
222                     minBytesNeeded)) {
223 
224                 int read = bytesRead.get(JAVA_INT, 0);
225                 int offset = 0;
226 
227                 while (offset < read) {
228                     MemorySegment record = buffer.asSlice(offset, WinNTFFM.EVENTLOGRECORD.byteSize());
229 
230                     int eventId = record.get(JAVA_INT, (int) OFFSET_EVENTID);
231                     long timeGenerated = Integer.toUnsignedLong(record.get(JAVA_INT, (int) OFFSET_TIME_GENERATED));
232 
233                     if (eventId == 12) { // system boot
234                         Advapi32FFM.CloseEventLog(hEventLog);
235                         return timeGenerated;
236                     } else if (eventId == 6005) { // event log startup
237                         if (event6005Time > 0) {
238                             Advapi32FFM.CloseEventLog(hEventLog);
239                             return event6005Time;
240                         }
241                         event6005Time = timeGenerated;
242                     }
243 
244                     // Advance to next record
245                     int length = record.get(JAVA_INT, (int) OFFSET_LENGTH);
246                     offset += length;
247                 }
248             }
249 
250             Advapi32FFM.CloseEventLog(hEventLog);
251 
252             if (event6005Time > 0) {
253                 return event6005Time;
254             }
255         } catch (Throwable t) {
256             LOG.error("Exception while querying system boottime, fallback to boot time from uptime", t);
257         }
258         // Fallback: approximate boot time from uptime
259         return System.currentTimeMillis() / 1000L - Kernel32UtilFFM.querySystemUptime();
260     }

262     public static String querySystemLog() {
263         String systemLog = GlobalConfig.get(GlobalConfig.OSHI_OS_WINDOWS_EVENTLOG, "System");
264         if (systemLog.isEmpty()) {
265             return null;
266         }
267 
268         try (Arena arena = Arena.ofConfined()) {
269             MemorySegment sourceName = toWideString(arena, systemLog);
270 
271             Optional<MemorySegment> hEventLog = Advapi32FFM.OpenEventLog(NULL, sourceName);
272             if (hEventLog.isEmpty()) {
273                 LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
274                         systemLog);
275                 return null;
276             }
277 
278             return systemLog;
279         } catch (Throwable t) {
280             LOG.error("Exception while opening system event log", t);
281             return null;
282         }
