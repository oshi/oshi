38  /**
39   * <p>
40   * PerfCounterQuery class.
41   * </p>
42   */
43  public class PerfCounterQuery<T extends Enum<T>> {
44  
45      private static final Logger LOG = LoggerFactory.getLogger(PerfCounterQuery.class);
46  
47      /*
48       * Set on instantiation
49       */
50      protected final Class<T> propertyEnum;
51      protected final String perfObject;
52      protected final String perfWmiClass;
53      protected final String queryKey;
54      protected CounterDataSource source;
55      protected PerfCounterQueryHandler pdhQueryHandler;
56      protected WmiQueryHandler wmiQueryHandler;
57      /*
58       * Only one will be non-null depending on source
59       */
60      private EnumMap<T, PerfCounter> counterMap = null;
61      protected WmiQuery<T> counterQuery = null;
62  
63      /*
64       * Multiple classes use these constants
65       */
66      /** Constant <code>TOTAL_INSTANCE="_Total"</code> */
67      public static final String TOTAL_INSTANCE = "_Total";
68      /** Constant <code>TOTAL_INSTANCES="*_Total"</code> */
69      public static final String TOTAL_INSTANCES = "*_Total";
70      /** Constant <code>NOT_TOTAL_INSTANCE="^ + TOTAL_INSTANCE"</code> */
71      public static final String NOT_TOTAL_INSTANCE = "^" + TOTAL_INSTANCE;
72      /** Constant <code>NOT_TOTAL_INSTANCES="^ + TOTAL_INSTANCES"</code> */
73      public static final String NOT_TOTAL_INSTANCES = "^" + TOTAL_INSTANCES;
74  
75      /**
76       * Construct a new object to hold performance counter data source and results
77       *
78       * @param propertyEnum
79       *            An enum which implements
80       *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
81       *            and contains the WMI field (Enum value) and PDH Counter string
82       *            (instance and counter)
83       * @param perfObject
84       *            The PDH object for this counter; all counters on this object will
85       *            be refreshed at the same time
86       * @param perfWmiClass
87       *            The WMI PerfData_RawData_* class corresponding to the PDH object
88       */
89      public PerfCounterQuery(Class<T> propertyEnum, String perfObject, String perfWmiClass) {
90          this(propertyEnum, perfObject, perfWmiClass, perfObject);
91      }
92  
93      /**
94       * Construct a new object to hold performance counter data source and results
95       *
96       * @param propertyEnum
97       *            An enum which implements
98       *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
99       *            and contains the WMI field (Enum value) and PDH Counter string
100      *            (instance and counter)
101      * @param perfObject
102      *            The PDH object for this counter; all counters on this object will
103      *            be refreshed at the same time
104      * @param perfWmiClass
105      *            The WMI PerfData_RawData_* class corresponding to the PDH object
106      * @param queryKey
107      *            An optional key for PDH counter updates; defaults to the PDH
108      *            object name
109      */
110     public PerfCounterQuery(Class<T> propertyEnum, String perfObject, String perfWmiClass, String queryKey) {
111         this.propertyEnum = propertyEnum;
112         this.perfObject = perfObject;
113         this.perfWmiClass = perfWmiClass;
114         this.queryKey = queryKey;
115         this.pdhQueryHandler = PerfCounterQueryHandler.getInstance();
116         this.wmiQueryHandler = WmiQueryHandler.createInstance();
117         // Start off with PDH as source; if query here fails we will permanently
118         // fall back to WMI
119         this.source = CounterDataSource.PDH;
120     }
121 
122     /**
123      * Set the Data Source for these counters
124      *
125      * @param source
126      *            The source of data
127      * @return Whether the data source was successfully set
128      */
129     public boolean setDataSource(CounterDataSource source) {
130         this.source = source;
131         switch (source) {
132         case PDH:
133             LOG.debug("Attempting to set PDH Data Source.");
134             unInitWmiCounters();
135             return initPdhCounters();
136         case WMI:
137             LOG.debug("Attempting to set WMI Data Source.");
138             unInitPdhCounters();
139             initWmiCounters();
140             return true;
141         default:
142             // This should never happen unless you've added a new source and
143             // forgot to add a case for it
144             throw new IllegalArgumentException("Invalid Data Source specified.");
145         }
146     }
147 
148     /**
149      * Initialize PDH counters for this data source. Adds necessary counters to a
150      * PDH Query.
151      *
152      * @return True if the counters were successfully added.
153      */
154     protected boolean initPdhCounters() {
155         this.counterMap = new EnumMap<>(propertyEnum);
156         for (T prop : propertyEnum.getEnumConstants()) {
157             PerfCounter counter = PerfDataUtil.createCounter(perfObject, ((PdhCounterProperty) prop).getInstance(),
158                     ((PdhCounterProperty) prop).getCounter());
159             counterMap.put(prop, counter);
160             if (!pdhQueryHandler.addCounterToQuery(counter, this.queryKey)) {
161                 unInitPdhCounters();
162                 return false;
163             }
164         }
165         return true;
166     }
167 
168     /**
169      * Uninitialize PDH counters for this data source. Removes necessary counters
170      * from the PDH Query, releasing their handles.
171      */
172     protected void unInitPdhCounters() {
173         pdhQueryHandler.removeAllCountersFromQuery(this.queryKey);
174         this.counterMap = null;
175     }
176 
177     /**
178      * Initialize the WMI query object needed to retrieve counters for this data
179      * source.
180      */
181     protected void initWmiCounters() {
182         this.counterQuery = new WmiQuery<>(perfWmiClass, propertyEnum);
183     }
184 
185     /**
186      * Uninitializes the WMI query object needed to retrieve counters for this data
187      * source, allowing it to be garbage collected.
188      */
189     protected void unInitWmiCounters() {
190         this.counterQuery = null;
191     }
192 
193     /**
194      * Query the current data source (PDH or WMI) for the Performance Counter values
195      * corresponding to the property enum.
196      *
197      * @return A map of the values by the counter enum.
198      */
199     public Map<T, Long> queryValues() {
200         EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
201         T[] props = this.propertyEnum.getEnumConstants();
202         if (source.equals(CounterDataSource.PDH)) {
203             // Set up the query and counter handles, and query
204             if (initPdhCounters() && queryPdh(valueMap, props)) {
205                 // If both init and query return true, then valueMap contains
206                 // the results. Release the handles.
207                 unInitPdhCounters();
208             } else {
209                 // If either init or query failed, switch to WMI
210                 setDataSource(CounterDataSource.WMI);
211             }
212         }
213         if (source.equals(CounterDataSource.WMI)) {
214             queryWmi(valueMap, props);
215         }
216         return valueMap;
217     }
218 
219     private boolean queryPdh(Map<T, Long> valueMap, T[] props) {
220         if (counterMap != null && 0 < pdhQueryHandler.updateQuery(this.queryKey)) {
221             for (T prop : props) {
222                 valueMap.put(prop, pdhQueryHandler.queryCounter(counterMap.get(prop)));
223             }
224             return true;
225         }
226         // Zero timestamp means update failed after muliple
227         // attempts; fallback to WMI
228         return false;
229     }
230 
231     private void queryWmi(Map<T, Long> valueMap, T[] props) {
232         WmiResult<T> result = wmiQueryHandler.queryWMI(this.counterQuery);
233         if (result.getResultCount() > 0) {
234             for (T prop : props) {
235                 switch (result.getCIMType(prop)) {
236                 case Wbemcli.CIM_UINT16:
237                     valueMap.put(prop, Long.valueOf(WmiUtil.getUint16(result, prop, 0)));
238                     break;
239                 case Wbemcli.CIM_UINT32:
240                     valueMap.put(prop, WmiUtil.getUint32asLong(result, prop, 0));
241                     break;
242                 case Wbemcli.CIM_UINT64:
243                     valueMap.put(prop, WmiUtil.getUint64(result, prop, 0));
244                     break;
245                 default:
246                     throw new ClassCastException("Unimplemented CIM Type Mapping.");
247                 }
248             }
249         }
250     }

252     /**
253      * Source of performance counter data.
254      */
255     public enum CounterDataSource {
256         /**
257          * Performance Counter data will be pulled from a PDH Counter
258          */
259         PDH,
260         /**
261          * Performance Counter data will be pulled from a WMI PerfData_RawData_* table
262          */
263         WMI;
264     }
265 
266     /**
267      * Contract for Counter Property Enums
268      */
269     public interface PdhCounterProperty {
270         /**
271          * @return Returns the instance.
272          */
273         String getInstance();
274 
275         /**
276          * @return Returns the counter.
277          */
278         String getCounter();
279     }
