20  /**
21   * FFM-based utility providing WMI query and result classes that mirror the JNA WbemcliUtil API.
22   */
23  public final class WbemcliUtilFFM {
24  
25      private static final Logger LOG = LoggerFactory.getLogger(WbemcliUtilFFM.class);
26  
27      private WbemcliUtilFFM() {
28      }
29  
30      /**
31       * Helper class wrapping information required for a WMI query.
32       *
33       * @param <T> An enum whose constants define the WMI properties to query
34       */
35      public static class WmiQuery<T extends Enum<T>> {
36  
37          private String nameSpace;
38          private final String wmiClassName;
39          private final Class<T> propertyEnum;
40  
41          /**
42           * Instantiate a WmiQuery in the default namespace.
43           *
44           * @param wmiClassName The WMI Class to use. May include a WHERE clause.
45           * @param propertyEnum An Enum that contains the properties to query
46           */
47          public WmiQuery(String wmiClassName, Class<T> propertyEnum) {
48              this(WbemcliFFM.DEFAULT_NAMESPACE, wmiClassName, propertyEnum);
49          }
50  
51          /**
52           * Instantiate a WmiQuery.
53           *
54           * @param nameSpace    The WMI namespace to use
55           * @param wmiClassName The WMI class to use. May include a WHERE clause.
56           * @param propertyEnum An enum for type mapping
57           */
58          public WmiQuery(String nameSpace, String wmiClassName, Class<T> propertyEnum) {
59              this.nameSpace = nameSpace;
60              this.wmiClassName = wmiClassName;
61              this.propertyEnum = propertyEnum;
62          }
63  
64          public Class<T> getPropertyEnum() {
65              return propertyEnum;
66          }
67  
68          public String getNameSpace() {
69              return nameSpace;
70          }
71  
72          public void setNameSpace(String nameSpace) {
73              this.nameSpace = nameSpace;
74          }
75  
76          public String getWmiClassName() {
77              return wmiClassName;
78          }
79      }
80  
81      /**
82       * Helper class wrapping the results of a WMI query, indexed by enum property and row.
83       *
84       * @param <T> An enum whose constants define the WMI properties
85       */
86      public static class WmiResult<T extends Enum<T>> {
87  
88          private final Map<T, List<Object>> propertyMap;
89          private final Map<T, Integer> vtTypeMap;
90          private final Map<T, Integer> cimTypeMap;
91          private int resultCount;
92  
93          public WmiResult(Class<T> propertyEnum) {
94              propertyMap = new EnumMap<>(propertyEnum);
95              vtTypeMap = new EnumMap<>(propertyEnum);
96              cimTypeMap = new EnumMap<>(propertyEnum);
97              for (T prop : propertyEnum.getEnumConstants()) {
98                  propertyMap.put(prop, new ArrayList<>());
99                  vtTypeMap.put(prop, 0); // VT_EMPTY
100                 cimTypeMap.put(prop, WbemcliFFM.CIM_EMPTY);
101             }
102         }
103 
104         public Object getValue(T property, int index) {
105             return propertyMap.get(property).get(index);
106         }
107 
108         public int getVtType(T property) {
109             return vtTypeMap.get(property);
110         }
111 
112         public int getCIMType(T property) {
113             return cimTypeMap.get(property);
114         }
115 
116         public int getResultCount() {
117             return resultCount;
118         }
119 
120         /**
121          * Adds a value for the given property.
122          *
123          * @param vtType   the VT type
124          * @param cimType  the CIM type
125          * @param property the property enum constant
126          * @param value    the value (may be null)
127          */
128         public void add(int vtType, int cimType, T property, Object value) {
129             propertyMap.get(property).add(value);
130             if (vtType != oshi.ffm.windows.com.VariantFFM.VT_NULL && vtType != oshi.ffm.windows.com.VariantFFM.VT_EMPTY
131                     && vtTypeMap.get(property) == 0) {
132                 vtTypeMap.put(property, vtType);
133             }
134             if (cimType != WbemcliFFM.CIM_EMPTY && cimTypeMap.get(property) == WbemcliFFM.CIM_EMPTY) {
135                 cimTypeMap.put(property, cimType);
136             }
137         }
138 
139         /**
140          * Increments the result count.
141          */
142         public void incrementResultCount() {
143             resultCount++;
144         }
145     }
146 
147     /**
148      * Populates a WmiResult row from an IWbemClassObject by reading each property defined in the enum.
149      *
150      * @param <T>     the property enum type
151      * @param pObject the IWbemClassObject pointer
152      * @param arena   the arena for memory allocation
153      * @param result  the WmiResult to populate
154      */
155     public static <T extends Enum<T>> void populateRow(MemorySegment pObject, Arena arena, WmiResult<T> result) {
156         for (T property : result.cimTypeMap.keySet()) {
157             IWbemClassObjectFFM.GetResult getResult = IWbemClassObjectFFM.get(pObject, property.name(), arena);
158             try {
159                 if (!getResult.succeeded()) {
160                     result.add(0, WbemcliFFM.CIM_EMPTY, property, null);
161                     continue;
162                 }
163                 int vt = oshi.ffm.windows.com.VariantFFM.getVt(getResult.variant());
164                 int cimType = getResult.cimType();
165                 switch (vt) {
166                     case oshi.ffm.windows.com.VariantFFM.VT_BSTR:
167                         result.add(vt, cimType, property,
168                                 oshi.ffm.windows.com.VariantFFM.getBstrVal(getResult.variant(), arena));
169                         break;
170                     case oshi.ffm.windows.com.VariantFFM.VT_I4:
171                     case oshi.ffm.windows.com.VariantFFM.VT_INT:
172                     case oshi.ffm.windows.com.VariantFFM.VT_UI4:
173                     case oshi.ffm.windows.com.VariantFFM.VT_UINT:
174                         result.add(vt, cimType, property,
175                                 oshi.ffm.windows.com.VariantFFM.getIntVal(getResult.variant()));
176                         break;
177                     case oshi.ffm.windows.com.VariantFFM.VT_I8:
178                     case oshi.ffm.windows.com.VariantFFM.VT_UI8:
179                         result.add(vt, cimType, property,
180                                 oshi.ffm.windows.com.VariantFFM.getLongVal(getResult.variant()));
181                         break;
182                     case oshi.ffm.windows.com.VariantFFM.VT_I2:
183                         result.add(vt, cimType, property,
184                                 (int) oshi.ffm.windows.com.VariantFFM.getShortVal(getResult.variant()));
185                         break;
186                     case oshi.ffm.windows.com.VariantFFM.VT_UI2:
187                         result.add(vt, cimType, property,
188                                 (int) oshi.ffm.windows.com.VariantFFM.getShortVal(getResult.variant()) & 0xFFFF);
189                         break;
190                     case oshi.ffm.windows.com.VariantFFM.VT_I1:
191                         result.add(vt, cimType, property,
192                                 (int) oshi.ffm.windows.com.VariantFFM.getByteVal(getResult.variant()));
193                         break;
194                     case oshi.ffm.windows.com.VariantFFM.VT_UI1:
195                         result.add(vt, cimType, property,
196                                 (int) (oshi.ffm.windows.com.VariantFFM.getByteVal(getResult.variant()) & 0xFF));
197                         break;
198                     case oshi.ffm.windows.com.VariantFFM.VT_BOOL:
199                         result.add(vt, cimType, property,
200                                 oshi.ffm.windows.com.VariantFFM.getBoolVal(getResult.variant()));
201                         break;
202                     case oshi.ffm.windows.com.VariantFFM.VT_R4:
203                         result.add(vt, cimType, property,
204                                 oshi.ffm.windows.com.VariantFFM.getFloatVal(getResult.variant()));
205                         break;
206                     case oshi.ffm.windows.com.VariantFFM.VT_R8:
207                         result.add(vt, cimType, property,
208                                 oshi.ffm.windows.com.VariantFFM.getDoubleVal(getResult.variant()));
209                         break;
210                     case oshi.ffm.windows.com.VariantFFM.VT_NULL:
211                     case oshi.ffm.windows.com.VariantFFM.VT_EMPTY:
212                         result.add(vt, cimType, property, null);
213                         break;
214                     default:
215                         LOG.debug("Unhandled VT type {} for property {}", vt, property.name());
216                         result.add(oshi.ffm.windows.com.VariantFFM.VT_EMPTY, cimType, property, null);
217                         break;
218                 }
219             } finally {
220                 oshi.ffm.windows.com.VariantFFM.clear(getResult.variant());
221             }
222         }
223         result.incrementResultCount();
224     }
