68      public static String[] getDnsServers() {
69          try (Arena arena = Arena.ofConfined()) {
70              MemorySegment sizeSegment = arena.allocate(JAVA_INT);
71              int ret = GetNetworkParams(NULL, sizeSegment);
72              if (ret != ERROR_BUFFER_OVERFLOW) {
73                  LOG.error("Failed to get network parameters size. Error: {}", ret);
74                  return new String[0];
75              }
76  
77              int size = sizeSegment.get(JAVA_INT, 0);
78              MemorySegment buffer = arena.allocate(size);
79              ret = GetNetworkParams(buffer, sizeSegment);
80              if (ret != 0) {
81                  LOG.error("Failed to get network parameters. Error: {}", ret);
82                  return new String[0];
83              }
84  
85              MemorySegment fixedInfo = buffer.asSlice(0, FIXED_INFO_LAYOUT.byteSize());
86              List<String> dnsServers = new ArrayList<>();
87  
88              MemorySegment dnsServerList = fixedInfo.asSlice(
89                      FIXED_INFO_LAYOUT.byteOffset(PathElement.groupElement("DnsServerList")),
90                      IP_ADDR_STRING_LAYOUT.byteSize());
91  
92              while (dnsServerList != NULL) {
93                  MemorySegment ipStringSeg = dnsServerList
94                          .asSlice(IP_ADDR_STRING_LAYOUT.byteOffset(PathElement.groupElement("IpAddress")), 16);
95  
96                  String ipString = readAnsiString(ipStringSeg, 16);
97                  if (!ipString.isEmpty()) {
98                      dnsServers.add(ipString);
99                  }
100 
101                 dnsServerList = dnsServerList.get(ADDRESS,
102                         IP_ADDR_STRING_LAYOUT.byteOffset(PathElement.groupElement("Next")));
103                 if (dnsServerList.address() == 0) {
104                     break;
105                 }
106             }
107 
108             return dnsServers.toArray(new String[0]);
109         } catch (Throwable t) {
110             LOG.error("GetNetworkParams failed: {}", t.getMessage());
111             return new String[0];
112         }
113     }

115     public static TcpStats getTcpStats(int family) {
116         try (Arena arena = Arena.ofConfined()) {
117             MemorySegment stats = arena.allocate(MIB_TCPSTATS_LAYOUT);
118 
119             int rc = GetTcpStatisticsEx(stats, family);
120             if (rc != ERROR_SUCCESS) {
121                 throw new Win32Exception(rc);
122             }
123 
124             long connectionsEstablished = stats.get(JAVA_INT,
125                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwCurrEstab")));
126             long connectionsActive = stats.get(JAVA_INT,
127                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwActiveOpens")));
128             long connectionsPassive = stats.get(JAVA_INT,
129                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwPassiveOpens")));
130             long connectionFailures = stats.get(JAVA_INT,
131                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwAttemptFails")));
132             long connectionsReset = stats.get(JAVA_INT,
133                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwEstabResets")));
134             long segmentsSent = stats.get(JAVA_INT,
135                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwOutSegs")));
136             long segmentsReceived = stats.get(JAVA_INT,
137                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwInSegs")));
138             long segmentsRetransmitted = stats.get(JAVA_INT,
139                     MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwRetransSegs")));
140             long inErrors = stats.get(JAVA_INT, MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwInErrs")));
141             long outResets = stats.get(JAVA_INT, MIB_TCPSTATS_LAYOUT.byteOffset(PathElement.groupElement("dwOutRsts")));
142 
143             return new TcpStats(connectionsEstablished, connectionsActive, connectionsPassive, connectionFailures,
144                     connectionsReset, segmentsSent, segmentsReceived, segmentsRetransmitted, inErrors, outResets);
145         } catch (Throwable t) {
146             LOG.debug("getTcpStats failed: {}", t.getMessage());
147             return null;
148         }

174     public static List<IPConnection> queryTCPv4Connections() {
175         List<IPConnection> conns = new ArrayList<>();
176         try (Arena arena = Arena.ofConfined()) {
177 
178             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
179             int ret = GetExtendedTcpTable(NULL, sizeSegment, 0, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);
180 
181             if (ret != ERROR_INSUFFICIENT_BUFFER && ret != ERROR_SUCCESS) {
182                 throw new Win32Exception(ret);
183             }
184 
185             int size = sizeSegment.get(JAVA_INT, 0);
186             MemorySegment buffer = arena.allocate(size);
187 
188             ret = GetExtendedTcpTable(buffer, sizeSegment, 0, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);
189             if (ret != ERROR_SUCCESS) {
190                 throw new Win32Exception(ret);
191             }
192 
193             int numEntries = buffer.get(JAVA_INT, 0);
194             long entryOffset = JAVA_INT.byteSize();
195 
196             for (int i = 0; i < numEntries; i++) {
197                 MemorySegment rowSeg = buffer.asSlice(entryOffset + i * MIB_TCPROW_OWNER_PID_LAYOUT.byteSize(),
198                         MIB_TCPROW_OWNER_PID_LAYOUT.byteSize());
199 
200                 int localAddr = rowSeg.get(JAVA_INT,
201                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwLocalAddr")));
202                 int localPortBE = rowSeg.get(JAVA_INT,
203                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwLocalPort")));
204                 int remoteAddr = rowSeg.get(JAVA_INT,
205                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwRemoteAddr")));
206                 int remotePortBE = rowSeg.get(JAVA_INT,
207                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwRemotePort")));
208                 int state = rowSeg.get(JAVA_INT,
209                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwState")));
210                 int pid = rowSeg.get(JAVA_INT,
211                         MIB_TCPROW_OWNER_PID_LAYOUT.byteOffset(PathElement.groupElement("dwOwningPid")));
212 
213                 conns.add(new IPConnection("tcp4", ParseUtil.parseIntToIP(localAddr),
214                         ParseUtil.bigEndian16ToLittleEndian(localPortBE), ParseUtil.parseIntToIP(remoteAddr),
215                         ParseUtil.bigEndian16ToLittleEndian(remotePortBE), stateLookup(state), 0, 0, pid));
216             }
217             return conns;
218         } catch (Throwable t) {
219             LOG.debug("queryTCPv4Connections failed: {}", t.getMessage());
220             return Collections.emptyList();
221         }

369     private static TcpState stateLookup(int state) {
370         switch (state) {
371             case 1:
372             case 12:
373                 return CLOSED;
374             case 2:
375                 return LISTEN;
376             case 3:
377                 return SYN_SENT;
378             case 4:
379                 return SYN_RECV;
380             case 5:
381                 return ESTABLISHED;
382             case 6:
383                 return FIN_WAIT_1;
384             case 7:
385                 return FIN_WAIT_2;
386             case 8:
387                 return CLOSE_WAIT;
388             case 9:
389                 return CLOSING;
390             case 10:
391                 return LAST_ACK;
392             case 11:
393                 return TIME_WAIT;
394             default:
395                 return UNKNOWN;
396         }
