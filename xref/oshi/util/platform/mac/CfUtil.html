19  package oshi.util.platform.mac;
20  
21  import java.util.Map;
22  import java.util.concurrent.ConcurrentHashMap;
23  
24  import com.sun.jna.Memory;
25  import com.sun.jna.Pointer;
26  import com.sun.jna.PointerType;
27  import com.sun.jna.ptr.IntByReference;
28  import com.sun.jna.ptr.LongByReference;
29  
30  import oshi.jna.platform.mac.CoreFoundation;
31  import oshi.jna.platform.mac.CoreFoundation.CFAllocatorRef;
32  import oshi.jna.platform.mac.CoreFoundation.CFStringRef;
33  
34  /**
35   * Provides utilities for Core Foundations
36   *
37   * @author widdis[at]gmail[dot]com
38   */
39  public class CfUtil {
40      public static final CFAllocatorRef ALLOCATOR = CoreFoundation.INSTANCE.CFAllocatorGetDefault();
41  
42      /**
43       * Cache cfStrings
44       */
45      private static Map<String, CFStringRef> cfStringMap = new ConcurrentHashMap<>();
46  
47      /**
48       * Return a CFStringRef representing a string, caching the result
49       *
50       * @param key
51       *            The string, usually a registry key
52       * @return the corresponding CFString
53       */
54      public static CFStringRef getCFString(String key) {
55          return cfStringMap.computeIfAbsent(key, CFStringRef::toCFString);
56      }
57  
58      /**
59       * Enum values used for number type in CFNumberGetValue(). Use ordinal() to
60       * fetch the corresponding constant.
61       */
62      public enum CFNumberType {
63          unusedZero, kCFNumberSInt8Type, kCFNumberSInt16Type, kCFNumberSInt32Type, kCFNumberSInt64Type, kCFNumberFloat32Type, kCFNumberFloat64Type, kCFNumberCharType, kCFNumberShortType, kCFNumberIntType, kCFNumberLongType, kCFNumberLongLongType, kCFNumberFloatType, kCFNumberDoubleType, kCFNumberCFIndexType, kCFNumberNSIntegerType, kCFNumberCGFloatType, kCFNumberMaxType
64      }
65  
66      /**
67       * Convert a pointer representing a Core Foundations LongLong into its long
68       *
69       * @param p
70       *            The pointer to a 64-bit integer
71       * @return The corresponding long
72       */
73      public static long cfPointerToLong(Pointer p) {
74          LongByReference lbr = new LongByReference();
75          CoreFoundation.INSTANCE.CFNumberGetValue(p, CFNumberType.kCFNumberLongLongType.ordinal(), lbr);
76          return lbr.getValue();
77      }
78  
79      /**
80       * Convert a pointer representing a Core Foundations LongLong into its long
81       *
82       * @param p
83       *            The pointer to an integer
84       * @return The corresponding int
85       */
86      public static int cfPointerToInt(Pointer p) {
87          IntByReference ibr = new IntByReference();
88          CoreFoundation.INSTANCE.CFNumberGetValue(p, CFNumberType.kCFNumberIntType.ordinal(), ibr);
89          return ibr.getValue();
90      }
91  
92      /**
93       * Convert a pointer representing a Core Foundations Boolean into its
94       * boolean
95       *
96       * @param p
97       *            The pointer to a boolean
98       * @return The corresponding boolean
99       */
100     public static boolean cfPointerToBoolean(Pointer p) {
101         return CoreFoundation.INSTANCE.CFBooleanGetValue(p);
102     }
103 
104     /**
105      * Convert a pointer representing a Core Foundations String into its string
106      *
107      * @param p
108      *            The pointer to a CFString
109      * @return The corresponding string
110      */
111     public static String cfPointerToString(Pointer p) {
112         if (p == null) {
113             return "null";
114         }
115         long length = CoreFoundation.INSTANCE.CFStringGetLength(p);
116         long maxSize = CoreFoundation.INSTANCE.CFStringGetMaximumSizeForEncoding(length, CoreFoundation.UTF_8);
117         if (maxSize == 0) {
118             maxSize = 1;
119         }
120         Pointer buf = new Memory(maxSize);
121         CoreFoundation.INSTANCE.CFStringGetCString(p, buf, maxSize, CoreFoundation.UTF_8);
122         return buf.getString(0);
123     }
124 
125     /**
126      * Releases a CF reference. Mandatory when an object is owned (using
127      * 'create' or 'copy' methods).
128      *
129      * @param ref
130      *            The reference to release
131      */
132     public static void release(PointerType ref) {
133         if (ref != null) {
134             CoreFoundation.INSTANCE.CFRelease(ref);
135         }
136     }
137 
138 }
