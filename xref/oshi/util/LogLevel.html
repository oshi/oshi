5   package oshi.util;

7   /**
8    * A logging severity level, used by OSHI's exception-handling and native-call helpers (such as {@link ExceptionUtil})
9    * to select which {@code Logger#debug/info/warn/error/trace} method a caught or handled condition is logged with.
10   * <p>
11   * OSHI carries the level with this enum rather than {@code org.slf4j.event.Level} so that it never class-loads a type
12   * introduced in slf4j-api 1.7.15; dispatch uses only classic {@code Logger} methods present since slf4j 1.0. This keeps
13   * OSHI loggable against host-provided slf4j-api versions older than 1.7.15 (e.g. the 1.7.5 bundled by Apache Maven
14   * 3.3.x, which loads OSHI-based extensions against its own slf4j).
15   */
16  public enum LogLevel {
17      /**
18       * The TRACE level.
19       */
20      TRACE,
21      /**
22       * The DEBUG level.
23       */
24      DEBUG,
25      /**
26       * The INFO level.
27       */
28      INFO,
29      /**
30       * The WARN level.
31       */
32      WARN,
33      /**
34       * The ERROR level.
35       */
36      ERROR
37  }
