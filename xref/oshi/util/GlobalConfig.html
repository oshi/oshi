20  /**
21   * The global configuration utility. See {@code src/main/resources/oshi.properties} for default values.
22   * <p>
23   * Configuration is resolved in the following precedence order (highest wins):
24   * <ol>
25   * <li>{@link #set(String, Object)} — programmatic, at runtime</li>
26   * <li>Java system properties ({@code -Doshi.*})</li>
27   * <li>Environment variables ({@code OSHI_*}) — mapped by uppercasing and replacing dots with underscores</li>
28   * <li>External properties file ({@code -Doshi.properties.file=/path/to/file})</li>
29   * <li>{@code oshi.properties} on the classpath (defaults)</li>
30   * </ol>
31   * <p>
32   * This class is not thread safe if methods manipulating the configuration are used. These methods are intended for use
33   * by a single thread at startup, before instantiation of any other OSHI classes. OSHI does not guarantee re-reading of
34   * any configuration changes.
35   */
36  @NotThreadSafe
37  public final class GlobalConfig {
38  
39      private static final String OSHI_PROPERTIES = "oshi.properties";
40  
41      private static final Logger LOG = LoggerFactory.getLogger(GlobalConfig.class);
42  
43      private static final Properties CONFIG = FileUtil.readPropertiesFromFilename(OSHI_PROPERTIES);
44      static {
45          loadExternalConfig(CONFIG);

48      // package-private for testing
49      static void loadExternalConfig(Properties config) {
50          // External properties file: system property takes precedence, env var as fallback
51          String externalFile = System.getProperty("oshi.properties.file");
52          if (externalFile == null || externalFile.isEmpty()) {
53              externalFile = System.getenv("OSHI_PROPERTIES_FILE");
54          }
55          if (externalFile != null && !externalFile.isEmpty()) {
56              try (InputStream is = new FileInputStream(externalFile)) {
57                  config.load(is);
58              } catch (FileNotFoundException e) {
59                  LOG.debug("External properties file '{}' not found, continuing", externalFile);
60              } catch (IOException e) {
61                  LOG.debug("Failed to load external properties file '{}'", externalFile, e);
62              }
63          }
64          // Environment variables (OSHI_*) override external file and classpath defaults
65          System.getenv().forEach((k, v) -> {
66              if (k.startsWith("OSHI_") && !"OSHI_PROPERTIES_FILE".equals(k)) {
67                  config.setProperty(envKeyToProperty(k), v);
68              }
69          });
70          // System properties (-Doshi.*) override everything else
71          System.getProperties().forEach((k, v) -> {
72              String key = k.toString();
73              if (key.startsWith("oshi.") && !"oshi.properties.file".equals(key)) {
74                  config.setProperty(key, v.toString());
75              }
76          });
77      }

79      /**
80       * Converts an environment variable name to a property key. For example, {@code OSHI_OS_LINUX_PRIVILEGED_PREFIX}
81       * becomes {@code oshi.os.linux.privileged.prefix}.
82       *
83       * @param envKey the environment variable name (e.g., {@code OSHI_UTIL_MEMOIZER_EXPIRATION})
84       * @return the corresponding property key
85       */
86      // package-private for testing
87      static String envKeyToProperty(String envKey) {
88          return envKey.toLowerCase(Locale.ROOT).replace('_', '.');
89      }

192      * Whether to use "Processor Utility" for System and per-processor CPU Load ticks (on Windows 8 and higher) to match

194      * <p>
195      * By default, OSHI uses "Processor Time" which measures active and idle times, matching Unix-based systems and
196      * Windows 7 Task Manager. Setting this to {@code true} switches to "Processor Utility" which measures work
197      * completed relative to nominal frequency. Because of features like Intel Turbo Boost and AMD Precision Boost, this
198      * value can exceed 100% (though Task Manager caps it). OSHI does not cap the value.
199      * <p>
200      * For this calculation to produce meaningful data, the ticks must come from the same CentralProcessor instance and
201      * the first polling interval must be less than 7 minutes. Default is {@code false}.
202      *
203      * @see CentralProcessor#getSystemCpuLoadTicks()
204      */

339     /**
340      * Set the given property, overwriting any existing value. If the given value is {@code null}, the property is
341      * removed.
342      *
343      * @param key The property key
344      * @param val The new value
345      */
346     public static void set(String key, Object val) {
347         if (val == null) {
348             CONFIG.remove(key);
349         } else {
350             CONFIG.setProperty(key, val.toString());
351         }
352     }

379     /**
380      * Indicates that a configuration value is invalid.
381      */
382     public static class PropertyException extends RuntimeException {
383 
384         private static final long serialVersionUID = -7482581936621748005L;
385 
386         /**
387          * Creates a PropertyException for an invalid property value.
388          *
389          * @param property The property name
390          */
391         public PropertyException(String property) {
392             super("Invalid property: \"" + property + "\" = " + GlobalConfig.get(property, null));
393         }
394 
395         /**
396          * Creates a PropertyException with a custom message.
397          *
398          * @param property The property name
399          * @param message  An exception message
400          */
401         public PropertyException(String property, String message) {
402             super("Invalid property \"" + property + "\": " + message);
403         }
404     }
