24  package oshi.util;

26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  /**
30   * General utility methods
31   */
32  public class Util {
33      private static final Logger LOG = LoggerFactory.getLogger(Util.class);
34  
35      private Util() {
36      }
37  
38      /**
39       * Sleeps for the specified number of milliseconds.
40       *
41       * @param ms
42       *            How long to sleep
43       */
44      public static void sleep(long ms) {
45          try {
46              LOG.trace("Sleeping for {} ms", ms);
47              Thread.sleep(ms);
48          } catch (InterruptedException e) { // NOSONAR squid:S2142
49              LOG.warn("Interrupted while sleeping for {} ms: {}", ms, e);
50              Thread.currentThread().interrupt();
51          }
52      }
53  
54      /**
55       * Tests if a String matches another String with a wildcard pattern.
56       *
57       * @param text
58       *            The String to test
59       * @param pattern
60       *            The String containing a wildcard pattern where ? represents a
61       *            single character and * represents any number of characters. If the
62       *            first character of the pattern is a carat (^) the test is
63       *            performed against the remaining characters and the result of the
64       *            test is the opposite.
65       * @return True if the String matches or if the first character is ^ and the
66       *         remainder of the String does not match.
67       */
68      public static boolean wildcardMatch(String text, String pattern) {
69          if (pattern.length() > 0 && pattern.charAt(0) == '^') {
70              return !wildcardMatch(text, pattern.substring(1));
71          }
72          return text.matches(pattern.replace("?", ".?").replace("*", ".*?"));
73      }
74  }
