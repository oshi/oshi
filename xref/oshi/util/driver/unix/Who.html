5   package oshi.util.driver.unix;
6   
7   import java.time.LocalDateTime;
8   import java.time.Year;
9   import java.time.ZoneId;
10  import java.time.format.DateTimeFormatter;
11  import java.time.format.DateTimeFormatterBuilder;
12  import java.time.format.DateTimeParseException;
13  import java.time.temporal.ChronoField;
14  import java.time.temporal.ChronoUnit;
15  import java.util.ArrayList;
16  import java.util.List;
17  import java.util.Locale;
18  import java.util.regex.Matcher;
19  import java.util.regex.Pattern;
20  
21  import oshi.annotation.concurrent.ThreadSafe;
22  import oshi.software.os.OSSession;
23  import oshi.util.ExecutingCommand;
24  
25  /**
26   * Utility to query logged in users using the {@code who} command with Unix date format parsing.
27   */
28  @ThreadSafe
29  public final class Who {
30  
31      // oshi ttys000 May 4 23:50 (192.168.1.23)
32      // middle 12 characters from Thu Nov 24 18:22:48 1986
33      private static final Pattern WHO_FORMAT_UNIX = Pattern
34              .compile("(\\S+)\\s+(\\S+)\\s+(\\S+)\\s+(\\d+)\\s+(\\d{2}:\\d{2})\\s*(?:\\((.+)\\))?");
35      private static final DateTimeFormatter WHO_DATE_FORMAT_UNIX = new DateTimeFormatterBuilder()
36              .appendPattern("MMM d HH:mm").parseDefaulting(ChronoField.YEAR, Year.now(ZoneId.systemDefault()).getValue())
37              .toFormatter(Locale.US);
38  
39      private Who() {
40      }
41  
42      /**
43       * Query {@code who} to get logged in users, parsing Unix date format.
44       *
45       * @return A list of logged in user sessions
46       */
47      public static synchronized List<OSSession> queryWho() {
48          List<OSSession> whoList = new ArrayList<>();
49          for (String s : ExecutingCommand.runNative("who")) {
50              matchUnix(whoList, s);
51          }
52          return whoList;
53      }
54  
55      /**
56       * Attempt to match Unix WHO format and add to the list.
57       *
58       * @param whoList the list to add to
59       * @param s       the string to match
60       * @return true if successful, false otherwise
61       */
62      public static boolean matchUnix(List<OSSession> whoList, String s) {
63          Matcher m = WHO_FORMAT_UNIX.matcher(s);
64          if (m.matches()) {
65              try {
66                  // Missing year, parse date time with current year
67                  LocalDateTime login = LocalDateTime.parse(m.group(3) + " " + m.group(4) + " " + m.group(5),
68                          WHO_DATE_FORMAT_UNIX);
69                  // If this date is in the future, subtract a year
70                  if (login.isAfter(LocalDateTime.now(ZoneId.systemDefault()))) {
71                      login = login.minus(1, ChronoUnit.YEARS);
72                  }
73                  long millis = login.atZone(ZoneId.systemDefault()).toInstant().toEpochMilli();
74                  whoList.add(new OSSession(m.group(1), m.group(2), millis, m.group(6) == null ? "" : m.group(6)));
75                  return true;
76              } catch (DateTimeParseException | NullPointerException e) {
77                  // shouldn't happen if regex matches and OS is producing sensible dates
78              }
79          }
80          return false;
81      }
82  }
