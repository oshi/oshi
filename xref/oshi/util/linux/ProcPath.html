5   package oshi.util.linux;
6   
7   import java.io.File;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.util.GlobalConfig;
11  
12  /**
13   * Provides constants for paths in the {@code /proc} filesystem on Linux.
14   * <p>
15   * If the user desires to configure a custom {@code /proc} path, it must be declared in the OSHI configuration file or
16   * updated in the {@link GlobalConfig} class prior to initializing this class.
17   */
18  @ThreadSafe
19  public final class ProcPath {
20  
21      /**
22       * The /proc filesystem location.
23       */
24      public static final String PROC = queryProcConfig();
25  
26      /** Path to asound. */
27      public static final String ASOUND = PROC + "/asound/";
28      /** Path to auxv. */
29      public static final String AUXV = PROC + "/self/auxv";
30      /** Path to cpuinfo. */
31      public static final String CPUINFO = PROC + "/cpuinfo";
32      /** Path to diskstats. */
33      public static final String DISKSTATS = PROC + "/diskstats";
34      /** Path to loadavg. */
35      public static final String LOADAVG = PROC + "/loadavg";
36      /** Path to meminfo. */
37      public static final String MEMINFO = PROC + "/meminfo";
38      /** Path to model. */
39      public static final String MODEL = PROC + "/device-tree/model";
40      /** Path to mounts. */
41      public static final String MOUNTS = PROC + "/mounts";
42      /** Path to net. */
43      public static final String NET = PROC + "/net";
44      /** Path to /proc/[pid]/cmdline. */
45      public static final String PID_CMDLINE = PROC + "/%d/cmdline";
46      /** Path to /proc/[pid]/cwd. */
47      public static final String PID_CWD = PROC + "/%d/cwd";
48      /** Path to /proc/[pid]/exe. */
49      public static final String PID_EXE = PROC + "/%d/exe";
50      /** Path to /proc/[pid]/environ. */
51      public static final String PID_ENVIRON = PROC + "/%d/environ";
52      /** Path to /proc/[pid]/fd. */
53      public static final String PID_FD = PROC + "/%d/fd";
54      /** Path to /proc/[pid]/io. */
55      public static final String PID_IO = PROC + "/%d/io";
56      /** Path to /proc/[pid]/stat. */
57      public static final String PID_STAT = PROC + "/%d/stat";
58      /** Path to /proc/[pid]/statm. */
59      public static final String PID_STATM = PROC + "/%d/statm";
60      /** Path to /proc/[pid]/status. */
61      public static final String PID_STATUS = PROC + "/%d/status";
62      /** Path to /proc/self/stat. */
63      public static final String SELF_STAT = PROC + "/self/stat";
64      /** Path to snmp. */
65      public static final String SNMP = NET + "/snmp";
66      /** Path to snmp6. */
67      public static final String SNMP6 = NET + "/snmp6";
68      /** Path to stat. */
69      public static final String STAT = PROC + "/stat";
70      /** Path to /proc/sys/fs/file nr. */
71      public static final String SYS_FS_FILE_NR = PROC + "/sys/fs/file-nr";
72      /** Path to /proc/sys/fs/file max. */
73      public static final String SYS_FS_FILE_MAX = PROC + "/sys/fs/file-max";
74      /** Path to /proc/[pid]/task/path. */
75      public static final String TASK_PATH = PROC + "/%d/task";
76      /** Path to /proc/[pid]/task/comm. */
77      public static final String TASK_COMM = TASK_PATH + "/%d/comm";
78      /** Path to /proc/[pid]/task/status. */
79      public static final String TASK_STATUS = TASK_PATH + "/%d/status";
80      /** Path to /proc/[pid]/task/stat. */
81      public static final String TASK_STAT = TASK_PATH + "/%d/stat";
82      /** Path to thread self. */
83      public static final String THREAD_SELF = PROC + "/thread-self";
84      /** Path to uptime. */
85      public static final String UPTIME = PROC + "/uptime";
86      /** Path to version. */
87      public static final String VERSION = PROC + "/version";
88      /** Path to vmstat. */
89      public static final String VMSTAT = PROC + "/vmstat";
90      /** Path to /proc/self/cgroup. */
91      public static final String SELF_CGROUP = PROC + "/self/cgroup";
92      /** Path to filesystems. */
93      public static final String FILESYSTEMS = PROC + "/filesystems";
94  
95      private ProcPath() {
96      }
97  
98      private static String queryProcConfig() {
99          String procPath = GlobalConfig.get(GlobalConfig.OSHI_UTIL_PROC_PATH, "/proc");
100         // Ensure prefix begins with path separator, but doesn't end with one
101         procPath = '/' + procPath.replaceAll("/$|^/", "");
102         if (!new File(procPath).exists()) {
103             throw new GlobalConfig.PropertyException(GlobalConfig.OSHI_UTIL_PROC_PATH, "The path does not exist");
104         }
105         return procPath;
106     }
107 }
