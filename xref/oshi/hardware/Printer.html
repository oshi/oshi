5   package oshi.hardware;
6   
7   import oshi.annotation.concurrent.Immutable;
8   
9   /**
10   * Printer interface representing a printer device.
11   */
12  @Immutable
13  public interface Printer {
14  
15      /**
16       * Retrieves the name of the printer.
17       *
18       * @return The printer name.
19       */
20      String getName();
21  
22      /**
23       * Retrieves the driver name or make/model of the printer.
24       *
25       * @return The driver or model name.
26       */
27      String getDriverName();
28  
29      /**
30       * Retrieves the user-friendly description of the printer.
31       *
32       * @return The printer description.
33       */
34      String getDescription();
35  
36      /**
37       * Retrieves the current status of the printer.
38       *
39       * @return The printer status.
40       */
41      PrinterStatus getStatus();
42  
43      /**
44       * Retrieves the reason for the current printer status, if available.
45       *
46       * @return A string describing the status reason (e.g., "Paper Jam", "media-empty"), or empty string if none.
47       */
48      String getStatusReason();
49  
50      /**
51       * Indicates whether this is the default printer.
52       *
53       * @return {@code true} if this is the default printer, {@code false} otherwise.
54       */
55      boolean isDefault();
56  
57      /**
58       * Indicates whether this is a local printer (as opposed to a network printer).
59       *
60       * @return {@code true} if this is a local printer, {@code false} if it is a network printer.
61       */
62      boolean isLocal();
63  
64      /**
65       * Retrieves the port name or device URI of the printer.
66       *
67       * @return The port name or URI.
68       */
69      String getPortName();
70  
71      /**
72       * Printer status enumeration.
73       */
74      enum PrinterStatus {
75          IDLE, PRINTING, ERROR, OFFLINE, UNKNOWN
76      }
77  }
