25  /**
26   * A CentralProcessor for NetBSD that uses JNA native sysctl calls when the JNA native library is available, falling
27   * back to the command-line {@link NetBsdCentralProcessor} implementation otherwise.
28   */
29  @ThreadSafe
30  public class NetBsdCentralProcessorJNA extends NetBsdCentralProcessor {
31  
32      @Override
33      protected long[] querySystemCpuLoadTicks() {
34          if (!NetBsdSysctlUtil.JNA_AVAILABLE) {
35              return super.querySystemCpuLoadTicks();
36          }
37          long[] ticks = new long[TickType.values().length];
38          int[] mib = { CTL_KERN, KERN_CP_TIME };
39          try (Memory m = NetBsdSysctlUtil.sysctl(mib)) {
40              if (m != null) {
41                  long[] cpuTicks = cpTimeToTicks(m);
42                  if (cpuTicks.length >= CPUSTATES) {
43                      ticks[TickType.USER.getIndex()] = cpuTicks[CP_USER];
44                      ticks[TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
45                      ticks[TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
46                      ticks[TickType.IRQ.getIndex()] = cpuTicks[CP_INTR];
47                      ticks[TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE];
48                  }
49              }
50          }
51          return ticks;
52      }
53  
54      @Override
55      protected long[][] queryProcessorCpuLoadTicks() {
56          if (!NetBsdSysctlUtil.JNA_AVAILABLE) {
57              return super.queryProcessorCpuLoadTicks();
58          }
59          long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
60          // On NetBSD, kern.cp_time with a third MIB element gives per-CPU data
61          int[] mib = { CTL_KERN, KERN_CP_TIME, 0 };
62          for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
63              mib[2] = cpu;
64              try (Memory m = NetBsdSysctlUtil.sysctl(mib)) {
65                  if (m != null) {
66                      long[] cpuTicks = cpTimeToTicks(m);
67                      if (cpuTicks.length >= CPUSTATES) {
68                          ticks[cpu][TickType.USER.getIndex()] = cpuTicks[CP_USER];
69                          ticks[cpu][TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
70                          ticks[cpu][TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
71                          ticks[cpu][TickType.IRQ.getIndex()] = cpuTicks[CP_INTR];
72                          ticks[cpu][TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE];
73                      }
74                  }
75              }
76          }
77          return ticks;
78      }
79  
80      /**
81       * Parse memory buffer returned from sysctl kern.cp_time to an array of longs representing CPU states.
82       *
83       * @param m A buffer containing the array of 64-bit values.
84       * @return The array
85       */
86      private static long[] cpTimeToTicks(Memory m) {
87          int arraySize = (int) (m.size() / 8L);
88          return m.getLongArray(0, arraySize);
89      }
90  
91      @Override
92      public double[] getSystemLoadAverage(int nelem) {
93          if (!NetBsdSysctlUtil.JNA_AVAILABLE) {
94              return super.getSystemLoadAverage(nelem);
95          }
96          if (nelem < 1 || nelem > 3) {
97              throw new IllegalArgumentException("Must include from one to three elements.");
98          }
99          double[] average = new double[nelem];
100         int retval = NetBsdLibc.INSTANCE.getloadavg(average, nelem);
101         if (retval < nelem) {
102             Arrays.fill(average, -1d);
103         }
104         return average;
105     }
106 }
