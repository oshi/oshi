7   import java.util.ArrayList;
8   import java.util.List;
9   import java.util.function.Supplier;
10  
11  import oshi.annotation.concurrent.Immutable;
12  import oshi.hardware.GraphicsCard;
13  import oshi.hardware.common.AbstractGraphicsCard;
14  import oshi.util.Constants;
15  import oshi.util.ParseUtil;
16  import oshi.util.Util;
17  
18  /**
19   * Graphics Card info obtained from lscfg
20   */
21  @Immutable
22  final class AixGraphicsCard extends AbstractGraphicsCard {
23  
24      /**
25       * Constructor for AixGraphicsCard
26       *
27       * @param name        The name
28       * @param deviceId    The device ID
29       * @param vendor      The vendor
30       * @param versionInfo The version info
31       * @param vram        The VRAM
32       */
33      AixGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
34          super(name, deviceId, vendor, versionInfo, vram);
35      }
36  
37      /**
38       * Gets graphics cards
39       *
40       * @param lscfg A memoized lscfg list
41       *
42       * @return List of graphics cards
43       */
44      public static List<GraphicsCard> getGraphicsCards(Supplier<List<String>> lscfg) {
45          List<GraphicsCard> cardList = new ArrayList<>();
46          boolean display = false;
47          String name = null;
48          String vendor = null;
49          List<String> versionInfo = new ArrayList<>();
50          for (String line : lscfg.get()) {
51              String s = line.trim();
52              if (s.startsWith("Name:") && s.contains("display")) {
53                  display = true;
54              } else if (display && s.toLowerCase().contains("graphics")) {
55                  name = s;
56              } else if (display && name != null) {
57                  if (s.startsWith("Manufacture ID")) {
58                      vendor = ParseUtil.removeLeadingDots(s.substring(14));
59                  } else if (s.contains("Level")) {
60                      versionInfo.add(s.replaceAll("\\.\\.+", "="));
61                  } else if (s.startsWith("Hardware Location Code")) {
62                      cardList.add(new AixGraphicsCard(name, Constants.UNKNOWN,
63                              Util.isBlank(vendor) ? Constants.UNKNOWN : vendor,
64                              versionInfo.isEmpty() ? Constants.UNKNOWN : String.join(",", versionInfo), 0L));
65                      display = false;
66                  }
67              }
68          }
69          return cardList;
70      }
71  }
