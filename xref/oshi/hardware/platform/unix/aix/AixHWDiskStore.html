47  /**
48   * AIX hard disk implementation.
49   */
50  @ThreadSafe
51  public final class AixHWDiskStore extends AbstractHWDiskStore {
52  
53      private final Supplier<perfstat_disk_t[]> diskStats;
54  
55      private long reads = 0L;
56      private long readBytes = 0L;
57      private long writes = 0L;
58      private long writeBytes = 0L;
59      private long currentQueueLength = 0L;
60      private long transferTime = 0L;
61      private long timeStamp = 0L;
62      private List<HWPartition> partitionList;
63  
64      private AixHWDiskStore(String name, String model, String serial, long size, Supplier<perfstat_disk_t[]> diskStats) {
65          super(name, model, serial, size);
66          this.diskStats = diskStats;
67      }
68  
69      @Override
70      public long getReads() {
71          return reads;
72      }
73  
74      @Override
75      public long getReadBytes() {
76          return readBytes;
77      }
78  
79      @Override
80      public long getWrites() {
81          return writes;
82      }
83  
84      @Override
85      public long getWriteBytes() {
86          return writeBytes;
87      }
88  
89      @Override
90      public long getCurrentQueueLength() {
91          return currentQueueLength;
92      }
93  
94      @Override
95      public long getTransferTime() {
96          return transferTime;
97      }
98  
99      @Override
100     public long getTimeStamp() {
101         return timeStamp;
102     }
103 
104     @Override
105     public List<HWPartition> getPartitions() {
106         return this.partitionList;
107     }
108 
109     @Override
110     public boolean updateAttributes() {
111         for (perfstat_disk_t stat : diskStats.get()) {
112             String name = Native.toString(stat.name);
113             if (name.equals(this.getName())) {
114                 // we only have total transfers so estimate read/write ratio from blocks
115                 long blks = stat.rblks + stat.wblks;
116                 this.reads = stat.xfers;
117                 if (blks > 0L) {
118                     this.writes = stat.xfers * stat.wblks / blks;
119                     this.reads -= this.writes;
120                 }
121                 this.readBytes = stat.rblks * stat.bsize;
122                 this.writeBytes = stat.wblks * stat.bsize;
123                 this.currentQueueLength = stat.qdepth;
124                 this.transferTime = stat.time;
125                 return true;
126             }
127         }
128         return false;
129     }
130 
131     /**
132      * Gets the disks on this machine
133      *
134      * @param diskStats
135      *            Memoized supplier of disk statistics
136      *
137      * @return a list of {@link HWDiskStore} objects representing the disks
138      */
139     public static List<HWDiskStore> getDisks(Supplier<perfstat_disk_t[]> diskStats) {
140         Map<String, Pair<Integer, Integer>> majMinMap = Ls.queryDeviceMajorMinor();
141         List<AixHWDiskStore> storeList = new ArrayList<>();
142         for (perfstat_disk_t disk : diskStats.get()) {
143             String storeName = Native.toString(disk.name);
144             Pair<String, String> ms = Lscfg.queryModelSerial(storeName);
145             String model = ms.getA() == null ? Native.toString(disk.description) : ms.getA();
146             String serial = ms.getB() == null ? Constants.UNKNOWN : ms.getB();
147             storeList.add(createStore(storeName, model, serial, disk.size << 20, diskStats, majMinMap));
148         }
149         return storeList.stream()
150                 .sorted(Comparator.comparingInt(
151                         s -> s.getPartitions().isEmpty() ? Integer.MAX_VALUE : s.getPartitions().get(0).getMajor()))
152                 .collect(Collectors.toList());
153     }
154 
155     private static AixHWDiskStore createStore(String diskName, String model, String serial, long size,
156             Supplier<perfstat_disk_t[]> diskStats, Map<String, Pair<Integer, Integer>> majMinMap) {
157         AixHWDiskStore store = new AixHWDiskStore(diskName, model.isEmpty() ? Constants.UNKNOWN : model, serial, size,
158                 diskStats);
159         store.partitionList = Collections.unmodifiableList(Lspv.queryLogicalVolumes(diskName, majMinMap).stream()
160                 .sorted(Comparator.comparing(HWPartition::getMinor).thenComparing(HWPartition::getName))
161                 .collect(Collectors.toList()));
162         store.updateAttributes();
163         return store;
164     }
165 }
