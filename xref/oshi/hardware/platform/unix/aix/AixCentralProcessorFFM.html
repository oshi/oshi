5   package oshi.hardware.platform.unix.aix;
6   
7   import static oshi.util.Memoizer.defaultExpiration;
8   import static oshi.util.Memoizer.memoize;
9   
10  import java.util.function.Supplier;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.driver.unix.aix.perfstat.PerfstatConfigFFM;
14  import oshi.driver.unix.aix.perfstat.PerfstatCpuFFM;
15  import oshi.hardware.common.platform.unix.aix.AixCentralProcessor;
16  
17  /**
18   * FFM-backed AIX CentralProcessor.
19   */
20  @ThreadSafe
21  final class AixCentralProcessorFFM extends AixCentralProcessor {
22  
23      private final Supplier<PerfstatCpuFFM.CpuTotal> cpuTotal = memoize(PerfstatCpuFFM::queryCpuTotal,
24              defaultExpiration());
25      private final Supplier<PerfstatCpuFFM.Cpu[]> cpuProc = memoize(PerfstatCpuFFM::queryCpu, defaultExpiration());
26  
27      @Override
28      protected PartitionInfo queryPartitionInfo() {
29          PerfstatConfigFFM.PartitionConfig config = PerfstatConfigFFM.queryConfig();
30          PartitionInfo info = new PartitionInfo();
31          info.vcpusMax = config.vcpusMax;
32          info.smtthreads = config.smtthreads;
33          info.machineID = config.machineID;
34          info.processorMHz = config.processorMHz;
35          return info;
36      }
37  
38      @Override
39      protected CpuTotalRow queryCpuTotal() {
40          PerfstatCpuFFM.CpuTotal p = cpuTotal.get();
41          CpuTotalRow row = new CpuTotalRow();
42          row.ncpus = p.ncpus;
43          row.processorHZ = p.processorHZ;
44          row.user = p.user;
45          row.sys = p.sys;
46          row.idle = p.idle;
47          row.wait = p.wait;
48          row.pswitch = p.pswitch;
49          row.devintrs = p.devintrs;
50          row.softintrs = p.softintrs;
51          row.idle_stolen_purr = p.idle_stolen_purr;
52          row.busy_stolen_purr = p.busy_stolen_purr;
53          row.loadavg[0] = p.loadavg[0];
54          row.loadavg[1] = p.loadavg[1];
55          row.loadavg[2] = p.loadavg[2];
56          return row;
57      }
58  
59      @Override
60      protected CpuTickRow[] queryPerCpuTicks() {
61          PerfstatCpuFFM.Cpu[] cpus = cpuProc.get();
62          CpuTickRow[] rows = new CpuTickRow[cpus.length];
63          for (int i = 0; i < cpus.length; i++) {
64              CpuTickRow row = new CpuTickRow();
65              row.user = cpus[i].user;
66              row.sys = cpus[i].sys;
67              row.idle = cpus[i].idle;
68              row.wait = cpus[i].wait;
69              row.devintrs = cpus[i].devintrs;
70              row.softintrs = cpus[i].softintrs;
71              row.idle_stolen_purr = cpus[i].idle_stolen_purr;
72              row.busy_stolen_purr = cpus[i].busy_stolen_purr;
73              rows[i] = row;
74          }
75          return rows;
76      }
77  }
