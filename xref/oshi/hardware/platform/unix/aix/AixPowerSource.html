24  package oshi.hardware.platform.unix.aix;
25  
26  import java.time.LocalDate;
27  import java.util.Collections;
28  import java.util.List;
29  
30  import oshi.annotation.concurrent.ThreadSafe;
31  import oshi.hardware.PowerSource;
32  import oshi.hardware.common.AbstractPowerSource;
33  
34  /**
35   * A Power Source
36   */
37  @ThreadSafe
38  public final class AixPowerSource extends AbstractPowerSource {
39  
40      public AixPowerSource(String name, String deviceName, double remainingCapacityPercent,
41              double timeRemainingEstimated, double timeRemainingInstant, double powerUsageRate, double voltage,
42              double amperage, boolean powerOnLine, boolean charging, boolean discharging, CapacityUnits capacityUnits,
43              int currentCapacity, int maxCapacity, int designCapacity, int cycleCount, String chemistry,
44              LocalDate manufactureDate, String manufacturer, String serialNumber, double temperature) {
45          super(name, deviceName, remainingCapacityPercent, timeRemainingEstimated, timeRemainingInstant, powerUsageRate,
46                  voltage, amperage, powerOnLine, charging, discharging, capacityUnits, currentCapacity, maxCapacity,
47                  designCapacity, cycleCount, chemistry, manufactureDate, manufacturer, serialNumber, temperature);
48      }
49  
50      /**
51       * Gets Battery Information. AIX does not provide any battery statistics, as
52       * most servers are not designed to be run on battery.
53       *
54       * @return An empty list.
55       */
56      public static List<PowerSource> getPowerSources() {
57          return Collections.emptyList();
58      }
59  }
