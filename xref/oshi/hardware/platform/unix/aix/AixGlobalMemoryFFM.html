5   package oshi.hardware.platform.unix.aix;
6   
7   import static oshi.util.Memoizer.defaultExpiration;
8   import static oshi.util.Memoizer.memoize;
9   
10  import java.util.List;
11  import java.util.function.Supplier;
12  
13  import oshi.annotation.concurrent.ThreadSafe;
14  import oshi.driver.unix.aix.perfstat.PerfstatMemoryFFM;
15  import oshi.hardware.VirtualMemory;
16  import oshi.hardware.common.platform.unix.aix.AixGlobalMemory;
17  
18  /**
19   * FFM-backed AIX GlobalMemory.
20   */
21  @ThreadSafe
22  final class AixGlobalMemoryFFM extends AixGlobalMemory {
23  
24      private final Supplier<PerfstatMemoryFFM.MemoryTotal> perfstatMem = memoize(PerfstatMemoryFFM::queryMemoryTotal,
25              defaultExpiration());
26      private final Supplier<VirtualMemory> vm = memoize(this::createVirtualMemory);
27  
28      AixGlobalMemoryFFM(Supplier<List<String>> lscfg) {
29          super(lscfg);
30      }
31  
32      @Override
33      public VirtualMemory getVirtualMemory() {
34          return vm.get();
35      }
36  
37      @Override
38      protected long queryRealAvail() {
39          return perfstatMem.get().real_avail;
40      }
41  
42      @Override
43      protected long queryRealTotal() {
44          return perfstatMem.get().real_total;
45      }
46  
47      private VirtualMemory createVirtualMemory() {
48          return new AixVirtualMemoryFFM(perfstatMem);
49      }
50  }
