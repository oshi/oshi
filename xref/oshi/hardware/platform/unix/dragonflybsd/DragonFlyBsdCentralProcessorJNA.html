5   package oshi.hardware.platform.unix.dragonflybsd;

7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   
10  import com.sun.jna.Memory;
11  import com.sun.jna.Native;
12  import com.sun.jna.platform.unix.LibCAPI.size_t;
13  
14  import oshi.hardware.CentralProcessor;
15  import oshi.hardware.platform.unix.freebsd.FreeBsdCentralProcessorJNA;
16  import oshi.jna.ByRef.CloseableSizeTByReference;
17  import oshi.jna.platform.unix.DragonFlyBsdLibc;
18  import oshi.jna.platform.unix.FreeBsdLibc;
19  
20  /**
21   * DragonFly BSD Central Processor. Overrides CPU tick retrieval to use DragonFly's kern.cputime sysctl via
22   * {@link DragonFlyBsdLibc#INSTANCE}.
23   */
24  public class DragonFlyBsdCentralProcessorJNA extends FreeBsdCentralProcessorJNA {
25  
26      private static final Logger LOG = LoggerFactory.getLogger(DragonFlyBsdCentralProcessorJNA.class);
27  
28      @Override
29      public long[] querySystemCpuLoadTicks() {
30          long[] ticks = new long[TickType.values().length];
31          try (FreeBsdLibc.CpTime cpTime = new FreeBsdLibc.CpTime();
32                  CloseableSizeTByReference size = new CloseableSizeTByReference(cpTime.size())) {
33              if (0 == DragonFlyBsdLibc.INSTANCE.sysctlbyname("kern.cp_time", cpTime.getPointer(), size, null,
34                      size_t.ZERO)) {
35                  cpTime.read();
36                  ticks[TickType.USER.getIndex()] = cpTime.cpu_ticks[FreeBsdLibc.CP_USER];
37                  ticks[TickType.NICE.getIndex()] = cpTime.cpu_ticks[FreeBsdLibc.CP_NICE];
38                  ticks[TickType.SYSTEM.getIndex()] = cpTime.cpu_ticks[FreeBsdLibc.CP_SYS];
39                  ticks[TickType.IRQ.getIndex()] = cpTime.cpu_ticks[FreeBsdLibc.CP_INTR];
40                  ticks[TickType.IDLE.getIndex()] = cpTime.cpu_ticks[FreeBsdLibc.CP_IDLE];
41              }
42          }
43          return ticks;
44      }
45  
46      @Override
47      public long[][] queryProcessorCpuLoadTicks() {
48          long[][] ticks = new long[getLogicalProcessorCount()][CentralProcessor.TickType.values().length];
49          String name = "kern.cputime";
50          // Two-step sysctl: first get size, then fetch data
51          try (CloseableSizeTByReference size = new CloseableSizeTByReference()) {
52              if (0 != DragonFlyBsdLibc.INSTANCE.sysctlbyname(name, null, size, null, size_t.ZERO)) {
53                  LOG.debug("Failed sysctl size query: {}, Error code: {}", name, Native.getLastError());
54                  return ticks;
55              }
56              long arraySize = size.getValue().longValue();
57              if (arraySize <= 0) {
58                  return ticks;
59              }
60              long structSize = arraySize / getLogicalProcessorCount();
61              try (Memory m = new Memory(arraySize);
62                      CloseableSizeTByReference dataSize = new CloseableSizeTByReference(arraySize)) {
63                  if (0 != DragonFlyBsdLibc.INSTANCE.sysctlbyname(name, m, dataSize, null, size_t.ZERO)) {
64                      LOG.debug("Failed sysctl data query: {}, Error code: {}", name, Native.getLastError());
65                      return ticks;
66                  }
67                  for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
68                      ticks[cpu][TickType.USER.getIndex()] = m
69                              .getLong(structSize * cpu + FreeBsdLibc.CP_USER * FreeBsdLibc.UINT64_SIZE);
70                      ticks[cpu][TickType.NICE.getIndex()] = m
71                              .getLong(structSize * cpu + FreeBsdLibc.CP_NICE * FreeBsdLibc.UINT64_SIZE);
72                      ticks[cpu][TickType.SYSTEM.getIndex()] = m
73                              .getLong(structSize * cpu + FreeBsdLibc.CP_SYS * FreeBsdLibc.UINT64_SIZE);
74                      ticks[cpu][TickType.IRQ.getIndex()] = m
75                              .getLong(structSize * cpu + FreeBsdLibc.CP_INTR * FreeBsdLibc.UINT64_SIZE);
76                      ticks[cpu][TickType.IDLE.getIndex()] = m
77                              .getLong(structSize * cpu + FreeBsdLibc.CP_IDLE * FreeBsdLibc.UINT64_SIZE);
78                  }
79              }
80          }
81          return ticks;
82      }
83  }
