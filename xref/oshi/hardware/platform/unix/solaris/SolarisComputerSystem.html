80      private static SmbiosStrings readSmbios() {
81          String biosVendor = null;
82          String biosVersion = null;
83          String biosDate = null;
84  
85          String manufacturer = null;
86          String model = null;
87          String serialNumber = null;
88          String uuid = null;
89  
90          String boardManufacturer = null;
91          String boardModel = null;
92          String boardVersion = null;
93          String boardSerialNumber = null;
94  
95          // $ smbios
96          // ID SIZE TYPE
97          // 0 87 SMB_TYPE_BIOS (BIOS Information)
98          //
99          // Vendor: Parallels Software International Inc.
100         // Version String: 11.2.1 (32686)
101         // Release Date: 07/15/2016
102         // Address Segment: 0xf000
103         // ... <snip> ...
104         //
105         // ID SIZE TYPE
106         // 1 177 SMB_TYPE_SYSTEM (system information)
107         //
108         // Manufacturer: Parallels Software International Inc.
109         // Product: Parallels Virtual Platforom
110         // Version: None
111         // Serial Number: Parallels-45 2E 7E 2D 57 5C 4B 59 B1 30 28 81 B7 81 89
112         // 34
113         //
114         // UUID: 452e7e2d-575c04b59-b130-2881b7818934
115         // Wake-up Event: 0x6 (Power Switch)
116         // SKU Number: Undefined
117         // Family: Parallels VM
118         //
119         // ID SIZE TYPE
120         // 2 90 SMB_TYPE_BASEBOARD (base board)
121         //
122         // Manufacturer: Parallels Software International Inc.
123         // Product: Parallels Virtual Platform
124         // Version: None
125         // Serial Number: None
126         // ... <snip> ...
127         //
128         // ID SIZE TYPE
129         // 3 .... <snip> ...
130 
131         final String vendorMarker = "Vendor:";
132         final String biosDateMarker = "Release Date:";
133         final String biosVersionMarker = "VersionString:";
134 
135         final String manufacturerMarker = "Manufacturer:";
136         final String productMarker = "Product:";
137         final String serialNumMarker = "Serial Number:";
138         final String uuidMarker = "UUID:";
139         final String versionMarker = "Version:";
140 
141         int smbTypeId = -1;
142         // Only works with root permissions but it's all we've got
143         for (final String checkLine : ExecutingCommand.runNative("smbios")) {
144             // Change the smbTypeId when hitting a new header
145             if (checkLine.contains("SMB_TYPE_") && (smbTypeId = getSmbType(checkLine)) == Integer.MAX_VALUE) {
146                 // If we get past what we need, stop iterating
147                 break;
148             }
149             // Based on the smbTypeID we are processing for
150             switch (smbTypeId) {
151             case 0: // BIOS
152                 if (checkLine.contains(vendorMarker)) {
153                     biosVendor = checkLine.split(vendorMarker)[1].trim();
154                 } else if (checkLine.contains(biosVersionMarker)) {
155                     biosVersion = checkLine.split(biosVersionMarker)[1].trim();
156                 } else if (checkLine.contains(biosDateMarker)) {
157                     biosDate = checkLine.split(biosDateMarker)[1].trim();
158                 }
159                 break;
160             case 1: // SYSTEM
161                 if (checkLine.contains(manufacturerMarker)) {
162                     manufacturer = checkLine.split(manufacturerMarker)[1].trim();
163                 } else if (checkLine.contains(productMarker)) {
164                     model = checkLine.split(productMarker)[1].trim();
165                 } else if (checkLine.contains(serialNumMarker)) {
166                     serialNumber = checkLine.split(serialNumMarker)[1].trim();
167                 } else if (checkLine.contains(uuidMarker)) {
168                     uuid = checkLine.split(uuidMarker)[1].trim();
169                 }
170                 break;
171             case 2: // BASEBOARD
172                 if (checkLine.contains(manufacturerMarker)) {
173                     boardManufacturer = checkLine.split(manufacturerMarker)[1].trim();
174                 } else if (checkLine.contains(productMarker)) {
175                     boardModel = checkLine.split(productMarker)[1].trim();
176                 } else if (checkLine.contains(versionMarker)) {
177                     boardVersion = checkLine.split(versionMarker)[1].trim();
178                 } else if (checkLine.contains(serialNumMarker)) {
179                     boardSerialNumber = checkLine.split(serialNumMarker)[1].trim();
180                 }
181                 break;
182             default:
183                 break;
184             }
185         }
186         // If we get to end and haven't assigned, use fallback
187         if (Util.isBlank(serialNumber)) {
188             serialNumber = readSerialNumber();
189         }
190         return new SmbiosStrings(biosVendor, biosVersion, biosDate, manufacturer, model, serialNumber, uuid,
191                 boardManufacturer, boardModel, boardVersion, boardSerialNumber);
192     }
193 
194     private static int getSmbType(String checkLine) {
195         if (checkLine.contains("SMB_TYPE_BIOS")) {
196             return 0; // BIOS
197         } else if (checkLine.contains("SMB_TYPE_SYSTEM")) {
198             return 1; // SYSTEM
199         } else if (checkLine.contains("SMB_TYPE_BASEBOARD")) {
200             return 2; // BASEBOARD
201         } else {
202             // First 3 SMB_TYPEs are what we need. After that no need to
203             // continue processing the output
204             return Integer.MAX_VALUE;
205         }
206     }
207 
208     private static String readSerialNumber() {
209         // If they've installed STB (Sun Explorer) this should work
210         String serialNumber = ExecutingCommand.getFirstAnswer("sneep");
211         // if that didn't work, try...
212         if (serialNumber.isEmpty()) {
213             String marker = "chassis-sn:";
214             for (String checkLine : ExecutingCommand.runNative("prtconf -pv")) {
215                 if (checkLine.contains(marker)) {
216                     serialNumber = ParseUtil.getSingleQuoteStringValue(checkLine);
217                     break;
218                 }
219             }
220         }
221         return serialNumber;
222     }

224     private static final class SmbiosStrings {
225         private final String biosVendor;
226         private final String biosVersion;
227         private final String biosDate;
228 
229         private final String manufacturer;
230         private final String model;
231         private final String serialNumber;
232         private final String uuid;
233 
234         private final String boardManufacturer;
235         private final String boardModel;
236         private final String boardVersion;
237         private final String boardSerialNumber;
238 
239         private SmbiosStrings(String biosVendor, String biosVersion, String biosDate, //
240                 String manufacturer, String model, String serialNumber, String uuid, //
241                 String boardManufacturer, String boardModel, String boardVersion, String boardSerialNumber) {
242             this.biosVendor = Util.isBlank(biosVendor) ? Constants.UNKNOWN : biosVendor;
243             this.biosVersion = Util.isBlank(biosVersion) ? Constants.UNKNOWN : biosVersion;
244             this.biosDate = Util.isBlank(biosDate) ? Constants.UNKNOWN : biosDate;
245 
246             this.manufacturer = Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer;
247             this.model = Util.isBlank(model) ? Constants.UNKNOWN : model;
248             this.serialNumber = Util.isBlank(serialNumber) ? Constants.UNKNOWN : serialNumber;
249             this.uuid = Util.isBlank(uuid) ? Constants.UNKNOWN : uuid;
250 
251             this.boardManufacturer = Util.isBlank(boardManufacturer) ? Constants.UNKNOWN : boardManufacturer;
252             this.boardModel = Util.isBlank(boardModel) ? Constants.UNKNOWN : boardModel;
253             this.boardVersion = Util.isBlank(boardVersion) ? Constants.UNKNOWN : boardVersion;
254             this.boardSerialNumber = Util.isBlank(boardSerialNumber) ? Constants.UNKNOWN : boardSerialNumber;
255         }
256     }
