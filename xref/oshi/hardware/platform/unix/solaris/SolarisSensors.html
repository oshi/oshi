19  package oshi.hardware.platform.unix.solaris;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import oshi.hardware.Sensors;
25  import oshi.util.ExecutingCommand;
26  import oshi.util.ParseUtil;
27  
28  public class SolarisSensors implements Sensors {
29  
30      private static final long serialVersionUID = 1L;
31  
32      /**
33       * {@inheritDoc}
34       */
35      @Override
36      public double getCpuTemperature() {
37          double maxTemp = 0d;
38          // Return max found temp
39          for (String line : ExecutingCommand.runNative("/usr/sbin/prtpicl -v -c temperature-sensor")) {
40              if (line.trim().startsWith("Temperature:")) {
41                  int temp = ParseUtil.parseLastInt(line, 0);
42                  if (temp > maxTemp) {
43                      maxTemp = temp;
44                  }
45              }
46          }
47          // If it's in millidegrees:
48          if (maxTemp > 1000) {
49              maxTemp /= 1000;
50          }
51          return maxTemp;
52      }
53  
54      /**
55       * {@inheritDoc}
56       */
57      @Override
58      public int[] getFanSpeeds() {
59          List<Integer> speedList = new ArrayList<>();
60          // Return max found temp
61          for (String line : ExecutingCommand.runNative("/usr/sbin/prtpicl -v -c fan")) {
62              if (line.trim().startsWith("Speed:")) {
63                  speedList.add(ParseUtil.parseLastInt(line, 0));
64              }
65          }
66          int[] fans = new int[speedList.size()];
67          for (int i = 0; i < speedList.size(); i++) {
68              fans[i] = speedList.get(i);
69          }
70          return fans;
71      }
72  
73      /**
74       * {@inheritDoc}
75       */
76      @Override
77      public double getCpuVoltage() {
78          double voltage = 0d;
79          // TODO This is entirely a guess!
80          for (String line : ExecutingCommand.runNative("/usr/sbin/prtpicl -v -c voltage-sensor")) {
81              if (line.trim().startsWith("Voltage:")) {
82                  voltage = ParseUtil.parseDoubleOrDefault(line.replace("Voltage:", "").trim(), 0d);
83                  break;
84              }
85          }
86          return voltage;
87      }
88  }
