24  package oshi.hardware.platform.unix.solaris;
25  
26  import com.sun.jna.platform.unix.solaris.LibKstat.Kstat; // NOSONAR
27  
28  import oshi.hardware.NetworkIF;
29  import oshi.hardware.common.AbstractNetworks;
30  import oshi.util.platform.unix.solaris.KstatUtil;
31  
32  /**
33   * @author widdis[at]gmail[dot]com
34   */
35  public class SolarisNetworks extends AbstractNetworks {
36  
37      private static final long serialVersionUID = 1L;
38  
39      /**
40       * Updates interface network statistics on the given interface. Statistics
41       * include packets and bytes sent and received, and interface speed.
42       *
43       * @param netIF
44       *            The interface on which to update statistics
45       */
46      public static void updateNetworkStats(NetworkIF netIF) {
47          Kstat ksp = KstatUtil.kstatLookup("link", -1, netIF.getName());
48          if (ksp == null) { // Solaris 10 compatibility
49              ksp = KstatUtil.kstatLookup(null, -1, netIF.getName());
50          }
51          if (ksp != null && KstatUtil.kstatRead(ksp)) {
52              netIF.setBytesSent(KstatUtil.kstatDataLookupLong(ksp, "obytes64"));
53              netIF.setBytesRecv(KstatUtil.kstatDataLookupLong(ksp, "rbytes64"));
54              netIF.setPacketsSent(KstatUtil.kstatDataLookupLong(ksp, "opackets64"));
55              netIF.setPacketsRecv(KstatUtil.kstatDataLookupLong(ksp, "ipackets64"));
56              netIF.setOutErrors(KstatUtil.kstatDataLookupLong(ksp, "oerrors"));
57              netIF.setInErrors(KstatUtil.kstatDataLookupLong(ksp, "ierrors"));
58              netIF.setSpeed(KstatUtil.kstatDataLookupLong(ksp, "ifspeed"));
59              // Snap time in ns; convert to ms
60              netIF.setTimeStamp(ksp.ks_snaptime / 1_000_000L);
61          }
62      }
63  }
