5   package oshi.hardware.platform.unix.solaris;
6   
7   import java.lang.foreign.MemorySegment;
8   import java.util.List;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.ffm.platform.unix.solaris.LibKstatFunctions;
12  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM;
13  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM.KstatChain;
14  import oshi.hardware.HWDiskStore;
15  import oshi.hardware.common.platform.unix.solaris.SolarisHWDiskStore;
16  
17  /**
18   * FFM-backed Solaris HWDiskStore. Uses the legacy {@code kstat} chain only; Kstat2 exists only on the JDK 17-capped
19   * latest Solaris, so FFM (JDK 25) never needs it.
20   */
21  @ThreadSafe
22  public final class SolarisHWDiskStoreFFM extends SolarisHWDiskStore {
23  
24      private SolarisHWDiskStoreFFM(String name, String model, String serial, long size) {
25          super(name, model, serial, size);
26      }
27  
28      /**
29       * Gets the disks on this machine.
30       *
31       * @return a list of {@link HWDiskStore} objects representing the disks
32       */
33      public static List<HWDiskStore> getDisks() {
34          return getDisks(SolarisHWDiskStoreFFM::new);
35      }
36  
37      @Override
38      protected DiskStats queryStats() {
39          try (KstatChain kc = KstatUtilFFM.openChain()) {
40              MemorySegment ksp = kc.lookup(null, 0, getName());
41              if (ksp.address() != 0L && kc.read(ksp)) {
42                  MemorySegment dataPtr = LibKstatFunctions.kstatData(ksp);
43                  if (dataPtr.address() == 0L) {
44                      return null;
45                  }
46                  MemorySegment io = dataPtr.reinterpret(LibKstatFunctions.KSTAT_IO_LAYOUT.byteSize());
47                  DiskStats stats = new DiskStats();
48                  stats.reads = LibKstatFunctions.kstatIoReads(io);
49                  stats.writes = LibKstatFunctions.kstatIoWrites(io);
50                  stats.readBytes = LibKstatFunctions.kstatIoNread(io);
51                  stats.writeBytes = LibKstatFunctions.kstatIoNwritten(io);
52                  stats.currentQueueLength = (long) LibKstatFunctions.kstatIoWcnt(io) + LibKstatFunctions.kstatIoRcnt(io);
53                  stats.transferTime = LibKstatFunctions.kstatIoRtime(io) / 1_000_000L;
54                  stats.timeStamp = LibKstatFunctions.kstatSnaptime(ksp) / 1_000_000L;
55                  return stats;
56              }
57          }
58          return null;
59      }
60  }
