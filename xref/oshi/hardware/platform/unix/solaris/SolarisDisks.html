52      /**
53       * Updates the statistics on a disk store.
54       *
55       * @param diskStore
56       *            the {@link oshi.hardware.HWDiskStore} to update.
57       * @return {@code true} if the update was (probably) successful.
58       */
59      public static boolean updateDiskStats(HWDiskStore diskStore) {
60          try (KstatChain kc = KstatUtil.openChain()) {
61              Kstat ksp = kc.lookup(null, 0, diskStore.getName());
62              if (ksp != null && kc.read(ksp)) {
63                  KstatIO data = new KstatIO(ksp.ks_data);
64                  diskStore.setReads(data.reads);
65                  diskStore.setWrites(data.writes);
66                  diskStore.setReadBytes(data.nread);
67                  diskStore.setWriteBytes(data.nwritten);
68                  diskStore.setCurrentQueueLength((long) data.wcnt + data.rcnt);
69                  // rtime and snaptime are nanoseconds, convert to millis
70                  diskStore.setTransferTime(data.rtime / 1_000_000L);
71                  diskStore.setTimeStamp(ksp.ks_snaptime / 1_000_000L);
72                  return true;
73              }
74          }
75          return false;
76      }

78      /**
79       * Gets the disks on this machine
80       *
81       * @return an array of {@link HWDiskStore} objects representing the disks
82       */
83      public static HWDiskStore[] getDisks() {
84          // Create map indexed by device name for multiple command reference
85          Map<String, HWDiskStore> diskMap = new HashMap<>();
86          // First, run iostat -er to enumerate disks by name. Sample output:
87          // errors
88          // device,s/w,h/w,trn,tot
89          // cmdk0,0,0,0,0
90          // sd0,0,0,0
91          List<String> disks = ExecutingCommand.runNative("iostat -er");
92  
93          // Create map to correlate disk name with block device mount point for
94          // later use in partition info
95          Map<String, String> deviceMap = new HashMap<>();
96          // Also run iostat -ern to get the same list by mount point. Sample
97          // output:
98          // errors
99          // s/w,h/w,trn,tot,device
100         // 0,0,0,0,c1d0
101         // 0,0,0,0,c1t1d0
102         List<String> mountpoints = ExecutingCommand.runNative("iostat -ern");
103         String disk;
104         for (int i = 0; i < disks.size() && i < mountpoints.size(); i++) {
105             // Map disk
106             disk = disks.get(i);
107             String[] diskSplit = disk.split(",");
108             if (diskSplit.length >= 5 && !"device".equals(diskSplit[0])) {
109                 HWDiskStorel#HWDiskStore">HWDiskStore store = new HWDiskStore();
110                 store.setName(diskSplit[0]);
111                 diskMap.put(diskSplit[0], store);
112                 // Map mount
113                 String mount = mountpoints.get(i);
114                 String[] mountSplit = mount.split(",");
115                 if (mountSplit.length >= 5 && !"device".equals(mountSplit[4])) {
116                     deviceMap.put(diskSplit[0], mountSplit[4]);
117                 }
118             }
119         }
120 
121         // Create map to correlate disk name with blick device mount point for
122         // later use in partition info
123         Map<String, Integer> majorMap = new HashMap<>();
124         // Run lshal, if available, to get block device major (we'll use
125         // partition # for minor)
126         List<String> lshal = ExecutingCommand.runNative("lshal");
127         disk = "";
128         for (String line : lshal) {
129             if (line.startsWith("udi ")) {
130                 String udi = ParseUtil.getSingleQuoteStringValue(line);
131                 disk = udi.substring(udi.lastIndexOf('/') + 1);
132                 continue;
133             }
134             line = line.trim();
135             if (line.startsWith("block.major")) {
136                 majorMap.put(disk, ParseUtil.getFirstIntValue(line));
137             }
138         }
139 
140         // Next, run iostat -Er to get model, etc.
141         disks = ExecutingCommand.runNative("iostat -Er");
142         // We'll use Model if available, otherwise Vendor+Product
143         disk = "";
144         String model = "";
145         String vendor = "";
146         String product = "";
147         String serial = "";
148         long size = 0;
149         for (String line : disks) {
150             // The -r switch enables comma delimited for easy parsing!
151             // No guarantees on which line the results appear so we'll nest
152             // a loop iterating on the comma splits
153             String[] split = line.split(",");
154             for (String keyValue : split) {
155                 keyValue = keyValue.trim();
156                 // If entry is tne name of a disk, this is beginning of new
157                 // output for that disk.
158                 if (diskMap.keySet().contains(keyValue)) {
159                     // First, if we have existing output from previous,
160                     // update
161                     if (!disk.isEmpty()) {
162                         updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
163                                 majorMap.getOrDefault(disk, 0));
164                     }
165                     // Reset values for next iteration
166                     disk = keyValue;
167                     model = "";
168                     vendor = "";
169                     product = "";
170                     serial = "";
171                     size = 0L;
172                     continue;
173                 }
174                 // Otherwise update variables
175                 if (keyValue.startsWith("Model:")) {
176                     model = keyValue.replace("Model:", "").trim();
177                 } else if (keyValue.startsWith("Serial No:")) {
178                     serial = keyValue.replace("Serial No:", "").trim();
179                 } else if (keyValue.startsWith("Vendor:")) {
180                     vendor = keyValue.replace("Vendor:", "").trim();
181                 } else if (keyValue.startsWith("Product:")) {
182                     product = keyValue.replace("Product:", "").trim();
183                 } else if (keyValue.startsWith("Size:")) {
184                     // Size: 1.23GB <1227563008 bytes>
185                     String[] bytes = keyValue.split("<");
186                     if (bytes.length > 1) {
187                         bytes = ParseUtil.whitespaces.split(bytes[1]);
188                         size = ParseUtil.parseLongOrDefault(bytes[0], 0L);
189                     }
190                 }
191             }
192             // At end of output update last entry
193             if (!disk.isEmpty()) {
194                 updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
195                         majorMap.getOrDefault(disk, 0));
196             }
197         }
198 
199         // Finally use kstat to get reads/writes
200         // simultaneously populate result array
201         HWDiskStoreDiskStore">HWDiskStore[] results = new HWDiskStore[diskMap.keySet().size()];
202         int index = 0;
203         for (Entry<String, HWDiskStore> entry : diskMap.entrySet()) {
204             updateDiskStats(entry.getValue());
205             results[index++] = entry.getValue();
206         }
207 
208         return results;
209     }

211     /**
212      * Updates the HWDiskStore. If model name is nonempty it is used, otherwise
213      * vendor+product are used for model
214      *
215      * @param store
216      *            A HWDiskStore
217      * @param model
218      *            model name, or empty string if none
219      * @param vendor
220      *            vendor name, or empty string if none
221      * @param product
222      *            product nmae, or empty string if none
223      * @param serial
224      *            serial number, or empty string if none
225      * @param size
226      *            size of the drive in bytes
227      * @param mount
228      *            The mount point of this store, used to fetch partition info
229      * @param major
230      *            The major device number for the partition
231      */
232     private static void updateStore(HWDiskStore store, String model, String vendor, String product, String serial,
233             long size, String mount, int major) {
234         store.setModel(model.isEmpty() ? (vendor + " " + product).trim() : model);
235         store.setSerial(serial);
236         store.setSize(size);
237 
238         // Temporary list to hold partitions
239         List<HWPartition> partList = new ArrayList<>();
240 
241         // Now grab prtvotc output for partitions
242         // This requires sudo permissions; will result in "permission denied
243         // otherwise" in which case we return empty partition list
244         List<String> prtvotc = ExecutingCommand.runNative("prtvtoc /dev/dsk/" + mount);
245         // Sample output - see man prtvtoc
246         if (prtvotc.size() > 1) {
247             int bytesPerSector = 0;
248             String[] split;
249             // We have a result, parse partition table
250             for (String line : prtvotc) {
251                 // If line starts with asterisk we ignore except for the one
252                 // specifying bytes per sector
253                 if (line.startsWith("*")) {
254                     if (line.endsWith("bytes/sector")) {
255                         split = ParseUtil.whitespaces.split(line);
256                         if (split.length > 0) {
257                             bytesPerSector = ParseUtil.parseIntOrDefault(split[1], 0);
258                         }
259                     }
260                 } else if (bytesPerSector > 0) {
261                     // If bytes/sector is still 0, these are not real partitions so
262                     // ignore.
263                     // Lines without asterisk have 6 or 7 whitespaces-split values
264                     // representing (last field optional):
265                     // Partition Tag Flags Sector Count Sector Mount
266                     split = ParseUtil.whitespaces.split(line.trim());
267                     // Partition 2 is always the whole disk so we ignore it
268                     if (split.length >= 6 && !"2".equals(split[0])) {
269                         HWPartitionPartition">HWPartition partition = new HWPartition();
270                         // First field is partition number
271                         partition.setIdentification(mount + "s" + split[0]);
272                         partition.setMajor(major);
273                         partition.setMinor(ParseUtil.parseIntOrDefault(split[0], 0));
274                         // Second field is tag. Parse:
275                         switch (ParseUtil.parseIntOrDefault(split[1], 0)) {
276                         case 0x01:
277                         case 0x18:
278                             partition.setName("boot");
279                             break;
280                         case 0x02:
281                             partition.setName("root");
282                             break;
283                         case 0x03:
284                             partition.setName("swap");
285                             break;
286                         case 0x04:
287                             partition.setName("usr");
288                             break;
289                         case 0x05:
290                             partition.setName("backup");
291                             break;
292                         case 0x06:
293                             partition.setName("stand");
294                             break;
295                         case 0x07:
296                             partition.setName("var");
297                             break;
298                         case 0x08:
299                             partition.setName("home");
300                             break;
301                         case 0x09:
302                             partition.setName("altsctr");
303                             break;
304                         case 0x0a:
305                             partition.setName("cache");
306                             break;
307                         case 0x0b:
308                             partition.setName("reserved");
309                             break;
310                         case 0x0c:
311                             partition.setName("system");
312                             break;
313                         case 0x0e:
314                             partition.setName("public region");
315                             break;
316                         case 0x0f:
317                             partition.setName("private region");
318                             break;
319                         default:
320                             partition.setName("unknown");
321                             break;
322                         }
323                         // Third field is flags.
324                         // First character writable, second is mountable
325                         switch (split[2]) {
326                         case "00":
327                             partition.setType("wm");
328                             break;
329                         case "10":
330                             partition.setType("rm");
331                             break;
332                         case "01":
333                             partition.setType("wu");
334                             break;
335                         default:
336                             partition.setType("ru");
337                             break;
338                         }
339                         // Fifth field is sector count
340                         partition.setSize(bytesPerSector * ParseUtil.parseLongOrDefault(split[4], 0L));
341                         // Seventh field (if present) is mount point
342                         if (split.length > 6) {
343                             partition.setMountPoint(split[6]);
344                         }
345                         partList.add(partition);
346                     }
347                 }
348             }
349             store.setPartitions(partList.toArray(new HWPartition[0]));
350         }
351     }
