24  /**
25   * SolarisNetworks class.
26   */
27  @ThreadSafe
28  public final class SolarisNetworkIF extends AbstractNetworkIF {
29  
30      private static final Logger LOG = LoggerFactory.getLogger(SolarisNetworkIF.class);
31  
32      public SolarisNetworkIF(NetworkInterface netint) throws InstantiationException {
33          super(netint);
34          updateAttributes();
35      }
36  
37      /**
38       * Gets all network interfaces on this machine
39       *
40       * @param includeLocalInterfaces include local interfaces in the result
41       * @return A list of {@link NetworkIF} objects representing the interfaces
42       */
43      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
44          List<NetworkIF> ifList = new ArrayList<>();
45          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
46              try {
47                  ifList.add(new SolarisNetworkIF(ni));
48              } catch (InstantiationException e) {
49                  LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
50              }
51          }
52          return ifList;
53      }
54  
55      @Override
56      public boolean updateAttributes() {
57          // Initialize to a sane default value
58          setTimeStamp(System.currentTimeMillis());
59          if (HAS_KSTAT2) {
60              // Use Kstat2 implementation
61              return updateAttributes2();
62          }
63          try (KstatChain kc = KstatUtil.openChain()) {
64              Kstat ksp = kc.lookup("link", -1, getName());
65              if (ksp == null) { // Solaris 10 compatibility
66                  ksp = kc.lookup(null, -1, getName());
67              }
68              if (ksp != null && kc.read(ksp)) {
69                  setBytesSent(KstatUtil.dataLookupLong(ksp, "obytes64"));
70                  setBytesRecv(KstatUtil.dataLookupLong(ksp, "rbytes64"));
71                  setPacketsSent(KstatUtil.dataLookupLong(ksp, "opackets64"));
72                  setPacketsRecv(KstatUtil.dataLookupLong(ksp, "ipackets64"));
73                  setOutErrors(KstatUtil.dataLookupLong(ksp, "oerrors"));
74                  setInErrors(KstatUtil.dataLookupLong(ksp, "ierrors"));
75                  setCollisions(KstatUtil.dataLookupLong(ksp, "collisions"));
76                  setInDrops(KstatUtil.dataLookupLong(ksp, "dl_idrops"));
77                  setSpeed(KstatUtil.dataLookupLong(ksp, "ifspeed"));
78                  // Snap time in ns; convert to ms
79                  setTimeStamp(ksp.ks_snaptime / 1_000_000L);
80                  return true;
81              }
82          }
83          return false;
84      }
85  
86      private boolean updateAttributes2() {
87          Object[] results = KstatUtil.queryKstat2("kstat:/net/link/" + getName() + "/0", "obytes64", "rbytes64",
88                  "opackets64", "ipackets64", "oerrors", "ierrors", "collisions", "dl_idrops", "ifspeed", "snaptime");
89          if (results[results.length - 1] == null) {
90              return false;
91          }
92          setBytesSent(results[0] == null ? 0L : (long) results[0]);
93          setBytesRecv(results[1] == null ? 0L : (long) results[1]);
94          setPacketsSent(results[2] == null ? 0L : (long) results[2]);
95          setPacketsRecv(results[3] == null ? 0L : (long) results[3]);
96          setOutErrors(results[4] == null ? 0L : (long) results[4]);
97          setInErrors(results[5] == null ? 0L : (long) results[5]);
98          setCollisions(results[6] == null ? 0L : (long) results[6]);
99          setInDrops(results[7] == null ? 0L : (long) results[7]);
100         setSpeed(results[8] == null ? 0L : (long) results[8]);
101         // Snap time in ns; convert to ms
102         setTimeStamp((long) results[9] / 1_000_000L);
103         return true;
104     }
