24  package oshi.hardware.platform.unix.solaris;
25  
26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  import oshi.annotation.concurrent.Immutable;
32  import oshi.hardware.SoundCard;
33  import oshi.hardware.common.AbstractSoundCard;
34  import oshi.util.ExecutingCommand;
35  import oshi.util.ParseUtil;
36  
37  /**
38   * Solaris Sound Card.
39   */
40  @Immutable
41  final class SolarisSoundCard extends AbstractSoundCard {
42  
43      private static final String LSHAL = "lshal";
44      private static final String DEFAULT_AUDIO_DRIVER = "audio810";
45  
46      /**
47       * Constructor for SolarisSoundCard.
48       *
49       * @param kernelVersion
50       *            The version
51       * @param name
52       *            The name
53       * @param codec
54       *            The codec
55       */
56      SolarisSoundCard(String kernelVersion, String name, String codec) {
57          super(kernelVersion, name, codec);
58      }
59  
60      /**
61       * <p>
62       * getSoundCards.
63       * </p>
64       *
65       * @return a {@link java.util.List} object.
66       */
67      public static List<SoundCard> getSoundCards() {
68          Map<String, String> vendorMap = new HashMap<>();
69          Map<String, String> productMap = new HashMap<>();
70          List<String> sounds = new ArrayList<>();
71          String key = "";
72          for (String line : ExecutingCommand.runNative(LSHAL)) {
73              line = line.trim();
74              if (line.startsWith("udi =")) {
75                  // we have the key.
76                  key = ParseUtil.getSingleQuoteStringValue(line);
77              } else if (!key.isEmpty() && !line.isEmpty()) {
78                  if (line.contains("info.solaris.driver =")
79                          && DEFAULT_AUDIO_DRIVER.equals(ParseUtil.getSingleQuoteStringValue(line))) {
80                      sounds.add(key);
81                  } else if (line.contains("info.product")) {
82                      productMap.put(key, ParseUtil.getStringBetween(line, '\''));
83                  } else if (line.contains("info.vendor")) {
84                      vendorMap.put(key, ParseUtil.getStringBetween(line, '\''));
85                  }
86              }
87          }
88          List<SoundCard> soundCards = new ArrayList<>();
89          for (String _key : sounds) {
90              soundCards.add(new SolarisSoundCard(productMap.get(_key) + " " + DEFAULT_AUDIO_DRIVER,
91                      vendorMap.get(_key) + " " + productMap.get(_key), productMap.get(_key)));
92          }
93          return soundCards;
94      }
95  }
