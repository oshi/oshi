5   package oshi.hardware.platform.unix.freebsd;
6   
7   import static java.lang.foreign.ValueLayout.JAVA_DOUBLE;
8   import static java.lang.foreign.ValueLayout.JAVA_LONG;
9   import static oshi.ffm.ForeignFunctions.callInArenaIntOrDefault;
10  
11  import java.lang.foreign.MemorySegment;
12  
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  import oshi.annotation.concurrent.ThreadSafe;
17  import oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions;
18  import oshi.ffm.util.platform.unix.freebsd.BsdSysctlUtilFFM;
19  import oshi.hardware.common.platform.unix.freebsd.FreeBsdCentralProcessor;
20  import oshi.util.LogLevel;
21  
22  /**
23   * FFM-backed FreeBSD central processor.
24   */
25  @ThreadSafe
26  public class FreeBsdCentralProcessorFFM extends FreeBsdCentralProcessor {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(FreeBsdCentralProcessorFFM.class);
29  
30      @Override
31      protected String sysctlString(String name, String def) {
32          return BsdSysctlUtilFFM.sysctl(name, def);
33      }
34  
35      @Override
36      protected long sysctlLong(String name, long def) {
37          return BsdSysctlUtilFFM.sysctl(name, def);
38      }
39  
40      @Override
41      protected int sysctlInt(String name, int def) {
42          return BsdSysctlUtilFFM.sysctl(name, def);
43      }
44  
45      @Override
46      protected long[] queryCpTimes(String name) {
47          MemorySegment buf = BsdSysctlUtilFFM.sysctl(name);
48          if (buf == null) {
49              return null;
50          }
51          int count = (int) (buf.byteSize() / Long.BYTES);
52          long[] result = new long[count];
53          for (int i = 0; i < count; i++) {
54              result[i] = buf.getAtIndex(JAVA_LONG, i);
55          }
56          return result;
57      }
58  
59      @Override
60      protected int getloadavgNative(double[] loadavg, int nelem) {
61          return callInArenaIntOrDefault(arena -> {
62              MemorySegment avg = arena.allocate(JAVA_DOUBLE, nelem);
63              int retval = FreeBsdLibcFunctions.getloadavg(avg, nelem);
64              for (int i = 0; i < nelem && i < retval; i++) {
65                  loadavg[i] = avg.getAtIndex(JAVA_DOUBLE, i);
66              }
67              return retval;
68          }, LOG, LogLevel.WARN, "Failed to read load average", 0);
69      }
70  }
