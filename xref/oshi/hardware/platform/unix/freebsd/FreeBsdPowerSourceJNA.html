5   package oshi.hardware.platform.unix.freebsd;
6   
7   import java.time.LocalDate;
8   import java.util.List;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.hardware.PowerSource;
12  import oshi.hardware.common.platform.unix.freebsd.FreeBsdPowerSource;
13  import oshi.util.platform.unix.freebsd.BsdSysctlUtil;
14  
15  /**
16   * A Power Source
17   */
18  @ThreadSafe
19  public final class FreeBsdPowerSourceJNA extends FreeBsdPowerSource {
20  
21      public FreeBsdPowerSourceJNA(String psName, String psDeviceName, double psRemainingCapacityPercent,
22              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
23              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
24              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
25              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
26              String psSerialNumber, double psTemperature) {
27          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
28                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
29                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
30                  psManufacturer, psSerialNumber, psTemperature);
31      }
32  
33      @Override
34      protected List<PowerSource> queryPowerSources() {
35          return getPowerSources();
36      }
37  
38      /**
39       * Gets Battery Information.
40       *
41       * @return A list of PowerSource objects representing batteries, etc.
42       */
43      public static List<PowerSource> getPowerSources() {
44          return queryBattery("BAT0", BsdSysctlUtil::sysctl, FreeBsdPowerSourceJNA::new);
45      }
46  }
