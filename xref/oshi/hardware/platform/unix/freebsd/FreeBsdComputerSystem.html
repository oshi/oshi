19  package oshi.hardware.platform.unix.freebsd;
20  
21  import oshi.hardware.common.AbstractComputerSystem;
22  import oshi.util.ExecutingCommand;
23  import oshi.util.ParseUtil;
24  
25  /**
26   * Hardware data obtained from dmidecode
27   *
28   * @author widdis [at] gmail [dot] com
29   */
30  final class FreeBsdComputerSystem extends AbstractComputerSystem {
31  
32      private static final long serialVersionUID = 1L;
33  
34      FreeBsdComputerSystem() {
35          init();
36      }
37  
38      private void init() {
39  
40          // $ sudo dmidecode -t system
41          // # dmidecode 3.0
42          // Scanning /dev/mem for entry point.
43          // SMBIOS 2.7 present.
44          //
45          // Handle 0x0001, DMI type 1, 27 bytes
46          // System Information
47          // Manufacturer: Parallels Software International Inc.
48          // Product Name: Parallels Virtual Platform
49          // Version: None
50          // Serial Number: Parallels-47 EC 38 2A 33 1B 4C 75 94 0F F7 AF 86 63 C0
51          // C4
52          // UUID: 2A38EC47-1B33-854C-940F-F7AF8663C0C4
53          // Wake-up Type: Power Switch
54          // SKU Number: Undefined
55          // Family: Parallels VM
56          //
57          // Handle 0x0016, DMI type 32, 20 bytes
58          // System Boot Information
59          // Status: No errors detected
60  
61          String manufacturer = "";
62          final String manufacturerMarker = "Manufacturer:";
63          String productName = "";
64          final String productNameMarker = "Product Name:";
65          String serialNumber = "";
66          final String serialNumMarker = "Serial Number:";
67  
68          // Only works with root permissions but it's all we've got
69          for (final String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
70              if (checkLine.contains(manufacturerMarker)) {
71                  manufacturer = checkLine.split(manufacturerMarker)[1].trim();
72              }
73              if (checkLine.contains(productNameMarker)) {
74                  productName = checkLine.split(productNameMarker)[1].trim();
75              }
76              if (checkLine.contains(serialNumMarker)) {
77                  serialNumber = checkLine.split(serialNumMarker)[1].trim();
78              }
79          }
80          if (!manufacturer.isEmpty()) {
81              setManufacturer(manufacturer);
82          }
83          if (!productName.isEmpty()) {
84              setModel(productName);
85          }
86          if (serialNumber.isEmpty()) {
87              serialNumber = getSystemSerialNumber();
88          }
89          setSerialNumber(serialNumber);
90  
91          setFirmware(new FreeBsdFirmware());
92  
93          setBaseboard(new FreeBsdBaseboard());
94      }
95  
96      private String getSystemSerialNumber() {
97          String marker = "system.hardware.serial =";
98          for (String checkLine : ExecutingCommand.runNative("lshal")) {
99              if (checkLine.contains(marker)) {
100                 return ParseUtil.getSingleQuoteStringValue(checkLine);
101             }
102         }
103         return "unknown";
104     }
105 }
