5   package oshi.hardware.platform.unix.freebsd;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.ffm.util.platform.unix.freebsd.BsdSysctlUtilFFM;
9   import oshi.hardware.VirtualMemory;
10  import oshi.hardware.common.platform.unix.freebsd.FreeBsdGlobalMemory;
11  
12  /**
13   * FFM-backed FreeBSD global memory. Native reads use {@link BsdSysctlUtilFFM}; shared parsing and the public API live
14   * on {@link FreeBsdGlobalMemory}.
15   */
16  @ThreadSafe
17  public class FreeBsdGlobalMemoryFFM extends FreeBsdGlobalMemory {
18  
19      @Override
20      protected long queryVmStats() {
21          // cached removed in FreeBSD 12 but was always set to 0.
22          // Counters are uint32 from the kernel; widen to long before adding so the sum and the page-size multiply
23          // don't truncate or sign-extend if either counter exceeds 2^31.
24          long inactive = Integer.toUnsignedLong(BsdSysctlUtilFFM.sysctl("vm.stats.vm.v_inactive_count", 0));
25          long free = Integer.toUnsignedLong(BsdSysctlUtilFFM.sysctl("vm.stats.vm.v_free_count", 0));
26          return (inactive + free) * getPageSize();
27      }
28  
29      @Override
30      protected long queryPhysMem() {
31          return BsdSysctlUtilFFM.sysctl("hw.physmem", 0L);
32      }
33  
34      @Override
35      protected VirtualMemory createVirtualMemory() {
36          return new FreeBsdVirtualMemoryFFM(this);
37      }
38  }
