60      @Override
61      protected long[] querySystemCpuLoadTicks() {
62          long[] ticks = new long[TickType.values().length];
63          int[] mib = { CTL_KERN, KERN_CPTIME };
64          MemorySegment buf = OpenBsdSysctlUtilFFM.sysctl(mib);
65          if (buf != null) {
66              int arraySize = (int) (buf.byteSize() / JAVA_LONG.byteSize());
67              if (arraySize >= CPUSTATES) {
68                  ticks[TickType.USER.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_USER);
69                  ticks[TickType.NICE.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_NICE);
70                  ticks[TickType.SYSTEM.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_SYS);
71                  int offset = arraySize > CPUSTATES ? 1 : 0;
72                  ticks[TickType.IRQ.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_INTR + offset);
73                  ticks[TickType.IDLE.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_IDLE + offset);
74              }
75          }
76          return ticks;
77      }
78  
79      @Override
80      protected long[][] queryProcessorCpuLoadTicks() {
81          long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
82          for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
83              int[] mib = { CTL_KERN, KERN_CPTIME2, cpu };
84              MemorySegment buf = OpenBsdSysctlUtilFFM.sysctl(mib);
85              if (buf != null) {
86                  int arraySize = (int) (buf.byteSize() / JAVA_LONG.byteSize());
87                  if (arraySize >= CPUSTATES) {
88                      ticks[cpu][TickType.USER.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_USER);
89                      ticks[cpu][TickType.NICE.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_NICE);
90                      ticks[cpu][TickType.SYSTEM.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_SYS);
91                      int offset = arraySize > CPUSTATES ? 1 : 0;
92                      ticks[cpu][TickType.IRQ.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_INTR + offset);
93                      ticks[cpu][TickType.IDLE.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_IDLE + offset);
94                  }
95              }
96          }
97          return ticks;
98      }

100     @Override
101     public double[] getSystemLoadAverage(int nelem) {
102         if (nelem < 1 || nelem > 3) {
103             throw new IllegalArgumentException("Must include from one to three elements.");
104         }
105         double[] average = new double[nelem];
106         Arrays.fill(average, -1d);
107         return ForeignFunctions.callInArenaOrDefault(arena -> {
108             MemorySegment avg = arena.allocate(JAVA_DOUBLE, nelem);
109             int retval = OpenBsdLibcFunctions.getloadavg(avg, nelem);
110             double[] result = new double[nelem];
111             for (int i = 0; i < nelem; i++) {
112                 result[i] = (i < retval) ? avg.getAtIndex(JAVA_DOUBLE, i) : -1d;
113             }
114             return result;
115         }, LOG, Level.WARN, "Failed to read load average", average);
116     }
