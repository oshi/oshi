62      private static final Pattern DMESG_CPU = Pattern.compile("cpu(\\d+): smt (\\d+), core (\\d+), package (\\d+)");

64      @Override
65      protected ProcessorIdentifier queryProcessorId() {
66          String cpuVendor = OpenBsdSysctlUtilFFM.sysctl("machdep.cpuvendor", "");
67          int[] mib = { CTL_HW, HW_MODEL };
68          String cpuName = OpenBsdSysctlUtilFFM.sysctl(mib, "");
69          int cpuid = ParseUtil.hexStringToInt(OpenBsdSysctlUtilFFM.sysctl("machdep.cpuid", ""), 0);
70          int cpufeature = ParseUtil.hexStringToInt(OpenBsdSysctlUtilFFM.sysctl("machdep.cpufeature", ""), 0);
71          Triplet<Integer, Integer, Integer> cpu = cpuidToFamilyModelStepping(cpuid);
72          String cpuFamily = cpu.getA().toString();
73          String cpuModel = cpu.getB().toString();
74          String cpuStepping = cpu.getC().toString();
75          long cpuFreq = ParseUtil.parseHertz(cpuName);
76          if (cpuFreq < 0) {
77              cpuFreq = queryMaxFreq();
78          }
79          mib = new int[] { CTL_HW, HW_MACHINE };
80          String machine = OpenBsdSysctlUtilFFM.sysctl(mib, "");
81          boolean cpu64bit = machine != null && machine.contains("64")
82                  || ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64");
83          String processorID = String.format(Locale.ROOT, "%08x%08x", cpufeature, cpuid);
84  
85          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
86                  cpuFreq);
87      }
88  
89      private static Triplet<Integer, Integer, Integer> cpuidToFamilyModelStepping(int cpuid) {
90          int family = cpuid >> 16 & 0xff0 | cpuid >> 8 & 0xf;
91          int model = cpuid >> 12 & 0xf0 | cpuid >> 4 & 0xf;
92          int stepping = cpuid & 0xf;
93          return new Triplet<>(family, model, stepping);
94      }

96      @Override
97      protected long[] queryCurrentFreq() {
98          long[] freq = new long[1];
99          int[] mib = { CTL_HW, HW_CPUSPEED };
100         freq[0] = OpenBsdSysctlUtilFFM.sysctl(mib, 0) * 1_000_000L;
101         return freq;
102     }

104     @Override
105     protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
106         Map<Integer, Integer> coreMap = new HashMap<>();
107         Map<Integer, Integer> packageMap = new HashMap<>();
108         for (String line : ExecutingCommand.runNative("dmesg")) {
109             Matcher m = DMESG_CPU.matcher(line);
110             if (m.matches()) {
111                 int cpu = ParseUtil.parseIntOrDefault(m.group(1), 0);
112                 coreMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(3), 0));
113                 packageMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(4), 0));
114             }
115         }
116         int logicalProcessorCount = OpenBsdSysctlUtilFFM.sysctl("hw.ncpuonline", 1);
117         if (logicalProcessorCount < coreMap.keySet().size()) {
118             logicalProcessorCount = coreMap.keySet().size();
119         }
120         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
121         for (int i = 0; i < logicalProcessorCount; i++) {
122             logProcs.add(new LogicalProcessor(i, coreMap.getOrDefault(i, 0), packageMap.getOrDefault(i, 0)));
123         }
124         Map<Integer, String> cpuMap = new HashMap<>();
125         Pattern p = Pattern.compile("cpu(\\d+).*: ((ARM|AMD|Intel|Apple).+)");
126 
127         Set<ProcessorCache> caches = new HashSet<>();
128         Pattern q = Pattern.compile("cpu(\\d+).*: (.+(I-|D-|L\\d+\\s)cache)");
129         Set<String> featureFlags = new LinkedHashSet<>();
130         for (String s : ExecutingCommand.runNative("dmesg")) {
131             Matcher m = p.matcher(s);
132             if (m.matches()) {
133                 int coreId = ParseUtil.parseIntOrDefault(m.group(1), 0);
134                 cpuMap.put(coreId, m.group(2).trim());
135             } else {
136                 Matcher n = q.matcher(s);
137                 if (n.matches()) {
138                     for (String cacheStr : n.group(2).split(",")) {
139                         ProcessorCache cache = parseCacheStr(cacheStr);
140                         if (cache != null) {
141                             caches.add(cache);
142                         }
143                     }
144                 }
145             }
146             if (s.startsWith("cpu")) {
147                 String[] ss = s.trim().split(": ");
148                 if (ss.length == 2 && ss[1].split(",").length > 3) {
149                     featureFlags.add(ss[1]);
150                 }
151             }
152         }
153         List<PhysicalProcessor> physProcs = cpuMap.isEmpty() ? null : createProcListFromDmesg(logProcs, cpuMap);
154         return new Quartet<>(logProcs, physProcs, orderedProcCaches(caches), new ArrayList<>(featureFlags));
155     }

157     private ProcessorCache parseCacheStr(String cacheStr) {
158         String[] split = ParseUtil.whitespaces.split(cacheStr.trim());
159         if (split.length > 3) {
160             switch (split[split.length - 1]) {
161                 case "I-cache":
162                     return new ProcessorCache(1, ParseUtil.getFirstIntValue(split[2]),
163                             ParseUtil.getFirstIntValue(split[1]), ParseUtil.parseDecimalMemorySizeToBinary(split[0]),
164                             Type.INSTRUCTION);
165                 case "D-cache":
166                     return new ProcessorCache(1, ParseUtil.getFirstIntValue(split[2]),
167                             ParseUtil.getFirstIntValue(split[1]), ParseUtil.parseDecimalMemorySizeToBinary(split[0]),
168                             Type.DATA);
169                 default:
170                     return new ProcessorCache(ParseUtil.getFirstIntValue(split[3]),
171                             ParseUtil.getFirstIntValue(split[2]), ParseUtil.getFirstIntValue(split[1]),
172                             ParseUtil.parseDecimalMemorySizeToBinary(split[0]), Type.UNIFIED);
173             }
174         }
175         return null;
176     }

188     private static Pair<Long, Long> queryVmStats() {
189         long contextSwitches = 0L;
190         long interrupts = 0L;
191         for (String line : ExecutingCommand.runNative("vmstat -s")) {
192             if (line.endsWith("cpu context switches")) {
193                 contextSwitches = ParseUtil.getFirstIntValue(line);
194             } else if (line.endsWith("interrupts")) {
195                 interrupts = ParseUtil.getFirstIntValue(line);
196             }
197         }
198         return new Pair<>(contextSwitches, interrupts);
199     }

201     @Override
202     protected long[] querySystemCpuLoadTicks() {
203         long[] ticks = new long[TickType.values().length];
204         int[] mib = { CTL_KERN, KERN_CPTIME };
205         MemorySegment buf = OpenBsdSysctlUtilFFM.sysctl(mib);
206         if (buf != null) {
207             int arraySize = (int) (buf.byteSize() / JAVA_LONG.byteSize());
208             if (arraySize >= CPUSTATES) {
209                 ticks[TickType.USER.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_USER);
210                 ticks[TickType.NICE.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_NICE);
211                 ticks[TickType.SYSTEM.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_SYS);
212                 int offset = arraySize > CPUSTATES ? 1 : 0;
213                 ticks[TickType.IRQ.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_INTR + offset);
214                 ticks[TickType.IDLE.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_IDLE + offset);
215             }
216         }
217         return ticks;
218     }

220     @Override
221     protected long[][] queryProcessorCpuLoadTicks() {
222         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
223         for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
224             int[] mib = { CTL_KERN, KERN_CPTIME2, cpu };
225             MemorySegment buf = OpenBsdSysctlUtilFFM.sysctl(mib);
226             if (buf != null) {
227                 int arraySize = (int) (buf.byteSize() / JAVA_LONG.byteSize());
228                 if (arraySize >= CPUSTATES) {
229                     ticks[cpu][TickType.USER.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_USER);
230                     ticks[cpu][TickType.NICE.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_NICE);
231                     ticks[cpu][TickType.SYSTEM.getIndex()] = buf.getAtIndex(JAVA_LONG, CP_SYS);
232                     int offset = arraySize > CPUSTATES ? 1 : 0;
233                     ticks[cpu][TickType.IRQ.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_INTR + offset);
234                     ticks[cpu][TickType.IDLE.getIndex()] = buf.getAtIndex(JAVA_LONG, (long) CP_IDLE + offset);
235                 }
236             }
237         }
238         return ticks;
239     }

241     @Override
242     public double[] getSystemLoadAverage(int nelem) {
243         if (nelem < 1 || nelem > 3) {
244             throw new IllegalArgumentException("Must include from one to three elements.");
245         }
246         double[] average = new double[nelem];
247         Arrays.fill(average, -1d);
248         return ForeignFunctions.callInArenaOrDefault(arena -> {
249             MemorySegment avg = arena.allocate(JAVA_DOUBLE, nelem);
250             int retval = OpenBsdLibcFunctions.getloadavg(avg, nelem);
251             double[] result = new double[nelem];
252             for (int i = 0; i < nelem; i++) {
253                 result[i] = (i < retval) ? avg.getAtIndex(JAVA_DOUBLE, i) : -1d;
254             }
255             return result;
256         }, LOG, Level.WARN, "Failed to read load average", average);
257     }
