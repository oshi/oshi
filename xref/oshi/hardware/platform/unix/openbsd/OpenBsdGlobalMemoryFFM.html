5   package oshi.hardware.platform.unix.openbsd;
6   
7   import static oshi.ffm.unix.openbsd.OpenBsdLibcFunctions.CTL_VFS;
8   import static oshi.ffm.unix.openbsd.OpenBsdLibcFunctions.VFS_BCACHESTAT;
9   import static oshi.ffm.unix.openbsd.OpenBsdLibcFunctions.VFS_GENERIC;
10  
11  import java.lang.foreign.Arena;
12  import java.lang.foreign.MemorySegment;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.ffm.unix.openbsd.OpenBsdLibcFunctions;
16  import oshi.ffm.util.platform.unix.openbsd.OpenBsdSysctlUtilFFM;
17  import oshi.hardware.VirtualMemory;
18  import oshi.hardware.common.platform.unix.openbsd.OpenBsdVirtualMemory;
19  import oshi.util.ExecutingCommand;
20  import oshi.util.ParseUtil;
21  
22  @ThreadSafe
23  final class OpenBsdGlobalMemoryFFM extends oshi.hardware.common.platform.unix.openbsd.OpenBsdGlobalMemory {
24  
25      @Override
26      protected long queryAvailable() {
27          long free = 0L;
28          long inactive = 0L;
29          for (String line : ExecutingCommand.runNative("vmstat -s")) {
30              if (line.endsWith("pages free")) {
31                  free = ParseUtil.getFirstIntValue(line);
32              } else if (line.endsWith("pages inactive")) {
33                  inactive = ParseUtil.getFirstIntValue(line);
34              }
35          }
36          long bufPages = 0L;
37          int[] mib = { CTL_VFS, VFS_GENERIC, VFS_BCACHESTAT };
38          try (Arena arena = Arena.ofConfined()) {
39              MemorySegment seg = arena.allocate(OpenBsdLibcFunctions.BCACHESTATS_LAYOUT);
40              if (OpenBsdSysctlUtilFFM.sysctl(mib, seg)) {
41                  bufPages = OpenBsdLibcFunctions.bcachestatsNumbufpages(seg);
42              }
43          }
44          return (bufPages + free + inactive) * getPageSize();
45      }
46  
47      @Override
48      protected long queryPhysMem() {
49          return OpenBsdSysctlUtilFFM.sysctl("hw.physmem", 0L);
50      }
51  
52      @Override
53      protected long queryPageSize() {
54          return OpenBsdSysctlUtilFFM.sysctl("hw.pagesize", 4096L);
55      }
56  
57      @Override
58      protected VirtualMemory createVirtualMemory() {
59          return new OpenBsdVirtualMemory(this);
60      }
61  }
