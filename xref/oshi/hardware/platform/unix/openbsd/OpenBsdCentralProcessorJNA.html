52      /**
53       * Get the system CPU load ticks
54       *
55       * @return The system CPU load ticks
56       */
57      @Override
58      protected long[] querySystemCpuLoadTicks() {
59          long[] ticks = new long[TickType.values().length];
60          int[] mib = new int[2];
61          mib[0] = CTL_KERN;
62          mib[1] = KERN_CPTIME;
63          try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
64              // array of 5 or 6 native longs
65              long[] cpuTicks = cpTimeToTicks(m, false);
66              if (cpuTicks.length >= 5) {
67                  ticks[TickType.USER.getIndex()] = cpuTicks[CP_USER];
68                  ticks[TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
69                  ticks[TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
70                  int offset = cpuTicks.length > 5 ? 1 : 0;
71                  ticks[TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
72                  ticks[TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
73              }
74          }
75          return ticks;
76      }
77  
78      /**
79       * Get the processor CPU load ticks
80       *
81       * @return The processor CPU load ticks
82       */
83      @Override
84      protected long[][] queryProcessorCpuLoadTicks() {
85          long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
86          int[] mib = new int[3];
87          mib[0] = CTL_KERN;
88          mib[1] = KERN_CPTIME2;
89          for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
90              mib[2] = cpu;
91              try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
92                  // array of 5 or 6 longs
93                  long[] cpuTicks = cpTimeToTicks(m, true);
94                  if (cpuTicks.length >= 5) {
95                      ticks[cpu][TickType.USER.getIndex()] = cpuTicks[CP_USER];
96                      ticks[cpu][TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
97                      ticks[cpu][TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
98                      int offset = cpuTicks.length > 5 ? 1 : 0;
99                      ticks[cpu][TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
100                     ticks[cpu][TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
101                 }
102             }
103         }
104         return ticks;
105     }

107     /**
108      * Parse memory buffer returned from sysctl kern.cptime or kern.cptime2 to an array of 5 or 6 longs depending on
109      * version.
110      * <p>
111      * Versions 6.4 and later have a 6-element array while earlier versions have only 5 elements. Additionally
112      * kern.cptime uses a native-sized long (32- or 64-bit) value while kern.cptime2 is always a 64-bit value.
113      *
114      * @param m          A buffer containing the array.
115      * @param force64bit True if the buffer is filled with 64-bit longs, false if native long sized values
116      * @return The array
117      */
118     private static long[] cpTimeToTicks(Memory m, boolean force64bit) {
119         long longBytes = force64bit ? 8L : Native.LONG_SIZE;
120         int arraySize = m == null ? 0 : (int) (m.size() / longBytes);
121         if (force64bit && m != null) {
122             return m.getLongArray(0, arraySize);
123         }
124         long[] ticks = new long[arraySize];
125         for (int i = 0; i < arraySize; i++) {
126             ticks[i] = m.getNativeLong(i * longBytes).longValue();
127         }
128         return ticks;
129     }

131     /**
132      * Returns the system load average for the number of elements specified, up to 3, representing 1, 5, and 15 minutes.
133      * The system load average is the sum of the number of runnable entities queued to the available processors and the
134      * number of runnable entities running on the available processors averaged over a period of time. The way in which
135      * the load average is calculated is operating system specific but is typically a damped time-dependent average. If
136      * the load average is not available, a negative value is returned. This method is designed to provide a hint about
137      * the system load and may be queried frequently.
138      * <p>
139      * The load average may be unavailable on some platforms (e.g., Windows) where it is expensive to implement this
140      * method.
141      *
142      * @param nelem Number of elements to return.
143      * @return an array of the system load averages for 1, 5, and 15 minutes with the size of the array specified by
144      *         nelem; or negative values if not available.
145      */
146     @Override
147     public double[] getSystemLoadAverage(int nelem) {
148         if (nelem < 1 || nelem > 3) {
149             throw new IllegalArgumentException("Must include from one to three elements.");
150         }
151         double[] average = new double[nelem];
152         int retval = OpenBsdLibc.INSTANCE.getloadavg(average, nelem);
153         if (retval < nelem) {
154             Arrays.fill(average, -1d);
155         }
156         return average;
157     }
