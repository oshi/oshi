43          return OpenBsdSysctlUtil.sysctl(mib, def);

46      @Override
47      protected long[] querySystemCpTime() {
48          int[] mib = { CTL_KERN, KERN_CPTIME };
49          try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
50              return cpTimeToTicks(m, false);
51          }
52      }
53  
54      @Override
55      protected long[] queryProcessorCpTime(int cpu) {
56          int[] mib = { CTL_KERN, KERN_CPTIME2, cpu };
57          try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
58              return cpTimeToTicks(m, true);
59          }
60      }
61  
62      /**
63       * Parse memory buffer returned from sysctl kern.cptime or kern.cptime2 to an array of 5 or 6 longs depending on
64       * version.
65       * <p>
66       * Versions 6.4 and later have a 6-element array while earlier versions have only 5 elements. Additionally
67       * kern.cptime uses a native-sized long (32- or 64-bit) value while kern.cptime2 is always a 64-bit value.
68       *
69       * @param m          A buffer containing the array.
70       * @param force64bit True if the buffer is filled with 64-bit longs, false if native long sized values
71       * @return The array
72       */
73      private static long[] cpTimeToTicks(Memory m, boolean force64bit) {
74          long longBytes = force64bit ? 8L : Native.LONG_SIZE;
75          int arraySize = m == null ? 0 : (int) (m.size() / longBytes);
76          if (force64bit && m != null) {
77              return m.getLongArray(0, arraySize);
78          }
79          long[] ticks = new long[arraySize];
80          for (int i = 0; i < arraySize; i++) {
81              ticks[i] = m.getNativeLong(i * longBytes).longValue();
82          }
83          return ticks;
84      }
85  
86      @Override
87      protected int getloadavgNative(double[] loadavg, int nelem) {
88          return OpenBsdLibc.INSTANCE.getloadavg(loadavg, nelem);
89      }
90  }
