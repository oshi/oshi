53      private static final Pattern DMESG_CPU = Pattern.compile("cpu(\\d+): smt (\\d+), core (\\d+), package (\\d+)");

55      @Override
56      protected ProcessorIdentifier queryProcessorId() {
57          String cpuVendor = OpenBsdSysctlUtil.sysctl("machdep.cpuvendor", "");
58          int[] mib = new int[2];
59          mib[0] = CTL_HW;
60          mib[1] = HW_MODEL;
61          String cpuName = OpenBsdSysctlUtil.sysctl(mib, "");
62          // CPUID: first 32 bits is cpufeature, last 32 bits is cpuid
63          int cpuid = ParseUtil.hexStringToInt(OpenBsdSysctlUtil.sysctl("machdep.cpuid", ""), 0);
64          int cpufeature = ParseUtil.hexStringToInt(OpenBsdSysctlUtil.sysctl("machdep.cpufeature", ""), 0);
65          Triplet<Integer, Integer, Integer> cpu = cpuidToFamilyModelStepping(cpuid);
66          String cpuFamily = cpu.getA().toString();
67          String cpuModel = cpu.getB().toString();
68          String cpuStepping = cpu.getC().toString();
69          long cpuFreq = ParseUtil.parseHertz(cpuName);
70          if (cpuFreq < 0) {
71              cpuFreq = queryMaxFreq();
72          }
73          mib[1] = HW_MACHINE;
74          String machine = OpenBsdSysctlUtil.sysctl(mib, "");
75          boolean cpu64bit = machine != null && machine.contains("64")
76                  || ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64");
77          String processorID = String.format("%08x%08x", cpufeature, cpuid);
78  
79          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
80                  cpuFreq);
81      }

83      private static Triplet<Integer, Integer, Integer> cpuidToFamilyModelStepping(int cpuid) {
84          // family is bits 27:20 | 11:8
85          int family = cpuid >> 16 & 0xff0 | cpuid >> 8 & 0xf;
86          // model is bits 19:16 | 7:4
87          int model = cpuid >> 12 & 0xf0 | cpuid >> 4 & 0xf;
88          // stepping is bits 3:0
89          int stepping = cpuid & 0xf;
90          return new Triplet<>(family, model, stepping);
91      }

93      @Override
94      protected long queryMaxFreq() {
95          return queryCurrentFreq()[0];
96      }

98      @Override
99      protected long[] queryCurrentFreq() {
100         long[] freq = new long[1];
101         int[] mib = new int[2];
102         mib[0] = CTL_HW;
103         mib[1] = HW_CPUSPEED;
104         freq[0] = OpenBsdSysctlUtil.sysctl(mib, 0L) * 1_000_000L;
105         return freq;
106     }

108     @Override
109     protected Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> initProcessorCounts() {
110         // Iterate dmesg, look for lines:
111         // cpu0: smt 0, core 0, package 0
112         // cpu1: smt 0, core 1, package 0
113         Map<Integer, Integer> coreMap = new HashMap<>();
114         Map<Integer, Integer> packageMap = new HashMap<>();
115         for (String line : ExecutingCommand.runNative("dmesg")) {
116             Matcher m = DMESG_CPU.matcher(line);
117             if (m.matches()) {
118                 int cpu = ParseUtil.parseIntOrDefault(m.group(1), 0);
119                 coreMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(3), 0));
120                 packageMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(4), 0));
121             }
122         }
123         // native call seems to fail here, use fallback
124         int logicalProcessorCount = OpenBsdSysctlUtil.sysctl("hw.ncpuonline", 1);
125         // If we found more procs in dmesg, update
126         if (logicalProcessorCount < coreMap.keySet().size()) {
127             logicalProcessorCount = coreMap.keySet().size();
128         }
129         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
130         for (int i = 0; i < logicalProcessorCount; i++) {
131             logProcs.add(new LogicalProcessor(i, coreMap.getOrDefault(i, 0), packageMap.getOrDefault(i, 0)));
132         }
133         Map<Integer, String> cpuMap = new HashMap<>();
134         // cpu0 at mainbus0 mpidr 0: ARM Cortex-A7 r0p5
135         // but NOT: cpu0 at mainbus0: apid 0 (boot processor)
136         // cpu0: AMD GX-412TC SOC, 998.28 MHz, 16-30-01
137         // cpu0: AMD EPYC 7313P 16-Core Processor, 2994.74 MHz, 19-01-01
138         // cpu0: Intel(R) Celeron(R) N4000 CPU @ 1.10GHz, 2491.67 MHz, 06-7a-01
139         Pattern p = Pattern.compile("cpu(\\\\d+).*: ((ARM|AMD|Intel|Apple).+)");
140 
141         Set<ProcessorCache> caches = new HashSet<>();
142         // cpu0: 48KB 64b/line 12-way D-cache, 32KB 64b/line 8-way I-cache,
143         // ... 1MB 64b/line 10-way L2 cache, 24MB 64b/line 12-way L3 cache
144         // cpu0: 32KB 64b/line 8-way D-cache, 64KB 64b/line 4-way I-cache,
145         // ... 512KB 64b/line 8-way L2 cache, 4MB 64b/line 16-way L3 cache
146         // cpu0: 32KB 64b/line 2-way I-cache, 32KB 64b/line 8-way D-cache,
147         // ... 2MB 64b/line 16-way L2 cache
148         // cpu0: 32KB 64b/line 8-way I-cache, 32KB 64b/line 8-way D-cache,
149         // ... 512KB 64b/line 8-way L2 cache
150         // cpu0: 256KB 64b/line disabled L2 cache
151         // cpu0: 1MB 64b/line 16-way L2 cache
152         // --- Mac M1 example ---
153         // shows different for icestorm/firestrorm and multiple lines for L1 vs. L2
154         // cpu0 at mainbus0 mpidr 0: Apple Icestorm Pro r2p0
155         // cpu0: 128KB 64b/line 8-way L1 VIPT I-cache, 64KB 64b/line 8-way L1 D-cache
156         // cpu0: 4096KB 128b/line 16-way L2 cache
157         // cpu2 at mainbus0 mpidr 10100: Apple Firestorm Pro r2p0
158         // cpu2: 192KB 64b/line 6-way L1 VIPT I-cache, 128KB 64b/line 8-way L1 D-cache
159         // cpu2: 12288KB 128b/line 12-way L2 cache
160         // --- BIG:little example ---
161         // shows different for A53/A72 and multiple lines for L1 vs. L2
162         // cpu0 at mainbus0 mpidr 0: ARM Cortex-A53 r0p4
163         // cpu0: 32KB 64b/line 2-way L1 VIPT I-cache, 32KB 64b/line 4-way L1 D-cache
164         // cpu0: 512KB 64b/line 16-way L2 cache
165         // cpu4 at mainbus0 mpidr 100: ARM Cortex-A72 r0p2
166         // cpu4: 48KB 64b/line 3-way L1 PIPT I-cache, 32KB 64b/line 2-way L1 D-cache
167         // cpu4: 1024KB 64b/line 16-way L2 cache
168         Pattern q = Pattern.compile("cpu(\\\\d+).*: (.+(I-|D-|L\\d+\\s)cache)");
169         for (String s : ExecutingCommand.runNative("dmesg")) {
170             Matcher m = p.matcher(s);
171             if (m.matches()) {
172                 int coreId = ParseUtil.parseIntOrDefault(m.group(1), 0);
173                 cpuMap.put(coreId, m.group(2).trim());
174             } else {
175                 Matcher n = q.matcher(s);
176                 if (n.matches()) {
177                     for (String cacheStr : n.group(1).split(",")) {
178                         ProcessorCache cache = parseCacheStr(cacheStr);
179                         if (cache != null) {
180                             caches.add(cache);
181                         }
182                     }
183                 }
184             }
185         }
186         List<PhysicalProcessor> physProcs = cpuMap.isEmpty() ? null : createProcListFromDmesg(logProcs, cpuMap);
187         return new Triplet<>(logProcs, physProcs, orderedProcCaches(caches));
188     }

190     private ProcessorCache parseCacheStr(String cacheStr) {
191         String[] split = ParseUtil.whitespaces.split(cacheStr);
192         if (split.length > 3) {
193             switch (split[split.length - 1]) {
194             case "I-cache":
195                 return new ProcessorCache(1, ParseUtil.getFirstIntValue(split[2]), ParseUtil.getFirstIntValue(split[1]),
196                         ParseUtil.parseDecimalMemorySizeToBinary(split[0]), Type.INSTRUCTION);
197             case "D-cache":
198                 return new ProcessorCache(1, ParseUtil.getFirstIntValue(split[2]), ParseUtil.getFirstIntValue(split[1]),
199                         ParseUtil.parseDecimalMemorySizeToBinary(split[0]), Type.DATA);
200             default:
201                 return new ProcessorCache(ParseUtil.getFirstIntValue(split[3]), ParseUtil.getFirstIntValue(split[2]),
202                         ParseUtil.getFirstIntValue(split[1]), ParseUtil.parseDecimalMemorySizeToBinary(split[0]),
203                         Type.UNIFIED);
204             }
205         }
206         return null;
207     }

229     private static Pair<Long, Long> queryVmStats() {
230         long contextSwitches = 0L;
231         long interrupts = 0L;
232         List<String> vmstat = ExecutingCommand.runNative("vmstat -s");
233         for (String line : vmstat) {
234             if (line.endsWith("cpu context switches")) {
235                 contextSwitches = ParseUtil.getFirstIntValue(line);
236             } else if (line.endsWith("interrupts")) {
237                 interrupts = ParseUtil.getFirstIntValue(line);
238             }
239         }
240         return new Pair<>(contextSwitches, interrupts);
241     }

243     /**
244      * Get the system CPU load ticks
245      *
246      * @return The system CPU load ticks
247      */
248     @Override
249     protected long[] querySystemCpuLoadTicks() {
250         long[] ticks = new long[TickType.values().length];
251         int[] mib = new int[2];
252         mib[0] = CTL_KERN;
253         mib[1] = KERN_CPTIME;
254         try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
255             // array of 5 or 6 native longs
256             long[] cpuTicks = cpTimeToTicks(m, false);
257             if (cpuTicks.length >= 5) {
258                 ticks[TickType.USER.getIndex()] = cpuTicks[CP_USER];
259                 ticks[TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
260                 ticks[TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
261                 int offset = cpuTicks.length > 5 ? 1 : 0;
262                 ticks[TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
263                 ticks[TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
264             }
265         }
266         return ticks;
267     }

269     /**
270      * Get the processor CPU load ticks
271      *
272      * @return The processor CPU load ticks
273      */
274     @Override
275     protected long[][] queryProcessorCpuLoadTicks() {
276         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
277         int[] mib = new int[3];
278         mib[0] = CTL_KERN;
279         mib[1] = KERN_CPTIME2;
280         for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
281             mib[2] = cpu;
282             try (Memory m = OpenBsdSysctlUtil.sysctl(mib)) {
283                 // array of 5 or 6 longs
284                 long[] cpuTicks = cpTimeToTicks(m, true);
285                 if (cpuTicks.length >= 5) {
286                     ticks[cpu][TickType.USER.getIndex()] = cpuTicks[CP_USER];
287                     ticks[cpu][TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
288                     ticks[cpu][TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
289                     int offset = cpuTicks.length > 5 ? 1 : 0;
290                     ticks[cpu][TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
291                     ticks[cpu][TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
292                 }
293             }
294         }
295         return ticks;
296     }

298     /**
299      * Parse memory buffer returned from sysctl kern.cptime or kern.cptime2 to an array of 5 or 6 longs depending on
300      * version.
301      * <p>
302      * Versions 6.4 and later have a 6-element array while earlier versions have only 5 elements. Additionally
303      * kern.cptime uses a native-sized long (32- or 64-bit) value while kern.cptime2 is always a 64-bit value.
304      *
305      * @param m          A buffer containing the array.
306      * @param force64bit True if the buffer is filled with 64-bit longs, false if native long sized values
307      * @return The array
308      */
309     private static long[] cpTimeToTicks(Memory m, boolean force64bit) {
310         long longBytes = force64bit ? 8L : Native.LONG_SIZE;
311         int arraySize = m == null ? 0 : (int) (m.size() / longBytes);
312         if (force64bit && m != null) {
313             return m.getLongArray(0, arraySize);
314         }
315         long[] ticks = new long[arraySize];
316         for (int i = 0; i < arraySize; i++) {
317             ticks[i] = m.getNativeLong(i * longBytes).longValue();
318         }
319         return ticks;
320     }

322     /**
323      * Returns the system load average for the number of elements specified, up to 3, representing 1, 5, and 15 minutes.
324      * The system load average is the sum of the number of runnable entities queued to the available processors and the
325      * number of runnable entities running on the available processors averaged over a period of time. The way in which
326      * the load average is calculated is operating system specific but is typically a damped time-dependent average. If
327      * the load average is not available, a negative value is returned. This method is designed to provide a hint about
328      * the system load and may be queried frequently.
329      * <p>
330      * The load average may be unavailable on some platforms (e.g., Windows) where it is expensive to implement this
331      * method.
332      *
333      * @param nelem Number of elements to return.
334      * @return an array of the system load averages for 1, 5, and 15 minutes with the size of the array specified by
335      *         nelem; or negative values if not available.
336      */
337     @Override
338     public double[] getSystemLoadAverage(int nelem) {
339         if (nelem < 1 || nelem > 3) {
340             throw new IllegalArgumentException("Must include from one to three elements.");
341         }
342         double[] average = new double[nelem];
343         int retval = OpenBsdLibc.INSTANCE.getloadavg(average, nelem);
344         if (retval < nelem) {
345             Arrays.fill(average, -1d);
346         }
347         return average;
348     }
