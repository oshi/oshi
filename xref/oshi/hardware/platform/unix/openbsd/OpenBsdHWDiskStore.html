26  /**
27   * OpenBSD hard disk implementation.
28   */
29  @ThreadSafe
30  public final class OpenBsdHWDiskStore extends AbstractHWDiskStore {
31  
32      private final Supplier<List<String>> iostat = memoize(OpenBsdHWDiskStore::querySystatIostat, defaultExpiration());
33  
34      private OpenBsdHWDiskStore(String name, String model, String serial, long size) {
35          super(name, model, serial, size);
36      }
37  
38      /**
39       * Gets the disks on this machine.
40       *
41       * @return a list of {@link HWDiskStore} objects representing the disks
42       */
43      public static List<HWDiskStore> getDisks() {
44          List<HWDiskStore> diskList = new ArrayList<>();
45          List<String> dmesg = null; // Lazily fetch in loop if needed
46  
47          // Get list of disks from sysctl
48          // hw.disknames=sd0:2cf69345d371cd82,cd0:,sd1:
49          String[] devices = OpenBsdSysctlUtil.sysctl("hw.disknames", "").split(",");
50          OpenBsdHWDiskStore store;
51          String diskName;
52          for (String device : devices) {
53              diskName = device.split(":")[0];
54              // get partitions using disklabel command (requires root)
55              Quartet<String, String, Long, List<HWPartition>> diskdata = Disklabel.getDiskParams(diskName);
56              String model = diskdata.getA();
57              long size = diskdata.getC();
58              if (size <= 1) {
59                  if (dmesg == null) {
60                      dmesg = ExecutingCommand.runNative("dmesg");
61                  }
62                  Pattern diskAt = Pattern.compile(diskName + " at .*<(.+)>.*");
63                  Pattern diskMB = Pattern
64                          .compile(diskName + ":.* (\\d+)MB, (?:(\\d+) bytes\\/sector, )?(?:(\\d+) sectors).*");
65                  for (String line : dmesg) {
66                      Matcher m = diskAt.matcher(line);
67                      if (m.matches()) {
68                          model = m.group(1);
69                      }
70                      m = diskMB.matcher(line);
71                      if (m.matches()) {
72                          // Group 3 is sectors
73                          long sectors = ParseUtil.parseLongOrDefault(m.group(3), 0L);
74                          // Group 2 is optional capture of bytes per sector
75                          long bytesPerSector = ParseUtil.parseLongOrDefault(m.group(2), 0L);
76                          if (bytesPerSector == 0 && sectors > 0) {
77                              // if we don't have bytes per sector guess at it based on total size and number
78                              // of sectors
79                              // Group 1 is size in MB, which may round
80                              size = ParseUtil.parseLongOrDefault(m.group(1), 0L) << 20;
81                              // Estimate bytes per sector. Should be "near" a power of 2
82                              bytesPerSector = size / sectors;
83                              // Multiply by 1.5 and round down to nearest power of 2:
84                              bytesPerSector = Long.highestOneBit(bytesPerSector + (bytesPerSector >> 1));
85                          }
86                          size = bytesPerSector * sectors;
87                          break;
88                      }
89                  }
90              }
91              store = new OpenBsdHWDiskStore(diskName, model, diskdata.getB(), size);
92              store.setPartitionList(diskdata.getD());
93              store.updateAttributes();
94  
95              diskList.add(store);
96          }
97          return diskList;
98      }
99  
100     @Override
101     public boolean updateAttributes() {
102         long now = System.currentTimeMillis();
103         boolean diskFound = false;
104         for (String line : iostat.get()) {
105             String[] split = ParseUtil.whitespaces.split(line);
106             if (split.length >= 6 && split[0].equals(getName())) {
107                 diskFound = true;
108                 setReadBytes(ParseUtil.parseMultipliedToLongs(split[1]));
109                 setWriteBytes(ParseUtil.parseMultipliedToLongs(split[2]));
110                 setReads((long) ParseUtil.parseDoubleOrDefault(split[3], 0d));
111                 setWrites((long) ParseUtil.parseDoubleOrDefault(split[4], 0d));
112                 // In seconds, multiply for ms
113                 setTransferTime((long) (ParseUtil.parseDoubleOrDefault(split[5], 0d) * 1000));
114                 setTimeStamp(now);
115             }
116         }
117         return diskFound;
118     }
119 
120     private static List<String> querySystatIostat() {
121         return ExecutingCommand.runNative("systat -ab iostat");
122     }
