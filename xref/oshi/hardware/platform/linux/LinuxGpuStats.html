7   import java.io.File;
8   import java.util.Locale;

10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.hardware.GpuStats;

17  /**
18   * Linux {@link GpuStats} session. Dynamic metrics are sourced in priority order: NVML (NVIDIA GPUs), then sysfs DRM
19   * driver files under {@code /sys/class/drm/cardN/device/}. The hwmon path and driver-specific sysfs paths are resolved
20   * once at construction time.
21   *
22   * <p>
23   * GPU ticks are not available on Linux and always return {@code (0L, 0L)}. Shared memory is not available and always
24   * returns -1.
25   */
26  @ThreadSafe
27  final class LinuxGpuStats implements GpuStats {
28  
29      private final String drmDevicePath;
30      private final String driverName;
31      private final String pciBusId;
32      private final String cardName;
33  
34      // Cached at construction; empty string if not found
35      private final String hwmonPath;
36      // Cached Intel gt0 path
37      private final String gt0Path;
38  
39      // Cached NVML device id; null means not yet resolved, empty string means unavailable
40      private String nvmlDeviceId;
41  
42      private boolean closed;
43  
44      LinuxGpuStats(String drmDevicePath, String driverName, String pciBusId, String cardName) {
45          this.drmDevicePath = drmDevicePath;
46          this.driverName = driverName;
47          this.pciBusId = pciBusId;
48          this.cardName = cardName;
49          this.hwmonPath = resolveHwmonPath(drmDevicePath);
50          this.gt0Path = drmDevicePath.isEmpty() ? "" : drmDevicePath + "/../gt/gt0";
51      }
52  
53      @Override
54      public synchronized void close() {
55          closed = true;
56      }
57  
58      @Override
59      public synchronized boolean isClosed() {
60          return closed;
61      }
62  
63      @Override
64      public synchronized GpuTicks getGpuTicks() {
65          checkOpen();
66          return new GpuTicks(0L, 0L);
67      }
68  
69      @Override
70      public synchronized double getGpuUtilization() {
71          checkOpen();
72          if (drmDevicePath.isEmpty()) {
73              return -1d;
74          }
75          String driver = driverName.toLowerCase(Locale.ROOT);
76          if ("amdgpu".equals(driver)) {
77              int pct = FileUtil.getIntFromFile(drmDevicePath + "/gpu_busy_percent");
78              return pct >= 0 ? pct : -1d;
79          }
80          if ("i915".equals(driver) || "xe".equals(driver)) {
81              // Approximates utilization as actual_freq / max_freq; not a true busy-time percentage
82              // but the best available metric without kernel tracepoints.
83              long actual = FileUtil.getLongFromFile(gt0Path + "/rps_act_freq_mhz");
84              long max = FileUtil.getLongFromFile(gt0Path + "/rps_max_freq_mhz");
85              if (actual >= 0 && max > 0) {
86                  return actual == 0 ? 0.0 : Math.min(100.0, actual * 100.0 / max);
87              }
88          }
89          return -1d;
90      }
91  
92      @Override
93      public synchronized long getVramUsed() {
94          checkOpen();
95          String nvmlDevice = findNvmlDevice();
96          if (nvmlDevice != null) {
97              long val = NvmlUtil.getVramUsed(nvmlDevice);
98              if (val >= 0) {
99                  return val;
100             }
101         }
102         if (drmDevicePath.isEmpty()) {
103             return -1L;
104         }
105         if ("amdgpu".equals(driverName.toLowerCase(Locale.ROOT))) {
106             long used = FileUtil.getLongFromFile(drmDevicePath + "/mem_info_vram_used");
107             return used >= 0 ? used : -1L;
108         }
109         return -1L;
110     }
111 
112     @Override
113     public synchronized long getSharedMemoryUsed() {
114         checkOpen();
115         return -1L;
116     }
117 
118     @Override
119     public synchronized double getTemperature() {
120         checkOpen();
121         String nvmlDevice = findNvmlDevice();
122         if (nvmlDevice != null) {
123             double val = NvmlUtil.getTemperature(nvmlDevice);
124             if (val >= 0) {
125                 return val;
126             }
127         }
128         if (!hwmonPath.isEmpty()) {
129             long milliC = FileUtil.getLongFromFile(hwmonPath + "/temp1_input");
130             if (milliC >= 0) {
131                 return milliC / 1000.0;
132             }
133         }
134         return -1d;
135     }
136 
137     @Override
138     public synchronized double getPowerDraw() {
139         checkOpen();
140         String nvmlDevice = findNvmlDevice();
141         if (nvmlDevice != null) {
142             double val = NvmlUtil.getPowerDraw(nvmlDevice);
143             if (val >= 0) {
144                 return val;
145             }
146         }
147         if (!hwmonPath.isEmpty()) {
148             long microW = FileUtil.getLongFromFile(hwmonPath + "/power1_average");
149             if (microW >= 0) {
150                 return microW / 1_000_000.0;
151             }
152         }
153         return -1d;
154     }
155 
156     @Override
157     public synchronized long getCoreClockMhz() {
158         checkOpen();
159         String nvmlDevice = findNvmlDevice();
160         if (nvmlDevice != null) {
161             long val = NvmlUtil.getCoreClockMhz(nvmlDevice);
162             if (val >= 0) {
163                 return val;
164             }
165         }
166         if (drmDevicePath.isEmpty()) {
167             return -1L;
168         }
169         String driver = driverName.toLowerCase(Locale.ROOT);
170         if ("amdgpu".equals(driver)) {
171             if (!hwmonPath.isEmpty()) {
172                 long hz = FileUtil.getLongFromFile(hwmonPath + "/freq1_input");
173                 if (hz > 0) {
174                     return hz / 1_000_000L;
175                 }
176             }
177             return parseDpmActiveMhz(drmDevicePath + "/pp_dpm_sclk");
178         }
179         if ("i915".equals(driver) || "xe".equals(driver)) {
180             long mhz = FileUtil.getLongFromFile(gt0Path + "/rps_cur_freq_mhz");
181             return mhz > 0 ? mhz : -1L;
182         }
183         return -1L;
184     }
185 
186     @Override
187     public synchronized long getMemoryClockMhz() {
188         checkOpen();
189         String nvmlDevice = findNvmlDevice();
190         if (nvmlDevice != null) {
191             long val = NvmlUtil.getMemoryClockMhz(nvmlDevice);
192             if (val >= 0) {
193                 return val;
194             }
195         }
196         if (drmDevicePath.isEmpty()) {
197             return -1L;
198         }
199         if ("amdgpu".equals(driverName.toLowerCase(Locale.ROOT))) {
200             if (!hwmonPath.isEmpty()) {
201                 long hz = FileUtil.getLongFromFile(hwmonPath + "/freq2_input");
202                 if (hz > 0) {
203                     return hz / 1_000_000L;
204                 }
205             }
206             return parseDpmActiveMhz(drmDevicePath + "/pp_dpm_mclk");
207         }
208         return -1L;
209     }
210 
211     @Override
212     public synchronized double getFanSpeedPercent() {
213         checkOpen();
214         String nvmlDevice = findNvmlDevice();
215         if (nvmlDevice != null) {
216             double val = NvmlUtil.getFanSpeedPercent(nvmlDevice);
217             if (val >= 0) {
218                 return val;
219             }
220         }
221         if (!hwmonPath.isEmpty()) {
222             long fanRpm = FileUtil.getLongFromFile(hwmonPath + "/fan1_input");
223             long fanMax = FileUtil.getLongFromFile(hwmonPath + "/fan1_max");
224             if (fanRpm >= 0 && fanMax > 0) {
225                 return Math.min(100.0, fanRpm * 100.0 / fanMax);
226             }
227             long pwm = FileUtil.getLongFromFile(hwmonPath + "/pwm1");
228             if (pwm >= 0) {
229                 return pwm / 255.0 * 100.0;
230             }
231         }
232         return -1d;
233     }
234 
235     private void checkOpen() {
236         if (closed) {
237             throw new IllegalStateException(
238                     "GpuStats session has been closed. Obtain a new session via GraphicsCard.createStatsSession().");
239         }
240     }
241 
242     private String findNvmlDevice() {
243         if (nvmlDeviceId != null) {
244             return nvmlDeviceId.isEmpty() ? null : nvmlDeviceId;
245         }
246         if (!NvmlUtil.isAvailable()) {
247             nvmlDeviceId = "";
248             return null;
249         }
250         String id = null;
251         if (!pciBusId.isEmpty()) {
252             id = NvmlUtil.findDevice(pciBusId);
253         }
254         if (id == null) {
255             id = NvmlUtil.findDeviceByName(cardName);
256         }
257         nvmlDeviceId = id != null ? id : "";
258         return id;
259     }
260 
261     private static String resolveHwmonPath(String drmDevicePath) {
262         if (drmDevicePath.isEmpty()) {
263             return "";
264         }
265         File hwmonDir = new File(drmDevicePath + "/hwmon");
266         File[] entries = hwmonDir.listFiles(f -> f.getName().startsWith("hwmon"));
267         if (entries != null && entries.length > 0) {
268             return entries[0].getAbsolutePath();
269         }
270         return "";
271     }
272 
273     private static long parseDpmActiveMhz(String path) {
274         for (String line : FileUtil.readFile(path, false)) {
275             if (line.endsWith("*")) {
276                 int mhzIdx = line.toLowerCase(Locale.ROOT).indexOf("mhz");
277                 if (mhzIdx > 0) {
278                     int start = line.lastIndexOf(' ', mhzIdx - 1);
279                     if (start >= 0) {
280                         return ParseUtil.parseLongOrDefault(line.substring(start + 1, mhzIdx), -1L);
281                     }
282                 }
283             }
284         }
285         return -1L;
286     }
