31  /**
32   * JNA-based Linux hard disk implementation.
33   */
34  @ThreadSafe
35  public final class LinuxHWDiskStoreJNA extends LinuxHWDiskStore {
36  
37      private static final Logger LOG = LoggerFactory.getLogger(LinuxHWDiskStoreJNA.class);
38  
39      LinuxHWDiskStoreJNA(String name, String model, String serial, long size) {
40          super(name, model, serial, size);
41      }
42  
43      /**
44       * Gets the disks on this machine
45       *
46       * @return a list of {@link HWDiskStore} objects representing the disks
47       */
48      public static List<HWDiskStore> getDisks() {
49          return getDisks(null);
50      }
51  
52      static List<HWDiskStore> getDisks(LinuxHWDiskStore storeToUpdate) {
53          if (!HAS_UDEV) {
54              LOG.warn("Disk Store information requires libudev, which is not present.");
55              return Collections.emptyList();
56          }
57          LinuxHWDiskStoreJNA store = null;
58          List<HWDiskStore> result = new ArrayList<>();
59          Map<String, String> mountsMap = readMountsMap();
60  
61          UdevContext udev = Udev.INSTANCE.udev_new();
62          if (udev == null) {
63              return Collections.emptyList();
64          }
65          try {
66              UdevEnumerate enumerate = udev.enumerateNew();
67              try {
68                  enumerate.addMatchSubsystem(BLOCK);
69                  enumerate.scanDevices();
70                  for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
71                      String syspath = entry.getName();
72                      UdevDevice device = udev.deviceNewFromSyspath(syspath);
73                      if (device != null) {
74                          try {
75                              String devnode = device.getDevnode();
76                              if (devnode != null && !devnode.startsWith(DevPath.LOOP)
77                                      && !devnode.startsWith(DevPath.RAM)) {
78                                  if (DISK.equals(device.getDevtype())) {
79                                      String devModel = device.getPropertyValue(ID_MODEL);
80                                      String devSerial = device.getPropertyValue(ID_SERIAL_SHORT);
81                                      long devSize = ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
82                                              * SECTORSIZE;
83                                      if (devnode.startsWith(DevPath.DM)) {
84                                          devModel = LOGICAL_VOLUME_GROUP;
85                                          devSerial = device.getPropertyValue(DM_UUID);
86                                          store = new LinuxHWDiskStoreJNA(devnode, devModel,
87                                                  devSerial == null ? Constants.UNKNOWN : devSerial, devSize);
88                                          String vgName = device.getPropertyValue(DM_VG_NAME);
89                                          String lvName = device.getPropertyValue(DM_LV_NAME);
90                                          if (vgName != null && lvName != null && devSerial != null
91                                                  && devSerial.startsWith("LVM-")) {
92                                              store.getMutablePartitionList().add(new HWPartition(
93                                                      getPartitionNameForDmDevice(vgName, lvName), device.getSysname(),
94                                                      device.getPropertyValue(ID_FS_TYPE) == null ? PARTITION
95                                                              : device.getPropertyValue(ID_FS_TYPE),
96                                                      device.getPropertyValue(ID_FS_UUID) == null ? ""
97                                                              : device.getPropertyValue(ID_FS_UUID),
98                                                      device.getPropertyValue(ID_FS_LABEL) == null ? ""
99                                                              : device.getPropertyValue(ID_FS_LABEL),
100                                                     ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
101                                                             * SECTORSIZE,
102                                                     ParseUtil.parseIntOrDefault(device.getPropertyValue(MAJOR), 0),
103                                                     ParseUtil.parseIntOrDefault(device.getPropertyValue(MINOR), 0),
104                                                     getMountPointOfDmDevice(vgName, lvName)));
105                                         }
106                                     } else {
107                                         store = new LinuxHWDiskStoreJNA(devnode,
108                                                 devModel == null ? Constants.UNKNOWN : devModel,
109                                                 devSerial == null ? Constants.UNKNOWN : devSerial, devSize);
110                                     }
111                                     if (storeToUpdate == null) {
112                                         computeDiskStats(store, device.getSysattrValue(STAT));
113                                         result.add(store);
114                                     } else if (store.getName().equals(storeToUpdate.getName())
115                                             && store.getModel().equals(storeToUpdate.getModel())
116                                             && store.getSerial().equals(storeToUpdate.getSerial())
117                                             && store.getSize() == storeToUpdate.getSize()) {
118                                         computeDiskStats(storeToUpdate, device.getSysattrValue(STAT));
119                                         result.add(storeToUpdate);
120                                         break;
121                                     }
122                                 } else if (storeToUpdate == null && store != null
123                                         && PARTITION.equals(device.getDevtype())) {
124                                     UdevDevice parent = device.getParentWithSubsystemDevtype(BLOCK, DISK);
125                                     if (parent != null && store.getName().equals(parent.getDevnode())) {
126                                         String name = device.getDevnode();
127                                         store.getMutablePartitionList().add(new HWPartition(name, device.getSysname(),
128                                                 device.getPropertyValue(ID_FS_TYPE) == null ? PARTITION
129                                                         : device.getPropertyValue(ID_FS_TYPE),
130                                                 device.getPropertyValue(ID_FS_UUID) == null ? ""
131                                                         : device.getPropertyValue(ID_FS_UUID),
132                                                 device.getPropertyValue(ID_FS_LABEL) == null ? ""
133                                                         : device.getPropertyValue(ID_FS_LABEL),
134                                                 ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
135                                                         * SECTORSIZE,
136                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MAJOR), 0),
137                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MINOR), 0),
138                                                 mountsMap.getOrDefault(name,
139                                                         getDependentNamesFromHoldersDirectory(device.getSyspath()))));
140                                     }
141                                 }
142                             }
143                         } finally {
144                             device.unref();
145                         }
146                     }
147                 }
148             } finally {
149                 enumerate.unref();
150             }
151         } finally {
152             udev.unref();
153         }
154         finalizePartitions(result);
155         return result;
156     }
157 
158     @Override
159     public boolean updateAttributes() {
160         return !getDisks(this).isEmpty();
161     }
162 }
