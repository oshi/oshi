5   package oshi.hardware.platform.linux;
6   
7   import java.util.List;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.GpuStats;
11  import oshi.hardware.GraphicsCard;
12  import oshi.hardware.common.platform.linux.LinuxGraphicsCard;
13  
14  /**
15   * JNA-based Linux graphics card with NVML-backed GPU stats.
16   */
17  @ThreadSafe
18  final class LinuxGraphicsCardJNA extends LinuxGraphicsCard {
19  
20      LinuxGraphicsCardJNA(String name, String deviceId, String vendor, String versionInfo, long vram,
21              String drmDevicePath, String driverName, String pciBusId) {
22          super(name, deviceId, vendor, versionInfo, vram, drmDevicePath, driverName, pciBusId);
23      }
24  
25      @Override
26      public GpuStats createStatsSession() {
27          return new LinuxGpuStatsJNA(getDrmDevicePath(), getDriverName(), getPciBusId(), getName());
28      }
29  
30      /**
31       * Gets graphics cards using JNA-based GPU stats.
32       *
33       * @return list of graphics cards
34       */
35      public static List<GraphicsCard> getGraphicsCards() {
36          return LinuxGraphicsCard
37                  .getGraphicsCards(a -> new LinuxGraphicsCardJNA(a.getName(), a.getDeviceId(), a.getVendor(),
38                          a.getVersionInfo(), a.getVram(), a.getDrmDevicePath(), a.getDriverName(), a.getPciBusId()));
39      }
40  }
