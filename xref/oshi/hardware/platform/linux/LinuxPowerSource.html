48      /**
49       * Initialization-on-demand holder: the map is only constructed if the non-udev path is reached, avoiding
50       * unnecessary work on systems where udev is available.
51       */
52      private static final class PropByName {
53          static final Map<String, Prop> MAP = new HashMap<>();
54          static {
55              for (Prop p : Prop.values()) {
56                  MAP.put(p.name(), p);
57              }
58          }
59      }

73      /**
74       * Gets Battery Information
75       *
76       * @return An array of PowerSource objects representing batteries, etc.
77       */
78      public static List<PowerSource> getPowerSources() {
79          List<PowerSource> psList = new ArrayList<>();
80          if (HAS_UDEV) {
81              UdevContext udev = Udev.INSTANCE.udev_new();
82              try {
83                  UdevEnumerate enumerate = udev.enumerateNew();
84                  try {
85                      enumerate.addMatchSubsystem("power_supply");
86                      enumerate.scanDevices();
87                      for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
88                          String syspath = entry.getName();
89                          String name = syspath.substring(syspath.lastIndexOf(File.separatorChar) + 1);
90                          if (!name.startsWith("ADP") && !name.startsWith("AC") && !name.contains("USBC")) {
91                              UdevDevice device = udev.deviceNewFromSyspath(syspath);
92                              if (device != null) {
93                                  try {
94                                      if (ParseUtil.parseIntOrDefault(
95                                              device.getPropertyValue(Prop.POWER_SUPPLY_PRESENT.name()), 1) > 0
96                                              && ParseUtil.parseIntOrDefault(
97                                                      device.getPropertyValue(Prop.POWER_SUPPLY_ONLINE.name()), 1) > 0) {
98                                          Map<Prop, String> props = new EnumMap<>(Prop.class);
99                                          for (Prop p : Prop.values()) {
100                                             String val = device.getPropertyValue(p.name());
101                                             if (val != null) {
102                                                 props.put(p, val);
103                                             }
104                                         }
105                                         psList.add(buildPowerSource(name, props));
106                                     }
107                                 } finally {
108                                     device.unref();
109                                 }
110                             }
111                         }
112                     }
113                 } finally {
114                     enumerate.unref();
115                 }
116             } finally {
117                 udev.unref();
118             }
119         } else {
120             File psDir = new File(SysPath.POWER_SUPPLY);
121             File[] psArr = psDir.listFiles();
122             if (psArr == null) {
123                 return Collections.emptyList();
124             }
125             for (File ps : psArr) {
126                 String name = ps.getName();
127                 if (!name.startsWith("ADP") && !name.startsWith("AC") && !name.contains("USBC")) {
128                     List<String> psInfo = FileUtil.readFile(SysPath.POWER_SUPPLY + "/" + name + "/uevent", false);
129                     Map<Prop, String> props = new EnumMap<>(Prop.class);
130                     for (String line : psInfo) {
131                         String[] split = line.split("=", 2);
132                         if (split.length > 1 && !split[1].isEmpty()) {
133                             Prop p = PropByName.MAP.get(split[0]);
134                             if (p != null) {
135                                 props.put(p, split[1]);
136                             }
137                         }
138                     }
139                     if (ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_PRESENT), 1) > 0) {
140                         psList.add(buildPowerSource(name, props));
141                     }
142                 }
143             }
144         }
145         return psList;
146     }

148     static LinuxPowerSource buildPowerSource(String name, Map<Prop, String> props) {
149         String psName = props.getOrDefault(Prop.POWER_SUPPLY_NAME, name);
150         String status = props.get(Prop.POWER_SUPPLY_STATUS);
151         boolean psCharging = "Charging".equals(status);
152         boolean psDischarging = "Discharging".equals(status);
153         double psRemainingCapacityPercent = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CAPACITY), -100)
154                 / 100d;
155 
156         // Debian/Ubuntu provides Energy (µWh). Fedora/RHEL provides Charge (µAh).
157         // Parse raw values in micro-units first, fall back across Energy/Charge, then
158         // divide by 1000 at the end to avoid integer division truncating the -1 sentinel.
159         int rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_NOW), -1);
160         CapacityUnits psCapacityUnits;
161         if (rawNow >= 0) {
162             psCapacityUnits = CapacityUnits.MWH;
163         } else {
164             rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_NOW), -1);
165             psCapacityUnits = CapacityUnits.MAH;
166         }
167         int psCurrentCapacity = rawNow >= 0 ? rawNow / 1000 : -1;
168 
169         int rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL), -1);
170         if (rawFull < 0) {
171             rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL), -1);
172         }
173         int psMaxCapacity = rawFull >= 0 ? rawFull / 1000 : -1;
174 
175         int rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL_DESIGN), -1);
176         if (rawDesign < 0) {
177             rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL_DESIGN), -1);
178         }
179         int psDesignCapacity = rawDesign >= 0 ? rawDesign / 1000 : -1;
180 
181         // Debian/Ubuntu provides Voltage and Power. Fedora/RHEL provides Voltage and Current.
182         double psVoltage = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_VOLTAGE_NOW), -1d) / 1_000_000d;
183         double psPowerUsageRate = 0d;
184         double psAmperage = 0d;
185         // From Physics we know P = IV so I = P/V
186         // Linux always reports POWER_NOW and CURRENT_NOW as non-negative; sign is
187         // conveyed by STATUS. Negate when discharging to match the PowerSource contract
188         // (positive = charging, negative = discharging).
189         double sign = psDischarging ? -1d : 1d;
190         if (psVoltage > 0) {
191             double rawPower = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_POWER_NOW), -1d);
192             double rawCurrent = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_CURRENT_NOW), -1d);
193             if (rawPower >= 0) {
194                 psPowerUsageRate = sign * rawPower / 1000d;
195             }
196             if (rawCurrent >= 0) {
197                 psAmperage = sign * rawCurrent / 1000d;
198             }
199             if (rawPower < 0 && rawCurrent >= 0) {
200                 psPowerUsageRate = psAmperage * psVoltage;
201             } else if (rawCurrent < 0 && rawPower >= 0) {
202                 psAmperage = psPowerUsageRate / psVoltage;
203             }
204         }
205 
206         int psCycleCount = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CYCLE_COUNT), -1);
207         String psChemistry = props.getOrDefault(Prop.POWER_SUPPLY_TECHNOLOGY, UNKNOWN);
208         String psDeviceName = props.getOrDefault(Prop.POWER_SUPPLY_MODEL_NAME, UNKNOWN);
209         String psManufacturer = props.getOrDefault(Prop.POWER_SUPPLY_MANUFACTURER, UNKNOWN);
210         String psSerialNumber = props.getOrDefault(Prop.POWER_SUPPLY_SERIAL_NUMBER, UNKNOWN);
211 
212         // TEMP is in tenths of degrees Celsius
213         int rawTemp = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_TEMP), -1);
214         double psTemperature = rawTemp >= 0 ? rawTemp / 10d : 0d;
215 
216         // TIME_TO_EMPTY_NOW when discharging, TIME_TO_FULL_NOW when charging; both in seconds
217         Prop timeKey = psCharging ? Prop.POWER_SUPPLY_TIME_TO_FULL_NOW : Prop.POWER_SUPPLY_TIME_TO_EMPTY_NOW;
218         double psTimeRemainingInstant = ParseUtil.parseIntOrDefault(props.get(timeKey), -1);
219 
220         int year = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_YEAR), -1);
221         int month = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_MONTH), -1);
222         int day = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_DAY), -1);
223         LocalDate psManufactureDate = (year > 0 && month > 0 && day > 0) ? LocalDate.of(year, month, day) : null;
224 
225         return new LinuxPowerSource(psName, psDeviceName, psRemainingCapacityPercent, -1d, psTimeRemainingInstant,
226                 psPowerUsageRate, psVoltage, psAmperage, false, psCharging, psDischarging, psCapacityUnits,
227                 psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
228                 psManufacturer, psSerialNumber, psTemperature);
229     }
