42  
43  public class LinuxHardwareAbstractionLayer implements HardwareAbstractionLayer {
44      private static final Logger LOG = LoggerFactory.getLogger(LinuxHardwareAbstractionLayer.class);
45  
46      private static final String SEPARATOR = "\\s+:\\s";
47  
48      private CentralProcessor processor;
49  
50      private GlobalMemory _memory;
51  
52      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
53  
54      @Override
55      public GlobalMemory getMemory() {
56          if (this._memory == null) {
57              this._memory = new LinuxGlobalMemory();
58          }
59          return this._memory;
60      }
61  
62      @Override
63      public CentralProcessor getProcessor() {
64  
65          if (this.processor == null) {
66              List<String> cpuInfo = null;
67              try {
68                  cpuInfo = FileUtil.readFile("/proc/cpuinfo");
69              } catch (IOException e) {
70                  LOG.error("Problem with /proc/cpuinfo: {}", e.getMessage());
71                  return null;
72              }
73              for (String toBeAnalyzed : cpuInfo) {
74                  if (toBeAnalyzed.equals("")) {
75                      break;
76                  }
77                  if (processor == null) {
78                      processor = new LinuxCentralProcessor();
79                  }
80                  if (toBeAnalyzed.startsWith("model name\t")) {
81                      processor.setName(toBeAnalyzed.split(SEPARATOR)[1]);
82                      continue;
83                  }
84                  if (toBeAnalyzed.startsWith("flags\t")) {
85                      String[] flags = toBeAnalyzed.split(SEPARATOR)[1].split(" ");
86                      boolean found = false;
87                      for (String flag : flags) {
88                          if (flag.equalsIgnoreCase("LM")) {
89                              found = true;
90                              break;
91                          }
92                      }
93                      processor.setCpu64(found);
94                      continue;
95                  }
96                  if (toBeAnalyzed.startsWith("cpu family\t")) {
97                      processor.setFamily(toBeAnalyzed.split(SEPARATOR)[1]);
98                      continue;
99                  }
100                 if (toBeAnalyzed.startsWith("model\t")) {
101                     processor.setModel(toBeAnalyzed.split(SEPARATOR)[1]);
102                     continue;
103                 }
104                 if (toBeAnalyzed.startsWith("stepping\t")) {
105                     processor.setStepping(toBeAnalyzed.split(SEPARATOR)[1]);
106                     continue;
107                 }
108                 if (toBeAnalyzed.startsWith("vendor_id")) {
109                     processor.setVendor(toBeAnalyzed.split(SEPARATOR)[1]);
110                     continue;
111                 }
112             }
113         }
114         return this.processor;
115     }
116 
117     @Override
118     public PowerSource[] getPowerSources() {
119         return LinuxPowerSource.getPowerSources();
120     }
121 
122     @Override
123     public OSFileStore[] getFileStores() {
124         return LinuxFileSystem.getFileStores();
125     }
126 
127     @Override
128     public Display[] getDisplays() {
129         return LinuxDisplay.getDisplays();
130     }
131 
132     @Override
133     public JsonObject toJSON() {
134         JsonArrayBuilder powerSourceArrayBuilder = jsonFactory.createArrayBuilder();
135         for (PowerSource powerSource : getPowerSources()) {
136             powerSourceArrayBuilder.add(powerSource.toJSON());
137         }
138         JsonArrayBuilder fileStoreArrayBuilder = jsonFactory.createArrayBuilder();
139         for (OSFileStore fileStore : getFileStores()) {
140             fileStoreArrayBuilder.add(fileStore.toJSON());
141         }
142         JsonArrayBuilder displayArrayBuilder = jsonFactory.createArrayBuilder();
143         for (Display display : getDisplays()) {
144             displayArrayBuilder.add(display.toJSON());
145         }
146         return jsonFactory.createObjectBuilder().add("processor", getProcessor().toJSON())
147                 .add("memory", getMemory().toJSON()).add("powerSources", powerSourceArrayBuilder.build())
148                 .add("fileStores", fileStoreArrayBuilder.build()).add("displays", displayArrayBuilder.build()).build();
149     }
150 }
