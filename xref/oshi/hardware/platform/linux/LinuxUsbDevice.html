35  public class LinuxUsbDevice extends AbstractUsbDevice {
36  
37      private static final long serialVersionUID = 2L;
38  
39      /*
40       * Maps to store information using device node path as the key
41       */
42      private static Map<String, String> nameMap = new HashMap<>();
43      private static Map<String, String> vendorMap = new HashMap<>();
44      private static Map<String, String> vendorIdMap = new HashMap<>();
45      private static Map<String, String> productIdMap = new HashMap<>();
46      private static Map<String, String> serialMap = new HashMap<>();
47      private static Map<String, List<String>> hubMap = new HashMap<>();
48  
49      public LinuxUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
50              UsbDevice[] connectedDevices) {
51          super(name, vendor, vendorId, productId, serialNumber, connectedDevices);
52      }
53  
54      /**
55       * {@inheritDoc}
56       */
57      public static UsbDevice[] getUsbDevices(boolean tree) {
58          UsbDevice[] devices = getUsbDevices();
59          if (tree) {
60              return devices;
61          }
62          List<UsbDevice> deviceList = new ArrayList<>();
63          // Top level is controllers; they won't be added to the list, but all
64          // their connected devices will be
65          for (UsbDevice device : devices) {
66              deviceList.add(new LinuxUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
67                      device.getProductId(), device.getSerialNumber(), new MacUsbDevice[0]));
68              addDevicesToList(deviceList, device.getConnectedDevices());
69          }
70          return deviceList.toArray(new UsbDevice[deviceList.size()]);
71      }
72  
73      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
74          for (UsbDevice device : connectedDevices) {
75              deviceList.add(device);
76              addDevicesToList(deviceList, device.getConnectedDevices());
77          }
78      }
79  
80      private static UsbDevice[] getUsbDevices() {
81          // Enumerate all usb devices and build information maps
82          Udev.UdevHandle udev = Udev.INSTANCE.udev_new();
83          // Create a list of the devices in the 'usb' subsystem.
84          UdevEnumerate enumerate = Udev.INSTANCE.udev_enumerate_new(udev);
85          Udev.INSTANCE.udev_enumerate_add_match_subsystem(enumerate, "usb");
86          Udev.INSTANCE.udev_enumerate_scan_devices(enumerate);
87          UdevListEntry devices = Udev.INSTANCE.udev_enumerate_get_list_entry(enumerate);
88  
89          // Build a list of devices with no parent; these will be the roots
90          List<String> usbControllers = new ArrayList<>();
91          // Empty out maps
92          nameMap.clear();
93          vendorMap.clear();
94          vendorIdMap.clear();
95          productIdMap.clear();
96          serialMap.clear();
97          hubMap.clear();
98  
99          // For each item enumerated, store information in the maps
100         for (UdevListEntry dev_list_entry = devices; dev_list_entry != null; dev_list_entry = Udev.INSTANCE
101                 .udev_list_entry_get_next(dev_list_entry)) {
102 
103             // Get the filename of the /sys entry for the device and create a
104             // udev_device object (dev) representing it
105             String path = Udev.INSTANCE.udev_list_entry_get_name(dev_list_entry);
106             UdevDevice dev = Udev.INSTANCE.udev_device_new_from_syspath(udev, path);
107             // Ignore interfaces
108             if (!"usb_device".equals(Udev.INSTANCE.udev_device_get_devtype(dev))) {
109                 continue;
110             }
111 
112             // Use the path as the key for the maps
113             String value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "product");
114             if (value != null) {
115                 nameMap.put(path, value);
116             }
117             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "manufacturer");
118             if (value != null) {
119                 vendorMap.put(path, value);
120             }
121             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "idVendor");
122             if (value != null) {
123                 vendorIdMap.put(path, value);
124             }
125             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "idProduct");
126             if (value != null) {
127                 productIdMap.put(path, value);
128             }
129             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "serial");
130             if (value != null) {
131                 serialMap.put(path, value);
132             }
133             UdevDevice parent = Udev.INSTANCE.udev_device_get_parent_with_subsystem_devtype(dev, "usb", "usb_device");
134             if (parent == null) {
135                 // This is a controller with no parent, add to list
136                 usbControllers.add(path);
137             } else {
138                 // Add child path (path variable) to parent's path
139                 String parentPath = Udev.INSTANCE.udev_device_get_syspath(parent);
140                 hubMap.computeIfAbsent(parentPath, k -> new ArrayList<String>()).add(path);
141             }
142             Udev.INSTANCE.udev_device_unref(dev);
143         }
144         // Free the enumerator object
145         Udev.INSTANCE.udev_enumerate_unref(enumerate);
146         Udev.INSTANCE.udev_unref(udev);
147 
148         // Build tree and return
149         List<UsbDevice> controllerDevices = new ArrayList<>();
150         for (String controller : usbControllers) {
151             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000"));
152         }
153         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
154     }
155 
156     /**
157      * Recursively creates LinuxUsbDevices by fetching information from maps to
158      * populate fields
159      *
160      * @param devPath
161      *            The device node path.
162      * @param vid
163      *            The default (parent) vendor ID
164      * @param pid
165      *            The default (parent) product ID
166      * @return A LinuxUsbDevice corresponding to this device
167      */
168     private static LinuxUsbDevice getDeviceAndChildren(String devPath, String vid, String pid) {
169         String vendorId = vendorIdMap.getOrDefault(devPath, vid);
170         String productId = productIdMap.getOrDefault(devPath, pid);
171         List<String> childPaths = hubMap.getOrDefault(devPath, new ArrayList<String>());
172         List<LinuxUsbDevice> usbDevices = new ArrayList<>();
173         for (String path : childPaths) {
174             usbDevices.add(getDeviceAndChildren(path, vendorId, productId));
175         }
176         Collections.sort(usbDevices);
177         return new LinuxUsbDevice(nameMap.getOrDefault(devPath, vendorId + ":" + productId),
178                 vendorMap.getOrDefault(devPath, ""), vendorId, productId, serialMap.getOrDefault(devPath, ""),
179                 usbDevices.toArray(new UsbDevice[usbDevices.size()]));
180     }
