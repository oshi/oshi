19  package oshi.hardware.platform.linux;
20  
21  import java.io.File;
22  import java.nio.file.Files;
23  import java.nio.file.Paths;
24  import java.util.ArrayList;
25  import java.util.HashMap;
26  import java.util.List;
27  import java.util.Map;
28  
29  import oshi.hardware.Sensors;
30  import oshi.util.FileUtil;
31  
32  public class LinuxSensors implements Sensors {
33  
34      private static final long serialVersionUID = 1L;
35  
36      // Possible sensor types. See sysfs documentation for others, e.g. current
37      private static final String TEMP = "temp";
38      private static final String FAN = "fan";
39      private static final String VOLTAGE = "in";
40      private static final String[] SENSORS = { TEMP, FAN, VOLTAGE };
41  
42      // Base HWMON path, adds 0, 1, etc. to end for various sensors
43      private static final String HWMON = "/sys/class/hwmon/hwmon";
44  
45      // Map from sensor to path
46      private Map<String, String> hwmonMap = new HashMap<>();
47  
48      public LinuxSensors() {
49          // Iterate over all hwmon* directories and look for sensor files
50          // e.g. /sys/class/hwmon/hwmon0/temp1_input
51          int i = 0;
52          while (Files.isDirectory(Paths.get(HWMON + i))) {
53              for (String sensor : SENSORS) {
54                  String path = HWMON + i;
55                  // Final to pass to anonymous class
56                  final String prefix = sensor;
57                  // Find any *_input files in that path
58                  File dir = new File(path);
59                  File[] matchingFiles = dir
60                          .listFiles(f -> f.getName().startsWith(prefix) && f.getName().endsWith("_input"));
61                  if (matchingFiles != null && matchingFiles.length > 0) {
62                      this.hwmonMap.put(sensor, String.format("%s/%s", path, sensor));
63                  }
64              }
65              i++;
66          }
67      }
68  
69      /**
70       * {@inheritDoc}
71       */
72      @Override
73      public double getCpuTemperature() {
74          if (!this.hwmonMap.containsKey(TEMP)) {
75              return 0d;
76          }
77          String hwmon = this.hwmonMap.get(TEMP);
78          // First attempt should be CPU temperature at index 1, if available
79          long millidegrees = FileUtil.getLongFromFile(String.format("%s1_input", hwmon));
80          // Should return a single line of millidegrees Celsius
81          if (millidegrees > 0) {
82              return millidegrees / 1000d;
83          }
84          // If temp1_input doesn't exist, iterate over temp2..temp6_input
85          // and average
86          int sum = 0;
87          int count = 0;
88          for (int i = 2; i <= 6; i++) {
89              millidegrees = FileUtil.getLongFromFile(String.format("%s%d_input", hwmon, i));
90              if (millidegrees > 0) {
91                  sum += millidegrees;
92                  count++;
93              }
94          }
95          if (count > 0) {
96              return sum / (count * 1000d);
97          }
98          return 0d;
99      }
100 
101     /**
102      * {@inheritDoc}
103      */
104     @Override
105     public int[] getFanSpeeds() {
106         if (this.hwmonMap.containsKey(FAN)) {
107             String hwmon = this.hwmonMap.get(FAN);
108             List<Integer> speeds = new ArrayList<>();
109             int fan = 1;
110             for (;;) {
111                 String fanPath = String.format("%s%d_input", hwmon, fan);
112                 if (!new File(fanPath).exists()) {
113                     // No file found, we've reached max fans
114                     break;
115                 }
116                 // Should return a single line of RPM
117                 speeds.add(FileUtil.getIntFromFile(fanPath));
118                 // Done reading data for current fan, read next fan
119                 fan++;
120             }
121             int[] fanSpeeds = new int[speeds.size()];
122             for (int i = 0; i < speeds.size(); i++) {
123                 fanSpeeds[i] = speeds.get(i);
124             }
125             return fanSpeeds;
126         }
127         return new int[0];
128     }
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     public double getCpuVoltage() {
135         if (this.hwmonMap.containsKey(VOLTAGE)) {
136             String hwmon = this.hwmonMap.get(VOLTAGE);
137             // Should return a single line of millidegrees Celsius
138             return FileUtil.getIntFromFile(String.format("%s1_input", hwmon)) / 1000d;
139         }
140         return 0d;
141     }
142 }
