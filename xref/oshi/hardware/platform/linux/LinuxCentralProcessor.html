62      @Override
63      protected ProcessorIdentifier queryProcessorId() {
64          String cpuVendor = "";
65          String cpuName = "";
66          String cpuFamily = "";
67          String cpuModel = "";
68          String cpuStepping = "";
69          String processorID;
70          long cpuFreq = 0L;
71          boolean cpu64bit = false;
72  
73          StringBuilder armStepping = new StringBuilder(); // For ARM equivalent
74          String[] flags = new String[0];
75          List<String> cpuInfo = FileUtil.readFile(CPUINFO);
76          for (String line : cpuInfo) {
77              String[] splitLine = ParseUtil.whitespacesColonWhitespace.split(line);
78              if (splitLine.length < 2) {
79                  // special case
80                  if (line.startsWith("CPU architecture: ")) {
81                      cpuFamily = line.replace("CPU architecture: ", "").trim();
82                  }
83                  continue;
84              }
85              switch (splitLine[0].toLowerCase(Locale.ROOT)) {
86              case "vendor_id":
87              case "cpu implementer":
88                  cpuVendor = splitLine[1];
89                  break;
90              case "model name":
91              case "processor": // some ARM chips
92                  // Ignore processor number
93                  if (!splitLine[1].matches("[0-9]+")) {
94                      cpuName = splitLine[1];
95                  }
96                  break;
97              case "flags":
98                  flags = splitLine[1].toLowerCase(Locale.ROOT).split(" ");
99                  for (String flag : flags) {
100                     if ("lm".equals(flag)) {
101                         cpu64bit = true;
102                         break;
103                     }
104                 }
105                 break;
106             case "stepping":
107                 cpuStepping = splitLine[1];
108                 break;
109             case "cpu variant":
110                 if (!armStepping.toString().startsWith("r")) {
111                     // CPU variant format always starts with 0x
112                     int rev = ParseUtil.parseLastInt(splitLine[1], 0);
113                     armStepping.insert(0, "r" + rev);
114                 }
115                 break;
116             case "cpu revision":
117                 if (!armStepping.toString().contains("p")) {
118                     armStepping.append('p').append(splitLine[1]);
119                 }
120                 break;
121             case "model":
122             case "cpu part":
123                 cpuModel = splitLine[1];
124                 break;
125             case "cpu family":
126                 cpuFamily = splitLine[1];
127                 break;
128             case "cpu mhz":
129                 cpuFreq = ParseUtil.parseHertz(splitLine[1]);
130                 break;
131             default:
132                 // Do nothing
133             }
134         }
135         if (cpuName.isEmpty()) {
136             cpuName = FileUtil.getStringFromFile(MODEL);
137         }
138         if (cpuName.contains("Hz")) {
139             // if Name contains CPU vendor frequency, ignore cpuinfo and use it
140             cpuFreq = -1L;
141         } else {
142             // Try lshw and use it in preference to cpuinfo
143             long cpuCapacity = Lshw.queryCpuCapacity();
144             if (cpuCapacity > cpuFreq) {
145                 cpuFreq = cpuCapacity;
146             }
147         }
148         if (cpuStepping.isEmpty()) {
149             cpuStepping = armStepping.toString();
150         }
151         processorID = getProcessorID(cpuVendor, cpuStepping, cpuModel, cpuFamily, flags);
152         if (cpuVendor.startsWith("0x") || cpuModel.isEmpty()) {
153             List<String> lscpu = ExecutingCommand.runNative("lscpu");
154             for (String line : lscpu) {
155                 if (line.startsWith("Architecture:") && cpuVendor.startsWith("0x")) {
156                     cpuVendor = line.replace("Architecture:", "").trim();
157                 } else if (line.startsWith("Vendor ID:")) {
158                     cpuVendor = line.replace("Vendor ID:", "").trim();
159                 } else if (line.startsWith("Model name:") && cpuModel.isEmpty()) {
160                     cpuModel = line.replace("Model name:", "").trim();
161                 }
162             }
163         }
164         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
165                 cpuFreq);
166     }

168     @Override
169     protected Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> initProcessorCounts() {
170         // Attempt to read from sysfs
171         Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> topology = HAS_UDEV
172                 ? readTopologyFromUdev()
173                 : readTopologyFromSysfs();
174         // This sometimes fails so fall back to CPUID
175         if (topology.getA().isEmpty()) {
176             topology = readTopologyFromCpuinfo();
177         }
178         List<LogicalProcessor> logProcs = topology.getA();
179         List<ProcessorCache> caches = topology.getB();
180         Map<Integer, Integer> coreEfficiencyMap = topology.getC();
181         Map<Integer, String> modAliasMap = topology.getD();
182         // Failsafe
183         if (logProcs.isEmpty()) {
184             logProcs.add(new LogicalProcessor(0, 0, 0));
185         }
186         if (coreEfficiencyMap.isEmpty()) {
187             coreEfficiencyMap.put(0, 0);
188         }
189         // Sort
190         logProcs.sort(Comparator.comparingInt(LogicalProcessor::getProcessorNumber));
191 
192         List<PhysicalProcessor> physProcs = coreEfficiencyMap.entrySet().stream().sorted(Map.Entry.comparingByKey())
193                 .map(e -> {
194                     int pkgId = e.getKey() >> 16;
195                     int coreId = e.getKey() & 0xffff;
196                     return new PhysicalProcessor(pkgId, coreId, e.getValue(), modAliasMap.getOrDefault(e.getKey(), ""));
197                 }).collect(Collectors.toList());
198 
199         return new Triplet<>(logProcs, physProcs, caches);
200     }

202     private static Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyFromUdev() {
203         List<LogicalProcessor> logProcs = new ArrayList<>();
204         Set<ProcessorCache> caches = new HashSet<>();
205         Map<Integer, Integer> coreEfficiencyMap = new HashMap<>();
206         Map<Integer, String> modAliasMap = new HashMap<>();
207         // Enumerate CPU topology from sysfs via udev
208         UdevContext udev = Udev.INSTANCE.udev_new();
209         try {
210             UdevEnumerate enumerate = udev.enumerateNew();
211             try {
212                 enumerate.addMatchSubsystem("cpu");
213                 enumerate.scanDevices();
214                 for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
215                     String syspath = entry.getName(); // /sys/devices/system/cpu/cpuX
216                     UdevDevice device = udev.deviceNewFromSyspath(syspath);
217                     String modAlias = null;
218                     if (device != null) {
219                         try {
220                             modAlias = device.getPropertyValue("MODALIAS");
221                         } finally {
222                             device.unref();
223                         }
224                     }
225                     logProcs.add(
226                             getLogicalProcessorFromSyspath(syspath, caches, modAlias, coreEfficiencyMap, modAliasMap));
227                 }
228             } finally {
229                 enumerate.unref();
230             }
231         } finally {
232             udev.unref();
233         }
234         return new Quartet<>(logProcs, orderedProcCaches(caches), coreEfficiencyMap, modAliasMap);
235     }

237     private static Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyFromSysfs() {
238         List<LogicalProcessor> logProcs = new ArrayList<>();
239         Set<ProcessorCache> caches = new HashSet<>();
240         Map<Integer, Integer> coreEfficiencyMap = new HashMap<>();
241         Map<Integer, String> modAliasMap = new HashMap<>();
242         String cpuPath = "/sys/devices/system/cpu/";
243         try {
244             try (Stream<Path> cpuFiles = Files.find(Paths.get(cpuPath), Integer.MAX_VALUE,
245                     (path, basicFileAttributes) -> path.toFile().getName().matches("cpu\\d+"))) {
246                 cpuFiles.forEach(cpu -> {
247                     String syspath = cpu.toString(); // /sys/devices/system/cpu/cpuX
248                     Map<String, String> uevent = FileUtil.getKeyValueMapFromFile(syspath + "/uevent", "=");
249                     String modAlias = uevent.get("MODALIAS");
250                     // updates caches as a side-effect
251                     logProcs.add(
252                             getLogicalProcessorFromSyspath(syspath, caches, modAlias, coreEfficiencyMap, modAliasMap));
253                 });
254             }
255         } catch (IOException e) {
256             // No udev and no cpu info in sysfs? Bad.
257             LOG.warn("Unable to find CPU information in sysfs at path {}", cpuPath);
258         }
259         return new Quartet<>(logProcs, orderedProcCaches(caches), coreEfficiencyMap, modAliasMap);
260     }

262     private static LogicalProcessor getLogicalProcessorFromSyspath(String syspath, Set<ProcessorCache> caches,
263             String modAlias, Map<Integer, Integer> coreEfficiencyMap, Map<Integer, String> modAliasMap) {
264         int processor = ParseUtil.getFirstIntValue(syspath);
265         int coreId = FileUtil.getIntFromFile(syspath + "/topology/core_id");
266         int pkgId = FileUtil.getIntFromFile(syspath + "/topology/physical_package_id");
267         int pkgCoreKey = (pkgId << 16) + coreId;
268         // The cpu_capacity value may not exist, this will just store 0
269         coreEfficiencyMap.put(pkgCoreKey, FileUtil.getIntFromFile(syspath + "/cpu_capacity"));
270         if (!Util.isBlank(modAlias)) {
271             modAliasMap.put(pkgCoreKey, modAlias);
272         }
273         int nodeId = 0;
274         final String nodePrefix = syspath + "/node";
275         try (Stream<Path> path = Files.list(Paths.get(syspath))) {
276             Optional<Path> first = path.filter(p -> p.toString().startsWith(nodePrefix)).findFirst();
277             if (first.isPresent()) {
278                 nodeId = ParseUtil.getFirstIntValue(first.get().getFileName().toString());
279             }
280         } catch (IOException e) {
281             // ignore
282         }
283         final String cachePath = syspath + "/cache";
284         final String indexPrefix = cachePath + "/index";
285         try (Stream<Path> path = Files.list(Paths.get(cachePath))) {
286             path.filter(p -> p.toString().startsWith(indexPrefix)).forEach(c -> {
287                 int level = FileUtil.getIntFromFile(c + "/level"); // 1
288                 Type type = parseCacheType(FileUtil.getStringFromFile(c + "/type")); // Data
289                 int associativity = FileUtil.getIntFromFile(c + "/ways_of_associativity"); // 8
290                 int lineSize = FileUtil.getIntFromFile(c + "/coherency_line_size"); // 64
291                 long size = ParseUtil.parseDecimalMemorySizeToBinary(FileUtil.getStringFromFile(c + "/size")); // 32K
292                 caches.add(new ProcessorCache(level, associativity, lineSize, size, type));
293             });
294         } catch (IOException e) {
295             // ignore
296         }
297         return new LogicalProcessor(processor, coreId, pkgId, nodeId);
298     }

300     private static ProcessorCache.Type parseCacheType(String type) {
301         try {
302             return ProcessorCache.Type.valueOf(type.toUpperCase(Locale.ROOT));
303         } catch (IllegalArgumentException e) {
304             return ProcessorCache.Type.UNIFIED;
305         }

308     private static Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyFromCpuinfo() {
309         List<LogicalProcessor> logProcs = new ArrayList<>();
310         Set<ProcessorCache> caches = mapCachesFromLscpu();
311         Map<Integer, Integer> numaNodeMap = mapNumaNodesFromLscpu();
312         Map<Integer, Integer> coreEfficiencyMap = new HashMap<>();
313 
314         List<String> procCpu = FileUtil.readFile(CPUINFO);
315         int currentProcessor = 0;
316         int currentCore = 0;
317         int currentPackage = 0;
318 
319         boolean first = true;
320         for (String cpu : procCpu) {
321             // Count logical processors
322             if (cpu.startsWith("processor")) {
323                 if (first) {
324                     first = false;
325                 } else {
326                     // add from the previous iteration
327                     logProcs.add(new LogicalProcessor(currentProcessor, currentCore, currentPackage,
328                             numaNodeMap.getOrDefault(currentProcessor, 0)));
329                     // Count unique combinations of core id and physical id.
330                     coreEfficiencyMap.put((currentPackage << 16) + currentCore, 0);
331                 }
332                 // start creating for this iteration
333                 currentProcessor = ParseUtil.parseLastInt(cpu, 0);
334             } else if (cpu.startsWith("core id") || cpu.startsWith("cpu number")) {
335                 currentCore = ParseUtil.parseLastInt(cpu, 0);
336             } else if (cpu.startsWith("physical id")) {
337                 currentPackage = ParseUtil.parseLastInt(cpu, 0);
338             }
339         }
340         logProcs.add(new LogicalProcessor(currentProcessor, currentCore, currentPackage,
341                 numaNodeMap.getOrDefault(currentProcessor, 0)));
342         coreEfficiencyMap.put((currentPackage << 16) + currentCore, 0);
343         return new Quartet<>(logProcs, orderedProcCaches(caches), coreEfficiencyMap, Collections.emptyMap());
344     }

346     private static Map<Integer, Integer> mapNumaNodesFromLscpu() {
347         Map<Integer, Integer> numaNodeMap = new HashMap<>();
348         // Get numa node info from lscpu
349         List<String> lscpu = ExecutingCommand.runNative("lscpu -p=cpu,node");
350         // Format:
351         // # comment lines starting with #
352         // # then comma-delimited cpu,node
353         // 0,0
354         // 1,0
355         for (String line : lscpu) {
356             if (!line.startsWith("#")) {
357                 int pos = line.indexOf(',');
358                 if (pos > 0 && pos < line.length()) {
359                     numaNodeMap.put(ParseUtil.parseIntOrDefault(line.substring(0, pos), 0),
360                             ParseUtil.parseIntOrDefault(line.substring(pos + 1), 0));
361                 }
362             }
363         }
364         return numaNodeMap;
365     }

367     private static Set<ProcessorCache> mapCachesFromLscpu() {
368         Set<ProcessorCache> caches = new HashSet<>();
369         int level = 0;
370         Type type = null;
371         int associativity = 0;
372         int lineSize = 0;
373         long size = 0L;
374         // Get numa node info from lscpu
375         List<String> lscpu = ExecutingCommand.runNative("lscpu -B -C --json");
376         for (String line : lscpu) {
377             String s = line.trim();
378             if (s.startsWith("}")) {
379                 // done with this entry, save it
380                 if (level > 0 && type != null) {
381                     caches.add(new ProcessorCache(level, associativity, lineSize, size, type));
382                 }
383                 level = 0;
384                 type = null;
385                 associativity = 0;
386                 lineSize = 0;
387                 size = 0L;
388             } else if (s.contains("one-size")) {
389                 // "one-size": "65536",
390                 String[] split = ParseUtil.notDigits.split(s);
391                 if (split.length > 1) {
392                     size = ParseUtil.parseLongOrDefault(split[1], 0L);
393                 }
394             } else if (s.contains("ways")) {
395                 // "ways": null,
396                 // "ways": 4,
397                 String[] split = ParseUtil.notDigits.split(s);
398                 if (split.length > 1) {
399                     associativity = ParseUtil.parseIntOrDefault(split[1], 0);
400                 }
401             } else if (s.contains("type")) {
402                 // "type": "Unified",
403                 String[] split = s.split("\"");
404                 if (split.length > 2) {
405                     type = parseCacheType(split[split.length - 2]);
406                 }
407             } else if (s.contains("level")) {
408                 // "level": 3,
409                 String[] split = ParseUtil.notDigits.split(s);
410                 if (split.length > 1) {
411                     level = ParseUtil.parseIntOrDefault(split[1], 0);
412                 }
413             } else if (s.contains("coherency-size")) {
414                 // "coherency-size": 64
415                 String[] split = ParseUtil.notDigits.split(s);
416                 if (split.length > 1) {
417                     lineSize = ParseUtil.parseIntOrDefault(split[1], 0);
418                 }
419             }
420         }
421         return caches;
422     }

424     @Override
425     public long[] querySystemCpuLoadTicks() {
426         // convert the Linux Jiffies to Milliseconds.
427         long[] ticks = CpuStat.getSystemCpuLoadTicks();
428         // In rare cases, /proc/stat reading fails. If so, try again.
429         if (LongStream.of(ticks).sum() == 0) {
430             ticks = CpuStat.getSystemCpuLoadTicks();
431         }
432         long hz = LinuxOperatingSystem.getHz();
433         for (int i = 0; i < ticks.length; i++) {
434             ticks[i] = ticks[i] * 1000L / hz;
435         }
436         return ticks;
437     }

439     @Override
440     public long[] queryCurrentFreq() {
441         long[] freqs = new long[getLogicalProcessorCount()];
442         // Attempt to fill array from cpu-freq source
443         long max = 0L;
444         UdevContext udev = Udev.INSTANCE.udev_new();
445         try {
446             UdevEnumerate enumerate = udev.enumerateNew();
447             try {
448                 enumerate.addMatchSubsystem("cpu");
449                 enumerate.scanDevices();
450                 for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
451                     String syspath = entry.getName(); // /sys/devices/system/cpu/cpuX
452                     int cpu = ParseUtil.getFirstIntValue(syspath);
453                     if (cpu >= 0 && cpu < freqs.length) {
454                         freqs[cpu] = FileUtil.getLongFromFile(syspath + "/cpufreq/scaling_cur_freq");
455                         if (freqs[cpu] == 0) {
456                             freqs[cpu] = FileUtil.getLongFromFile(syspath + "/cpufreq/cpuinfo_cur_freq");
457                         }
458                     }
459                     if (max < freqs[cpu]) {
460                         max = freqs[cpu];
461                     }
462                 }
463                 if (max > 0L) {
464                     // If successful, array is filled with values in KHz.
465                     for (int i = 0; i < freqs.length; i++) {
466                         freqs[i] *= 1000L;
467                     }
468                     return freqs;
469                 }
470             } finally {
471                 enumerate.unref();
472             }
473         } finally {
474             udev.unref();
475         }
476         // If unsuccessful, try from /proc/cpuinfo
477         Arrays.fill(freqs, -1);
478         List<String> cpuInfo = FileUtil.readFile(CPUINFO);
479         int proc = 0;
480         for (String s : cpuInfo) {
481             if (s.toLowerCase(Locale.ROOT).contains("cpu mhz")) {
482                 freqs[proc] = Math.round(ParseUtil.parseLastDouble(s, 0d) * 1_000_000d);
483                 if (++proc >= freqs.length) {
484                     break;
485                 }
486             }
487         }
488         return freqs;
489     }

491     @Override
492     public long queryMaxFreq() {
493         long max = Arrays.stream(this.getCurrentFreq()).max().orElse(-1L);
494         // Max of current freq, if populated, is in units of Hz, convert to kHz
495         if (max > 0) {
496             max /= 1000L;
497         }
498         // Iterating CPUs only gets the existing policy, so we need to iterate the
499         // policy directories to find the system-wide policy max
500         UdevContext udev = Udev.INSTANCE.udev_new();
501         try {
502             UdevEnumerate enumerate = udev.enumerateNew();
503             try {
504                 enumerate.addMatchSubsystem("cpu");
505                 enumerate.scanDevices();
506                 // Find the parent directory of cpuX paths
507                 // We only need the first one of the iteration
508                 UdevListEntry entry = enumerate.getListEntry();
509                 if (entry != null) {
510                     String syspath = entry.getName(); // /sys/devices/system/cpu/cpu0
511                     String cpuFreqPath = syspath.substring(0, syspath.lastIndexOf(File.separatorChar)) + "/cpuFreq";
512                     String policyPrefix = cpuFreqPath + "/policy";
513                     try (Stream<Path> path = Files.list(Paths.get(cpuFreqPath))) {
514                         Optional<Long> maxPolicy = path.filter(p -> p.toString().startsWith(policyPrefix)).map(p -> {
515                             long freq = FileUtil.getLongFromFile(p.toString() + "/scaling_max_freq");
516                             if (freq == 0) {
517                                 freq = FileUtil.getLongFromFile(p.toString() + "/cpuinfo_max_freq");
518                             }
519                             return freq;
520                         }).max(Long::compare);
521                         if (maxPolicy.isPresent() && max < maxPolicy.get()) {
522                             max = maxPolicy.get();
523                         }
524                     } catch (IOException e) {
525                         // ignore
526                     }
527                 }
528             } finally {
529                 enumerate.unref();
530             }
531         } finally {
532             udev.unref();
533         }
534         if (max == 0L) {
535             return -1L;
536         }
537         // If successful, value is in KHz.
538         max *= 1000L;
539         // Cpufreq result assumes intel pstates and is unreliable for AMD processors.
540         // Check lshw as a backup
541         long lshwMax = Lshw.queryCpuCapacity();
542         return lshwMax > max ? lshwMax : max;
543     }

545     @Override
546     public double[] getSystemLoadAverage(int nelem) {
547         if (nelem < 1 || nelem > 3) {
548             throw new IllegalArgumentException("Must include from one to three elements.");
549         }
550         double[] average = new double[nelem];
551         int retval = LinuxLibc.INSTANCE.getloadavg(average, nelem);
552         if (retval < nelem) {
553             for (int i = Math.max(retval, 0); i < average.length; i++) {
554                 average[i] = -1d;
555             }
556         }
557         return average;
558     }

560     @Override
561     public long[][] queryProcessorCpuLoadTicks() {
562         long[][] ticks = CpuStat.getProcessorCpuLoadTicks(getLogicalProcessorCount());
563         // In rare cases, /proc/stat reading fails. If so, try again.
564         // In theory we should check all of them, but on failure we can expect all 0's
565         // so we only need to check for processor 0
566         if (LongStream.of(ticks[0]).sum() == 0) {
567             ticks = CpuStat.getProcessorCpuLoadTicks(getLogicalProcessorCount());
568         }
569         // convert the Linux Jiffies to Milliseconds.
570         long hz = LinuxOperatingSystem.getHz();
571         for (int i = 0; i < ticks.length; i++) {
572             for (int j = 0; j < ticks[i].length; j++) {
573                 ticks[i][j] = ticks[i][j] * 1000L / hz;
574             }
575         }
576         return ticks;
577     }

579     /**
580      * Fetches the ProcessorID from dmidecode (if possible with root permissions), the cpuid command (if installed) or
581      * by encoding the stepping, model, family, and feature flags.
582      *
583      * @param vendor   The vendor
584      * @param stepping The stepping
585      * @param model    The model
586      * @param family   The family
587      * @param flags    The flags
588      * @return The Processor ID string
589      */
590     private static String getProcessorID(String vendor, String stepping, String model, String family, String[] flags) {
591         boolean procInfo = false;
592         String marker = "Processor Information";
593         for (String checkLine : ExecutingCommand.runNative("dmidecode -t 4")) {
594             if (!procInfo && checkLine.contains(marker)) {
595                 marker = "ID:";
596                 procInfo = true;
597             } else if (procInfo && checkLine.contains(marker)) {
598                 return checkLine.split(marker)[1].trim();
599             }
600         }
601         // If we've gotten this far, dmidecode failed. Try cpuid.
602         marker = "eax=";
603         for (String checkLine : ExecutingCommand.runNative("cpuid -1r")) {
604             if (checkLine.contains(marker) && checkLine.trim().startsWith("0x00000001")) {
605                 String eax = "";
606                 String edx = "";
607                 for (String register : ParseUtil.whitespaces.split(checkLine)) {
608                     if (register.startsWith("eax=")) {
609                         eax = ParseUtil.removeMatchingString(register, "eax=0x");
610                     } else if (register.startsWith("edx=")) {
611                         edx = ParseUtil.removeMatchingString(register, "edx=0x");
612                     }
613                 }
614                 return edx + eax;
615             }
616         }
617         // If we've gotten this far, dmidecode failed. Encode arguments
618         if (vendor.startsWith("0x")) {
619             return createMIDR(vendor, stepping, model, family) + "00000000";
620         }
621         return createProcessorID(stepping, model, family, flags);
622     }
623 
624     /**
625      * Creates the MIDR, the ARM equivalent of CPUID ProcessorID
626      *
627      * @param vendor   the CPU implementer
628      * @param stepping the "rnpn" variant and revision
629      * @param model    the partnum
630      * @param family   the architecture
631      * @return A 32-bit hex string for the MIDR
632      */
633     private static String createMIDR(String vendor, String stepping, String model, String family) {
634         int midrBytes = 0;
635         // Build 32-bit MIDR
636         if (stepping.startsWith("r") && stepping.contains("p")) {
637             String[] rev = stepping.substring(1).split("p");
638             // 3:0  Revision: last n in rnpn
639             midrBytes |= ParseUtil.parseLastInt(rev[1], 0);
640             // 23:20 - Variant: first n in rnpn
641             midrBytes |= ParseUtil.parseLastInt(rev[0], 0) << 20;
642         }
643         // 15:4 - PartNum = model
644         midrBytes |= ParseUtil.parseLastInt(model, 0) << 4;
645         // 19:16 - Architecture = family
646         midrBytes |= ParseUtil.parseLastInt(family, 0) << 16;
647         // 31:24 - Implementer = vendor
648         midrBytes |= ParseUtil.parseLastInt(vendor, 0) << 24;
649 
650         return String.format(Locale.ROOT, "%08X", midrBytes);
651     }
