5   package oshi.hardware.platform.linux;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.hardware.common.platform.linux.LinuxGpuStats;
9   import oshi.util.gpu.NvmlUtilJNA;
10  
11  /**
12   * JNA-based Linux {@link LinuxGpuStats} subclass providing NVML integration via JNA.
13   */
14  @ThreadSafe
15  final class LinuxGpuStatsJNA extends LinuxGpuStats {
16  
17      LinuxGpuStatsJNA(String drmDevicePath, String driverName, String pciBusId, String cardName) {
18          super(drmDevicePath, driverName, pciBusId, cardName);
19      }
20  
21      @Override
22      protected boolean nvmlIsAvailable() {
23          return NvmlUtilJNA.isAvailable();
24      }
25  
26      @Override
27      protected String nvmlFindDevice(String busId) {
28          return NvmlUtilJNA.findDevice(busId);
29      }
30  
31      @Override
32      protected String nvmlFindDeviceByName(String name) {
33          return NvmlUtilJNA.findDeviceByName(name);
34      }
35  
36      @Override
37      protected long nvmlGetVramUsed(String deviceId) {
38          return NvmlUtilJNA.getVramUsed(deviceId);
39      }
40  
41      @Override
42      protected double nvmlGetTemperature(String deviceId) {
43          return NvmlUtilJNA.getTemperature(deviceId);
44      }
45  
46      @Override
47      protected double nvmlGetPowerDraw(String deviceId) {
48          return NvmlUtilJNA.getPowerDraw(deviceId);
49      }
50  
51      @Override
52      protected long nvmlGetCoreClockMhz(String deviceId) {
53          return NvmlUtilJNA.getCoreClockMhz(deviceId);
54      }
55  
56      @Override
57      protected long nvmlGetMemoryClockMhz(String deviceId) {
58          return NvmlUtilJNA.getMemoryClockMhz(deviceId);
59      }
60  
61      @Override
62      protected double nvmlGetFanSpeedPercent(String deviceId) {
63          return NvmlUtilJNA.getFanSpeedPercent(deviceId);
64      }
65  }
