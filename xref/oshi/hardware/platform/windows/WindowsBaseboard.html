24  package oshi.hardware.platform.windows;
25  
26  import static oshi.util.Memoizer.memoize;
27  
28  import java.util.function.Supplier;
29  
30  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult; // NOSONAR squid:S1191
31  
32  import oshi.annotation.concurrent.Immutable;
33  import oshi.driver.windows.wmi.Win32BaseBoard;
34  import oshi.driver.windows.wmi.Win32BaseBoard.BaseBoardProperty;
35  import oshi.hardware.common.AbstractBaseboard;
36  import oshi.util.Constants;
37  import oshi.util.Util;
38  import oshi.util.platform.windows.WmiUtil;
39  import oshi.util.tuples.Quartet;
40  
41  /**
42   * Baseboard data obtained from WMI
43   */
44  @Immutable
45  final class WindowsBaseboard extends AbstractBaseboard {
46  
47      private final Supplier<Quartet<String, String, String, String>> manufModelVersSerial = memoize(
48              WindowsBaseboard::queryManufModelVersSerial);
49  
50      @Override
51      public String getManufacturer() {
52          return manufModelVersSerial.get().getA();
53      }
54  
55      @Override
56      public String getModel() {
57          return manufModelVersSerial.get().getB();
58      }
59  
60      @Override
61      public String getVersion() {
62          return manufModelVersSerial.get().getC();
63      }
64  
65      @Override
66      public String getSerialNumber() {
67          return manufModelVersSerial.get().getD();
68      }
69  
70      private static Quartet<String, String, String, String> queryManufModelVersSerial() {
71          String manufacturer = null;
72          String model = null;
73          String version = null;
74          String serialNumber = null;
75          WmiResult<BaseBoardProperty> win32BaseBoard = Win32BaseBoard.queryBaseboardInfo();
76          if (win32BaseBoard.getResultCount() > 0) {
77              manufacturer = WmiUtil.getString(win32BaseBoard, BaseBoardProperty.MANUFACTURER, 0);
78              model = WmiUtil.getString(win32BaseBoard, BaseBoardProperty.MODEL, 0);
79              version = WmiUtil.getString(win32BaseBoard, BaseBoardProperty.VERSION, 0);
80              serialNumber = WmiUtil.getString(win32BaseBoard, BaseBoardProperty.SERIALNUMBER, 0);
81          }
82          return new Quartet<>(Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer,
83                  Util.isBlank(model) ? Constants.UNKNOWN : model, Util.isBlank(version) ? Constants.UNKNOWN : version,
84                  Util.isBlank(serialNumber) ? Constants.UNKNOWN : serialNumber);
85      }
86  }
