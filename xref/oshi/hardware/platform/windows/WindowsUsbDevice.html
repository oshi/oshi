64      public WindowsUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
65              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
66          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
67      }

69      /**
70       * {@inheritDoc}
71       *
72       * @param tree
73       *            a boolean.
74       * @return an array of {@link oshi.hardware.UsbDevice} objects.
75       */
76      public static List<UsbDevice> getUsbDevices(boolean tree) {
77          List<UsbDevice> devices = getUsbDevices();
78          if (tree) {
79              return devices;
80          }
81          List<UsbDevice> deviceList = new ArrayList<>();
82          // Top level is controllers; they won't be added to the list, but all
83          // their connected devices will be
84          for (UsbDevice device : devices) {
85              addDevicesToList(deviceList, device.getConnectedDevices());
86          }
87          return deviceList;
88      }
89  
90      private static List<UsbDevice> getUsbDevices() {
91          // Map to build the recursive tree structure
92          Map<String, List<String>> deviceTreeMap = new HashMap<>();
93          // Track devices seen in the process
94          Set<String> devicesSeen = new HashSet<>();
95  
96          // Navigate the device tree to track what devices are present
97          List<UsbDevice> controllerDevices = new ArrayList<>();
98          List<String> controllerDeviceIdList = getControllerDeviceIdList();
99          for (String controllerDeviceId : controllerDeviceIdList) {
100             putChildrenInDeviceTree(controllerDeviceId, 0, deviceTreeMap, devicesSeen);
101         }
102         // Map to store information using PNPDeviceID as the key.
103         Map<String, Triplet<String, String, String>> deviceStringMap = queryDeviceStringsMap(devicesSeen);
104         // recursively build results
105         for (String controllerDeviceId : controllerDeviceIdList) {
106             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(controllerDeviceId, "0000", "0000", deviceTreeMap,
107                     deviceStringMap);
108             if (deviceAndChildren != null) {
109                 controllerDevices.add(deviceAndChildren);
110             }
111         }
112         return controllerDevices;
113     }

115     private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
116         for (UsbDevice device : list) {
117             deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
118                     device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(),
119                     Collections.emptyList()));
120             addDevicesToList(deviceList, device.getConnectedDevices());
121         }
122     }

124     private static Map<String, Triplet<String, String, String>> queryDeviceStringsMap(Set<String> devicesToAdd) {
125         Map<String, Triplet<String, String, String>> deviceStringCache = new HashMap<>();
126         // Add devices not in the tree
127         if (!devicesToAdd.isEmpty()) {
128             StringBuilder sb = new StringBuilder();
129             boolean first = true;
130             for (String deviceID : devicesToAdd) {
131                 if (first) {
132                     sb.append(" WHERE (PnPDeviceID=\"");
133                     first = false;
134                 } else {
135                     sb.append(" OR (PnPDeviceID=\"");
136                 }
137                 sb.append(deviceID).append("\")");
138             }
139             String whereClause = sb.toString();
140             // Get serial # for disk drives or other physical media
141             Map<String, String> pnpToSerialMap = new HashMap<>();
142             WmiResult<DeviceIdProperty> serialNumbers = Win32DiskDrive.queryDiskDriveId(whereClause);
143             for (int i = 0; i < serialNumbers.getResultCount(); i++) {
144                 String pnpDeviceID = WmiUtil.getString(serialNumbers, DeviceIdProperty.PNPDEVICEID, i);
145                 if (deviceStringCache.containsKey(pnpDeviceID)) {
146                     pnpToSerialMap.put(pnpDeviceID, ParseUtil
147                             .hexStringToString(WmiUtil.getString(serialNumbers, DeviceIdProperty.SERIALNUMBER, i)));
148                 }
149             }
150             // Query Win32_PnPEntity to populate the maps
151             WmiResult<PnPEntityProperty> pnpEntity = Win32PnPEntity.queryDeviceId(whereClause);
152             for (int i = 0; i < pnpEntity.getResultCount(); i++) {
153                 String pnpDeviceID = WmiUtil.getString(pnpEntity, PnPEntityProperty.PNPDEVICEID, i);
154                 String name = WmiUtil.getString(pnpEntity, PnPEntityProperty.NAME, i);
155                 String vendor = WmiUtil.getString(pnpEntity, PnPEntityProperty.MANUFACTURER, i);
156                 deviceStringCache.put(pnpDeviceID,
157                         new Triplet<>(name, vendor, pnpToSerialMap.getOrDefault(pnpDeviceID, "")));
158                 LOG.debug("Adding {} to USB device cache.", pnpDeviceID);
159             }
160         }
161         return deviceStringCache;
162     }

164     /**
165      * Navigates the Device Tree to place all children PNPDeviceIDs into the map for
166      * the specified deviceID. Recursively adds children's children, etc.
167      *
168      * @param deviceId
169      *            The device to add respective children to the map
170      * @param deviceInstance
171      *            The device instance (devnode handle), if known. If set to 0, the
172      *            code will search for a match.
173      * @param deviceTreeMap
174      *            The overall device tree map that starts at the controllers
175      * @param devicesSeen
176      *            Devices we've seen so we can add/remove from cache later
177      */
178     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance,
179             Map<String, List<String>> deviceTreeMap, Set<String> devicesSeen) {
180         devicesSeen.add(deviceId);
181         // If no devInst provided, find it
182         int devInst = deviceInstance;
183         if (devInst == 0) {
184             IntByReference pdnDevInst = new IntByReference();
185             Cfgmgr32.INSTANCE.CM_Locate_DevNode(pdnDevInst, deviceId, 0);
186             devInst = pdnDevInst.getValue();
187         }
188         // Now iterate the children. Call CM_Get_Child to get first child
189         IntByReference child = new IntByReference();
190         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
191             // Add first child to a list
192             List<String> childList = new ArrayList<>();
193             String childId = Cfgmgr32Util.CM_Get_Device_ID(child.getValue());
194             childList.add(childId);
195             deviceTreeMap.put(deviceId, childList);
196             putChildrenInDeviceTree(childId, child.getValue(), deviceTreeMap, devicesSeen);
197             // Find any other children
198             IntByReference sibling = new IntByReference();
199             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
200                 // Add to the list
201                 String siblingId = Cfgmgr32Util.CM_Get_Device_ID(sibling.getValue());
202                 deviceTreeMap.get(deviceId).add(siblingId);
203                 putChildrenInDeviceTree(siblingId, sibling.getValue(), deviceTreeMap, devicesSeen);
204                 // Make this sibling the new child to find other siblings
205                 child = sibling;
206             }
207         }
208     }

210     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid,
211             Map<String, List<String>> deviceTreeMap, Map<String, Triplet<String, String, String>> deviceStringMap) {
212         String vendorId = vid;
213         String productId = pid;
214         Pair<String, String> idPair = ParseUtil.parsePnPDeviceIdToVendorProductId(hubDeviceId);
215         if (idPair != null) {
216             vendorId = idPair.getA();
217             productId = idPair.getB();
218         }
219         List<String> pnpDeviceIds = deviceTreeMap.getOrDefault(hubDeviceId, new ArrayList<>());
220         List<UsbDevice> usbDevices = new ArrayList<>();
221         for (String pnpDeviceId : pnpDeviceIds) {
222             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(pnpDeviceId, vendorId, productId, deviceTreeMap,
223                     deviceStringMap);
224             if (deviceAndChildren != null) {
225                 usbDevices.add(deviceAndChildren);
226             }
227         }
228         Collections.sort(usbDevices);
229         if (deviceStringMap.containsKey(hubDeviceId)) {
230             // name, vendor, serial
231             Triplet<String, String, String> device = deviceStringMap.get(hubDeviceId);
232             String name = device.getA();
233             if (name.isEmpty()) {
234                 name = vendorId + ":" + productId;
235             }
236             return new WindowsUsbDevice(name, device.getB(), vendorId, productId, device.getC(), hubDeviceId,
237                     usbDevices);
238         }
239         return null;
240     }

242     /**
243      * Queries the USB Controller list
244      *
245      * @return A list of Strings of USB Controller PNPDeviceIDs
246      */
247     private static List<String> getControllerDeviceIdList() {
248         List<String> controllerDeviceIdsList = new ArrayList<>();
249         // One time lookup of USB Controller PnP Device IDs which don't
250         // change
251         WmiResult<USBControllerProperty> usbController = Win32USBController.queryUSBControllers();
252         for (int i = 0; i < usbController.getResultCount(); i++) {
253             controllerDeviceIdsList.add(WmiUtil.getString(usbController, USBControllerProperty.PNPDEVICEID, i));
254         }
255         return controllerDeviceIdsList;
256     }
