54      /**
55       * {@inheritDoc}
56       */
57      @Override
58      public double getCpuTemperature() {
59          // Initialize
60          double tempC = 0d;
61          // If Open Hardware Monitor identifier is set, we couldn't get through
62          // normal WMI, and got ID from OHM at least once so go directly to OHM
63          if (this.tempIdentifierStr != null) {
64              Map<String, List<Float>> vals = WmiUtil.selectFloatsFrom("root\\OpenHardwareMonitor", "Sensor", "Value",
65                      "WHERE Parent=\"" + this.tempIdentifierStr + "\" AND SensorType=\"Temperature\"");
66              if (!vals.get("Value").isEmpty()) {
67                  double sum = 0;
68                  for (double val : vals.get("Value")) {
69                      sum += val;
70                  }
71                  tempC = sum / vals.get("Value").size();
72              }
73              return tempC;
74          }
75          // This branch is used the first time and all subsequent times if
76          // successful (tempIdenifierStr == null)
77          // Try to get value using initial or updated successful values
78          long tempK;
79          if (this.wmiTempClass == null) {
80              this.wmiTempNamespace = "root\\cimv2";
81              this.wmiTempClass = "Win32_Temperature";
82              this.wmiTempProperty = "CurrentReading";
83              tempK = WmiUtil.selectUint32From(this.wmiTempNamespace, this.wmiTempClass, this.wmiTempProperty, null);
84              if (tempK == 0) {
85                  this.wmiTempClass = "Win32_TemperatureProbe";
86                  tempK = WmiUtil.selectUint32From(this.wmiTempNamespace, this.wmiTempClass, this.wmiTempProperty, null);
87              }
88              if (tempK == 0) {
89                  this.wmiTempClass = "Win32_PerfFormattedData_Counters_ThermalZoneInformation";
90                  this.wmiTempProperty = "Temperature";
91                  tempK = WmiUtil.selectUint32From(this.wmiTempNamespace, this.wmiTempClass, this.wmiTempProperty, null);
92              }
93              if (tempK == 0) {
94                  this.wmiTempNamespace = "root\\wmi";
95                  this.wmiTempClass = "MSAcpi_ThermalZoneTemperature";
96                  this.wmiTempProperty = "CurrentTemperature";
97                  tempK = WmiUtil.selectUint32From(this.wmiTempNamespace, this.wmiTempClass, this.wmiTempProperty, null);
98              }
99          } else {
100             // We've successfully read a previous time, or failed both here and
101             // with OHM, so keep using same values
102             tempK = WmiUtil.selectUint32From(this.wmiTempNamespace, this.wmiTempClass, this.wmiTempProperty, null);
103         }
104         // Convert K to C and return result
105         if (tempK > 2732) {
106             tempC = tempK / 10d - 273.15;
107         } else if (tempK > 274) {
108             tempC = tempK - 273d;
109         }
110         if (tempC <= 0d) {
111             // Unable to get temperature via WMI. Future attempts will be
112             // attempted via Open Hardware Monitor WMI if successful
113             String cpuIdentifier = WmiUtil.selectStringFrom("root\\OpenHardwareMonitor", "Hardware", "Identifier",
114                     "WHERE HardwareType=\"CPU\"");
115             this.tempIdentifierStr = cpuIdentifier.length() > 0 ? cpuIdentifier : null;
116             // If not null, recurse and get value via OHM
117             if (this.tempIdentifierStr != null) {
118                 return getCpuTemperature();
119             }
120             tempC = 0d;
121         }
122         return tempC;
123     }

125     /**
126      * {@inheritDoc}
127      */
128     @Override
129     public int[] getFanSpeeds() {
130         // Initialize
131         int[] fanSpeeds = new int[1];
132         // If we couldn't get through normal WMI go directly to OHM
133         if (!this.fanSpeedWMI) {
134             Map<String, List<Float>> vals = WmiUtil.selectFloatsFrom("root\\OpenHardwareMonitor", "Sensor", "Value",
135                     "WHERE Parent=\"" + this.tempIdentifierStr + "\" AND SensorType=\"Fan\"");
136             if (!vals.get("Value").isEmpty()) {
137                 fanSpeeds = new int[vals.get("Value").size()];
138                 for (int i = 0; i < vals.get("Value").size(); i++) {
139                     fanSpeeds[i] = vals.get("Value").get(i).intValue();
140                 }
141             }
142             return fanSpeeds;
143         }
144         // This branch is used the first time and all subsequent times if
145         // successful (fanSpeedWMI == true)
146         // Try to get value
147         int rpm = WmiUtil.selectUint32From(null, "Win32_Fan", "DesiredSpeed", null).intValue();
148         // Set in array and return
149         if (rpm > 0) {
150             fanSpeeds[0] = rpm;
151         } else {
152             // Fail, switch to OHM
153             this.fanSpeedWMI = false;
154             return getFanSpeeds();
155         }
156         return fanSpeeds;
157     }

159     /**
160      * {@inheritDoc}
161      */
162     @Override
163     public double getCpuVoltage() {
164         // Initialize
165         double volts = 0d;
166         // If we couldn't get through normal WMI go directly to OHM
167         if (this.voltIdentifierStr != null) {
168             return WmiUtil.selectFloatFrom("root\\OpenHardwareMonitor", "Sensor", "Value",
169                     "WHERE Parent=\"" + this.voltIdentifierStr + "\" AND SensorType=\"Voltage\"");
170         }
171         // This branch is used the first time and all subsequent times if
172         // successful (voltIdenifierStr == null)
173         // Try to get value
174         // Try to get value using initial or updated successful values
175         int decivolts;
176         if (this.wmiVoltClass == null) {
177             this.wmiVoltNamespace = "root\\cimv2";
178             this.wmiVoltClass = "Win32_Processor";
179             this.wmiVoltProperty = "CurrentVoltage";
180             decivolts = WmiUtil.selectUint32From(this.wmiVoltNamespace, this.wmiVoltClass, this.wmiVoltProperty, null)
181                     .intValue();
182             // If the eighth bit is set, bits 0-6 contain the voltage
183             // multiplied by 10. If the eighth bit is not set, then the bit
184             // setting in VoltageCaps represents the voltage value.
185             if ((decivolts & 0x80) == 0 && decivolts > 0) {
186                 this.wmiVoltProperty = "VoltageCaps";
187                 // really a bit setting, not decivolts, test later
188                 decivolts = WmiUtil
189                         .selectUint32From(this.wmiVoltNamespace, this.wmiVoltClass, this.wmiVoltProperty, null)
190                         .intValue();
191             }
192         } else {
193             // We've successfully read a previous time, or failed both here and
194             // with OHM
195             decivolts = WmiUtil.selectUint32From(this.wmiVoltNamespace, this.wmiVoltClass, this.wmiVoltProperty, null)
196                     .intValue();
197         }
198         // Convert dV to V and return result
199         if (decivolts > 0) {
200             if ("VoltageCaps".equals(this.wmiVoltProperty)) {
201                 // decivolts are bits
202                 if ((decivolts & 0x1) > 0) {
203                     volts = 5.0;
204                 } else if ((decivolts & 0x2) > 0) {
205                     volts = 3.3;
206                 } else if ((decivolts & 0x4) > 0) {
207                     volts = 2.9;
208                 }
209             } else {
210                 // Value from bits 0-6
211                 volts = (decivolts & 0x7F) / 10d;
212             }
213         }
214         if (volts <= 0d) {
215             // Unable to get voltage via WMI. Future attempts will be
216             // attempted via Open Hardware Monitor WMI if successful
217             Map<String, List<String>> voltIdentifiers = WmiUtil.selectStringsFrom("root\\OpenHardwareMonitor",
218                     "Hardware", "Identifier", "WHERE SensorType=\"Voltage\"");
219             // Look for identifier containing "cpu"
220 
221             for (String id : voltIdentifiers.get("Identifier")) {
222                 if (id.toLowerCase().contains("cpu")) {
223                     this.voltIdentifierStr = id;
224                     break;
225                 }
226             }
227             // If none contain cpu just grab the first one
228             if (this.voltIdentifierStr == null && !voltIdentifiers.get("Identifier").isEmpty()) {
229                 this.voltIdentifierStr = voltIdentifiers.get("Identifier").get(0);
230             }
231             // If not null, recurse and get value via OHM
232             if (this.voltIdentifierStr != null) {
233                 return getCpuVoltage();
234             }
235         }
236         return volts;
237     }
