5   package oshi.hardware.platform.windows;
6   
7   import java.lang.foreign.MemorySegment;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import oshi.annotation.concurrent.Immutable;
15  import oshi.ffm.platform.windows.Win32Exception;
16  import oshi.ffm.platform.windows.WinErrorFFM;
17  import oshi.ffm.platform.windows.WinRegFFM;
18  import oshi.ffm.util.platform.windows.Advapi32UtilFFM;
19  import oshi.hardware.SoundCard;
20  import oshi.hardware.common.AbstractSoundCard;
21  
22  /**
23   * Sound Card data obtained from registry using FFM.
24   */
25  @Immutable
26  final class WindowsSoundCardFFM extends AbstractSoundCard {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(WindowsSoundCardFFM.class);
29  
30      private static final String REGISTRY_SOUNDCARDS = "SYSTEM\\CurrentControlSet\\Control\\Class\\{4d36e96c-e325-11ce-bfc1-08002be10318}\\";
31  
32      private static final MemorySegment HKLM = MemorySegment.ofAddress(WinRegFFM.HKEY_LOCAL_MACHINE);
33  
34      WindowsSoundCardFFM(String kernelVersion, String name, String codec) {
35          super(kernelVersion, name, codec);
36      }
37  
38      public static List<SoundCard> getSoundCards() {
39          List<SoundCard> soundCards = new ArrayList<>();
40          String[] keys;
41          try {
42              keys = Advapi32UtilFFM.registryGetKeys(HKLM, REGISTRY_SOUNDCARDS, 0);
43          } catch (Win32Exception e) {
44              if (e.getErrorCode() != WinErrorFFM.ERROR_ACCESS_DENIED) {
45                  throw e;
46              }
47              return soundCards;
48          } catch (Throwable t) { // NOSONAR squid:S1181
49              LOG.debug("Failed to enumerate sound card registry keys: {}", t.getMessage());
50              return soundCards;
51          }
52          for (String key : keys) {
53              try {
54                  String fullKey = REGISTRY_SOUNDCARDS + key;
55                  if (Advapi32UtilFFM.registryValueExists(HKLM, fullKey, "Driver")) {
56                      String driver = getRegString(fullKey, "Driver");
57                      String driverVersion = getRegString(fullKey, "DriverVersion");
58                      String providerName = getRegString(fullKey, "ProviderName");
59                      String driverDesc = getRegString(fullKey, "DriverDesc");
60                      soundCards.add(new WindowsSoundCardFFM(driver + " " + driverVersion,
61                              providerName + " " + driverDesc, driverDesc));
62                  }
63              } catch (Win32Exception e) {
64                  if (e.getErrorCode() != WinErrorFFM.ERROR_ACCESS_DENIED) {
65                      throw e;
66                  }
67              } catch (Throwable t) { // NOSONAR squid:S1181
68                  LOG.debug("Failed to read sound card registry key {}: {}", key, t.getMessage());
69              }
70          }
71          return soundCards;
72      }
73  
74      private static String getRegString(String keyPath, String valueName) {
75          Object val = Advapi32UtilFFM.registryGetValue(HKLM, keyPath, valueName);
76          return val instanceof String string ? string : "";
77      }
78  }
