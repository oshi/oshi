92          return Arrays.asList(getPowerSource("System Battery"));

95      private static WindowsPowerSource getPowerSource(String name) {
96          String psName = name;
97          String psDeviceName = Constants.UNKNOWN;
98          double psRemainingCapacityPercent = 1d;
99          double psTimeRemainingEstimated = -1d;
100         double psTimeRemainingInstant = 0d;
101         int psPowerUsageRate = 0;
102         double psVoltage = -1d;
103         double psAmperage = 0d;
104         boolean psPowerOnLine = false;
105         boolean psCharging = false;
106         boolean psDischarging = false;
107         CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
108         int psCurrentCapacity = 0;
109         int psMaxCapacity = 1;
110         int psDesignCapacity = 1;
111         int psCycleCount = -1;
112         String psChemistry = Constants.UNKNOWN;
113         LocalDate psManufactureDate = null;
114         String psManufacturer = Constants.UNKNOWN;
115         String psSerialNumber = Constants.UNKNOWN;
116         double psTemperature = 0d;
117 
118         // Note: CallNtPowerInformation(SystemBatteryState) was previously used as a
119         // first pass here but no longer returns accurate data as of Windows 11 23H2.
120         // All battery information is now sourced exclusively via DeviceIoControl.
121         //
122         // Enumerate batteries and ask each one for information.
123         // Ported from:
124         // https://docs.microsoft.com/en-us/windows/win32/power/enumerating-battery-devices
125         try (Arena arena = Arena.ofConfined()) {
126             MemorySegment classGuid = GuidFFM.GUID_DEVCLASS_BATTERY(arena);
127             Optional<MemorySegment> hdevOpt = SetupApiFFM.SetupDiGetClassDevs(classGuid,
128                     DIGCF_PRESENT | DIGCF_DEVICEINTERFACE);
129             if (hdevOpt.isEmpty()) {
130                 LOG.warn("SetupDiGetClassDevs failed for battery class");
131             } else {
132                 MemorySegment hdev = hdevOpt.get();
133                 try {
134                     boolean batteryFound = false;
135                     for (int idev = 0; !batteryFound && idev < 100; idev++) {
136                         MemorySegment did = arena.allocate(SP_DEVICE_INTERFACE_DATA);
137                         did.set(JAVA_INT, 0, (int) SP_DEVICE_INTERFACE_DATA.byteSize());
138 
139                         int enumResult = SetupApiFFM.SetupDiEnumDeviceInterfaces(hdev, classGuid, idev, did);
140                         if (enumResult == 0) {
141                             break;
142                         }
143                         if (enumResult < 0) {
144                             continue;
145                         }
146 
147                         int requiredSize = SetupApiFFM.SetupDiGetDeviceInterfaceDetailSize(hdev, did, arena);
148                         if (requiredSize == 0) {
149                             continue;
150                         }
151 
152                         Optional<String> devicePathOpt = SetupApiFFM.SetupDiGetDeviceInterfaceDetail(hdev, did,
153                                 requiredSize, arena);
154                         if (devicePathOpt.isEmpty()) {
155                             continue;
156                         }
157 
158                         MemorySegment devicePathSeg = WindowsForeignFunctions.toWideString(arena, devicePathOpt.get());
159                         Optional<MemorySegment> hBatteryOpt = Kernel32FFM.CreateFile(devicePathSeg,
160                                 GENERIC_READ | GENERIC_WRITE, FILE_SHARE_READ | FILE_SHARE_WRITE, OPEN_EXISTING,
161                                 FILE_ATTRIBUTE_NORMAL);
162                         if (hBatteryOpt.isEmpty()) {
163                             continue;
164                         }
165                         MemorySegment hBattery = hBatteryOpt.get();
166                         try {
167                             // Ask the battery for its tag
168                             MemorySegment dwWait = arena.allocate(JAVA_INT);
169                             MemorySegment dwTag = arena.allocate(JAVA_INT);
170                             if (!Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_TAG, dwWait,
171                                     (int) dwWait.byteSize(), dwTag, (int) dwTag.byteSize())) {
172                                 continue;
173                             }
174                             int batteryTag = dwTag.get(JAVA_INT, 0);
175                             if (batteryTag == 0) {
176                                 continue;
177                             }
178 
179                             MemorySegment bqi = arena.allocate(BATTERY_QUERY_INFORMATION);
180                             bqi.set(JAVA_INT, OFF_BQI_TAG, batteryTag);
181                             bqi.set(JAVA_INT, OFF_BQI_LEVEL, BATTERY_INFORMATION_LEVEL);
182 
183                             MemorySegment bi = arena.allocate(BATTERY_INFORMATION);
184                             if (!Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi,
185                                     (int) bqi.byteSize(), bi, (int) bi.byteSize())) {
186                                 continue;
187                             }
188 
189                             int capabilities = bi.get(JAVA_INT, OFF_BI_CAPABILITIES);
190                             // Only non-UPS system batteries count
191                             if (0 == (capabilities & BATTERY_SYSTEM_BATTERY)
192                                     || 0 != (capabilities & BATTERY_IS_SHORT_TERM)) {
193                                 continue;
194                             }
195 
196                             if (0 == (capabilities & BATTERY_CAPACITY_RELATIVE)) {
197                                 psCapacityUnits = CapacityUnits.MWH;
198                             }
199                             byte[] chemBytes = new byte[4];
200                             for (int i = 0; i < 4; i++) {
201                                 chemBytes[i] = bi.get(JAVA_BYTE, OFF_BI_CHEMISTRY + i);
202                             }
203                             psChemistry = new String(chemBytes, StandardCharsets.US_ASCII).trim();
204                             psDesignCapacity = bi.get(JAVA_INT, OFF_BI_DESIGNED);
205                             psMaxCapacity = bi.get(JAVA_INT, OFF_BI_FULL);
206                             psCycleCount = bi.get(JAVA_INT, OFF_BI_CYCLE);
207                             int maxCapacitySafe = psMaxCapacity > 0 ? psMaxCapacity
208                                     : psDesignCapacity > 0 ? psDesignCapacity : 1;
209 
210                             // Query battery status
211                             MemorySegment bws = arena.allocate(BATTERY_WAIT_STATUS);
212                             bws.set(JAVA_INT, OFF_BWS_TAG, batteryTag);
213                             MemorySegment bs = arena.allocate(BATTERY_STATUS);
214                             if (Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_STATUS, bws,
215                                     (int) bws.byteSize(), bs, (int) bs.byteSize())) {
216                                 int powerState = bs.get(JAVA_INT, OFF_BS_POWERSTATE);
217                                 if (0 != (powerState & BATTERY_POWER_ON_LINE)) {
218                                     psPowerOnLine = true;
219                                 }
220                                 if (0 != (powerState & BATTERY_DISCHARGING)) {
221                                     psDischarging = true;
222                                 }
223                                 if (0 != (powerState & BATTERY_CHARGING)) {
224                                     psCharging = true;
225                                     psTimeRemainingEstimated = -2d;
226                                 }
227                                 psCurrentCapacity = bs.get(JAVA_INT, OFF_BS_CAPACITY);
228                                 int voltage = bs.get(JAVA_INT, OFF_BS_VOLTAGE);
229                                 psVoltage = voltage > 0 ? voltage / 1000d : voltage;
230                                 psPowerUsageRate = bs.get(JAVA_INT, OFF_BS_RATE);
231                                 if (psVoltage > 0) {
232                                     psAmperage = psPowerUsageRate / psVoltage;
233                                 }
234                                 psRemainingCapacityPercent = Math.min(1d, (double) psCurrentCapacity / maxCapacitySafe);
235                             }
236 
237                             MemorySegment nameBuf = arena.allocate(1024);
238                             psDeviceName = batteryQueryString(hBattery, batteryTag, BATTERY_DEVICE_NAME_LEVEL, bqi,
239                                     nameBuf);
240                             psManufacturer = batteryQueryString(hBattery, batteryTag, BATTERY_MANUFACTURE_NAME_LEVEL,
241                                     bqi, nameBuf);
242                             psSerialNumber = batteryQueryString(hBattery, batteryTag, BATTERY_SERIAL_NUMBER_LEVEL, bqi,
243                                     nameBuf);
244 
245                             // Manufacture date
246                             bqi.set(JAVA_INT, OFF_BQI_LEVEL, BATTERY_MANUFACTURE_DATE_LEVEL);
247                             MemorySegment bmd = arena.allocate(BATTERY_MANUFACTURE_DATE);
248                             if (Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi,
249                                     (int) bqi.byteSize(), bmd, (int) bmd.byteSize())) {
250                                 int year = bmd.get(JAVA_SHORT, OFF_BMD_YEAR) & 0xFFFF;
251                                 int month = bmd.get(JAVA_BYTE, OFF_BMD_MONTH) & 0xFF;
252                                 int day = bmd.get(JAVA_BYTE, OFF_BMD_DAY) & 0xFF;
253                                 if (year > 1900 && month >= 1 && month <= 12 && day >= 1 && day <= 31) {
254                                     try {
255                                         psManufactureDate = LocalDate.of(year, month, day);
256                                     } catch (java.time.DateTimeException ignored) {
257                                         // malformed firmware date — leave null
258                                     }
259                                 }
260                             }
261 
262                             // Temperature (1/10 degree K)
263                             bqi.set(JAVA_INT, OFF_BQI_LEVEL, BATTERY_TEMPERATURE_LEVEL);
264                             MemorySegment tempK = arena.allocate(JAVA_INT);
265                             if (Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi,
266                                     (int) bqi.byteSize(), tempK, (int) tempK.byteSize())) {
267                                 psTemperature = tempK.get(JAVA_INT, 0) / 10d - 273.15;
268                             }
269 
270                             // Estimated time — put last because we change AtRate
271                             bqi.set(JAVA_INT, OFF_BQI_LEVEL, BATTERY_ESTIMATED_TIME_LEVEL);
272                             if (psPowerUsageRate != 0) {
273                                 bqi.set(JAVA_INT, OFF_BQI_ATRATE, psPowerUsageRate);
274                             }
275                             MemorySegment tr = arena.allocate(JAVA_INT);
276                             if (Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi,
277                                     (int) bqi.byteSize(), tr, (int) tr.byteSize())) {
278                                 psTimeRemainingInstant = tr.get(JAVA_INT, 0);
279                             }
280                             // Fallback if BatteryEstimatedTime query failed
281                             if (psTimeRemainingInstant <= 0 && psPowerUsageRate != 0) {
282                                 psTimeRemainingInstant = psDischarging
283                                         ? Math.max(0d, psCurrentCapacity * 3600d / Math.abs(psPowerUsageRate))
284                                         : Math.max(0d, (maxCapacitySafe - psCurrentCapacity) * 3600d
285                                                 / Math.abs(psPowerUsageRate));
286                             }
287                             if (psDischarging && psTimeRemainingInstant > 0) {
288                                 psTimeRemainingEstimated = psTimeRemainingInstant;
289                             }
290 
291                             batteryFound = true;
292                         } finally {
293                             Kernel32FFM.CloseHandle(hBattery);
294                         }
295                     }
296                 } finally {
297                     SetupApiFFM.SetupDiDestroyDeviceInfoList(hdev);
298                 }
299             }
300         }
301         return new WindowsPowerSourceFFM(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated,
302                 psTimeRemainingInstant, psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging,
303                 psDischarging, psCapacityUnits, psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount,
304                 psChemistry, psManufactureDate, psManufacturer, psSerialNumber, psTemperature);
305     }

307     private static String batteryQueryString(MemorySegment hBattery, int tag, int infoLevel, MemorySegment bqi,
308             MemorySegment nameBuf) {
309         bqi.set(JAVA_INT, OFF_BQI_TAG, tag);
310         bqi.set(JAVA_INT, OFF_BQI_LEVEL, infoLevel);
311         if (!Kernel32FFM.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi, (int) bqi.byteSize(), nameBuf,
312                 (int) nameBuf.byteSize())) {
313             return Constants.UNKNOWN;
314         }
315         String result = WindowsForeignFunctions.readWideString(nameBuf);
316         return result.isEmpty() ? Constants.UNKNOWN : result;
317     }
