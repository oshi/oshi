19  package oshi.hardware.platform.windows;
20  
21  import java.util.ArrayList;
22  import java.util.HashMap;
23  import java.util.List;
24  import java.util.Map;
25  
26  import oshi.hardware.common.AbstractSoundCard;
27  import oshi.jna.platform.windows.WbemcliUtil;
28  import oshi.util.platform.windows.WmiUtil;
29  
30  /**
31   * Sound Card data obtained by Wmi Queries for Windows.
32   *
33   * @author : BilalAM
34   */
35  public class WindowsSoundCard extends AbstractSoundCard {
36      enum SoundCardName {
37          MANUFACTURER, NAME;
38      }
39  
40      enum SoundCardKernel {
41          DRIVERPROVIDERNAME, DRIVERNAME, DRIVERVERSION, DEVICENAME;
42      }
43  
44      private static final Map<String, String> NAME_MAP = new HashMap<>();
45      private static final String DRIVER_QUERY;
46  
47      /**
48       * Runs the Win32_SoundDevice query only once and initializes a Map with
49       * Name as key and Manufacturer as Value
50       *
51       * Also calls the createClause() method to build our WHERE clause query only
52       * once
53       */
54      static {
55          WbemcliUtil.WmiQuery<SoundCardName> soundCardQuery = new WbemcliUtil.WmiQuery<>("Win32_SoundDevice",
56                  SoundCardName.class);
57          WbemcliUtil.WmiResult<SoundCardName> soundCardResult = WmiUtil.queryWMI(soundCardQuery);
58          for (int i = 0; i < soundCardResult.getResultCount(); i++) {
59              NAME_MAP.put(WmiUtil.getString(soundCardResult, SoundCardName.NAME, i),
60                      WmiUtil.getString(soundCardResult, SoundCardName.MANUFACTURER, i));
61          }
62          DRIVER_QUERY = createClause();
63      }
64  
65      public WindowsSoundCard(String kernelVersion, String name, String codec) {
66          super(kernelVersion, name, codec);
67      }
68  
69      /**
70       * Creates our Win32_PnPSignedDevice query with the WHERE clause taking the
71       * attributes from our map.
72       *
73       * @return The WHERE clause
74       */
75      private static String createClause() {
76          StringBuilder sb = new StringBuilder("Win32_PnPSignedDriver");
77          boolean first = true;
78          for (String key : NAME_MAP.keySet()) {
79              if (first) {
80                  sb.append(" WHERE");
81                  first = false;
82              } else {
83                  sb.append(" OR");
84              }
85              sb.append(" DeviceName LIKE \"%").append(key).append("%\"");
86          }
87          return sb.toString();
88      }
89  
90      /**
91       * Short method to build our kernel string.
92       *
93       * @param index
94       *            The row to fetch data.
95       * @param cardKernelQueryResult
96       *            The result to take data from.
97       * @return The kernel string.
98       */
99      private static String getAudioCardKernelVersion(int index,
100             WbemcliUtil.WmiResult<SoundCardKernel> cardKernelQueryResult) {
101         String audioCardKernel;
102         audioCardKernel = WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERPROVIDERNAME, index) + " "
103                 + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, index) + " "
104                 + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERNAME, index) + " "
105                 + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERVERSION, index);
106         return audioCardKernel;
107     }
108 
109     /**
110      * Does the following :
111      * <ul>
112      * <li>Creates and runs our 'built where clause' query</li>
113      * <li>Then iterates over the key values of our map and compares the key
114      * with the DeviceName attribute returned by our where-clause query.If its a
115      * match then we create our SoundCard object</li>
116      * </ul>
117      * <br>
118      * NOTE : The reason why the codec name is same as the card name is because
119      * windows does not provide the name of the codec chip but sometimes the
120      * name of the card returned is infact the name of the codec chip also.
121      * Example : Realtek ALC887 HD Audio Device
122      *
123      * @return List of sound cards
124      */
125     public static List<WindowsSoundCard> getSoundCards() {
126         WbemcliUtil.WmiQuery<SoundCardKernel> cardKernelQuery = new WbemcliUtil.WmiQuery<>(DRIVER_QUERY,
127                 SoundCardKernel.class);
128         WbemcliUtil.WmiResult<SoundCardKernel> cardKernelQueryResult = WmiUtil.queryWMI(cardKernelQuery);
129 
130         List<WindowsSoundCard> soundCards = new ArrayList<>();
131         for (int i = 0; i < cardKernelQueryResult.getResultCount(); i++) {
132             // If the map has a key that is equal to the value returned by
133             // cardKernelQuery
134             if (NAME_MAP.containsKey(WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i))) {
135                 // then build a sound card by extracting values from
136                 // cardKernelQuery
137                 soundCards.add(new WindowsSoundCard(getAudioCardKernelVersion(i, cardKernelQueryResult),
138                         WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERPROVIDERNAME, i) + " "
139                                 + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i),
140                         WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i)));
141             }
142         }
143         return soundCards;
144     }
