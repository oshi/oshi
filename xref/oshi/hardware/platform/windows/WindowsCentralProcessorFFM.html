90      @Override
91      protected ProcessorIdentifier queryProcessorId() {
92          String cpuVendor = "";
93          String cpuName = "";
94          String cpuIdentifier = "";
95          String cpuFamily = "";
96          String cpuModel = "";
97          String cpuStepping = "";
98          long cpuVendorFreq = 0L;
99          String processorID;
100         boolean cpu64bit = false;
101 
102         final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
103         try (Arena arena = Arena.ofConfined()) {
104             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
105             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, cpuRegistryRoot);
106             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
107                 try (NativeHandle key = NativeHandle.of(hKey.get(ValueLayout.ADDRESS, 0), Advapi32FFM::RegCloseKey)) {
108                     // Get first subkey name
109                     MemorySegment nameBuffer = arena.allocate(256 * 2L); // WCHAR[256]
110                     MemorySegment nameLen = arena.allocate(ValueLayout.JAVA_INT);
111                     nameLen.set(ValueLayout.JAVA_INT, 0, 256);
112                     if (Advapi32FFM.RegEnumKeyEx(key.get(), 0, nameBuffer, nameLen, MemorySegment.NULL,
113                             MemorySegment.NULL, MemorySegment.NULL, MemorySegment.NULL) == 0) {
114                         String firstKey = WindowsForeignFunctions.readWideString(nameBuffer);
115                         String cpuRegistryPath = cpuRegistryRoot + firstKey;
116                         cpuVendor = registryGetString(arena, cpuRegistryPath, "VendorIdentifier");
117                         cpuName = registryGetString(arena, cpuRegistryPath, "ProcessorNameString");
118                         cpuIdentifier = registryGetString(arena, cpuRegistryPath, "Identifier");
119                         cpuVendorFreq = registryGetDword(arena, cpuRegistryPath, "~MHz") * 1_000_000L;
120                     }
121                 }
122             }
123         } catch (Throwable t) {
124             LOG.debug("Failed to read processor registry info: {}", t.getMessage());
125         }
126 
127         if (!cpuIdentifier.isEmpty()) {
128             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
129             cpuModel = parseIdentifier(cpuIdentifier, "Model");
130             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
131         }
132 
133         // GetNativeSystemInfo to determine 64-bit
134         try (Arena arena = Arena.ofConfined()) {
135             // SYSTEM_INFO is 48 bytes on x64
136             MemorySegment sysInfo = arena.allocate(48);
137             if (Kernel32FFM.GetNativeSystemInfo(sysInfo)) {
138                 // wProcessorArchitecture is at offset 0 (WORD in union)
139                 int arch = Short.toUnsignedInt(sysInfo.get(ValueLayout.JAVA_SHORT, 0));
140                 if (arch == 9 || arch == 12 || arch == 6) { // AMD64, ARM64, IA64
141                     cpu64bit = true;
142                 }
143             }
144         }
145 
146         var processorIdResult = Win32ProcessorFFM.queryProcessorId();
147         if (processorIdResult.getResultCount() > 0) {
148             processorID = WmiUtil.getString(processorIdResult, ProcessorIdProperty.PROCESSORID, 0);
149         } else {
150             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
151                     cpu64bit ? new String[] { "ia64" } : new String[0]);
152         }
153         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
154                 cpuVendorFreq);
155     }

157     @Override
158     protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
159         Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> lpi = LogicalProcessorInformationFFM
160                 .getLogicalProcessorInformationEx();
161         buildNumaNodeProcMap(lpi.getA());
162 
163         List<String> featureFlags = IntStream.range(0, PROCESSOR_FEATURES.length)
164                 .filter(i -> Kernel32FFM.IsProcessorFeaturePresent(PROCESSOR_FEATURES[i]))
165                 .mapToObj(i -> PROCESSOR_FEATURE_NAMES[i]).toList();
166         return new Quartet<>(lpi.getA(), lpi.getB(), lpi.getC(), featureFlags);
167     }

169     @Override
170     public long[] querySystemCpuLoadTicks() {
171         long[] ticks = new long[TickType.values().length];
172         if (USE_LEGACY_SYSTEM_COUNTERS) {
173             try (Arena arena = Arena.ofConfined()) {
174                 // FILETIME is 8 bytes (dwLowDateTime + dwHighDateTime)
175                 MemorySegment lpIdleTime = arena.allocate(8);
176                 MemorySegment lpKernelTime = arena.allocate(8);
177                 MemorySegment lpUserTime = arena.allocate(8);
178                 if (!Kernel32FFM.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
179                     LOG.error("Failed to update system idle/kernel/user times. Error code: {}",
180                             Kernel32FFM.GetLastError().orElse(-1));
181                     return ticks;
182                 }
183 
184                 Map<SystemTickCountProperty, Long> valueMap = ProcessorInformationFFM.querySystemCounters();
185                 ticks[TickType.IRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTINTERRUPTTIME, 0L)
186                         / 10_000L;
187                 ticks[TickType.SOFTIRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTDPCTIME, 0L)
188                         / 10_000L;
189 
190                 ticks[TickType.IDLE.getIndex()] = lpIdleTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L;
191                 ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L
192                         - ticks[TickType.IDLE.getIndex()];
193                 ticks[TickType.USER.getIndex()] = lpUserTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L;
194                 ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()]
195                         + ticks[TickType.SOFTIRQ.getIndex()];
196             }
197             return ticks;
198         }
199         long[][] procTicks = getProcessorCpuLoadTicks();
200         for (int i = 0; i < ticks.length; i++) {
201             for (long[] procTick : procTicks) {
202                 ticks[i] += procTick[i];
203             }
204         }
205         return ticks;
206     }

208     @Override
209     public long[] queryCurrentFreq() {
210         if (VersionHelpersFFM.IsWindows7OrGreater()) {
211             long maxFreq = this.getMaxFreq();
212             if (maxFreq > 0) {
213                 // Prefer % Processor Performance from WMI formatted table (Win8+, reports >100% with turbo)
214                 Pair<List<String>, Map<ProcessorPerformanceProperty, List<Long>>> perfPair = ProcessorInformationFFM
215                         .queryProcessorPerformanceCounters();
216                 List<Long> perfList = perfPair.getB().get(ProcessorPerformanceProperty.PERCENTPROCESSORPERFORMANCE);
217                 long[] freqs = mapPercentToFreqs(perfPair.getA(), perfList, maxFreq);
218                 if (freqs != null) {
219                     return freqs;
220                 }
221                 // Fall back to % of Maximum Frequency (Win7, caps at 100%)
222                 Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> freqPair = ProcessorInformationFFM
223                         .queryFrequencyCounters();
224                 List<Long> percentMaxList = freqPair.getB().get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
225                 freqs = mapPercentToFreqs(freqPair.getA(), percentMaxList, maxFreq);
226                 if (freqs != null) {
227                     return freqs;
228                 }
229             }
230         }
231         return queryNTPower(2);
232     }

234     private long[] mapPercentToFreqs(List<String> instances, List<Long> percentList, long maxFreq) {
235         if (instances.isEmpty() || percentList == null) {
236             return null;
237         }
238         long[] freqs = new long[getLogicalProcessorCount()];
239         boolean populated = false;
240         for (int i = 0; i < instances.size(); i++) {
241             String instance = instances.get(i);
242             int cpu;
243             if (instance.contains(",")) {
244                 if (!getNumaNodeProcToLogicalProcMap().containsKey(instance)) {
245                     continue;
246                 }
247                 cpu = getNumaNodeProcToLogicalProcMap().get(instance);
248             } else {
249                 cpu = ParseUtil.parseIntOrDefault(instance, -1);
250             }
251             if (cpu < 0 || cpu >= getLogicalProcessorCount() || i >= percentList.size()) {
252                 continue;
253             }
254             freqs[cpu] = percentList.get(i) * maxFreq / 100L;
255             if (freqs[cpu] > 0) {
256                 populated = true;
257             }
258         }
259         return populated ? freqs : null;
260     }

264         long[] freqs = queryNTPower(1);
265         return Arrays.stream(freqs).max().orElse(-1L);

268     private long[] queryNTPower(int fieldIndex) {
269         long[] freqs = new long[getLogicalProcessorCount()];
270         int totalSize = PPI_SIZE * freqs.length;
271         try (Arena arena = Arena.ofConfined()) {
272             MemorySegment buffer = arena.allocate(totalSize);
273             int status = Kernel32FFM.CallNtPowerInformation(PROCESSOR_INFORMATION_LEVEL, MemorySegment.NULL, 0, buffer,
274                     totalSize);
275             if (status != 0) {
276                 LOG.error("Unable to get Processor Information, status: {}", status);
277                 Arrays.fill(freqs, -1L);
278                 return freqs;
279             }
280             for (int i = 0; i < freqs.length; i++) {
281                 long offset = i * (long) PPI_SIZE;
282                 // ProcessorPowerInformation: number(4), maxMhz(4), currentMhz(4), mhzLimit(4), maxIdleState(4),
283                 // currentIdleState(4)
284                 freqs[i] = switch (fieldIndex) {
285                     case 1 -> Integer.toUnsignedLong(buffer.get(ValueLayout.JAVA_INT, offset + 4)) * 1_000_000L;
286                     case 2 -> Integer.toUnsignedLong(buffer.get(ValueLayout.JAVA_INT, offset + 8)) * 1_000_000L;
287                     default -> -1L;
288                 };
289                 if (freqs[i] == 0) {
290                     freqs[i] = getProcessorIdentifier().getVendorFreq();
291                 }
292             }
293         }
294         return freqs;
295     }

326     private static String registryGetString(Arena arena, String path, String valueName) {
327         try {
328             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
329             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, path);
330             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
331                 try (NativeHandle key = NativeHandle.of(hKey.get(ValueLayout.ADDRESS, 0), Advapi32FFM::RegCloseKey)) {
332                     MemorySegment valueNameSeg = WindowsForeignFunctions.toWideString(arena, valueName);
333                     MemorySegment dataSize = arena.allocate(ValueLayout.JAVA_INT);
334                     dataSize.set(ValueLayout.JAVA_INT, 0, 512);
335                     MemorySegment data = arena.allocate(512);
336                     if (Advapi32FFM.RegQueryValueEx(key.get(), valueNameSeg, 0, MemorySegment.NULL, data,
337                             dataSize) == 0) {
338                         return WindowsForeignFunctions.readWideString(data);
339                     }
340                 }
341             }
342         } catch (Throwable t) {
343             LOG.debug("Failed to read registry string {}/{}: {}", path, valueName, t.getMessage());
344         }
345         return "";
346     }

348     private static long registryGetDword(Arena arena, String path, String valueName) {
349         try {
350             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
351             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, path);
352             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
353                 try (NativeHandle key = NativeHandle.of(hKey.get(ValueLayout.ADDRESS, 0), Advapi32FFM::RegCloseKey)) {
354                     MemorySegment valueNameSeg = WindowsForeignFunctions.toWideString(arena, valueName);
355                     MemorySegment dataSize = arena.allocate(ValueLayout.JAVA_INT);
356                     dataSize.set(ValueLayout.JAVA_INT, 0, 4);
357                     MemorySegment data = arena.allocate(4);
358                     if (Advapi32FFM.RegQueryValueEx(key.get(), valueNameSeg, 0, MemorySegment.NULL, data,
359                             dataSize) == 0) {
360                         return Integer.toUnsignedLong(data.get(ValueLayout.JAVA_INT, 0));
361                     }
362                 }
363             }
364         } catch (Throwable t) {
365             LOG.debug("Failed to read registry DWORD {}/{}: {}", path, valueName, t.getMessage());
366         }
367         return 0L;
368     }
