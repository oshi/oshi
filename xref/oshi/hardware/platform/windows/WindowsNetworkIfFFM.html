25  /**
26   * WindowsNetworks class using FFM.
27   */
28  @ThreadSafe
29  public final class WindowsNetworkIfFFM extends AbstractNetworkIF {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(WindowsNetworkIfFFM.class);
32  
33      private static final byte CONNECTOR_PRESENT_BIT = 0b00000100;
34  
35      private int ifType;
36      private int ndisPhysicalMediumType;
37      private boolean connectorPresent;
38      private String ifAlias;
39      private IfOperStatus ifOperStatus;
40  
41      public WindowsNetworkIfFFM(NetworkInterface netint) throws InstantiationException {
42          super(netint);
43          updateAttributes();
44      }
45  
46      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
47          List<NetworkIF> ifList = new ArrayList<>();
48          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
49              try {
50                  ifList.add(new WindowsNetworkIfFFM(ni));
51              } catch (InstantiationException e) {
52                  LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
53              }
54          }
55          return ifList;
56      }
57  
58      @Override
59      public int getIfType() {
60          return this.ifType;
61      }
62  
63      @Override
64      public int getNdisPhysicalMediumType() {
65          return this.ndisPhysicalMediumType;
66      }
67  
68      @Override
69      public boolean isConnectorPresent() {
70          return this.connectorPresent;
71      }
72  
73      @Override
74      public String getIfAlias() {
75          return ifAlias;
76      }
77  
78      @Override
79      public IfOperStatus getIfOperStatus() {
80          return ifOperStatus;
81      }
82  
83      @Override
84      public boolean updateAttributes() {
85          // FFM targets JDK 25+ which requires Vista+; use GetIfEntry2 directly
86          boolean success = callInArenaBooleanOrDefault(arena -> {
87              MemorySegment ifRow = arena.allocate(IPHlpAPIFFM.MIB_IF_ROW2_SIZE);
88              ifRow.fill((byte) 0);
89              // Set InterfaceIndex at offset 8
90              ifRow.set(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_INTERFACE_INDEX, queryNetworkInterface().getIndex());
91              if (0 != IPHlpAPIFFM.GetIfEntry2(ifRow)) {
92                  LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
93                          getName());
94                  return false;
95              }
96              this.ifType = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_TYPE);
97              this.ndisPhysicalMediumType = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_PHYSICAL_MEDIUM_TYPE);
98              byte flags = ifRow.get(ValueLayout.JAVA_BYTE, IPHlpAPIFFM.OFFSET_FLAGS);
99              this.connectorPresent = (flags & CONNECTOR_PRESENT_BIT) > 0;
100             this.speed = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_RECEIVE_LINK_SPEED);
101             this.bytesRecv = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_OCTETS);
102             this.packetsRecv = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_UCAST_PKTS);
103             this.inErrors = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_ERRORS);
104             this.inDrops = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_DISCARDS);
105             this.bytesSent = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_OCTETS);
106             this.packetsSent = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_UCAST_PKTS);
107             this.outErrors = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_ERRORS);
108             this.collisions = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_DISCARDS);
109             // Alias is WCHAR[257] at OFFSET_ALIAS
110             MemorySegment aliasSlice = ifRow.asSlice(IPHlpAPIFFM.OFFSET_ALIAS, 257 * 2L);
111             this.ifAlias = readWideString(aliasSlice);
112             int operStatus = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_OPER_STATUS);
113             this.ifOperStatus = IfOperStatus.byValue(operStatus);
114             return true;
115         }, LOG, ERROR, "Failed to retrieve data for interface " + queryNetworkInterface().getIndex() + ", " + getName(),
116                 false);
117         if (success) {
118             this.timeStamp = System.currentTimeMillis();
119         }
120         return success;
121     }
122 
123     private static String readWideString(MemorySegment seg) {
124         int len = 0;
125         while (len < seg.byteSize() / 2 && seg.get(ValueLayout.JAVA_SHORT, len * 2L) != 0) {
126             len++;
127         }
128         byte[] bytes = new byte[len * 2];
129         MemorySegment.copy(seg, ValueLayout.JAVA_BYTE, 0, bytes, 0, bytes.length);
130         return new String(bytes, StandardCharsets.UTF_16LE);
131     }
