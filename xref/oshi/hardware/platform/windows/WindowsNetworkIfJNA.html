5   package oshi.hardware.platform.windows;
6   
7   import java.net.NetworkInterface;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import com.sun.jna.Native;
15  import com.sun.jna.platform.win32.IPHlpAPI;
16  import com.sun.jna.platform.win32.VersionHelpers;
17  
18  import oshi.annotation.concurrent.ThreadSafe;
19  import oshi.hardware.NetworkIF;
20  import oshi.hardware.common.AbstractNetworkIF;
21  import oshi.jna.Struct.CloseableMibIfRow;
22  import oshi.jna.Struct.CloseableMibIfRow2;
23  import oshi.util.ParseUtil;
24  
25  /**
26   * WindowsNetworks class.
27   */
28  @ThreadSafe
29  public final class WindowsNetworkIfJNA extends AbstractNetworkIF {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(WindowsNetworkIfJNA.class);
32  
33      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
34      private static final byte CONNECTOR_PRESENT_BIT = 0b00000100;
35  
36      private int ifType;
37      private int ndisPhysicalMediumType;
38      private boolean connectorPresent;
39      private String ifAlias;
40      private IfOperStatus ifOperStatus;
41  
42      public WindowsNetworkIfJNA(NetworkInterface netint) throws InstantiationException {
43          super(netint);
44          updateAttributes();
45      }
46  
47      /**
48       * Gets all network interfaces on this machine
49       *
50       * @param includeLocalInterfaces include local interfaces in the result
51       * @return A list of {@link NetworkIF} objects representing the interfaces
52       */
53      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
54          List<NetworkIF> ifList = new ArrayList<>();
55          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
56              try {
57                  ifList.add(new WindowsNetworkIfJNA(ni));
58              } catch (InstantiationException e) {
59                  LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
60              }
61          }
62          return ifList;
63      }
64  
65      @Override
66      public int getIfType() {
67          return this.ifType;
68      }
69  
70      @Override
71      public int getNdisPhysicalMediumType() {
72          return this.ndisPhysicalMediumType;
73      }
74  
75      @Override
76      public boolean isConnectorPresent() {
77          return this.connectorPresent;
78      }
79  
80      @Override
81      public String getIfAlias() {
82          return ifAlias;
83      }
84  
85      @Override
86      public IfOperStatus getIfOperStatus() {
87          return ifOperStatus;
88      }
89  
90      @Override
91      public boolean updateAttributes() {
92          // MIB_IFROW2 requires Vista (6.0) or later.
93          if (IS_VISTA_OR_GREATER) {
94              // Create new MIB_IFROW2 and set index to this interface index
95              try (CloseableMibIfRow2 ifRow = new CloseableMibIfRow2()) {
96                  ifRow.InterfaceIndex = queryNetworkInterface().getIndex();
97                  if (0 != IPHlpAPI.INSTANCE.GetIfEntry2(ifRow)) {
98                      // Error, abort
99                      LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
100                             getName());
101                     return false;
102                 }
103                 this.ifType = ifRow.Type;
104                 this.ndisPhysicalMediumType = ifRow.PhysicalMediumType;
105                 this.connectorPresent = (ifRow.InterfaceAndOperStatusFlags & CONNECTOR_PRESENT_BIT) > 0;
106                 setBytesSent(ifRow.OutOctets);
107                 setBytesRecv(ifRow.InOctets);
108                 setPacketsSent(ifRow.OutUcastPkts);
109                 setPacketsRecv(ifRow.InUcastPkts);
110                 setOutErrors(ifRow.OutErrors);
111                 setInErrors(ifRow.InErrors);
112                 setCollisions(ifRow.OutDiscards);
113                 setInDrops(ifRow.InDiscards);
114                 setSpeed(ifRow.ReceiveLinkSpeed);
115                 this.ifAlias = Native.toString(ifRow.Alias);
116                 this.ifOperStatus = IfOperStatus.byValue(ifRow.OperStatus);
117             }
118         } else {
119             // Create new MIB_IFROW and set index to this interface index
120             try (CloseableMibIfRow ifRow = new CloseableMibIfRow()) {
121                 ifRow.dwIndex = queryNetworkInterface().getIndex();
122                 if (0 != IPHlpAPI.INSTANCE.GetIfEntry(ifRow)) {
123                     // Error, abort
124                     LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
125                             getName());
126                     return false;
127                 }
128                 this.ifType = ifRow.dwType;
129                 // These are unsigned ints. Widen them to longs.
130                 setBytesSent(ParseUtil.unsignedIntToLong(ifRow.dwOutOctets));
131                 setBytesRecv(ParseUtil.unsignedIntToLong(ifRow.dwInOctets));
132                 setPacketsSent(ParseUtil.unsignedIntToLong(ifRow.dwOutUcastPkts));
133                 setPacketsRecv(ParseUtil.unsignedIntToLong(ifRow.dwInUcastPkts));
134                 setOutErrors(ParseUtil.unsignedIntToLong(ifRow.dwOutErrors));
135                 setInErrors(ParseUtil.unsignedIntToLong(ifRow.dwInErrors));
136                 setCollisions(ParseUtil.unsignedIntToLong(ifRow.dwOutDiscards));
137                 setInDrops(ParseUtil.unsignedIntToLong(ifRow.dwInDiscards));
138                 setSpeed(ParseUtil.unsignedIntToLong(ifRow.dwSpeed));
139                 this.ifAlias = ""; // not supported by MIB_IFROW
140                 this.ifOperStatus = IfOperStatus.UNKNOWN; // not supported
141             }
142         }
143         setTimeStamp(System.currentTimeMillis());
144         return true;
145     }
146 }
