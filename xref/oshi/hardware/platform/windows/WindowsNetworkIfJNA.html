6   
7   import java.net.NetworkInterface;
8   import java.util.List;
9   
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import com.sun.jna.Native;
14  import com.sun.jna.platform.win32.IPHlpAPI;
15  import com.sun.jna.platform.win32.VersionHelpers;
16  
17  import oshi.annotation.concurrent.ThreadSafe;
18  import oshi.hardware.NetworkIF;
19  import oshi.hardware.common.platform.windows.WindowsNetworkIF;
20  import oshi.jna.Struct.CloseableMibIfRow;
21  import oshi.jna.Struct.CloseableMibIfRow2;
22  import oshi.util.ParseUtil;
23  
24  /**
25   * WindowsNetworks class.
26   * <p>
27   * Not {@code final} so that tests can subclass it to force the {@link #isVistaOrGreater()} version gate and exercise
28   * the pre-Vista {@code MIB_IFROW} query path against the real system.
29   */
30  @ThreadSafe
31  public class WindowsNetworkIfJNA extends WindowsNetworkIF {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(WindowsNetworkIfJNA.class);
34  
35      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
36  
37      public WindowsNetworkIfJNA(NetworkInterface netint) throws InstantiationException {
38          super(netint);
39      }
40  
41      /**
42       * Gets all network interfaces on this machine
43       *
44       * @param includeLocalInterfaces include local interfaces in the result
45       * @return A list of {@link NetworkIF} objects representing the interfaces
46       */
47      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
48          return getNetworks(includeLocalInterfaces, WindowsNetworkIfJNA::new);
49      }
50  
51      /**
52       * Whether {@code MIB_IFROW2} (Vista+) is available. Overridable so tests can force the pre-Vista {@code MIB_IFROW}
53       * path.
54       *
55       * @return {@code true} on Windows Vista or later
56       */
57      protected boolean isVistaOrGreater() {
58          return IS_VISTA_OR_GREATER;
59      }
60  
61      @Override
62      protected IfStats queryStats() {
63          return isVistaOrGreater() ? queryStatsVista() : queryStatsLegacy();
64      }
65  
66      private IfStats queryStatsVista() {
67          // MIB_IFROW2 requires Vista (6.0) or later.
68          try (CloseableMibIfRow2 ifRow = new CloseableMibIfRow2()) {
69              ifRow.InterfaceIndex = queryNetworkInterface().getIndex();
70              if (0 != IPHlpAPI.INSTANCE.GetIfEntry2(ifRow)) {
71                  LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
72                          getName());
73                  return null;
74              }
75              IfStats stats = new IfStats();
76              stats.ifType = ifRow.Type;
77              stats.ndisPhysicalMediumType = ifRow.PhysicalMediumType;
78              stats.connectorPresent = (ifRow.InterfaceAndOperStatusFlags & CONNECTOR_PRESENT_BIT) > 0;
79              stats.bytesSent = ifRow.OutOctets;
80              stats.bytesRecv = ifRow.InOctets;
81              stats.packetsSent = ifRow.OutUcastPkts;
82              stats.packetsRecv = ifRow.InUcastPkts;
83              stats.outErrors = ifRow.OutErrors;
84              stats.inErrors = ifRow.InErrors;
85              stats.collisions = ifRow.OutDiscards;
86              stats.inDrops = ifRow.InDiscards;
87              stats.speed = ifRow.ReceiveLinkSpeed;
88              stats.ifAlias = Native.toString(ifRow.Alias);
89              stats.ifOperStatus = IfOperStatus.byValue(ifRow.OperStatus);
90              return stats;
91          }
92      }
93  
94      private IfStats queryStatsLegacy() {
95          // Pre-Vista MIB_IFROW: a narrower field set with unsigned 32-bit counters widened to long. Physical medium
96          // type, connector-present, alias, and oper-status are not available and keep their IfStats defaults.
97          try (CloseableMibIfRow ifRow = new CloseableMibIfRow()) {
98              ifRow.dwIndex = queryNetworkInterface().getIndex();
99              if (0 != IPHlpAPI.INSTANCE.GetIfEntry(ifRow)) {
100                 LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
101                         getName());
102                 return null;
103             }
104             IfStats stats = new IfStats();
105             stats.ifType = ifRow.dwType;
106             stats.bytesSent = ParseUtil.unsignedIntToLong(ifRow.dwOutOctets);
107             stats.bytesRecv = ParseUtil.unsignedIntToLong(ifRow.dwInOctets);
108             stats.packetsSent = ParseUtil.unsignedIntToLong(ifRow.dwOutUcastPkts);
109             stats.packetsRecv = ParseUtil.unsignedIntToLong(ifRow.dwInUcastPkts);
110             stats.outErrors = ParseUtil.unsignedIntToLong(ifRow.dwOutErrors);
111             stats.inErrors = ParseUtil.unsignedIntToLong(ifRow.dwInErrors);
112             stats.collisions = ParseUtil.unsignedIntToLong(ifRow.dwOutDiscards);
113             stats.inDrops = ParseUtil.unsignedIntToLong(ifRow.dwInDiscards);
114             stats.speed = ParseUtil.unsignedIntToLong(ifRow.dwSpeed);
115             return stats;
116         }
117     }
