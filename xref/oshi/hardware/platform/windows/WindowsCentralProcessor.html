217     /**
218      * Initializes Class variables
219      */
220     private void initVars() {
221         final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
222         String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
223         if (processorIds.length > 0) {
224             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
225             setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
226                     "VendorIdentifier"));
227             setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
228                     "ProcessorNameString"));
229             setIdentifier(
230                     Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
231         }
232         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
233         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
234         if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
235                 || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
236             setCpu64(true);
237         } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
238             setCpu64(false);
239         }
240 
241         WmiQuery<ProcessorProperty> processorIdQuery = new WmiQuery<>("Win32_Processor", ProcessorProperty.class);
242         WmiResult<ProcessorProperty> processorId = WmiQueryHandler.createInstance().queryWMI(processorIdQuery);
243         if (processorId.getResultCount() > 0) {
244             setProcessorID(WmiUtil.getString(processorId, ProcessorProperty.PROCESSORID, 0));
245         }
246     }

248     /**
249      * Updates logical and physical processor counts
250      */
251     @Override
252     protected void calculateProcessorCounts() {
253         // Get number of logical processors
254         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
255         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
256         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
257 
258         // Get number of physical processors
259         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
260         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
261             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
262                 this.physicalPackageCount++;
263             }
264             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
265                 this.physicalProcessorCount++;
266             }
267         }
268     }

270     /**
271      * {@inheritDoc}
272      */
273     @Override
274     public long[] getSystemCpuLoadTicks() {
275         long[] ticks = new long[TickType.values().length];
276         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
277         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
278         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
279         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
280             LOG.error("Failed to update system idle/kernel/user times. Error code: {}", Native.getLastError());
281             return ticks;
282         }
283         // IOwait:
284         // Windows does not measure IOWait.
285 
286         // IRQ and ticks:
287         // Percent time raw value is cumulative 100NS-ticks
288         // Divide by 10_000 to get milliseconds
289 
290         Map<SystemTickCountProperty, Long> valueMap = this.systemTickPerfCounters.queryValues();
291         ticks[TickType.IRQ.getIndex()] = MapUtil.getOrDefault(valueMap, SystemTickCountProperty.PERCENTINTERRUPTTIME,
292                 0L) / 10_000L;
293         ticks[TickType.SOFTIRQ.getIndex()] = MapUtil.getOrDefault(valueMap, SystemTickCountProperty.PERCENTDPCTIME, 0L)
294                 / 10_000L;
295 
296         ticks[TickType.IDLE.getIndex()] = lpIdleTime.toDWordLong().longValue() / 10_000L;
297         ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.toDWordLong().longValue() / 10_000L
298                 - ticks[TickType.IDLE.getIndex()];
299         ticks[TickType.USER.getIndex()] = lpUserTime.toDWordLong().longValue() / 10_000L;
300         // Additional decrement to avoid double counting in the total array
301         ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()] + ticks[TickType.SOFTIRQ.getIndex()];
302         return ticks;
303     }

305     /**
306      * {@inheritDoc}
307      */
308     @Override
309     public double[] getSystemLoadAverage(int nelem) {
310         if (nelem < 1 || nelem > 3) {
311             throw new IllegalArgumentException("Must include from one to three elements.");
312         }
313         double[] average = new double[nelem];
314         // Windows doesn't have load average
315         for (int i = 0; i < average.length; i++) {
316             average[i] = -1;
317         }
318         return average;
319     }

321     /**
322      * {@inheritDoc}
323      */
324     @Override
325     public long[][] getProcessorCpuLoadTicks() {
326         Map<ProcessorTickCountProperty, List<Long>> valueMap = this.processorTickPerfCounters.queryValuesWildcard();
327         List<String> instances = this.processorTickPerfCounters.getInstancesFromLastQuery();
328         List<Long> systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
329         List<Long> userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
330         List<Long> irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
331         List<Long> softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
332         // % Processor Time is actually Idle time
333         List<Long> idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
334 
335         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
336         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
337                 || idleList == null) {
338             return ticks;
339         }
340         for (int p = 0; p < instances.size(); p++) {
341             int cpu = ParseUtil.parseIntOrDefault(instances.get(p), 0);
342             if (cpu >= this.logicalProcessorCount) {
343                 continue;
344             }
345             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
346             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
347             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
348             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
349             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
350 
351             // Additional decrement to avoid double counting in the
352             // total array
353             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
354                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
355 
356             // Raw value is cumulative 100NS-ticks
357             // Divide by 10_000 to get milliseconds
358             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
359             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
360             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
361             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
362             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
363         }
364         // Skipping nice and IOWait, they'll stay 0
365         return ticks;
366     }

368     /**
369      * {@inheritDoc}
370      */
371     @Override
372     public long getSystemUptime() {
373         // Uptime is in seconds so divide milliseconds
374         // GetTickCount64 requires Vista (6.0) or later
375         if (IS_VISTA_OR_GREATER) {
376             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
377         } else {
378             // 32 bit rolls over at ~ 49 days
379             return Kernel32.INSTANCE.GetTickCount() / 1000L;
380         }
381     }
