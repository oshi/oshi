19  package oshi.hardware.platform.windows;
20  
21  import java.util.List;
22  import java.util.Map;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import com.sun.jna.platform.win32.Kernel32;
28  
29  import oshi.hardware.common.AbstractGlobalMemory;
30  import oshi.jna.platform.windows.Psapi;
31  import oshi.jna.platform.windows.Psapi.PERFORMANCE_INFORMATION;
32  import oshi.util.platform.windows.WmiUtil;
33  
34  /**
35   * Memory obtained by GlobalMemoryStatusEx.
36   *
37   * @author dblock[at]dblock[dot]org
38   */
39  public class WindowsGlobalMemory extends AbstractGlobalMemory {
40  
41      private static final long serialVersionUID = 1L;
42  
43      private static final Logger LOG = LoggerFactory.getLogger(WindowsGlobalMemory.class);
44  
45      private transient PERFORMANCE_INFORMATION perfInfo = new PERFORMANCE_INFORMATION();
46  
47      private long lastUpdate = 0;
48  
49      /**
50       * Update the performance information no more frequently than every 100ms
51       */
52      @Override
53      protected void updateMeminfo() {
54          long now = System.currentTimeMillis();
55          if (now - this.lastUpdate > 100) {
56              if (!Psapi.INSTANCE.GetPerformanceInfo(this.perfInfo, this.perfInfo.size())) {
57                  LOG.error("Failed to get Performance Info. Error code: {}", Kernel32.INSTANCE.GetLastError());
58                  return;
59              }
60              this.memAvailable = this.perfInfo.PageSize.longValue() * this.perfInfo.PhysicalAvailable.longValue();
61              this.memTotal = this.perfInfo.PageSize.longValue() * this.perfInfo.PhysicalTotal.longValue();
62              this.swapTotal = this.perfInfo.PageSize.longValue()
63                      * (this.perfInfo.CommitLimit.longValue() - this.perfInfo.PhysicalTotal.longValue());
64              this.lastUpdate = now;
65          }
66      }
67  
68      /**
69       * {@inheritDoc}
70       */
71      @Override
72      protected void updateSwap() {
73          updateMeminfo();
74          Map<String, List<Long>> usage = WmiUtil.selectUint32sFrom(null, "Win32_PerfRawData_PerfOS_PagingFile",
75                  "PercentUsage,PercentUsage_Base", "WHERE Name=\"_Total\"");
76          if (!usage.get("PercentUsage").isEmpty()) {
77              this.swapUsed = this.swapTotal * usage.get("PercentUsage").get(0) / usage.get("PercentUsage_Base").get(0);
78          }
79      }
80  }
