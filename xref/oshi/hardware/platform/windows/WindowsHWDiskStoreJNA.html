44      private static final Logger LOG = LoggerFactory.getLogger(WindowsHWDiskStoreJNA.class);
45  
46      private WindowsHWDiskStoreJNA(String name, String model, String serial, long size) {
47          super(name, model, serial, size);
48      }
49  
50      @Override
51      protected DiskStats queryReadWriteStats(String index) {
52          return populateDiskStats(index, PhysicalDiskJNA.queryDiskCounters());
53      }
54  
55      /**
56       * Gets the disks on this machine
57       *
58       * @return a list of {@link HWDiskStore} objects representing the disks
59       */
60      public static List<HWDiskStore> getDisks() {
61          WmiQueryHandler h = Objects.requireNonNull(WmiQueryHandler.createInstance());
62          boolean comInit = false;
63          try {
64              comInit = h.initCOM();
65              List<HWDiskStore> result = new ArrayList<>();
66              DiskStats stats = populateDiskStats(null, PhysicalDiskJNA.queryDiskCounters());
67              PartitionMaps maps = queryPartitionMaps(h);
68  
69              WmiResult<DiskDriveProperty> vals = Win32DiskDriveJNA.queryDiskDrive(h);
70              for (int i = 0; i < vals.getResultCount(); i++) {
71                  WindowsHWDiskStoreJNA ds = new WindowsHWDiskStoreJNA(WmiUtil.getString(vals, DiskDriveProperty.NAME, i),
72                          String.format(Locale.ROOT, "%s %s", WmiUtil.getString(vals, DiskDriveProperty.MODEL, i),
73                                  WmiUtil.getString(vals, DiskDriveProperty.MANUFACTURER, i)).trim(),
74                          ParseUtil.hexStringToString(WmiUtil.getString(vals, DiskDriveProperty.SERIALNUMBER, i)),
75                          WmiUtil.getUint64(vals, DiskDriveProperty.SIZE, i));
76  
77                  String index = Integer.toString(WmiUtil.getUint32(vals, DiskDriveProperty.INDEX, i));
78                  ds.setDiskStats(stats, index);
79                  ds.setPartitionList(buildPartitionList(maps, ds.getName()));
80                  result.add(ds);
81              }
82              return result;
83          } catch (COMException e) {
84              LOG.warn("COM exception: {}", e.getMessage());
85              return Collections.emptyList();
86          } finally {
87              if (comInit) {
88                  h.unInitCOM();
89              }
90          }
91      }
92  
93      private static PartitionMaps queryPartitionMaps(WmiQueryHandler h) {
94          PartitionMaps maps = new PartitionMaps();
95  
96          // Map drives to partitions
97          WmiResult<DriveToPartitionProperty> drivePartitionMap = Win32DiskDriveToDiskPartitionJNA
98                  .queryDriveToPartition(h);
99          for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
100             mapDriveToPartition(maps, WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.ANTECEDENT, i),
101                     WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.DEPENDENT, i));
102         }
103 
104         // Map partitions to logical disks
105         WmiResult<DiskToPartitionProperty> diskPartitionMap = Win32LogicalDiskToPartitionJNA.queryDiskToPartition(h);
106         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
107             long size = WmiUtil.getUint64(diskPartitionMap, DiskToPartitionProperty.ENDINGADDRESS, i)
108                     - WmiUtil.getUint64(diskPartitionMap, DiskToPartitionProperty.STARTINGADDRESS, i) + 1L;
109             mapPartitionToLogicalDrive(maps,
110                     WmiUtil.getRefString(diskPartitionMap, DiskToPartitionProperty.ANTECEDENT, i),
111                     WmiUtil.getRefString(diskPartitionMap, DiskToPartitionProperty.DEPENDENT, i), size);
112         }
113 
114         // Get all partitions and create objects
115         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = Win32DiskPartitionJNA.queryPartition(h);
116         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
117             String deviceID = WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DEVICEID, i);
118             List<Pair<String, Long>> logicalDrives = maps.getPartitionToLogicalDriveMap().get(deviceID);
119             if (logicalDrives == null) {
120                 continue;
121             }
122             for (int j = 0; j < logicalDrives.size(); j++) {
123                 Pair<String, Long> logicalDrive = logicalDrives.get(j);
124                 if (logicalDrive != null && !logicalDrive.getA().isEmpty()) {
125                     char[] volumeChr = new char[GUID_BUFSIZE];
126                     Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive.getA(), volumeChr, GUID_BUFSIZE);
127                     String uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
128                     char[] labelChr = new char[LABEL_BUFSIZE];
129                     String label = "";
130                     if (Kernel32.INSTANCE.GetVolumeInformation(logicalDrive.getA(), labelChr, LABEL_BUFSIZE, null, null,
131                             null, null, 0)) {
132                         label = new String(labelChr).trim();
133                     } else {
134                         int error = Kernel32.INSTANCE.GetLastError();
135                         LOG.debug("Failed to get volume label for {}: error code {}", logicalDrive.getA(), error);
136                     }
137                     HWPartition pt = new HWPartition(
138                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.NAME, i),
139                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.TYPE, i),
140                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DESCRIPTION, i), uuid, label,
141                             logicalDrive.getB(),
142                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.DISKINDEX, i),
143                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.INDEX, i),
144                             logicalDrive.getA());
145                     maps.getPartitionMap().computeIfAbsent(deviceID, x -> new ArrayList<>()).add(pt);
146                 }
147             }
148         }
149         return maps;
150     }
