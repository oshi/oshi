29  /**
30   * Virtual memory obtained from Performance Info using FFM.
31   */
32  @ThreadSafe
33  final class WindowsVirtualMemoryFFM extends WindowsVirtualMemory {
34  
35      private static final Logger LOG = LoggerFactory.getLogger(WindowsVirtualMemoryFFM.class);
36  
37      WindowsVirtualMemoryFFM(WindowsGlobalMemoryFFM windowsGlobalMemory) {
38          super(windowsGlobalMemory);
39      }
40  
41      @Override
42      protected long querySwapUsed() {
43          return PagingFileFFM.querySwapUsed().getOrDefault(PagingPercentProperty.PERCENTUSAGE, 0L);
44      }
45  
46      @Override
47      protected Triplet<Long, Long, Long> querySwapTotalVirtMaxVirtUsed() {
48          return callInArenaOrDefault(arena -> {
49              MemorySegment perfInfo = arena.allocate(PsapiFFM.PERFORMANCE_INFORMATION_LAYOUT);
50              int size = (int) PsapiFFM.PERFORMANCE_INFORMATION_LAYOUT.byteSize();
51              if (!PsapiFFM.GetPerformanceInfo(perfInfo, size)) {
52                  LOG.error("Failed to get Performance Info. Error code: {}", Kernel32FFM.GetLastError().orElse(-1));
53                  return new Triplet<>(0L, 0L, 0L);
54              }
55              long commitLimit = perfInfo.get(ValueLayout.JAVA_LONG, PsapiFFM.PERFORMANCE_INFORMATION_LAYOUT
56                      .byteOffset(MemoryLayout.PathElement.groupElement("CommitLimit")));
57              long commitTotal = perfInfo.get(ValueLayout.JAVA_LONG, PsapiFFM.PERFORMANCE_INFORMATION_LAYOUT
58                      .byteOffset(MemoryLayout.PathElement.groupElement("CommitTotal")));
59              long physTotal = perfInfo.get(ValueLayout.JAVA_LONG, PsapiFFM.PERFORMANCE_INFORMATION_LAYOUT
60                      .byteOffset(MemoryLayout.PathElement.groupElement("PhysicalTotal")));
61              return new Triplet<>(commitLimit - physTotal, commitLimit, commitTotal);
62          }, LOG, ERROR, "Failed to get Performance Info", new Triplet<>(0L, 0L, 0L));
63      }
64  
65      @Override
66      protected Pair<Long, Long> queryPageSwaps() {
67          Map<PageSwapProperty, Long> valueMap = MemoryInformationFFM.queryPageSwaps();
68          return new Pair<>(valueMap.getOrDefault(PageSwapProperty.PAGESINPUTPERSEC, 0L),
69                  valueMap.getOrDefault(PageSwapProperty.PAGESOUTPUTPERSEC, 0L));
70      }
71  }
