24  /**
25   * Windows USB Device using FFM.
26   */
27  @Immutable
28  public class WindowsUsbDeviceFFM extends AbstractUsbDevice {
29  
30      // GUID_DEVINTERFACE_USB_HOST_CONTROLLER {3ABF6F2D-71C4-462A-8A92-1E6861E6AF27}
31      private static final byte[] GUID_DEVINTERFACE_USB_HOST_CONTROLLER = { 0x2D, 0x6F, (byte) 0xBF, 0x3A, (byte) 0xC4,
32              0x71, 0x2A, 0x46, (byte) 0x8A, (byte) 0x92, 0x1E, 0x68, 0x61, (byte) 0xE6, (byte) 0xAF, 0x27 };
33  
34      public WindowsUsbDeviceFFM(String name, String vendor, String vendorId, String productId, String serialNumber,
35              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
36          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
37      }
38  
39      public static List<UsbDevice> getUsbDevices(boolean tree) {
40          List<UsbDevice> devices = queryUsbDevices();
41          if (tree) {
42              return devices;
43          }
44          List<UsbDevice> deviceList = new ArrayList<>();
45          for (UsbDevice device : devices) {
46              addDevicesToList(deviceList, device.getConnectedDevices());
47          }
48          return deviceList;
49      }
50  
51      private static List<UsbDevice> queryUsbDevices() {
52          Quintet<Set<Integer>, Map<Integer, Integer>, Map<Integer, String>, Map<Integer, String>, Map<Integer, String>> controllerDevices = DeviceTreeFFM
53                  .queryDeviceTree(GUID_DEVINTERFACE_USB_HOST_CONTROLLER);
54          Map<Integer, Integer> parentMap = controllerDevices.getB();
55          Map<Integer, String> nameMap = controllerDevices.getC();
56          Map<Integer, String> deviceIdMap = controllerDevices.getD();
57          Map<Integer, String> mfgMap = controllerDevices.getE();
58  
59          List<UsbDevice> usbDevices = new ArrayList<>();
60          for (Integer controllerDevice : controllerDevices.getA()) {
61              WindowsUsbDeviceFFM deviceAndChildren = queryDeviceAndChildren(controllerDevice, parentMap, nameMap,
62                      deviceIdMap, mfgMap, "0000", "0000", "");
63              if (deviceAndChildren != null) {
64                  usbDevices.add(deviceAndChildren);
65              }
66          }
67          return usbDevices;
68      }
69  
70      private static WindowsUsbDeviceFFM queryDeviceAndChildren(Integer device, Map<Integer, Integer> parentMap,
71              Map<Integer, String> nameMap, Map<Integer, String> deviceIdMap, Map<Integer, String> mfgMap, String vid,
72              String pid, String parentSerial) {
73          String vendorId = vid;
74          String productId = pid;
75          String serial = parentSerial;
76          Triplet<String, String, String> idsAndSerial = ParseUtil
77                  .parseDeviceIdToVendorProductSerial(deviceIdMap.get(device));
78          if (idsAndSerial != null) {
79              vendorId = idsAndSerial.getA();
80              productId = idsAndSerial.getB();
81              serial = idsAndSerial.getC();
82              if (serial.isEmpty() && vendorId.equals(vid) && productId.equals(pid)) {
83                  serial = parentSerial;
84              }
85          }
86          Set<Integer> childDeviceSet = parentMap.entrySet().stream().filter(e -> e.getValue().equals(device))
87                  .map(Entry::getKey).collect(Collectors.toSet());
88          List<UsbDevice> childDevices = new ArrayList<>();
89          for (Integer child : childDeviceSet) {
90              WindowsUsbDeviceFFM deviceAndChildren = queryDeviceAndChildren(child, parentMap, nameMap, deviceIdMap,
91                      mfgMap, vendorId, productId, serial);
92              if (deviceAndChildren != null) {
93                  childDevices.add(deviceAndChildren);
94              }
95          }
96          sort(childDevices);
97          if (nameMap.containsKey(device)) {
98              String name = nameMap.get(device);
99              if (name.isEmpty()) {
100                 name = vendorId + ":" + productId;
101             }
102             String deviceId = deviceIdMap.get(device);
103             String mfg = mfgMap.get(device);
104             return new WindowsUsbDeviceFFM(name, mfg, vendorId, productId, serial, deviceId, childDevices);
105         }
106         return null;
107     }
