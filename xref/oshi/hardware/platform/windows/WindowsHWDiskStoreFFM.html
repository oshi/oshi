51      private WindowsHWDiskStoreFFM(String name, String model, String serial, long size) {
52          super(name, model, serial, size);
53      }

55      @Override
56      protected DiskStats queryReadWriteStats(String index) {
57          return populateDiskStats(index, PhysicalDiskFFM.queryDiskCounters());
58      }

60      /**
61       * Gets the disks on this machine
62       *
63       * @return a list of {@link HWDiskStore} objects representing the disks
64       */
65      public static List<HWDiskStore> getDisks() {
66          WmiQueryHandlerFFM h = Objects.requireNonNull(WmiQueryHandlerFFM.createInstance());
67          boolean comInit = false;
68          try {
69              comInit = h.initCOM();
70              List<HWDiskStore> result = new ArrayList<>();
71              DiskStats stats = populateDiskStats(null, PhysicalDiskFFM.queryDiskCounters());
72              PartitionMaps maps = queryPartitionMaps(h);
73  
74              WmiResult<DiskDriveProperty> vals = Win32DiskDriveFFM.queryDiskDrive(h);
75              for (int i = 0; i < vals.getResultCount(); i++) {
76                  WindowsHWDiskStoreFFM ds = new WindowsHWDiskStoreFFM(
77                          WmiUtilFFM.getString(vals, DiskDriveProperty.NAME, i),
78                          String.format(Locale.ROOT, "%s %s", WmiUtilFFM.getString(vals, DiskDriveProperty.MODEL, i),
79                                  WmiUtilFFM.getString(vals, DiskDriveProperty.MANUFACTURER, i)).trim(),
80                          ParseUtil.hexStringToString(WmiUtilFFM.getString(vals, DiskDriveProperty.SERIALNUMBER, i)),
81                          WmiUtilFFM.getUint64(vals, DiskDriveProperty.SIZE, i));
82  
83                  String index = Integer.toString(WmiUtilFFM.getUint32(vals, DiskDriveProperty.INDEX, i));
84                  ds.setDiskStats(stats, index);
85                  ds.setPartitionList(buildPartitionList(maps, ds.getName()));
86                  result.add(ds);
87              }
88              return result;
89          } catch (FfmComException e) {
90              LOG.warn("COM exception: {}", e.getMessage());
91              return Collections.emptyList();
92          } finally {
93              if (comInit) {
94                  h.unInitCOM();
95              }
96          }
97      }

99      private static PartitionMaps queryPartitionMaps(WmiQueryHandlerFFM h) {
100         PartitionMaps maps = new PartitionMaps();
101 
102         // Map drives to partitions
103         WmiResult<DriveToPartitionProperty> drivePartitionMap = Win32DiskDriveToDiskPartitionFFM
104                 .queryDriveToPartition(h);
105         for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
106             mapDriveToPartition(maps,
107                     WmiUtilFFM.getRefString(drivePartitionMap, DriveToPartitionProperty.ANTECEDENT, i),
108                     WmiUtilFFM.getRefString(drivePartitionMap, DriveToPartitionProperty.DEPENDENT, i));
109         }
110 
111         // Map partitions to logical disks
112         WmiResult<DiskToPartitionProperty> diskPartitionMap = Win32LogicalDiskToPartitionFFM.queryDiskToPartition(h);
113         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
114             long size = WmiUtilFFM.getUint64(diskPartitionMap, DiskToPartitionProperty.ENDINGADDRESS, i)
115                     - WmiUtilFFM.getUint64(diskPartitionMap, DiskToPartitionProperty.STARTINGADDRESS, i) + 1L;
116             mapPartitionToLogicalDrive(maps,
117                     WmiUtilFFM.getRefString(diskPartitionMap, DiskToPartitionProperty.ANTECEDENT, i),
118                     WmiUtilFFM.getRefString(diskPartitionMap, DiskToPartitionProperty.DEPENDENT, i), size);
119         }
120 
121         // Get all partitions and create objects
122         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = Win32DiskPartitionFFM.queryPartition(h);
123         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
124             String deviceID = WmiUtilFFM.getString(hwPartitionQueryMap, DiskPartitionProperty.DEVICEID, i);
125             List<Pair<String, Long>> logicalDrives = maps.getPartitionToLogicalDriveMap().get(deviceID);
126             if (logicalDrives == null) {
127                 continue;
128             }
129             for (int j = 0; j < logicalDrives.size(); j++) {
130                 Pair<String, Long> logicalDrive = logicalDrives.get(j);
131                 if (logicalDrive != null && !logicalDrive.getA().isEmpty()) {
132                     String uuid;
133                     String label = "";
134                     try (Arena arena = Arena.ofConfined()) {
135                         MemorySegment mountPoint = WindowsForeignFunctions.toWideString(arena, logicalDrive.getA());
136                         MemorySegment volumeBuf = arena.allocate(JAVA_CHAR, GUID_BUFSIZE);
137                         Kernel32FFM.GetVolumeNameForVolumeMountPoint(mountPoint, volumeBuf, GUID_BUFSIZE);
138                         uuid = ParseUtil.parseUuidOrDefault(WindowsForeignFunctions.readWideString(volumeBuf), "");
139 
140                         MemorySegment labelBuf = arena.allocate(JAVA_CHAR, LABEL_BUFSIZE);
141                         OptionalInt volInfoResult = Kernel32FFM.GetVolumeInformation(mountPoint, labelBuf,
142                                 LABEL_BUFSIZE, MemorySegment.NULL, MemorySegment.NULL, MemorySegment.NULL,
143                                 MemorySegment.NULL, 0);
144                         if (volInfoResult.isPresent() && volInfoResult.getAsInt() != 0) {
145                             label = WindowsForeignFunctions.readWideString(labelBuf);
146                         } else {
147                             OptionalInt error = Kernel32FFM.GetLastError();
148                             LOG.debug("Failed to get volume label for {}: error code {}", logicalDrive.getA(),
149                                     error.orElse(-1));
150                         }
151                     }
152                     HWPartition pt = new HWPartition(
153                             WmiUtilFFM.getString(hwPartitionQueryMap, DiskPartitionProperty.NAME, i),
154                             WmiUtilFFM.getString(hwPartitionQueryMap, DiskPartitionProperty.TYPE, i),
155                             WmiUtilFFM.getString(hwPartitionQueryMap, DiskPartitionProperty.DESCRIPTION, i), uuid,
156                             label, logicalDrive.getB(),
157                             WmiUtilFFM.getUint32(hwPartitionQueryMap, DiskPartitionProperty.DISKINDEX, i),
158                             WmiUtilFFM.getUint32(hwPartitionQueryMap, DiskPartitionProperty.INDEX, i),
159                             logicalDrive.getA());
160                     maps.getPartitionMap().computeIfAbsent(deviceID, x -> new ArrayList<>()).add(pt);
161                 }
162             }
163         }
164         return maps;
165     }
