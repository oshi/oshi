39  /**
40   * A Power Source
41   *
42   * @author widdis[at]gmail[dot]com
43   */
44  public class MacPowerSource extends AbstractPowerSource {
45  
46      private static final long serialVersionUID = 1L;
47  
48      private static final Logger LOG = LoggerFactory.getLogger(MacPowerSource.class);
49  
50      public MacPowerSource(String newName, double newRemainingCapacity, double newTimeRemaining) {
51          super(newName, newRemainingCapacity, newTimeRemaining);
52          LOG.debug("Initialized MacPowerSource");
53      }
54  
55      /**
56       * Gets Battery Information
57       *
58       * @return An array of PowerSource objects representing batteries, etc.
59       */
60      public static PowerSource[] getPowerSources() {
61          // Get the blob containing current power source state
62          CFTypeRef powerSourcesInfo = IOKit.INSTANCE.IOPSCopyPowerSourcesInfo();
63          CFArrayRef powerSourcesList = IOKit.INSTANCE.IOPSCopyPowerSourcesList(powerSourcesInfo);
64          int powerSourcesCount = CoreFoundation.INSTANCE.CFArrayGetCount(powerSourcesList);
65  
66          // Get time remaining
67          // -1 = unknown, -2 = unlimited
68          double timeRemaining = IOKit.INSTANCE.IOPSGetTimeRemainingEstimate();
69  
70          // For each power source, output various info
71          List<MacPowerSource> psList = new ArrayList<>(powerSourcesCount);
72          for (int ps = 0; ps < powerSourcesCount; ps++) {
73              // Get the dictionary for that Power Source
74              CFTypeRef powerSource = CoreFoundation.INSTANCE.CFArrayGetValueAtIndex(powerSourcesList, ps);
75              CFDictionaryRef dictionary = IOKit.INSTANCE.IOPSGetPowerSourceDescription(powerSourcesInfo, powerSource);
76  
77              // Get values from dictionary (See IOPSKeys.h)
78              // Skip if not present
79              boolean isPresent = false;
80              Pointer isPresentRef = CoreFoundation.INSTANCE.CFDictionaryGetValue(dictionary, IOKit.IOPS_IS_PRESENT_KEY);
81              if (isPresentRef != null) {
82                  isPresent = CoreFoundation.INSTANCE.CFBooleanGetValue(isPresentRef);
83              }
84              if (!isPresent) {
85                  continue;
86              }
87  
88              // Get name
89              String name = CfUtil
90                      .cfPointerToString(CoreFoundation.INSTANCE.CFDictionaryGetValue(dictionary, IOKit.IOPS_NAME_KEY));
91  
92              // Remaining Capacity = current / max
93              IntByReference currentCapacity = new IntByReference();
94              if (!CoreFoundation.INSTANCE.CFDictionaryGetValueIfPresent(dictionary, IOKit.IOPS_CURRENT_CAPACITY_KEY,
95                      currentCapacity)) {
96                  currentCapacity = new IntByReference(0);
97              }
98              IntByReference maxCapacity = new IntByReference();
99              if (!CoreFoundation.INSTANCE.CFDictionaryGetValueIfPresent(dictionary, IOKit.IOPS_MAX_CAPACITY_KEY,
100                     maxCapacity)) {
101                 maxCapacity = new IntByReference(1);
102             }
103 
104             // Add to list
105             psList.add(new MacPowerSource(name, (double) currentCapacity.getValue() / maxCapacity.getValue(),
106                     timeRemaining));
107         }
108         // Release the blob
109         CoreFoundation.INSTANCE.CFRelease(powerSourcesInfo);
110 
111         return psList.toArray(new MacPowerSource[psList.size()]);
112     }
113 }
