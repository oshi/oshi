23  /**
24   * MacNetworks JNA implementation.
25   */
26  @ThreadSafe
27  public final class MacNetworkIfJNA extends MacNetworkIF {
28  
29      public MacNetworkIfJNA(NetworkInterface netint, Map<Integer, IFdata> data) throws InstantiationException {
30          super(netint, queryIfDisplayName(netint));
31          updateNetworkStats(data);
32      }
33  
34      private static String queryIfDisplayName(NetworkInterface netint) {
35          String name = netint.getName();
36          CFArrayRef ifArray = SystemConfiguration.INSTANCE.SCNetworkInterfaceCopyAll();
37          if (ifArray != null) {
38              try {
39                  int count = ifArray.getCount();
40                  for (int i = 0; i < count; i++) {
41                      Pointer pNetIf = ifArray.getValueAtIndex(i);
42                      SCNetworkInterfaceRef scNetIf = new SCNetworkInterfaceRef(pNetIf);
43                      CFStringRef cfName = SystemConfiguration.INSTANCE.SCNetworkInterfaceGetBSDName(scNetIf);
44                      if (cfName != null && name.equals(cfName.stringValue())) {
45                          CFStringRef cfDisplayName = SystemConfiguration.INSTANCE
46                                  .SCNetworkInterfaceGetLocalizedDisplayName(scNetIf);
47                          if (cfDisplayName != null) {
48                              return cfDisplayName.stringValue();
49                          }
50                      }
51                  }
52              } finally {
53                  ifArray.release();
54              }
55          }
56          return name;
57      }
58  
59      /**
60       * Gets all network interfaces on this machine
61       *
62       * @param includeLocalInterfaces include local interfaces in the result
63       * @return A list of {@link NetworkIF} objects representing the interfaces
64       */
65      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
66          Map<Integer, IFdata> data = NetStat.queryIFdata(-1);
67          return getNetworks(includeLocalInterfaces, ni -> new MacNetworkIfJNA(ni, data));
68      }
69  
70      @Override
71      protected Map<Integer, IFdata> queryIFdata(int index) {
72          return NetStat.queryIFdata(index);
73      }
74  }
