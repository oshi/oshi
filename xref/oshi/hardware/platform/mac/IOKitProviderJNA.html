5   package oshi.hardware.platform.mac;
6   
7   import java.util.function.Consumer;
8   import java.util.function.Function;
9   
10  import com.sun.jna.platform.mac.IOKit.IOIterator;
11  import com.sun.jna.platform.mac.IOKit.IORegistryEntry;
12  import com.sun.jna.platform.mac.IOKitUtil;
13  
14  import oshi.hardware.common.platform.mac.IOKitProvider;
15  
16  /**
17   * JNA-based {@link IOKitProvider} implementation.
18   */
19  final class IOKitProviderJNA implements IOKitProvider {
20  
21      static final IOKitProviderJNA INSTANCE = new IOKitProviderJNA();
22  
23      private IOKitProviderJNA() {
24      }
25  
26      @Override
27      public <T> T withMatchingService(String serviceName, Function<RegistryEntry, T> extractor) {
28          IORegistryEntry entry = IOKitUtil.getMatchingService(serviceName);
29          if (entry != null) {
30              try {
31                  return extractor.apply(new JnaRegistryEntry(entry));
32              } finally {
33                  entry.release();
34              }
35          }
36          return null;
37      }
38  
39      @Override
40      public void forEachMatchingService(String serviceName, Consumer<RegistryEntry> consumer) {
41          forEachMatchingServiceUntil(serviceName, entry -> {
42              consumer.accept(entry);
43              return false;
44          });
45      }
46  
47      @Override
48      public void forEachMatchingServiceUntil(String serviceName, Function<RegistryEntry, Boolean> visitor) {
49          IOIterator iter = IOKitUtil.getMatchingServices(serviceName);
50          if (iter != null) {
51              try {
52                  IORegistryEntry entry = iter.next();
53                  while (entry != null) {
54                      try {
55                          if (Boolean.TRUE.equals(visitor.apply(new JnaRegistryEntry(entry)))) {
56                              return;
57                          }
58                      } finally {
59                          entry.release();
60                      }
61                      entry = iter.next();
62                  }
63              } finally {
64                  iter.release();
65              }
66          }
67      }
68  
69      private static final class JnaRegistryEntry implements RegistryEntry {
70          private final IORegistryEntry entry;
71  
72          JnaRegistryEntry(IORegistryEntry entry) {
73              this.entry = entry;
74          }
75  
76          @Override
77          public String getName() {
78              return entry.getName();
79          }
80  
81          @Override
82          public byte[] getByteArrayProperty(String key) {
83              return entry.getByteArrayProperty(key);
84          }
85  
86          @Override
87          public String getStringProperty(String key) {
88              return entry.getStringProperty(key);
89          }
90      }
91  }
