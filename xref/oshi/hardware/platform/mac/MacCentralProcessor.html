62       * Create a Processor

64      public MacCentralProcessor() {
65          // Initialize class variables
66          initVars();
67          // Initialize tick arrays
68          initTicks();
69          // Set max processes
70          this.maxProc = SysctlUtil.sysctl("kern.maxproc", 0x1000);

72          LOG.debug("Initialized Processor");

75      private void initVars() {
76          setVendor(SysctlUtil.sysctl("machdep.cpu.vendor", ""));
77          setName(SysctlUtil.sysctl("machdep.cpu.brand_string", ""));
78          setCpu64(SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0);
79          int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
80          setStepping(i < 0 ? "" : Integer.toString(i));
81          i = SysctlUtil.sysctl("machdep.cpu.model", -1);
82          setModel(i < 0 ? "" : Integer.toString(i));
83          i = SysctlUtil.sysctl("machdep.cpu.family", -1);
84          setFamily(i < 0 ? "" : Integer.toString(i));
85      }

95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public long[] getSystemCpuLoadTicks() {
100         long[] ticks = new long[curTicks.length];
101         int machPort = SystemB.INSTANCE.mach_host_self();
102         HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
103         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
104                 new IntByReference(cpuLoadInfo.size()))) {
105             LOG.error("Failed to get System CPU ticks. Error code: " + Native.getLastError());
106             return ticks;
107         }
108         // Switch order to match linux
109         ticks[0] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
110         ticks[1] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
111         ticks[2] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
112         ticks[3] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
113         return ticks;
114     }

134     /**
135      * {@inheritDoc}
136      */
137     @Override
138     public double[] getSystemLoadAverage(int nelem) {
139         if (nelem < 1) {
140             throw new IllegalArgumentException("Must include at least one element.");
141         }
142         if (nelem > 3) {
143             LOG.warn("Max elements of SystemLoadAverage is 3. " + nelem + " specified. Ignoring extra.");
144             nelem = 3;
145         }
146         double[] average = new double[nelem];
147         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
148         if (retval < nelem) {
149             for (int i = Math.max(retval, 0); i < average.length; i++) {
150                 average[i] = -1d;
151             }
152         }
153         return average;
154     }

156     /**
157      * {@inheritDoc}
158      */
159     @Override
160     public long[][] getProcessorCpuLoadTicks() {
161         long[][] ticks = new long[logicalProcessorCount][4];
162 
163         int machPort = SystemB.INSTANCE.mach_host_self();
164 
165         IntByReference procCount = new IntByReference();
166         PointerByReference procCpuLoadInfo = new PointerByReference();
167         IntByReference procInfoCount = new IntByReference();
168         if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
169                 procCpuLoadInfo, procInfoCount)) {
170             LOG.error("Failed to update CPU Load. Error code: " + Native.getLastError());
171             return ticks;
172         }
173 
174         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
175         for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
176             for (int j = 0; j < 4; j++) {
177                 int offset = cpu * SystemB.CPU_STATE_MAX;
178                 ticks[cpu][0] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
179                 ticks[cpu][1] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
180                 ticks[cpu][2] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
181                 ticks[cpu][3] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
182             }
183         }
184         return ticks;
185     }

187     /**
188      * {@inheritDoc}
189      */
190     @Override
191     public long getSystemUptime() {
192         Timeval tv = new Timeval();
193         if (!SysctlUtil.sysctl("kern.boottime", tv)) {
194             return 0L;
195         }
196         // tv now points to a 16-bit timeval structure for boot time.
197         // First 8 bytes are seconds, second 8 bytes are microseconds (ignore)
198         return System.currentTimeMillis() / 1000 - tv.tv_sec;
199     }

205     public String getSystemSerialNumber() {
206         if (this.cpuSerialNumber == null) {
207             int service = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
208             if (service != 0) {
209                 // Fetch the serial number
210                 this.cpuSerialNumber = IOKitUtil.getIORegistryStringProperty(service, "IOPlatformSerialNumber");
211                 IOKit.INSTANCE.IOObjectRelease(service);
212             }
213             if (this.cpuSerialNumber == null) {
214                 this.cpuSerialNumber = "unknown";
215             }
216         }
217         return this.cpuSerialNumber;
218     }

220     /**
221      * {@inheritDoc}
222      */
223     @Override
224     public OSProcess[] getProcesses() {
225         List<OSProcess> procs = new ArrayList<>();
226         int[] pids = new int[this.maxProc];
227         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids, pids.length)
228                 / SystemB.INT_SIZE;
229         for (int i = 0; i < numberOfProcesses; i++) {
230             OSProcess proc = getProcess(pids[i]);
231             if (proc != null) {
232                 procs.add(proc);
233             }
234         }
235         return procs.toArray(new OSProcess[procs.size()]);
236     }

238     /**
239      * {@inheritDoc}
240      */
241     @Override
242     public OSProcess getProcess(int pid) {
243         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
244         if (0 > SystemB.INSTANCE.proc_pidinfo(pid, SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo, taskAllInfo.size())) {
245             return null;
246         }
247         String name = null;
248         String path = "";
249         Pointer buf = new Memory(SystemB.PROC_PIDPATHINFO_MAXSIZE);
250         if (0 < SystemB.INSTANCE.proc_pidpath(pid, buf, SystemB.PROC_PIDPATHINFO_MAXSIZE)) {
251             path = buf.getString(0).trim();
252             // Overwrite name with last part of path
253             String[] pathSplit = path.split("/");
254             if (pathSplit.length > 0) {
255                 name = pathSplit[pathSplit.length - 1];
256             }
257         }
258         // If process is gone, return null
259         if (taskAllInfo.ptinfo.pti_threadnum < 1) {
260             return null;
261         }
262         if (name == null) {
263             // pbi_comm contains first 16 characters of name
264             // null terminated
265             for (int t = 0; t < taskAllInfo.pbsd.pbi_comm.length; t++) {
266                 if (taskAllInfo.pbsd.pbi_comm[t] == 0) {
267                     name = new String(taskAllInfo.pbsd.pbi_comm, 0, t);
268                     break;
269                 }
270             }
271         }
272         return new MacProcess(name, path, taskAllInfo.pbsd.pbi_status, pid, taskAllInfo.pbsd.pbi_ppid,
273                 taskAllInfo.ptinfo.pti_threadnum, taskAllInfo.ptinfo.pti_priority, taskAllInfo.ptinfo.pti_virtual_size,
274                 taskAllInfo.ptinfo.pti_resident_size, taskAllInfo.ptinfo.pti_total_system / 1000000L,
275                 taskAllInfo.ptinfo.pti_total_user / 1000000L,
276                 taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L,
277                 System.currentTimeMillis());
278     }

296     /**
297      * {@inheritDoc}
298      */
299     @Override
300     public int getThreadCount() {
301         // Get current pids, then slightly pad in case new process starts while
302         // allocating array space
303         int[] pids = new int[getProcessCount() + 10];
304         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids, pids.length)
305                 / SystemB.INT_SIZE;
306         int numberOfThreads = 0;
307         ProcTaskInfo taskInfo = new ProcTaskInfo();
308         for (int i = 0; i < numberOfProcesses; i++) {
309             SystemB.INSTANCE.proc_pidinfo(pids[i], SystemB.PROC_PIDTASKINFO, 0, taskInfo, taskInfo.size());
310             numberOfThreads += taskInfo.pti_threadnum;
311         }
312         return numberOfThreads;
313     }
