70      private static final int ROSETTA_CPUTYPE = 0x00000007;
71      private static final int ROSETTA_CPUFAMILY = 0x573b5eec;
72      private static final int M1_CPUTYPE = 0x0100000C;
73      private static final int M1_CPUFAMILY = 0x1b588bb3;

75      @Override
76      protected ProcessorIdentifier queryProcessorId() {
77          String cpuName = SysctlUtil.sysctl("machdep.cpu.brand_string", "");
78          String cpuVendor;
79          String cpuStepping;
80          String cpuModel;
81          String cpuFamily;
82          String processorID;
83          long cpuFreq = 0L;
84          if (cpuName.startsWith("Apple")) {
85              // Processing an M1 chip
86              cpuVendor = vendor.get();
87              cpuStepping = "0"; // No correlation yet
88              cpuModel = "0"; // No correlation yet
89              int type = SysctlUtil.sysctl("hw.cputype", 0);
90              int family = SysctlUtil.sysctl("hw.cpufamily", 0);
91              // M1 should have hw.cputype 0x0100000C (ARM64) and hw.cpufamily 0x1b588bb3 for
92              // an ARM SoC. However, under Rosetta 2, low level cpuid calls in the translated
93              // environment report hw.cputype for x86 (0x00000007) and hw.cpufamily for an
94              // Intel Westmere chip (0x573b5eec), family 6, model 44, stepping 0.
95              // Test if under Rosetta and generate correct chip
96              Quartet<Integer, Integer, Long, Map<Integer, String>> armCpu = queryArmCpu();
97              if (family == ROSETTA_CPUFAMILY) {
98                  type = armCpu.getA();
99                  family = armCpu.getB();
100             }
101             cpuFreq = armCpu.getC();
102             // Translate to output
103             cpuFamily = String.format("0x%08x", family);
104             // Processor ID is an intel concept but CPU type + family conveys same info
105             processorID = String.format("%08x%08x", type, family);
106         } else {
107             // Processing an Intel chip
108             cpuVendor = SysctlUtil.sysctl("machdep.cpu.vendor", "");
109             int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
110             cpuStepping = i < 0 ? "" : Integer.toString(i);
111             i = SysctlUtil.sysctl("machdep.cpu.model", -1);
112             cpuModel = i < 0 ? "" : Integer.toString(i);
113             i = SysctlUtil.sysctl("machdep.cpu.family", -1);
114             cpuFamily = i < 0 ? "" : Integer.toString(i);
115             long processorIdBits = 0L;
116             processorIdBits |= SysctlUtil.sysctl("machdep.cpu.signature", 0);
117             processorIdBits |= (SysctlUtil.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
118             processorID = String.format("%016x", processorIdBits);
119         }
120         if (cpuFreq == 0) {
121             cpuFreq = SysctlUtil.sysctl("hw.cpufrequency", 0L);
122         }
123         boolean cpu64bit = SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0;
124 
125         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
126                 cpuFreq);
127     }

129     @Override
130     protected Pair<List<LogicalProcessor>, List<PhysicalProcessor>> initProcessorCounts() {
131         int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
132         int physicalProcessorCount = SysctlUtil.sysctl("hw.physicalcpu", 1);
133         int physicalPackageCount = SysctlUtil.sysctl("hw.packages", 1);
134         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
135         Set<Integer> pkgCoreKeys = new HashSet<>();
136         for (int i = 0; i < logicalProcessorCount; i++) {
137             int coreId = i * physicalProcessorCount / logicalProcessorCount;
138             int pkgId = i * physicalPackageCount / logicalProcessorCount;
139             logProcs.add(new LogicalProcessor(i, coreId, pkgId));
140             pkgCoreKeys.add((pkgId << 16) + coreId);
141         }
142         Map<Integer, String> compatMap = queryArmCpu().getD();
143         List<PhysicalProcessor> physProcs = pkgCoreKeys.stream().sorted().map(k -> {
144             String compat = compatMap.getOrDefault(k, "");
145             int efficiency = 0; // default, for E-core icestorm
146             if (compat.toLowerCase().contains("firestorm")) {
147                 efficiency = 1; // P-core, more performance
148             }
149             return new PhysicalProcessor(k >> 16, k & 0xffff, efficiency, compat);
150         }).collect(Collectors.toList());
151         return new Pair<>(logProcs, physProcs);
152     }

154     @Override
155     public long[] querySystemCpuLoadTicks() {
156         long[] ticks = new long[TickType.values().length];
157         int machPort = SystemB.INSTANCE.mach_host_self();
158         try (CloseableHostCpuLoadInfo cpuLoadInfo = new CloseableHostCpuLoadInfo();
159                 CloseableIntByReference size = new CloseableIntByReference(cpuLoadInfo.size())) {
160             if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo, size)) {
161                 LOG.error("Failed to get System CPU ticks. Error code: {} ", Native.getLastError());
162                 return ticks;
163             }
164 
165             ticks[TickType.USER.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
166             ticks[TickType.NICE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
167             ticks[TickType.SYSTEM.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
168             ticks[TickType.IDLE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
169         }
170         // Leave IOWait and IRQ values as 0
171         return ticks;
172     }

186     @Override
187     public double[] getSystemLoadAverage(int nelem) {
188         if (nelem < 1 || nelem > 3) {
189             throw new IllegalArgumentException("Must include from one to three elements.");
190         }
191         double[] average = new double[nelem];
192         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
193         if (retval < nelem) {
194             Arrays.fill(average, -1d);
195         }
196         return average;
197     }

199     @Override
200     public long[][] queryProcessorCpuLoadTicks() {
201         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
202 
203         int machPort = SystemB.INSTANCE.mach_host_self();
204         try (CloseableIntByReference procCount = new CloseableIntByReference();
205                 CloseablePointerByReference procCpuLoadInfo = new CloseablePointerByReference();
206                 CloseableIntByReference procInfoCount = new CloseableIntByReference()) {
207             if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
208                     procCpuLoadInfo, procInfoCount)) {
209                 LOG.error("Failed to update CPU Load. Error code: {}", Native.getLastError());
210                 return ticks;
211             }
212 
213             int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
214             for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
215                 int offset = cpu * SystemB.CPU_STATE_MAX;
216                 ticks[cpu][TickType.USER.getIndex()] = FormatUtil
217                         .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
218                 ticks[cpu][TickType.NICE.getIndex()] = FormatUtil
219                         .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
220                 ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil
221                         .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
222                 ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil
223                         .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
224             }
225         }
226         return ticks;
227     }

245     private static String platformExpert() {
246         String manufacturer = null;
247         IORegistryEntry platformExpert = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
248         if (platformExpert != null) {
249             // Get manufacturer from IOPlatformExpertDevice
250             byte[] data = platformExpert.getByteArrayProperty("manufacturer");
251             if (data != null) {
252                 manufacturer = Native.toString(data, StandardCharsets.UTF_8);
253             }
254             platformExpert.release();
255         }
256         return Util.isBlank(manufacturer) ? "Apple Inc." : manufacturer;
257     }

259     private static Quartet<Integer, Integer, Long, Map<Integer, String>> queryArmCpu() {
260         int type = ROSETTA_CPUTYPE;
261         int family = ROSETTA_CPUFAMILY;
262         long freq = 0L;
263         Map<Integer, String> compatibleStrMap = new HashMap<>();
264         // All CPUs are an IOPlatformDevice
265         // Iterate each CPU and save frequency and "compatible" strings
266         IOIterator iter = IOKitUtil.getMatchingServices("IOPlatformDevice");
267         if (iter != null) {
268             Set<String> compatibleStrSet = new HashSet<>();
269             IORegistryEntry cpu = iter.next();
270             while (cpu != null) {
271                 if (cpu.getName().toLowerCase().startsWith("cpu")) {
272                     int procId = ParseUtil.getFirstIntValue(cpu.getName());
273                     // Accurate CPU vendor frequency in kHz as little-endian byte array
274                     byte[] data = cpu.getByteArrayProperty("clock-frequency");
275                     if (data != null) {
276                         long cpuFreq = ParseUtil.byteArrayToLong(data, data.length, false) * 1000L;
277                         if (cpuFreq > freq) {
278                             freq = cpuFreq;
279                         }
280                     }
281                     // Compatible key is null-delimited C string array in byte array
282                     data = cpu.getByteArrayProperty("compatible");
283                     if (data != null) {
284                         for (String s : new String(data, StandardCharsets.UTF_8).split("\0")) {
285                             if (!s.isEmpty()) {
286                                 compatibleStrSet.add(s);
287                                 if (compatibleStrMap.containsKey(procId)) {
288                                     compatibleStrMap.put(procId, compatibleStrMap.get(procId) + " " + s);
289                                 } else {
290                                     compatibleStrMap.put(procId, s);
291                                 }
292                             }
293                         }
294                     }
295                 }
296                 cpu.release();
297                 cpu = iter.next();
298             }
299             iter.release();
300             // Match strings in "compatible" field with expectation for M1 chip
301             // Hard coded for M1 for now. Need to update and make more configurable for M1X,
302             // M2, etc.
303             List<String> m1compatible = Arrays.asList("ARM,v8", "apple,firestorm", "apple,icestorm");
304             compatibleStrSet.retainAll(m1compatible);
305             if (compatibleStrSet.size() == m1compatible.size()) {
306                 type = M1_CPUTYPE;
307                 family = M1_CPUFAMILY;
308             }
309         }
310         return new Quartet<>(type, family, freq, compatibleStrMap);
311     }
