113     /**
114      * Create a Processor
115      */
116     public MacCentralProcessor()
117     {
118         // Processor counts
119         calculateProcessorCounts();
120 
121         // System ticks
122         this.prevTicks = new long[4];
123         this.curTicks = new long[4];
124         updateSystemTicks();
125 
126         // Per-processor ticks
127         this.prevProcTicks = new long[logicalProcessorCount][4];
128         this.curProcTicks = new long[logicalProcessorCount][4];
129         updateProcessorTicks();
130 
131         LOG.debug( "Initialized Processor" );
132     }

134     /**
135      * Updates logical and physical processor counts from sysctl calls
136      */
137     private void calculateProcessorCounts()
138     {
139         IntByReference size = new IntByReference( SystemB.INT_SIZE );
140         Pointer p = new Memory( size.getValue() );
141 
142         // Get number of logical processors
143         if ( 0 != SystemB.INSTANCE.sysctlbyname( "hw.logicalcpu", p, size, null, 0 ) )
144         {
145             LOG.error( "Failed to get number of logical CPUs. Error code: " + Native.getLastError() );
146             this.logicalProcessorCount = 1;
147         }
148         else
149             this.logicalProcessorCount = p.getInt( 0 );
150 
151         // Get number of physical processors
152         if ( 0 != SystemB.INSTANCE.sysctlbyname( "hw.physicalcpu", p, size, null, 0 ) )
153         {
154             LOG.error( "Failed to get number of physical CPUs. Error code: " + Native.getLastError() );
155             this.physicalProcessorCount = 1;
156         }
157         else
158             this.physicalProcessorCount = p.getInt( 0 );
159     }

232     @Override
233     public long getVendorFreq()
234     {
235         if ( this.cpuVendorFreq == null )
236         {
237             Pattern pattern = Pattern.compile( "@ (.*)$" );
238             Matcher matcher = pattern.matcher( getName() );
239 
240             if ( matcher.find() )
241             {
242                 String unit = matcher.group( 1 );
243                 this.cpuVendorFreq = Long.valueOf( ParseUtil.parseHertz( unit ) );
244             }
245             else
246             {
247                 this.cpuVendorFreq = Long.valueOf( -1L );
248             }
249         }
250 
251         return this.cpuVendorFreq.longValue();
252     }

413     /**
414      * {@inheritDoc}
415      */
416     @Override
417     public synchronized double getSystemCpuLoadBetweenTicks()
418     {
419         // Check if > ~ 0.95 seconds since last tick count.
420         long now = System.currentTimeMillis();
421         LOG.trace( "Current time: {}  Last tick time: {}", now, tickTime );
422         if ( now - tickTime > 950 )
423         {
424             // Enough time has elapsed.
425             updateSystemTicks();
426         }
427         // Calculate total
428         long total = 0;
429         for ( int i = 0; i < curTicks.length; i++ )
430         {
431             total += ( curTicks[i] - prevTicks[i] );
432         }
433         // Calculate idle from last field [3]
434         long idle = curTicks[3] - prevTicks[3];
435         LOG.trace( "Total ticks: {}  Idle ticks: {}", total, idle );
436 
437         return ( total > 0 && idle >= 0 ) ? (double) ( total - idle ) / total : 0d;
438     }

440     /**
441      * {@inheritDoc}
442      */
443     @Override
444     public long[] getSystemCpuLoadTicks()
445     {
446         long[] ticks = new long[curTicks.length];
447         int machPort = SystemB.INSTANCE.mach_host_self();
448         HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
449         if ( 0 != SystemB.INSTANCE.host_statistics( machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
450                                                     new IntByReference( cpuLoadInfo.size() ) ) )
451         {
452             LOG.error( "Failed to get System CPU ticks. Error code: " + Native.getLastError() );
453             return ticks;
454         }
455         // Switch order to match linux
456         ticks[0] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
457         ticks[1] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
458         ticks[2] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
459         ticks[3] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
460         return ticks;
461     }

463     /**
464      * Updates system tick information from native host_statistics query. Stores in array with four elements
465      * representing clock ticks or milliseconds (platform dependent) spent in User (0), Nice (1), System (2), and Idle
466      * (3) states. By measuring the difference between ticks across a time interval, CPU load over that interval may be
467      * calculated.
468      */
469     private void updateSystemTicks()
470     {
471         LOG.trace( "Updating System Ticks" );
472         // Copy to previous
473         System.arraycopy( curTicks, 0, prevTicks, 0, curTicks.length );
474         this.tickTime = System.currentTimeMillis();
475         long[] ticks = getSystemCpuLoadTicks();
476         System.arraycopy( ticks, 0, curTicks, 0, ticks.length );
477     }

501     /**
502      * {@inheritDoc}
503      */
504     @Override
505     public double[] getProcessorCpuLoadBetweenTicks()
506     {
507         // Check if > ~ 0.95 seconds since last tick count.
508         long now = System.currentTimeMillis();
509         LOG.trace( "Current time: {}  Last tick time: {}", now, procTickTime );
510         if ( now - procTickTime > 950 )
511         {
512             // Enough time has elapsed.
513             // Update latest
514             updateProcessorTicks();
515         }
516         double[] load = new double[logicalProcessorCount];
517         for ( int cpu = 0; cpu < logicalProcessorCount; cpu++ )
518         {
519             long total = 0;
520             for ( int i = 0; i < this.curProcTicks[cpu].length; i++ )
521             {
522                 total += ( this.curProcTicks[cpu][i] - this.prevProcTicks[cpu][i] );
523             }
524             // Calculate idle from last field [3]
525             long idle = this.curProcTicks[cpu][3] - this.prevProcTicks[cpu][3];
526             LOG.trace( "CPU: {}  Total ticks: {}  Idle ticks: {}", cpu, total, idle );
527             // update
528             load[cpu] = ( total > 0 && idle >= 0 ) ? (double) ( total - idle ) / total : 0d;
529         }
530         return load;
531     }

533     /**
534      * {@inheritDoc}
535      */
536     @Override
537     public long[][] getProcessorCpuLoadTicks()
538     {
539         long[][] ticks = new long[logicalProcessorCount][4];
540 
541         int machPort = SystemB.INSTANCE.mach_host_self();
542 
543         IntByReference procCount = new IntByReference();
544         PointerByReference procCpuLoadInfo = new PointerByReference();
545         IntByReference procInfoCount = new IntByReference();
546         if ( 0 != SystemB.INSTANCE.host_processor_info( machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
547                                                         procCpuLoadInfo, procInfoCount ) )
548         {
549             LOG.error( "Failed to update CPU Load. Error code: " + Native.getLastError() );
550             return ticks;
551         }
552 
553         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray( 0, procInfoCount.getValue() );
554         for ( int cpu = 0; cpu < procCount.getValue(); cpu++ )
555         {
556             for ( int j = 0; j < 4; j++ )
557             {
558                 int offset = cpu * SystemB.CPU_STATE_MAX;
559                 ticks[cpu][0] = FormatUtil.getUnsignedInt( cpuTicks[offset + SystemB.CPU_STATE_USER] );
560                 ticks[cpu][1] = FormatUtil.getUnsignedInt( cpuTicks[offset + SystemB.CPU_STATE_NICE] );
561                 ticks[cpu][2] = FormatUtil.getUnsignedInt( cpuTicks[offset + SystemB.CPU_STATE_SYSTEM] );
562                 ticks[cpu][3] = FormatUtil.getUnsignedInt( cpuTicks[offset + SystemB.CPU_STATE_IDLE] );
563             }
564         }
565         return ticks;
566     }

568     /**
569      * Updates the tick array for all processors by calling host_processor_info(). Stores in 2D array; an array for each
570      * logical processor with four elements representing clock ticks or milliseconds (platform dependent) spent in User
571      * (0), Nice (1), System (2), and Idle (3) states. By measuring the difference between ticks across a time interval,
572      * CPU load over that interval may be calculated.
573      */
574     private void updateProcessorTicks()
575     {
576         LOG.trace( "Updating Processor Ticks" );
577         // Copy to previous
578         for ( int cpu = 0; cpu < logicalProcessorCount; cpu++ )
579         {
580             System.arraycopy( curProcTicks[cpu], 0, prevProcTicks[cpu], 0, curProcTicks[cpu].length );
581         }
582         this.procTickTime = System.currentTimeMillis();
583         long[][] ticks = getProcessorCpuLoadTicks();
584         for ( int cpu = 0; cpu < logicalProcessorCount; cpu++ )
585         {
586             System.arraycopy( ticks[cpu], 0, curProcTicks[cpu], 0, ticks[cpu].length );
587         }
588     }

590     /**
591      * {@inheritDoc}
592      */
593     @Override
594     public long getSystemUptime()
595     {
596         IntByReference size = new IntByReference();
597         if ( 0 != SystemB.INSTANCE.sysctlbyname( "kern.boottime", null, size, null, 0 ) )
598         {
599             LOG.error( "Failed to get Boot Time. Error code: " + Native.getLastError() );
600             return 0L;
601         }
602         // This should point to a 16-byte structure. If not, this code is valid
603         if ( size.getValue() != 16 )
604             throw new UnsupportedOperationException( "sysctl kern.boottime should be 16 bytes but isn't." );
605         Pointer p = new Memory( size.getValue() + 1 );
606         if ( 0 != SystemB.INSTANCE.sysctlbyname( "kern.boottime", p, size, null, 0 ) )
607         {
608             LOG.error( "Failed to get Boot Time. Error code: " + Native.getLastError() );
609             return 0L;
610         }
611         // p now points to a 16-bit timeval structure for boot time.
612         // First 8 bytes are seconds, second 8 bytes are microseconds (ignore)
613         return System.currentTimeMillis() / 1000 - p.getLong( 0 );
614     }

616     /**
617      * {@inheritDoc}
618      */
619     @Override
620     public String getSystemSerialNumber()
621     {
622         String sn = null;
623         ArrayList<String> hwInfo = ExecutingCommand.runNative( "system_profiler SPHardwareDataType" );
624         // Mavericks and later
625         for ( String checkLine : hwInfo )
626         {
627             if ( checkLine.contains( "Serial Number (system)" ) )
628             {
629                 String[] snSplit = checkLine.split( "\\s+" );
630                 sn = snSplit[snSplit.length - 1];
631                 break;
632             }
633         }
634         // Panther and later
635         if ( sn == null )
636         {
637             for ( String checkLine : hwInfo )
638             {
639                 if ( checkLine.contains( "r (system)" ) )
640                 {
641                     String[] snSplit = checkLine.split( "\\s+" );
642                     sn = snSplit[snSplit.length - 1];
643                     break;
644                 }
645             }
646         }
647         return ( sn == null ) ? "unknown" : sn;
648     }
