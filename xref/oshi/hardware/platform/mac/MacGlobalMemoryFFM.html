5   package oshi.hardware.platform.mac;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.ffm.mac.MacSystemFunctions;
11  import oshi.hardware.VirtualMemory;
12  import oshi.util.platform.mac.SysctlUtilFFM;
13  
14  import java.lang.foreign.Arena;
15  import java.lang.foreign.MemorySegment;
16  
17  import static java.lang.foreign.ValueLayout.JAVA_INT;
18  import static java.lang.foreign.ValueLayout.JAVA_LONG;
19  import static oshi.ffm.mac.MacSystem.VM_FREE_COUNT;
20  import static oshi.ffm.mac.MacSystem.VM_INACTIVE_COUNT;
21  import static oshi.ffm.mac.MacSystem.VM_STATISTICS;
22  import static oshi.ffm.mac.MacSystemFunctions.mach_host_self;
23  
24  @ThreadSafe
25  final class MacGlobalMemoryFFM extends MacGlobalMemory {
26  
27      private static final Logger LOG = LoggerFactory.getLogger(MacGlobalMemoryFFM.class);
28  
29      @Override
30      protected long queryVmStats() {
31          try (Arena arena = Arena.ofConfined()) {
32              // Allocate memory for VM statistics structure and count
33              MemorySegment vmStats = arena.allocate(VM_STATISTICS);
34              if (MacMemoryUtilFFM.callVmStat(arena, vmStats)) {
35                  // Read free_count and inactive_count from the structure
36                  int freeCount = vmStats.get(JAVA_INT, VM_STATISTICS.byteOffset(VM_FREE_COUNT));
37                  int inactiveCount = vmStats.get(JAVA_INT, VM_STATISTICS.byteOffset(VM_INACTIVE_COUNT));
38  
39                  return (freeCount + inactiveCount) * getPageSize();
40              }
41          } catch (Throwable e) {
42              // Ignore
43          }
44          return 0L;
45      }
46  
47      @Override
48      protected long sysctl(String name, long defaultValue) {
49          return SysctlUtilFFM.sysctl(name, defaultValue);
50      }
51  
52      @Override
53      protected long host_page_size() {
54          try (Arena arena = Arena.ofConfined()) {
55              MemorySegment pageSize = arena.allocate(JAVA_LONG);
56              int result = MacSystemFunctions.host_page_size(mach_host_self(), pageSize);
57              if (result == 0) {
58                  return pageSize.get(JAVA_LONG, 0);
59              }
60          } catch (Throwable t) {
61              // IGNORED
62          }
63          return -1;
64      }
65  
66      @Override
67      protected VirtualMemory createVirtualMemory() {
68          return new MacVirtualMemoryFFM(this);
69      }
70  }
