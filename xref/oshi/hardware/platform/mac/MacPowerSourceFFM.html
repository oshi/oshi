51      /**
52       * Gets Battery Information.
53       *
54       * @return An array of PowerSource objects representing batteries, etc.
55       */
56      public static List<PowerSource> getPowerSources() {
57          CFStringRef nameKey = CFStringRef.createCFString("Name");
58          CFStringRef isPresentKey = CFStringRef.createCFString("Is Present");
59          CFStringRef currentCapacityKey = CFStringRef.createCFString("Current Capacity");
60          CFStringRef maxCapacityKey = CFStringRef.createCFString("Max Capacity");
61          try (nameKey; isPresentKey; currentCapacityKey; maxCapacityKey) {
62              MemorySegment powerSourcesInfo = IOPSCopyPowerSourcesInfo();
63              // wrapped only to release the native CF object on close
64              try (var _ = new CFTypeRef(powerSourcesInfo)) {
65                  CFArrayRef cfList = new CFArrayRef(IOPSCopyPowerSourcesList(powerSourcesInfo));
66                  try (cfList) {
67                      double psTimeRemainingEstimated = IOPSGetTimeRemainingEstimate();
68                      int powerSourcesCount = cfList.getCount();
69  
70                      List<PowerSourceData> sources = new ArrayList<>(powerSourcesCount);
71                      for (int ps = 0; ps < powerSourcesCount; ps++) {
72                          MemorySegment pwrSrcPtr = cfList.getValueAtIndex(ps);
73                          MemorySegment dictionary = IOPSGetPowerSourceDescription(powerSourcesInfo, pwrSrcPtr);
74                          @SuppressWarnings("resource") // borrowed ref from IOPSGetPowerSourceDescription
75                          CFDictionaryRef dict = new CFDictionaryRef(dictionary);
76  
77                          MemorySegment result = dict.getValue(isPresentKey);
78                          if (!result.equals(MemorySegment.NULL) && CFBooleanRef.booleanValue(result)) {
79                              result = dict.getValue(nameKey);
80                              String psName = CFUtilFFM.cfPointerToString(result);
81  
82                              double currentCapacity = 0d;
83                              if (dict.getValueIfPresent(currentCapacityKey, MemorySegment.NULL)) {
84                                  result = dict.getValue(currentCapacityKey);
85                                  currentCapacity = CFNumberRef.intValue(result);
86                              }
87                              double maxCapacity = 1d;
88                              if (dict.getValueIfPresent(maxCapacityKey, MemorySegment.NULL)) {
89                                  result = dict.getValue(maxCapacityKey);
90                                  maxCapacity = CFNumberRef.intValue(result);
91                              }
92                              sources.add(new PowerSourceData(psName, currentCapacity, maxCapacity));
93                          }
94                      }
95                      return buildPowerSources(IOKitProviderFFM.INSTANCE, psTimeRemainingEstimated, sources,
96                              MacPowerSourceFFM::new);
97                  }
98              }
99          } catch (Throwable _) {
100             return new ArrayList<>();
101         }
102     }
