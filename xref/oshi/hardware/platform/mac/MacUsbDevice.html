47       * Maps to store information using RegistryEntryID as the key

49      private static Map<Long, String> nameMap = new HashMap<>();
50      private static Map<Long, String> vendorMap = new HashMap<>();
51      private static Map<Long, String> vendorIdMap = new HashMap<>();
52      private static Map<Long, String> productIdMap = new HashMap<>();
53      private static Map<Long, String> serialMap = new HashMap<>();
54      private static Map<Long, List<Long>> hubMap = new HashMap<>();

56      public MacUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
57              UsbDevice[] connectedDevices) {
58          super(name, vendor, vendorId, productId, serialNumber, connectedDevices);
59      }

61      /**
62       * {@inheritDoc}
63       */
64      public static UsbDevice[] getUsbDevices(boolean tree) {
65          UsbDevice[] devices = getUsbDevices();
66          if (tree) {
67              return devices;
68          }
69          List<UsbDevice> deviceList = new ArrayList<>();
70          // Top level is controllers; they won't be added to the list, but all
71          // their connected devices will be
72          for (UsbDevice device : devices) {
73              addDevicesToList(deviceList, device.getConnectedDevices());
74          }
75          return deviceList.toArray(new UsbDevice[deviceList.size()]);
76      }
77  
78      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
79          for (UsbDevice device : connectedDevices) {
80              deviceList.add(new MacUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
81                      device.getProductId(), device.getSerialNumber(), new MacUsbDevice[0]));
82              addDevicesToList(deviceList, device.getConnectedDevices());
83          }
84      }

86      private static UsbDevice[] getUsbDevices() {
87          // Reusable buffer for getting IO name strings
88          Pointer buffer = new Memory(128); // io_name_t is char[128]
89  
90          // Empty out maps
91          nameMap.clear();
92          vendorMap.clear();
93          vendorIdMap.clear();
94          productIdMap.clear();
95          serialMap.clear();
96          hubMap.clear();
97  
98          // Iterate over USB Controllers. All devices are children of one of
99          // these controllers in the "IOService" plane
100         List<Long> usbControllers = new ArrayList<>();
101         IntByReference iter = new IntByReference();
102         IOKitUtil.getMatchingServices("IOUSBController", iter);
103         int device = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
104         while (device != 0) {
105             // Unique global identifier for this device
106             LongByReference id = new LongByReference();
107             IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(device, id);
108             usbControllers.add(id.getValue());
109 
110             // Get device name and store in map
111             IOKit.INSTANCE.IORegistryEntryGetName(device, buffer);
112             nameMap.put(id.getValue(), buffer.getString(0));
113             // The only information we have in registry for this device is the
114             // locationID. Use that to search for matching PCI device to obtain
115             // more information.
116             CFTypeRef ref = IOKit.INSTANCE.IORegistryEntryCreateCFProperty(device, CfUtil.getCFString("locationID"),
117                     CfUtil.ALLOCATOR, 0);
118             if (ref != null && ref.getPointer() != null) {
119                 getControllerIdByLocation(id.getValue(), ref);
120             }
121             CfUtil.release(ref);
122 
123             // Now iterate the children of this device in the "IOService" plane.
124             // If device parent is root, link to the controller
125             IntByReference childIter = new IntByReference();
126             IOKit.INSTANCE.IORegistryEntryGetChildIterator(device, "IOService", childIter);
127             int childDevice = IOKit.INSTANCE.IOIteratorNext(childIter.getValue());
128             while (childDevice != 0) {
129                 // Unique global identifier for this device
130                 LongByReference childId = new LongByReference();
131                 IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(childDevice, childId);
132 
133                 // Get this device's parent in the "IOUSB" plane
134                 IntByReference parent = new IntByReference();
135                 IOKit.INSTANCE.IORegistryEntryGetParentEntry(childDevice, "IOUSB", parent);
136                 // If the parent is not an IOUSBDevice (will be root), set the
137                 // parentId to the controller
138                 LongByReference parentId = new LongByReference();
139                 if (!IOKit.INSTANCE.IOObjectConformsTo(parent.getValue(), "IOUSBDevice")) {
140                     parentId = id;
141                 } else {
142                     // Unique global identifier for the parent
143                     IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(parent.getValue(), parentId);
144                 }
145                 // Store parent in map
146                 MapUtil.createNewListIfAbsent(hubMap, parentId.getValue()).add(childId.getValue());
147 
148                 // Get device name and store in map
149                 IOKit.INSTANCE.IORegistryEntryGetName(childDevice, buffer);
150                 nameMap.put(childId.getValue(), buffer.getString(0));
151                 // Get vendor and store in map
152                 String vendor = IOKitUtil.getIORegistryStringProperty(childDevice, "USB Vendor Name");
153                 if (vendor != null) {
154                     vendorMap.put(childId.getValue(), vendor);
155                 }
156                 // Get vendorId and store in map
157                 long vendorId = IOKitUtil.getIORegistryLongProperty(childDevice, "idVendor");
158                 if (vendorId != 0) {
159                     vendorIdMap.put(childId.getValue(), String.format("%04x", 0xffff & vendorId));
160                 }
161                 // Get productId and store in map
162                 long productId = IOKitUtil.getIORegistryLongProperty(childDevice, "idProduct");
163                 if (productId != 0) {
164                     productIdMap.put(childId.getValue(), String.format("%04x", 0xffff & productId));
165                 }
166                 // Get serial and store in map
167                 String serial = IOKitUtil.getIORegistryStringProperty(childDevice, "USB Serial Number");
168                 if (serial != null) {
169                     serialMap.put(childId.getValue(), serial);
170                 }
171                 IOKit.INSTANCE.IOObjectRelease(childDevice);
172                 childDevice = IOKit.INSTANCE.IOIteratorNext(childIter.getValue());
173             }
174             IOKit.INSTANCE.IOObjectRelease(childIter.getValue());
175 
176             IOKit.INSTANCE.IOObjectRelease(device);
177             device = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
178         }
179         IOKit.INSTANCE.IOObjectRelease(iter.getValue());
180 
181         // Build tree and return
182         List<UsbDevice> controllerDevices = new ArrayList<>();
183         for (Long controller : usbControllers) {
184             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000"));
185         }
186         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
187     }

189     /**
190      * Looks up vendor and product id information for a USB Host Controller by
191      * cross-referencing the location
192      *
193      * @param id
194      *            The global unique ID for the host controller used as a key for
195      *            maps
196      * @param locationId
197      *            The locationID of this controller returned from the registry
198      */
199     private static void getControllerIdByLocation(long id, CFTypeRef locationId) {
200         // Create a matching property dictionary from the locationId
201         CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0,
202                 null, null);
203         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, CfUtil.getCFString("locationID"), locationId);
204         CFMutableDictionaryRef matchingDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0,
205                 null, null);
206         CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict, CfUtil.getCFString("IOPropertyMatch"), propertyDict);
207 
208         // search for all IOservices that match the locationID
209         IntByReference serviceIterator = new IntByReference();
210         IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
211         // getMatchingServices releases matchingDict
212         CfUtil.release(propertyDict);
213 
214         // Iterate matching services looking for devices whose parents have the
215         // vendor and device ids
216         boolean found = false;
217         int matchingService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
218         while (matchingService != 0 && !found) {
219             // Get the parent, which contains the keys we need
220             IntByReference parent = new IntByReference();
221             IOKit.INSTANCE.IORegistryEntryGetParentEntry(matchingService, "IOService", parent);
222             // look up the vendor-id by key
223             // vendor-id is a byte array of 4 bytes
224             byte[] vid = IOKitUtil.getIORegistryByteArrayProperty(parent.getValue(), "vendor-id");
225             if (vid != null && vid.length >= 2) {
226                 vendorIdMap.put(id, String.format("%02x%02x", vid[1], vid[0]));
227                 found = true;
228             }
229             // look up the device-id by key
230             // device-id is a byte array of 4 bytes
231             byte[] pid = IOKitUtil.getIORegistryByteArrayProperty(parent.getValue(), "device-id");
232             if (pid != null && pid.length >= 2) {
233                 productIdMap.put(id, String.format("%02x%02x", pid[1], pid[0]));
234                 found = true;
235             }
236             // iterate
237             matchingService = IOKit.INSTANCE.IOIteratorNext(matchingService);
238         }
239     }

241     /**
242      * Recursively creates MacUsbDevices by fetching information from maps to
243      * populate fields
244      *
245      * @param registryEntryId
246      *            The device unique registry id.
247      * @param vid
248      *            The default (parent) vendor ID
249      * @param pid
250      *            The default (parent) product ID
251      * @return A MacUsbDevice corresponding to this device
252      */
253     private static MacUsbDevice getDeviceAndChildren(Long registryEntryId, String vid, String pid) {
254         String vendorId = MapUtil.getOrDefault(vendorIdMap, registryEntryId, vid);
255         String productId = MapUtil.getOrDefault(productIdMap, registryEntryId, pid);
256         List<Long> childIds = MapUtil.getOrDefault(hubMap, registryEntryId, new ArrayList<Long>());
257         List<MacUsbDevice> usbDevices = new ArrayList<>();
258         for (Long id : childIds) {
259             usbDevices.add(getDeviceAndChildren(id, vendorId, productId));
260         }
261         Collections.sort(usbDevices);
262         return new MacUsbDevice(MapUtil.getOrDefault(nameMap, registryEntryId, vendorId + ":" + productId),
263                 MapUtil.getOrDefault(vendorMap, registryEntryId, ""), vendorId, productId,
264                 MapUtil.getOrDefault(serialMap, registryEntryId, ""),
265                 usbDevices.toArray(new UsbDevice[usbDevices.size()]));
266     }
