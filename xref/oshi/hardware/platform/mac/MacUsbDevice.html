78      /**
79       * {@inheritDoc}
80       *
81       * @param tree
82       *            a boolean.
83       * @return an array of {@link oshi.hardware.UsbDevice} objects.
84       */
85      public static UsbDevice[] getUsbDevices(boolean tree) {
86          UsbDevice[] devices = getUsbDevices();
87          if (tree) {
88              return devices;
89          }
90          List<UsbDevice> deviceList = new ArrayList<>();
91          // Top level is controllers; they won't be added to the list, but all
92          // their connected devices will be
93          for (UsbDevice device : devices) {
94              addDevicesToList(deviceList, device.getConnectedDevices());
95          }
96          return deviceList.toArray(new UsbDevice[0]);
97      }

99      private static UsbDevice[] getUsbDevices() {
100         // Maps to store information using RegistryEntryID as the key
101         Map<Long, String> nameMap = new HashMap<>();
102         Map<Long, String> vendorMap = new HashMap<>();
103         Map<Long, String> vendorIdMap = new HashMap<>();
104         Map<Long, String> productIdMap = new HashMap<>();
105         Map<Long, String> serialMap = new HashMap<>();
106         Map<Long, List<Long>> hubMap = new HashMap<>();
107 
108         // Iterate over USB Controllers. All devices are children of one of
109         // these controllers in the "IOService" plane
110         List<Long> usbControllers = new ArrayList<>();
111         IOIterator iter = IOKitUtil.getMatchingServices("IOUSBController");
112         if (iter != null) {
113             // Define keys
114             CFStringRef locationIDKey = CFStringRef.createCFString("locationID");
115             CFStringRef ioPropertyMatchKey = CFStringRef.createCFString("IOPropertyMatch");
116 
117             IORegistryEntry device = iter.next();
118             while (device != null) {
119                 // Unique global identifier for this device
120                 long id = device.getRegistryEntryID();
121                 usbControllers.add(id);
122 
123                 // Get device name and store in map
124                 nameMap.put(id, device.getName());
125                 // The only information we have in registry for this device is
126                 // the
127                 // locationID. Use that to search for matching PCI device to
128                 // obtain
129                 // more information.
130                 CFTypeRef ref = device.createCFProperty(locationIDKey);
131                 if (ref != null) {
132                     getControllerIdByLocation(id, ref, locationIDKey, ioPropertyMatchKey, vendorIdMap, productIdMap);
133                     ref.release();
134                 }
135 
136                 // Now iterate the children of this device in the "IOService"
137                 // plane.
138                 // If device parent is root, link to the controller
139                 IOIterator childIter = device.getChildIterator("IOService");
140                 IORegistryEntry childDevice = childIter.next();
141                 while (childDevice != null) {
142                     // Unique global identifier for this device
143                     long childId = childDevice.getRegistryEntryID();
144 
145                     // Get this device's parent in the "IOUSB" plane
146                     IORegistryEntry parent = childDevice.getParentEntry("IOUSB");
147                     // If the parent is not an IOUSBDevice (will be root), set
148                     // the parentId to the controller
149                     long parentId;
150                     if (parent == null || !parent.conformsTo("IOUSBDevice")) {
151                         parentId = id;
152                     } else {
153                         // Unique global identifier for the parent
154                         parentId = parent.getRegistryEntryID();
155                     }
156                     if (parent != null) {
157                         parent.release();
158                     }
159                     // Store parent in map
160                     hubMap.computeIfAbsent(parentId, x -> new ArrayList<>()).add(childId);
161                     // Get device name and store in map
162                     nameMap.put(childId, childDevice.getName().trim());
163                     // Get vendor and store in map
164                     String vendor = childDevice.getStringProperty("USB Vendor Name");
165                     if (vendor != null) {
166                         vendorMap.put(childId, vendor.trim());
167                     }
168                     // Get vendorId and store in map
169                     Long vendorId = childDevice.getLongProperty("idVendor");
170                     if (vendorId != null) {
171                         vendorIdMap.put(childId, String.format("%04x", 0xffff & vendorId));
172                     }
173                     // Get productId and store in map
174                     Long productId = childDevice.getLongProperty("idProduct");
175                     if (productId != null) {
176                         productIdMap.put(childId, String.format("%04x", 0xffff & productId));
177                     }
178                     // Get serial and store in map
179                     String serial = childDevice.getStringProperty("USB Serial Number");
180                     if (serial != null) {
181                         serialMap.put(childId, serial.trim());
182                     }
183                     childDevice.release();
184                     childDevice = childIter.next();
185                 }
186                 childIter.release();
187                 device.release();
188                 device = iter.next();
189             }
190             iter.release();
191             locationIDKey.release();
192             ioPropertyMatchKey.release();
193         }
194 
195         // Build tree and return
196         List<UsbDevice> controllerDevices = new ArrayList<>();
197         for (Long controller : usbControllers) {
198             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
199                     productIdMap, serialMap, hubMap));
200         }
201         return controllerDevices.toArray(new UsbDevice[0]);
202     }

204     private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
205         for (UsbDevice device : connectedDevices) {
206             deviceList.add(new MacUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
207                     device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(), new MacUsbDevice[0]));
208             addDevicesToList(deviceList, device.getConnectedDevices());
209         }
210     }

212     /**
213      * Looks up vendor and product id information for a USB Host Controller by
214      * cross-referencing the location
215      *
216      * @param id
217      *            The global unique ID for the host controller used as a key for
218      *            maps
219      * @param locationId
220      *            The locationID of this controller returned from the registry
221      * @param locationIDKey
222      *            A pointer to the locationID string
223      * @param ioPropertyMatchKey
224      *            A pointer to the IOPropertyMatch string
225      * @param productIdMap
226      * @param vendorIdMap
227      */
228     private static void getControllerIdByLocation(long id, CFTypeRef locationId, CFStringRef locationIDKey,
229             CFStringRef ioPropertyMatchKey, Map<Long, String> vendorIdMap, Map<Long, String> productIdMap) {
230         // Create a matching property dictionary from the locationId
231         CFMutableDictionaryRef propertyDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(), new CFIndex(0),
232                 null, null);
233         propertyDict.setValue(locationIDKey, locationId);
234         CFMutableDictionaryRef matchingDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(), new CFIndex(0),
235                 null, null);
236         matchingDict.setValue(ioPropertyMatchKey, propertyDict);
237 
238         // search for all IOservices that match the locationID
239         IOIterator serviceIterator = IOKitUtil.getMatchingServices(matchingDict);
240         // getMatchingServices releases matchingDict
241         propertyDict.release();
242 
243         // Iterate matching services looking for devices whose parents have the
244         // vendor and device ids
245         boolean found = false;
246         if (serviceIterator != null) {
247             IORegistryEntry matchingService = serviceIterator.next();
248             while (matchingService != null && !found) {
249                 // Get the parent, which contains the keys we need
250                 IORegistryEntry parent = matchingService.getParentEntry("IOService");
251                 // look up the vendor-id by key
252                 // vendor-id is a byte array of 4 bytes
253                 if (parent != null) {
254                     byte[] vid = parent.getByteArrayProperty("vendor-id");
255                     if (vid != null && vid.length >= 2) {
256                         vendorIdMap.put(id, String.format("%02x%02x", vid[1], vid[0]));
257                         found = true;
258                     }
259                     // look up the device-id by key
260                     // device-id is a byte array of 4 bytes
261                     byte[] pid = parent.getByteArrayProperty("device-id");
262                     if (pid != null && pid.length >= 2) {
263                         productIdMap.put(id, String.format("%02x%02x", pid[1], pid[0]));
264                         found = true;
265                     }
266                     parent.release();
267                 }
268                 // iterate
269                 matchingService.release();
270                 matchingService = serviceIterator.next();
271             }
272             serviceIterator.release();
273         }
274     }

276     /**
277      * Recursively creates MacUsbDevices by fetching information from maps to
278      * populate fields
279      *
280      * @param registryEntryId
281      *            The device unique registry id.
282      * @param vid
283      *            The default (parent) vendor ID
284      * @param pid
285      *            The default (parent) product ID
286      * @param nameMap
287      * @param vendorMap
288      * @param vendorIdMap
289      * @param productIdMap
290      * @param serialMap
291      * @param hubMap
292      * @return A MacUsbDevice corresponding to this device
293      */
294     private static MacUsbDevice getDeviceAndChildren(Long registryEntryId, String vid, String pid,
295             Map<Long, String> nameMap, Map<Long, String> vendorMap, Map<Long, String> vendorIdMap,
296             Map<Long, String> productIdMap, Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
297         String vendorId = vendorIdMap.getOrDefault(registryEntryId, vid);
298         String productId = productIdMap.getOrDefault(registryEntryId, pid);
299         List<Long> childIds = hubMap.getOrDefault(registryEntryId, new ArrayList<Long>());
300         List<MacUsbDevice> usbDevices = new ArrayList<>();
301         for (Long id : childIds) {
302             usbDevices.add(getDeviceAndChildren(id, vendorId, productId, nameMap, vendorMap, vendorIdMap, productIdMap,
303                     serialMap, hubMap));
304         }
305         Collections.sort(usbDevices);
306         return new MacUsbDevice(nameMap.getOrDefault(registryEntryId, vendorId + ":" + productId),
307                 vendorMap.getOrDefault(registryEntryId, ""), vendorId, productId,
308                 serialMap.getOrDefault(registryEntryId, ""), "0x" + Long.toHexString(registryEntryId),
309                 usbDevices.toArray(new UsbDevice[0]));
310     }
