27  /**
28   * Enumerates Mac USB devices via FFM/IOKit, adapting the FFM registry-entry wrapper to {@link MacUsbDevice}'s
29   * backend-neutral view so the tree-building logic is shared.
30   */
31  public final class MacUsbDeviceFFM {
32  
33      private MacUsbDeviceFFM() {
34      }
35  
36      private static final String IOSERVICE = "IOService";
37      private static final Logger LOG = LoggerFactory.getLogger(MacUsbDeviceFFM.class);
38  
39      /**
40       * Instantiates the USB controller device tree. The flat form is derived by the caller (the HAL).
41       *
42       * @return a list of USB controllers, each with its connected-device tree.
43       */
44      public static List<UsbDevice> getUsbDevices() {
45          IORegistryEntry root = IOKitUtilFFM.getRoot();
46          return MacUsbDevice.getUsbDevices(root == null ? null : new FfmRegistryEntry(root));
47      }

49      /**
50       * Adapts an FFM {@link IORegistryEntry} to {@link MacUsbDevice.RegistryEntry}.
51       */
52      private static final class FfmRegistryEntry implements MacUsbDevice.RegistryEntry {
53          private final IORegistryEntry entry;
54  
55          FfmRegistryEntry(IORegistryEntry entry) {
56              this.entry = entry;
57          }
58  
59          @Override
60          public long getRegistryEntryID() {
61              return entry.getRegistryEntryID();
62          }
63  
64          @Override
65          public String getName() {
66              return entry.getName();
67          }
68  
69          @Override
70          public String getStringProperty(String key) {
71              return entry.getStringProperty(key);
72          }
73  
74          @Override
75          public Long getLongProperty(String key) {
76              return entry.getLongProperty(key);
77          }
78  
79          @Override
80          public MacUsbDevice.RegistryEntry getParentEntry(String plane) {
81              IORegistryEntry parent = entry.getParentEntry(plane);
82              return parent == null ? null : new FfmRegistryEntry(parent);
83          }
84  
85          @Override
86          public MacUsbDevice.RegistryIterator getChildIterator(String plane) {
87              IOIterator iter = entry.getChildIterator(plane);
88              return iter == null ? null : new FfmRegistryIterator(iter);
89          }
90  
91          @Override
92          public String[] lookupControllerVidPid() {
93              String[] result = new String[2];
94              CFStringRef locationIDKey = CFStringRef.createCFString("locationID");
95              CFStringRef ioPropertyMatchKey = CFStringRef.createCFString("IOPropertyMatch");
96              try (locationIDKey; ioPropertyMatchKey) {
97                  MemorySegment ref = entry.createCFProperty(locationIDKey.segment());
98                  if (ref == null || ref.equals(MemorySegment.NULL)) {
99                      return result;
100                 }
101                 CFTypeRef locationId = new CFTypeRef(ref);
102                 try (locationId) {
103                     // Build matching dict: { IOPropertyMatch: { locationID: <locationId> } }
104                     runOrLog(() -> {
105                         CFAllocatorRef alloc = new CFAllocatorRef(CoreFoundationFunctions.CFAllocatorGetDefault());
106                         CFMutableDictionaryRef propertyDict = new CFMutableDictionaryRef(CoreFoundationFunctions
107                                 .CFDictionaryCreateMutable(alloc.segment(), 0, MemorySegment.NULL, MemorySegment.NULL));
108                         propertyDict.setValue(locationIDKey, locationId);
109                         CFMutableDictionaryRef matchingDict = new CFMutableDictionaryRef(CoreFoundationFunctions
110                                 .CFDictionaryCreateMutable(alloc.segment(), 0, MemorySegment.NULL, MemorySegment.NULL));
111                         matchingDict.setValue(ioPropertyMatchKey, propertyDict);
112 
113                         IOIterator serviceIterator = IOKitUtilFFM.getMatchingServices(matchingDict.segment());
114                         propertyDict.release();
115                         readVidPid(serviceIterator, result);
116                     }, LOG, "Failed to retrieve controller vendor/product IDs");
117                 }
118             }
119             return result;
120         }
121 
122         /**
123          * Iterates matching services, reading vendor-id/device-id from the first parent that carries them.
124          *
125          * @param serviceIterator the iterator of matching services, or {@code null}
126          * @param result          the two-element {@code {vendorId, productId}} array to populate
127          */
128         private static void readVidPid(IOIterator serviceIterator, String[] result) {
129             if (serviceIterator == null) {
130                 return;
131             }
132             try (serviceIterator) {
133                 boolean found = false;
134                 IORegistryEntry matchingService = serviceIterator.next();
135                 while (matchingService != null && !found) {
136                     // The parent holds the vendor-id/device-id keys, each a 4-byte array
137                     IORegistryEntry parent = matchingService.getParentEntry(IOSERVICE);
138                     if (parent != null) {
139                         byte[] vid = parent.getByteArrayProperty("vendor-id");
140                         if (vid != null && vid.length >= 2) {
141                             result[0] = String.format(Locale.ROOT, "%02x%02x", vid[1], vid[0]);
142                             found = true;
143                         }
144                         byte[] pid = parent.getByteArrayProperty("device-id");
145                         if (pid != null && pid.length >= 2) {
146                             result[1] = String.format(Locale.ROOT, "%02x%02x", pid[1], pid[0]);
147                             found = true;
148                         }
149                         parent.release();
150                     }
151                     matchingService.release();
152                     matchingService = serviceIterator.next();
153                 }
154             }
155         }
156 
157         @Override
158         public void release() {
159             entry.release();
160         }
161     }

163     /**
164      * Adapts an FFM {@link IOIterator} to {@link MacUsbDevice.RegistryIterator}.
165      */
166     private static final class FfmRegistryIterator implements MacUsbDevice.RegistryIterator {
167         private final IOIterator iter;
168 
169         FfmRegistryIterator(IOIterator iter) {
170             this.iter = iter;
171         }
172 
173         @Override
174         public MacUsbDevice.RegistryEntry next() {
175             IORegistryEntry next = iter.next();
176             return next == null ? null : new FfmRegistryEntry(next);
177         }
178 
179         @Override
180         public void release() {
181             iter.release();
182         }
183     }
