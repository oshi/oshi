26  import static oshi.util.Memoizer.memoize;

28  import java.nio.charset.StandardCharsets;
29  import java.util.function.Supplier;
30  
31  import com.sun.jna.Native; // NOSONAR squid:S1191
32  import com.sun.jna.platform.mac.IOKit.IOIterator;
33  import com.sun.jna.platform.mac.IOKit.IORegistryEntry;
34  import com.sun.jna.platform.mac.IOKitUtil;
35  
36  import oshi.annotation.concurrent.Immutable;
37  import oshi.hardware.common.AbstractFirmware;
38  import oshi.util.Constants;
39  import oshi.util.Util;
40  import oshi.util.tuples.Quintet;
41  
42  /**
43   * Firmware data obtained from ioreg.
44   */
45  @Immutable
46  final class MacFirmware extends AbstractFirmware {
47  
48      private final Supplier<Quintet<String, String, String, String, String>> manufNameDescVersRelease = memoize(
49              MacFirmware::queryEfi);
50  
51      @Override
52      public String getManufacturer() {
53          return manufNameDescVersRelease.get().getA();
54      }
55  
56      @Override
57      public String getName() {
58          return manufNameDescVersRelease.get().getB();
59      }
60  
61      @Override
62      public String getDescription() {
63          return manufNameDescVersRelease.get().getC();
64      }
65  
66      @Override
67      public String getVersion() {
68          return manufNameDescVersRelease.get().getD();
69      }
70  
71      @Override
72      public String getReleaseDate() {
73          return manufNameDescVersRelease.get().getE();
74      }
75  
76      private static Quintet<String, String, String, String, String> queryEfi() {
77          String manufacturer = null;
78          String name = null;
79          String description = null;
80          String version = null;
81          String releaseDate = null;
82  
83          IORegistryEntry platformExpert = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
84          byte[] data;
85          if (platformExpert != null) {
86              IOIterator iter = platformExpert.getChildIterator("IODeviceTree");
87              if (iter != null) {
88                  IORegistryEntry entry = iter.next();
89                  while (entry != null) {
90                      switch (entry.getName()) {
91                      case "rom":
92                          data = entry.getByteArrayProperty("vendor");
93                          if (data != null) {
94                              manufacturer = Native.toString(data, StandardCharsets.UTF_8);
95                          }
96                          data = entry.getByteArrayProperty("version");
97                          if (data != null) {
98                              version = Native.toString(data, StandardCharsets.UTF_8);
99                          }
100                         data = entry.getByteArrayProperty("release-date");
101                         if (data != null) {
102                             releaseDate = Native.toString(data, StandardCharsets.UTF_8);
103                         }
104                         break;
105                     case "chosen":
106                         data = entry.getByteArrayProperty("booter-name");
107                         if (data != null) {
108                             name = Native.toString(data, StandardCharsets.UTF_8);
109                         }
110                         break;
111                     case "efi":
112                         data = entry.getByteArrayProperty("firmware-abi");
113                         if (data != null) {
114                             description = Native.toString(data, StandardCharsets.UTF_8);
115                         }
116                         break;
117                     default:
118                         if (Util.isBlank(name)) {
119                             name = entry.getStringProperty("IONameMatch");
120                         }
121                         break;
122                     }
123                     entry.release();
124                     entry = iter.next();
125                 }
126                 iter.release();
127             }
128             if (Util.isBlank(manufacturer)) {
129                 data = platformExpert.getByteArrayProperty("manufacturer");
130                 if (data != null) {
131                     manufacturer = Native.toString(data, StandardCharsets.UTF_8);
132                 }
133             }
134             if (Util.isBlank(version)) {
135                 data = platformExpert.getByteArrayProperty("target-type");
136                 if (data != null) {
137                     version = Native.toString(data, StandardCharsets.UTF_8);
138                 }
139             }
140             if (Util.isBlank(name)) {
141                 data = platformExpert.getByteArrayProperty("device_type");
142                 if (data != null) {
143                     name = Native.toString(data, StandardCharsets.UTF_8);
144                 }
145             }
146             platformExpert.release();
147         }
148         return new Quintet<>(Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer,
149                 Util.isBlank(name) ? Constants.UNKNOWN : name,
150                 Util.isBlank(description) ? Constants.UNKNOWN : description,
151                 Util.isBlank(version) ? Constants.UNKNOWN : version,
152                 Util.isBlank(releaseDate) ? Constants.UNKNOWN : releaseDate);
153     }
