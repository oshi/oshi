74      private enum CFKey {
75          IO_PROPERTY_MATCH("IOPropertyMatch"), //
76  
77          STATISTICS("Statistics"), //
78          READ_OPS("Operations (Read)"), READ_BYTES("Bytes (Read)"), READ_TIME("Total Time (Read)"), //
79          WRITE_OPS("Operations (Write)"), WRITE_BYTES("Bytes (Write)"), WRITE_TIME("Total Time (Write)"), //
80  
81          BSD_UNIT("BSD Unit"), LEAF("Leaf"), WHOLE("Whole"), //
82  
83          DA_MEDIA_NAME("DAMediaName"), DA_VOLUME_NAME("DAVolumeName"), DA_MEDIA_SIZE("DAMediaSize"), //
84          DA_DEVICE_MODEL("DADeviceModel"), MODEL("Model");
85  
86          private final String key;
87  
88          CFKey(String key) {
89              this.key = key;
90          }
91  
92          public String getKey() {
93              return this.key;
94          }
95      }

97      /**
98       * Temporarily cache pointers to keys. The values from this map must be
99       * released after use.}
100      * 
101      * @return A map of keys in the {@link CFKey} enum to corresponding
102      *         {@link CFStringRef}.
103      */
104     private static Map<CFKey, CFStringRef> mapCFKeys() {
105         Map<CFKey, CFStringRef> keyMap = new EnumMap<>(CFKey.class);
106         for (CFKey cfKey : CFKey.values()) {
107             keyMap.put(cfKey, CFStringRef.toCFString(cfKey.getKey()));
108         }
109         return keyMap;
110     }

112     private static Map<String, String> queryMountPointMap() {
113         final Map<String, String> mountPointMap = new HashMap<>();
114         // Use statfs to populate mount point map
115         int numfs = SystemB.INSTANCE.getfsstat64(null, 0, 0);
116         // Create array to hold results
117         Statfs[] fs = new Statfs[numfs];
118         // Fill array with results
119         SystemB.INSTANCE.getfsstat64(fs, numfs * new Statfs().size(), SystemB.MNT_NOWAIT);
120         // Iterate all mounted file systems
121         for (Statfs f : fs) {
122             String mntFrom = new String(f.f_mntfromname).trim();
123             mountPointMap.put(mntFrom.replace("/dev/", ""), new String(f.f_mntonname).trim());
124         }
125         return mountPointMap;
126     }

128     private static Map<String, String> queryLogicalVolumeMap() {
129         final Map<String, String> logicalVolumeMap = new HashMap<>();
130         // Parse `diskutil cs list` to populate logical volume map
131         Set<String> physicalVolumes = new HashSet<>();
132         boolean logicalVolume = false;
133         for (String line : ExecutingCommand.runNative("diskutil cs list")) {
134             if (line.contains("Logical Volume Group")) {
135                 // Logical Volume Group defines beginning of grouping which will
136                 // list multiple physical volumes followed by the logical volume
137                 // they are associated with. Each physical volume will be a key
138                 // with the logical volume as its value, but since the value
139                 // doesn't appear until the end we collect the keys in a list
140                 physicalVolumes.clear();
141                 logicalVolume = false;
142             } else if (line.contains("Logical Volume Family")) {
143                 // Done collecting physical volumes, prepare to store logical
144                 // volume
145                 logicalVolume = true;
146             } else if (line.contains("Disk:")) {
147                 String volume = ParseUtil.parseLastString(line);
148                 if (logicalVolume) {
149                     // Store this disk as the logical volume value for all the
150                     // physical volume keys
151                     for (String pv : physicalVolumes) {
152                         logicalVolumeMap.put(pv, volume);
153                     }
154                     physicalVolumes.clear();
155                 } else {
156                     physicalVolumes.add(ParseUtil.parseLastString(line));
157                 }
158             }
159         }
160         return logicalVolumeMap;
161     }

163     private static boolean updateDiskStats(HWDiskStore diskStore, DASessionRef session,
164             Map<String, String> mountPointMap, Map<String, String> logicalVolumeMap, Map<CFKey, CFStringRef> cfKeyMap) {
165         // Now look up the device using the BSD Name to get its
166         // statistics
167         String bsdName = diskStore.getName();
168         CFMutableDictionaryRef matchingDict = IOKitUtil.getBSDNameMatchingDict(bsdName);
169         if (matchingDict != null) {
170             // search for all IOservices that match the bsd name
171             IntByReference driveList = new IntByReference();
172             IOKitUtil.getMatchingServices(matchingDict, driveList);
173             // getMatchingServices releases matchingDict
174             int drive = IOKit.INSTANCE.IOIteratorNext(driveList.getValue());
175             // Should only match one drive
176             if (drive != 0) {
177                 // Should be an IOMedia object with a parent
178                 // IOBlockStorageDriver object
179                 // Get the properties from the parent
180                 IntByReference parent = new IntByReference();
181                 if (IOKit.INSTANCE.IOObjectConformsTo(drive, "IOMedia")
182                         && IOKit.INSTANCE.IORegistryEntryGetParentEntry(drive, "IOService", parent) == 0) {
183                     PointerByReference propsPtr = new PointerByReference();
184                     if (IOKit.INSTANCE.IOObjectConformsTo(parent.getValue(), "IOBlockStorageDriver") && IOKit.INSTANCE
185                             .IORegistryEntryCreateCFProperties(parent.getValue(), propsPtr, CfUtil.ALLOCATOR, 0) == 0) {
186                         CFMutableDictionaryRef properties = new CFMutableDictionaryRef();
187                         properties.setPointer(propsPtr.getValue());
188                         // We now have a properties object with the
189                         // statistics we need on it. Fetch them
190                         Pointer statsPtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties,
191                                 cfKeyMap.get(CFKey.STATISTICS));
192                         diskStore.setTimeStamp(System.currentTimeMillis());
193                         CFDictionaryRef statistics = new CFDictionaryRef();
194                         statistics.setPointer(statsPtr);
195 
196                         // Now get the stats we want
197                         Pointer stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
198                                 cfKeyMap.get(CFKey.READ_OPS));
199                         diskStore.setReads(CfUtil.cfPointerToLong(stat));
200                         stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics, cfKeyMap.get(CFKey.READ_BYTES));
201                         diskStore.setReadBytes(CfUtil.cfPointerToLong(stat));
202 
203                         stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics, cfKeyMap.get(CFKey.WRITE_OPS));
204                         diskStore.setWrites(CfUtil.cfPointerToLong(stat));
205                         stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
206                                 cfKeyMap.get(CFKey.WRITE_BYTES));
207                         diskStore.setWriteBytes(CfUtil.cfPointerToLong(stat));
208 
209                         // Total time is in nanoseconds. Add read+write
210                         // and convert total to ms
211                         stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics, cfKeyMap.get(CFKey.READ_TIME));
212                         long xferTime = CfUtil.cfPointerToLong(stat);
213                         stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics, cfKeyMap.get(CFKey.WRITE_TIME));
214                         xferTime += CfUtil.cfPointerToLong(stat);
215                         diskStore.setTransferTime(xferTime / 1_000_000L);
216 
217                         CfUtil.release(properties);
218                     } else {
219                         // This is normal for FileVault drives, Fusion
220                         // drives, and other virtual bsd names
221                         LOG.debug("Unable to find block storage driver properties for {}", bsdName);
222                     }
223                     // Now get partitions for this disk.
224                     List<HWPartition> partitions = new ArrayList<>();
225                     if (IOKit.INSTANCE.IORegistryEntryCreateCFProperties(drive, propsPtr, CfUtil.ALLOCATOR, 0) == 0) {
226                         CFMutableDictionaryRef properties = new CFMutableDictionaryRef();
227                         properties.setPointer(propsPtr.getValue());
228                         // Partitions will match BSD Unit property
229                         Pointer p = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties,
230                                 cfKeyMap.get(CFKey.BSD_UNIT));
231                         CFNumberRef bsdUnit = new CFNumberRef();
232                         bsdUnit.setPointer(p);
233                         // We need a CFBoolean that's false.
234                         // Whole disk has 'true' for Whole and 'false'
235                         // for leaf; store the boolean false
236                         p = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties, cfKeyMap.get(CFKey.LEAF));
237                         CFBooleanRef cfFalse = new CFBooleanRef();
238                         cfFalse.setPointer(p);
239                         // create a matching dict for BSD Unit
240                         CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE
241                                 .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
242                         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, cfKeyMap.get(CFKey.BSD_UNIT),
243                                 bsdUnit);
244                         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, cfKeyMap.get(CFKey.WHOLE), cfFalse);
245                         matchingDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null,
246                                 null);
247                         CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict,
248                                 cfKeyMap.get(CFKey.IO_PROPERTY_MATCH), propertyDict);
249 
250                         // search for IOservices that match the BSD Unit
251                         // with whole=false; these are partitions
252                         IntByReference serviceIterator = new IntByReference();
253                         IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
254                         // getMatchingServices releases matchingDict
255                         CfUtil.release(properties);
256                         CfUtil.release(propertyDict);
257 
258                         // Iterate disks
259                         int sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
260                         while (sdService != 0) {
261                             // look up the BSD Name
262                             String partBsdName = IOKitUtil.getIORegistryStringProperty(sdService, "BSD Name");
263                             String name = partBsdName;
264                             String type = "";
265                             // Get the DiskArbitration dictionary for
266                             // this partition
267                             DADiskRef disk = DiskArbitration.INSTANCE.DADiskCreateFromBSDName(CfUtil.ALLOCATOR, session,
268                                     partBsdName);
269                             if (disk != null) {
270                                 CFDictionaryRef diskInfo = DiskArbitration.INSTANCE.DADiskCopyDescription(disk);
271                                 if (diskInfo != null) {
272                                     // get volume name from its key
273                                     Pointer volumePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
274                                             cfKeyMap.get(CFKey.DA_MEDIA_NAME));
275                                     type = CfUtil.cfPointerToString(volumePtr);
276                                     volumePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
277                                             cfKeyMap.get(CFKey.DA_VOLUME_NAME));
278                                     if (volumePtr == null) {
279                                         name = type;
280                                     } else {
281                                         name = CfUtil.cfPointerToString(volumePtr);
282                                     }
283                                     CfUtil.release(diskInfo);
284                                 }
285                                 CfUtil.release(disk);
286                             }
287                             String mountPoint;
288                             if (logicalVolumeMap.containsKey(partBsdName)) {
289                                 mountPoint = "Logical Volume: " + logicalVolumeMap.get(partBsdName);
290                             } else {
291                                 mountPoint = mountPointMap.getOrDefault(partBsdName, "");
292                             }
293                             partitions.add(new HWPartition(partBsdName, name, type,
294                                     IOKitUtil.getIORegistryStringProperty(sdService, "UUID"),
295                                     IOKitUtil.getIORegistryLongProperty(sdService, "Size"),
296                                     IOKitUtil.getIORegistryIntProperty(sdService, "BSD Major"),
297                                     IOKitUtil.getIORegistryIntProperty(sdService, "BSD Minor"), mountPoint));
298                             IOKit.INSTANCE.IOObjectRelease(sdService);
299                             // iterate
300                             sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
301                         }
302                         IOKit.INSTANCE.IOObjectRelease(serviceIterator.getValue());
303 
304                     } else {
305                         LOG.error("Unable to find properties for {}", bsdName);
306                     }
307                     Collections.sort(partitions);
308                     diskStore.setPartitions(partitions.toArray(new HWPartition[0]));
309                     IOKit.INSTANCE.IOObjectRelease(parent.getValue());
310                 } else {
311                     LOG.error("Unable to find IOMedia device or parent for {}", bsdName);
312                 }
313                 IOKit.INSTANCE.IOObjectRelease(drive);
314             }
315             IOKit.INSTANCE.IOObjectRelease(driveList.getValue());
316             return true;
317         } else {
318             return false;
319         }
320     }

322     public static boolean updateDiskStats(HWDiskStore diskStore) {
323         DASessionRef session = DiskArbitration.INSTANCE.DASessionCreate(CfUtil.ALLOCATOR);
324         if (session == null) {
325             LOG.error("Unable to open session to DiskArbitration framework.");
326             return false;
327         }
328         Map<CFKey, CFStringRef> cfKeyMap = mapCFKeys();
329 
330         boolean diskFound = updateDiskStats(diskStore, session, queryMountPointMap(), queryLogicalVolumeMap(),
331                 cfKeyMap);
332 
333         CfUtil.release(session);
334         CfUtil.releaseAll(cfKeyMap.values());
335 
336         return diskFound;
337     }

339     @Override
340     public HWDiskStore[] getDisks() {
341         Map<String, String> mountPointMap = queryMountPointMap();
342         Map<String, String> logicalVolumeMap = queryLogicalVolumeMap();
343         Map<CFKey, CFStringRef> cfKeyMap = mapCFKeys();
344 
345         List<HWDiskStore> result = new ArrayList<>();
346 
347         // Open a DiskArbitration session
348         DASessionRef session = DiskArbitration.INSTANCE.DASessionCreate(CfUtil.ALLOCATOR);
349         if (session == null) {
350             LOG.error("Unable to open session to DiskArbitration framework.");
351             return new HWDiskStore[0];
352         }
353 
354         // Get IOMedia objects representing whole drives
355         List<String> bsdNames = new ArrayList<>();
356         IntByReference iter = new IntByReference();
357         IOKitUtil.getMatchingServices("IOMedia", iter);
358         int media = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
359         while (media != 0) {
360             if (IOKitUtil.getIORegistryBooleanProperty(media, "Whole")) {
361                 DADiskRef disk = DiskArbitration.INSTANCE.DADiskCreateFromIOMedia(CfUtil.ALLOCATOR, session, media);
362                 bsdNames.add(DiskArbitration.INSTANCE.DADiskGetBSDName(disk));
363             }
364             IOKit.INSTANCE.IOObjectRelease(media);
365             media = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
366         }
367 
368         // Now iterate the bsdNames
369         for (String bsdName : bsdNames) {
370             String model = "";
371             String serial = "";
372             long size = 0L;
373 
374             // Get a reference to the disk - only matching /dev/disk*
375             String path = "/dev/" + bsdName;
376 
377             // Get the DiskArbitration dictionary for this disk, which has model
378             // and size (capacity)
379             DADiskRef disk = DiskArbitration.INSTANCE.DADiskCreateFromBSDName(CfUtil.ALLOCATOR, session, path);
380             if (disk != null) {
381                 CFDictionaryRef diskInfo = DiskArbitration.INSTANCE.DADiskCopyDescription(disk);
382                 if (diskInfo != null) {
383                     // Parse out model and size from their respective keys
384                     Pointer modelPtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
385                             cfKeyMap.get(CFKey.DA_DEVICE_MODEL));
386                     model = CfUtil.cfPointerToString(modelPtr);
387                     Pointer sizePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
388                             cfKeyMap.get(CFKey.DA_MEDIA_SIZE));
389                     size = CfUtil.cfPointerToLong(sizePtr);
390                     CfUtil.release(diskInfo);
391 
392                     // Use the model as a key to get serial from IOKit
393                     if (!"Disk Image".equals(model)) {
394                         CFStringRef modelNameRef = CFStringRef.toCFString(model);
395                         CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE
396                                 .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
397                         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, cfKeyMap.get(CFKey.MODEL),
398                                 modelNameRef);
399                         CFMutableDictionaryRef matchingDict = CoreFoundation.INSTANCE
400                                 .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
401                         CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict,
402                                 cfKeyMap.get(CFKey.IO_PROPERTY_MATCH), propertyDict);
403 
404                         // search for all IOservices that match the model
405                         IntByReference serviceIterator = new IntByReference();
406                         IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
407                         // getMatchingServices releases matchingDict
408                         CfUtil.release(modelNameRef);
409                         CfUtil.release(propertyDict);
410                         int sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
411                         while (sdService != 0) {
412                             // look up the serial number
413                             serial = IOKitUtil.getIORegistryStringProperty(sdService, "Serial Number");
414                             IOKit.INSTANCE.IOObjectRelease(sdService);
415                             if (serial != null) {
416                                 break;
417                             }
418                             // iterate
419                             sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
420                         }
421                         if (serial == null) {
422                             serial = "";
423                         }
424                         IOKit.INSTANCE.IOObjectRelease(serviceIterator.getValue());
425                     }
426                 }
427                 CfUtil.release(disk);
428 
429                 // If empty, ignore
430                 if (size <= 0) {
431                     continue;
432                 }
433                 HWDiskStorekStore">HWDiskStore diskStore = new HWDiskStore();
434                 diskStore.setName(bsdName);
435                 diskStore.setModel(model.trim());
436                 diskStore.setSerial(serial.trim());
437                 diskStore.setSize(size);
438 
439                 updateDiskStats(diskStore, session, mountPointMap, logicalVolumeMap, cfKeyMap);
440                 result.add(diskStore);
441             }
442         }
443         // Close DA session
444         CfUtil.release(session);
445         CfUtil.releaseAll(cfKeyMap.values());
446         Collections.sort(result);
447         return result.toArray(new HWDiskStore[0]);
448     }
