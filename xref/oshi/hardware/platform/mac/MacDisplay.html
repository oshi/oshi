24  package oshi.hardware.platform.mac;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  import com.sun.jna.Pointer; // NOSONAR squid:S1191
33  import com.sun.jna.platform.mac.CoreFoundation.CFDataRef;
34  import com.sun.jna.platform.mac.CoreFoundation.CFStringRef;
35  import com.sun.jna.platform.mac.CoreFoundation.CFTypeRef;
36  import com.sun.jna.platform.mac.IOKit.IOIterator;
37  import com.sun.jna.platform.mac.IOKit.IORegistryEntry;
38  import com.sun.jna.platform.mac.IOKitUtil;
39  
40  import oshi.annotation.concurrent.Immutable;
41  import oshi.hardware.Display;
42  import oshi.hardware.common.AbstractDisplay;
43  
44  /**
45   * A Display
46   */
47  @Immutable
48  final class MacDisplay extends AbstractDisplay {
49  
50      private static final Logger LOG = LoggerFactory.getLogger(MacDisplay.class);
51  
52      /**
53       * Constructor for MacDisplay.
54       *
55       * @param edid
56       *            a byte array representing a display EDID
57       */
58      MacDisplay(byte[] edid) {
59          super(edid);
60          LOG.debug("Initialized MacDisplay");
61      }
62  
63      /**
64       * Gets Display Information
65       *
66       * @return An array of Display objects representing monitors, etc.
67       */
68      public static List<Display> getDisplays() {
69          List<Display> displays = new ArrayList<>();
70          // Iterate IO Registry IODisplayConnect
71          IOIterator serviceIterator = IOKitUtil.getMatchingServices("IODisplayConnect");
72          if (serviceIterator != null) {
73              CFStringRef cfEdid = CFStringRef.createCFString("IODisplayEDID");
74              IORegistryEntry sdService = serviceIterator.next();
75              while (sdService != null) {
76                  // Display properties are in a child entry
77                  IORegistryEntry properties = sdService.getChildEntry("IOService");
78                  if (properties != null) {
79                      // look up the edid by key
80                      CFTypeRef edidRaw = properties.createCFProperty(cfEdid);
81                      if (edidRaw != null) {
82                          CFDataRef edid = new CFDataRef(edidRaw.getPointer());
83                          // Edid is a byte array of 128 bytes
84                          int length = edid.getLength();
85                          Pointer p = edid.getBytePtr();
86                          displays.add(new MacDisplay(p.getByteArray(0, length)));
87                          edid.release();
88                      }
89                      properties.release();
90                  }
91                  // iterate
92                  sdService.release();
93                  sdService = serviceIterator.next();
94              }
95              serviceIterator.release();
96              cfEdid.release();
97          }
98          return displays;
99      }
100 }
