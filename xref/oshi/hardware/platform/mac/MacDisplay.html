17  package oshi.hardware.platform.mac;
18  
19  import java.util.ArrayList;
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import javax.json.Json;
24  import javax.json.JsonBuilderFactory;
25  import javax.json.JsonObject;
26  
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import oshi.hardware.Display;
31  import oshi.json.NullAwareJsonObjectBuilder;
32  import oshi.util.EdidUtil;
33  import oshi.util.ExecutingCommand;
34  import oshi.util.ParseUtil;
35  
36  /**
37   * A Display
38   * 
39   * @author widdis[at]gmail[dot]com
40   */
41  public class MacDisplay implements Display {
42      private static final Logger LOG = LoggerFactory.getLogger(MacDisplay.class);
43  
44      private byte[] edid;
45  
46      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
47  
48      public MacDisplay(byte[] edid) {
49          this.edid = edid;
50          LOG.debug("Initialized MacDisplay");
51      }
52  
53      @Override
54      public byte[] getEdid() {
55          return Arrays.copyOf(edid, edid.length);
56      }
57  
58      @Override
59      public JsonObject toJSON() {
60          return NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder()).add("edid", EdidUtil.toString(getEdid())).build();
61      }
62  
63      /**
64       * Gets Display Information
65       * 
66       * @return An array of Display objects representing monitors, etc.
67       */
68      public static Display[] getDisplays() {
69          List<Display> displays = new ArrayList<Display>();
70          ArrayList<String> ioReg = ExecutingCommand.runNative("ioreg -lw0 -r -c IODisplayConnect");
71          for (String s : ioReg) {
72              if (s.contains("IODisplayEDID")) {
73                  String edidStr = s.substring(s.indexOf("<") + 1, s.indexOf(">"));
74                  LOG.debug("Parsed EDID: {}", edidStr);
75                  Display display = new MacDisplay(ParseUtil.hexStringToByteArray(edidStr));
76                  displays.add(display);
77              }
78          }
79  
80          return displays.toArray(new Display[displays.size()]);
81      }
82  }
