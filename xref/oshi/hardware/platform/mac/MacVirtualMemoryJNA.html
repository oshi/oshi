5   package oshi.hardware.platform.mac;
6   
7   import com.sun.jna.Native;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.jna.ByRef.CloseableIntByReference;
12  import oshi.jna.Struct.CloseableVMStatistics;
13  import oshi.jna.Struct.CloseableXswUsage;
14  import oshi.jna.platform.mac.SystemB;
15  import oshi.util.ParseUtil;
16  import oshi.util.platform.mac.SysctlUtil;
17  import oshi.util.tuples.Pair;
18  
19  @ThreadSafe
20  final class MacVirtualMemoryJNA extends MacVirtualMemory {
21      private static final Logger LOG = LoggerFactory.getLogger(MacVirtualMemoryJNA.class);
22  
23      MacVirtualMemoryJNA(MacGlobalMemoryJNA macGlobalMemory) {
24          super(macGlobalMemory);
25      }
26  
27      @Override
28      protected Pair<Long, Long> querySwapUsage() {
29          long swapUsed = 0L;
30          long swapTotal = 0L;
31          try (CloseableXswUsage xswUsage = new CloseableXswUsage()) {
32              if (SysctlUtil.sysctl("vm.swapusage", xswUsage)) {
33                  swapUsed = xswUsage.xsu_used;
34                  swapTotal = xswUsage.xsu_total;
35              }
36          }
37          return new Pair<>(swapUsed, swapTotal);
38      }
39  
40      @Override
41      protected Pair<Long, Long> queryVmStat() {
42          long swapPagesIn = 0L;
43          long swapPagesOut = 0L;
44          try (CloseableVMStatistics vmStats = new CloseableVMStatistics();
45                  CloseableIntByReference size = new CloseableIntByReference(vmStats.size() / SystemB.INT_SIZE)) {
46              if (0 == SystemB.INSTANCE.host_statistics(SystemB.INSTANCE.mach_host_self(), SystemB.HOST_VM_INFO, vmStats,
47                      size)) {
48                  swapPagesIn = ParseUtil.unsignedIntToLong(vmStats.pageins);
49                  swapPagesOut = ParseUtil.unsignedIntToLong(vmStats.pageouts);
50              } else {
51                  LOG.error("Failed to get host VM info. Error code: {}", Native.getLastError());
52              }
53          }
54          return new Pair<>(swapPagesIn, swapPagesOut);
55      }
56  
57  }
