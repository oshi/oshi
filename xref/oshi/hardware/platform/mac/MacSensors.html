5   package oshi.hardware.platform.mac;
6   
7   import java.util.Locale;
8   
9   import com.sun.jna.platform.mac.IOKit.IOConnect;
10  
11  import oshi.annotation.concurrent.ThreadSafe;
12  import oshi.hardware.common.AbstractSensors;
13  import oshi.util.platform.mac.SmcUtil;
14  import static oshi.util.platform.mac.SmcUtil.SMC_KEY_CPU_TEMP;
15  import static oshi.util.platform.mac.SmcUtil.SMC_KEY_CPU_VOLTAGE;
16  import static oshi.util.platform.mac.SmcUtil.SMC_KEY_CPU_VOLTAGE_AS;
17  import static oshi.util.platform.mac.SmcUtil.SMC_KEY_FAN_NUM;
18  import static oshi.util.platform.mac.SmcUtil.SMC_KEY_FAN_SPEED;
19  import static oshi.util.platform.mac.SmcUtil.SMC_KEYS_CPU_TEMP_AS;
20  
21  /**
22   * Sensors from SMC
23   */
24  @ThreadSafe
25  final class MacSensors extends AbstractSensors {
26  
27      private volatile int numFans = 0;
28  
29      @Override
30      public double queryCpuTemperature() {
31          IOConnect conn = SmcUtil.smcOpen();
32          if (conn == null) {
33              return 0d;
34          }
35          try {
36              double temp = SmcUtil.smcGetFirstFloat(conn, SMC_KEYS_CPU_TEMP_AS);
37              if (temp <= 0d) {
38                  temp = SmcUtil.smcGetFloat(conn, SMC_KEY_CPU_TEMP);
39              }
40              return temp;
41          } finally {
42              SmcUtil.smcClose(conn);
43          }
44      }
45  
46      @Override
47      public int[] queryFanSpeeds() {
48          IOConnect conn = SmcUtil.smcOpen();
49          if (conn == null) {
50              return new int[this.numFans];
51          }
52          try {
53              int fans = this.numFans;
54              if (fans == 0) {
55                  fans = (int) SmcUtil.smcGetLong(conn, SMC_KEY_FAN_NUM);
56                  this.numFans = fans;
57              }
58              int[] fanSpeeds = new int[fans];
59              for (int i = 0; i < fans; i++) {
60                  fanSpeeds[i] = (int) SmcUtil.smcGetFloat(conn, String.format(Locale.ROOT, SMC_KEY_FAN_SPEED, i));
61              }
62              return fanSpeeds;
63          } finally {
64              SmcUtil.smcClose(conn);
65          }
66      }
67  
68      @Override
69      public double queryCpuVoltage() {
70          IOConnect conn = SmcUtil.smcOpen();
71          if (conn == null) {
72              return 0d;
73          }
74          try {
75              // Apple Silicon: VP0C is flt already in volts, no scaling needed
76              double volts = SmcUtil.smcGetFloat(conn, SMC_KEY_CPU_VOLTAGE_AS);
77              if (volts > 0d) {
78                  return volts;
79              }
80              // Intel: VC0C is FPE2 in millivolts, divide by 1000 to get volts
81              return SmcUtil.smcGetFloat(conn, SMC_KEY_CPU_VOLTAGE) / 1000d;
82          } finally {
83              SmcUtil.smcClose(conn);
84          }
85      }
86  }
