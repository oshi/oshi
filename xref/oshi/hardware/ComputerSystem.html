19  package oshi.hardware;
20  
21  import java.io.Serializable;
22  
23  /**
24   * The ComputerSystem represents the physical hardware, of a computer
25   * system/product and includes BIOS/firmware and a motherboard, logic board,
26   * etc.
27   *
28   * @author SchiTho1 [at] Securiton AG
29   * @author widdis [at] gmail [dot] com
30   */
31  public interface ComputerSystem extends Serializable {
32      /**
33       * Get the computer system manufacturer.
34       *
35       * @return The manufacturer.
36       */
37      String getManufacturer();
38  
39      /**
40       * Get the computer system model.
41       *
42       * @return The model.
43       */
44      String getModel();
45  
46      /**
47       * Get the computer system serial number, if available.
48       *
49       * Performs a best-effort attempt to retrieve a unique serial number from
50       * the computer system. This may originate from the baseboard, BIOS,
51       * processor, hardware UUID, etc.
52       *
53       * This value is provided for information only. Caution should be exercised
54       * if using this result to "fingerprint" a system for licensing or other
55       * purposes, as the result may change based on program permissions or
56       * installation of software packages. Specifically, on Linux and FreeBSD,
57       * this requires either root permissions, or installation of the
58       * (deprecated) HAL library (lshal command). Linux also attempts to read the
59       * dmi/id serial number files in sysfs, which are read-only root by default
60       * but may have permissions altered by the user.
61       *
62       * @return the System Serial Number, if available, otherwise returns
63       *         "unknown"
64       */
65      String getSerialNumber();
66  
67      /**
68       * Get the computer system firmware/BIOS
69       *
70       * @return A {@link Firmware} object for this system
71       */
72      Firmware getFirmware();
73  
74      /**
75       * Get the computer system baseboard/motherboard
76       *
77       * @return A {@link Baseboard} object for this system
78       */
79      Baseboard getBaseboard();
80  
81  }
