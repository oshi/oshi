5   package oshi.hardware.common.platform.linux;
6   
7   import java.util.ArrayList;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map;
11  
12  import oshi.annotation.concurrent.Immutable;
13  import oshi.hardware.UsbDevice;
14  import oshi.hardware.common.AbstractUsbDevice;
15  
16  /**
17   * A Linux USB device. Backends (udev via JNA or FFM, or the native-free sysfs reader) enumerate raw
18   * {@link UdevUsbDevice} attributes; this class builds the controller device tree from them, so all the
19   * parsing/parent-child/sorting logic lives in one place.
20   */
21  @Immutable
22  public final class LinuxUsbDevice extends AbstractUsbDevice {
23  
24      private LinuxUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
25              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
26          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
27      }
28  
29      /**
30       * Builds the USB controller device tree from raw per-device attributes gathered by a backend.
31       *
32       * @param rawDevices the enumerated {@code usb_device} entries, in any order
33       * @return a list of USB controllers, each with its connected-device tree
34       */
35      public static List<UsbDevice> getUsbDevices(List<UdevUsbDevice> rawDevices) {
36          // Build lookup maps keyed by syspath, plus the parent -> children (hub) map and the list of root controllers
37          Map<String, String> nameMap = new HashMap<>();
38          Map<String, String> vendorMap = new HashMap<>();
39          Map<String, String> vendorIdMap = new HashMap<>();
40          Map<String, String> productIdMap = new HashMap<>();
41          Map<String, String> serialMap = new HashMap<>();
42          Map<String, List<String>> hubMap = new HashMap<>();
43          List<String> usbControllers = new ArrayList<>();
44          for (UdevUsbDevice device : rawDevices) {
45              String syspath = device.getSyspath();
46              if (device.getProduct() != null) {
47                  nameMap.put(syspath, device.getProduct());
48              }
49              if (device.getManufacturer() != null) {
50                  vendorMap.put(syspath, device.getManufacturer());
51              }
52              if (device.getVendorId() != null) {
53                  vendorIdMap.put(syspath, device.getVendorId());
54              }
55              if (device.getProductId() != null) {
56                  productIdMap.put(syspath, device.getProductId());
57              }
58              if (device.getSerial() != null) {
59                  serialMap.put(syspath, device.getSerial());
60              }
61              String parent = device.getParentSyspath();
62              if (parent == null) {
63                  usbControllers.add(syspath);
64              } else {
65                  hubMap.computeIfAbsent(parent, x -> new ArrayList<>()).add(syspath);
66              }
67          }
68  
69          List<UsbDevice> controllerDevices = new ArrayList<>();
70          for (String controller : usbControllers) {
71              controllerDevices.add(buildDeviceTree(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
72                      productIdMap, serialMap, hubMap, LinuxUsbDevice::new));
73          }
74          return controllerDevices;
75      }
76  
77  }
