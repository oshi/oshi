21  /**
22   * Native-free Linux disk store implementation. Enumerates block devices from {@code /sys/block/} and reads stats from
23   * sysfs, without requiring udev.
24   */
25  @ThreadSafe
26  public final class LinuxHWDiskStoreNF extends LinuxHWDiskStore {
27  
28      private static final String SYS_BLOCK = SysPath.SYS + "block/";
29  
30      LinuxHWDiskStoreNF(String name, String model, String serial, long size, String diskType) {
31          super(name, model, serial, size, diskType);
32      }
33  
34      /**
35       * Gets disk stores on this machine.
36       *
37       * @return a list of {@link HWDiskStore} objects
38       */
39      public static List<HWDiskStore> getDisks() {
40          List<HWDiskStore> result = new ArrayList<>();
41          Map<String, String> mountsMap = readMountsMap();
42  
43          File[] blockDevices = new File(SYS_BLOCK).listFiles();
44          if (blockDevices == null) {
45              return result;
46          }
47          for (File blockDev : blockDevices) {
48              String name = blockDev.getName();
49              String sysPath = SYS_BLOCK + name + "/";
50  
51              // Filter to real disks (skip loop, ram, etc.)
52              String devType = FileUtil.getStringFromFile(sysPath + "device/type").trim();
53              // No device/type means it's a virtual device; check if it has a device/ subdir
54              if (devType.isEmpty() && !new File(sysPath + "device").exists()) {
55                  continue;
56              }
57  
58              String model = FileUtil.getStringFromFile(sysPath + "device/model").trim();
59              String serial = FileUtil.getStringFromFile(sysPath + "device/serial").trim();
60              long size = ParseUtil.parseLongOrDefault(FileUtil.getStringFromFile(sysPath + SIZE).trim(), 0L)
61                      * SECTORSIZE;
62              String diskType = detectDiskType(sysPath);
63  
64              LinuxHWDiskStoreNF store = new LinuxHWDiskStoreNF(DevPath.DEV + name, model, serial, size, diskType);
65  
66              // Read stats
67              String statStr = FileUtil.getStringFromFile(sysPath + STAT).trim();
68              if (!statStr.isEmpty()) {
69                  computeDiskStats(store, statStr);
70              }
71  
72              // Enumerate partitions
73              File[] partDirs = blockDev.listFiles(f -> f.isDirectory() && f.getName().startsWith(name));
74              if (partDirs != null) {
75                  for (File partDir : partDirs) {
76                      String partName = partDir.getName();
77                      String partPath = sysPath + partName + "/";
78                      long partSize = ParseUtil.parseLongOrDefault(FileUtil.getStringFromFile(partPath + SIZE).trim(), 0L)
79                              * SECTORSIZE;
80                      String devStr = FileUtil.getStringFromFile(partPath + "dev").trim();
81                      int major = 0;
82                      int minor = 0;
83                      if (devStr.contains(":")) {
84                          String[] majMin = devStr.split(":");
85                          major = ParseUtil.parseIntOrDefault(majMin[0], 0);
86                          minor = ParseUtil.parseIntOrDefault(majMin[1], 0);
87                      }
88                      String mountPoint = mountsMap.getOrDefault(DevPath.DEV + partName, "");
89                      store.getMutablePartitionList()
90                              .add(new HWPartition(partName, partName, "", "", "", partSize, major, minor, mountPoint));
91                  }
92              }
93  
94              result.add(store);
95          }
96          finalizePartitions(result);
97          return result;
98      }
99  
100     /**
101      * Detects the disk type from sysfs attributes, mirroring the udev-based backends' {@code detectDiskType}.
102      *
103      * @param sysPath the {@code /sys/block/<device>/} path for the device
104      * @return {@code "Removable"}, {@code "SSD"}, {@code "HDD"}, or {@code "Unknown"}
105      */
106     private static String detectDiskType(String sysPath) {
107         String removable = FileUtil.getStringFromFile(sysPath + "removable").trim();
108         if ("1".equals(removable)) {
109             return "Removable";
110         }
111         String rotational = FileUtil.getStringFromFile(sysPath + "queue/rotational").trim();
112         if ("0".equals(rotational)) {
113             return "SSD";
114         } else if ("1".equals(rotational)) {
115             return "HDD";
116         }
117         return "Unknown";
118     }
119 
120     @Override
121     public boolean updateAttributes() {
122         String devName = getName().replace(DevPath.DEV, "");
123         String statStr = FileUtil.getStringFromFile(SYS_BLOCK + devName + "/" + STAT).trim();
124         if (statStr.isEmpty()) {
125             return false;
126         }
127         computeDiskStats(this, statStr);
128         return true;
129     }
130 }
