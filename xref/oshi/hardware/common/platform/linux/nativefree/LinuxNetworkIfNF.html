5   package oshi.hardware.common.platform.linux.nativefree;
6   
7   import java.net.NetworkInterface;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.hardware.NetworkIF;
16  import oshi.hardware.common.platform.linux.LinuxNetworkIF;
17  
18  /**
19   * Native-free Linux network interface implementation. Uses sysfs for model identification.
20   */
21  @ThreadSafe
22  public final class LinuxNetworkIfNF extends LinuxNetworkIF {
23  
24      private static final Logger LOG = LoggerFactory.getLogger(LinuxNetworkIfNF.class);
25  
26      LinuxNetworkIfNF(NetworkInterface netint) throws InstantiationException {
27          super(netint, queryIfModelFromSysfs(netint.getName()));
28      }
29  
30      /**
31       * Gets network interfaces on this machine.
32       *
33       * @param includeLocalInterfaces include local interfaces in the result
34       * @return a list of {@link NetworkIF} objects
35       */
36      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
37          List<NetworkIF> ifList = new ArrayList<>();
38          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
39              try {
40                  ifList.add(new LinuxNetworkIfNF(ni));
41              } catch (InstantiationException e) {
42                  LOG.debug("Network Interface Instantiation failed: {}", e.toString());
43              }
44          }
45          return ifList;
46      }
47  }
