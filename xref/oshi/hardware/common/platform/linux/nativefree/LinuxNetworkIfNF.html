5   package oshi.hardware.common.platform.linux.nativefree;
6   
7   import java.net.NetworkInterface;
8   import java.util.List;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.hardware.NetworkIF;
12  import oshi.hardware.common.platform.linux.LinuxNetworkIF;
13  
14  /**
15   * Native-free Linux network interface implementation. Uses sysfs for model identification.
16   */
17  @ThreadSafe
18  public final class LinuxNetworkIfNF extends LinuxNetworkIF {
19  
20      LinuxNetworkIfNF(NetworkInterface netint) throws InstantiationException {
21          super(netint, queryIfModelFromSysfs(netint.getName()));
22      }
23  
24      /**
25       * Gets network interfaces on this machine.
26       *
27       * @param includeLocalInterfaces include local interfaces in the result
28       * @return a list of {@link NetworkIF} objects
29       */
30      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
31          return getNetworks(includeLocalInterfaces, LinuxNetworkIfNF::new);
32      }
33  }
