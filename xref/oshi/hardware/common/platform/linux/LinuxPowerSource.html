87      /**
88       * Initialization-on-demand holder: the map is only constructed if the non-udev path is reached, avoiding
89       * unnecessary work on systems where udev is available.
90       */
91      private static final class PropByName {
92          static final Map<String, Prop> MAP = new HashMap<>();
93          static {
94              for (Prop p : Prop.values()) {
95                  MAP.put(p.name(), p);
96              }
97          }
98      }

165     /**
166      * Gets Battery Information from the specified power supply path.
167      *
168      * @param powerSupplyPath The path to the power supply directory.
169      * @return An array of PowerSource objects representing batteries, etc.
170      */
171     static List<PowerSource> getPowerSources(String powerSupplyPath) {
172         List<PowerSource> psList = new ArrayList<>();
173         File psDir = new File(powerSupplyPath);
174         File[] psArr = psDir.listFiles();
175         if (psArr == null) {
176             return Collections.emptyList();
177         }
178         for (File ps : psArr) {
179             String name = ps.getName();
180             if (!name.startsWith("ADP") && !name.startsWith("AC") && !name.contains("USBC")) {
181                 List<String> psInfo = FileUtil.readFile(powerSupplyPath + "/" + name + "/uevent", false);
182                 Map<Prop, String> props = new EnumMap<>(Prop.class);
183                 for (String line : psInfo) {
184                     String[] split = line.split("=", 2);
185                     if (split.length > 1 && !split[1].isEmpty()) {
186                         Prop p = PropByName.MAP.get(split[0]);
187                         if (p != null) {
188                             props.put(p, split[1]);
189                         }
190                     }
191                 }
192                 if (ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_PRESENT), 1) > 0) {
193                     psList.add(buildPowerSource(name, props));
194                 }
195             }
196         }
197         return psList;
198     }

200     /**
201      * Builds a LinuxPowerSource from parsed uevent properties.
202      *
203      * @param name  the power supply name
204      * @param props the parsed properties map
205      * @return a new LinuxPowerSource instance
206      */
207     protected static LinuxPowerSource buildPowerSource(String name, Map<Prop, String> props) {
208         String psName = props.getOrDefault(Prop.POWER_SUPPLY_NAME, name);
209         String status = props.get(Prop.POWER_SUPPLY_STATUS);
210         boolean psCharging = "Charging".equals(status);
211         boolean psDischarging = "Discharging".equals(status);
212         double psRemainingCapacityPercent = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CAPACITY), -100)
213                 / 100d;
214 
215         // Debian/Ubuntu provides Energy (µWh). Fedora/RHEL provides Charge (µAh).
216         // Parse raw values in micro-units first, fall back across Energy/Charge, then
217         // divide by 1000 at the end to avoid integer division truncating the -1 sentinel.
218         int rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_NOW), -1);
219         CapacityUnits psCapacityUnits;
220         if (rawNow >= 0) {
221             psCapacityUnits = CapacityUnits.MWH;
222         } else {
223             rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_NOW), -1);
224             psCapacityUnits = CapacityUnits.MAH;
225         }
226         int psCurrentCapacity = rawNow >= 0 ? rawNow / 1000 : -1;
227 
228         int rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL), -1);
229         if (rawFull < 0) {
230             rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL), -1);
231         }
232         int psMaxCapacity = rawFull >= 0 ? rawFull / 1000 : -1;
233 
234         int rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL_DESIGN), -1);
235         if (rawDesign < 0) {
236             rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL_DESIGN), -1);
237         }
238         int psDesignCapacity = rawDesign >= 0 ? rawDesign / 1000 : -1;
239 
240         // Debian/Ubuntu provides Voltage and Power. Fedora/RHEL provides Voltage and Current.
241         double psVoltage = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_VOLTAGE_NOW), -1d) / 1_000_000d;
242         double psPowerUsageRate = 0d;
243         double psAmperage = 0d;
244         // From Physics we know P = IV so I = P/V
245         // Linux always reports POWER_NOW and CURRENT_NOW as non-negative; sign is
246         // conveyed by STATUS. Negate when discharging to match the PowerSource contract
247         // (positive = charging, negative = discharging).
248         double sign = psDischarging ? -1d : 1d;
249         if (psVoltage > 0) {
250             double rawPower = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_POWER_NOW), -1d);
251             double rawCurrent = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_CURRENT_NOW), -1d);
252             if (rawPower >= 0) {
253                 psPowerUsageRate = sign * rawPower / 1000d;
254             }
255             if (rawCurrent >= 0) {
256                 psAmperage = sign * rawCurrent / 1000d;
257             }
258             if (rawPower < 0 && rawCurrent >= 0) {
259                 psPowerUsageRate = psAmperage * psVoltage;
260             } else if (rawCurrent < 0 && rawPower >= 0) {
261                 psAmperage = psPowerUsageRate / psVoltage;
262             }
263         }
264 
265         int psCycleCount = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CYCLE_COUNT), -1);
266         String psChemistry = props.getOrDefault(Prop.POWER_SUPPLY_TECHNOLOGY, UNKNOWN);
267         String psDeviceName = props.getOrDefault(Prop.POWER_SUPPLY_MODEL_NAME, UNKNOWN);
268         String psManufacturer = props.getOrDefault(Prop.POWER_SUPPLY_MANUFACTURER, UNKNOWN);
269         String psSerialNumber = props.getOrDefault(Prop.POWER_SUPPLY_SERIAL_NUMBER, UNKNOWN);
270 
271         // TEMP is in tenths of degrees Celsius
272         int rawTemp = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_TEMP), -1);
273         double psTemperature = rawTemp >= 0 ? rawTemp / 10d : 0d;
274 
275         // TIME_TO_EMPTY_NOW when discharging, TIME_TO_FULL_NOW when charging; both in seconds
276         Prop timeKey = psCharging ? Prop.POWER_SUPPLY_TIME_TO_FULL_NOW : Prop.POWER_SUPPLY_TIME_TO_EMPTY_NOW;
277         double psTimeRemainingInstant = ParseUtil.parseIntOrDefault(props.get(timeKey), -1);
278 
279         int year = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_YEAR), -1);
280         int month = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_MONTH), -1);
281         int day = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_DAY), -1);
282         LocalDate psManufactureDate = (year > 0 && month > 0 && day > 0) ? LocalDate.of(year, month, day) : null;
283 
284         return new LinuxPowerSource(psName, psDeviceName, psRemainingCapacityPercent, -1d, psTimeRemainingInstant,
285                 psPowerUsageRate, psVoltage, psAmperage, false, psCharging, psDischarging, psCapacityUnits,
286                 psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
287                 psManufacturer, psSerialNumber, psTemperature);
288     }
