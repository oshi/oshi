66      /**
67       * Checks whether the OS version is at least the given Windows {@code major.minor} using system property
68       * 'os.version', without requiring native access.
69       *
70       * @param major the minimum major version
71       * @param minor the minimum minor version
72       * @return true if the OS version is at least {@code major.minor}
73       */
74      private static boolean isWindowsVersionOrGreater(int major, int minor) {
75          String[] parts = System.getProperty("os.version", "").split("\\.");
76          if (parts.length >= 2) {
77              int osMajor = ParseUtil.parseIntOrDefault(parts[0], 0);
78              int osMinor = ParseUtil.parseIntOrDefault(parts[1], 0);
79              return osMajor > major || (osMajor == major && osMinor >= minor);
80          }
81          return false;
82      }

131     /**
132      * Builds the numaNodeProcToLogicalProcMap from the logical processor list.
133      *
134      * @param logProcs the list of logical processors
135      */
136     protected void buildNumaNodeProcMap(List<LogicalProcessor> logProcs) {
137         Map<Integer, Integer> nextProcIndexByNode = new HashMap<>();
138         int lp = 0;
139         this.numaNodeProcToLogicalProcMap = new HashMap<>();
140         for (LogicalProcessor logProc : logProcs) {
141             int node = logProc.getNumaNode();
142             int procNum = nextProcIndexByNode.getOrDefault(node, 0);
143             numaNodeProcToLogicalProcMap.put(String.format(Locale.ROOT, "%d,%d", node, procNum), lp++);
144             nextProcIndexByNode.put(node, procNum + 1);
145         }
146     }

148     /**
149      * Parses identifier string
150      *
151      * @param identifier the full identifier string
152      * @param key        the key to retrieve
153      * @return the string following id
154      */
155     protected static String parseIdentifier(String identifier, String key) {
156         String[] idSplit = ParseUtil.whitespaces.split(identifier);
157         boolean found = false;
158         for (String s : idSplit) {
159             if (found) {
160                 return s;
161             }
162             found = s.equals(key);
163         }
164         return "";
165     }

167     /**
168      * Lazily calculate the capacity tick multiplier once.
169      *
170      * @param deltaBase The difference in base ticks.
171      * @param deltaT    The difference in elapsed 100NS time
172      * @return The ratio of elapsed time to base ticks
173      */
174     protected synchronized long lazilyCalculateMultiplier(long deltaBase, long deltaT) {
175         if (utilityBaseMultiplier == null) {
176             // If too much time has elapsed from class instantiation, re-initialize the
177             // ticks and return without calculating. Approx 7 minutes for 100NS counter to
178             // exceed max unsigned int.
179             if (deltaT >> 32 > 0) {
180                 setInitialUtilityCounters(queryProcessorUtilityCounters());
181                 return 0L;
182             }
183             // Base counter wraps approximately every 115 minutes
184             // If deltaBase is nonpositive assume it has wrapped
185             if (deltaBase <= 0) {
186                 deltaBase += 1L << 32;
187             }
188             long multiplier = Math.round((double) deltaT / deltaBase);
189             // If not enough time has elapsed, return the value this one time but don't
190             // persist. 5000 ms = 50 million 100NS ticks
191             if (deltaT < 50_000_000L) {
192                 return multiplier;
193             }
194             utilityBaseMultiplier = multiplier;
195         }
196         return utilityBaseMultiplier;
197     }

254     @Override
255     public long[] queryCurrentFreq() {
256         if (isWindows7OrGreater()) {
257             long maxFreq = this.getMaxFreq();
258             if (maxFreq > 0) {
259                 // Prefer % Processor Performance from WMI formatted table (Win8+, reports >100% with turbo)
260                 Pair<List<String>, Map<ProcessorPerformanceProperty, List<Long>>> perfPair = queryProcessorPerformanceCounters();
261                 List<Long> perfList = perfPair.getB().get(ProcessorPerformanceProperty.PERCENTPROCESSORPERFORMANCE);
262                 long[] freqs = mapPercentToFreqs(perfPair.getA(), perfList, maxFreq);
263                 if (freqs != null) {
264                     return freqs;
265                 }
266                 // Fall back to % of Maximum Frequency (Win7, caps at 100%)
267                 Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> freqPair = queryFrequencyCounters();
268                 List<Long> percentMaxList = freqPair.getB().get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
269                 freqs = mapPercentToFreqs(freqPair.getA(), percentMaxList, maxFreq);
270                 if (freqs != null) {
271                     return freqs;
272                 }
273             }
274         }
275         // If <Win7 or anything failed in PDH/WMI, use the native call
276         return queryNTPower(2); // Current is field index 2
277     }

279     private long[] mapPercentToFreqs(List<String> instances, List<Long> percentList, long maxFreq) {
280         if (instances.isEmpty() || percentList == null) {
281             return null;
282         }
283         long[] freqs = new long[getLogicalProcessorCount()];
284         boolean populated = false;
285         // instances and percentList are parallel lists; read by position i and map the instance name to the logical
286         // processor index the same way processTickData does (numa "group,proc" via the map, else the plain number).
287         for (int i = 0; i < instances.size(); i++) {
288             String instance = instances.get(i);
289             int cpu;
290             if (instance.contains(",")) {
291                 if (!getNumaNodeProcToLogicalProcMap().containsKey(instance)) {
292                     continue;
293                 }
294                 cpu = getNumaNodeProcToLogicalProcMap().get(instance);
295             } else {
296                 cpu = ParseUtil.parseIntOrDefault(instance, -1);
297             }
298             if (cpu < 0 || cpu >= getLogicalProcessorCount() || i >= percentList.size()) {
299                 continue;
300             }
301             freqs[cpu] = percentList.get(i) * maxFreq / 100L;
302             if (freqs[cpu] > 0) {
303                 populated = true;
304             }
305         }
306         return populated ? freqs : null;
307     }

309     @Override
310     public long queryMaxFreq() {
311         long[] freqs = queryNTPower(1); // Max is field index 1
312         return Arrays.stream(freqs).max().orElse(-1L);
313     }

315     @Override
316     public long[][] queryProcessorCpuLoadTicks() {
317         // These are used in all cases
318         List<String> instances;
319         List<Long> systemList;
320         List<Long> userList;
321         List<Long> irqList;
322         List<Long> softIrqList;
323         List<Long> idleList;
324         // These are only used with USE_CPU_UTILITY
325         List<Long> baseList = null;
326         List<Long> systemUtility = null;
327         List<Long> processorUtility = null;
328         List<Long> processorUtilityBase = null;
329         List<Long> initSystemList = null;
330         List<Long> initUserList = null;
331         List<Long> initBase = null;
332         List<Long> initSystemUtility = null;
333         List<Long> initProcessorUtility = null;
334         List<Long> initProcessorUtilityBase = null;
335         if (USE_CPU_UTILITY) {
336             Pair<List<String>, Map<ProcessorUtilityTickCountProperty, List<Long>>> instanceValuePair = processorUtilityCounters
337                     .get();
338             instances = instanceValuePair.getA();
339             Map<ProcessorUtilityTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
340             systemList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
341             userList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
342             irqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTINTERRUPTTIME);
343             softIrqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTDPCTIME);
344             // % Processor Time is actually Idle time
345             idleList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORTIME);
346             baseList = valueMap.get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
347             // Utility ticks, if configured
348             systemUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
349             processorUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
350             processorUtilityBase = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
351 
352             initSystemList = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
353             initUserList = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
354             initBase = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
355             initSystemUtility = getInitialUtilityCounters()
356                     .get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
357             initProcessorUtility = getInitialUtilityCounters()
358                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
359             initProcessorUtilityBase = getInitialUtilityCounters()
360                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
361         } else {
362             Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = queryProcessorCounters();
363             instances = instanceValuePair.getA();
364             Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
365             systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
366             userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
367             irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
368             softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
369             // % Processor Time is actually Idle time
370             idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
371         }
372 
373         return processTickData(instances, systemList, userList, irqList, softIrqList, idleList, baseList, systemUtility,
374                 processorUtility, processorUtilityBase, initSystemList, initUserList, initBase, initSystemUtility,
375                 initProcessorUtility, initProcessorUtilityBase);
376     }

378     /**
379      * Processes raw perfmon tick data into the standard tick array format. Handles both legacy and utility-based
380      * approaches.
381      *
382      * @param instances                the perfmon instance names
383      * @param systemList               system (privileged) time ticks
384      * @param userList                 user time ticks
385      * @param irqList                  IRQ time ticks
386      * @param softIrqList              soft IRQ (DPC) time ticks
387      * @param idleList                 idle time ticks
388      * @param baseList                 timestamp base (for utility mode), may be null
389      * @param systemUtility            system utility ticks, may be null
390      * @param processorUtility         processor utility ticks, may be null
391      * @param processorUtilityBase     processor utility base, may be null
392      * @param initSystemList           initial system ticks (for utility mode), may be null
393      * @param initUserList             initial user ticks (for utility mode), may be null
394      * @param initBase                 initial base (for utility mode), may be null
395      * @param initSystemUtility        initial system utility, may be null
396      * @param initProcessorUtility     initial processor utility, may be null
397      * @param initProcessorUtilityBase initial processor utility base, may be null
398      * @return the processed tick array
399      */
400     protected long[][] processTickData(List<String> instances, List<Long> systemList, List<Long> userList,
401             List<Long> irqList, List<Long> softIrqList, List<Long> idleList, List<Long> baseList,
402             List<Long> systemUtility, List<Long> processorUtility, List<Long> processorUtilityBase,
403             List<Long> initSystemList, List<Long> initUserList, List<Long> initBase, List<Long> initSystemUtility,
404             List<Long> initProcessorUtility, List<Long> initProcessorUtilityBase) {
405 
406         int ncpu = getLogicalProcessorCount();
407         long[][] ticks = new long[ncpu][TickType.values().length];
408         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
409                 || idleList == null
410                 || (USE_CPU_UTILITY && (baseList == null || systemUtility == null || processorUtility == null
411                         || processorUtilityBase == null || initSystemList == null || initUserList == null
412                         || initBase == null || initSystemUtility == null || initProcessorUtility == null
413                         || initProcessorUtilityBase == null))) {
414             return ticks;
415         }
416         int size = instances.size();
417         if (systemList.size() < size || userList.size() < size || irqList.size() < size || softIrqList.size() < size
418                 || idleList.size() < size) {
419             return ticks;
420         }
421         if (USE_CPU_UTILITY && (baseList.size() < size || systemUtility.size() < size || processorUtility.size() < size
422                 || processorUtilityBase.size() < size || initSystemList.size() < size || initUserList.size() < size
423                 || initBase.size() < size || initSystemUtility.size() < size || initProcessorUtility.size() < size
424                 || initProcessorUtilityBase.size() < size)) {
425             return ticks;
426         }
427         for (int i = 0; i < instances.size(); i++) {
428             String instance = instances.get(i);
429             int cpu;
430             if (instance.contains(",")) {
431                 if (!getNumaNodeProcToLogicalProcMap().containsKey(instance)) {
432                     continue;
433                 }
434                 cpu = getNumaNodeProcToLogicalProcMap().get(instance);
435             } else {
436                 cpu = ParseUtil.parseIntOrDefault(instance, -1);
437             }
438             if (cpu < 0 || cpu >= ncpu) {
439                 continue;
440             }
441             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(i);
442             ticks[cpu][TickType.USER.getIndex()] = userList.get(i);
443             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(i);
444             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(i);
445             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(i);
446 
447             if (USE_CPU_UTILITY) {
448                 long deltaT = baseList.get(i) - initBase.get(i);
449                 if (deltaT > 0) {
450                     long deltaBase = processorUtilityBase.get(i) - initProcessorUtilityBase.get(i);
451                     long multiplier = lazilyCalculateMultiplier(deltaBase, deltaT);
452                     if (multiplier > 0) {
453                         long deltaProc = processorUtility.get(i) - initProcessorUtility.get(i);
454                         long deltaSys = systemUtility.get(i) - initSystemUtility.get(i);
455                         long newUser = initUserList.get(i) + multiplier * (deltaProc - deltaSys) / 100;
456                         long newSystem = initSystemList.get(i) + multiplier * deltaSys / 100;
457                         long delta = newUser - ticks[cpu][TickType.USER.getIndex()];
458                         ticks[cpu][TickType.USER.getIndex()] = newUser;
459                         delta += newSystem - ticks[cpu][TickType.SYSTEM.getIndex()];
460                         ticks[cpu][TickType.SYSTEM.getIndex()] = newSystem;
461                         ticks[cpu][TickType.IDLE.getIndex()] -= delta;
462                     }
463                 }
464             }
465 
466             // Decrement IRQ from system to avoid double counting in the total array
467             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
468                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
469 
470             // Raw value is cumulative 100NS-ticks
471             // Divide by 10_000 to get milliseconds
472             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
473             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
474             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
475             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
476             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
477         }
478         return ticks;
479     }
