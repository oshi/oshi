5   package oshi.hardware.common.platform.mac;
6   
7   import java.net.NetworkInterface;
8   import java.util.Map;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.driver.common.mac.IFdata;
12  import oshi.hardware.common.AbstractNetworkIF;
13  
14  /**
15   * Base class for macOS NetworkIF implementations. The per-interface statistics come from a shared {@link IFdata} map;
16   * subclasses provide the native display-name resolution and the {@code IFdata} query (JNA or FFM).
17   */
18  @ThreadSafe
19  public abstract class MacNetworkIF extends AbstractNetworkIF {
20  
21      private volatile int ifType;
22  
23      /**
24       * Creates a MacNetworkIF.
25       *
26       * @param netint      the network interface
27       * @param displayName the display name
28       * @throws InstantiationException if the interface cannot be instantiated
29       */
30      protected MacNetworkIF(NetworkInterface netint, String displayName) throws InstantiationException {
31          super(netint, displayName);
32      }
33  
34      @Override
35      public int getIfType() {
36          return this.ifType;
37      }
38  
39      @Override
40      public boolean updateAttributes() {
41          return updateNetworkStats(queryIFdata(queryNetworkInterface().getIndex()));
42      }
43  
44      /**
45       * Populates this interface's statistics from the given {@link IFdata} map.
46       *
47       * @param data map of interface index to {@link IFdata}
48       * @return {@code true} if this interface's index was present in the map
49       */
50      protected boolean updateNetworkStats(Map<Integer, IFdata> data) {
51          int index = queryNetworkInterface().getIndex();
52          if (data.containsKey(index)) {
53              IFdata ifData = data.get(index);
54              this.ifType = ifData.getIfType();
55              this.bytesSent = ifData.getOBytes();
56              this.bytesRecv = ifData.getIBytes();
57              this.packetsSent = ifData.getOPackets();
58              this.packetsRecv = ifData.getIPackets();
59              this.outErrors = ifData.getOErrors();
60              this.inErrors = ifData.getIErrors();
61              this.collisions = ifData.getCollisions();
62              this.inDrops = ifData.getIDrops();
63              this.speed = ifData.getSpeed();
64              this.timeStamp = ifData.getTimeStamp();
65              return true;
66          }
67          return false;
68      }
69  
70      /**
71       * Queries interface statistics via the platform-specific (JNA or FFM) {@code getifaddrs} backend.
72       *
73       * @param index the interface index, or {@code -1} for all interfaces
74       * @return map of interface index to {@link IFdata}
75       */
76      protected abstract Map<Integer, IFdata> queryIFdata(int index);
77  }
