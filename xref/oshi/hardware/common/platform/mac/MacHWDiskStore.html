5   package oshi.hardware.common.platform.mac;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.hardware.common.AbstractHWDiskStore;
9   
10  /**
11   * Base class for macOS HWDiskStore implementations. Subclasses provide platform-specific disk enumeration and
12   * statistics updates.
13   */
14  @ThreadSafe
15  public abstract class MacHWDiskStore extends AbstractHWDiskStore {
16  
17      /**
18       * Creates a MacHWDiskStore with unknown disk type.
19       *
20       * @param name   the disk name
21       * @param model  the disk model
22       * @param serial the serial number
23       * @param size   the disk size in bytes
24       */
25      protected MacHWDiskStore(String name, String model, String serial, long size) {
26          super(name, model, serial, size);
27      }
28  
29      /**
30       * Creates a MacHWDiskStore.
31       *
32       * @param name     the disk name
33       * @param model    the disk model
34       * @param serial   the serial number
35       * @param size     the disk size in bytes
36       * @param diskType the disk type
37       */
38      protected MacHWDiskStore(String name, String model, String serial, long size, String diskType) {
39          super(name, model, serial, size, diskType);
40      }
41  
42      /**
43       * Strings to convert to CFStringRef for pointer lookups.
44       */
45      protected enum CFKey {
46          /** IOPropertyMatch key. */
47          IO_PROPERTY_MATCH("IOPropertyMatch"),
48          /** Statistics dictionary key. */
49          STATISTICS("Statistics"),
50          /** Read operations count. */
51          READ_OPS("Operations (Read)"),
52          /** Read bytes count. */
53          READ_BYTES("Bytes (Read)"),
54          /** Read time. */
55          READ_TIME("Total Time (Read)"),
56          /** Write operations count. */
57          WRITE_OPS("Operations (Write)"),
58          /** Write bytes count. */
59          WRITE_BYTES("Bytes (Write)"),
60          /** Write time. */
61          WRITE_TIME("Total Time (Write)"),
62          /** BSD unit number. */
63          BSD_UNIT("BSD Unit"),
64          /** Leaf node indicator. */
65          LEAF("Leaf"),
66          /** Whole disk indicator. */
67          WHOLE("Whole"),
68          /** DiskArbitration media name. */
69          DA_MEDIA_NAME("DAMediaName"),
70          /** DiskArbitration volume name. */
71          DA_VOLUME_NAME("DAVolumeName"),
72          /** DiskArbitration media size. */
73          DA_MEDIA_SIZE("DAMediaSize"),
74          /** DiskArbitration device model. */
75          DA_DEVICE_MODEL("DADeviceModel"),
76          /** IOKit model property. */
77          MODEL("Model");
78  
79          private final String key;
80  
81          CFKey(String key) {
82              this.key = key;
83          }
84  
85          /**
86           * Gets the CoreFoundation key string.
87           *
88           * @return the key string
89           */
90          public String getKey() {
91              return this.key;
92          }
93      }
94  }
