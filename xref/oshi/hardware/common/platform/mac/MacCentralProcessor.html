88      @Override
89      protected ProcessorIdentifier queryProcessorId() {
90          String cpuName = sysctlString("machdep.cpu.brand_string", "");
91          String cpuVendor;
92          String cpuStepping;
93          String cpuModel;
94          String cpuFamily;
95          String processorID;
96          if (cpuName.startsWith("Apple")) {
97              cpuVendor = vendor.get();
98              cpuStepping = "0";
99              cpuModel = "0";
100             int type;
101             int family;
102             if (isArmCpu) {
103                 type = ARM_CPUTYPE;
104                 family = sysctlInt("hw.cpufamily", 0);
105                 if (family == 0) {
106                     int mSeries = ParseUtil.getFirstIntValue(cpuName);
107                     switch (mSeries) {
108                         case 2:
109                             family = M2_CPUFAMILY;
110                             break;
111                         case 3:
112                             family = M3_CPUFAMILY;
113                             break;
114                         default:
115                             family = M1_CPUFAMILY;
116                     }
117                 }
118             } else {
119                 type = sysctlInt("hw.cputype", 0);
120                 family = sysctlInt("hw.cpufamily", 0);
121             }
122             cpuFamily = String.format(Locale.ROOT, "0x%08x", family);
123             processorID = String.format(Locale.ROOT, "%08x%08x", type, family);
124         } else {
125             cpuVendor = sysctlString("machdep.cpu.vendor", "");
126             int i = sysctlInt("machdep.cpu.stepping", -1);
127             cpuStepping = i < 0 ? "" : Integer.toString(i);
128             i = sysctlInt("machdep.cpu.model", -1);
129             cpuModel = i < 0 ? "" : Integer.toString(i);
130             i = sysctlInt("machdep.cpu.family", -1);
131             cpuFamily = i < 0 ? "" : Integer.toString(i);
132             long processorIdBits = 0L;
133             processorIdBits |= sysctlInt("machdep.cpu.signature", 0);
134             processorIdBits |= (sysctlLong("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
135             processorID = String.format(Locale.ROOT, "%016x", processorIdBits);
136         }
137         if (isArmCpu) {
138             calculateNominalFrequencies();
139         }
140         long cpuFreq = isArmCpu ? performanceCoreFrequency : sysctlLong("hw.cpufrequency", 0L);
141         boolean cpu64bit = sysctlInt("hw.cpu64bit_capable", 0) != 0;
142 
143         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
144                 cpuFreq);
145     }

147     @Override
148     protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
149         int logicalProcessorCount = sysctlInt("hw.logicalcpu", 1);
150         int physicalProcessorCount = sysctlInt("hw.physicalcpu", 1);
151         int physicalPackageCount = sysctlInt("hw.packages", 1);
152         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
153         Set<Integer> pkgCoreKeys = new HashSet<>();
154         for (int i = 0; i < logicalProcessorCount; i++) {
155             int coreId = i * physicalProcessorCount / logicalProcessorCount;
156             int pkgId = i * physicalPackageCount / logicalProcessorCount;
157             logProcs.add(new LogicalProcessor(i, coreId, pkgId));
158             pkgCoreKeys.add((pkgId << 16) + coreId);
159         }
160         Map<Integer, String> compatMap = queryCompatibleStrings();
161         int perflevels = sysctlIntNoWarn("hw.nperflevels", 1);
162         List<PhysicalProcessor> physProcs = pkgCoreKeys.stream().sorted().map(k -> {
163             String compat = compatMap.getOrDefault(k, "").toLowerCase(Locale.ROOT);
164             int efficiency = ARM_P_CORES.contains(compat) ? 1 : 0;
165             return new PhysicalProcessor(k >> 16, k & 0xffff, efficiency, compat);
166         }).collect(Collectors.toList());
167         List<ProcessorCache> caches = orderedProcCaches(getCacheValues(perflevels));
168         List<String> featureFlags = getFeatureFlagsFromSysctl();
169         return new Quartet<>(logProcs, physProcs, caches, featureFlags);
170     }
171 
172     private Set<ProcessorCache> getCacheValues(int perflevels) {
173         int linesize = (int) sysctlLong("hw.cachelinesize", 0L);
174         int l1associativity = sysctlIntNoWarn("machdep.cpu.cache.L1_associativity", 0);
175         int l2associativity = sysctlIntNoWarn("machdep.cpu.cache.L2_associativity", 0);
176         Set<ProcessorCache> caches = new HashSet<>();
177         for (int i = 0; i < perflevels; i++) {
178             int size = sysctlIntNoWarn("hw.perflevel" + i + ".l1icachesize", 0);
179             if (size > 0) {
180                 caches.add(new ProcessorCache(1, l1associativity, linesize, size, Type.INSTRUCTION));
181             }
182             size = sysctlIntNoWarn("hw.perflevel" + i + ".l1dcachesize", 0);
183             if (size > 0) {
184                 caches.add(new ProcessorCache(1, l1associativity, linesize, size, Type.DATA));
185             }
186             size = sysctlIntNoWarn("hw.perflevel" + i + ".l2cachesize", 0);
187             if (size > 0) {
188                 caches.add(new ProcessorCache(2, l2associativity, linesize, size, Type.UNIFIED));
189             }
190             size = sysctlIntNoWarn("hw.perflevel" + i + ".l3cachesize", 0);
191             if (size > 0) {
192                 caches.add(new ProcessorCache(3, 0, linesize, size, Type.UNIFIED));
193             }
194         }
195         return caches;
196     }

198     private List<String> getFeatureFlagsFromSysctl() {
199         List<String> x86Features = parseX86FeatureFlags();
200         return x86Features.isEmpty() ? ExecutingCommand.runNative("sysctl -a hw.optional") : x86Features;
201     }
202 
203     List<String> parseX86FeatureFlags() {
204         return Stream.of("features", "extfeatures", "leaf7_features").map(f -> {
205             String key = "machdep.cpu." + f;
206             String features = sysctlStringNoWarn(key, "");
207             return Util.isBlank(features) ? null : (key + ": " + features);
208         }).filter(Objects::nonNull).collect(Collectors.toList());
209     }

211     @Override
212     public long[] queryCurrentFreq() {
213         if (isArmCpu) {
214             Map<Integer, Long> physFreqMap = new HashMap<>();
215             getPhysicalProcessors().forEach(p -> physFreqMap.put(p.getPhysicalProcessorNumber(),
216                     p.getEfficiency() > 0 ? performanceCoreFrequency : efficiencyCoreFrequency));
217             return getLogicalProcessors().stream().map(LogicalProcessor::getPhysicalProcessorNumber)
218                     .map(p -> physFreqMap.getOrDefault(p, performanceCoreFrequency)).mapToLong(f -> f).toArray();
219         }
220         return new long[] { getProcessorIdentifier().getVendorFreq() };
221     }
222 
223     @Override
224     public long queryMaxFreq() {
225         if (isArmCpu) {
226             return performanceCoreFrequency;
227         }
228         return sysctlLong("hw.cpufrequency_max", getProcessorIdentifier().getVendorFreq());
229     }

241     private boolean isArmCpu() {
242         return getPhysicalProcessors().stream().map(PhysicalProcessor::getIdString).anyMatch(id -> id.contains("arm"));
243     }
244 
245     protected long getMaxFreqFromByteArray(byte[] data) {
246         if (data != null && data.length >= 8) {
247             byte[] freqData = Arrays.copyOfRange(data, data.length - 8, data.length - 4);
248             return ParseUtil.byteArrayToLong(freqData, 4, false);
249         }
250         return DEFAULT_FREQUENCY;
251     }
