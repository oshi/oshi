18  /**
19   * Abstract base for the macOS PowerSource. The {@code AppleSmartBattery} registry read and the per-power-source field
20   * computation are shared; the JNA and FFM subclasses supply an {@link IOKitProvider} and read the IOKit Power Sources
21   * (IOPS) list into {@link PowerSourceData} carriers via {@link #buildPowerSources}.
22   */
23  @ThreadSafe
24  public abstract class MacPowerSource extends AbstractPowerSource {

63      /** Battery-wide fields read from the {@code AppleSmartBattery} registry entry, shared by every power source. */
64      private static final class BatteryReadings {
65          private String deviceName = Constants.UNKNOWN;
66          private String manufacturer = Constants.UNKNOWN;
67          private String serialNumber = Constants.UNKNOWN;
68          private LocalDate manufactureDate;
69          private CapacityUnits capacityUnits = CapacityUnits.RELATIVE;
70          private int designCapacity = -1;
71          private int maxCapacity = 1;
72          private int currentCapacity;
73          private int cycleCount = -1;
74          private double timeRemainingInstant;
75          private double temperature;
76          private double voltage = -1d;
77          private double amperage;
78          private double powerUsageRate;
79          private boolean powerOnLine;
80          private boolean charging;
81          private boolean discharging;
82      }

84      /** Raw values read from a single IOKit Power Source (IOPS) dictionary by the JNA/FFM subclasses. */
85      public static final class PowerSourceData {
86          private final String name;
87          private final double currentCapacity;
88          private final double maxCapacity;
89  
90          /**
91           * Creates a single power-source reading.
92           *
93           * @param name            the power source name
94           * @param currentCapacity the current capacity
95           * @param maxCapacity     the maximum capacity
96           */
97          public PowerSourceData(String name, double currentCapacity, double maxCapacity) {
98              this.name = name;
99              this.currentCapacity = currentCapacity;
100             this.maxCapacity = maxCapacity;
101         }
102     }

115     /**
116      * Reads the {@code AppleSmartBattery} registry entry (via {@code ioKit}) and merges it with the per-source IOPS
117      * readings into a list of power sources.
118      * <p>
119      * Mac PowerSource information comes from two sources: the IORegistry's {@code AppleSmartBattery} entry
120      * (battery-wide fields, read here through the shared {@link IOKitProvider}) and the IOKit's IOPS functions (one
121      * entry per power source, read natively by the caller into {@code sources}).
122      *
123      * @param ioKit                  the IOKit provider backing the registry read
124      * @param timeRemainingEstimated the estimated time remaining (-1 unknown, -2 unlimited) from IOPS
125      * @param sources                the present power sources read from the IOPS list
126      * @param factory                creates the platform-specific instance
127      * @return a list of power sources, one per element of {@code sources}
128      */
129     protected static List<PowerSource> buildPowerSources(IOKitProvider ioKit, double timeRemainingEstimated,
130             List<PowerSourceData> sources, PowerSourceFactory factory) {
131         BatteryReadings b = ioKit.withMatchingService("AppleSmartBattery", MacPowerSource::readBattery);
132         if (b == null) {
133             b = new BatteryReadings();
134         }
135         List<PowerSource> psList = new ArrayList<>(sources.size());
136         for (PowerSourceData src : sources) {
137             double psRemainingCapacityPercent = src.maxCapacity <= 0 ? 0d
138                     : Math.min(1d, src.currentCapacity / src.maxCapacity);
139             psList.add(factory.create(src.name, b.deviceName, psRemainingCapacityPercent, timeRemainingEstimated,
140                     b.timeRemainingInstant, b.powerUsageRate, b.voltage, b.amperage, b.powerOnLine, b.charging,
141                     b.discharging, b.capacityUnits, b.currentCapacity, b.maxCapacity, b.designCapacity, b.cycleCount,
142                     Constants.UNKNOWN, b.manufactureDate, b.manufacturer, b.serialNumber, b.temperature));
143         }
144         return psList;
145     }

147     private static BatteryReadings readBattery(RegistryEntry smartBattery) {
148         BatteryReadings b = new BatteryReadings();
149         String s = smartBattery.getStringProperty("DeviceName");
150         if (s != null) {
151             b.deviceName = s;
152         }
153         s = smartBattery.getStringProperty("Manufacturer");
154         if (s != null) {
155             b.manufacturer = s;
156         }
157         s = smartBattery.getStringProperty("BatterySerialNumber");
158         if (s != null) {
159             b.serialNumber = s;
160         }
161 
162         Integer temp = smartBattery.getIntegerProperty("ManufactureDate");
163         if (temp != null) {
164             // Bits 0...4 => day (value 1-31; 5 bits)
165             // Bits 5...8 => month (value 1-12; 4 bits)
166             // Bits 9...15 => years since 1980 (value 0-127; 7 bits)
167             int day = temp & 0x1f;
168             int month = (temp >> 5) & 0xf;
169             int year80 = (temp >> 9) & 0x7f;
170             try {
171                 b.manufactureDate = LocalDate.of(1980 + year80, month, day);
172             } catch (DateTimeException e) {
173                 // Corrupt bitfield — leave manufactureDate as null
174             }
175         }
176 
177         temp = smartBattery.getIntegerProperty("DesignCapacity");
178         if (temp != null) {
179             b.designCapacity = temp;
180         }
181         temp = smartBattery.getIntegerProperty("MaxCapacity");
182         if (temp != null) {
183             b.maxCapacity = temp;
184         }
185         temp = smartBattery.getIntegerProperty("CurrentCapacity");
186         if (temp != null) {
187             b.currentCapacity = temp;
188         }
189         b.capacityUnits = CapacityUnits.MAH;
190 
191         temp = smartBattery.getIntegerProperty("TimeRemaining");
192         if (temp != null) {
193             b.timeRemainingInstant = temp * 60d;
194         }
195         temp = smartBattery.getIntegerProperty("CycleCount");
196         if (temp != null) {
197             b.cycleCount = temp;
198         }
199         temp = smartBattery.getIntegerProperty("Temperature");
200         if (temp != null) {
201             b.temperature = temp / 100d;
202         }
203         temp = smartBattery.getIntegerProperty("Voltage");
204         if (temp != null) {
205             b.voltage = temp / 1000d;
206         }
207         temp = smartBattery.getIntegerProperty("Amperage");
208         if (temp != null) {
209             b.amperage = temp;
210         }
211         b.powerUsageRate = b.voltage * b.amperage;
212 
213         Boolean bool = smartBattery.getBooleanProperty("ExternalConnected");
214         if (bool != null) {
215             b.powerOnLine = bool;
216         }
217         bool = smartBattery.getBooleanProperty("IsCharging");
218         if (bool != null) {
219             b.charging = bool;
220         }
221         b.discharging = !b.charging && !b.powerOnLine;
222 
223         return b;
224     }
