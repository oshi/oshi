5   package oshi.hardware.common.platform.mac;
6   
7   import static oshi.util.Memoizer.memoize;
8   
9   import java.nio.charset.StandardCharsets;
10  import java.util.function.Supplier;
11  
12  import oshi.annotation.concurrent.Immutable;
13  import oshi.hardware.common.AbstractComputerSystem;
14  import oshi.util.Constants;
15  import oshi.util.Util;
16  import oshi.util.tuples.Quartet;
17  
18  /**
19   * Hardware data obtained from ioreg.
20   */
21  @Immutable
22  public abstract class MacComputerSystem extends AbstractComputerSystem {
23  
24      /**
25       * Default constructor.
26       */
27      protected MacComputerSystem() {
28      }
29  
30      private final Supplier<Quartet<String, String, String, String>> manufacturerModelSerialUUID = memoize(
31              this::platformExpert);
32  
33      @Override
34      public String getManufacturer() {
35          return manufacturerModelSerialUUID.get().getA();
36      }
37  
38      @Override
39      public String getModel() {
40          return manufacturerModelSerialUUID.get().getB();
41      }
42  
43      @Override
44      public String getSerialNumber() {
45          return manufacturerModelSerialUUID.get().getC();
46      }
47  
48      @Override
49      public String getHardwareUUID() {
50          return manufacturerModelSerialUUID.get().getD();
51      }
52  
53      /**
54       * Returns the IOKit provider for this implementation.
55       *
56       * @return the IOKit provider
57       */
58      protected abstract IOKitProvider ioKitProvider();
59  
60      /**
61       * Queries platform expert computer system information.
62       *
63       * @return a quartet of manufacturer, model, serial number, UUID
64       */
65      protected Quartet<String, String, String, String> platformExpert() {
66          Quartet<String, String, String, String> result = ioKitProvider().withMatchingService("IOPlatformExpertDevice",
67                  entry -> {
68                      byte[] data = entry.getByteArrayProperty("manufacturer");
69                      String mfr = data != null ? new String(data, StandardCharsets.UTF_8).replace("\0", "").trim()
70                              : null;
71                      data = entry.getByteArrayProperty("model");
72                      String mdl = data != null ? new String(data, StandardCharsets.UTF_8).replace("\0", "").trim()
73                              : null;
74                      String sn = entry.getStringProperty("IOPlatformSerialNumber");
75                      String uuid = entry.getStringProperty("IOPlatformUUID");
76                      return new Quartet<>(mfr, mdl, sn, uuid);
77                  });
78          if (result == null) {
79              result = new Quartet<>(null, null, null, null);
80          }
81          return new Quartet<>(Util.isBlank(result.getA()) ? "Apple Inc." : result.getA(),
82                  Util.isBlank(result.getB()) ? Constants.UNKNOWN : result.getB(),
83                  Util.isBlank(result.getC()) ? Constants.UNKNOWN : result.getC(),
84                  Util.isBlank(result.getD()) ? Constants.UNKNOWN : result.getD());
85      }
86  }
