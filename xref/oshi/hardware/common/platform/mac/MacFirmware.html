5   package oshi.hardware.common.platform.mac;
6   
7   import static oshi.util.Memoizer.memoize;
8   
9   import java.util.function.Supplier;
10  
11  import oshi.annotation.concurrent.Immutable;
12  import oshi.hardware.common.AbstractFirmware;
13  import oshi.util.tuples.Quintet;
14  
15  /**
16   * Firmware data obtained from ioreg. Subclasses provide platform-specific IOKit queries.
17   */
18  @Immutable
19  public abstract class MacFirmware extends AbstractFirmware {
20  
21      private final Supplier<Quintet<String, String, String, String, String>> manufNameDescVersRelease = memoize(
22              this::queryEfi);
23  
24      @Override
25      public String getManufacturer() {
26          return manufNameDescVersRelease.get().getA();
27      }
28  
29      @Override
30      public String getName() {
31          return manufNameDescVersRelease.get().getB();
32      }
33  
34      @Override
35      public String getDescription() {
36          return manufNameDescVersRelease.get().getC();
37      }
38  
39      @Override
40      public String getVersion() {
41          return manufNameDescVersRelease.get().getD();
42      }
43  
44      @Override
45      public String getReleaseDate() {
46          return manufNameDescVersRelease.get().getE();
47      }
48  
49      protected abstract Quintet<String, String, String, String, String> queryEfi();
50  }
