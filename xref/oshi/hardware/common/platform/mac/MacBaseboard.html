5   package oshi.hardware.common.platform.mac;
6   
7   import static oshi.util.Memoizer.memoize;
8   
9   import java.util.function.Supplier;
10  
11  import oshi.annotation.concurrent.Immutable;
12  import oshi.hardware.common.AbstractBaseboard;
13  import oshi.util.tuples.Quartet;
14  
15  /**
16   * Baseboard data obtained from ioreg
17   */
18  @Immutable
19  public abstract class MacBaseboard extends AbstractBaseboard {
20  
21      private final Supplier<Quartet<String, String, String, String>> manufModelVersSerial = memoize(this::queryPlatform);
22  
23      @Override
24      public String getManufacturer() {
25          return manufModelVersSerial.get().getA();
26      }
27  
28      @Override
29      public String getModel() {
30          return manufModelVersSerial.get().getB();
31      }
32  
33      @Override
34      public String getVersion() {
35          return manufModelVersSerial.get().getC();
36      }
37  
38      @Override
39      public String getSerialNumber() {
40          return manufModelVersSerial.get().getD();
41      }
42  
43      protected abstract Quartet<String, String, String, String> queryPlatform();
44  }
