5   package oshi.hardware.common.platform.mac;
6   
7   import static java.util.Collections.emptyList;
8   import static java.util.Collections.sort;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  import java.util.Map;
13  
14  import oshi.annotation.concurrent.Immutable;
15  import oshi.hardware.UsbDevice;
16  import oshi.hardware.common.AbstractUsbDevice;
17  
18  /**
19   * Mac USB device base class.
20   */
21  @Immutable
22  public class MacUsbDevice extends AbstractUsbDevice {
23  
24      protected MacUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
25              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
26          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
27      }
28  
29      /**
30       * No-arg constructor required so that subclasses, which extend this class solely to inherit its static helper
31       * methods, can compile without an explicit constructor. No subclass is ever instantiated; only this class is, via
32       * the full constructor.
33       */
34      protected MacUsbDevice() {
35          this("", "", "", "", "", "", emptyList());
36      }
37  
38      /**
39       * Recursively creates MacUsbDevices by fetching information from maps to populate fields
40       *
41       * @param registryEntryId The device unique registry id.
42       * @param vid             The default (parent) vendor ID
43       * @param pid             The default (parent) product ID
44       * @param nameMap         the map of names
45       * @param vendorMap       the map of vendors
46       * @param vendorIdMap     the map of vendorIds
47       * @param productIdMap    the map of productIds
48       * @param serialMap       the map of serial numbers
49       * @param hubMap          the map of hubs
50       * @return A MacUsbDevice corresponding to this device
51       */
52      protected static MacUsbDevice getDeviceAndChildren(Long registryEntryId, String vid, String pid,
53              Map<Long, String> nameMap, Map<Long, String> vendorMap, Map<Long, String> vendorIdMap,
54              Map<Long, String> productIdMap, Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
55          String vendorId = vendorIdMap.getOrDefault(registryEntryId, vid);
56          String productId = productIdMap.getOrDefault(registryEntryId, pid);
57          List<Long> childIds = hubMap.getOrDefault(registryEntryId, new ArrayList<>());
58          List<UsbDevice> usbDevices = new ArrayList<>();
59          for (Long childId : childIds) {
60              usbDevices.add(getDeviceAndChildren(childId, vendorId, productId, nameMap, vendorMap, vendorIdMap,
61                      productIdMap, serialMap, hubMap));
62          }
63          sort(usbDevices);
64          return new MacUsbDevice(nameMap.getOrDefault(registryEntryId, vendorId + ":" + productId),
65                  vendorMap.getOrDefault(registryEntryId, ""), vendorId, productId,
66                  serialMap.getOrDefault(registryEntryId, ""), "0x" + Long.toHexString(registryEntryId), usbDevices);
67      }
68  }
