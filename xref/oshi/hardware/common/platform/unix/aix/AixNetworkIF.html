5   package oshi.hardware.common.platform.unix.aix;
6   
7   import java.net.NetworkInterface;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.AbstractNetworkIF;
11  
12  /**
13   * Abstract base for AIX NetworkIF. {@link #updateAttributes()} populates fields from a per-interface stats POJO that
14   * subclasses fetch from their respective perfstat driver.
15   */
16  @ThreadSafe
17  public abstract class AixNetworkIF extends AbstractNetworkIF {
18  
19      /** Per-interface stats fields the AIX HAL reads. JNA and FFM concrete subclasses both fill this. */
20      public static final class IfStats {
21          public long bytesSent;
22          public long bytesRecv;
23          public long packetsSent;
24          public long packetsRecv;
25          public long outErrors;
26          public long inErrors;
27          public long collisions;
28          public long inDrops;
29          public long speed;
30      }
31  
32      protected AixNetworkIF(NetworkInterface netint) throws InstantiationException {
33          super(netint);
34      }
35  
36      @Override
37      public boolean updateAttributes() {
38          IfStats stats = queryStats();
39          if (stats == null) {
40              return false;
41          }
42          this.bytesSent = stats.bytesSent;
43          this.bytesRecv = stats.bytesRecv;
44          this.packetsSent = stats.packetsSent;
45          this.packetsRecv = stats.packetsRecv;
46          this.outErrors = stats.outErrors;
47          this.inErrors = stats.inErrors;
48          this.collisions = stats.collisions;
49          this.inDrops = stats.inDrops;
50          this.speed = stats.speed;
51          this.timeStamp = System.currentTimeMillis();
52          return true;
53      }
54  
55      /**
56       * Looks up this interface's per-interface stats from the subclass's perfstat data source.
57       *
58       * @return stats POJO, or {@code null} if no entry was found for this interface name
59       */
60      protected abstract IfStats queryStats();
61  }
