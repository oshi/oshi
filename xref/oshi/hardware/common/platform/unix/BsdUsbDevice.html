7   import java.util.ArrayList;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map;
11  
12  import oshi.annotation.concurrent.Immutable;
13  import oshi.hardware.UsbDevice;
14  import oshi.hardware.common.AbstractUsbDevice;
15  import oshi.util.ExecutingCommand;
16  
17  /**
18   * Shared USB device implementation for BSDs that use {@code usbdevs -v} to enumerate devices (NetBSD, OpenBSD).
19   */
20  @Immutable
21  public final class BsdUsbDevice extends AbstractUsbDevice {
22  
23      public BsdUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
24              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
25          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
26      }
27  
28      /**
29       * Instantiates the USB controller device tree. The flat form is derived by the caller (the HAL).
30       *
31       * @return a list of USB controllers, each with its connected-device tree.
32       */
33      public static List<UsbDevice> getUsbDevices() {
34          Map<String, String> nameMap = new HashMap<>();
35          Map<String, String> vendorMap = new HashMap<>();
36          Map<String, String> vendorIdMap = new HashMap<>();
37          Map<String, String> productIdMap = new HashMap<>();
38          Map<String, String> serialMap = new HashMap<>();
39          Map<String, List<String>> hubMap = new HashMap<>();
40  
41          List<String> rootHubs = new ArrayList<>();
42          String key = "";
43          String parent = "";
44          for (String line : ExecutingCommand.runNative("usbdevs -v")) {
45              if (line.startsWith("Controller ")) {
46                  parent = line.substring(11);
47              } else if (line.startsWith("addr ")) {
48                  if (line.indexOf(':') == 7 && line.indexOf(',') >= 18) {
49                      key = parent + line.substring(0, 7);
50                      String[] split = line.substring(8).trim().split(",");
51                      if (split.length > 1) {
52                          String vendorStr = split[0].trim();
53                          int idx1 = vendorStr.indexOf(':');
54                          int idx2 = vendorStr.indexOf(' ');
55                          if (idx1 >= 0 && idx2 >= 0) {
56                              vendorIdMap.put(key, vendorStr.substring(0, idx1));
57                              productIdMap.put(key, vendorStr.substring(idx1 + 1, idx2));
58                              vendorMap.put(key, vendorStr.substring(idx2 + 1));
59                          }
60                          nameMap.put(key, split[1].trim());
61                          hubMap.computeIfAbsent(parent, x -> new ArrayList<>()).add(key);
62                          if (!parent.contains("addr")) {
63                              parent = key;
64                              rootHubs.add(parent);
65                          }
66                      }
67                  }
68              } else if (!key.isEmpty()) {
69                  int idx = line.indexOf("iSerial ");
70                  if (idx >= 0) {
71                      serialMap.put(key, line.substring(idx + 8).trim());
72                  }
73                  key = "";
74              }
75          }
76  
77          List<UsbDevice> controllerDevices = new ArrayList<>();
78          for (String devusb : rootHubs) {
79              controllerDevices.add(buildDeviceTree(devusb, "0000", "0000", nameMap, vendorMap, vendorIdMap, productIdMap,
80                      serialMap, hubMap, BsdUsbDevice::new));
81          }
82          return controllerDevices;
83      }
84  }
