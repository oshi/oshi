5   package oshi.hardware.common.platform.unix.openbsd;
6   
7   import java.util.List;
8   import java.util.Locale;
9   
10  import oshi.hardware.common.platform.unix.BsdCentralProcessor;
11  import oshi.util.ExecutingCommand;
12  import oshi.util.ParseUtil;
13  import oshi.util.tuples.Pair;
14  import oshi.util.tuples.Triplet;
15  
16  /**
17   * Shared OpenBSD CentralProcessor logic. The processor identification, frequency, and {@code vmstat} parsing are common
18   * to the JNA and FFM backends; subclasses supply only the sysctl reads (mib- and name-based) and the native CPU-tick
19   * and load-average calls.
20   */
21  public abstract class OpenBsdCentralProcessor extends BsdCentralProcessor {
22  
23      // OpenBSD sysctl(3) ABI identifiers, stable across releases (mirrored from the JNA/FFM libc backends).
24      private static final int CTL_HW = 6;
25      private static final int HW_MACHINE = 1;
26      private static final int HW_MODEL = 2;
27      private static final int HW_CPUSPEED = 12;
28  
29      /**
30       * Reads a string sysctl value by name.
31       *
32       * @param name the sysctl name
33       * @param def  the default value
34       * @return the sysctl string value or the default
35       */
36      protected abstract String sysctl(String name, String def);
37  
38      /**
39       * Reads a string sysctl value addressed by a Management Information Base (MIB) array.
40       *
41       * @param mib the sysctl MIB
42       * @param def the default value
43       * @return the sysctl string value or the default
44       */
45      protected abstract String sysctl(int[] mib, String def);
46  
47      /**
48       * Reads an integer sysctl value addressed by a Management Information Base (MIB) array.
49       *
50       * @param mib the sysctl MIB
51       * @param def the default value
52       * @return the sysctl integer value or the default
53       */
54      protected abstract int sysctl(int[] mib, int def);
55  
56      @Override
57      protected ProcessorIdentifier queryProcessorId() {
58          String cpuVendor = sysctl("machdep.cpuvendor", "");
59          int[] mib = { CTL_HW, HW_MODEL };
60          String cpuName = sysctl(mib, "");
61          int cpuid = ParseUtil.hexStringToInt(sysctl("machdep.cpuid", ""), 0);
62          int cpufeature = ParseUtil.hexStringToInt(sysctl("machdep.cpufeature", ""), 0);
63          Triplet<Integer, Integer, Integer> cpu = cpuidToFamilyModelStepping(cpuid);
64          String cpuFamily = cpu.getA().toString();
65          String cpuModel = cpu.getB().toString();
66          String cpuStepping = cpu.getC().toString();
67          long cpuFreq = ParseUtil.parseHertz(cpuName);
68          if (cpuFreq < 0) {
69              cpuFreq = queryMaxFreq();
70          }
71          mib = new int[] { CTL_HW, HW_MACHINE };
72          String machine = sysctl(mib, "");
73          boolean cpu64bit = machine != null && machine.contains("64")
74                  || ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64");
75          String processorID = String.format(Locale.ROOT, "%08x%08x", cpufeature, cpuid);
76  
77          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
78                  cpuFreq);
79      }
80  
81      private static Triplet<Integer, Integer, Integer> cpuidToFamilyModelStepping(int cpuid) {
82          // family is bits 27:20 | 11:8
83          int family = cpuid >> 16 & 0xff0 | cpuid >> 8 & 0xf;
84          // model is bits 19:16 | 7:4
85          int model = cpuid >> 12 & 0xf0 | cpuid >> 4 & 0xf;
86          // stepping is bits 3:0
87          int stepping = cpuid & 0xf;
88          return new Triplet<>(family, model, stepping);
89      }
90  
91      @Override
92      protected long[] queryCurrentFreq() {
93          long[] freq = new long[1];
94          int[] mib = { CTL_HW, HW_CPUSPEED };
95          freq[0] = sysctl(mib, 0) * 1_000_000L;
96          return freq;
97      }
98  
99      @Override
100     protected Pair<Long, Long> queryVmStats() {
101         long contextSwitches = 0L;
102         long interrupts = 0L;
103         List<String> vmstat = ExecutingCommand.runNative("vmstat -s");
104         for (String line : vmstat) {
105             if (line.endsWith("cpu context switches")) {
106                 contextSwitches = ParseUtil.parseLongOrDefault(line.trim().split("\\s+")[0], 0L);
107             } else if (line.endsWith("interrupts")) {
108                 interrupts = ParseUtil.parseLongOrDefault(line.trim().split("\\s+")[0], 0L);
109             }
110         }
111         return new Pair<>(contextSwitches, interrupts);
112     }
113 }
