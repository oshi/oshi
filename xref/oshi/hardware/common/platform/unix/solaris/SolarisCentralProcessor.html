22  /**
23   * Abstract base for the Solaris CPU. The command-based pieces (NUMA mapping, processor-ID feature flags, the {@code
24   * dmesg} physical-processor parsing, the averaged tick rollup, and the {@code kstat -p} context-switch/interrupt
25   * fallbacks) are shared; the {@code kstat}-chain queries are native and implemented by the JNA and FFM subclasses.
26   */
27  @ThreadSafe
28  public abstract class SolarisCentralProcessor extends AbstractCentralProcessor {
29  
30      @Override
31      protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
32          // Defensive copy: the base appends a fallback entry, so don't assume the subclass returned a mutable list
33          List<LogicalProcessor> logProcs = new ArrayList<>(queryLogicalProcessors());
34          if (logProcs.isEmpty()) {
35              logProcs.add(new LogicalProcessor(0, 0, 0));
36          }
37          Map<Integer, String> dmesg = parseDmesgCpuInfo(ExecutingCommand.runNative("dmesg"));
38          if (dmesg.isEmpty()) {
39              return new Quartet<>(logProcs, null, null, Collections.emptyList());
40          }
41          List<String> featureFlags = ExecutingCommand.runNative("isainfo -x");
42          return new Quartet<>(logProcs, createProcListFromDmesg(logProcs, dmesg), null, featureFlags);
43      }
44  
45      @Override
46      public long[] querySystemCpuLoadTicks() {
47          long[] ticks = new long[TickType.values().length];
48          // Average processor ticks
49          long[][] procTicks = getProcessorCpuLoadTicks();
50          for (int i = 0; i < ticks.length; i++) {
51              for (long[] procTick : procTicks) {
52                  ticks[i] += procTick[i];
53              }
54              ticks[i] /= procTicks.length;
55          }
56          return ticks;
57      }
58  
59      @Override
60      public long queryContextSwitches() {
61          // Shell-escaped pipe: Java \\\\| -> shell \\| -> kstat regex \| -> matches literal pipe.
62          return sumKstatLong(ExecutingCommand.runNative("kstat -p cpu_stat:::/pswitch\\\\|inv_swtch/"));
63      }
64  
65      @Override
66      public long queryInterrupts() {
67          return sumKstatLong(ExecutingCommand.runNative("kstat -p cpu_stat:::/intr/"));
68      }

77      /**
78       * Maps each CPU id to its NUMA node (lgroup) via {@code lgrpinfo}.
79       *
80       * @return a map of processor id to NUMA node
81       */
82      protected static Map<Integer, Integer> mapNumaNodes() {
83          return parseNumaNodes(ExecutingCommand.runNative("lgrpinfo -c leaves"));
84      }

86      /**
87       * Fetches the ProcessorID by encoding the stepping, model, family, and feature flags.
88       *
89       * @param stepping the stepping
90       * @param model    the model
91       * @param family   the family
92       * @return the Processor ID string
93       */
94      protected static String getProcessorID(String stepping, String model, String family) {
95          String[] flags = parseIsainfoFlags(ExecutingCommand.runNative("isainfo -v"));
96          return createProcessorID(stepping, model, family, flags);
97      }

99      /**
100      * Parses the output of {@code dmesg} to extract CPU name strings keyed by core id.
101      *
102      * @param dmesg the lines emitted by {@code dmesg}
103      * @return a map of core id to CPU name string
104      */
105     static Map<Integer, String> parseDmesgCpuInfo(List<String> dmesg) {
106         Map<Integer, String> cpuMap = new HashMap<>();
107         // Jan 9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: Intel(r) Celeron(r)
108         // CPU J3455 @ 1.50GHz
109         // but NOT: Jan 9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: x86 (chipid
110         // 0x0 GenuineIntel 506C9 family 6 model 92 step 9 clock 1500 MHz)
111         Pattern p = Pattern.compile(".* cpu(\\d+): ((ARM|AMD|Intel).+)");
112         for (String s : dmesg) {
113             Matcher m = p.matcher(s);
114             if (m.matches()) {
115                 int coreId = ParseUtil.parseIntOrDefault(m.group(1), 0);
116                 cpuMap.put(coreId, m.group(2).trim());
117             }
118         }
119         return cpuMap;
120     }

122     /**
123      * Parses the output of {@code lgrpinfo -c leaves} into a map of CPU id to NUMA node.
124      *
125      * @param lgrpinfo the lines emitted by {@code lgrpinfo -c leaves}
126      * @return a map of processor id to NUMA node number
127      */
128     static Map<Integer, Integer> parseNumaNodes(List<String> lgrpinfo) {
129         // Get numa node info from lgrpinfo
130         Map<Integer, Integer> numaNodeMap = new HashMap<>();
131         int lgroup = 0;
132         for (String line : lgrpinfo) {
133             // Format:
134             // lgroup 0 (root):
135             // CPUs: 0 1
136             // CPUs: 0-7
137             // CPUs: 0-3 6 7 12 13
138             // CPU: 0
139             // CPU: 1
140             if (line.startsWith("lgroup")) {
141                 lgroup = ParseUtil.getFirstIntValue(line);
142             } else if (line.contains("CPUs:") || line.contains("CPU:")) {
143                 for (Integer cpu : ParseUtil.parseHyphenatedIntList(line.split(":")[1])) {
144                     numaNodeMap.put(cpu, lgroup);
145                 }
146             }
147         }
148         return numaNodeMap;
149     }
150 
151     /**
152      * Parses the output of {@code isainfo -v} to extract the 64-bit feature flags.
153      *
154      * @param isainfo the lines emitted by {@code isainfo -v}
155      * @return an array of feature flag strings (lowercase)
156      */
157     static String[] parseIsainfoFlags(List<String> isainfo) {
158         StringBuilder flags = new StringBuilder();
159         for (String line : isainfo) {
160             if (line.startsWith("32-bit")) {
161                 break;
162             } else if (!line.startsWith("64-bit")) {
163                 flags.append(' ').append(line.trim());
164             }
165         }
166         return ParseUtil.whitespaces.split(flags.toString().toLowerCase(Locale.ROOT));
167     }
168 
169     /**
170      * Sums the last long value from each line of {@code kstat -p} output.
171      *
172      * @param kstat the lines emitted by a {@code kstat -p} command
173      * @return the sum of the trailing long values across all lines
174      */
175     static long sumKstatLong(List<String> kstat) {
176         long total = 0L;
177         for (String s : kstat) {
178             total += ParseUtil.parseLastLong(s, 0L);
179         }
180         return total;
181     }
