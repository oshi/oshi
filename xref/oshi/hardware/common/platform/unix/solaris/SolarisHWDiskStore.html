18  import oshi.driver.common.unix.solaris.disk.Prtvtoc;

24  /**
25   * Abstract base for Solaris HWDiskStore. The disk enumeration ({@link #getDisks}) and partition assembly are shared and
26   * call the common {@code iostat}/{@code lshal}/{@code prtvtoc} drivers; the {@code kstat}-based per-disk I/O stats are
27   * native and fetched by the JNA and FFM subclasses via {@link #queryStats()}.
28   */
29  @ThreadSafe
30  public abstract class SolarisHWDiskStore extends AbstractHWDiskStore {
31  
32      /** Per-disk I/O stats the Solaris HAL reads. Both JNA and FFM concrete subclasses populate this. */
33      public static final class DiskStats {
34          public long reads;
35          public long writes;
36          public long readBytes;
37          public long writeBytes;
38          public long currentQueueLength;
39          public long transferTime;
40          public long timeStamp;
41      }
42  
43      /** Creates a concrete (JNA or FFM) Solaris disk store. */
44      @FunctionalInterface
45      protected interface DiskStoreFactory {
46          SolarisHWDiskStore create(String name, String model, String serial, long size);
47      }
48  
49      protected SolarisHWDiskStore(String name, String model, String serial, long size) {
50          super(name, model, serial, size);
51      }
52  
53      @Override
54      public boolean updateAttributes() {
55          this.timeStamp = System.currentTimeMillis();
56          DiskStats stats = queryStats();
57          if (stats == null) {
58              return false;
59          }
60          this.reads = stats.reads;
61          this.writes = stats.writes;
62          this.readBytes = stats.readBytes;
63          this.writeBytes = stats.writeBytes;
64          this.currentQueueLength = stats.currentQueueLength;
65          this.transferTime = stats.transferTime;
66          this.timeStamp = stats.timeStamp;
67          return true;
68      }
69  
70      /**
71       * Gets the disks on this machine, building each store with the given concrete factory.
72       *
73       * @param factory creates the platform-specific store instances
74       * @return a list of {@link HWDiskStore} objects representing the disks
75       */
76      protected static List<HWDiskStore> getDisks(DiskStoreFactory factory) {
77          // Create map to correlate disk name with block device mount point for later use in partition info
78          Map<String, String> deviceMap = Iostat.queryPartitionToMountMap();
79  
80          // Run lshal, if available, to get block device major (we'll use partition # for minor)
81          Map<String, Integer> majorMap = Lshal.queryDiskToMajorMap();
82  
83          // Create map of model, vendor, product, serial, size
84          // We'll use Model if available, otherwise Vendor+Product
85          Map<String, Quintet<String, String, String, String, Long>> deviceStringMap = Iostat
86                  .queryDeviceStrings(deviceMap.keySet());
87  
88          List<HWDiskStore> storeList = new ArrayList<>();
89          for (Entry<String, Quintet<String, String, String, String, Long>> entry : deviceStringMap.entrySet()) {
90              String storeName = entry.getKey();
91              Quintet<String, String, String, String, Long> val = entry.getValue();
92              storeList.add(createStore(factory, storeName, val.getA(), val.getB(), val.getC(), val.getD(), val.getE(),
93                      deviceMap.getOrDefault(storeName, ""), majorMap.getOrDefault(storeName, 0)));
94          }
95  
96          return storeList;
97      }
98  
99      private static SolarisHWDiskStore createStore(DiskStoreFactory factory, String diskName, String model,
100             String vendor, String product, String serial, long size, String mount, int major) {
101         SolarisHWDiskStore store = factory.create(diskName, model.isEmpty() ? (vendor + " " + product).trim() : model,
102                 serial, size);
103         store.setPartitionList(Collections.unmodifiableList(Prtvtoc.queryPartitions(mount, major).stream()
104                 .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList())));
105         store.updateAttributes();
106         return store;
107     }
108 
109     /**
110      * Looks up this disk's per-disk I/O stats from the subclass's kstat data source.
111      *
112      * @return stats POJO, or {@code null} if no entry was found for this disk name
113      */
114     protected abstract DiskStats queryStats();
115 }
