5   package oshi.hardware.common;
6   
7   import java.util.Locale;
8   
9   import oshi.annotation.concurrent.Immutable;
10  import oshi.hardware.GraphicsCard;
11  import oshi.hardware.GpuStats;
12  
13  /**
14   * An abstract Graphics Card
15   */
16  @Immutable
17  public abstract class AbstractGraphicsCard implements GraphicsCard {
18  
19      private final String name;
20      private final String deviceId;
21      private final String vendor;
22      private final String versionInfo;
23      private final long vram;
24  
25      /**
26       * Constructor for AbstractGraphicsCard
27       *
28       * @param name        The name
29       * @param deviceId    The device ID
30       * @param vendor      The vendor
31       * @param versionInfo The version info
32       * @param vram        The VRAM
33       */
34      protected AbstractGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
35          this.name = name;
36          this.deviceId = deviceId;
37          this.vendor = vendor;
38          this.versionInfo = versionInfo;
39          this.vram = vram;
40      }
41  
42      @Override
43      public String getName() {
44          return name;
45      }
46  
47      @Override
48      public String getDeviceId() {
49          return deviceId;
50      }
51  
52      @Override
53      public String getVendor() {
54          return vendor;
55      }
56  
57      @Override
58      public String getVersionInfo() {
59          return versionInfo;
60      }
61  
62      @Override
63      public long getVRam() {
64          return vram;
65      }
66  
67      @Override
68      public GpuStats createStatsSession() {
69          return new NoOpGpuStats();
70      }
71  
72      /**
73       * Returns a string representation of this graphics card including any available dynamic metrics.
74       *
75       * <p>
76       * <strong>Performance note:</strong> this method opens a native {@link GpuStats} session via
77       * {@link #createStatsSession()}, samples all available metrics, then closes the session. On platforms with native
78       * GPU subscriptions (e.g. Apple Silicon via IOReport) this involves kernel-level IPC and should not be called in
79       * hot paths. Cache the result if repeated string representations are needed.
80       *
81       * <p>
82       * Delta-based metric backends ({@link GpuStats#getGpuUtilization()} and {@link GpuStats#getPowerDraw()}) may return
83       * -1 on the first call after a session is opened while they record the initial baseline. Backends that read
84       * instantaneous values (e.g. Linux sysfs and the macOS IOAccelerator fallback) can return a valid value on the
85       * first call. Use a persistent {@link GpuStats} session with a polling loop for reliable delta values.
86       *
87       * @return a human-readable description of this graphics card
88       */
89      @Override
90      public String toString() {
91          StringBuilder builder = new StringBuilder();
92          builder.append("GraphicsCard@");
93          builder.append(Integer.toHexString(hashCode()));
94          builder.append(" [name=");
95          builder.append(this.name);
96          builder.append(", deviceId=");
97          builder.append(this.deviceId);
98          builder.append(", vendor=");
99          builder.append(this.vendor);
100         builder.append(", vRam=");
101         builder.append(this.vram);
102         GpuStats stats = null;
103         try {
104             stats = createStatsSession();
105             long vramUsed = stats.getVramUsed();
106             if (vramUsed >= 0) {
107                 builder.append(", vramUsed=");
108                 builder.append(vramUsed);
109             }
110             long sharedUsed = stats.getSharedMemoryUsed();
111             if (sharedUsed >= 0) {
112                 builder.append(", sharedMemUsed=");
113                 builder.append(sharedUsed);
114             }
115             double utilization = stats.getGpuUtilization();
116             if (utilization >= 0) {
117                 builder.append(", utilization=");
118                 builder.append(String.format(Locale.ROOT, "%.1f%%", utilization));
119             }
120             double temp = stats.getTemperature();
121             if (temp >= 0) {
122                 builder.append(", temp=");
123                 builder.append(String.format(Locale.ROOT, "%.1f°C", temp));
124             }
125             double power = stats.getPowerDraw();
126             if (power >= 0) {
127                 builder.append(", power=");
128                 builder.append(String.format(Locale.ROOT, "%.1fW", power));
129             }
130             long coreClock = stats.getCoreClockMhz();
131             if (coreClock >= 0) {
132                 builder.append(", coreClock=");
133                 builder.append(coreClock);
134                 builder.append("MHz");
135             }
136             long memClock = stats.getMemoryClockMhz();
137             if (memClock >= 0) {
138                 builder.append(", memClock=");
139                 builder.append(memClock);
140                 builder.append("MHz");
141             }
142             double fan = stats.getFanSpeedPercent();
143             if (fan >= 0) {
144                 builder.append(", fan=");
145                 builder.append(String.format(Locale.ROOT, "%.1f%%", fan));
146             }
147         } catch (Exception e) {
148             builder.append(", metricsUnavailable");
149         } finally {
150             if (stats != null) {
151                 try {
152                     stats.close();
153                 } catch (Exception e) {
154                     // best-effort cleanup; toString must not propagate close() failures
155                 }
156             }
157         }
158         builder.append(", versionInfo=[");
159         builder.append(this.versionInfo);
160         builder.append("]]");
161         return builder.toString();
162     }
163 }
