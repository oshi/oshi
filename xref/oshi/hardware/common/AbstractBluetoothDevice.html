5   package oshi.hardware.common;
6   
7   import oshi.annotation.concurrent.Immutable;
8   import oshi.hardware.BluetoothDevice;
9   
10  /**
11   * Abstract base class for BluetoothDevice implementations.
12   */
13  @Immutable
14  public abstract class AbstractBluetoothDevice implements BluetoothDevice {
15  
16      private final String name;
17      private final String address;
18      private final String majorDeviceClass;
19      private final boolean connected;
20      private final boolean paired;
21      private final int batteryLevel;
22      private final String adapterName;
23  
24      /**
25       * Creates an AbstractBluetoothDevice with the given parameters.
26       *
27       * @param name             the device name
28       * @param address          the MAC address
29       * @param majorDeviceClass the major device class
30       * @param connected        whether the device is connected
31       * @param paired           whether the device is paired
32       * @param batteryLevel     the battery level (0–100), or -1 if unavailable
33       * @param adapterName      the adapter name
34       */
35      protected AbstractBluetoothDevice(String name, String address, String majorDeviceClass, boolean connected,
36              boolean paired, int batteryLevel, String adapterName) {
37          this.name = name;
38          this.address = address;
39          this.majorDeviceClass = majorDeviceClass;
40          this.connected = connected;
41          this.paired = paired;
42          this.batteryLevel = batteryLevel < 0 ? -1 : Math.min(100, batteryLevel);
43          this.adapterName = adapterName;
44      }
45  
46      @Override
47      public String getName() {
48          return name;
49      }
50  
51      @Override
52      public String getAddress() {
53          return address;
54      }
55  
56      @Override
57      public String getMajorDeviceClass() {
58          return majorDeviceClass;
59      }
60  
61      @Override
62      public boolean isConnected() {
63          return connected;
64      }
65  
66      @Override
67      public boolean isPaired() {
68          return paired;
69      }
70  
71      @Override
72      public int getBatteryLevel() {
73          return batteryLevel;
74      }
75  
76      @Override
77      public String getAdapterName() {
78          return adapterName;
79      }
80  
81      @Override
82      public String toString() {
83          return "BluetoothDevice [name=" + name + ", address=" + address + ", class=" + majorDeviceClass + ", connected="
84                  + connected + ", paired=" + paired + ", battery=" + batteryLevel + ", adapter=" + adapterName + "]";
85      }
86  
87      /**
88       * Parses the major device class from the Bluetooth Class of Device (CoD) integer.
89       * <p>
90       * The major device class is bits 12–8 of the CoD value.
91       *
92       * @param cod the Class of Device integer
93       * @return a human-readable major device class string
94       */
95      public static String parseMajorDeviceClass(int cod) {
96          int major = (cod >> 8) & 0x1F;
97          switch (major) {
98              case 0:
99                  return "Miscellaneous";
100             case 1:
101                 return "Computer";
102             case 2:
103                 return "Phone";
104             case 3:
105                 return "Networking";
106             case 4:
107                 return "Audio/Video";
108             case 5:
109                 return "Peripheral";
110             case 6:
111                 return "Imaging";
112             case 7:
113                 return "Wearable";
114             case 8:
115                 return "Toy";
116             case 9:
117                 return "Health";
118             case 31:
119                 return "Uncategorized";
120             default:
121                 return "";
122         }
123     }
124 }
