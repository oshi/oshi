19  /**
20   * Memory refers to the state information of a computing system, as it is kept
21   * active in some physical structure. The term "memory" is used for the
22   * information in physical systems which are fast (ie. RAM), as a distinction
23   * from physical systems which are slow to access (ie. data storage). By design,
24   * the term "memory" refers to temporary state devices, whereas the term
25   * "storage" is reserved for permanent data.
26   * 
27   * @author dblock[at]dblock[dot]org
28   */
29  public interface Memory {
30  	/**
31  	 * Total memory.
32  	 * 
33  	 * @return Total number of bytes.
34  	 */
35  	long getTotal();
36  
37  	/**
38  	 * Currently available.
39  	 * 
40  	 * @return Available number of bytes.
41  	 */
42  	long getAvailable();
43  }
