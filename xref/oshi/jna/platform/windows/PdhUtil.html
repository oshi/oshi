32  /**
33   * Pdh utility API.
34   * 
35   * @author widdis[at]gmail[dot]com
36   */
37  public abstract class PdhUtil {
38      private static final int CHAR_TO_BYTES = Boolean.getBoolean("w32.ascii") ? 1 : Native.WCHAR_SIZE;
39  
40      // This REG_MULTI_SZ value in HKLM provides English counters regardless of
41      // the current locale setting
42      private static final String ENGLISH_COUNTER_KEY = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Perflib\\009";
43      private static final String ENGLISH_COUNTER_VALUE = "Counter";
44  
45      /**
46       * Utility method to call Pdh's PdhLookupPerfNameByIndex that allocates the
47       * required memory for the szNameBuffer parameter based on the type mapping
48       * used, calls to PdhLookupPerfNameByIndex, and returns the received string.
49       * 
50       * @param szMachineName
51       *            Null-terminated string that specifies the name of the computer
52       *            where the specified performance object or counter is located.
53       *            The computer name can be specified by the DNS name or the IP
54       *            address. If NULL, the function uses the local computer.
55       * @param dwNameIndex
56       *            Index of the performance object or counter.
57       * @return Returns the name of the performance object or counter.
58       */
59      public static String PdhLookupPerfNameByIndex(String szMachineName, int dwNameIndex) {
60          // Call once to get required buffer size
61          DWORDByReference pcchNameBufferSize = new DWORDByReference(new DWORD(0));
62          int result = Pdh.INSTANCE.PdhLookupPerfNameByIndex(szMachineName, dwNameIndex, null, pcchNameBufferSize);
63          if (result != WinError.ERROR_SUCCESS && result != Pdh.PDH_MORE_DATA) {
64              throw new PdhException(result);
65          }
66  
67          // Can't allocate 0 memory
68          if (pcchNameBufferSize.getValue().longValue() < 1) {
69              return "";
70          }
71          // Allocate buffer and call again
72          Memory mem = new Memory(pcchNameBufferSize.getValue().longValue() * CHAR_TO_BYTES);
73          result = Pdh.INSTANCE.PdhLookupPerfNameByIndex(szMachineName, dwNameIndex, mem, pcchNameBufferSize);
74  
75          if (result != WinError.ERROR_SUCCESS) {
76              throw new PdhException(result);
77          }
78  
79          // Convert buffer to Java String
80          if (CHAR_TO_BYTES == 1) {
81              return mem.getString(0);
82          } else {
83              return mem.getWideString(0);
84          }
85      }

87      /**
88       * Utility method similar to Pdh's PdhLookupPerfIndexByName that returns the
89       * counter index corresponding to the specified counter name in English.
90       * Uses the registry on the local machine to find the index in the English
91       * locale, regardless of the current language setting on the machine.
92       * 
93       * @param szNameBuffer
94       *            The English name of the performance counter
95       * @return The counter's index if it exists, or 0 otherwise.
96       */
97      public static int PdhLookupPerfIndexByEnglishName(String szNameBuffer) {
98          // Look up list of english names and ids
99          String[] counters = Advapi32Util.registryGetStringArray(WinReg.HKEY_LOCAL_MACHINE, ENGLISH_COUNTER_KEY,
100                 ENGLISH_COUNTER_VALUE);
101         // Array contains alternating index/name pairs
102         // {"1", "1847", "2", "System", "4", "Memory", ... }
103         // Get position of name in the array (odd index), return parsed value of
104         // previous even index
105         for (int i = 1; i < counters.length; i += 2) {
106             if (counters[i].equals(szNameBuffer)) {
107                 try {
108                     return Integer.parseInt(counters[i - 1]);
109                 } catch (NumberFormatException e) {
110                     // Unexpected but handle anyway
111                     return 0;
112                 }
113             }
114         }
115         // Didn't find the String
116         return 0;
117     }

119     /**
120      * Utility method to call Pdh's PdhEnumObjectItems that allocates the
121      * required memory for the lists parameters based on the type mapping used,
122      * calls to PdhEnumObjectItems, and returns the received lists of strings.
123      * 
124      * @param szDataSource
125      *            String that specifies the name of the log file used to
126      *            enumerate the counter and instance names. If NULL, the
127      *            function uses the computer specified in the szMachineName
128      *            parameter to enumerate the names.
129      * @param szMachineName
130      *            String that specifies the name of the computer that contains
131      *            the counter and instance names that you want to enumerate.
132      *            Include the leading slashes in the computer name, for example,
133      *            \\computername. If the szDataSource parameter is NULL, you can
134      *            set szMachineName to NULL to specify the local computer.
135      * @param szObjectName
136      *            String that specifies the name of the object whose counter and
137      *            instance names you want to enumerate.
138      * @param dwDetailLevel
139      *            Detail level of the performance items to return. All items
140      *            that are of the specified detail level or less will be
141      *            returned.
142      * @return Returns a List of Strings of the counters for the object.
143      */
144     public static PdhEnumObjectItems PdhEnumObjectItems(String szDataSource, String szMachineName, String szObjectName,
145             int dwDetailLevel) {
146         List<String> counters = new ArrayList<>();
147         List<String> instances = new ArrayList<>();
148 
149         // Call once to get string lengths
150         DWORDByReference pcchCounterListLength = new DWORDByReference(new DWORD(0));
151         DWORDByReference pcchInstanceListLength = new DWORDByReference(new DWORD(0));
152         int result = Pdh.INSTANCE.PdhEnumObjectItems(szDataSource, szMachineName, szObjectName, null,
153                 pcchCounterListLength, null, pcchInstanceListLength, dwDetailLevel, 0);
154         if (result != WinError.ERROR_SUCCESS && result != Pdh.PDH_MORE_DATA) {
155             throw new PdhException(result);
156         }
157 
158         Memory mszCounterList = null;
159         Memory mszInstanceList = null;
160 
161         if (pcchCounterListLength.getValue().longValue() > 0) {
162             mszCounterList = new Memory(pcchCounterListLength.getValue().longValue() * CHAR_TO_BYTES);
163         }
164 
165         if (pcchInstanceListLength.getValue().longValue() > 0) {
166             mszInstanceList = new Memory(pcchInstanceListLength.getValue().longValue() * CHAR_TO_BYTES);
167         }
168 
169         result = Pdh.INSTANCE.PdhEnumObjectItems(szDataSource, szMachineName, szObjectName, mszCounterList,
170                 pcchCounterListLength, mszInstanceList, pcchInstanceListLength, dwDetailLevel, 0);
171 
172         if (result != WinError.ERROR_SUCCESS) {
173             throw new PdhException(result);
174         }
175 
176         // Fetch counters
177         if (mszCounterList != null) {
178             int offset = 0;
179             while (offset < mszCounterList.size()) {
180                 String s = null;
181                 if (CHAR_TO_BYTES == 1) {
182                     s = mszCounterList.getString(offset);
183                 } else {
184                     s = mszCounterList.getWideString(offset);
185                 }
186                 // list ends with double null
187                 if (s.isEmpty()) {
188                     break;
189                 }
190                 counters.add(s);
191                 // Increment for string + null terminator
192                 offset += (s.length() + 1) * CHAR_TO_BYTES;
193             }
194         }
195 
196         if (mszInstanceList != null) {
197             int offset = 0;
198             while (offset < mszInstanceList.size()) {
199                 String s = null;
200                 if (CHAR_TO_BYTES == 1) {
201                     s = mszInstanceList.getString(offset);
202                 } else {
203                     s = mszInstanceList.getWideString(offset);
204                 }
205                 // list ends with double null
206                 if (s.isEmpty()) {
207                     break;
208                 }
209                 instances.add(s);
210                 // Increment for string + null terminator
211                 offset += (s.length() + 1) * CHAR_TO_BYTES;
212             }
213         }
214 
215         return new PdhEnumObjectItems(counters, instances);
216     }

218     /**
219      * Holder Object for PdhEnumObjectsItems. The embedded lists are modifiable
220      * lists and can be accessed through the {@link #getCounters()} and
221      * {@link #getInstances()} accessors.
222      */
223     public static class PdhEnumObjectItems {
224         private final List<String> counters;
225         private final List<String> instances;
226 
227         public PdhEnumObjectItems(List<String> counters, List<String> instances) {
228             this.counters = copyAndEmptyListForNullList(counters);
229             this.instances = copyAndEmptyListForNullList(instances);
230         }
231 
232         /**
233          * @return the embedded counters list, all calls to this function
234          *         receive the same list and thus share modifications
235          */
236         public List<String> getCounters() {
237             return counters;
238         }
239 
240         /**
241          * @return the embedded instances list, all calls to this function
242          *         receive the same list and thus share modifications
243          */
244         public List<String> getInstances() {
245             return instances;
246         }
247 
248         private List<String> copyAndEmptyListForNullList(List<String> inputList) {
249             if (inputList == null) {
250                 return new ArrayList<>();
251             } else {
252                 return new ArrayList<>(inputList);
253             }
254         }
255 
256         @Override
257         public String toString() {
258             return "PdhEnumObjectItems{" + "counters=" + counters + ", instances=" + instances + '}';
259         }
260     }

262     @SuppressWarnings("serial")
263     public static final class PdhException extends RuntimeException {
264         private final int errorCode;
265 
266         public PdhException(int errorCode) {
267             super(String.format("Pdh call failed with error code 0x%08X", errorCode));
268             this.errorCode = errorCode;
269         }
270 
271         public int getErrorCode() {
272             return errorCode;
273         }
274     }
