68      /**
69       * IOKitLib implements non-kernel task access to common IOKit object types -
70       * IORegistryEntry, IOService, IOIterator etc. These functions are generic -
71       * families may provide API that is more specific.
72       * <p>
73       * IOKitLib represents IOKit objects outside the kernel with the types
74       * io_object_t, io_registry_entry_t, io_service_t, and io_connect_t. Function
75       * names usually begin with the type of object they are compatible with - e.g.,
76       * IOObjectRelease can be used with any io_object_t. Inside the kernel, the c++
77       * class hierarchy allows the subclasses of each object type to receive the same
78       * requests from user level clients, for example in the kernel, IOService is a
79       * subclass of IORegistryEntry, which means any of the IORegistryEntryXXX
80       * functions in IOKitLib may be used with io_service_t's as well as
81       * io_registry_t's. There are functions available to introspect the class of the
82       * kernel object which any io_object_t et al. represents. IOKit objects returned
83       * by all functions should be released with {@link IOKit#IOObjectRelease}.
84       */
85      class IOObject extends PointerType {
86          public IOObject() {
87              super();
88          }
89  
90          public IOObject(Pointer p) {
91              super(p);
92          }
93  
94          /**
95           * Convenience method for {@link IOKit#IOObjectConformsTo} on this object.
96           *
97           * @param className
98           *            The name of the class.
99           * @return If the object handle is valid, and represents an object in the kernel
100          *         that dynamic casts to the class true is returned, otherwise false.
101          */
102         public boolean conformsTo(String className) {
103             return INSTANCE.IOObjectConformsTo(this, className);
104         }
105 
106         /**
107          * Convenience method for {@link IOKit#IOObjectRelease} on this object.
108          *
109          * @return 0 if successful, otherwise a {@code kern_return_t} error code.
110          */
111         public int release() {
112             return INSTANCE.IOObjectRelease(this);
113         }

307         /**
308          * Convenience method to get a {@link java.lang.String} value from this IO
309          * Registry Entry.
310          *
311          * @param key
312          *            The string name of the key to retrieve
313          * @return The value of the registry entry if it exists; {@code null} otherwise
314          */
315         public String getStringProperty(String key) {
316             String value = null;
317             CFStringRef keyAsCFString = CFStringRef.createCFString(key);
318             CFTypeRef valueAsCFType = this.createCFProperty(keyAsCFString);
319             keyAsCFString.release();
320             if (valueAsCFType != null) {
321                 CFStringRef valueAsCFString = new CFStringRef(valueAsCFType.getPointer());
322                 value = valueAsCFString.stringValue();
323                 valueAsCFType.release();
324             }
325             return value;
326         }
