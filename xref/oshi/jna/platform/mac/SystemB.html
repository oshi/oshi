5   package oshi.jna.platform.mac;
6   
7   import com.sun.jna.Native;
8   import com.sun.jna.NativeLong;
9   import com.sun.jna.Structure;
10  import com.sun.jna.Structure.FieldOrder;
11  
12  import oshi.jna.platform.unix.CLibrary;
13  
14  /**
15   * System class. This class should be considered non-API as it may be removed if/when its code is incorporated into the
16   * JNA project.
17   */
18  public interface SystemB extends com.sun.jna.platform.mac.SystemB, CLibrary {
19  
20      SystemB INSTANCE = Native.load("System", SystemB.class);
21  
22      int UTX_USERSIZE = 256;
23      int UTX_LINESIZE = 32;
24      int UTX_IDSIZE = 4;
25      int UTX_HOSTSIZE = 256;
26  
27      /**
28       * Mac connection info
29       */
30      @FieldOrder({ "ut_user", "ut_id", "ut_line", "ut_pid", "ut_type", "ut_tv", "ut_host", "ut_pad" })
31      class MacUtmpx extends Structure {
32          public byte[] ut_user = new byte[UTX_USERSIZE]; // login name
33          public byte[] ut_id = new byte[UTX_IDSIZE]; // id
34          public byte[] ut_line = new byte[UTX_LINESIZE]; // tty name
35          public int ut_pid; // process id creating the entry
36          public short ut_type; // type of this entry
37          public Timeval ut_tv; // time entry was created
38          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
39          public byte[] ut_pad = new byte[16]; // reserved for future use
40      }
41  
42      /**
43       * Reads a line from the current file position in the utmp file. It returns a pointer to a structure containing the
44       * fields of the line.
45       * <p>
46       * Not thread safe
47       *
48       * @return a {@link MacUtmpx} on success, and NULL on failure (which includes the "record not found" case)
49       */
50      MacUtmpx getutxent();
51  
52      @FieldOrder({ "ru_utime_sec", "ru_utime_usec", "ru_stime_sec", "ru_stime_usec", "ru_maxrss", "ru_ixrss", "ru_idrss",
53              "ru_isrss", "ru_minflt", "ru_majflt", "ru_nswap", "ru_inblock", "ru_oublock", "ru_msgsnd", "ru_msgrcv",
54              "ru_nsignals", "ru_nvcsw", "ru_nivcsw" })
55      class Rusage extends Structure {
56          public NativeLong ru_utime_sec;
57          public int ru_utime_usec;
58          public NativeLong ru_stime_sec;
59          public int ru_stime_usec;
60          public NativeLong ru_maxrss;
61          public NativeLong ru_ixrss;
62          public NativeLong ru_idrss;
63          public NativeLong ru_isrss;
64          public NativeLong ru_minflt;
65          public NativeLong ru_majflt;
66          public NativeLong ru_nswap;
67          public NativeLong ru_inblock;
68          public NativeLong ru_oublock;
69          public NativeLong ru_msgsnd;
70          public NativeLong ru_msgrcv;
71          public NativeLong ru_nsignals;
72          public NativeLong ru_nvcsw;
73          public NativeLong ru_nivcsw;
74      }
75  
76      int getrusage(int who, Rusage rusage);
77  }
