19  package oshi.jna.platform.mac;
20  
21  import com.sun.jna.Library;
22  import com.sun.jna.Native;
23  import com.sun.jna.PointerType;
24  
25  import oshi.jna.platform.mac.CoreFoundation.CFAllocatorRef;
26  import oshi.jna.platform.mac.CoreFoundation.CFDictionaryRef;
27  
28  /**
29   * DiskArbitration framework for disk stats. This class should be considered
30   * non-API as it may be removed if/when its code is incorporated into the JNA
31   * project.
32   *
33   * @author widdis[at]gmail[dot]com
34   */
35  public interface DiskArbitration extends Library {
36      DiskArbitration INSTANCE = Native.loadLibrary("DiskArbitration", DiskArbitration.class);
37  
38      class DASessionRef extends PointerType {
39      }
40  
41      class DADiskRef extends PointerType {
42      }
43  
44      /**
45       * Creates a new session. The caller of this function receives a reference
46       * to the returned object. The caller also implicitly retains the object and
47       * is responsible for releasing it with CFRelease().
48       *
49       * @param allocator
50       *            The allocator object to be used to allocate memory.
51       * @return A reference to a new DASession.
52       */
53      DASessionRef DASessionCreate(CFAllocatorRef allocator);
54  
55      /**
56       * Creates a new disk object. The caller of this function receives a
57       * reference to the returned object. The caller also implicitly retains the
58       * object and is responsible for releasing it with CFRelease().
59       *
60       * @param allocator
61       *            The allocator object to be used to allocate memory.
62       * @param session
63       *            The DASession in which to contact Disk Arbitration.
64       * @param diskName
65       *            he BSD device name.
66       * @return A reference to a new DADisk.
67       */
68      DADiskRef DADiskCreateFromBSDName(CFAllocatorRef allocator, DASessionRef session, String diskName);
69  
70      /**
71       * Creates a new disk object. The caller of this function receives a
72       * reference to the returned object. The caller also implicitly retains the
73       * object and is responsible for releasing it with CFRelease().
74       *
75       * @param allocator
76       *            The allocator object to be used to allocate memory.
77       * @param session
78       *            The DASession in which to contact Disk Arbitration.
79       * @param media
80       *            The I/O Kit media object.
81       * @return A reference to a new DADisk.
82       */
83      DADiskRef DADiskCreateFromIOMedia(CFAllocatorRef allocator, DASessionRef session, int media);
84  
85      /**
86       * Obtains the Disk Arbitration description of the specified disk. This
87       * function will contact Disk Arbitration to acquire the latest description
88       * of the specified disk, unless this function is called on a disk object
89       * passed within the context of a registered callback, in which case the
90       * description is current as of that callback event.
91       *
92       * The caller of this function receives a reference to the returned object.
93       * The caller also implicitly retains the object and is responsible for
94       * releasing it with CFRelease().
95       *
96       * @param disk
97       *            The DADisk for which to obtain the Disk Arbitration
98       *            description.
99       * @return The disk's Disk Arbitration description.
100      */
101     CFDictionaryRef DADiskCopyDescription(DADiskRef disk);
102 
103     /**
104      * Obtains the BSD device name for the specified disk.
105      *
106      * @param disk
107      *            The DADisk for which to obtain the BSD device name.
108      *
109      * @return The disk's BSD device name.
110      */
111     String DADiskGetBSDName(DADiskRef disk);
112 }
