44      int UTX_USERSIZE = 32;
45      int UTX_LINESIZE = 32;
46      int UTX_IDSIZE = 4;
47      int UTX_HOSTSIZE = 257;
48  
49      int PRCLSZ = 8;
50      int PRFNSZ = 16;
51      int PRLNSZ = 32;
52      int PRARGSZ = 80;
53  
54      /**
55       * Connection info
56       */
57      @FieldOrder({ "ut_user", "ut_id", "ut_line", "ut_pid", "ut_type", "ut_exit", "ut_tv", "ut_session", "pad",
58              "ut_syslen", "ut_host" })
59      class SolarisUtmpx extends Structure {
60          public byte[] ut_user = new byte[UTX_USERSIZE]; // user login name
61          public byte[] ut_id = new byte[UTX_IDSIZE]; // etc/inittab id (usually line #)
62          public byte[] ut_line = new byte[UTX_LINESIZE]; // device name
63          public int ut_pid; // process id
64          public short ut_type; // type of entry
65          public Exit_status ut_exit; // process termination/exit status
66          public Timeval ut_tv; // time entry was made
67          public int ut_session; // session ID, used for windowing
68          public int[] pad = new int[5]; // reserved for future use
69          public short ut_syslen; // significant length of ut_host including terminating null
70          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
71      }
72  
73      /**
74       * Part of utmpx structure
75       */
76      @FieldOrder({ "e_termination", "e_exit" })
77      class Exit_status extends Structure {
78          public short e_termination; // Process termination status
79          public short e_exit; // Process exit status
80      }
81  
82      /**
83       * 32/64-bit timeval required for utmpx structure
84       */
85      @FieldOrder({ "tv_sec", "tv_usec" })
86      class Timeval extends Structure {
87          public NativeLong tv_sec; // seconds
88          public NativeLong tv_usec; // microseconds
89      }

96       *
97       * @return a {@link SolarisUtmpx} on success, and NULL on failure (which

100     SolarisUtmpx getutxent();

102     /**
103      * Structure for psinfo file
104      */
105     class SolarisPsInfo {
106         public int pr_flag; // process flags (DEPRECATED; do not use)
107         public int pr_nlwp; // number of active lwps in the process
108         public int pr_pid; // unique process id
109         public int pr_ppid; // process id of parent
110         public int pr_pgid; // pid of process group leader
111         public int pr_sid; // session id
112         public int pr_uid; // real user id
113         public int pr_euid; // effective user id
114         public int pr_gid; // real group id
115         public int pr_egid; // effective group id
116         public Pointer pr_addr; // address of process
117         public size_t pr_size; // size of process image in Kbytes
118         public size_t pr_rssize; // resident set size in Kbytes
119         public size_t pr_rssizepriv; // resident set size of private mappings
120         public NativeLong pr_ttydev; // controlling tty device (or PRNODEV)
121         // The following percent numbers are 16-bit binary
122         // fractions [0 .. 1] with the binary point to the
123         // right of the high-order bit (1.0 == 0x8000)
124         public short pr_pctcpu; // % of recent cpu time used by all lwps
125         public short pr_pctmem; // % of system memory used by process
126         public Timestruc pr_start; // process start time, from the epoch
127         public Timestruc pr_time; // usr+sys cpu time for this process
128         public Timestruc pr_ctime; // usr+sys cpu time for reaped children
129         public byte[] pr_fname = new byte[PRFNSZ]; // name of exec'ed file
130         public byte[] pr_psargs = new byte[PRARGSZ]; // initial characters of arg list
131         public int pr_wstat; // if zombie, the wait() status
132         public int pr_argc; // initial argument count
133         public Pointer pr_argv; // address of initial argument vector
134         public Pointer pr_envp; // address of initial environment vector
135         public byte pr_dmodel; // data model of the process
136         public byte[] pr_pad2 = new byte[3];
137         public int pr_taskid; // task id
138         public int pr_projid; // project id
139         public int pr_nzomb; // number of zombie lwps in the process
140         public int pr_poolid; // pool id
141         public int pr_zoneid; // zone id
142         public int pr_contract; // process contract id
143         public int pr_filler; // 4 bytes reserved for future use
144         public SolarisLwpsInfo pr_lwp; // information for representative lwp
145 
146         public SolarisPsInfo(ByteBuffer buff) {
147             this.pr_flag = FileUtil.readIntFromBuffer(buff);
148             this.pr_nlwp = FileUtil.readIntFromBuffer(buff);
149             this.pr_pid = FileUtil.readIntFromBuffer(buff);
150             this.pr_ppid = FileUtil.readIntFromBuffer(buff);
151             this.pr_pgid = FileUtil.readIntFromBuffer(buff);
152             this.pr_sid = FileUtil.readIntFromBuffer(buff);
153             this.pr_uid = FileUtil.readIntFromBuffer(buff);
154             this.pr_euid = FileUtil.readIntFromBuffer(buff);
155             this.pr_gid = FileUtil.readIntFromBuffer(buff);
156             this.pr_egid = FileUtil.readIntFromBuffer(buff);
157             this.pr_addr = FileUtil.readPointerFromBuffer(buff);
158             this.pr_size = FileUtil.readSizeTFromBuffer(buff);
159             this.pr_rssize = FileUtil.readSizeTFromBuffer(buff);
160             this.pr_rssizepriv = FileUtil.readSizeTFromBuffer(buff);
161             this.pr_ttydev = FileUtil.readNativeLongFromBuffer(buff);
162             this.pr_pctcpu = FileUtil.readShortFromBuffer(buff);
163             this.pr_pctmem = FileUtil.readShortFromBuffer(buff);
164             // Force 8 byte alignment
165             if (Native.LONG_SIZE > 4) {
166                 FileUtil.readIntFromBuffer(buff);
167             }
168             this.pr_start = new Timestruc(buff);
169             this.pr_time = new Timestruc(buff);
170             this.pr_ctime = new Timestruc(buff);
171             FileUtil.readByteArrayFromBuffer(buff, this.pr_fname);
172             FileUtil.readByteArrayFromBuffer(buff, this.pr_psargs);
173             this.pr_wstat = FileUtil.readIntFromBuffer(buff);
174             this.pr_argc = FileUtil.readIntFromBuffer(buff);
175             this.pr_argv = FileUtil.readPointerFromBuffer(buff);
176             this.pr_envp = FileUtil.readPointerFromBuffer(buff);
177             this.pr_dmodel = FileUtil.readByteFromBuffer(buff);
178             FileUtil.readByteArrayFromBuffer(buff, this.pr_pad2);
179             this.pr_taskid = FileUtil.readIntFromBuffer(buff);
180             this.pr_projid = FileUtil.readIntFromBuffer(buff);
181             this.pr_nzomb = FileUtil.readIntFromBuffer(buff);
182             this.pr_poolid = FileUtil.readIntFromBuffer(buff);
183             this.pr_zoneid = FileUtil.readIntFromBuffer(buff);
184             this.pr_contract = FileUtil.readIntFromBuffer(buff);
185             this.pr_filler = FileUtil.readIntFromBuffer(buff);
186             this.pr_lwp = new SolarisLwpsInfo(buff);
187         }
188     }

190     /**
191      * Nested Structure for psinfo file
192      */
193     class SolarisLwpsInfo {
194         public int pr_flag; // lwp flags (DEPRECATED; do not use)
195         public int pr_lwpid; // lwp id
196         public Pointer pr_addr; // DEPRECATED was internal address of lwp
197         public Pointer pr_wchan; // DEPRECATED was wait addr for sleeping lwp
198         public byte pr_stype; // synchronization event type
199         public byte pr_state; // numeric lwp state
200         public byte pr_sname; // printable character for pr_state
201         public byte pr_nice; // nice for cpu usage
202         public short pr_syscall; // system call number (if in syscall)
203         public byte pr_oldpri; // pre-SVR4, low value is high priority
204         public byte pr_cpu; // pre-SVR4, cpu usage for scheduling
205         public int pr_pri; // priority, high value = high priority
206         // The following percent numbers are 16-bit binary
207         // fractions [0 .. 1] with the binary point to the
208         // right of the high-order bit (1.0 == 0x8000)
209         public short pr_pctcpu; // % of recent cpu time used by this lwp
210         public short pr_pad;
211         public Timestruc pr_start; // lwp start time, from the epoch
212         public Timestruc pr_time; // cpu time for this lwp
213         public byte[] pr_clname = new byte[PRCLSZ]; // scheduling class name
214         public byte[] pr_oldname = new byte[PRFNSZ]; // binary compatibility -- unused
215         public int pr_onpro; // processor which last ran this lwp
216         public int pr_bindpro; // processor to which lwp is bound
217         public int pr_bindpset; // processor set to which lwp is bound
218         public int pr_lgrp; // home lgroup
219         public long pr_last_onproc; // Timestamp of when thread last ran on a processor
220         public byte[] pr_name = new byte[PRLNSZ]; // name of system lwp
221 
222         public SolarisLwpsInfo(ByteBuffer buff) {
223             this.pr_flag = FileUtil.readIntFromBuffer(buff);
224             this.pr_lwpid = FileUtil.readIntFromBuffer(buff);
225             this.pr_addr = FileUtil.readPointerFromBuffer(buff);
226             this.pr_wchan = FileUtil.readPointerFromBuffer(buff);
227             this.pr_stype = FileUtil.readByteFromBuffer(buff);
228             this.pr_state = FileUtil.readByteFromBuffer(buff);
229             this.pr_sname = FileUtil.readByteFromBuffer(buff);
230             this.pr_nice = FileUtil.readByteFromBuffer(buff);
231             this.pr_syscall = FileUtil.readShortFromBuffer(buff);
232             this.pr_oldpri = FileUtil.readByteFromBuffer(buff);
233             this.pr_cpu = FileUtil.readByteFromBuffer(buff);
234             this.pr_pri = FileUtil.readIntFromBuffer(buff);
235             this.pr_pctcpu = FileUtil.readShortFromBuffer(buff);
236             this.pr_pad = FileUtil.readShortFromBuffer(buff);
237             this.pr_start = new Timestruc(buff);
238             this.pr_time = new Timestruc(buff);
239             FileUtil.readByteArrayFromBuffer(buff, this.pr_clname);
240             FileUtil.readByteArrayFromBuffer(buff, this.pr_oldname);
241             this.pr_onpro = FileUtil.readIntFromBuffer(buff);
242             this.pr_bindpro = FileUtil.readIntFromBuffer(buff);
243             this.pr_bindpset = FileUtil.readIntFromBuffer(buff);
244             this.pr_lgrp = FileUtil.readIntFromBuffer(buff);
245             this.pr_last_onproc = FileUtil.readLongFromBuffer(buff);
246             FileUtil.readByteArrayFromBuffer(buff, this.pr_name);
247         }
248     }

250     /**
251      * Structure for usage file
252      */
253     class SolarisPrUsage {
254         public int pr_lwpid; // lwp id. 0: process or defunct
255         public int pr_count; // number of contributing lwps
256         public Timestruc pr_tstamp; // current time stamp
257         public Timestruc pr_create; // process/lwp creation time stamp
258         public Timestruc pr_term; // process/lwp termination time stamp
259         public Timestruc pr_rtime; // total lwp real (elapsed) time
260         public Timestruc pr_utime; // user level cpu time
261         public Timestruc pr_stime; // system call cpu time
262         public Timestruc pr_ttime; // other system trap cpu time
263         public Timestruc pr_tftime; // text page fault sleep time
264         public Timestruc pr_dftime; // data page fault sleep time
265         public Timestruc pr_kftime; // kernel page fault sleep time
266         public Timestruc pr_ltime; // user lock wait sleep time
267         public Timestruc pr_slptime; // all other sleep time
268         public Timestruc pr_wtime; // wait-cpu (latency) time
269         public Timestruc pr_stoptime; // stopped time
270         public Timestruc[] filltime = new Timestruc[6]; // filler for future expansion
271         public NativeLong pr_minf; // minor page faults
272         public NativeLong pr_majf; // major page faults
273         public NativeLong pr_nswap; // swaps
274         public NativeLong pr_inblk; // input blocks
275         public NativeLong pr_oublk; // output blocks
276         public NativeLong pr_msnd; // messages sent
277         public NativeLong pr_mrcv; // messages received
278         public NativeLong pr_sigs; // signals received
279         public NativeLong pr_vctx; // voluntary context switches
280         public NativeLong pr_ictx; // involuntary context switches
281         public NativeLong pr_sysc; // system calls
282         public NativeLong pr_ioch; // chars read and written
283         public NativeLong[] filler = new NativeLong[10]; // filler for future expansion
284 
285         public SolarisPrUsage(ByteBuffer buff) {
286             this.pr_lwpid = FileUtil.readIntFromBuffer(buff);
287             this.pr_count = FileUtil.readIntFromBuffer(buff);
288             this.pr_tstamp = new Timestruc(buff);
289             this.pr_create = new Timestruc(buff);
290             this.pr_term = new Timestruc(buff);
291             this.pr_rtime = new Timestruc(buff);
292             this.pr_utime = new Timestruc(buff);
293             this.pr_stime = new Timestruc(buff);
294             this.pr_ttime = new Timestruc(buff);
295             this.pr_tftime = new Timestruc(buff);
296             this.pr_dftime = new Timestruc(buff);
297             this.pr_kftime = new Timestruc(buff);
298             this.pr_ltime = new Timestruc(buff);
299             this.pr_slptime = new Timestruc(buff);
300             this.pr_wtime = new Timestruc(buff);
301             this.pr_stoptime = new Timestruc(buff);
302             for (int i = 0; i < filltime.length; i++) {
303                 this.filltime[i] = new Timestruc(buff);
304             }
305             this.pr_minf = FileUtil.readNativeLongFromBuffer(buff);
306             this.pr_majf = FileUtil.readNativeLongFromBuffer(buff);
307             this.pr_nswap = FileUtil.readNativeLongFromBuffer(buff);
308             this.pr_inblk = FileUtil.readNativeLongFromBuffer(buff);
309             this.pr_oublk = FileUtil.readNativeLongFromBuffer(buff);
310             this.pr_msnd = FileUtil.readNativeLongFromBuffer(buff);
311             this.pr_mrcv = FileUtil.readNativeLongFromBuffer(buff);
312             this.pr_sigs = FileUtil.readNativeLongFromBuffer(buff);
313             this.pr_vctx = FileUtil.readNativeLongFromBuffer(buff);
314             this.pr_ictx = FileUtil.readNativeLongFromBuffer(buff);
315             this.pr_sysc = FileUtil.readNativeLongFromBuffer(buff);
316             this.pr_ioch = FileUtil.readNativeLongFromBuffer(buff);
317             for (int i = 0; i < filler.length; i++) {
318                 this.filler[i] = FileUtil.readNativeLongFromBuffer(buff);
319             }
320         }
321     }
