24  package oshi.jna.platform.unix;
25  
26  import com.sun.jna.Native; // NOSONAR squid:S1191
27  import com.sun.jna.Structure;
28  import com.sun.jna.Structure.FieldOrder;
29  
30  /**
31   * C library. This class should be considered non-API as it may be removed
32   * if/when its code is incorporated into the JNA project.
33   */
34  public interface FreeBsdLibc extends CLibrary {
35      FreeBsdLibc INSTANCE = Native.load("libc", FreeBsdLibc.class);
36  
37      int UTX_USERSIZE = 32;
38      int UTX_LINESIZE = 16;
39      int UTX_IDSIZE = 8;
40      int UTX_HOSTSIZE = 128;
41  
42      /**
43       * Connection info
44       */
45      @FieldOrder({ "ut_type", "ut_tv", "ut_id", "ut_pid", "ut_user", "ut_line", "ut_host", "ut_spare" })
46      class FreeBsdUtmpx extends Structure {
47          public short ut_type; // type of entry
48          public Timeval ut_tv; // time entry was made
49          public byte[] ut_id = new byte[UTX_IDSIZE]; // etc/inittab id (usually line #)
50          public int ut_pid; // process id
51          public byte[] ut_user = new byte[UTX_USERSIZE]; // user login name
52          public byte[] ut_line = new byte[UTX_LINESIZE]; // device name
53          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
54          public byte[] ut_spare = new byte[64];
55      }
56  
57      /*
58       * Data size
59       */
60      /** Constant <code>UINT64_SIZE=Native.getNativeSize(long.class)</code> */
61      int UINT64_SIZE = Native.getNativeSize(long.class);
62      /** Constant <code>INT_SIZE=Native.getNativeSize(int.class)</code> */
63      int INT_SIZE = Native.getNativeSize(int.class);
64  
65      /*
66       * CPU state indices
67       */
68      /** Constant <code>CPUSTATES=5</code> */
69      int CPUSTATES = 5;
70      /** Constant <code>CP_USER=0</code> */
71      int CP_USER = 0;
72      /** Constant <code>CP_NICE=1</code> */
73      int CP_NICE = 1;
74      /** Constant <code>CP_SYS=2</code> */
75      int CP_SYS = 2;
76      /** Constant <code>CP_INTR=3</code> */
77      int CP_INTR = 3;
78      /** Constant <code>CP_IDLE=4</code> */
79      int CP_IDLE = 4;
80  
81      /**
82       * Return type for BSD sysctl kern.boottime
83       */
84      @FieldOrder({ "tv_sec", "tv_usec" })
85      class Timeval extends Structure {
86          public long tv_sec; // seconds
87          public long tv_usec; // microseconds
88      }
89  
90      /**
91       * CPU Ticks
92       */
93      @FieldOrder({ "cpu_ticks" })
94      class CpTime extends Structure {
95          public long[] cpu_ticks = new long[CPUSTATES];
96      }
97  
98      /**
99       * Reads a line from the current file position in the utmp file. It returns a
100      * pointer to a structure containing the fields of the line.
101      * <p>
102      * Not thread safe
103      *
104      * @return a {@link FreeBsdUtmpx} on success, and NULL on failure (which
105      *         includes the "record not found" case)
106      */
107     FreeBsdUtmpx getutxent();
108 }
