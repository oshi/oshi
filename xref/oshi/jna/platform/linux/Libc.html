33  /**
34   * Linux C Library. This class should be considered non-API as it may be removed
35   * if/when its code is incorporated into the JNA project.
36   *
37   * @author widdis[at]gmail[dot]com
38   */
39  public interface Libc extends CLibrary {
40  
41      Libc INSTANCE = Native.load("c", Libc.class);
42  
43      @FieldOrder({ "uptime", "loads", "totalram", "freeram", "sharedram", "bufferram", "totalswap", "freeswap", "procs",
44              "totalhigh", "freehigh", "mem_unit", "_f" })
45      class Sysinfo extends Structure {
46          private static final int PADDING_SIZE = 20 - 2 * NativeLong.SIZE - 4;
47  
48          public NativeLong uptime; // Seconds since boot
49          // 1, 5, and 15 minute load averages, divide by 2^16
50          public NativeLong[] loads = new NativeLong[3];
51          public NativeLong totalram; // Total usable main memory size
52          public NativeLong freeram; // Available memory size
53          public NativeLong sharedram; // Amount of shared memory
54          public NativeLong bufferram; // Memory used by buffers
55          public NativeLong totalswap; // Total swap space size
56          public NativeLong freeswap; // swap space still available
57          public short procs; // Number of current processes
58          public NativeLong totalhigh; // Total high memory size
59          public NativeLong freehigh; // Available high memory size
60          public int mem_unit; // Memory unit size in bytes
61          // Padding to 64 bytes
62          public byte[] _f = new byte[PADDING_SIZE];
63  
64          /*
65           * getFieldList and getFieldOrder are overridden because PADDING_SIZE
66           * might be 0 - that is a GCC only extension and not supported by JNA
67           * 
68           * The dummy field at the end of the structure is just padding and so if
69           * the field is the zero length array, it is stripped from the fields
70           * and field order.
71           */
72          @Override
73          protected List<Field> getFieldList() {
74              List<Field> fields = new ArrayList<>(super.getFieldList());
75              if (PADDING_SIZE == 0) {
76                  Iterator<Field> fieldIterator = fields.iterator();
77                  while (fieldIterator.hasNext()) {
78                      Field field = fieldIterator.next();
79                      if ("_f".equals(field.getName())) {
80                          fieldIterator.remove();
81                      }
82                  }
83              }
84              return fields;
85          }
86  
87          @Override
88          protected List<String> getFieldOrder() {
89              List<String> fieldOrder = new ArrayList<>(super.getFieldOrder());
90              if (PADDING_SIZE == 0) {
91                  fieldOrder.remove("_f");
92              }
93              return fieldOrder;
94          }
95      }
96  
97      @FieldOrder({ "fsBlockSize", "fsFragmentSize", "fsSizeInBlocks", "fsBlocksFree", "fsBlocksFreeUnpriv",
98              "fsTotalInodeCount", "fsFreeInodeCount", "fsFreeInodeCountUnpriv", "fsId", "_fPad32bit", "fsMountFlags",
99              "fsMaxFilenameLength", "_fSpare" })
100     class Statvfs extends Structure {
101         private static final int PADDING_SIZE = 8 / NativeLong.SIZE - 1;
102 
103         public NativeLong fsBlockSize;
104         public NativeLong fsFragmentSize;
105         public NativeLong fsSizeInBlocks;
106         public NativeLong fsBlocksFree;
107         public NativeLong fsBlocksFreeUnpriv;
108         public NativeLong fsTotalInodeCount;
109         public NativeLong fsFreeInodeCount;
110         public NativeLong fsFreeInodeCountUnpriv;
111         public NativeLong fsId;
112         public int[] _fPad32bit = new int[PADDING_SIZE]; // 0 on 64-bit systems
113         public NativeLong fsMountFlags;
114         public NativeLong fsMaxFilenameLength;
115         public int[] _fSpare = new int[6];
116 
117         @Override
118         protected List<Field> getFieldList() {
119             List<Field> fields = new ArrayList<Field>(super.getFieldList());
120             if (PADDING_SIZE == 0) {
121                 Iterator<Field> fieldIterator = fields.iterator();
122                 while (fieldIterator.hasNext()) {
123                     Field field = fieldIterator.next();
124                     if ("_fPad32bit".equals(field.getName())) {
125                         fieldIterator.remove();
126                     }
127                 }
128             }
129             return fields;
130         }
131 
132         @Override
133         protected List<String> getFieldOrder() {
134             List<String> fieldOrder = new ArrayList<>(super.getFieldOrder());
135             if (PADDING_SIZE == 0) {
136                 fieldOrder.remove("_fPad32bit");
137             }
138             return fieldOrder;
139         }
140     }
141 
142     /**
143      * sysinfo() provides a simple way of getting overall system statistics.
144      * This is more portable than reading /dev/kmem.
145      * 
146      * @param info
147      *            A Sysinfo structure which will be populated
148      * @return On success, zero is returned. On error, -1 is returned, and errno
149      *         is set appropriately.
150      */
151     int sysinfo(Sysinfo info);
152 
153     /**
154      * The function statvfs() returns information about a mounted filesystem.
155      * 
156      * @param path
157      *            the pathname of any file within the mounted filesystem.
158      * @param buf
159      *            a pointer to a statvfs structure
160      * @return On success, zero is returned. On error, -1 is returned, and errno
161      *         is set appropriately.
162      */
163     int statvfs(String path, Statvfs buf);
164 }
