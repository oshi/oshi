19  package oshi.jna.platform.linux;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import com.sun.jna.Library;
25  import com.sun.jna.Native;
26  import com.sun.jna.NativeLong;
27  import com.sun.jna.Pointer;
28  import com.sun.jna.Structure;
29  
30  /**
31   * Linux C Library. This class should be considered non-API as it may be removed
32   * if/when its code is incorporated into the JNA project.
33   *
34   * @author widdis[at]gmail[dot]com
35   */
36  public interface Libc extends Library {
37  
38      Libc INSTANCE = (Libc) Native.loadLibrary("c", Libc.class);
39  
40      class Sysinfo extends Structure {
41          public NativeLong uptime; // Seconds since boot
42          // 1, 5, and 15 minute load averages
43  
44          public NativeLong[] loads = new NativeLong[3];
45  
46          public NativeLong totalram; // Total usable main memory size
47  
48          public NativeLong freeram; // Available memory size
49  
50          public NativeLong sharedram; // Amount of shared memory
51  
52          public NativeLong bufferram; // Memory used by buffers
53  
54          public NativeLong totalswap; // Total swap space size
55  
56          public NativeLong freeswap; // swap space still available
57  
58          public short procs; // Number of current processes
59  
60          public NativeLong totalhigh; // Total high memory size
61  
62          public NativeLong freehigh; // Available high memory size
63  
64          public int mem_unit; // Memory unit size in bytes
65  
66          public byte[] _f = new byte[8]; // Won't be written for 64-bit systems
67  
68          @Override
69          protected List<String> getFieldOrder() {
70              return Arrays.asList(new String[] { "uptime", "loads", "totalram", "freeram", "sharedram", "bufferram",
71                      "totalswap", "freeswap", "procs", "totalhigh", "freehigh", "mem_unit", "_f" });
72          }
73      }
74  
75      int sysinfo(Sysinfo info);
76  
77      /**
78       * The getloadavg() function returns the number of processes in the system
79       * run queue averaged over various periods of time. Up to nelem samples are
80       * retrieved and assigned to successive elements of loadavg[]. The system
81       * imposes a maximum of 3 samples, representing averages over the last 1, 5,
82       * and 15 minutes, respectively.
83       *
84       * @param loadavg
85       *            array to be filled
86       * @param nelem
87       *            number of elements in the array to fill
88       * @return If the load average was unobtainable, -1 is returned; otherwise,
89       *         the number of samples actually retrieved is returned.
90       */
91      int getloadavg(double[] loadavg, int nelem);
92  
93      /**
94       * Returns the process ID of the calling process. The ID is guaranteed to be
95       * unique and is useful for constructing temporary file names.
96       *
97       * @return the process ID of the calling process.
98       */
99      int getpid();
100 
101     /**
102      * Places the contents of the symbolic link path in the buffer buf, which
103      * has size bufsiz.
104      *
105      * @param path
106      *            A symbolic link
107      * @param buf
108      *            Holds actual path to location pointed to by symlink
109      * @param bufsize
110      *            size of data in buffer
111      * @return readlink() places the contents of the symbolic link path in the
112      *         buffer buf, which has size bufsiz. readlink() does not append a
113      *         null byte to buf. It will truncate the contents (to a length of
114      *         bufsiz characters), in case the buffer is too small to hold all
115      *         of the contents.
116      */
117     int readlink(String path, Pointer buf, int bufsize);
118 }
