5   package oshi.annotation;
6   
7   import java.lang.annotation.ElementType;
8   import java.lang.annotation.Retention;
9   import java.lang.annotation.RetentionPolicy;
10  import java.lang.annotation.Target;
11  
12  /**
13   * This annotation exists to permit suppression in the Forbidden APIs check and also indicate to consumers of these
14   * classes that they may need to handle these differently in their production code.
15   */
16  @Retention(RetentionPolicy.CLASS)
17  @Target({ ElementType.CONSTRUCTOR, ElementType.FIELD, ElementType.METHOD, ElementType.TYPE })
18  public @interface SuppressForbidden {
19      String reason();
20  }
