7   import java.lang.annotation.Documented;
8   import java.lang.annotation.ElementType;
9   import java.lang.annotation.Retention;
10  import java.lang.annotation.RetentionPolicy;
11  import java.lang.annotation.Target;
12  
13  /**
14   * Indicates that the annotated type is part of the OSHI public API. Public API types follow
15   * <a href="https://semver.org/">Semantic Versioning</a>: they are guaranteed to remain binary-compatible within the
16   * same major version.
17   * <p>
18   * Types <em>not</em> annotated with {@code @PublicApi} (such as platform-specific implementations, drivers, and utility
19   * classes) may change between minor releases.
20   */
21  @Documented
22  @Retention(RetentionPolicy.RUNTIME)
23  @Target(ElementType.TYPE)
24  public @interface PublicApi {
25  }
