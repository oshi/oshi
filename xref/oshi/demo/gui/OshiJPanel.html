24  package oshi.demo.gui;
25  
26  import java.awt.BorderLayout;
27  import java.awt.Dimension;
28  
29  import javax.swing.BorderFactory;
30  import javax.swing.JLabel;
31  import javax.swing.JPanel;
32  
33  /**
34   * Parent class combining code common to the other panels.
35   */
36  public class OshiJPanel extends JPanel {
37  
38      private static final long serialVersionUID = 1L;
39  
40      protected JLabel msgLabel = new JLabel();
41      protected JPanel msgPanel = new JPanel();
42  
43      public OshiJPanel() {
44          Dimension maxSize = getMaximumSize();
45          if (maxSize != null) {
46              setSize(maxSize);
47          }
48          setLayout(new BorderLayout());
49          setBorder(BorderFactory.createEmptyBorder(5, 10, 10, 10));
50          msgPanel.add(msgLabel);
51  
52          JPanel topPanel = new JPanel(new BorderLayout());
53          topPanel.add(msgPanel, BorderLayout.CENTER);
54          add(topPanel, BorderLayout.NORTH);
55      }
56  }
