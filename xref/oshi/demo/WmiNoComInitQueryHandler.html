24  package oshi.demo;
25  
26  import java.util.concurrent.TimeoutException;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import com.sun.jna.platform.win32.COM.COMException; // NOSONAR squid:S1191
32  import com.sun.jna.platform.win32.COM.Wbemcli;
33  import com.sun.jna.platform.win32.COM.WbemcliUtil;
34  
35  import oshi.util.platform.windows.WmiQueryHandler;
36  import oshi.util.platform.windows.WmiUtil;
37  
38  /**
39   * Query handler class that overrides WMI query method assuming COM is already
40   * initialized by the user.
41   */
42  public class WmiNoComInitQueryHandler extends WmiQueryHandler {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(WmiNoComInitQueryHandler.class);
45  
46      @Override
47      public <T extends Enum<T>> WbemcliUtil.WmiResult<T> queryWMI(WbemcliUtil.WmiQuery<T> query) {
48  
49          WbemcliUtil.WmiResult<T> result = WbemcliUtil.INSTANCE.new WmiResult<>(query.getPropertyEnum());
50          if (failedWmiClassNames.contains(query.getWmiClassName())) {
51              return result;
52          }
53          try {
54              result = query.execute(wmiTimeout);
55          } catch (COMException e) {
56              // Ignore any exceptions with OpenHardwareMonitor
57              if (!WmiUtil.OHM_NAMESPACE.equals(query.getNameSpace())) {
58                  final int hresult = e.getHresult() == null ? -1 : e.getHresult().intValue();
59                  switch (hresult) {
60                  case Wbemcli.WBEM_E_INVALID_NAMESPACE:
61                      LOG.warn("COM exception: Invalid Namespace {}", query.getNameSpace());
62                      break;
63                  case Wbemcli.WBEM_E_INVALID_CLASS:
64                      LOG.warn("COM exception: Invalid Class {}", query.getWmiClassName());
65                      break;
66                  case Wbemcli.WBEM_E_INVALID_QUERY:
67                      LOG.warn("COM exception: Invalid Query: {}", WmiUtil.queryToString(query));
68                      break;
69                  default:
70                      handleComException(query, e);
71                      break;
72                  }
73                  failedWmiClassNames.add(query.getWmiClassName());
74              }
75          } catch (TimeoutException e) {
76              LOG.error("WMI query timed out after {} ms: {}", wmiTimeout, WmiUtil.queryToString(query));
77          }
78          return result;
79      }
80  }
