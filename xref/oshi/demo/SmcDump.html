5   package oshi.demo;
6   
7   import com.sun.jna.NativeLong;
8   import com.sun.jna.platform.mac.IOKit.IOConnect;
9   
10  import java.util.Locale;
11  
12  import oshi.annotation.SuppressForbidden;
13  import oshi.jna.ByRef.CloseableNativeLongByReference;
14  import oshi.jna.platform.mac.IOKit;
15  import oshi.jna.platform.mac.IOKit.SMCKeyData;
16  import oshi.jna.platform.mac.IOKit.SMCVal;
17  import oshi.util.ParseUtil;
18  import oshi.util.platform.mac.SmcUtil;
19  
20  /**
21   * Dumps all readable SMC keys and their raw float values. Run on macOS to discover Apple Silicon key names for fans,
22   * voltage, and other sensors.
23   */
24  public final class SmcDump {
25  
26      private static final byte SMC_CMD_READ_INDEX = 8;
27      private static final IOKit IO = IOKit.INSTANCE;
28  
29      private SmcDump() {
30      }
31  
32      @SuppressForbidden(reason = "Using System.out in a demo class")
33      public static void main(String[] args) {
34          IOConnect conn = SmcUtil.smcOpen();
35          if (conn == null) {
36              System.err.println("Failed to open SMC connection"); // NOPMD
37              return;
38          }
39          try {
40              long keyCount = SmcUtil.smcGetLong(conn, "#KEY");
41              System.out.printf(Locale.ROOT, "Total SMC keys: %d%n%n", keyCount);
42              System.out.printf(Locale.ROOT, "%-6s %-6s %-8s %s%n", "Key", "Type", "Size", "Float value");
43              System.out.println("----------------------------------------------");
44  
45              for (int i = 0; i < keyCount; i++) {
46                  String keyName = readKeyAtIndex(conn, i);
47                  if (keyName == null) {
48                      continue;
49                  }
50                  try (SMCVal val = new SMCVal()) {
51                      int result = SmcUtil.smcReadKey(conn, keyName, val);
52                      String typeName = result == 0 ? asciiType(val.dataType) : "?";
53                      int size = result == 0 ? val.dataSize : 0;
54                      double floatVal = SmcUtil.smcGetFloat(conn, keyName);
55                      char first = keyName.charAt(0);
56                      if (first == 'F' || first == 'V' || first == 'T' || floatVal != 0d) {
57                          System.out.printf(Locale.ROOT, "%-6s %-6s %-8d %.4f%n", keyName, typeName, size, floatVal);
58                      }
59                  }
60              }
61          } finally {
62              SmcUtil.smcClose(conn);
63          }
64      }
65  
66      private static String readKeyAtIndex(IOConnect conn, int index) {
67          try (SMCKeyData input = new SMCKeyData();
68                  SMCKeyData output = new SMCKeyData();
69                  CloseableNativeLongByReference size = new CloseableNativeLongByReference(
70                          new NativeLong(output.size()))) {
71              input.data8 = SMC_CMD_READ_INDEX;
72              input.data32 = index;
73              int result = IO.IOConnectCallStructMethod(conn, SmcUtil.KERNEL_INDEX_SMC, input,
74                      new NativeLong(input.size()), output, size);
75              if (result != 0) {
76                  return null;
77              }
78              // Key is stored as a big-endian 4-byte int in output.key; convert to ASCII string
79              byte[] keyBytes = ParseUtil.longToByteArray(output.key, 4, 4);
80              StringBuilder sb = new StringBuilder(4);
81              for (byte b : keyBytes) {
82                  sb.append((char) (b & 0xFF));
83              }
84              return sb.toString();
85          }
86      }
87  
88      private static String asciiType(byte[] dataType) {
89          if (dataType == null) {
90              return "";
91          }
92          StringBuilder sb = new StringBuilder();
93          for (byte b : dataType) {
94              if (b == 0) {
95                  break;
96              }
97              sb.append((char) b);
98          }
99          return sb.toString();
100     }
