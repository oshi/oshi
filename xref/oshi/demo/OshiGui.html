26  import java.awt.BorderLayout;
27  import java.awt.Container;
28  
29  import javax.swing.JButton;

31  import javax.swing.JMenuBar;
32  import javax.swing.SwingUtilities;
33  import javax.swing.WindowConstants;
34  
35  import oshi.SystemInfo;
36  import oshi.demo.gui.Config;
37  import oshi.demo.gui.FileStorePanel;
38  import oshi.demo.gui.MemoryPanel;
39  import oshi.demo.gui.OsHwTextPanel;
40  import oshi.demo.gui.OshiJPanel;
41  import oshi.demo.gui.ProcessPanel;
42  import oshi.demo.gui.ProcessorPanel;
43  import oshi.demo.gui.UsbPanel;
44  
45  /**
46   * Basic Swing class to demonstrate potential uses for OSHI in a monitoring GUI.
47   * Not ready for production use and intended as inspiration/examples.
48   */
49  public class OshiGui {
50  
51      private JFrame mainFrame;
52      private JButton jMenu;
53      private JMenuBar menuBar;
54  
55      private SystemInfoystemInfo si = new SystemInfo();
56  
57      public static void main(String[] args) {
58          OshiGuiOshiGui gui = new OshiGui();
59          gui.init();
60          SwingUtilities.invokeLater(gui::setVisible);
61      }
62  
63      private void setVisible() {
64          mainFrame.setVisible(true);
65          jMenu.doClick();
66      }
67  
68      private void init() {
69          // Create the external frame
70          mainFrame = new JFrame(Config.GUI_TITLE);
71          mainFrame.setSize(Config.GUI_WIDTH, Config.GUI_HEIGHT);
72          mainFrame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
73          mainFrame.setResizable(true);
74          mainFrame.setLocationByPlatform(true);
75          mainFrame.setLayout(new BorderLayout());
76          // Add a menu bar
77          menuBar = new JMenuBar();
78          mainFrame.setJMenuBar(menuBar);
79          // Create the first menu option in this thread
80          jMenu = getJMenu("OS & HW Info", 'O', "Hardware & OS Summary", new OsHwTextPanel(si));
81          menuBar.add(jMenu);
82          // Add later menu items in their own threads
83          new Thread(new AddMenuBarTask("Memory", 'M', "Memory Summary", new MemoryPanel(si))).start();
84          new Thread(new AddMenuBarTask("CPU", 'C', "CPU Usage", new ProcessorPanel(si))).start();
85          new Thread(new AddMenuBarTask("FileStores", 'F', "FileStore Usage", new FileStorePanel(si))).start();
86          new Thread(new AddMenuBarTask("Processes", 'P', "Processes", new ProcessPanel(si))).start();
87          new Thread(new AddMenuBarTask("USB Devices", 'U', "USB Device list", new UsbPanel(si))).start();
88      }
89  
90      private JButton getJMenu(String title, char mnemonic, String toolTip, OshiJPanel panel) {
91          JButton button = new JButton(title);
92          button.setMnemonic(mnemonic);
93          button.setToolTipText(toolTip);
94          button.addActionListener(e -> {
95              Container contentPane = this.mainFrame.getContentPane();
96              if (contentPane.getComponents().length <= 0 || contentPane.getComponent(0) != panel) {
97                  resetMainGui();
98                  this.mainFrame.getContentPane().add(panel);
99                  refreshMainGui();
100             }
101         });
102 
103         return button;
104     }
105 
106     private void resetMainGui() {
107         this.mainFrame.getContentPane().removeAll();
108     }
109 
110     private void refreshMainGui() {
111         this.mainFrame.revalidate();
112         this.mainFrame.repaint();
113     }
114 
115     /**
116      * Runnable class to add a JMenu to the menubar.
117      */
118     private final class AddMenuBarTask implements Runnable {
119         private String title;
120         private char mnemonic;
121         private String toolTip;
122         private OshiJPanel panel;
123 
124         private AddMenuBarTask(String title, char mnemonic, String toolTip, OshiJPanel panel) {
125             this.title = title;
126             this.mnemonic = mnemonic;
127             this.toolTip = toolTip;
128             this.panel = panel;
129         }
130 
131         @Override
132         public void run() {
133             menuBar.add(getJMenu(title, mnemonic, toolTip, panel));
134         }
135     }
136 }
