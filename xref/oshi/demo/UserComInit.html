24  package oshi.demo;
25  
26  import com.sun.jna.platform.win32.COM.COMException; // NOSONAR squid:S1191
27  
28  import oshi.driver.windows.wmi.Win32OperatingSystem;
29  import oshi.util.platform.windows.WmiQueryHandler;
30  
31  import java.util.Objects;
32  
33  /**
34   * Class demonstrating WMI stat performance improvements if the user does COM
35   * initialization so OSHI doesn't have to
36   */
37  public class UserComInit {
38  
39      private static final int REPETITIONS = 300;
40  
41      public static void main(String[] args) {
42          wakeUpCom();
43          System.out.println("Collecting statistics with default COM setup");
44          loopWmiQueries();
45          System.out.println("Collecting statistics with user-init COM setup");
46          loopWmiQueriesWithUserCom();
47      }
48  
49      private static void wakeUpCom() {
50          // The first COM query can take extra long so just do one separately
51          Win32OperatingSystem.queryOsVersion();
52      }
53  
54      private static void loopWmiQueries() {
55          long t = System.nanoTime();
56          for (int i = 0; i < REPETITIONS; i++) {
57              Win32OperatingSystem.queryOsVersion();
58          }
59          t = System.nanoTime() - t;
60          System.out.println("Average ms per rep: " + t / (1_000_000d * REPETITIONS));
61      }
62  
63      private static void loopWmiQueriesWithUserCom() {
64          // Create instance using existing WmiQueryHandler class for convenience, only to
65          // be used for COM init/uninit. Not needed if user initializes COM.
66          WmiQueryHandler handlerForSingleCOMinit = Objects.requireNonNull(WmiQueryHandler.createInstance());
67  
68          boolean singleComInit = false;
69          try {
70              // Initialize using the default query handler. This is unnecessary in a user
71              // application if the application controls COM initialization elsewhere.
72              singleComInit = handlerForSingleCOMinit.initCOM();
73  
74              // Change query handler class to not initialize COM
75              WmiQueryHandler.setInstanceClass(WmiNoComInitQueryHandler.class);
76  
77              loopWmiQueries();
78          } catch (COMException e) {
79              // Ignore for demo. Production code should handle this!
80          } finally {
81              // User should ununitialize COM in their own application
82              if (singleComInit) {
83                  handlerForSingleCOMinit.unInitCOM();
84              }
85          }
86      }
87  }
