5   package oshi.demo;
6   
7   import com.fasterxml.jackson.core.JsonProcessingException;
8   import com.fasterxml.jackson.databind.ObjectMapper;
9   
10  import oshi.SystemInfo;
11  import oshi.hardware.CentralProcessor;
12  import oshi.hardware.GlobalMemory;
13  import oshi.hardware.HardwareAbstractionLayer;
14  
15  /**
16   * Demonstrates the use of Jackson's ObjectMapper to create JSON from OSHI objects
17   */
18  public class Json {
19      /**
20       * <p>
21       * main.
22       * </p>
23       *
24       * @param args an array of {@link java.lang.String} objects.
25       */
26      public static void main(String[] args) {
27          // Jackson ObjectMapper
28          ObjectMapper mapper = new ObjectMapper();
29  
30          // Fetch some OSHI objects
31          SystemInfo si = new SystemInfo();
32          HardwareAbstractionLayer hal = si.getHardware();
33  
34          try {
35              // Pretty print computer system
36              System.out.println("JSON for CPU:");
37              CentralProcessor cpu = hal.getProcessor();
38              System.out.println(mapper.writerWithDefaultPrettyPrinter().writeValueAsString(cpu));
39  
40              // Print memory
41              System.out.println("JSON for Memory:");
42              GlobalMemory mem = hal.getMemory();
43              System.out.println(mapper.writeValueAsString(mem));
44  
45          } catch (JsonProcessingException e) {
46              System.out.println("Exception encountered: " + e.getMessage());
47          }
48      }
49  }
