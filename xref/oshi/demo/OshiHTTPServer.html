24  package oshi.demo;
25  
26  import java.io.BufferedOutputStream;
27  import java.io.BufferedReader;
28  import java.io.IOException;
29  import java.io.InputStreamReader;
30  import java.io.OutputStreamWriter;
31  import java.io.PrintWriter;
32  import java.net.ServerSocket;
33  import java.net.Socket;
34  import java.nio.charset.StandardCharsets;
35  import java.time.Instant;
36  import java.util.StringTokenizer;
37  
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  import com.fasterxml.jackson.databind.ObjectMapper;
42  
43  import oshi.SystemInfo;
44  
45  /**
46   * Demo class to vend OSHI JSON data via an HTTP Webserver
47   * <p>
48   * This is for demonstration only not to be used in production code.
49   * <p>
50   * Code based on tutorial found on SSaurel's Blog :
51   * https://www.ssaurel.com/blog/create-a-simple-http-web-server-in-java Each
52   * Client Connection will be managed in a dedicated Thread
53   */
54  public class OshiHTTPServer implements Runnable {
55      // port to listen connection
56      private static final int PORT = 8080;
57  
58      private static final Logger logger = LoggerFactory.getLogger(OshiHTTPServer.class);
59  
60      // Client Connection via Socket Class
61      private Socket connect;
62  
63      public OshiHTTPServer(Socket c) {
64          connect = c;
65          logger.debug("Connecton opened.");
66      }
67  
68      public static void main(String[] args) {
69          try (ServerSocket serverConnect = new ServerSocket(PORT)) {
70              logger.info("Server started. Listening for connections on port {}", PORT);
71  
72              // we listen until user halts server execution
73              while (true) { // NOSONAR squid:S2189
74                  OshiHTTPServer myServer = new OshiHTTPServer(serverConnect.accept());
75  
76                  // create dedicated thread to manage the client connection
77                  Thread thread = new Thread(myServer);
78                  thread.start();
79              }
80          } catch (IOException e) {
81              logger.error("Server Connection error: {}", e.getMessage());
82          }
83      }
84  
85      @Override
86      public void run() {
87          try ( // read characters from the client via input stream on the socket
88                  BufferedReader in = new BufferedReader(
89                          new InputStreamReader(connect.getInputStream(), StandardCharsets.UTF_8));
90                  // get character output stream to client (for headers)
91                  PrintWriter out = new PrintWriter(
92                          new OutputStreamWriter(connect.getOutputStream(), StandardCharsets.UTF_8));
93                  // get binary output stream to client (for requested data)
94                  BufferedOutputStream dataOut = new BufferedOutputStream(connect.getOutputStream())) {
95  
96              // get first line of the request from the client
97              String input = in.readLine();
98              if (input == null) {
99                  throw new IOException("No characters read from input stream.");
100             }
101             // we parse the request with a string tokenizer
102             StringTokenizer parse = new StringTokenizer(input);
103             String method = parse.nextToken().toUpperCase(); // we get the HTTP method of the client
104             // we get fields requested
105             String fileRequested = parse.nextToken().toLowerCase();
106 
107             // we support only GET and HEAD methods, we check
108             if (!method.equals("GET") && !method.equals("HEAD")) {
109                 logger.debug("501 Not Implemented: {}", method);
110                 String contentMimeType = "text/html";
111 
112                 // we send HTTP Headers with data to client
113                 out.println("HTTP/1.1 501 Not Implemented");
114                 out.println("Server: OSHI HTTP Server");
115                 out.println("Date: " + Instant.now());
116                 out.println("Content-type: " + contentMimeType);
117                 out.println("Content-length: " + 0);
118                 out.println(); // blank line between headers and content, very important !
119                 out.flush(); // flush character output stream buffer
120 
121                 // Could return other information here...
122             } else {
123                 // Possibly could use the fileRequested value from user input to work down
124                 // OSHI's JSON tree and only return the relevant info instead of the entire
125                 // SystemInfo object.
126                 SystemInfo si = new SystemInfo();
127                 ObjectMapper mapper = new ObjectMapper();
128                 byte[] content = mapper.writerWithDefaultPrettyPrinter().writeValueAsString(si)
129                         .getBytes(StandardCharsets.UTF_8);
130 
131                 if (method.equals("GET")) { // GET method so we return content
132                     // send HTTP Headers
133                     out.println("HTTP/1.1 200 OK");
134                     out.println("Server: OSHI HTTP Server");
135                     out.println("Date: " + Instant.now());
136                     out.println("Content-type: application/json");
137                     out.println("Content-length: " + content.length);
138                     out.println(); // blank line between headers and content, very important !
139                     out.flush(); // flush character output stream buffer
140 
141                     dataOut.write(content, 0, content.length);
142                     dataOut.flush();
143                 }
144 
145                 logger.debug("Data {} returned", fileRequested);
146             }
147         } catch (IOException ioe) {
148             logger.error("Server error: {}", ioe.getMessage());
149         } finally {
150             try {
151                 // close socket connection, defined for this thread
152                 connect.close();
153             } catch (Exception e) {
154                 logger.error("Error closing connection: {}", e.getMessage());
155             }
156             logger.debug("Connection closed.");
157         }
158     }
159 }
