5   package oshi.demo.jmx.mbeans;
6   
7   import oshi.demo.jmx.demo.PropertiesAvailable;
8   
9   import javax.management.DynamicMBean;
10  import javax.management.MBeanInfo;
11  import javax.management.MBeanAttributeInfo;
12  import javax.management.MBeanNotificationInfo;
13  import javax.management.AttributeNotFoundException;
14  import javax.management.MBeanException;
15  import javax.management.ReflectionException;
16  import javax.management.InvalidAttributeValueException;
17  import javax.management.AttributeList;
18  import javax.management.Attribute;
19  import java.beans.IntrospectionException;
20  import java.beans.Introspector;
21  import java.beans.PropertyDescriptor;
22  import java.lang.reflect.InvocationTargetException;
23  import java.lang.reflect.Method;
24  import java.util.ArrayList;
25  import java.util.List;
26  
27  public class Baseboard implements DynamicMBean, PropertiesAvailable {
28  
29      private oshi.hardware.Baseboard baseboard;
30      private MBeanInfo dMBeanInfo = null;
31      private List<String> propertiesAvailable = new ArrayList<>();
32      private final String PROPERTIES = "Properties";
33      private final String GET_PREFIX = "get";
34  
35      private void setUpMBean() throws IntrospectionException, javax.management.IntrospectionException {
36  
37          PropertyDescriptor[] methods = Introspector.getBeanInfo(baseboard.getClass()).getPropertyDescriptors();
38          MBeanAttributeInfo[] attributeInfos = new MBeanAttributeInfo[methods.length];
39  
40          for (int i = 0; i < methods.length; i++) {
41              attributeInfos[i] = new MBeanAttributeInfo(methods[i].getName(), methods[i].getShortDescription(),
42                      methods[i].getReadMethod(), null);
43              propertiesAvailable
44                      .add(methods[i].getName().substring(0, 1).toUpperCase() + methods[i].getName().substring(1));
45          }
46  
47          dMBeanInfo = new MBeanInfo(baseboard.getClass().getName(), null, attributeInfos, null, null,
48                  new MBeanNotificationInfo[0]);
49      }
50  
51      public Baseboard(oshi.hardware.Baseboard baseboard)
52              throws IntrospectionException, javax.management.IntrospectionException {
53          this.baseboard = baseboard;
54          this.setUpMBean();
55      }
56  
57      @Override
58      public Object getAttribute(String attribute)
59              throws AttributeNotFoundException, MBeanException, ReflectionException {
60  
61          if (attribute.equals(PROPERTIES)) {
62              return this.getProperties();
63          }
64  
65          Method method = null;
66          try {
67              method = baseboard.getClass().getDeclaredMethod(GET_PREFIX + attribute, null);
68              method.setAccessible(true);
69              return method.invoke(baseboard, null);
70          } catch (NoSuchMethodException e) {
71              throw new RuntimeException(e);
72          } catch (InvocationTargetException e) {
73              throw new RuntimeException(e);
74          } catch (IllegalAccessException e) {
75              throw new RuntimeException(e);
76          }
77      }
78  
79      @Override
80      public MBeanInfo getMBeanInfo() {
81          return dMBeanInfo;
82      }
83  
84      @Override
85      public List<String> getProperties() {
86          return propertiesAvailable;
87      }
88  
89      @Override
90      public void setAttribute(Attribute attribute)
91              throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
92  
93      }
94  
95      @Override
96      public AttributeList getAttributes(String[] attributes) {
97          return null;
98      }
99  
100     @Override
101     public AttributeList setAttributes(AttributeList attributes) {
102         return null;
103     }
104 
105     @Override
106     public Object invoke(String actionName, Object[] params, String[] signature)
107             throws MBeanException, ReflectionException {
108         return null;
109     }
110 
111 }
