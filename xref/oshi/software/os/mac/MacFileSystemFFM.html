47  /**
48   * The Mac File System contains {@link oshi.software.os.OSFileStore}s which are a storage pool, device, partition,
49   * volume, concrete file system or other implementation specific means of file storage. In macOS, these are found in the
50   * /Volumes directory.
51   */
52  @ThreadSafe
53  public class MacFileSystemFFM extends MacFileSystem {
54  
55      private static final Logger LOG = LoggerFactory.getLogger(MacFileSystemFFM.class);
56  
57      @Override
58      public List<OSFileStore> getFileStores(boolean localOnly) {
59          // List of file systems
60          return getFileStoreMatching(null, localOnly);
61      }
62  
63      // Called by MacOSFileStore
64      static List<OSFileStore> getFileStoreMatching(String nameToMatch, boolean localOnly) {
65          List<OSFileStore> fsList = new ArrayList<>();
66          try (Arena arena = Arena.ofConfined()) {
67              // Use getfsstat to find fileSystems
68              // Query with null to get total # required
69              int numfs = getfsstat64(MemorySegment.NULL, 0, 0);
70              if (numfs <= 0) {
71                  return fsList;
72              }
73              // Create buffer to hold results
74              long statfsSize = STATFS.byteSize(); // Size of one structure
75              MemorySegment statfsBuffer = arena.allocate(statfsSize * numfs);
76              numfs = getfsstat64(statfsBuffer, (int) statfsBuffer.byteSize(), MNT_NOWAIT);
77              if (numfs <= 0) {
78                  return fsList;
79              }
80  
81              // Open a DiskArbitration session to get VolumeName of file systems
82              // with bsd names
83              CFAllocatorRef allocator = new CFAllocatorRef(CFAllocatorGetDefault());
84              DASessionRef session = new DASessionRef(DASessionCreate(allocator.segment()));
85              if (session.segment() == null) {
86                  LOG.error("Unable to open session to DiskArbitration framework.");
87                  return fsList;
88              }
89  
90              try {
91                  CFStringRef daVolumeNameKey = CFStringRef.createCFString("DAVolumeName");
92                  try {
93                      for (int f = 0; f < numfs; f++) {
94                          MemorySegment statfs = statfsBuffer.asSlice(f * statfsSize, statfsSize);
95                          // Mount on name will match mounted path, e.g. /Volumes/foo
96                          // Mount to name will match canonical path., e.g., /dev/disk0s2
97                          // Byte arrays are null-terminated strings
98                          String volume = statfs.getString(STATFS.byteOffset(F_MNTFROMNAME));
99                          String path = statfs.getString(STATFS.byteOffset(F_MNTONNAME));
100                         String type = statfs.getString(STATFS.byteOffset(F_FSTYPENAME));
101                         int flags = statfs.get(JAVA_INT, STATFS.byteOffset(F_FLAGS));
102                         long ffree = statfs.get(JAVA_LONG, STATFS.byteOffset(F_FFREE));
103                         long files = statfs.get(JAVA_LONG, STATFS.byteOffset(F_FILES));
104 
105                         // Skip non-local drives if requested, and exclude pseudo file systems
106                         boolean isLocal = (flags & MNT_LOCAL) != 0;
107                         if ((localOnly && !isLocal) || !path.equals("/")
108                                 && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path, volume,
109                                         FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
110                             continue;
111                         }
112 
113                         String description = "Volume";
114                         if (LOCAL_DISK.matcher(volume).matches()) {
115                             description = "Local Disk";
116                         } else if (volume.startsWith("localhost:") || volume.startsWith("//")
117                                 || volume.startsWith("smb://") || NETWORK_FS_TYPES.contains(type)) {
118                             description = "Network Drive";
119                         }
120                         File file = new File(path);
121                         String name = file.getName();
122                         // getName() for / is still blank, so:
123                         if (name.isEmpty()) {
124                             name = file.getPath();
125                         }
126 
127                         StringBuilder options = new StringBuilder((MNT_RDONLY & flags) == 0 ? "rw" : "ro");
128                         String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
129                                 .map(Map.Entry::getValue).collect(Collectors.joining(","));
130                         if (!moreOptions.isEmpty()) {
131                             options.append(',').append(moreOptions);
132                         }
133 
134                         String uuid = "";
135                         // Use volume to find DiskArbitration volume name and search for
136                         // the registry entry for UUID
137                         String bsdName = volume.replace("/dev/disk", "disk");
138                         if (bsdName.startsWith("disk")) {
139                             // Get the DiskArbitration dictionary for this disk
140                             DADiskRef disk = null;
141                             CFDictionaryRef diskInfo = null;
142                             try {
143                                 disk = DADiskRef.createFromBSDName(allocator, session, volume);
144                                 if (!disk.isNull()) {
145                                     diskInfo = disk.copyDescription();
146                                     if (!diskInfo.isNull()) {
147                                         // Get volume name
148                                         MemorySegment result = diskInfo.getValue(daVolumeNameKey);
149                                         if (!result.equals(MemorySegment.NULL)) {
150                                             name = CFUtilFFM.cfPointerToString(result);
151                                         }
152                                     }
153                                 }
154                             } finally {
155                                 if (diskInfo != null) {
156                                     diskInfo.release();
157                                 }
158                                 if (disk != null) {
159                                     disk.release();
160                                 }
161                             }
162                             // Search for bsd name in IOKit registry for UUID
163                             MemorySegment matchingDict = IOKitUtilFFM.getBSDNameMatchingDict(bsdName);
164                             if (matchingDict != null) {
165                                 // search for all IOservices that match the bsd name
166                                 IOIterator fsIter = IOKitUtilFFM.getMatchingServices(matchingDict);
167                                 if (fsIter != null) {
168                                     // getMatchingServices releases matchingDict
169                                     // Should only match one logical drive
170                                     IORegistryEntry fsEntry = fsIter.next();
171                                     if (fsEntry != null && fsEntry.conformsTo("IOMedia")) {
172                                         // Now get the UUID
173                                         uuid = fsEntry.getStringProperty("UUID");
174                                         if (uuid != null) {
175                                             uuid = uuid.toLowerCase(Locale.ROOT);
176                                         }
177                                         fsEntry.release();
178                                     }
179                                     fsIter.release();
180                                 }
181                             }
182                         }
183 
184                         if (nameToMatch != null && !nameToMatch.equals(name)) {
185                             continue;
186                         }
187 
188                         fsList.add(new MacOSFileStore(name, volume, name, path, options.toString(),
189                                 uuid == null ? "" : uuid, isLocal, "", description, type, file.getFreeSpace(),
190                                 file.getUsableSpace(), file.getTotalSpace(), ffree, files));
191                     }
192                 } finally {
193                     daVolumeNameKey.release();
194                 }
195             } finally {
196                 session.release();
197             }
198         } catch (Throwable e) {
199             LOG.warn("Failed to query file systems: {}", e.getMessage(), e);
200             return fsList;
201         }
202         return fsList;
203     }
