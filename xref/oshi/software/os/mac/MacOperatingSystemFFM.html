54      private static final long BOOTTIME;
55      static {
56          long bootTime = 0L;
57          try (Arena arena = Arena.ofConfined()) {
58              MemorySegment timeval = arena.allocate(TIMEVAL);
59              if (SysctlUtilFFM.sysctl("kern.boottime", timeval)) {
60                  bootTime = timeval.get(JAVA_LONG, 0L);
61              }
62          } catch (Throwable e) {
63              // do nothing, the bootTime == 0 conditional will handle
64          }
65          // Usually this works. If it doesn't, fall back to text parsing.
66          // Boot time will be the first consecutive string of digits.
67          if (bootTime == 0) {
68              bootTime = ParseUtil.parseLongOrDefault(
69                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
70                      System.currentTimeMillis() / 1000);
71          }
72          BOOTTIME = bootTime;
73      }

89      @Override
90      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
91          String family = this.major > 10 || (this.major == 10 && this.minor >= 12) ? "macOS"
92                  : System.getProperty("os.name");
93          String codeName = parseCodeName();
94          String buildNumber = SysctlUtilFFM.sysctl("kern.osversion", "");
95          return new Pair<>(family, new OSVersionInfo(this.osXVersion, codeName, buildNumber));
96      }

104     @Override
105     public List<OSProcess> queryAllProcesses() {
106         try (Arena arena = Arena.ofConfined()) {
107             // Calculate size needed, add a small buffer
108             int numberOfProcesses = proc_listpids(PROC_ALL_PIDS, 0, MemorySegment.NULL, 0) / INT_SIZE;
109             MemorySegment pidSegment = arena.allocate(JAVA_INT, numberOfProcesses + 10);
110             numberOfProcesses = proc_listpids(PROC_ALL_PIDS, 0, pidSegment, numberOfProcesses * INT_SIZE) / INT_SIZE;
111             // Use only the segment containing pids
112             return Arrays.stream(pidSegment.asSlice(0, numberOfProcesses * INT_SIZE).toArray(ValueLayout.JAVA_INT))
113                     .distinct().parallel().mapToObj(this::getProcess).filter(Objects::nonNull)
114                     .filter(ProcessFiltering.VALID_PROCESS).toList();
115         } catch (Throwable e) {
116             LOG.warn("Failed to query processes", e.getMessage());
117             return Collections.emptyList();
118         }
119     }

147     @Override
148     public int getThreadCount() {
149         try (Arena arena = Arena.ofConfined()) {
150             // Calculate size needed, add a small buffer
151             int numberOfProcesses = proc_listpids(PROC_ALL_PIDS, 0, MemorySegment.NULL, 0) / INT_SIZE;
152             MemorySegment pidSegment = arena.allocate(JAVA_INT, numberOfProcesses + 10);
153             numberOfProcesses = proc_listpids(PROC_ALL_PIDS, 0, pidSegment, numberOfProcesses * INT_SIZE) / INT_SIZE;
154             // Use only the segment containing pids
155             return Arrays.stream(pidSegment.asSlice(0, numberOfProcesses * INT_SIZE).toArray(ValueLayout.JAVA_INT))
156                     .distinct().parallel().map(this::threadsPerProc).sum();
157         } catch (Throwable e) {
158             LOG.warn("Failed to query processes", e.getMessage());
159             return 0;
160         }
161     }
162 
163     private int threadsPerProc(int pid) {
164         try (Arena arena = Arena.ofConfined()) {
165             MemorySegment buffer = arena.allocate(PROC_TASK_INFO);
166             int result = proc_pidinfo(pid, PROC_PIDTASKINFO, 0L, buffer, (int) PROC_TASK_INFO.byteSize());
167             if (result > 0) {
168                 return buffer.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_THREADNUM));
169             }
170             return 0;
171         } catch (Throwable e) {
172             // if this is a common warning for short-lived processes may need a lower log level
173             LOG.warn("Failed to get threads for process {}:", pid, e.getMessage());
174             return 0;
175         }
176     }
