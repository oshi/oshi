55      private static final int ARGMAX = SysctlUtil.sysctl("kern.argmax", 0);
56      private static final long TICKS_PER_MS;
57      static {
58          // default to 1 tick per nanosecond
59          long ticksPerSec = 1_000_000_000L;
60          IOIterator iter = IOKitUtil.getMatchingServices("IOPlatformDevice");
61          if (iter != null) {
62              IORegistryEntry cpu = iter.next();
63              while (cpu != null) {
64                  try {
65                      String s = cpu.getName().toLowerCase(Locale.ROOT);
66                      if (s.startsWith("cpu") && s.length() > 3) {
67                          byte[] data = cpu.getByteArrayProperty("timebase-frequency");
68                          if (data != null) {
69                              ticksPerSec = ParseUtil.byteArrayToLong(data, 4, false);
70                              break;
71                          }
72                      }
73                  } finally {
74                      cpu.release();
75                  }
76                  cpu = iter.next();
77              }
78              iter.release();
79          }
80          // Convert to ticks per millisecond
81          TICKS_PER_MS = ticksPerSec / 1000L;
82      }

89      @Override
90      protected Pair<List<String>, Map<String, String>> queryArgsAndEnvironment() {
91          int pid = getProcessID();
92          // Set up return objects
93          List<String> args = new ArrayList<>();
94          // API does not specify any particular order of entries, but it is reasonable to
95          // maintain whatever order the OS provided to the end user
96          Map<String, String> env = new LinkedHashMap<>();
97  
98          // Get command line via sysctl
99          int[] mib = new int[3];
100         mib[0] = 1; // CTL_KERN
101         mib[1] = 49; // KERN_PROCARGS2
102         mib[2] = pid;
103         // Allocate memory for arguments
104         try (Memory procargs = new Memory(ARGMAX)) {
105             procargs.clear();
106             size_t.ByReference size = new size_t.ByReference(ARGMAX);
107             // Fetch arguments
108             if (0 == SystemB.INSTANCE.sysctl(mib, mib.length, procargs, size, null, size_t.ZERO)) {
109                 // Procargs contains an int representing total # of args, followed by a
110                 // null-terminated execpath string and then the arguments, each
111                 // null-terminated (possible multiple consecutive nulls),
112                 // The execpath string is also the first arg.
113                 // Following this is an int representing total # of env, followed by
114                 // null-terminated envs in similar format
115                 int nargs = procargs.getInt(0);
116                 // Sanity check
117                 if (nargs > 0 && nargs <= 1024) {
118                     // Skip first int (containing value of nargs)
119                     long offset = INT_SIZE;
120                     // Skip exec_command, as
121                     offset += procargs.getString(offset).length();
122                     // Iterate character by character using offset
123                     // Build each arg and add to list
124                     while (offset < size.longValue()) {
125                         // Advance through additional nulls
126                         while (procargs.getByte(offset) == 0) {
127                             if (++offset >= size.longValue()) {
128                                 break;
129                             }
130                         }
131                         // Grab a string. This should go until the null terminator
132                         String arg = procargs.getString(offset);
133                         if (nargs-- > 0) {
134                             // If we havent found nargs yet, it's an arg
135                             args.add(arg);
136                         } else {
137                             // otherwise it's an env
138                             int idx = arg.indexOf('=');
139                             if (idx > 0) {
140                                 env.put(arg.substring(0, idx), arg.substring(idx + 1));
141                             }
142                         }
143                         // Advance offset to next null
144                         offset += arg.length();
145                     }
146                 }
147             } else {
148                 // Don't warn for pid 0
149                 if (pid > 0 && LOG_MAC_SYSCTL_WARNING) {
150                     LOG.warn(
151                             "Failed sysctl call for process arguments (kern.procargs2), process {} may not exist. Error code: {}",
152                             pid, Native.getLastError());
153                 }
154             }
155         }
156         return new Pair<>(Collections.unmodifiableList(args), Collections.unmodifiableMap(env));
157     }

159     @Override
160     protected int queryLogicalProcessorCount() {
161         return SysctlUtil.sysctl("hw.logicalcpu", 1);
162     }

164     @Override
165     public long getSoftOpenFileLimit() {
166         if (getProcessID() == this.os.getProcessId()) {
167             final Resource.Rlimit rlimit = new Resource.Rlimit();
168             SystemB.INSTANCE.getrlimit(MAC_RLIMIT_NOFILE, rlimit);
169             return rlimit.rlim_cur;
170         } else {
171             return -1L; // not supported
172         }
173     }
174 
175     @Override
176     public long getHardOpenFileLimit() {
177         if (getProcessID() == this.os.getProcessId()) {
178             final Resource.Rlimit rlimit = new Resource.Rlimit();
179             SystemB.INSTANCE.getrlimit(MAC_RLIMIT_NOFILE, rlimit);
180             return rlimit.rlim_max;
181         } else {
182             return -1L; // not supported
183         }
184     }

186     @Override
187     public boolean updateAttributes() {
188         long now = System.currentTimeMillis();
189         try (CloseableProcTaskAllInfo taskAllInfo = new CloseableProcTaskAllInfo()) {
190             if (0 > SystemB.INSTANCE.proc_pidinfo(getProcessID(), PROC_PIDTASKALLINFO, 0, taskAllInfo,
191                     taskAllInfo.size()) || taskAllInfo.ptinfo.pti_threadnum < 1) {
192                 this.state = INVALID;
193                 return false;
194             }
195             try (Memory buf = new Memory(PROC_PIDPATHINFO_MAXSIZE)) {
196                 if (0 < SystemB.INSTANCE.proc_pidpath(getProcessID(), buf, PROC_PIDPATHINFO_MAXSIZE)) {
197                     this.path = buf.getString(0).trim();
198                     // Overwrite name with last part of path
199                     String[] pathSplit = this.path.split("/");
200                     if (pathSplit.length > 0) {
201                         this.name = pathSplit[pathSplit.length - 1];
202                     }
203                 }
204             }
205             if (this.name.isEmpty()) {
206                 // pbi_comm contains first 16 characters of name
207                 this.name = Native.toString(taskAllInfo.pbsd.pbi_comm, StandardCharsets.UTF_8);
208             }
209 
210             this.state = stateFromStatus(taskAllInfo.pbsd.pbi_status);
211             this.parentProcessID = taskAllInfo.pbsd.pbi_ppid;
212             this.userID = Integer.toString(taskAllInfo.pbsd.pbi_uid);
213             Passwd pwuid = SystemB.INSTANCE.getpwuid(taskAllInfo.pbsd.pbi_uid);
214             this.user = pwuid == null ? Integer.toString(taskAllInfo.pbsd.pbi_uid) : pwuid.pw_name;
215             this.groupID = Integer.toString(taskAllInfo.pbsd.pbi_gid);
216             Group grgid = SystemB.INSTANCE.getgrgid(taskAllInfo.pbsd.pbi_gid);
217             this.group = grgid == null ? Integer.toString(taskAllInfo.pbsd.pbi_gid) : grgid.gr_name;
218             this.threadCount = taskAllInfo.ptinfo.pti_threadnum;
219             this.priority = taskAllInfo.ptinfo.pti_priority;
220             this.virtualSize = taskAllInfo.ptinfo.pti_virtual_size;
221             this.residentSetSize = taskAllInfo.ptinfo.pti_resident_size;
222             // Default/fallback: RSS. Will be overwritten by phys_footprint when available.
223             this.memoryFootprint = this.residentSetSize;
224             this.kernelTime = taskAllInfo.ptinfo.pti_total_system / TICKS_PER_MS;
225             this.userTime = taskAllInfo.ptinfo.pti_total_user / TICKS_PER_MS;
226             this.startTime = taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L;
227             this.upTime = now - this.startTime;
228             this.openFiles = taskAllInfo.pbsd.pbi_nfiles;
229             this.bitness = (taskAllInfo.pbsd.pbi_flags & P_LP64) == 0 ? 32 : 64;
230             this.majorFaults = taskAllInfo.ptinfo.pti_pageins;
231             // testing using getrusage confirms pti_faults includes both major and minor
232             this.minorFaults = taskAllInfo.ptinfo.pti_faults - taskAllInfo.ptinfo.pti_pageins; // NOSONAR squid:S2184
233             // getrusage(RUSAGE_SELF) aggregates across all threads for current process
234             if (getProcessID() == this.os.getProcessId()) {
235                 SystemB.Rusage rusage = new SystemB.Rusage();
236                 if (0 == SystemB.INSTANCE.getrusage(RUSAGE_SELF, rusage)) {
237                     this.voluntaryContextSwitches = rusage.ru_nvcsw.longValue();
238                     this.involuntaryContextSwitches = rusage.ru_nivcsw.longValue();
239                     this.contextSwitches = this.voluntaryContextSwitches + this.involuntaryContextSwitches;
240                 } else {
241                     // getrusage failed; fall back to pti_csw (combined only)
242                     this.contextSwitches = taskAllInfo.ptinfo.pti_csw;
243                     this.voluntaryContextSwitches = 0L;
244                     this.involuntaryContextSwitches = 0L;
245                 }
246             } else {
247                 // For other processes, only combined total is available via pti_csw
248                 this.contextSwitches = taskAllInfo.ptinfo.pti_csw;
249             }
250         }
251         if (this.majorVersion > 10 || (this.majorVersion == 10 && this.minorVersion >= 9)) {
252             try (CloseableRUsageInfoV2 rUsageInfoV2 = new CloseableRUsageInfoV2()) {
253                 if (0 == SystemB.INSTANCE.proc_pid_rusage(getProcessID(), RUSAGE_INFO_V2, rUsageInfoV2)) {
254                     this.bytesRead = rUsageInfoV2.ri_diskio_bytesread;
255                     this.bytesWritten = rUsageInfoV2.ri_diskio_byteswritten;
256                     this.memoryFootprint = rUsageInfoV2.ri_phys_footprint;
257                 }
258             }
259         }
260         try (CloseableVnodePathInfo vpi = new CloseableVnodePathInfo()) {
261             if (0 < SystemB.INSTANCE.proc_pidinfo(getProcessID(), PROC_PIDVNODEPATHINFO, 0, vpi, vpi.size())) {
262                 this.currentWorkingDirectory = Native.toString(vpi.pvi_cdir.vip_path, StandardCharsets.UTF_8);
263             }
264         }
265         return true;
266     }
