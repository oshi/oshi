129     private String queryCommandLine() {
130         // Get command line via sysctl
131         int[] mib = new int[3];
132         mib[0] = 1; // CTL_KERN
133         mib[1] = 49; // KERN_PROCARGS2
134         mib[2] = getProcessID();
135         // Allocate memory for arguments
136         int argmax = SysctlUtil.sysctl("kern.argmax", 0);
137         Pointer procargs = new Memory(argmax);
138         IntByReference size = new IntByReference(argmax);
139         // Fetch arguments
140         if (0 != SystemB.INSTANCE.sysctl(mib, mib.length, procargs, size, null, 0)) {
141             LOG.warn(
142                     "Failed syctl call for process arguments (kern.procargs2), process {} may not exist. Error code: {}",
143                     getProcessID(), Native.getLastError());
144             return "";
145         }
146         // Procargs contains an int representing total # of args, followed by a
147         // null-terminated execpath string and then the arguments, each
148         // null-terminated (possible multiple consecutive nulls),
149         // The execpath string is also the first arg.
150         int nargs = procargs.getInt(0);
151         // Sanity check
152         if (nargs < 0 || nargs > 1024) {
153             LOG.error("Nonsensical number of process arguments for pid {}: {}", getProcessID(), nargs);
154             return "";
155         }
156         List<String> args = new ArrayList<>(nargs);
157         // Skip first int (containing value of nargs)
158         long offset = SystemB.INT_SIZE;
159         // Skip exec_command
160         offset += procargs.getString(offset).length();
161         // Iterate character by character using offset
162         // Build each arg and add to list
163         while (nargs-- > 0 && offset < size.getValue()) {
164             // Advance through additional nulls
165             while (procargs.getByte(offset) == 0) {
166                 if (++offset >= size.getValue()) {
167                     break;
168                 }
169             }
170             // Grab a string. This should go until the null terminator
171             String arg = procargs.getString(offset);
172             args.add(arg);
173             // Advance offset to next null
174             offset += arg.length();
175         }
176         // Return args null-delimited
177         return String.join("\0", args);
178     }

220     @Override
221     public List<OSThread> getThreadDetails() {
222         long now = System.currentTimeMillis();
223         List<MacOSThread> details = new ArrayList<>();
224         List<ThreadStats> stats = ThreadInfo.queryTaskThreads(getProcessID());
225         for (ThreadStats stat : stats) {
226             // For long running threads the start time calculation can overestimate
227             long start = now - stat.getUpTime();
228             if (start < this.getStartTime()) {
229                 start = this.getStartTime();
230             }
231             details.add(new MacOSThread(getProcessID(), stat.getThreadId(), stat.getState(), stat.getSystemTime(),
232                     stat.getUserTime(), start, now - start, stat.getPriority()));
233         }
234         return Collections.unmodifiableList(details);
235     }

292     @Override
293     public long getAffinityMask() {
294         // macOS doesn't do affinity. Return a bitmask of the current processors.
295         int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
296         return logicalProcessorCount < 64 ? (1L << logicalProcessorCount) - 1 : -1L;
297     }

309     @Override
310     public boolean updateAttributes() {
311         long now = System.currentTimeMillis();
312         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
313         if (0 > SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo,
314                 taskAllInfo.size()) || taskAllInfo.ptinfo.pti_threadnum < 1) {
315             this.state = INVALID;
316             return false;
317         }
318         Pointer buf = new Memory(SystemB.PROC_PIDPATHINFO_MAXSIZE);
319         if (0 < SystemB.INSTANCE.proc_pidpath(getProcessID(), buf, SystemB.PROC_PIDPATHINFO_MAXSIZE)) {
320             this.path = buf.getString(0).trim();
321             // Overwrite name with last part of path
322             String[] pathSplit = this.path.split("/");
323             if (pathSplit.length > 0) {
324                 this.name = pathSplit[pathSplit.length - 1];
325             }
326         }
327         if (this.name.isEmpty()) {
328             // pbi_comm contains first 16 characters of name
329             // null terminated
330             for (int t = 0; t < taskAllInfo.pbsd.pbi_comm.length; t++) {
331                 if (taskAllInfo.pbsd.pbi_comm[t] == 0) {
332                     this.name = new String(taskAllInfo.pbsd.pbi_comm, 0, t, StandardCharsets.UTF_8);
333                     break;
334                 }
335             }
336         }
337 
338         switch (taskAllInfo.pbsd.pbi_status) {
339         case SSLEEP:
340             this.state = SLEEPING;
341             break;
342         case SWAIT:
343             this.state = WAITING;
344             break;
345         case SRUN:
346             this.state = RUNNING;
347             break;
348         case SIDL:
349             this.state = NEW;
350             break;
351         case SZOMB:
352             this.state = ZOMBIE;
353             break;
354         case SSTOP:
355             this.state = STOPPED;
356             break;
357         default:
358             this.state = OTHER;
359             break;
360         }
361         this.parentProcessID = taskAllInfo.pbsd.pbi_ppid;
362         this.userID = Integer.toString(taskAllInfo.pbsd.pbi_uid);
363         Passwd pwuid = SystemB.INSTANCE.getpwuid(taskAllInfo.pbsd.pbi_uid);
364         if (pwuid != null) {
365             this.user = pwuid.pw_name;
366         }
367         this.groupID = Integer.toString(taskAllInfo.pbsd.pbi_gid);
368         Group grgid = SystemB.INSTANCE.getgrgid(taskAllInfo.pbsd.pbi_gid);
369         if (grgid != null) {
370             this.group = grgid.gr_name;
371         }
372         this.threadCount = taskAllInfo.ptinfo.pti_threadnum;
373         this.priority = taskAllInfo.ptinfo.pti_priority;
374         this.virtualSize = taskAllInfo.ptinfo.pti_virtual_size;
375         this.residentSetSize = taskAllInfo.ptinfo.pti_resident_size;
376         this.kernelTime = taskAllInfo.ptinfo.pti_total_system / 1_000_000L;
377         this.userTime = taskAllInfo.ptinfo.pti_total_user / 1_000_000L;
378         this.startTime = taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L;
379         this.upTime = now - this.startTime;
380         this.openFiles = taskAllInfo.pbsd.pbi_nfiles;
381         this.bitness = (taskAllInfo.pbsd.pbi_flags & P_LP64) == 0 ? 32 : 64;
382         this.majorFaults = taskAllInfo.ptinfo.pti_pageins;
383         // testing using getrusage confirms pti_faults includes both major and minor
384         this.minorFaults = taskAllInfo.ptinfo.pti_faults - taskAllInfo.ptinfo.pti_pageins;
385         if (this.minorVersion >= 9) {
386             RUsageInfoV2 rUsageInfoV2 = new RUsageInfoV2();
387             if (0 == SystemB.INSTANCE.proc_pid_rusage(getProcessID(), SystemB.RUSAGE_INFO_V2, rUsageInfoV2)) {
388                 this.bytesRead = rUsageInfoV2.ri_diskio_bytesread;
389                 this.bytesWritten = rUsageInfoV2.ri_diskio_byteswritten;
390             }
391         }
392         VnodePathInfo vpi = new VnodePathInfo();
393         if (0 < SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDVNODEPATHINFO, 0, vpi, vpi.size())) {
394             int len = 0;
395             for (byte b : vpi.pvi_cdir.vip_path) {
396                 if (b == 0) {
397                     break;
398                 }
399                 len++;
400             }
401             this.currentWorkingDirectory = new String(vpi.pvi_cdir.vip_path, 0, len, StandardCharsets.US_ASCII);
402         }
403         return true;
404     }
