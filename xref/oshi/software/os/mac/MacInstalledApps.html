37      public static List<ApplicationInfo> queryInstalledApps() {
38          List<String> output = ExecutingCommand.runNative("system_profiler -xml SPApplicationsDataType");
39  
40          try {
41              List<Map<String, String>> plistValues = parseItems(String.join("", output));
42  
43              if (!plistValues.isEmpty()) {
44                  Set<ApplicationInfo> appInfoSet = new LinkedHashSet<>();
45  
46                  for (Map<String, String> dictValues : plistValues) {
47                      try {
48                          String obtainedFrom = ParseUtil.getStringValueOrUnknown(dictValues.get("obtained_from"));
49                          if ("apple".equals(obtainedFrom)) {
50                              obtainedFrom = "Apple";
51                          } else if ("mac_app_store".equals(obtainedFrom)) {
52                              obtainedFrom = "App Store";
53                          }
54                          String signedBy = ParseUtil.getStringValueOrUnknown(dictValues.get("signed_by"));
55                          String vendor;
56                          if ("identified_developer".equals(obtainedFrom)) {
57                              if (signedBy.startsWith("Developer ID Application: ")) {
58                                  vendor = signedBy.substring(26);
59                              } else {
60                                  vendor = signedBy;
61                              }
62                          } else if (Constants.UNKNOWN.equals(obtainedFrom) && !Constants.UNKNOWN.equals(signedBy)) {
63                              vendor = signedBy;
64                          } else {
65                              vendor = obtainedFrom;
66                          }
67  
68                          String version = dictValues.get("version");
69  
70                          String lastModified = dictValues.get("lastModified");
71                          long lastModifiedEpoch = ParseUtil.parseDateToEpoch(lastModified, "yyyy-MM-dd'T'HH:mm:ss'Z'");
72  
73                          Map<String, String> additionalInfo = new LinkedHashMap<>();
74                          additionalInfo.put("Kind", ParseUtil.getStringValueOrUnknown(dictValues.get("arch_kind")));
75                          String location = ParseUtil.getStringValueOrUnknown(dictValues.get("path"));
76                          additionalInfo.put("Location", location);
77                          if (!Constants.UNKNOWN.equals(location)) {
78                              File appPlistFile = new File(location, "/Contents/Info.plist");
79                              if (appPlistFile.exists()) {
80                                  if ("bplist00"
81                                          .equals(new String(readFirstBytes(appPlistFile), StandardCharsets.UTF_8))) {
82                                      // convert binary plist to xml
83                                      output = ExecutingCommand.runNative(new String[] { "plutil", "-convert", "xml1",
84                                              "-o", "-", appPlistFile.getAbsolutePath() });
85                                  } else {
86                                      output = FileUtil.readFile(appPlistFile.getAbsolutePath());
87                                  }
88                                  String xml = String.join("", output);
89                                  String getInfoString = readStringValue(xml, "CFBundleGetInfoString");
90                                  if (getInfoString != null && !getInfoString.isEmpty()) {
91                                      additionalInfo.put("Get Info String", getInfoString);
92                                  }
93                                  if (version == null || version.isEmpty()) {
94                                      version = readStringValue(xml, "CFBundleVersion");
95                                  }
96                              }
97                          }
98  
99                          if (version == null || version.isEmpty()) {
100                             version = Constants.UNKNOWN;
101                         }
102 
103                         appInfoSet.add(new ApplicationInfo(dictValues.get("_name"), version, vendor, lastModifiedEpoch,
104                                 additionalInfo));
105                     } catch (Exception e) {
106                         LOG.trace("Unable to parse dict values: " + e.getMessage() + " - " + dictValues, e);
107                     }
108                 }
109 
110                 return new ArrayList<>(appInfoSet);
111             }
112         } catch (Exception e) {
113             LOG.trace("Unable to read installed apps: " + e.getMessage(), e);
114         }
115         return Collections.emptyList();
116     }

118     private static byte[] readFirstBytes(File file) throws IOException {
119         byte[] buffer = new byte[8];
120         try (FileInputStream fis = new FileInputStream(file)) {
121             fis.read(buffer);
122             return buffer;
123         }
124     }

126     private static List<Map<String, String>> parseItems(String xml) {
127         if (xml == null) {
128             return Collections.emptyList();
129         }
130 
131         Matcher m = Pattern.compile("<key>\\s*_items\\s*</key>").matcher(xml);
132         if (!m.find()) {
133             return Collections.emptyList();
134         }
135 
136         int arrayOpen = xml.indexOf("<array>", m.end());
137         if (arrayOpen < 0) {
138             return Collections.emptyList();
139         }
140 
141         String arrayBody = extractBalancedInner(xml, arrayOpen, "<array>", "</array>");
142         if (arrayBody == null) {
143             return Collections.emptyList();
144         }
145 
146         List<String> dictBlocks = extractTopLevelBlocks(arrayBody, "<dict>", "</dict>");
147         List<Map<String, String>> out = new ArrayList<>();
148         for (String dictInner : dictBlocks) {
149             out.add(parseDict(dictInner));
150         }
151         return out;
152     }

154     private static Map<String, String> parseDict(String dictInner) {
155         Map<String, String> map = new LinkedHashMap<>();
156         int pos = 0;
157         while (true) {
158             int kStart = dictInner.indexOf("<key>", pos);
159             if (kStart < 0) {
160                 break;
161             }
162             int kEnd = dictInner.indexOf("</key>", kStart + 5);
163             if (kEnd < 0) {
164                 break;
165             }
166 
167             String key = unescape(dictInner.substring(kStart + 5, kEnd).trim());
168             int vOpen = dictInner.indexOf('<', kEnd + 6);
169             if (vOpen < 0) {
170                 break;
171             }
172 
173             String val;
174             if (startsWith(dictInner, vOpen, "<string>")) {
175                 String inner = extractSimpleInner(dictInner, vOpen, "<string>", "</string>");
176                 val = unescape(inner);
177                 pos = dictInner.indexOf("</string>", vOpen) + "</string>".length();
178             } else if (startsWith(dictInner, vOpen, "<date>")) {
179                 String inner = extractSimpleInner(dictInner, vOpen, "<date>", "</date>");
180                 val = inner.trim();
181                 pos = dictInner.indexOf("</date>", vOpen) + "</date>".length();
182             } else if (startsWith(dictInner, vOpen, "<array>")) {
183                 String inner = extractBalancedInner(dictInner, vOpen, "<array>", "</array>");
184                 String parsed = inner == null ? null : parseStringArray(inner);
185                 val = parsed == null ? "" : parsed;
186                 pos = dictInner.indexOf("</array>", vOpen) + "</array>".length();
187             } else {
188                 // other irrelevant tags: go to next tag
189                 pos = vOpen + 1;
190                 continue;
191             }
192 
193             map.put(key, val);
194         }
195         return map;
196     }

198     private static String parseStringArray(String arrayInner) {
199         int lt = arrayInner.indexOf('<');
200         if (lt >= 0) {
201             if (startsWith(arrayInner, lt, "<string>")) {
202                 String inner = extractSimpleInner(arrayInner, lt, "<string>", "</string>");
203                 return unescape(inner);
204             }
205         }
206         return null;
207     }

209     private static boolean startsWith(String s, int pos, String tag) {
210         int end = pos + tag.length();
211         return end <= s.length() && s.regionMatches(false, pos, tag, 0, tag.length());
212     }

214     private static String extractSimpleInner(String s, int openPos, String openTag, String closeTag) {
215         int start = s.indexOf(openTag, openPos);
216         if (start < 0) {
217             return "";
218         }
219         int end = s.indexOf(closeTag, start + openTag.length());
220         if (end < 0) {
221             return "";
222         }
223         return s.substring(start + openTag.length(), end);
224     }

226     private static String extractBalancedInner(String s, int openPos, String openTag, String closeTag) {
227         int pos = openPos;
228         if (!startsWith(s, pos, openTag)) {
229             return null;
230         }
231         pos += openTag.length();
232         int depth = 1;
233         while (pos < s.length()) {
234             int nextOpen = s.indexOf(openTag, pos);
235             int nextClose = s.indexOf(closeTag, pos);
236             if (nextClose == -1) {
237                 return null;
238             }
239             if (nextOpen != -1 && nextOpen < nextClose) {
240                 depth++;
241                 pos = nextOpen + openTag.length();
242             } else {
243                 depth--;
244                 if (depth == 0) {
245                     return s.substring(openPos + openTag.length(), nextClose);
246                 }
247                 pos = nextClose + closeTag.length();
248             }
249         }
250         return null;
251     }

253     private static List<String> extractTopLevelBlocks(String containerInner, String openTag, String closeTag) {
254         List<String> blocks = new ArrayList<>();
255         int pos = 0;
256         while (true) {
257             int open = containerInner.indexOf(openTag, pos);
258             if (open < 0) {
259                 break;
260             }
261             String inner = extractBalancedInner(containerInner, open, openTag, closeTag);
262             if (inner == null) {
263                 break;
264             }
265             blocks.add(inner);
266             int closeEnd = containerInner.indexOf(closeTag, open) + closeTag.length();
267             pos = closeEnd;
268         }
269         return blocks;
270     }

272     private static String unescape(String s) {
273         return s.replace("&amp;", "&").replace("&lt;", "<").replace("&gt;", ">").replace("&quot;", "\"")
274                 .replace("&apos;", "'");
275     }

277     private static String readStringValue(String xml, String keyName) throws Exception {
278         int i = xml.indexOf("<key>" + keyName + "</key>");
279         if (i > 0) {
280             i = xml.indexOf("<string>", i);
281             if (i > 0) {
282                 i += 8; // lengh of "<string>"
283                 int e = xml.indexOf("</string>", i);
284                 if (e > 0) {
285                     return xml.substring(i, e);
286                 }
287             }
288         }
289         return null;
290     }
