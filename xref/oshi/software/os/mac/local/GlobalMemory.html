17  package oshi.software.os.mac.local;
18  
19  import com.sun.jna.LastErrorException;
20  import com.sun.jna.Native;
21  import com.sun.jna.Pointer;
22  import com.sun.jna.ptr.IntByReference;
23  import com.sun.jna.ptr.LongByReference;
24  
25  import oshi.hardware.Memory;
26  import oshi.software.os.mac.local.SystemB.VMStatistics;
27  
28  /**
29   * Memory obtained by host_statistics (vm_stat) and sysctl
30   * 
31   * @author widdis[at]gmail[dot]com
32   */
33  public class GlobalMemory implements Memory {
34  
35  	long totalMemory = 0;
36  
37  	@Override
38  	public long getAvailable() {
39  		long availableMemory = 0;
40  		long pageSize = 4096;
41  
42  		int machPort = SystemB.INSTANCE.mach_host_self();
43  
44  		LongByReference pPageSize = new LongByReference();
45  		if (0 != SystemB.INSTANCE.host_page_size(machPort, pPageSize))
46  			throw new LastErrorException("Error code: " + Native.getLastError());
47  		pageSize = pPageSize.getValue();
48  
49  		VMStatistics vmStats = new VMStatistics();
50  		if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_VM_INFO, vmStats,
51  				new IntByReference(vmStats.size() / SystemB.INT_SIZE)))
52  			throw new LastErrorException("Error code: " + Native.getLastError());
53  		availableMemory = (vmStats.free_count + vmStats.inactive_count) * pageSize;
54  
55  		return availableMemory;
56  	}
57  
58  	@Override
59  	public long getTotal() {
60  		if (this.totalMemory == 0) {
61  			Pointer pMemSize = new com.sun.jna.Memory(SystemB.UINT64_SIZE);
62  			if (0 != SystemB.INSTANCE.sysctlbyname("hw.memsize", pMemSize, new IntByReference(SystemB.UINT64_SIZE),
63  					null, 0))
64  				throw new LastErrorException("Error code: " + Native.getLastError());
65  			this.totalMemory = pMemSize.getLong(0);
66  		}
67  		return this.totalMemory;
68  	}
69  }
