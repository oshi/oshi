24  package oshi.software.os;
25  
26  import java.time.Instant;
27  import java.time.LocalDateTime;
28  import java.time.ZoneId;
29  import java.time.format.DateTimeFormatter;
30  
31  import oshi.annotation.concurrent.Immutable;
32  
33  /**
34   * This class encapsulates information about users who are currently logged in
35   * to an operating system.
36   */
37  @Immutable
38  public class OSSession {
39      private static final DateTimeFormatter LOGIN_FORMAT = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm");
40  
41      private final String userName;
42      private final String terminalDevice;
43      private final long loginTime;
44      private final String host;
45  
46      public OSSession(String userName, String terminalDevice, long loginTime, String host) {
47          this.userName = userName;
48          this.terminalDevice = terminalDevice;
49          this.loginTime = loginTime;
50          this.host = host;
51      }
52  
53      /**
54       * Gets the login name of the user
55       *
56       * @return the userName
57       */
58      public String getUserName() {
59          return userName;
60      }
61  
62      /**
63       * Gets the terminal device (such as tty, pts, etc.) the user used to log in
64       *
65       * @return the terminalDevice
66       */
67      public String getTerminalDevice() {
68          return terminalDevice;
69      }
70  
71      /**
72       * Gets the time the user logged in
73       *
74       * @return the loginTime, in milliseconds since the 1970 epoch
75       */
76      public long getLoginTime() {
77          return loginTime;
78      }
79  
80      /**
81       * Gets the remote host from which the user logged in
82       *
83       * @return the host as either an IPv4 or IPv6 representation. If the host is
84       *         unspecified, may also be an empty string, depending on the platform.
85       */
86      public String getHost() {
87          return host;
88      }
89  
90      @Override
91      public String toString() {
92          String loginStr = loginTime == 0 ? "No login"
93                  : LocalDateTime.ofInstant(Instant.ofEpochMilli(loginTime), ZoneId.systemDefault()).format(LOGIN_FORMAT);
94          String hostStr = "";
95          if (!host.isEmpty() && !host.equals("::") && !host.equals("0.0.0.0")) {
96              hostStr = ", (" + host + ")";
97          }
98          return String.format("%s, %s, %s%s", userName, terminalDevice, loginStr, hostStr);
99      }
100 }
