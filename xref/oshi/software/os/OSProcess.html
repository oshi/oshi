23  /**
24   * A process is an instance of a computer program that is being executed. It
25   * contains the program code and its current activity. Depending on the
26   * operating system (OS), a process may be made up of multiple threads of
27   * execution that execute instructions concurrently.
28   *
29   * @author widdis[at]gmail[dot]com

60      /**
61       * Process Execution States
62       */
63      public enum State {
64          /**
65           * Intermediate state in process creation
66           */
67          NEW,
68          /**
69           * Actively executing process
70           */
71          RUNNING,
72          /**
73           * Interruptible sleep state
74           */
75          SLEEPING,
76          /**
77           * Blocked, uninterruptible sleep state
78           */
79          WAITING,
80          /**
81           * Intermediate state in process termination
82           */
83          ZOMBIE,
84          /**
85           * Stopped by the user, such as for debugging
86           */
87          STOPPED,
88          /**
89           * Other or unknown states not defined
90           */
91          OTHER
92      }

209     /**
210      * @return Returns the priority of this process.
211      *
212      *         For Linux and Unix, priority is a value in the range -20 to 19
213      *         (20 on some systems). The default priority is 0; lower priorities
214      *         cause more favorable scheduling.
215      *
216      *         For Windows, priority values can range from 0 (lowest priority)
217      *         to 31 (highest priority).
218      *
219      *         Mac OS X has 128 priority levels, ranging from 0 (lowest
220      *         priority) to 127 (highest priority). They are divided into
221      *         several major bands: 0 through 51 are the normal levels; the
222      *         default priority is 31. 52 through 79 are the highest priority
223      *         regular threads; 80 through 95 are for kernel mode threads; and
224      *         96 through 127 correspond to real-time threads, which are treated
225      *         differently than other threads by the scheduler.
226      */

552     /**
553      * Calculates CPU usage of this process.
554      *
555      * @return The proportion of up time that the process was executing in
556      *         kernel or user mode.
557      */
558     public double calculateCpuPercent() {
559         if (this.cpuPercent < 0d) {
560             this.cpuPercent = (getKernelTime() + getUserTime()) / (double) getUpTime();
561         }
562         return this.cpuPercent;
563     }

565     /**
566      * {@inheritDoc}
567      */
568     @Override
569     public String toString() {
570         StringBuilder builder = new StringBuilder("OSProcess@");
571         builder.append(Integer.toHexString(hashCode()));
572         builder.append("[processID=").append(this.processID);
573         builder.append(", name=").append(this.name).append(']');
574         return builder.toString();
575     }
