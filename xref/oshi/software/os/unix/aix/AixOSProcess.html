233     private int queryBitness() {
234         List<String> pflags = ExecutingCommand.runNative("pflags " + getProcessID());
235         for (String line : pflags) {
236             if (line.contains("data model")) {
237                 if (line.contains("LP32")) {
238                     return 32;
239                 } else if (line.contains("LP64")) {
240                     return 64;
241                 }
242             }
243         }
244         return 0;
245     }

247     @Override
248     public long getAffinityMask() {
249         // Need to capture BND field from ps
250         // ps -m -o THREAD -p 12345
251         // BND field for PID is either a dash (all processors) or the processor it's
252         // bound to, do 1L << # to get mask
253         long mask = 0L;
254         List<String> processAffinityInfoList = ExecutingCommand.runNative("ps -m -o THREAD -p " + getProcessID());
255         if (processAffinityInfoList.size() > 2) { // what happens when the process has not thread?
256             processAffinityInfoList.remove(0); // remove header row
257             processAffinityInfoList.remove(0); // remove process row
258             for (String processAffinityInfo : processAffinityInfoList) { // affinity information is in thread row
259                 Map<PsThreadColumns, String> threadMap = ParseUtil.stringToEnumMap(PsThreadColumns.class,
260                         processAffinityInfo.trim(), ' ');
261                 if (threadMap.containsKey(PsThreadColumns.COMMAND)
262                         && threadMap.get(PsThreadColumns.ST).charAt(0) != 'Z') { // only non-zombie threads
263                     String bnd = threadMap.get(PsThreadColumns.BND);
264                     if (bnd.charAt(0) == '-') { // affinity to all processors
265                         return this.affinityMask.get();
266                     } else {
267                         int affinity = ParseUtil.parseIntOrDefault(bnd, 0);
268                         mask |= 1L << affinity;
269                     }
270                 }
271             }
272         }
273         return mask;
274     }

276     @Override
277     public List<OSThread> getThreadDetails() {
278         List<String> threadListInfoPs = ExecutingCommand.runNative("ps -m -o THREAD -p " + getProcessID());
279         // 1st row is header, 2nd row is process data.
280         if (threadListInfoPs.size() > 2) {
281             List<OSThread> threads = new ArrayList<>();
282             threadListInfoPs.remove(0); // header removed
283             threadListInfoPs.remove(0); // process data removed
284             for (String threadInfo : threadListInfoPs) {
285                 Map<PsThreadColumns, String> threadMap = ParseUtil.stringToEnumMap(PsThreadColumns.class,
286                         threadInfo.trim(), ' ');
287                 if (threadMap.containsKey(PsThreadColumns.COMMAND)) {
288                     threads.add(new AixOSThread(getProcessID(), threadMap));
289                 }
290             }
291             return threads;
292         }
293         return Collections.emptyList();
294     }

301     @Override
302     public boolean updateAttributes() {
303         perfstat_process_t[] perfstat = procCpu.get();
304         List<String> procList = ExecutingCommand
305                 .runNative("ps -o " + AixOperatingSystem.PS_COMMAND_ARGS + " -p " + getProcessID());
306         // Parse array to map of user/system times
307         Map<Integer, Pair<Long, Long>> cpuMap = new HashMap<>();
308         for (perfstat_process_t stat : perfstat) {
309             cpuMap.put((int) stat.pid, new Pair<>((long) stat.ucpu_time, (long) stat.scpu_time));
310         }
311         if (procList.size() > 1) {
312             Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, procList.get(1).trim(), ' ');
313             // Check if last (thus all) value populated
314             if (psMap.containsKey(PsKeywords.ARGS)) {
315                 return updateAttributes(psMap, cpuMap);
316             }
317         }
318         this.state = State.INVALID;
319         return false;
320     }
321 
322     private boolean updateAttributes(Map<PsKeywords, String> psMap, Map<Integer, Pair<Long, Long>> cpuMap) {
323         long now = System.currentTimeMillis();
324         this.state = getStateFromOutput(psMap.get(PsKeywords.ST).charAt(0));
325         this.parentProcessID = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PPID), 0);
326         this.user = psMap.get(PsKeywords.USER);
327         this.userID = psMap.get(PsKeywords.UID);
328         this.group = psMap.get(PsKeywords.GROUP);
329         this.groupID = psMap.get(PsKeywords.GID);
330         this.threadCount = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.THCOUNT), 0);
331         this.priority = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PRI), 0);
332         // These are in KB, multiply
333         this.virtualSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.VSIZE), 0) << 10;
334         this.residentSetSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.RSSIZE), 0) << 10;
335         long elapsedTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.ETIME), 0L);
336         if (cpuMap.containsKey(getProcessID())) {
337             Pair<Long, Long> userSystem = cpuMap.get(getProcessID());
338             this.userTime = userSystem.getA();
339             this.kernelTime = userSystem.getB();
340         } else {
341             this.userTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.TIME), 0L);
342             this.kernelTime = 0L;
343         }
344         // Avoid divide by zero for processes up less than a second
345         this.upTime = elapsedTime < 1L ? 1L : elapsedTime;
346         while (this.upTime < this.userTime + this.kernelTime) {
347             this.upTime += 500L;
348         }
349         this.startTime = now - this.upTime;
350         this.name = psMap.get(PsKeywords.COMM);
351         this.majorFaults = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.PAGEIN), 0L);
352         this.commandLineBackup = psMap.get(PsKeywords.ARGS);
353         this.path = ParseUtil.whitespaces.split(this.commandLineBackup)[0];
354         return true;
355     }

357     /***
358      * Returns Enum STATE for the state value obtained from status string of
359      * thread/process.
360      *
361      * @param stateValue
362      *            state value from the status string
363      * @return The state
364      */
365     static State getStateFromOutput(char stateValue) {
366         State state;
367         switch (stateValue) {
368         case 'O':
369             state = INVALID;
370             break;
371         case 'R':
372         case 'A':
373             state = RUNNING;
374             break;
375         case 'I':
376             state = WAITING;
377             break;
378         case 'S':
379         case 'W':
380             state = SLEEPING;
381             break;
382         case 'Z':
383             state = ZOMBIE;
384             break;
385         case 'T':
386             state = STOPPED;
387             break;
388         default:
389             state = OTHER;
390             break;
391         }
392         return state;
393     }
