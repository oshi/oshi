5   package oshi.software.os.unix.freebsd;
6   
7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   import static oshi.util.Memoizer.defaultExpiration;
9   import static oshi.util.Memoizer.memoize;
10  
11  import java.lang.foreign.MemorySegment;
12  import java.util.function.Supplier;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.ffm.util.platform.unix.freebsd.BsdSysctlUtilFFM;
16  import oshi.software.common.os.unix.freebsd.FreeBsdInternetProtocolStats;
17  import oshi.util.ParseUtil;
18  import oshi.util.driver.unix.NetStat;
19  import oshi.util.tuples.Pair;
20  
21  /**
22   * FFM-backed FreeBSD Internet Protocol stats. Reads {@code net.inet.tcp.stats} and {@code net.inet.udp.stats} as raw
23   * memory and pulls counters at fixed offsets matching the kernel struct layout. Established-connection counts come from
24   * the shared {@link NetStat} command-line driver.
25   */
26  @ThreadSafe
27  public class FreeBsdInternetProtocolStatsFFM extends FreeBsdInternetProtocolStats {
28  
29      // Minimum sizes that contain all the fields we read. Sysctl outputs may grow over time but never shrink.
30      private static final int TCPSTAT_MIN_SIZE = 128;
31      private static final int UDPSTAT_MIN_SIZE = 84;
32  
33      private Supplier<Pair<Long, Long>> establishedv4v6 = memoize(NetStat::queryTcpnetstat, defaultExpiration());
34      private Supplier<MemorySegment> tcpstat = memoize(FreeBsdInternetProtocolStatsFFM::queryTcpstat,
35              defaultExpiration());
36      private Supplier<MemorySegment> udpstat = memoize(FreeBsdInternetProtocolStatsFFM::queryUdpstat,
37              defaultExpiration());
38  
39      @Override
40      public TcpStats getTCPv4Stats() {
41          MemorySegment m = tcpstat.get();
42          if (m == null || m.byteSize() < TCPSTAT_MIN_SIZE) {
43              return new TcpStats(establishedv4v6.get().getA(), 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L);
44          }
45          long connattempt = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 0));
46          long accepts = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 4));
47          long drops = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 12));
48          long conndrops = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 16));
49          long sndpack = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 64));
50          long sndrexmitpack = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 72));
51          long rcvpack = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 104));
52          long inErrors = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 112)) // rcvbadsum
53                  + ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 116)) // rcvbadoff
54                  + ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 120)) // rcvmemdrop
55                  + ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 124)); // rcvshort
56          return new TcpStats(establishedv4v6.get().getA(), connattempt, accepts, conndrops, drops, sndpack, rcvpack,
57                  sndrexmitpack, inErrors, 0L);
58      }
59  
60      @Override
61      public UdpStats getUDPv4Stats() {
62          MemorySegment m = udpstat.get();
63          if (m == null || m.byteSize() < UDPSTAT_MIN_SIZE) {
64              return new UdpStats(0L, 0L, 0L, 0L);
65          }
66          long ipackets = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 0));
67          long hdrops = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 4));
68          long badsum = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 8));
69          long badlen = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 12));
70          long opackets = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 36));
71          long noportmcast = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 48));
72          return new UdpStats(opackets, ipackets, noportmcast, hdrops + badsum + badlen);
73      }
74  
75      @Override
76      public UdpStats getUDPv6Stats() {
77          MemorySegment m = udpstat.get();
78          if (m == null || m.byteSize() < UDPSTAT_MIN_SIZE) {
79              return new UdpStats(0L, 0L, 0L, 0L);
80          }
81          long rcv6 = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 64));
82          long snd6 = ParseUtil.unsignedIntToLong(m.get(JAVA_INT, 80));
83          return new UdpStats(snd6, rcv6, 0L, 0L);
84      }
85  
86      private static MemorySegment queryTcpstat() {
87          return BsdSysctlUtilFFM.sysctl("net.inet.tcp.stats");
88      }
89  
90      private static MemorySegment queryUdpstat() {
91          return BsdSysctlUtilFFM.sysctl("net.inet.udp.stats");
92      }
93  }
