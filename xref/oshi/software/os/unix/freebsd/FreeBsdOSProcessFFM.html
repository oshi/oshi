33  /**
34   * FFM-backed FreeBSD OSProcess.
35   */
36  @ThreadSafe
37  public class FreeBsdOSProcessFFM extends FreeBsdOSProcess {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(FreeBsdOSProcessFFM.class);
40  
41      private static final int ARGMAX = BsdSysctlUtilFFM.sysctl("kern.argmax", 0);
42  
43      // CTL_KERN.KERN_PROC.<index>.<pid> sysctl MIB constants
44      private static final int CTL_KERN = 1;
45      private static final int KERN_PROC = 14;
46      private static final int KERN_PROC_ARGS = 7;
47      private static final int KERN_PROC_SV_NAME = 9;
48      private static final int KERN_PROC_ENV = 35;
49  
50      private final FreeBsdOperatingSystemFFM os;
51  
52      public FreeBsdOSProcessFFM(int pid, Map<BsdPsKeyword, String> psMap, FreeBsdOperatingSystemFFM os) {
53          super(pid);
54          this.os = os;
55          updateAttributes(psMap);
56      }
57  
58      @Override
59      protected List<String> queryArguments() {
60          if (ARGMAX <= 0) {
61              return Collections.emptyList();
62          }
63          return callInArenaOrDefault(arena -> {
64              MemorySegment mib = arena.allocateFrom(JAVA_INT, CTL_KERN, KERN_PROC, KERN_PROC_ARGS, getProcessID());
65              MemorySegment buf = arena.allocate(ARGMAX);
66              MemorySegment size = arena.allocateFrom(SIZE_T, (long) ARGMAX);
67              MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
68              int rc = FreeBsdLibcFunctions.sysctl(callState, mib, 4, buf, size, MemorySegment.NULL, 0L);
69              if (rc != 0) {
70                  LOG.warn(
71                          "Failed sysctl call for process arguments (kern.proc.args), process {} may not exist. Error code: {}",
72                          getProcessID(), getErrno(callState));
73                  return Collections.<String>emptyList();
74              }
75              long written = size.get(SIZE_T, 0);
76              byte[] bytes = getByteArrayFromNativePointer(buf, written, arena);
77              return Collections.unmodifiableList(ParseUtil.parseByteArrayToStrings(bytes));
78          }, LOG, org.slf4j.event.Level.WARN, "queryArguments failed", Collections.<String>emptyList());
79      }
80  
81      @Override
82      protected Map<String, String> queryEnvironmentVariables() {
83          if (ARGMAX <= 0) {
84              return Collections.emptyMap();
85          }
86          return callInArenaOrDefault(arena -> {
87              MemorySegment mib = arena.allocateFrom(JAVA_INT, CTL_KERN, KERN_PROC, KERN_PROC_ENV, getProcessID());
88              MemorySegment buf = arena.allocate(ARGMAX);
89              MemorySegment size = arena.allocateFrom(SIZE_T, (long) ARGMAX);
90              MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
91              int rc = FreeBsdLibcFunctions.sysctl(callState, mib, 4, buf, size, MemorySegment.NULL, 0L);
92              if (rc != 0) {
93                  LOG.warn(
94                          "Failed sysctl call for process environment variables (kern.proc.env), process {} may not exist. Error code: {}",
95                          getProcessID(), getErrno(callState));
96                  return Collections.<String, String>emptyMap();
97              }
98              long written = size.get(SIZE_T, 0);
99              byte[] bytes = getByteArrayFromNativePointer(buf, written, arena);
100             return Collections.unmodifiableMap(ParseUtil.parseByteArrayToStringMap(bytes));
101         }, LOG, org.slf4j.event.Level.WARN, "queryEnvironmentVariables failed", Collections.<String, String>emptyMap());
102     }
103 
104     @Override
105     public String getCurrentWorkingDirectory() {
106         return ProcstatUtil.getCwd(getProcessID());
107     }
108 
109     @Override
110     public long getOpenFiles() {
111         return ProcstatUtil.getOpenFiles(getProcessID());
112     }
113 
114     @Override
115     public long getSoftOpenFileLimit() {
116         if (getProcessID() == this.os.getProcessId()) {
117             return rlimitNofile(true);
118         }
119         return getProcessOpenFileLimit(getProcessID(), 1);
120     }
121 
122     @Override
123     public long getHardOpenFileLimit() {
124         if (getProcessID() == this.os.getProcessId()) {
125             return rlimitNofile(false);
126         }
127         return getProcessOpenFileLimit(getProcessID(), 2);
128     }
129 
130     private static long rlimitNofile(boolean soft) {
131         try (Arena arena = Arena.ofConfined()) {
132             MemorySegment rlim = arena.allocate(RLIMIT_LAYOUT);
133             if (FreeBsdLibcFunctions.getrlimit(RLIMIT_NOFILE, rlim) != 0) {
134                 return -1L;
135             }
136             return soft ? FreeBsdLibcFunctions.rlimitCur(rlim) : FreeBsdLibcFunctions.rlimitMax(rlim);
137         } catch (Throwable t) {
138             LOG.warn("getrlimit(RLIMIT_NOFILE) failed", t);
139             return -1L;
140         }
141     }
142 
143     @Override
144     protected int queryBitness() {
145         return callInArenaOrDefault(arena -> {
146             MemorySegment mib = arena.allocateFrom(JAVA_INT, CTL_KERN, KERN_PROC, KERN_PROC_SV_NAME, getProcessID());
147             MemorySegment buf = arena.allocate(32L);
148             MemorySegment size = arena.allocateFrom(SIZE_T, 32L);
149             MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
150             if (0 != FreeBsdLibcFunctions.sysctl(callState, mib, 4, buf, size, MemorySegment.NULL, 0L)) {
151                 return 0;
152             }
153             String elf = buf.getString(0);
154             if (elf.contains("ELF32")) {
155                 return 32;
156             } else if (elf.contains("ELF64")) {
157                 return 64;
158             }
159             return 0;
160         }, LOG, org.slf4j.event.Level.WARN, "queryBitness failed", 0);
161     }
