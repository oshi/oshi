39  /**
40   * The Solaris File System contains {@link oshi.software.os.OSFileStore}s which
41   * are a storage pool, device, partition, volume, concrete file system or other
42   * implementation specific means of file storage. In Linux, these are found in
43   * the /proc/mount filesystem, excluding temporary and kernel mounts.
44   */
45  public class FreeBsdFileSystem extends AbstractFileSystem {
46  
47      // Linux defines a set of virtual file systems
48      private static final List<String> PSEUDO_FS = Arrays.asList( //
49              "procfs", // Proc file system
50              "devfs", // Dev temporary file system
51              "ctfs", // Contract file system
52              "fdescfs", // fd
53              "objfs", // Object file system
54              "mntfs", // Mount file system
55              "sharefs", // Share file system
56              "lofs", // Library file system
57              "autofs" // Auto mounting fs
58      // "tmpfs", // Temporary file system
59      // NOTE: tmpfs is evaluated apart, because Solaris uses it for
60      // RAMdisks
61      );
62  
63      // System path mounted as tmpfs
64      private static final List<String> TMP_FS_PATHS = Arrays.asList("/system", "/tmp", "/dev/fd");
65  
66      @Override
67      public OSFileStore[] getFileStores(boolean localOnly) {
68          // Find any partition UUIDs and map them
69          Map<String, String> uuidMap = new HashMap<>();
70          // Now grab dmssg output
71          String device = "";
72          for (String line : ExecutingCommand.runNative("geom part list")) {
73              if (line.contains("Name: ")) {
74                  device = line.substring(line.lastIndexOf(' ') + 1);
75              }
76              // If we aren't working with a current partition, continue
77              if (device.isEmpty()) {
78                  continue;
79              }
80              line = line.trim();
81              if (line.startsWith("rawuuid:")) {
82                  uuidMap.put(device, line.substring(line.lastIndexOf(' ') + 1));
83                  device = "";
84              }
85          }
86  
87          List<OSFileStore> fsList = new ArrayList<>();
88  
89          // Get inode usage data
90          Map<String, Long> inodeFreeMap = new HashMap<>();
91          Map<String, Long> inodeTotalMap = new HashMap<>();
92          for (String line : ExecutingCommand.runNative("df -i")) {
93              /*- Sample Output:
94              Filesystem    1K-blocks   Used   Avail Capacity iused  ifree %iused  Mounted on
95              /dev/twed0s1a   2026030 584112 1279836    31%    2751 279871    1%   /
96              */
97              if (line.startsWith("/")) {
98                  String[] split = ParseUtil.whitespaces.split(line);
99                  if (split.length > 7) {
100                     inodeFreeMap.put(split[0], ParseUtil.parseLongOrDefault(split[6], 0L));
101                     // total is used + free
102                     inodeTotalMap.put(split[0],
103                             inodeFreeMap.get(split[0]) + ParseUtil.parseLongOrDefault(split[5], 0L));
104                 }
105             }
106         }
107 
108         // Get mount table
109         for (String fs : ExecutingCommand.runNative("mount -p")) {
110             String[] split = ParseUtil.whitespaces.split(fs);
111             if (split.length < 5) {
112                 continue;
113             }
114             // 1st field is volume name
115             // 2nd field is mount point
116             // 3rd field is fs type
117             // 4th field is options
118             // other fields ignored
119             String volume = split[0];
120             String path = split[1];
121             String type = split[2];
122             String options = split[3];
123 
124             // Skip non-local drives if requested, and exclude pseudo file systems
125             if ((localOnly && NETWORK_FS_TYPES.contains(type)) || PSEUDO_FS.contains(type) || path.equals("/dev")
126                     || ParseUtil.filePathStartsWith(TMP_FS_PATHS, path)
127                     || volume.startsWith("rpool") && !path.equals("/")) {
128                 continue;
129             }
130 
131             String name = path.substring(path.lastIndexOf('/') + 1);
132             // Special case for /, pull last element of volume instead
133             if (name.isEmpty()) {
134                 name = volume.substring(volume.lastIndexOf('/') + 1);
135             }
136             File f = new File(path);
137             long totalSpace = f.getTotalSpace();
138             long usableSpace = f.getUsableSpace();
139             long freeSpace = f.getFreeSpace();
140 
141             String description;
142             if (volume.startsWith("/dev") || path.equals("/")) {
143                 description = "Local Disk";
144             } else if (volume.equals("tmpfs")) {
145                 description = "Ram Disk";
146             } else if (NETWORK_FS_TYPES.contains(type)) {
147                 description = "Network Disk";
148             } else {
149                 description = "Mount Point";
150             }
151             // Match UUID
152             String uuid = uuidMap.getOrDefault(name, "");
153 
154             // Add to the list
155             OSFileStoreml#OSFileStore">OSFileStore osStore = new OSFileStore();
156             osStore.setName(name);
157             osStore.setVolume(volume);
158             osStore.setMount(path);
159             osStore.setDescription(description);
160             osStore.setType(type);
161             osStore.setOptions(options);
162             osStore.setUUID(uuid);
163             osStore.setFreeSpace(freeSpace);
164             osStore.setUsableSpace(usableSpace);
165             osStore.setTotalSpace(totalSpace);
166             osStore.setFreeInodes(inodeFreeMap.containsKey(path) ? inodeFreeMap.get(path) : 0L);
167             osStore.setTotalInodes(inodeTotalMap.containsKey(path) ? inodeTotalMap.get(path) : 0L);
168             fsList.add(osStore);
169         }
170         return fsList.toArray(new OSFileStore[0]);
171     }

173     /** {@inheritDoc} */
174     @Override
175     public long getOpenFileDescriptors() {
176         return BsdSysctlUtil.sysctl("kern.openfiles", 0);
177     }
178 
179     /** {@inheritDoc} */
180     @Override
181     public long getMaxFileDescriptors() {
182         return BsdSysctlUtil.sysctl("kern.maxfiles", 0);
183     }

185     /**
186      * <p>
187      * updateFileStoreStats.
188      * </p>
189      *
190      * @param osFileStore
191      *            a {@link oshi.software.os.OSFileStore} object.
192      * @return a boolean.
193      */
194     public static boolean updateFileStoreStats(OSFileStore osFileStore) {
195         // Just as fast to query all of them
196         for (OSFileStore fileStore : new FreeBsdFileSystem().getFileStores()) {
197             if (osFileStore.getName().equals(fileStore.getName())
198                     && osFileStore.getVolume().equals(fileStore.getVolume())
199                     && osFileStore.getMount().equals(fileStore.getMount())) {
200                 osFileStore.setLogicalVolume(fileStore.getLogicalVolume());
201                 osFileStore.setDescription(fileStore.getDescription());
202                 osFileStore.setType(fileStore.getType());
203                 osFileStore.setUUID(fileStore.getUUID());
204                 osFileStore.setFreeSpace(fileStore.getFreeSpace());
205                 osFileStore.setUsableSpace(fileStore.getUsableSpace());
206                 osFileStore.setTotalSpace(fileStore.getTotalSpace());
207                 osFileStore.setFreeInodes(fileStore.getFreeInodes());
208                 osFileStore.setTotalInodes(fileStore.getTotalInodes());
209                 return true;
210             }
211         }
212         return false;
213     }
