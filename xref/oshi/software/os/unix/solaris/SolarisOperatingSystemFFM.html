28  /**
29   * FFM-backed Solaris OperatingSystem. Uses the legacy {@code kstat} chain only; Kstat2 exists only on the JDK 17-capped
30   * latest Solaris, so FFM (JDK 25) never needs it.
31   */
32  @ThreadSafe
33  public class SolarisOperatingSystemFFM extends SolarisOperatingSystem {
34  
35      private static final Logger LOG = LoggerFactory.getLogger(SolarisOperatingSystemFFM.class);
36  
37      private static final long BOOTTIME = querySystemBootTime();
38  
39      @Override
40      public FileSystem getFileSystem() {
41          return new SolarisFileSystemFFM();
42      }
43  
44      @Override
45      protected OSProcess createProcess(int pid) {
46          return new SolarisOSProcessFFM(pid, this);
47      }
48  
49      @Override
50      public int getProcessId() {
51          return ForeignFunctions.callInArenaIntOrDefault(arena -> SolarisLibcFunctions.getpid(), LOG, LogLevel.WARN,
52                  "getpid failed", 0);
53      }
54  
55      @Override
56      public int getThreadId() {
57          return ForeignFunctions.callInArenaIntOrDefault(arena -> SolarisLibcFunctions.thr_self(), LOG, LogLevel.WARN,
58                  "thr_self failed", 0);
59      }
60  
61      @Override
62      public OSThread getCurrentThread() {
63          return new SolarisOSThreadFFM(getProcessId(), getThreadId());
64      }
65  
66      @Override
67      public long getSystemUptime() {
68          try (KstatChain kc = KstatUtilFFM.openChain()) {
69              MemorySegment ksp = kc.lookup("unix", 0, "system_misc");
70              if (ksp.address() != 0L && kc.read(ksp)) {
71                  return LibKstatFunctions.kstatSnaptime(ksp) / 1_000_000_000L;
72              }
73          }
74          return 0L;
75      }
76  
77      @Override
78      public long getSystemBootTime() {
79          return BOOTTIME;
80      }
81  
82      private static long querySystemBootTime() {
83          try (KstatChain kc = KstatUtilFFM.openChain()) {
84              MemorySegment ksp = kc.lookup("unix", 0, "system_misc");
85              if (ksp.address() != 0L && kc.read(ksp)) {
86                  return KstatUtilFFM.dataLookupLong(ksp, "boot_time");
87              }
88          }
89          return System.currentTimeMillis() / 1000L;
90      }
91  
92      @Override
93      public NetworkParams getNetworkParams() {
94          return new SolarisNetworkParamsFFM();
95      }
96  
97      @Override
98      public List<OSSession> getSessions() {
99          return USE_WHO_COMMAND ? super.getSessions() : WhoFFM.queryUtxent();
100     }
101 }
