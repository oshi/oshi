197     private int queryBitness() {
198         List<String> pflags = ExecutingCommand.runNative("pflags " + getProcessID());
199         for (String line : pflags) {
200             if (line.contains("data model")) {
201                 if (line.contains("LP32")) {
202                     return 32;
203                 } else if (line.contains("LP64")) {
204                     return 64;
205                 }
206             }
207         }
208         return 0;
209     }

211     @Override
212     public long getAffinityMask() {
213         long bitMask = 0L;
214         String cpuset = ExecutingCommand.getFirstAnswer("pbind -q " + getProcessID());
215         // Sample output:
216         // <empty string if no binding>
217         // pid 101048 strongly bound to processor(s) 0 1 2 3.
218         if (cpuset.isEmpty()) {
219             List<String> allProcs = ExecutingCommand.runNative("psrinfo");
220             for (String proc : allProcs) {
221                 String[] split = ParseUtil.whitespaces.split(proc);
222                 int bitToSet = ParseUtil.parseIntOrDefault(split[0], -1);
223                 if (bitToSet >= 0) {
224                     bitMask |= 1L << bitToSet;
225                 }
226             }
227             return bitMask;
228         } else if (cpuset.endsWith(".") && cpuset.contains("strongly bound to processor(s)")) {
229             String parse = cpuset.substring(0, cpuset.length() - 1);
230             String[] split = ParseUtil.whitespaces.split(parse);
231             for (int i = split.length - 1; i >= 0; i--) {
232                 int bitToSet = ParseUtil.parseIntOrDefault(split[i], -1);
233                 if (bitToSet >= 0) {
234                     bitMask |= 1L << bitToSet;
235                 } else {
236                     // Once we run into the word processor(s) we're done
237                     break;
238                 }
239             }
240         }
241         return bitMask;
242     }

244     @Override
245     public List<OSThread> getThreadDetails() {
246         List<String> threadListInfo1 = ExecutingCommand
247                 .runNative("ps -o lwp,s,etime,stime,time,addr,pri -p " + getProcessID());
248         List<String> threadListInfo2 = ExecutingCommand.runNative("prstat -L -v -p " + getProcessID() + " 1 1");
249         Map<Integer, String[]> threadMap = parseAndMergePSandPrstatInfo(threadListInfo1, 0, 7, threadListInfo2, true);
250         if (threadMap.keySet().size() > 1) {
251             return threadMap.entrySet().stream().map(entry -> new SolarisOSThread(getProcessID(), entry.getValue()))
252                     .collect(Collectors.toList());
253         }
254         return Collections.emptyList();
255     }

257     @Override
258     public boolean updateAttributes() {
259         int pid = getProcessID();
260         List<String> procList = ExecutingCommand
261                 .runNative("ps -o s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args -p " + pid);
262         List<String> procList2 = ExecutingCommand.runNative("prstat -v -p " + pid + " 1 1");
263         Map<Integer, String[]> processMap = parseAndMergePSandPrstatInfo(procList, 1, 15, procList2, false);
264         if (processMap.containsKey(pid)) {
265             return updateAttributes(processMap.get(getProcessID()));
266         }
267         this.state = State.INVALID;
268         return false;
269     }

271     private boolean updateAttributes(String[] split) {
272         long now = System.currentTimeMillis();
273         this.state = getStateFromOutput(split[0].charAt(0));
274         this.parentProcessID = ParseUtil.parseIntOrDefault(split[2], 0);
275         this.user = split[3];
276         this.userID = split[4];
277         this.group = split[5];
278         this.groupID = split[6];
279         this.threadCount = ParseUtil.parseIntOrDefault(split[7], 0);
280         this.priority = ParseUtil.parseIntOrDefault(split[8], 0);
281         // These are in KB, multiply
282         this.virtualSize = ParseUtil.parseLongOrDefault(split[9], 0) * 1024;
283         this.residentSetSize = ParseUtil.parseLongOrDefault(split[10], 0) * 1024;
284         // Avoid divide by zero for processes up less than a second
285         long elapsedTime = ParseUtil.parseDHMSOrDefault(split[11], 0L);
286         this.upTime = elapsedTime < 1L ? 1L : elapsedTime;
287         this.startTime = now - this.upTime;
288         this.kernelTime = 0L;
289         this.userTime = ParseUtil.parseDHMSOrDefault(split[12], 0L);
290         this.path = split[13];
291         this.name = this.path.substring(this.path.lastIndexOf('/') + 1);
292         this.commandLine = split[14];
293         long nonVoluntaryContextSwitches = ParseUtil.parseLongOrDefault(split[15], 0L);
294         long voluntaryContextSwitches = ParseUtil.parseLongOrDefault(split[16], 0L);
295         this.contextSwitches = voluntaryContextSwitches + nonVoluntaryContextSwitches;
296 
297         return true;
298     }

300     /**
301      * Merges results of a ps and prstat query, since Solaris process and thread
302      * details are not available in a single command. Package private to permit
303      * access by SolarisOperatingSystem and SolarisOSThread.
304      *
305      * @param psInfo
306      *            output from ps command.
307      * @param psKeyIndex
308      *            which field of the ps split should be the key (e.g., pid or tid)
309      * @param psLength
310      *            how many fields to split
311      * @param prstatInfo
312      *            output from the prstat command.
313      * @param useTid
314      *            If true, parses thread id (slash-delimited last field), otherwise
315      *            uses process id (field 0)
316      * @return a map with key as thread id and an array of command outputs as value
317      */
318     static Map<Integer, String[]> parseAndMergePSandPrstatInfo(List<String> psInfo, int psKeyIndex, int psLength,
319             List<String> prstatInfo, boolean useTid) {
320         Map<Integer, String[]> map = new HashMap<>();
321         if (psInfo.size() > 1) { // first row is header
322             psInfo.stream().skip(1).forEach(info -> {
323                 String[] psSplit = ParseUtil.whitespaces.split(info.trim(), psLength);
324                 String[] mergedSplit = new String[psLength + 2];
325                 if (psSplit.length == psLength) {
326                     for (int idx = 0; idx < psLength; idx++) {
327                         if (idx == psKeyIndex) {
328                             map.put(ParseUtil.parseIntOrDefault(psSplit[idx], 0), mergedSplit);
329                         }
330                         mergedSplit[idx] = psSplit[idx];
331                     }
332                 }
333             });
334             // 0-pid, 1-username, 2-usertime, 3-sys, 4-trp, 5-tfl, 6-dfl, 7-lck, 8-slp,
335             // 9-lat, 10-vcx, 11-icx, 12-scl, 13-sig, 14-process/lwpid
336             if (prstatInfo.size() > 1) { // first row is header
337                 prstatInfo.stream().skip(1).forEach(threadInfo -> {
338                     String[] splitPrstat = ParseUtil.whitespaces.split(threadInfo.trim());
339                     if (splitPrstat.length == 15) {
340                         String id = splitPrstat[0]; // pid
341                         if (useTid) {
342                             int idxAfterForwardSlash = splitPrstat[14].lastIndexOf('/') + 1; // format is process/lwpid
343                             if (idxAfterForwardSlash > 0 && idxAfterForwardSlash < splitPrstat[14].length()) {
344                                 id = splitPrstat[14].substring(idxAfterForwardSlash); // getting the thread id
345                             }
346                         }
347                         String[] existingSplit = map.get(Integer.parseInt(id));
348                         if (existingSplit != null) { // if thread wasn't in ps command output
349                             existingSplit[psLength] = splitPrstat[10]; // voluntary context switch
350                             existingSplit[psLength + 1] = splitPrstat[11]; // involuntary context switch
351                         }
352                     }
353                 });
354             }
355         }
356         return map;
357     }

359     /***
360      * Returns Enum STATE for the state value obtained from status string of
361      * thread/process.
362      *
363      * @param stateValue
364      *            state value from the status string
365      * @return The state
366      */
367     static State getStateFromOutput(char stateValue) {
368         State state;
369         switch (stateValue) {
370         case 'O':
371             state = RUNNING;
372             break;
373         case 'S':
374             state = SLEEPING;
375             break;
376         case 'R':
377         case 'W':
378             state = WAITING;
379             break;
380         case 'Z':
381             state = ZOMBIE;
382             break;
383         case 'T':
384             state = STOPPED;
385             break;
386         default:
387             state = OTHER;
388             break;
389         }
390         return state;
391     }
