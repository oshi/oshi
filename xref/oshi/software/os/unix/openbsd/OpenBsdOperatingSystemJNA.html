5   package oshi.software.os.unix.openbsd;
6   
7   import java.util.Map;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.jna.platform.unix.OpenBsdLibc;
11  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
12  import oshi.software.common.os.unix.openbsd.OpenBsdOperatingSystem;
13  import oshi.software.os.FileSystem;
14  import oshi.software.os.OSProcess;
15  import oshi.util.platform.unix.openbsd.OpenBsdSysctlUtil;
16  
17  /**
18   * JNA-backed OpenBSD operating system.
19   */
20  @ThreadSafe
21  public class OpenBsdOperatingSystemJNA extends OpenBsdOperatingSystem {
22  
23      @Override
24      protected String querySysctl(int[] mib, String def) {
25          return OpenBsdSysctlUtil.sysctl(mib, def);
26      }
27  
28      @Override
29      public FileSystem getFileSystem() {
30          return new OpenBsdFileSystemJNA();
31      }
32  
33      @Override
34      public int getProcessId() {
35          return OpenBsdLibc.INSTANCE.getpid();
36      }
37  
38      @Override
39      public int getThreadId() {
40          return OpenBsdLibc.INSTANCE.getthrid();
41      }
42  
43      @Override
44      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
45          return new OpenBsdOSProcessJNA(pid, psMap, this);
46      }
47  }
