5   package oshi.software.os.unix.openbsd;
6   
7   import static oshi.jna.platform.unix.OpenBsdLibc.CTL_KERN;
8   import static oshi.jna.platform.unix.OpenBsdLibc.KERN_OSRELEASE;
9   import static oshi.jna.platform.unix.OpenBsdLibc.KERN_OSTYPE;
10  import static oshi.jna.platform.unix.OpenBsdLibc.KERN_VERSION;
11  
12  import java.util.Map;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.jna.platform.unix.OpenBsdLibc;
16  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
17  import oshi.software.common.os.unix.openbsd.OpenBsdOperatingSystem;
18  import oshi.software.os.FileSystem;
19  import oshi.software.os.OSProcess;
20  import oshi.util.platform.unix.openbsd.OpenBsdSysctlUtil;
21  import oshi.util.tuples.Pair;
22  
23  /**
24   * JNA-backed OpenBSD operating system.
25   */
26  @ThreadSafe
27  public class OpenBsdOperatingSystemJNA extends OpenBsdOperatingSystem {
28  
29      @Override
30      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
31          int[] mib = new int[2];
32          mib[0] = CTL_KERN;
33          mib[1] = KERN_OSTYPE;
34          String family = OpenBsdSysctlUtil.sysctl(mib, "OpenBSD");
35          mib[1] = KERN_OSRELEASE;
36          String version = OpenBsdSysctlUtil.sysctl(mib, "");
37          mib[1] = KERN_VERSION;
38          String versionInfo = OpenBsdSysctlUtil.sysctl(mib, "");
39          String buildNumber = versionInfo.split(":")[0].replace(family, "").replace(version, "").trim();
40  
41          return new Pair<>(family, new OSVersionInfo(version, null, buildNumber));
42      }
43  
44      @Override
45      public FileSystem getFileSystem() {
46          return new OpenBsdFileSystemJNA();
47      }
48  
49      @Override
50      public int getProcessId() {
51          return OpenBsdLibc.INSTANCE.getpid();
52      }
53  
54      @Override
55      public int getThreadId() {
56          return OpenBsdLibc.INSTANCE.getthrid();
57      }
58  
59      @Override
60      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
61          return new OpenBsdOSProcessJNA(pid, psMap, this);
62      }
63  }
