5   package oshi.software.os.unix.openbsd;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.util.platform.unix.openbsd.OpenBsdSysctlUtil;
9   
10  /**
11   * The OpenBSD File System contains {@link oshi.software.os.OSFileStore}s which are a storage pool, device, partition,
12   * volume, concrete file system or other implementation specific means of file storage.
13   */
14  @ThreadSafe
15  public class OpenBsdFileSystem extends oshi.software.common.os.unix.openbsd.OpenBsdFileSystem {
16  
17      @Override
18      public long getOpenFileDescriptors() {
19          return OpenBsdSysctlUtil.sysctl("kern.nfiles", 0);
20      }
21  
22      @Override
23      public long getMaxFileDescriptors() {
24          return OpenBsdSysctlUtil.sysctl("kern.maxfiles", 0);
25      }
26  
27      @Override
28      public long getMaxFileDescriptorsPerProcess() {
29          return OpenBsdSysctlUtil.sysctl("kern.maxfilesperproc", 0);
30      }
31  }
