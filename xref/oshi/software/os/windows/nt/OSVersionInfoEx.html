33  /**
34   * Contains operating system version information. The information includes major
35   * and minor version numbers, a build number, a platform identifier, and
36   * descriptive text about the operating system.
37   * 
38   * @author dblock[at]dblock[dot]org
39   */
40  public class OSVersionInfoEx implements OperatingSystemVersion {
41  	private static final Logger LOG = LoggerFactory.getLogger(OSVersionInfoEx.class);
42  
43  	private OSVERSIONINFOEX _versionInfo;
44  
45  	public OSVersionInfoEx() {
46  		this._versionInfo = new OSVERSIONINFOEX();
47  		if (!Kernel32.INSTANCE.GetVersionEx(this._versionInfo)) {
48  			LOG.error("Failed to Initialize OSVersionInfoEx. Error code: {}", Kernel32.INSTANCE.GetLastError());
49  			this._versionInfo = null;
50  		} else
51  			LOG.debug("Initialized OSVersionInfoEx");
52  	}
53  
54  	/**
55  	 * The major version number of the operating system.
56  	 * 
57  	 * @return The major version within the following supported operating
58  	 *         systems. Windows 8: 6.2 Windows Server 2012: 6.2 Windows 7: 6.1
59  	 *         Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows
60  	 *         Vista: 6.0 Windows Server 2003 R2: 5.2 Windows Home Server: 5.2
61  	 *         Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
62  	 *         Windows XP: 5.1 Windows 2000: 5.0
63  	 */
64  	public int getMajor() {
65  		if (this._versionInfo == null) {
66  			LOG.warn("OSVersionInfoEx not initialized. No version data available");
67  			return 0;
68  		}
69  		return this._versionInfo.dwMajorVersion.intValue();
70  	}
71  
72  	/**
73  	 * The minor version number of the operating system.
74  	 * 
75  	 * @return The minor version within the following supported operating
76  	 *         systems. Windows 8: 6.2 Windows Server 2012: 6.2 Windows 7: 6.1
77  	 *         Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows
78  	 *         Vista: 6.0 Windows Server 2003 R2: 5.2 Windows Home Server: 5.2
79  	 *         Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
80  	 *         Windows XP: 5.1 Windows 2000: 5.0
81  	 */
82  	public int getMinor() {
83  		if (this._versionInfo == null) {
84  			LOG.warn("OSVersionInfoEx not initialized. No version data available");
85  			return 0;
86  		}
87  		return this._versionInfo.dwMinorVersion.intValue();
88  	}
89  
90  	/**
91  	 * The build number of the operating system.
92  	 * 
93  	 * @return Build number.
94  	 */
95  	public int getBuildNumber() {
96  		if (this._versionInfo == null) {
97  			LOG.warn("OSVersionInfoEx not initialized. No build number available");
98  			return 0;
99  		}
100 		return this._versionInfo.dwBuildNumber.intValue();
101 	}
102 
103 	/**
104 	 * The operating system platform. This member can be VER_PLATFORM_WIN32_NT.
105 	 * 
106 	 * @return Platform ID.
107 	 */
108 	public int getPlatformId() {
109 		if (this._versionInfo == null) {
110 			LOG.warn("OSVersionInfoEx not initialized. No platform id available");
111 			return 0;
112 		}
113 		return this._versionInfo.dwPlatformId.intValue();
114 	}
115 
116 	/**
117 	 * String, such as "Service Pack 3", that indicates the latest Service Pack
118 	 * installed on the system. If no Service Pack has been installed, the
119 	 * string is empty.
120 	 * 
121 	 * @return Service pack.
122 	 */
123 	public String getServicePack() {
124 		if (this._versionInfo == null) {
125 			LOG.warn("OSVersionInfoEx not initialized. No service pack data available");
126 			return "";
127 		}
128 		return Native.toString(this._versionInfo.szCSDVersion);
129 	}
130 
131 	/**
132 	 * A bit mask that identifies the product suites available on the system.
133 	 * 
134 	 * @return Suite mask.
135 	 */
136 	public int getSuiteMask() {
137 		if (this._versionInfo == null) {
138 			LOG.warn("OSVersionInfoEx not initialized. No suite mask data available");
139 			return 0;
140 		}
141 
142 		return this._versionInfo.wSuiteMask.intValue();
143 	}
144 
145 	/**
146 	 * Any additional information about the system.
147 	 * 
148 	 * @return Product type.
149 	 */
150 	public byte getProductType() {
151 		if (this._versionInfo == null) {
152 			LOG.warn("OSVersionInfoEx not initialized. No product type available");
153 			return 0;
154 		}
155 		return this._versionInfo.wProductType;
156 	}
157 
158 	@Override
159 	public String toString() {
160 		if (this._versionInfo == null) {
161 			LOG.warn("OSVersionInfoEx not initialized. No version data available");
162 			return "Unknown";
163 		}
164 
165 		String version = null;
166 
167 		// see
168 		// http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
169 		if (getPlatformId() == WinNT.VER_PLATFORM_WIN32_NT) {
170 			boolean ntWorkstation = getProductType() == WinNT.VER_NT_WORKSTATION;
171 			if (getMajor() == 10 && getMinor() == 0 && ntWorkstation) {
172 				version = "10";
173 			} else if (getMajor() == 10 && getMinor() == 0 && !ntWorkstation) {
174 				version = "Server 2016";
175 			} else if (getMajor() == 6 && getMinor() == 3 && ntWorkstation) {
176 				version = "8.1";
177 			} else if (getMajor() == 6 && getMinor() == 3 && !ntWorkstation) {
178 				version = "Server 2012 R2";
179 			} else if (getMajor() == 6 && getMinor() == 2 && ntWorkstation) {
180 				version = "8";
181 			} else if (getMajor() == 6 && getMinor() == 2 && !ntWorkstation) {
182 				version = "Server 2012";
183 			} else if (getMajor() == 6 && getMinor() == 1 && ntWorkstation) {
184 				version = "7";
185 			} else if (getMajor() == 6 && getMinor() == 1 && !ntWorkstation) {
186 				version = "Server 2008 R2";
187 			} else if (getMajor() == 6 && getMinor() == 0 && !ntWorkstation) {
188 				version = "Server 2008";
189 			} else if (getMajor() == 6 && getMinor() == 0 && ntWorkstation) {
190 				version = "Vista";
191 			} else if (getMajor() == 5 && getMinor() == 2 && !ntWorkstation) {
192 				version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003" : "Server 2003 R2";
193 			} else if (getMajor() == 5 && getMinor() == 2 && ntWorkstation) {
194 				version = "XP"; // 64 bits
195 			} else if (getMajor() == 5 && getMinor() == 1) {
196 				version = "XP"; // 32 bits
197 			} else if (getMajor() == 5 && getMinor() == 0) {
198 				version = "2000";
199 			} else if (getMajor() == 4) {
200 				version = "NT 4";
201 				if ("Service Pack 6".equals(getServicePack())) {
202 					if (Advapi32Util.registryKeyExists(WinReg.HKEY_LOCAL_MACHINE,
203 							"SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Hotfix\\Q246009")) {
204 						return "NT4 SP6a";
205 					}
206 				}
207 
208 			} else {
209 				throw new RuntimeException("Unsupported Windows NT version: " + this._versionInfo.toString());
210 			}
211 
212 			if (this._versionInfo.wServicePackMajor.intValue() > 0) {
213 				version = version + " SP" + this._versionInfo.wServicePackMajor.intValue();
214 			}
215 
216 		} else if (getPlatformId() == WinNT.VER_PLATFORM_WIN32_WINDOWS) {
217 			if (getMajor() == 4 && getMinor() == 90) {
218 				version = "ME";
219 			} else if (getMajor() == 4 && getMinor() == 10) {
220 				if (this._versionInfo.szCSDVersion[1] == 'A') {
221 					version = "98 SE";
222 				} else {
223 					version = "98";
224 				}
225 			} else if (getMajor() == 4 && getMinor() == 0) {
226 				if (this._versionInfo.szCSDVersion[1] == 'C' || this._versionInfo.szCSDVersion[1] == 'B') {
227 					version = "95 OSR2";
228 				} else {
229 					version = "95";
230 				}
231 			} else {
232 				throw new RuntimeException("Unsupported Windows 9x version: " + this._versionInfo.toString());
233 			}
234 		} else {
235 			throw new RuntimeException("Unsupported Windows platform: " + this._versionInfo.toString());
236 		}
237 
238 		return version;
239 	}
240 
241 	public OSVersionInfoEx(OSVERSIONINFOEX versionInfo) {
242 		this._versionInfo = versionInfo;
243 	}
244 }
