17  package oshi.software.os.windows.nt;
18  
19  import com.sun.jna.Native;
20  import com.sun.jna.platform.win32.Advapi32Util;
21  import com.sun.jna.platform.win32.Kernel32;
22  import com.sun.jna.platform.win32.User32;
23  import com.sun.jna.platform.win32.Win32Exception;
24  import com.sun.jna.platform.win32.WinNT;
25  import com.sun.jna.platform.win32.WinNT.OSVERSIONINFOEX;
26  import com.sun.jna.platform.win32.WinReg;
27  import com.sun.jna.platform.win32.WinUser;

29  import oshi.software.os.OperatingSystemVersion;

31  /**
32   * Contains operating system version information. The information includes major
33   * and minor version numbers, a build number, a platform identifier, and
34   * descriptive text about the operating system.
35   * 
36   * @author dblock[at]dblock[dot]org
37   */
38  public class OSVersionInfoEx implements OperatingSystemVersion {
39  	private OSVERSIONINFOEX _versionInfo;
40  
41  	public OSVersionInfoEx() {
42  		this._versionInfo = new OSVERSIONINFOEX();
43  		if (!Kernel32.INSTANCE.GetVersionEx(this._versionInfo)) {
44  			throw new Win32Exception(Kernel32.INSTANCE.GetLastError());
45  		}
46  	}
47  
48  	/**
49  	 * The major version number of the operating system.
50  	 * 
51  	 * @return The major version within the following supported operating
52  	 *         systems. Windows 8: 6.2 Windows Server 2012: 6.2 Windows 7: 6.1
53  	 *         Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows
54  	 *         Vista: 6.0 Windows Server 2003 R2: 5.2 Windows Home Server: 5.2
55  	 *         Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
56  	 *         Windows XP: 5.1 Windows 2000: 5.0
57  	 */
58  	public int getMajor() {
59  		return this._versionInfo.dwMajorVersion.intValue();
60  	}
61  
62  	/**
63  	 * The minor version number of the operating system.
64  	 * 
65  	 * @return The minor version within the following supported operating
66  	 *         systems. Windows 8: 6.2 Windows Server 2012: 6.2 Windows 7: 6.1
67  	 *         Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows
68  	 *         Vista: 6.0 Windows Server 2003 R2: 5.2 Windows Home Server: 5.2
69  	 *         Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
70  	 *         Windows XP: 5.1 Windows 2000: 5.0
71  	 */
72  	public int getMinor() {
73  		return this._versionInfo.dwMinorVersion.intValue();
74  	}
75  
76  	/**
77  	 * The build number of the operating system.
78  	 * 
79  	 * @return Build number.
80  	 */
81  	public int getBuildNumber() {
82  		return this._versionInfo.dwBuildNumber.intValue();
83  	}
84  
85  	/**
86  	 * The operating system platform. This member can be VER_PLATFORM_WIN32_NT.
87  	 * 
88  	 * @return Platform ID.
89  	 */
90  	public int getPlatformId() {
91  		return this._versionInfo.dwPlatformId.intValue();
92  	}
93  
94  	/**
95  	 * String, such as "Service Pack 3", that indicates the latest Service Pack
96  	 * installed on the system. If no Service Pack has been installed, the
97  	 * string is empty.
98  	 * 
99  	 * @return Service pack.
100 	 */
101 	public String getServicePack() {
102 		return Native.toString(this._versionInfo.szCSDVersion);
103 	}
104 
105 	/**
106 	 * A bit mask that identifies the product suites available on the system.
107 	 * 
108 	 * @return Suite mask.
109 	 */
110 	public int getSuiteMask() {
111 		return this._versionInfo.wSuiteMask.intValue();
112 	}
113 
114 	/**
115 	 * Any additional information about the system.
116 	 * 
117 	 * @return Product type.
118 	 */
119 	public byte getProductType() {
120 		return this._versionInfo.wProductType;
121 	}
122 
123 	@Override
124 	public String toString() {
125 		String version = null;
126 
127         // see http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
128         if (getPlatformId() == WinNT.VER_PLATFORM_WIN32_NT) {
129             boolean ntWorkstation = getProductType() == WinNT.VER_NT_WORKSTATION;
130             if (getMajor() == 10 && getMinor() == 0 && ntWorkstation) {
131                 version = "10";
132             } else if (getMajor() == 10 && getMinor() == 0 && !ntWorkstation) {
133                 version = "Server 2016";
134             } else if (getMajor() == 6 && getMinor() == 3 && ntWorkstation) {
135                 version = "8.1";
136             } else if (getMajor() == 6 && getMinor() == 3 && !ntWorkstation) {
137                 version = "Server 2012 R2";
138             } else if (getMajor() == 6 && getMinor() == 2 && ntWorkstation) {
139                 version = "8";
140             } else if (getMajor() == 6 && getMinor() == 2 && !ntWorkstation) {
141                 version = "Server 2012";
142             } else if (getMajor() == 6 && getMinor() == 1 && ntWorkstation) {
143                 version = "7";
144             } else if (getMajor() == 6 && getMinor() == 1 && !ntWorkstation) {
145                 version = "Server 2008 R2";
146             } else if (getMajor() == 6 && getMinor() == 0 && !ntWorkstation) {
147                 version = "Server 2008";
148             } else if (getMajor() == 6 && getMinor() == 0 && ntWorkstation) {
149                 version = "Vista";
150             } else if (getMajor() == 5 && getMinor() == 2 && !ntWorkstation) {
151                 version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003" : "Server 2003 R2";
152             } else if (getMajor() == 5 && getMinor() == 2 && ntWorkstation) {
153                 version = "XP"; // 64 bits
154             } else if (getMajor() == 5 && getMinor() == 1) {
155                 version = "XP"; // 32 bits
156             } else if (getMajor() == 5 && getMinor() == 0) {
157                 version = "2000";
158             } else if (getMajor() == 4) {
159                 version = "NT 4";
160                 if ("Service Pack 6".equals(getServicePack())) {
161                     if (Advapi32Util.registryKeyExists(WinReg.HKEY_LOCAL_MACHINE, "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Hotfix\\Q246009")) {
162                         return "NT4 SP6a";
163                     }
164                 }
165 
166 			} else {
167 				throw new RuntimeException("Unsupported Windows NT version: " + this._versionInfo.toString());
168 			}
169 
170 			if (this._versionInfo.wServicePackMajor.intValue() > 0) {
171 				version = version + " SP" + this._versionInfo.wServicePackMajor.intValue();
172 			}
173 
174 		} else if (getPlatformId() == WinNT.VER_PLATFORM_WIN32_WINDOWS) {
175 			if (getMajor() == 4 && getMinor() == 90) {
176 				version = "ME";
177 			} else if (getMajor() == 4 && getMinor() == 10) {
178 				if (this._versionInfo.szCSDVersion[1] == 'A') {
179 					version = "98 SE";
180 				} else {
181 					version = "98";
182 				}
183 			} else if (getMajor() == 4 && getMinor() == 0) {
184 				if (this._versionInfo.szCSDVersion[1] == 'C' || this._versionInfo.szCSDVersion[1] == 'B') {
185 					version = "95 OSR2";
186 				} else {
187 					version = "95";
188 				}
189 			} else {
190 				throw new RuntimeException("Unsupported Windows 9x version: " + this._versionInfo.toString());
191 			}
192 		} else {
193 			throw new RuntimeException("Unsupported Windows platform: " + this._versionInfo.toString());
194 		}
195 
196 		return version;
197 	}
198 
199 	public OSVersionInfoEx(OSVERSIONINFOEX versionInfo) {
200 		this._versionInfo = versionInfo;
201 	}
202 }
