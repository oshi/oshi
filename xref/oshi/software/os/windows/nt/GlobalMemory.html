17  package oshi.software.os.windows.nt;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import com.sun.jna.platform.win32.Kernel32;
23  import com.sun.jna.platform.win32.WinBase.MEMORYSTATUSEX;
24  
25  import oshi.hardware.Memory;
26  
27  /**
28   * Memory obtained by GlobalMemoryStatusEx.
29   * 
30   * @author dblock[at]dblock[dot]org
31   */
32  public class GlobalMemory implements Memory {
33  	private static final Logger LOG = LoggerFactory.getLogger(GlobalMemory.class);
34  
35  	private MEMORYSTATUSEX _memory = new MEMORYSTATUSEX();
36  
37  	public GlobalMemory() {
38  		if (!Kernel32.INSTANCE.GlobalMemoryStatusEx(this._memory)) {
39  			LOG.error("Failed to Initialize MemoryStatusEx. Error code: {}", Kernel32.INSTANCE.GetLastError());
40  			this._memory = null;
41  		}
42  	}
43  
44  	@Override
45  	public long getAvailable() {
46  		if (this._memory == null) {
47  			LOG.warn("MemoryStatusEx not initialized. No available memoroy data available");
48  			return 0L;
49  		}
50  		return this._memory.ullAvailPhys.longValue();
51  	}
52  
53  	@Override
54  	public long getTotal() {
55  		if (this._memory == null) {
56  			LOG.warn("MemoryStatusEx not initialized. No total memory data available");
57  			return 0L;
58  		}
59  		return this._memory.ullTotalPhys.longValue();
60  	}
61  }
