100     @Override
101     public List<OSFileStore> getFileStores(boolean localOnly) {
102         // Create list to hold results
103         ArrayList<OSFileStore> result;
104 
105         // Begin with all the local volumes
106         result = getLocalVolumes(null);
107 
108         // Build a map of existing mount point to OSFileStore
109         Map<String, OSFileStore> volumeMap = new HashMap<>();
110         for (OSFileStore volume : result) {
111             volumeMap.put(volume.getMount(), volume);
112         }
113 
114         // Iterate through volumes in WMI and update missing fields (if it exists)
115         // or add new if it doesn't (expected for network drives)
116         for (OSFileStore wmiVolume : getWmiVolumes(null, localOnly)) {
117             if (volumeMap.containsKey(wmiVolume.getMount())) {
118                 // If the volume is already in our list, preserve local values
119                 // and only backfill missing fields from WMI
120                 OSFileStore volume = volumeMap.get(wmiVolume.getMount());
121                 int idx = result.indexOf(volume);
122                 WindowsOSFileStoreFFM updated = new WindowsOSFileStoreFFM(volume.getName(), // preserve local name
123                         volume.getVolume(), volume.getLabel().isEmpty() ? wmiVolume.getLabel() : volume.getLabel(),
124                         volume.getMount(), volume.getOptions(), volume.getUUID(), "",
125                         volume.getDescription().isEmpty() ? wmiVolume.getDescription() : volume.getDescription(),
126                         volume.getType(), volume.getFreeSpace(), volume.getUsableSpace(), volume.getTotalSpace(), 0, 0,
127                         true);
128                 if (idx >= 0) {
129                     result.set(idx, updated);
130                 } else {
131                     result.add(updated);
132                 }
133             } else if (!localOnly) {
134                 // Otherwise add the new volume in its entirety
135                 result.add(wmiVolume);
136             }
137         }
138         return result;
139     }

141     /**
142      * method for getting all mounted local drives.
143      *
144      * @param volumeToMatch an optional string to filter match, null otherwise
145      * @return A list of {@link OSFileStore} objects representing all local mounted volumes
146      */
147     static ArrayList<OSFileStore> getLocalVolumes(String volumeToMatch) {
148         ArrayList<OSFileStore> fs = new ArrayList<>();
149 
150         try (Arena arena = Arena.ofConfined()) {
151             MemorySegment volumeNameBuf = arena.allocate(BUFSIZE * JAVA_CHAR.byteSize());
152             Optional<MemorySegment> hVolOpt = Kernel32FFM.FindFirstVolume(volumeNameBuf, BUFSIZE);
153             if (hVolOpt.isEmpty()) {
154                 return fs;
155             }
156             MemorySegment hVol = hVolOpt.get();
157             try {
158                 do {
159                     String volume = readWideString(volumeNameBuf);
160                     if (volumeToMatch != null && !volume.equals(volumeToMatch)) {
161                         continue;
162                     }
163                     MemorySegment fstypeBuf = arena.allocate(16 * JAVA_CHAR.byteSize());
164                     MemorySegment nameBuf = arena.allocate(BUFSIZE * JAVA_CHAR.byteSize());
165                     MemorySegment mountBuf = arena.allocate(BUFSIZE * JAVA_CHAR.byteSize());
166                     MemorySegment flagsBuf = arena.allocate(JAVA_INT);
167                     MemorySegment userFreeBytesBuf = arena.allocate(JAVA_LONG);
168                     MemorySegment totalBytesBuf = arena.allocate(JAVA_LONG);
169                     MemorySegment systemFreeBytesBuf = arena.allocate(JAVA_LONG);
170                     MemorySegment returnLengthBuf = arena.allocate(JAVA_INT);
171 
172                     // Get volume information - skip if call fails
173                     var volInfoResult = Kernel32FFM.GetVolumeInformation(toWideString(arena, volume), nameBuf, BUFSIZE,
174                             NULL, NULL, flagsBuf, fstypeBuf, 16);
175                     if (volInfoResult.isEmpty() || volInfoResult.getAsInt() == 0) {
176                         continue;
177                     }
178                     int flags = flagsBuf.get(JAVA_INT, 0);
179 
180                     // Get volume path names; retry with larger buffer if ERROR_MORE_DATA
181                     int mountBufSize = BUFSIZE;
182                     var pathNamesResult = Kernel32FFM.GetVolumePathNamesForVolumeName(toWideString(arena, volume),
183                             mountBuf, mountBufSize, returnLengthBuf);
184                     if (pathNamesResult.isEmpty() || pathNamesResult.getAsInt() == 0) {
185                         if (Kernel32FFM.GetLastError().orElse(0) == 0x7A) { // ERROR_MORE_DATA
186                             mountBufSize = returnLengthBuf.get(JAVA_INT, 0);
187                             mountBuf = arena.allocate((long) mountBufSize * JAVA_CHAR.byteSize());
188                             pathNamesResult = Kernel32FFM.GetVolumePathNamesForVolumeName(toWideString(arena, volume),
189                                     mountBuf, mountBufSize, returnLengthBuf);
190                         }
191                     }
192                     if (pathNamesResult.isEmpty() || pathNamesResult.getAsInt() == 0) {
193                         continue;
194                     }
195                     String mount = readWideString(mountBuf);
196 
197                     if (!mount.isEmpty()) {
198                         String name = readWideString(nameBuf);
199                         String fsType = readWideString(fstypeBuf);
200                         StringBuilder options = new StringBuilder((FILE_READ_ONLY_VOLUME & flags) == 0 ? "rw" : "ro");
201                         String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
202                                 .map(Map.Entry::getValue).collect(Collectors.joining(","));
203                         if (!moreOptions.isEmpty()) {
204                             options.append(',').append(moreOptions);
205                         }
206 
207                         // Get disk free space - skip if call fails
208                         var diskSpaceResult = Kernel32FFM.GetDiskFreeSpaceEx(toWideString(arena, volume),
209                                 userFreeBytesBuf, totalBytesBuf, systemFreeBytesBuf);
210                         if (diskSpaceResult.isEmpty() || diskSpaceResult.getAsInt() == 0) {
211                             continue;
212                         }
213                         long systemFreeBytes = systemFreeBytesBuf.get(JAVA_LONG, 0);
214                         long totalBytes = totalBytesBuf.get(JAVA_LONG, 0);
215                         long userFreeBytes = userFreeBytesBuf.get(JAVA_LONG, 0);
216 
217                         String uuid = ParseUtil.parseUuidOrDefault(volume, "");
218 
219                         fs.add(new WindowsOSFileStoreFFM(String.format(Locale.ROOT, "%s (%s)", name, mount), volume,
220                                 name, mount, options.toString(), uuid, "", getDriveType(mountBuf), fsType,
221                                 systemFreeBytes, userFreeBytes, totalBytes, 0, 0, true));
222                     }
223                 } while (Kernel32FFM.FindNextVolume(hVol, volumeNameBuf, BUFSIZE).orElse(0) != 0);
224                 return fs;
225             } finally {
226                 Kernel32FFM.FindVolumeClose(hVol);
227             }
228         }
229     }

231     /**
232      * Private method for getting mounted drive type.
233      *
234      * @param mountBuf memory segment containing the mounted drive path
235      * @return A drive type description
236      */
237     private static String getDriveType(MemorySegment mountBuf) {
238         int type = Kernel32FFM.GetDriveType(mountBuf).orElse(-1);
239         return getDriveTypeString(type);
240     }
241 
242     /**
243      * Converts a drive type integer to a description string.
244      *
245      * @param type the drive type from GetDriveType
246      * @return A drive type description
247      */
248     private static String getDriveTypeString(int type) {
249         return switch (type) {
250             case DRIVE_REMOVABLE -> "Removable drive";
251             case DRIVE_FIXED -> "Fixed drive";
252             case DRIVE_REMOTE -> "Network drive";
253             case DRIVE_CDROM -> "CD-ROM";
254             case DRIVE_RAMDISK -> "RAM drive";
255             default -> "Unknown drive type";
256         };
257     }

259     /**
260      * Gets logical drives listed in WMI.
261      *
262      * @param nameToMatch an optional string to filter match, null otherwise
263      * @param localOnly   whether to only search local drives
264      * @return A list of {@link OSFileStore} objects representing all WMI volumes
265      */
266     static List<OSFileStore> getWmiVolumes(String nameToMatch, boolean localOnly) {
267         List<OSFileStore> fs = new ArrayList<>();
268         List<Win32LogicalDiskFFM.LogicalDiskInfo> drives = Win32LogicalDiskFFM.queryLogicalDisk(nameToMatch, localOnly);
269 
270         try (Arena arena = Arena.ofConfined()) {
271             for (Win32LogicalDiskFFM.LogicalDiskInfo drive : drives) {
272                 long free = drive.getFreeSpace();
273                 long total = drive.getSize();
274                 String description = drive.getDescription();
275                 String name = drive.getName();
276                 String label = drive.getVolumeName();
277                 String options = drive.getAccess() == 1 ? "ro" : "rw";
278                 int type = drive.getDriveType();
279                 String volume;
280 
281                 if (type != 4) {
282                     // Local drive - get volume GUID path
283                     MemorySegment mountPoint = toWideString(arena, name + "\\");
284                     MemorySegment volumeBuf = arena.allocate(BUFSIZE * JAVA_CHAR.byteSize());
285                     var volResult = Kernel32FFM.GetVolumeNameForVolumeMountPoint(mountPoint, volumeBuf, BUFSIZE);
286                     volume = (volResult.isPresent() && volResult.getAsInt() != 0) ? readWideString(volumeBuf) : "";
287                 } else {
288                     // Network drive
289                     volume = drive.getProviderName();
290                     String[] split = volume.split("\\\\");
291                     if (split.length > 1 && !split[split.length - 1].isEmpty()) {
292                         description = split[split.length - 1];
293                     }
294                 }
295 
296                 fs.add(new WindowsOSFileStoreFFM(String.format(Locale.ROOT, "%s (%s)", description, name), volume,
297                         label, name + "\\", options, "", "", getDriveTypeString(type), drive.getFileSystem(), free,
298                         free, total, 0, 0, type != 4));
299             }
300         }
301         return fs;
302     }
