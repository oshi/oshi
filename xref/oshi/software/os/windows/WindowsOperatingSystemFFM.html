47      private static boolean enableDebugPrivilege() {
48  
49          MemorySegment hToken = null;
50          try (Arena arena = Arena.ofConfined()) {
51              MemorySegment hTokenPtr = arena.allocate(ADDRESS);
52  
53              Optional<MemorySegment> hProcess = Kernel32FFM.GetCurrentProcess();
54              if (hProcess.isEmpty()) {
55                  return false;
56              }
57              boolean success = Advapi32FFM.OpenProcessToken(hProcess.get(),
58                      WinNTFFM.TOKEN_QUERY | WinNTFFM.TOKEN_ADJUST_PRIVILEGES, hTokenPtr);
59              if (!success) {
60                  LOG.error("OpenProcessToken failed, error: " + GetLastError());
61                  return false;
62              }
63              hToken = hTokenPtr.get(ADDRESS, 0);
64  
65              MemorySegment luid = arena.allocate(WinNTFFM.LUID);
66              success = Advapi32FFM.LookupPrivilegeValue("SeDebugPrivilege", luid, arena);
67              if (!success) {
68                  LOG.error("LookupPrivilegeValue failed, error: " + GetLastError());
69                  return false;
70              }
71  
72              MemorySegment tkp = setupTokenPrivileges(arena, luid);
73              success = Advapi32FFM.AdjustTokenPrivileges(hToken, tkp);
74              if (!success) {
75                  LOG.error("AdjustTokenPrivileges failed, error: " + GetLastError());
76                  return false;
77              }
78  
79              return true;
80          } catch (Throwable t) {
81              LOG.error("enableDebugPrivilege exception: " + t);
82              return false;
83          } finally {
84              if (hToken != null && hToken.address() != 0) {
85                  Kernel32FFM.CloseHandle(hToken);
86              }
87          }
88      }

125     public int getThreadCount() {
126 
127         try (Arena arena = Arena.ofConfined()) {
128             MemorySegment perfInfo = arena.allocate(PERFORMANCE_INFORMATION);
129             int size = (int) PERFORMANCE_INFORMATION.byteSize();
130             perfInfo.set(JAVA_INT, PERFORMANCE_INFORMATION.byteOffset(MemoryLayout.PathElement.groupElement("cb")),
131                     size);
132             if (!PsapiFFM.GetPerformanceInfo(perfInfo, size)) {
133                 LOG.error("Failed to get Performance Info. Error code: {}", GetLastError());
134                 return 0;
135             }
136 
137             int threadCount = perfInfo.get(JAVA_INT,
138                     PERFORMANCE_INFORMATION.byteOffset(MemoryLayout.PathElement.groupElement("ThreadCount")));
139             return threadCount;
140         } catch (Throwable t) {
141             LOG.error("Exception getting thread count", t);
142             return 0;
143         }
144     }
