88      private static boolean enableDebugPrivilege() {
89          try (Arena arena = Arena.ofConfined()) {
90              MemorySegment hTokenPtr = arena.allocate(ADDRESS);
91  
92              Optional<MemorySegment> hProcess = Kernel32FFM.GetCurrentProcess();
93              if (hProcess.isEmpty()) {
94                  return false;
95              }
96              boolean success = Advapi32FFM.OpenProcessToken(hProcess.get(),
97                      WinNTFFM.TOKEN_QUERY | WinNTFFM.TOKEN_ADJUST_PRIVILEGES, hTokenPtr);
98              if (!success) {
99                  LOG.error("OpenProcessToken failed, error: {}", GetLastError());
100                 return false;
101             }
102             try (NativeHandle hToken = NativeHandle.of(hTokenPtr.get(ADDRESS, 0), Kernel32FFM::CloseHandle)) {
103                 MemorySegment luid = arena.allocate(WinNTFFM.LUID);
104                 success = Advapi32FFM.LookupPrivilegeValue("SeDebugPrivilege", luid, arena);
105                 if (!success) {
106                     LOG.error("LookupPrivilegeValue failed, error: {}", GetLastError());
107                     return false;
108                 }
109 
110                 MemorySegment tkp = setupTokenPrivileges(arena, luid);
111                 success = Advapi32FFM.AdjustTokenPrivileges(hToken.get(), tkp);
112                 if (!success) {
113                     LOG.error("AdjustTokenPrivileges failed, error: {}", GetLastError());
114                     return false;
115                 }
116 
117                 return true;
118             }
119         } catch (Throwable t) {
120             LOG.error("enableDebugPrivilege exception: {}", t.getMessage());
121             return false;
122         }
123     }

138     private static final int SC_MANAGER_ENUMERATE_SERVICE = 0x0004;
139     private static final int SERVICE_WIN32 = 0x30;
140     private static final int SERVICE_STATE_ALL = 3;
141     // ENUM_SERVICE_STATUS_PROCESSW on 64-bit: 8+8+36=52, padded to 56
142     private static final long ENUM_SERVICE_STATUS_PROCESS_SIZE = 56;
143     private static final long DISPLAY_NAME_OFFSET = 8;
144     private static final long CURRENT_STATE_OFFSET = 20;
145     private static final long PROCESS_ID_OFFSET = 44;
146 
147     private static List<OSService> queryServicesFFM() {
148         try (Arena arena = Arena.ofConfined()) {
149             MemorySegment hSCManager = Advapi32FFM.OpenSCManager(MemorySegment.NULL, MemorySegment.NULL,
150                     SC_MANAGER_ENUMERATE_SERVICE);
151             if (hSCManager == null || hSCManager.address() == 0) {
152                 LOG.error("Failed to open Service Control Manager");
153                 return Collections.emptyList();
154             }
155             try (NativeHandle ignored = NativeHandle.of(hSCManager, Advapi32FFM::CloseServiceHandle)) {
156                 // First call to get required buffer size
157                 MemorySegment pcbBytesNeeded = arena.allocate(JAVA_INT);
158                 MemorySegment lpServicesReturned = arena.allocate(JAVA_INT);
159                 MemorySegment lpResumeHandle = arena.allocate(JAVA_INT);
160                 lpResumeHandle.set(JAVA_INT, 0, 0);
161 
162                 Advapi32FFM.EnumServicesStatusEx(hSCManager, 0, SERVICE_WIN32, SERVICE_STATE_ALL, MemorySegment.NULL, 0,
163                         pcbBytesNeeded, lpServicesReturned, lpResumeHandle, MemorySegment.NULL);
164 
165                 int bytesNeeded = pcbBytesNeeded.get(JAVA_INT, 0);
166                 if (bytesNeeded == 0) {
167                     return Collections.emptyList();
168                 }
169 
170                 MemorySegment lpServices = arena.allocate(bytesNeeded);
171                 lpResumeHandle.set(JAVA_INT, 0, 0);
172                 if (!Advapi32FFM.EnumServicesStatusEx(hSCManager, 0, SERVICE_WIN32, SERVICE_STATE_ALL, lpServices,
173                         bytesNeeded, pcbBytesNeeded, lpServicesReturned, lpResumeHandle, MemorySegment.NULL)) {
174                     LOG.error("EnumServicesStatusEx failed, error: {}", Kernel32FFM.GetLastError());
175                     return Collections.emptyList();
176                 }
177 
178                 int count = lpServicesReturned.get(JAVA_INT, 0);
179                 List<OSService> svcArray = new ArrayList<>(count);
180 
181                 for (int i = 0; i < count; i++) {
182                     long base = (long) i * ENUM_SERVICE_STATUS_PROCESS_SIZE;
183                     MemorySegment pDisplayName = lpServices.get(ADDRESS, base + DISPLAY_NAME_OFFSET).reinterpret(512);
184                     String displayName = readWideString(pDisplayName);
185                     int currentState = lpServices.get(JAVA_INT, base + CURRENT_STATE_OFFSET);
186                     int processId = lpServices.get(JAVA_INT, base + PROCESS_ID_OFFSET);
187 
188                     State state;
189                     switch (currentState) {
190                         case 1:
191                             state = State.STOPPED;
192                             break;
193                         case 4:
194                             state = State.RUNNING;
195                             break;
196                         default:
197                             state = State.OTHER;
198                             break;
199                     }
200                     svcArray.add(new OSService(displayName, processId, state));
201                 }
202                 return svcArray;
203             }
204         } catch (Throwable t) {
205             LOG.error("Error enumerating services: {}", t.getMessage());
206             return Collections.emptyList();
207         }
208     }

260     private int getPerformanceInfoField(String fieldName) {
261         return callInArenaIntOrDefault(arena -> {
262             MemorySegment perfInfo = arena.allocate(PERFORMANCE_INFORMATION);
263             int size = (int) PERFORMANCE_INFORMATION.byteSize();
264             perfInfo.set(JAVA_INT, PERFORMANCE_INFORMATION.byteOffset(MemoryLayout.PathElement.groupElement("cb")),
265                     size);
266             if (!PsapiFFM.GetPerformanceInfo(perfInfo, size)) {
267                 LOG.error("Failed to get Performance Info. Error code: {}", GetLastError());
268                 return 0;
269             }
270             return perfInfo.get(JAVA_INT,
271                     PERFORMANCE_INFORMATION.byteOffset(MemoryLayout.PathElement.groupElement(fieldName)));
272         }, LOG, ERROR, "Exception getting " + fieldName, 0);
273     }

280     @Override
281     public OSThread getCurrentThread() {
282         final int tid = getThreadId();
283         OSProcess proc = getCurrentProcess();
284         if (proc == null) {
285             return new WindowsOSThreadFFM(0, tid, null, null);
286         }
287         return proc.getThreadDetails().stream().filter(t -> t.getThreadId() == tid).findFirst()
288                 .orElseGet(() -> new WindowsOSThreadFFM(proc.getProcessID(), tid, null, null));
289     }

319     private List<OSProcess> processMapToList(Collection<Integer> pids) {
320         // Get data from the registry if possible
321         Map<Integer, ProcessPerfCounterBlock> processMap = processMapFromRegistry.get();
322         // otherwise performance counters with WMI backup
323         if (processMap == null || processMap.isEmpty()) {
324             processMap = (pids == null) ? processMapFromPerfCounters.get()
325                     : ProcessPerformanceDataFFM.buildProcessMapFromPerfCounters(pids);
326         }
327         Map<Integer, ThreadPerfCounterBlock> threadMap = null;
328         if (USE_PROCSTATE_SUSPENDED) {
329             // Get data from the registry if possible
330             threadMap = threadMapFromRegistry.get();
331             // otherwise performance counters with WMI backup
332             if (threadMap == null || threadMap.isEmpty()) {
333                 threadMap = (pids == null) ? threadMapFromPerfCounters.get()
334                         : ThreadPerformanceDataFFM.buildThreadMapFromPerfCounters(pids);
335             }
336         }
337 
338         Map<Integer, WtsInfo> processWtsMap = ProcessWtsDataFFM.queryProcessWtsMap(pids);
339 
340         // Determine which PIDs to process: requested pids filtered by available data, or all available
341         Set<Integer> mapKeys;
342         if (pids == null) {
343             // All processes: use intersection of WTS and processMap (WTS enrichment required for full data)
344             mapKeys = new HashSet<>(processWtsMap.keySet());
345             mapKeys.retainAll(processMap.keySet());
346         } else {
347             // Specific PIDs: filter by what's available in processMap
348             mapKeys = new HashSet<>(pids);
349             mapKeys.retainAll(processMap.keySet());
350         }
351 
352         final Map<Integer, ProcessPerfCounterBlock> finalProcessMap = processMap;
353         final Map<Integer, ThreadPerfCounterBlock> finalThreadMap = threadMap;
354         return mapKeys.stream().parallel()
355                 .map(pid -> new WindowsOSProcessFFM(pid, this, finalProcessMap, processWtsMap, finalThreadMap))
356                 .filter(VALID_PROCESS).collect(Collectors.toList());
357     }

375     @Override
376     protected Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
377         String version = System.getProperty("os.name");
378         if (version.startsWith("Windows ")) {
379             version = version.substring(8);
380         }
381         String sp = null;
382         int suiteMask = 0;
383         String buildNumber = "";
384         WmiResult<OSVersionProperty> versionInfo = Win32OperatingSystemFFM.queryOsVersion();
385         if (versionInfo.getResultCount() > 0) {
386             sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
387             if (!sp.isEmpty() && !"unknown".equals(sp)) {
388                 version = version + " " + sp.replace("Service Pack ", "SP");
389             }
390             suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
391             buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
392         }
393         String codeName = parseCodeName(suiteMask);
394         if ("10".equals(version) && buildNumber.compareTo("22000") >= 0) {
395             version = "11";
396         }
397         if ("Server 2016".equals(version) && buildNumber.compareTo("17762") > 0) {
398             version = "Server 2019";
399         }
400         if ("Server 2019".equals(version) && buildNumber.compareTo("20347") > 0) {
401             version = "Server 2022";
402         }
403         if ("Server 2022".equals(version) && buildNumber.compareTo("26039") > 0) {
404             version = "Server 2025";
405         }
406         return new Pair<>("Windows", new OSVersionInfo(version, codeName, buildNumber));
407     }

409     @Override
410     protected int queryBitness(int jvmBitness) {
411         if (jvmBitness < 64 && System.getenv("ProgramFiles(x86)") != null) {
412             WmiResult<BitnessProperty> bitnessMap = Win32ProcessorFFM.queryBitness();
413             if (bitnessMap.getResultCount() > 0) {
414                 return WmiUtil.getUint16(bitnessMap, BitnessProperty.ADDRESSWIDTH, 0);
415             }
416         }
417         return jvmBitness;
418     }

434     private Map<Integer, Integer> getParentPidMap() {
435         Map<Integer, Integer> parentPidMap = new HashMap<>();
436         Map<Integer, ProcessPerfCounterBlock> processMap = processMapFromRegistry.get();
437         if (processMap == null || processMap.isEmpty()) {
438             processMap = processMapFromPerfCounters.get();
439         }
440         if (processMap != null) {
441             for (Map.Entry<Integer, ProcessPerfCounterBlock> entry : processMap.entrySet()) {
442                 parentPidMap.put(entry.getKey(), entry.getValue().getParentProcessID());
443             }
444         }
445         return parentPidMap;
446     }
