81      @Override
82      public long getAffinityMask() {
83          Optional<MemorySegment> pHandleOpt = Kernel32FFM.OpenProcess(PROCESS_QUERY_INFORMATION, false, getProcessID());
84          if (pHandleOpt.isPresent()) {
85              MemorySegment pHandle = pHandleOpt.get();
86              try (Arena arena = Arena.ofConfined()) {
87                  MemorySegment processAffinity = arena.allocate(JAVA_LONG);
88                  MemorySegment systemAffinity = arena.allocate(JAVA_LONG);
89                  if (Kernel32FFM.GetProcessAffinityMask(pHandle, processAffinity, systemAffinity)) {
90                      return processAffinity.get(JAVA_LONG, 0);
91                  }
92              } finally {
93                  Kernel32FFM.CloseHandle(pHandle);
94              }
95          }
96          return 0L;
97      }

99      @Override
100     public boolean updateAttributes() {
101         Set<Integer> pids = Collections.singleton(this.getProcessID());
102         // Get data from the registry if possible
103         Map<Integer, ProcessPerfCounterBlock> pcbMap = ProcessPerformanceDataFFM.buildProcessMapFromRegistry(pids);
104         // otherwise performance counters with WMI backup
105         if (pcbMap == null || pcbMap.isEmpty()) {
106             pcbMap = ProcessPerformanceDataFFM.buildProcessMapFromPerfCounters(pids);
107         }
108         ProcessPerfCounterBlock pcb = pcbMap == null ? null : pcbMap.get(this.getProcessID());
109         if (USE_PROCSTATE_SUSPENDED) {
110             // Populate name from pcb before querying threads, since the fallback path uses getName()
111             if (pcb != null) {
112                 setName(pcb.getName());
113             }
114             setTcb(queryMatchingThreads(pids));
115         }
116         Map<Integer, WtsInfo> wts = ProcessWtsDataFFM.queryProcessWtsMap(pids);
117         return updateAttributes(pcb, wts == null ? null : wts.get(this.getProcessID()));
118     }

120     @Override
121     protected boolean updateAttributes(ProcessPerfCounterBlock pcb, WtsInfo wts) {
122         if (!super.updateAttributes(pcb, wts)) {
123             return false;
124         }
125 
126         Optional<MemorySegment> pHandleOpt = Kernel32FFM.OpenProcess(PROCESS_QUERY_INFORMATION, false, getProcessID());
127         if (pHandleOpt.isPresent()) {
128             MemorySegment pHandle = pHandleOpt.get();
129             try (Arena arena = Arena.ofConfined()) {
130                 // Test for 32-bit process on 64-bit windows
131                 if (IS_VISTA_OR_GREATER && getBitness() == 64) {
132                     MemorySegment wow64 = arena.allocate(JAVA_INT);
133                     if (Kernel32FFM.IsWow64Process(pHandle, wow64) && wow64.get(JAVA_INT, 0) > 0) {
134                         setBitness(32);
135                     }
136                 }
137                 // EXECUTABLEPATH
138                 if (IS_WINDOWS7_OR_GREATER) {
139                     Optional<String> pathOpt = Kernel32FFM.QueryFullProcessImageName(pHandle, 0, arena);
140                     if (pathOpt.isPresent()) {
141                         setPath(pathOpt.get());
142                     }
143                 }
144             } finally {
145                 Kernel32FFM.CloseHandle(pHandle);
146             }
147         }
148 
149         return !getState().equals(State.INVALID);
150     }

157     @Override
158     protected Map<Integer, ThreadPerfCounterBlock> queryMatchingThreads(Set<Integer> pids) {
159         Map<Integer, ThreadPerfCounterBlock> threads = ThreadPerformanceDataFFM.buildThreadMapFromRegistry(pids);
160         if (threads == null || threads.isEmpty()) {
161             threads = ThreadPerformanceDataFFM.buildThreadMapFromPerfCounters(pids, this.getName(), -1);
162         }
163         return threads;
164     }

166     @Override
167     protected String queryCommandLine() {
168         String cwdCmdEnvB = getCwdCmdEnv().getB();
169         if (!cwdCmdEnvB.isEmpty()) {
170             return cwdCmdEnvB;
171         }
172         if (USE_BATCH_COMMANDLINE) {
173             return Win32ProcessCachedFFM.getInstance().getCommandLine(getProcessID(), getStartTime());
174         }
175         var commandLineProcs = Win32ProcessFFM.queryCommandLines(Collections.singleton(getProcessID()));
176         if (commandLineProcs.getResultCount() > 0) {
177             return WmiUtilFFM.getString(commandLineProcs, CommandLineProperty.COMMANDLINE, 0);
178         }
179         return "";
180     }

191     @Override
192     protected Pair<String, String> queryUserInfo() {
193         Pair<String, String> pair = null;
194         Optional<MemorySegment> pHandleOpt = Kernel32FFM.OpenProcess(PROCESS_QUERY_INFORMATION, false, getProcessID());
195         if (pHandleOpt.isPresent()) {
196             MemorySegment pHandle = pHandleOpt.get();
197             MemorySegment hToken = null;
198             try (Arena arena = Arena.ofConfined()) {
199                 MemorySegment hTokenPtr = arena.allocate(ADDRESS);
200                 if (Advapi32FFM.OpenProcessToken(pHandle, TOKEN_QUERY, hTokenPtr)) {
201                     hToken = hTokenPtr.get(ADDRESS, 0);
202                     pair = getTokenAccountInfo(hToken, TokenUser, arena);
203                 } else {
204                     int error = Kernel32FFM.GetLastError().orElse(0);
205                     if (error != WinErrorFFM.ERROR_ACCESS_DENIED) {
206                         LOG.error("Failed to get process token for process {}: {}", getProcessID(), error);
207                     }
208                 }
209             } catch (Throwable e) {
210                 LOG.warn("Failed to query user info for process {} ({}): {}", getProcessID(), getName(),
211                         e.getMessage());
212             } finally {
213                 if (hToken != null && hToken.address() != 0) {
214                     Kernel32FFM.CloseHandle(hToken);
215                 }
216                 Kernel32FFM.CloseHandle(pHandle);
217             }
218         }
219         return pair != null ? pair : defaultPair();
220     }

253     private Pair<String, String> getTokenAccountInfo(MemorySegment hToken, int tokenInfoClass, Arena arena)
254             throws Throwable {
255         MemorySegment returnLength = arena.allocate(JAVA_INT);
256         Advapi32FFM.GetTokenInformation(hToken, tokenInfoClass, MemorySegment.NULL, 0, returnLength);
257 
258         int size = returnLength.get(JAVA_INT, 0);
259         if (size <= 0) {
260             return null;
261         }
262 
263         MemorySegment tokenInfo = arena.allocate(size);
264         if (!Advapi32FFM.GetTokenInformation(hToken, tokenInfoClass, tokenInfo, size, returnLength)) {
265             return null;
266         }
267 
268         MemorySegment sidPtr = tokenInfo.get(ADDRESS, 0);
269         if (sidPtr.address() == 0) {
270             return null;
271         }
272 
273         int maxNameLen = 256;
274         MemorySegment nameBuffer = arena.allocate(JAVA_CHAR, maxNameLen);
275         MemorySegment nameLen = arena.allocate(JAVA_INT);
276         nameLen.set(JAVA_INT, 0, maxNameLen);
277 
278         MemorySegment domainBuffer = arena.allocate(JAVA_CHAR, maxNameLen);
279         MemorySegment domainLen = arena.allocate(JAVA_INT);
280         domainLen.set(JAVA_INT, 0, maxNameLen);
281 
282         MemorySegment sidUse = arena.allocate(JAVA_INT);
283 
284         String accountName = Constants.UNKNOWN;
285         if (Advapi32FFM.LookupAccountSid(MemorySegment.NULL, sidPtr, nameBuffer, nameLen, domainBuffer, domainLen,
286                 sidUse)) {
287             accountName = readWideString(nameBuffer);
288         }
289 
290         String sidString = Constants.UNKNOWN;
291         MemorySegment sidStringPtr = arena.allocate(ADDRESS);
292         if (Advapi32FFM.ConvertSidToStringSid(sidPtr, sidStringPtr)) {
293             MemorySegment strPtr = sidStringPtr.get(ADDRESS, 0);
294             if (strPtr.address() != 0) {
295                 sidString = readWideString(strPtr.reinterpret(512));
296                 Kernel32FFM.LocalFree(strPtr);
297             }
298         }
299 
300         return new Pair<>(accountName, sidString);
301     }

303     @Override
304     protected Triplet<String, String, Map<String, String>> queryCwdCommandlineEnvironment() {
305         Optional<MemorySegment> hOpt = Kernel32FFM.OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, false,
306                 getProcessID());
307         if (hOpt.isPresent()) {
308             MemorySegment h = hOpt.get();
309             try (Arena arena = Arena.ofConfined()) {
310                 if (WindowsOperatingSystemFFM.isX86() == isWow(h, arena)) {
311                     MemorySegment pbi = arena.allocate(PROCESS_BASIC_INFORMATION_STRUCT);
312                     MemorySegment nRead = arena.allocate(JAVA_INT);
313 
314                     int ret = NtDllFFM.NtQueryInformationProcess(h, PROCESS_BASIC_INFORMATION, pbi,
315                             (int) PROCESS_BASIC_INFORMATION_STRUCT.byteSize(), nRead);
316                     if (ret != 0) {
317                         return defaultCwdCommandlineEnvironment();
318                     }
319 
320                     MemorySegment pebAddress = pbi.get(ADDRESS, PBI_PEB_BASE_ADDRESS_OFFSET);
321                     if (pebAddress.address() == 0) {
322                         return defaultCwdCommandlineEnvironment();
323                     }
324 
325                     MemorySegment peb = arena.allocate(PEB);
326                     MemorySegment bytesRead = arena.allocate(JAVA_LONG);
327                     if (!Kernel32FFM.ReadProcessMemory(h, pebAddress, peb, PEB.byteSize(), bytesRead)) {
328                         return defaultCwdCommandlineEnvironment();
329                     }
330                     if (bytesRead.get(JAVA_LONG, 0) == 0) {
331                         return defaultCwdCommandlineEnvironment();
332                     }
333 
334                     MemorySegment processParamsAddress = peb.get(ADDRESS, PEB_PROCESS_PARAMETERS_OFFSET);
335                     if (processParamsAddress.address() == 0) {
336                         return defaultCwdCommandlineEnvironment();
337                     }
338 
339                     MemorySegment upp = arena.allocate(RTL_USER_PROCESS_PARAMETERS);
340                     if (!Kernel32FFM.ReadProcessMemory(h, processParamsAddress, upp,
341                             RTL_USER_PROCESS_PARAMETERS.byteSize(), bytesRead)) {
342                         return defaultCwdCommandlineEnvironment();
343                     }
344                     if (bytesRead.get(JAVA_LONG, 0) == 0) {
345                         return defaultCwdCommandlineEnvironment();
346                     }
347 
348                     MemorySegment cwdUnicodeString = upp.asSlice(UPP_CURRENT_DIRECTORY_OFFSET,
349                             UNICODE_STRING.byteSize());
350                     String cwd = readUnicodeString(h, cwdUnicodeString, arena);
351 
352                     MemorySegment cmdLineUnicodeString = upp.asSlice(UPP_COMMAND_LINE_OFFSET,
353                             UNICODE_STRING.byteSize());
354                     String cl = readUnicodeString(h, cmdLineUnicodeString, arena);
355 
356                     long envSize = upp.get(JAVA_LONG, UPP_ENVIRONMENT_SIZE_OFFSET);
357                     if (envSize > 0 && envSize < Integer.MAX_VALUE) {
358                         MemorySegment envAddress = upp.get(ADDRESS, UPP_ENVIRONMENT_OFFSET);
359                         if (envAddress.address() != 0) {
360                             MemorySegment envBuffer = arena.allocate((int) envSize);
361                             if (Kernel32FFM.ReadProcessMemory(h, envAddress, envBuffer, envSize, bytesRead)) {
362                                 if (bytesRead.get(JAVA_LONG, 0) > 0) {
363                                     char[] env = new char[(int) (envSize / 2)];
364                                     for (int i = 0; i < env.length; i++) {
365                                         env[i] = envBuffer.get(JAVA_CHAR, (long) i * 2);
366                                     }
367                                     Map<String, String> envMap = ParseUtil.parseCharArrayToStringMap(env);
368                                     envMap.remove("");
369                                     return new Triplet<>(cwd, cl, Collections.unmodifiableMap(envMap));
370                                 }
371                             }
372                         }
373                     }
374                     return new Triplet<>(cwd, cl, Collections.emptyMap());
375                 }
376             } finally {
377                 Kernel32FFM.CloseHandle(h);
378             }
379         }
380         return defaultCwdCommandlineEnvironment();
381     }

383     private static boolean isWow(MemorySegment h, Arena arena) {
384         MemorySegment wow64 = arena.allocate(JAVA_INT);
385         if (Kernel32FFM.IsWow64Process(h, wow64)) {
386             return wow64.get(JAVA_INT, 0) != 0;
387         }
388         return false;
