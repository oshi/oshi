133     @Override
134     public FamilyVersionInfo queryFamilyVersionInfo() {
135         WmiResult<OSVersionProperty> versionInfo = Win32OperatingSystem.queryOsVersion();
136         if (versionInfo.getResultCount() < 1) {
137             return new FamilyVersionInfo("Windows", new OSVersionInfo(System.getProperty("os.version"), null, null));
138         }
139         // Guaranteed that versionInfo is not null and lists non-empty
140         // before calling the parse*() methods
141         int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
142         String buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
143         String version = parseVersion(versionInfo, suiteMask, buildNumber);
144         String codeName = parseCodeName(suiteMask);
145         return new FamilyVersionInfo("Windows", new OSVersionInfo(version, codeName, buildNumber));
146     }

148     private String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask, String buildNumber) {
149         // Initialize a default, sane value
150         String version = System.getProperty("os.version");
151 
152         // Version is major.minor.build. Parse the version string for
153         // major/minor and get the build number separately
154         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
155         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
156         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
157 
158         // see
159         // https://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
160         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
161                 0) == WinNT.VER_NT_WORKSTATION;
162 
163         StringBuilder verLookup = new StringBuilder(major).append('.').append(minor);
164 
165         if (IS_VISTA_OR_GREATER && ntWorkstation) {
166             verLookup.append(".nt");
167         } else if (major == 10 && ParseUtil.parseLongOrDefault(buildNumber, 0L) > 17_762) {
168             verLookup.append(".17763+");
169         } else if (major == 5 && minor == 2) {
170             if (ntWorkstation && getBitness() == 64) {
171                 verLookup.append(".nt.x64");
172             } else if ((suiteMask & 0x00008000) != 0) { // VER_SUITE_WH_SERVER
173                 verLookup.append(".HS");
174             } else if (User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0) {
175                 verLookup.append(".R2");
176             }
177         }
178 
179         Properties verProps = FileUtil.readPropertiesFromFilename(WIN_VERSION_PROPERTIES);
180         version = verProps.getProperty(verLookup.toString()) != null ? verProps.getProperty(verLookup.toString())
181                 : version;
182 
183         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
184         if (!sp.isEmpty() && !"unknown".equals(sp)) {
185             version = version + " " + sp.replace("Service Pack ", "SP");
186         }
187 
188         return version;
189     }

191     /**
192      * Gets suites available on the system and return as a codename
193      *
194      * @param suiteMask
195      *            The suite mask bitmask
196      *
197      * @return Suites
198      */
199     private static String parseCodeName(int suiteMask) {
200         List<String> suites = new ArrayList<>();
201         if ((suiteMask & 0x00000002) != 0) {
202             suites.add("Enterprise");
203         }
204         if ((suiteMask & 0x00000004) != 0) {
205             suites.add("BackOffice");
206         }
207         if ((suiteMask & 0x00000008) != 0) {
208             suites.add("Communication Server");
209         }
210         if ((suiteMask & 0x00000080) != 0) {
211             suites.add("Datacenter");
212         }
213         if ((suiteMask & 0x00000200) != 0) {
214             suites.add("Home");
215         }
216         if ((suiteMask & 0x00000400) != 0) {
217             suites.add("Web Server");
218         }
219         if ((suiteMask & 0x00002000) != 0) {
220             suites.add("Storage Server");
221         }
222         if ((suiteMask & 0x00004000) != 0) {
223             suites.add("Compute Cluster");
224         }
225         // 0x8000, Home Server, is included in main version name
226         return String.join(",", suites);
227     }

231         if (jvmBitness < 64 && System.getenv("ProgramFiles(x86)") != null && IS_VISTA_OR_GREATER) {
232             WmiResult<BitnessProperty> bitnessMap = Win32Processor.queryBitness();
233             if (bitnessMap.getResultCount() > 0) {

236         }
237         return jvmBitness;

280     @Override
281     public List<OSProcess> getChildProcesses(int parentPid, int limit, ProcessSort sort) {
282         Set<Integer> childPids = new HashSet<>();
283         // Get processes from ToolHelp API for parent PID
284         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
285         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
286         try {
287             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
288                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
289                     childPids.add(processEntry.th32ProcessID.intValue());
290                 }
291             }
292         } finally {
293             Kernel32.INSTANCE.CloseHandle(snapshot);
294         }
295         List<OSProcess> procList = getProcesses(childPids);
296         List<OSProcess> sorted = processSort(procList, limit, sort);
297         return Collections.unmodifiableList(sorted);
298     }

306     private List<OSProcess> processMapToList(Collection<Integer> pids) {
307         // Get data from the registry if possible
308         Map<Integer, PerfCounterBlock> processMap = processMapFromRegistry.get();
309         // otherwise performance counters with WMI backup
310         if (processMap == null) {
311             processMap = (pids == null) ? processMapFromPerfCounters.get()
312                     : ProcessPerformanceData.buildProcessMapFromPerfCounters(pids);
313         }
314 
315         Map<Integer, WtsInfo> processWtsMap = ProcessWtsData.queryProcessWtsMap(pids);
316 
317         Set<Integer> mapKeys = new HashSet<>(processWtsMap.keySet());
318         mapKeys.retainAll(processMap.keySet());
319 
320         List<OSProcess> processList = new ArrayList<>();
321         for (Integer pid : mapKeys) {
322             processList.add(new WindowsOSProcess(pid, this, processMap, processWtsMap));
323         }
324         return processList;
325     }

365     private static long querySystemUptime() {
366         // Uptime is in seconds so divide milliseconds
367         // GetTickCount64 requires Vista (6.0) or later
368         if (IS_VISTA_OR_GREATER) {
369             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
370         } else {
371             // 32 bit rolls over at ~ 49 days
372             return Kernel32.INSTANCE.GetTickCount() / 1000L;
373         }
374     }

381     private static long querySystemBootTime() {
382         String eventLog = systemLog.get();
383         if (eventLog != null) {
384             try {
385                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
386                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
387                 // Startup" is enabled we may not see event 12, so also check for most recent ID
388                 // 6005 (Event log startup) as a reasonably close backup.
389                 long event6005Time = 0L;
390                 while (iter.hasNext()) {
391                     EventLogRecord record = iter.next();
392                     if (record.getStatusCode() == 12) {
393                         // Event 12 is system boot. We want this value unless we find two 6005 events
394                         // first (may occur with Fast Boot)
395                         return record.getRecord().TimeGenerated.longValue();
396                     } else if (record.getStatusCode() == 6005) {
397                         // If we already found one, this means we've found a second one without finding
398                         // an event 12. Return the latest one.
399                         if (event6005Time > 0) {
400                             return event6005Time;
401                         }
402                         // First 6005; tentatively assign
403                         event6005Time = record.getRecord().TimeGenerated.longValue();
404                     }
405                 }
406                 // Only one 6005 found, return
407                 if (event6005Time > 0) {
408                     return event6005Time;
409                 }
410             } catch (Win32Exception e) {
411                 LOG.warn("Can't open event log \"{}\".", eventLog);
412             }
413         }
414         // If we get this far, event log reading has failed, either from no log or no
415         // startup times. Subtract up time from current time as a reasonable proxy.
416         return System.currentTimeMillis() / 1000L - querySystemUptime();
417     }

424     /**
425      * Enables debug privileges for this process, required for OpenProcess() to get
426      * processes other than the current user
427      *
428      * @return {@code true} if debug privileges were successfully enabled.
429      */
430     private static boolean enableDebugPrivilege() {
431         HANDLEByReference hToken = new HANDLEByReference();
432         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
433                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
434         if (!success) {
435             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
436             return false;
437         }
438         try {
439             WinNT.LUID luid = new WinNT.LUID();
440             success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
441             if (!success) {
442                 LOG.error("LookupPrivilegeValue failed. Error: {}", Native.getLastError());
443                 return false;
444             }
445             WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
446             tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
447             success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
448             // Possible to partially succeed, only need to check LastError
449             int err = Native.getLastError();
450             if (err != WinError.ERROR_SUCCESS) {
451                 LOG.error("AdjustTokenPrivileges failed. Error: {}", err);
452                 return false;
453             }
454         } finally {
455             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
456         }
457         return true;
458     }

460     @Override
461     public OSService[] getServices() {
462         try (W32ServiceManager sm = new W32ServiceManager()) {
463             sm.open(Winsvc.SC_MANAGER_ENUMERATE_SERVICE);
464             Winsvc.ENUM_SERVICE_STATUS_PROCESS[] services = sm.enumServicesStatusExProcess(WinNT.SERVICE_WIN32,
465                     Winsvc.SERVICE_STATE_ALL, null);
466             OSServiceervice">OSService[] svcArray = new OSService[services.length];
467             for (int i = 0; i < services.length; i++) {
468                 State state;
469                 switch (services[i].ServiceStatusProcess.dwCurrentState) {
470                 case 1:
471                     state = STOPPED;
472                     break;
473                 case 4:
474                     state = RUNNING;
475                     break;
476                 default:
477                     state = OTHER;
478                     break;
479                 }
480                 svcArray[i] = new OSService(services[i].lpDisplayName, services[i].ServiceStatusProcess.dwProcessId,
481                         state);
482             }
483             return svcArray;
484         } catch (com.sun.jna.platform.win32.Win32Exception ex) {
485             LOG.error("Win32Exception: {}", ex.getMessage());
486             return new OSService[0];
487         }
488     }

490     private static String querySystemLog() {
491         String systemLog = GlobalConfig.get("oshi.os.windows.eventlog", "System");
492         if (systemLog.isEmpty()) {
493             // Use faster boot time approximation
494             return null;
495         }
496         // Check whether it works
497         HANDLE h = Advapi32.INSTANCE.OpenEventLog(null, systemLog);
498         if (h == null) {
499             LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
500                     systemLog);
501             return null;
502         }
503         return systemLog;
504     }
