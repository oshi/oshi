114     private void initRegistry() {
115         // Get the title indices
116         DWORDByReference index = new DWORDByReference();
117         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Process", index);
118         processIndex = index.getValue().intValue();
119         processIndexStr = Integer.toString(processIndex);
120         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Priority Base", index);
121         int priorityBaseIndex = index.getValue().intValue();
122         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Elapsed Time", index);
123         int elapsedTimeIndex = index.getValue().intValue();
124         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "ID Process", index);
125         int idProcessIndex = index.getValue().intValue();
126         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Creating Process ID", index);
127         int creatingProcessIdIndex = index.getValue().intValue();
128         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Read Bytes/sec", index);
129         int ioReadIndex = index.getValue().intValue();
130         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Write Bytes/sec", index);
131         int ioWriteIndex = index.getValue().intValue();
132         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Working Set - Private", index);
133         int workingSetPrivateIndex = index.getValue().intValue();
134 
135         // now load the Process registry to match up the offsets
136         // Sequentially increase the buffer until everything fits.
137         // Save this buffer size for later use
138         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
139         Pointer pPerfData = new Memory(this.perfDataBufferSize);
140         while (WinError.ERROR_MORE_DATA == Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA,
141                 processIndexStr, 0, null, pPerfData, lpcbData)) {
142             this.perfDataBufferSize += 4096;
143             lpcbData.setValue(this.perfDataBufferSize);
144             pPerfData = new Memory(this.perfDataBufferSize);
145         }
146 
147         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
148 
149         // See format at
150         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
151         // [ ] Object Type
152         // [ ][ ][ ] Multiple counter definitions
153         // Then multiple:
154         // [ ] Instance Definition
155         // [ ] Instance name
156         // [ ] Counter Block
157         // [ ][ ][ ] Counter data for each definition above
158 
159         long perfObjectOffset = perfData.HeaderLength;
160 
161         // Iterate object types. For Process should only be one here
162         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
163             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
164             // Identify where counter definitions start
165             long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
166             // If this isn't the Process object, ignore
167             if (perfObject.ObjectNameTitleIndex == processIndex) {
168                 for (int counter = 0; counter < perfObject.NumCounters; counter++) {
169                     PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
170                             pPerfData.share(perfCounterOffset));
171                     if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
172                         this.priorityBaseOffset = perfCounter.CounterOffset;
173                     } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
174                         this.elapsedTimeOffset = perfCounter.CounterOffset;
175                     } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
176                         this.creatingProcessIdOffset = perfCounter.CounterOffset;
177                     } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
178                         this.idProcessOffset = perfCounter.CounterOffset;
179                     } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
180                         this.ioReadOffset = perfCounter.CounterOffset;
181                     } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
182                         this.ioWriteOffset = perfCounter.CounterOffset;
183                     } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
184                         this.workingSetPrivateOffset = perfCounter.CounterOffset;
185                     }
186                     // Increment for next Counter
187                     perfCounterOffset += perfCounter.ByteLength;
188                 }
189                 // We're done, break the loop
190                 break;
191             }
192             // Increment for next object (should never need this)
193             perfObjectOffset += perfObject.TotalByteLength;
194         }
195     }

197     private void initBitness() {
198         // If bitness is 64 we are 64 bit.
199         // If 32 test if we are on 64-bit OS
200         if (bitness < 64) {
201             // Try the easy way
202             if (System.getenv("ProgramFiles(x86)") != null) {
203                 this.bitness = 64;
204             } else {
205                 // Fetch from WMI
206                 this.bitness = (WmiUtil.selectUint32From(null, "Win32_Processor", "AddressWidth", null)).intValue();
207             }
208         }

232     @Override
233     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
234         Set<Integer> childPids = new HashSet<>();
235         // Get processes from ToolHelp API for parent PID
236         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
237         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
238         try {
239             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
240                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
241                     childPids.add(processEntry.th32ProcessID.intValue());
242                 }
243             }
244         } finally {
245             Kernel32.INSTANCE.CloseHandle(snapshot);
246         }
247         List<OSProcess> procList = getProcesses(childPids);
248         List<OSProcess> sorted = processSort(procList, limit, sort);
249         return sorted.toArray(new OSProcess[sorted.size()]);
250     }

271     /**
272      * Private method to do the heavy lifting for all the getProcess functions.
273      *
274      * @param pids
275      *            A collection of pids to query. If null, the entire process
276      *            list will be queried.
277      * @return A corresponding list of processes
278      */
279     private List<OSProcess> processMapToList(Collection<Integer> pids) {
280         long now = System.currentTimeMillis();
281         List<OSProcess> procList = new ArrayList<>();
282         // define here to avoid object creation overhead later
283         // For groups
284         List<String> groupList = new ArrayList<>();
285         List<String> groupIDList = new ArrayList<>();
286 
287         int myPid = getProcessId();
288         Map<Integer, OSProcess> tempProcessMap = new HashMap<>();
289 
290         // Get processes from WTS
291         final PointerByReference ppProcessInfo = new PointerByReference();
292         IntByReference pCount = new IntByReference(0);
293         if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
294                 Wtsapi32.WTSTypeProcessInfoLevel1, Wtsapi32.WTS_ANY_SESSION, ppProcessInfo, pCount)) {
295             LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
296             return procList;
297         }
298         // extract the pointed-to pointer and create array
299         final Pointer pProcessInfo = ppProcessInfo.getValue();
300         final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
301         final WTS_PROCESS_INFO_EX[] processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
302 
303         for (WTS_PROCESS_INFO_EX procInfo : processInfo) {
304             // Only consider processes passed to this method
305             if (pids != null && !pids.contains(procInfo.ProcessId.intValue())) {
306                 continue;
307             }
308             OSProcess proc = new OSProcess();
309             proc.setProcessID(procInfo.ProcessId.intValue());
310             proc.setName(procInfo.pProcessName);
311             // For my own process, set CWD
312             if (myPid == proc.getProcessID()) {
313                 proc.setCurrentWorkingDirectory(new File(".").getAbsolutePath());
314             }
315 
316             proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
317             proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
318 
319             // Get a handle to the process for various extended info. Only gets
320             // current user unless running as administrator
321             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
322                     proc.getProcessID());
323             if (pHandle != null) {
324                 // Full path
325                 proc.setPath(Kernel32Util.QueryFullProcessImageName(pHandle, 0));
326 
327                 final HANDLEByReference phToken = new HANDLEByReference();
328                 if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY, phToken)) {
329                     Account account = Advapi32Util.getTokenAccount(phToken.getValue());
330                     proc.setUser(account.name);
331                     proc.setUserID(account.sidString);
332                     // Fetching group information incurs ~10ms latency per
333                     // process. Skip for full process list
334                     if (pids != null) {
335                         Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
336                         // get groups
337                         groupList.clear();
338                         groupIDList.clear();
339                         for (Account a : accounts) {
340                             groupList.add(a.name);
341                             groupIDList.add(a.sidString);
342                         }
343                         proc.setGroup(FormatUtil.join(",", groupList));
344                         proc.setGroupID(FormatUtil.join(",", groupIDList));
345                     }
346                 } else {
347                     int error = Kernel32.INSTANCE.GetLastError();
348                     // Access denied errors are common. Fail silently.
349                     if (error != WinError.ERROR_ACCESS_DENIED) {
350                         LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
351                                 Kernel32.INSTANCE.GetLastError());
352                     }
353                 }
354             }
355             Kernel32.INSTANCE.CloseHandle(pHandle);
356 
357             // TODO: There is no easy way to get ExecutuionState for a process.
358             // The WMI value is null. It's possible to get thread Execution
359             // State and possibly roll up.
360             proc.setState(OSProcess.State.RUNNING);
361 
362             proc.setThreadCount(procInfo.NumberOfThreads.intValue());
363             proc.setVirtualSize(procInfo.PagefileUsage.longValue());
364 
365             proc.setOpenFiles(procInfo.HandleCount.intValue());
366             tempProcessMap.put(proc.getProcessID(), proc);
367         }
368         // Clean up memory allocated in C
369         if (!Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTSTypeProcessInfoLevel1.getValue(), pProcessInfo,
370                 pCount.getValue())) {
371             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
372             return procList;
373         }
374 
375         // Get the rest of the data we don't already have from the registry
376         updateRegistryStats(pids, tempProcessMap, now);
377         // Command Line only accessible via WMI.
378         // Utilize cache to only update new processes
379         Set<Integer> emptyCommandLines = new HashSet<>();
380         for (Integer pid : tempProcessMap.keySet()) {
381             if (this.processMap.containsKey(pid) && this.processMap.get(pid).getCommandLine().isEmpty()) {
382                 emptyCommandLines.add(pid);
383             }
384         }
385         if (!emptyCommandLines.isEmpty()) {
386             StringBuilder query = new StringBuilder("WHERE ");
387             for (Integer pid : emptyCommandLines) {
388                 query.append(String.format("ProcessID=%d OR ", pid));
389             }
390             query.setLength(query.length() - 3);
391             Map<String, List<Object>> commandLineProcs = WmiUtil.selectObjectsFrom(null, "Win32_Process",
392                     PROCESS_PROPERTIES, query.toString(), PROCESS_PROPERTY_TYPES);
393             for (int p = 0; p < commandLineProcs.get("ProcessID").size(); p++) {
394                 int pid = ((Long) commandLineProcs.get("ProcessID").get(p)).intValue();
395                 if (tempProcessMap.containsKey(pid)) {
396                     OSProcess proc = tempProcessMap.get(pid);
397                     proc.setCommandLine((String) commandLineProcs.get("CommandLine").get(p));
398                 }
399             }
400         }
401 
402         // Update cached list
403         // TODO: this is ugly as we replace a perfectly good cached value with
404         // something new. Better to grab the cached value earlier in the process
405         // and update it instead of creating new
406         for (OSProcess p : tempProcessMap.values()) {
407             this.processMap.put(p.getProcessID(), p);
408             procList.add(p);
409         }
410         return new ArrayList<>(tempProcessMap.values());
411     }

413     private void updateRegistryStats(Collection<Integer> pids, Map<Integer, OSProcess> tempProcessMap, long now) {
414 
415         List<Integer> pidsToKeep = new ArrayList<>(tempProcessMap.keySet().size());
416 
417         // Sequentially increase the buffer until everything fits.
418         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
419         Pointer pPerfData = new Memory(this.perfDataBufferSize);
420         while (WinError.ERROR_MORE_DATA == Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA,
421                 processIndexStr, 0, null, pPerfData, lpcbData)) {
422             this.perfDataBufferSize += 4096;
423             lpcbData.setValue(this.perfDataBufferSize);
424             pPerfData = new Memory(this.perfDataBufferSize);
425         }
426 
427         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
428         long perfTime100nSec = perfData.PerfTime100nSec.getValue();
429 
430         // See format at
431         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
432         // [ ] Object Type
433         // [ ][ ][ ] Multiple counter definitions
434         // Then multiple:
435         // [ ] Instance Definition
436         // [ ] Instance name
437         // [ ] Counter Block
438         // [ ][ ][ ] Counter data for each definition above
439 
440         long perfObjectOffset = perfData.HeaderLength;
441 
442         // Iterate object types. For Process should only be one here
443         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
444             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
445             // If this isn't the Process object, ignore
446             if (perfObject.ObjectNameTitleIndex == processIndex) {
447                 // Skip over counter definitions
448                 // There will be many of these, this points to the first one
449                 long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
450 
451                 // We need this for every process, initialize outside loop to
452                 // save overhead
453                 PERF_COUNTER_BLOCK perfCounterBlock = null;
454                 // Iterate instances.
455                 // The last instance is _Total so subtract 1 from max
456                 for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
457                     PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
458                             pPerfData.share(perfInstanceOffset));
459                     long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
460 
461                     int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
462                     if ((pids == null || pids.contains(pid)) && tempProcessMap.containsKey(pid)) {
463                         pidsToKeep.add(pid);
464                         OSProcess proc = tempProcessMap.get(pid);
465 
466                         // If name is empty, populate from registry
467                         if (proc.getName().isEmpty()) {
468                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
469                         }
470 
471                         proc.setUpTime(
472                                 (perfTime100nSec - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset))
473                                         / 10000L);
474                         proc.setStartTime(now - proc.getUpTime());
475                         proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
476                         proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
477                         proc.setResidentSetSize(
478                                 pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
479                         proc.setParentProcessID(
480                                 pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
481                         proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
482 
483                         // If start time is newer than cached version, delete
484                         // cache
485                         if (this.processMap.containsKey(proc.getProcessID())
486                                 && this.processMap.get(proc.getProcessID()).getStartTime() < proc.getStartTime()) {
487                             this.processMap.remove(proc.getProcessID());
488                         }
489                     }
490 
491                     // Increment to next instance
492                     perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
493                     perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
494                 }
495                 // We're done, break the loop
496                 break;
497             }
498             // Increment for next object (should never need this)
499             perfObjectOffset += perfObject.TotalByteLength;
500         }
501         if (pids == null) {
502             for (Integer pid : new HashSet<>(this.processMap.keySet())) {
503                 if (!pidsToKeep.contains(pid)) {
504                     this.processMap.remove(pid);
505                 }
506             }
507         }
508     }

552     /**
553      * Enables debug privileges for this process, required for OpenProcess() to
554      * get processes other than the current user
555      */
556     private static void enableDebugPrivilege() {
557         HANDLEByReference hToken = new HANDLEByReference();
558         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
559                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
560         if (!success) {
561             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
562             return;
563         }
564         WinNT.LUID luid = new WinNT.LUID();
565         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
566         if (!success) {
567             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
568             return;
569         }
570         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
571         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
572         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
573         if (!success) {
574             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
575         }
576         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
577     }
