5   package oshi.software.os.windows;
6   
7   import java.util.List;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.software.common.os.windows.WindowsOSFileStore;
11  import oshi.software.os.OSFileStore;
12  
13  /**
14   * FFM-based Windows OSFileStore implementation.
15   */
16  @ThreadSafe
17  public class WindowsOSFileStoreFFM extends WindowsOSFileStore {
18  
19      public WindowsOSFileStoreFFM(String name, String volume, String label, String mount, String options, String uuid,
20              boolean local, String logicalVolume, String description, String fsType, long freeSpace, long usableSpace,
21              long totalSpace, long freeInodes, long totalInodes) {
22          super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
23                  usableSpace, totalSpace, freeInodes, totalInodes);
24      }
25  
26      @Override
27      public boolean updateAttributes() {
28          // Check if we have the volume locally
29          List<OSFileStore> volumes;
30          if (isLocal()) {
31              volumes = WindowsFileSystemFFM.getLocalVolumes(getVolume());
32          } else {
33              // Not locally, search WMI
34              String nameToMatch = getMount().length() < 2 ? null : getMount().substring(0, 2);
35              volumes = WindowsFileSystemFFM.getWmiVolumes(nameToMatch, false);
36          }
37          for (OSFileStore fileStore : volumes) {
38              if (getVolume().equals(fileStore.getVolume()) && getMount().equals(fileStore.getMount())) {
39                  updateFrom(fileStore);
40                  return true;
41              }
42          }
43          return false;
44      }
45  }
