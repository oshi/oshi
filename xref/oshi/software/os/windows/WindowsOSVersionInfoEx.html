33  /**
34   * Contains operating system version information. The information includes major and minor version numbers, a build
35   * number, a platform identifier, and descriptive text about the operating system.
36   * 
37   * @author dblock[at]dblock[dot]org
38   */
39  public class WindowsOSVersionInfoEx
40      implements OperatingSystemVersion
41  {
42      private static final Logger LOG = LoggerFactory.getLogger( WindowsOSVersionInfoEx.class );
43  
44      private OSVERSIONINFOEX _versionInfo;
45  
46      public WindowsOSVersionInfoEx()
47      {
48          this._versionInfo = new OSVERSIONINFOEX();
49          if ( !Kernel32.INSTANCE.GetVersionEx( this._versionInfo ) )
50          {
51              LOG.error( "Failed to Initialize OSVersionInfoEx. Error code: {}", Kernel32.INSTANCE.GetLastError() );
52              this._versionInfo = null;
53          }
54          else
55          {
56              LOG.debug( "Initialized OSVersionInfoEx" );
57          }
58      }
59  
60      /**
61       * The major version number of the operating system.
62       * 
63       * @return The major version within the following supported operating systems. Windows 8: 6.2 Windows Server 2012:
64       *         6.2 Windows 7: 6.1 Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows Vista: 6.0 Windows Server
65       *         2003 R2: 5.2 Windows Home Server: 5.2 Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
66       *         Windows XP: 5.1 Windows 2000: 5.0
67       */
68      public int getMajor()
69      {
70          if ( this._versionInfo == null )
71          {
72              LOG.warn( "OSVersionInfoEx not initialized. No version data available" );
73              return 0;
74          }
75          return this._versionInfo.dwMajorVersion.intValue();
76      }
77  
78      /**
79       * The minor version number of the operating system.
80       * 
81       * @return The minor version within the following supported operating systems. Windows 8: 6.2 Windows Server 2012:
82       *         6.2 Windows 7: 6.1 Windows Server 2008 R2: 6.1 Windows Server 2008: 6.0 Windows Vista: 6.0 Windows Server
83       *         2003 R2: 5.2 Windows Home Server: 5.2 Windows Server 2003: 5.2 Windows XP Professional x64 Edition: 5.2
84       *         Windows XP: 5.1 Windows 2000: 5.0
85       */
86      public int getMinor()
87      {
88          if ( this._versionInfo == null )
89          {
90              LOG.warn( "OSVersionInfoEx not initialized. No version data available" );
91              return 0;
92          }
93          return this._versionInfo.dwMinorVersion.intValue();
94      }
95  
96      /**
97       * The build number of the operating system.
98       * 
99       * @return Build number.
100      */
101     public int getBuildNumber()
102     {
103         if ( this._versionInfo == null )
104         {
105             LOG.warn( "OSVersionInfoEx not initialized. No build number available" );
106             return 0;
107         }
108         return this._versionInfo.dwBuildNumber.intValue();
109     }
110 
111     /**
112      * The operating system platform. This member can be VER_PLATFORM_WIN32_NT.
113      * 
114      * @return Platform ID.
115      */
116     public int getPlatformId()
117     {
118         if ( this._versionInfo == null )
119         {
120             LOG.warn( "OSVersionInfoEx not initialized. No platform id available" );
121             return 0;
122         }
123         return this._versionInfo.dwPlatformId.intValue();
124     }
125 
126     /**
127      * String, such as "Service Pack 3", that indicates the latest Service Pack installed on the system. If no Service
128      * Pack has been installed, the string is empty.
129      * 
130      * @return Service pack.
131      */
132     public String getServicePack()
133     {
134         if ( this._versionInfo == null )
135         {
136             LOG.warn( "OSVersionInfoEx not initialized. No service pack data available" );
137             return "";
138         }
139         return Native.toString( this._versionInfo.szCSDVersion );
140     }
141 
142     /**
143      * A bit mask that identifies the product suites available on the system.
144      * 
145      * @return Suite mask.
146      */
147     public int getSuiteMask()
148     {
149         if ( this._versionInfo == null )
150         {
151             LOG.warn( "OSVersionInfoEx not initialized. No suite mask data available" );
152             return 0;
153         }
154 
155         return this._versionInfo.wSuiteMask.intValue();
156     }
157 
158     /**
159      * Any additional information about the system.
160      * 
161      * @return Product type.
162      */
163     public byte getProductType()
164     {
165         if ( this._versionInfo == null )
166         {
167             LOG.warn( "OSVersionInfoEx not initialized. No product type available" );
168             return 0;
169         }
170         return this._versionInfo.wProductType;
171     }
172 
173     @Override
174     public String toString()
175     {
176         if ( this._versionInfo == null )
177         {
178             LOG.warn( "OSVersionInfoEx not initialized. No version data available" );
179             return "Unknown";
180         }
181 
182         String version = null;
183 
184         // see
185         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
186         if ( getPlatformId() == WinNT.VER_PLATFORM_WIN32_NT )
187         {
188             boolean ntWorkstation = getProductType() == WinNT.VER_NT_WORKSTATION;
189             if ( getMajor() == 10 && getMinor() == 0 && ntWorkstation )
190             {
191                 version = "10";
192             }
193             else if ( getMajor() == 10 && getMinor() == 0 && !ntWorkstation )
194             {
195                 version = "Server 2016";
196             }
197             else if ( getMajor() == 6 && getMinor() == 3 && ntWorkstation )
198             {
199                 version = "8.1";
200             }
201             else if ( getMajor() == 6 && getMinor() == 3 && !ntWorkstation )
202             {
203                 version = "Server 2012 R2";
204             }
205             else if ( getMajor() == 6 && getMinor() == 2 && ntWorkstation )
206             {
207                 version = "8";
208             }
209             else if ( getMajor() == 6 && getMinor() == 2 && !ntWorkstation )
210             {
211                 version = "Server 2012";
212             }
213             else if ( getMajor() == 6 && getMinor() == 1 && ntWorkstation )
214             {
215                 version = "7";
216             }
217             else if ( getMajor() == 6 && getMinor() == 1 && !ntWorkstation )
218             {
219                 version = "Server 2008 R2";
220             }
221             else if ( getMajor() == 6 && getMinor() == 0 && !ntWorkstation )
222             {
223                 version = "Server 2008";
224             }
225             else if ( getMajor() == 6 && getMinor() == 0 && ntWorkstation )
226             {
227                 version = "Vista";
228             }
229             else if ( getMajor() == 5 && getMinor() == 2 && !ntWorkstation )
230             {
231                 version =
232                     User32.INSTANCE.GetSystemMetrics( WinUser.SM_SERVERR2 ) != 0 ? "Server 2003" : "Server 2003 R2";
233             }
234             else if ( getMajor() == 5 && getMinor() == 2 && ntWorkstation )
235             {
236                 version = "XP"; // 64 bits
237             }
238             else if ( getMajor() == 5 && getMinor() == 1 )
239             {
240                 version = "XP"; // 32 bits
241             }
242             else if ( getMajor() == 5 && getMinor() == 0 )
243             {
244                 version = "2000";
245             }
246             else if ( getMajor() == 4 )
247             {
248                 version = "NT 4";
249                 if ( "Service Pack 6".equals( getServicePack() ) )
250                 {
251                     if ( Advapi32Util.registryKeyExists( WinReg.HKEY_LOCAL_MACHINE,
252                                                          "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Hotfix\\Q246009" ) )
253                     {
254                         return "NT4 SP6a";
255                     }
256                 }
257 
258             }
259             else
260             {
261                 throw new RuntimeException( "Unsupported Windows NT version: " + this._versionInfo.toString() );
262             }
263 
264             if ( this._versionInfo.wServicePackMajor.intValue() > 0 )
265             {
266                 version = version + " SP" + this._versionInfo.wServicePackMajor.intValue();
267             }
268 
269         }
270         else if ( getPlatformId() == WinNT.VER_PLATFORM_WIN32_WINDOWS )
271         {
272             if ( getMajor() == 4 && getMinor() == 90 )
273             {
274                 version = "ME";
275             }
276             else if ( getMajor() == 4 && getMinor() == 10 )
277             {
278                 if ( this._versionInfo.szCSDVersion[1] == 'A' )
279                 {
280                     version = "98 SE";
281                 }
282                 else
283                 {
284                     version = "98";
285                 }
286             }
287             else if ( getMajor() == 4 && getMinor() == 0 )
288             {
289                 if ( this._versionInfo.szCSDVersion[1] == 'C' || this._versionInfo.szCSDVersion[1] == 'B' )
290                 {
291                     version = "95 OSR2";
292                 }
293                 else
294                 {
295                     version = "95";
296                 }
297             }
298             else
299             {
300                 throw new RuntimeException( "Unsupported Windows 9x version: " + this._versionInfo.toString() );
301             }
302         }
303         else
304         {
305             throw new RuntimeException( "Unsupported Windows platform: " + this._versionInfo.toString() );
306         }
307 
308         return version;
309     }
310 
311     public WindowsOSVersionInfoEx( OSVERSIONINFOEX versionInfo )
312     {
313         this._versionInfo = versionInfo;
314     }
