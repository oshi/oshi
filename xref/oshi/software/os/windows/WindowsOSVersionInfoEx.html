19  package oshi.software.os.windows;

21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import com.sun.jna.platform.win32.User32;
28  import com.sun.jna.platform.win32.WinNT;
29  import com.sun.jna.platform.win32.WinUser;
30  
31  import oshi.software.common.AbstractOSVersionInfoEx;
32  import oshi.util.ParseUtil;
33  import oshi.util.StringUtil;
34  import oshi.util.platform.windows.WmiUtil;
35  import oshi.util.platform.windows.WmiUtil.WmiQuery;
36  import oshi.util.platform.windows.WmiUtil.WmiResult;
37  
38  public class WindowsOSVersionInfoEx extends AbstractOSVersionInfoEx {
39  
40      private static final long serialVersionUID = 1L;
41  
42      private static final Logger LOG = LoggerFactory.getLogger(WindowsOSVersionInfoEx.class);
43  
44      enum OSVersionProperty {
45          VERSION, PRODUCTTYPE, BUILDNUMBER, CSDVERSION, SUITEMASK;
46      }
47  
48      public WindowsOSVersionInfoEx() {
49          init();
50      }
51  
52      private void init() {
53          // Populate a key-value map from WMI
54          WmiQuery<OSVersionProperty> osVersionQuery = WmiUtil.createQuery("Win32_OperatingSystem",
55                  OSVersionProperty.class);
56          WmiResult<OSVersionProperty> versionInfo = WmiUtil.queryWMI(osVersionQuery);
57          if (versionInfo.getResultCount() < 1) {
58              LOG.warn("No version data available.");
59              setVersion(System.getProperty("os.version"));
60              setCodeName("");
61              setBuildNumber("");
62          } else {
63              // Guaranteed that versionInfo is not null and lists non-empty
64              // before calling the parse*() methods
65              int suiteMask = versionInfo.getInteger(OSVersionProperty.SUITEMASK, 0);
66              setVersion(parseVersion(versionInfo, suiteMask));
67              setCodeName(parseCodeName(suiteMask));
68              setBuildNumber(versionInfo.getString(OSVersionProperty.BUILDNUMBER, 0));
69              LOG.debug("Initialized OSVersionInfoEx");
70          }
71      }
72  
73      /**
74       * Gets the operating system version
75       * 
76       * @param suiteMask
77       *
78       * @return Version
79       */
80      private String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask) {
81  
82          // Initialize a default, sane value
83          String version = System.getProperty("os.version");
84  
85          // Version is major.minor.build. Parse the version string for
86          // major/minor and get the build number separately
87          String[] verSplit = (versionInfo.getString(OSVersionProperty.VERSION, 0)).split("\\D");
88          int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
89          int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
90  
91          // see
92          // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
93          boolean ntWorkstation = versionInfo.getInteger(OSVersionProperty.PRODUCTTYPE, 0)
94                  .equals(WinNT.VER_NT_WORKSTATION);
95          if (major == 10) {
96              if (minor == 0) {
97                  version = ntWorkstation ? "10" : "Server 2016";
98              }
99          } else if (major == 6) {
100             if (minor == 3) {
101                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
102             } else if (minor == 2) {
103                 version = ntWorkstation ? "8" : "Server 2012";
104             } else if (minor == 1) {
105                 version = ntWorkstation ? "7" : "Server 2008 R2";
106             } else if (minor == 0) {
107                 version = ntWorkstation ? "Vista" : "Server 2008";
108             }
109         } else if (major == 5) {
110             if (minor == 2) {
111                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
112                     version = "Home Server";
113                 } else if (ntWorkstation) {
114                     version = "XP"; // 64 bits
115                 } else {
116                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
117                             : "Server 2003 R2";
118                 }
119             } else if (minor == 1) {
120                 version = "XP"; // 32 bits
121             } else if (minor == 0) {
122                 version = "2000";
123             }
124         }
125 
126         String sp = versionInfo.getString(OSVersionProperty.CSDVERSION, 0);
127         if (!sp.isEmpty() && !"unknown".equals(sp)) {
128             version = version + " " + sp.replace("Service Pack ", "SP");
129         }
130 
131         return version;
132     }
133 
134     /**
135      * Gets suites available on the system and return as a codename
136      * 
137      * @param suiteMask
138      *
139      * @return Suites
140      */
141     private String parseCodeName(int suiteMask) {
142         List<String> suites = new ArrayList<>();
143         if ((suiteMask & 0x00000002) != 0) {
144             suites.add("Enterprise");
145         }
146         if ((suiteMask & 0x00000004) != 0) {
147             suites.add("BackOffice");
148         }
149         if ((suiteMask & 0x00000008) != 0) {
150             suites.add("Communication Server");
151         }
152         if ((suiteMask & 0x00000080) != 0) {
153             suites.add("Datacenter");
154         }
155         if ((suiteMask & 0x00000200) != 0) {
156             suites.add("Home");
157         }
158         if ((suiteMask & 0x00000400) != 0) {
159             suites.add("Web Server");
160         }
161         if ((suiteMask & 0x00002000) != 0) {
162             suites.add("Storage Server");
163         }
164         if ((suiteMask & 0x00004000) != 0) {
165             suites.add("Compute Cluster");
166         }
167         // 0x8000, Home Server, is included in main version name
168         return StringUtil.join(",", suites);
169     }
