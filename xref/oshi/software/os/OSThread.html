24  package oshi.software.os;
25  
26  import oshi.software.os.OSProcess.State;
27  
28  /**
29   * Represents a Thread/Task on the operating system.
30   */
31  public interface OSThread {
32  
33      /**
34       * The thread id. The meaning of this value is OS-dependent.
35       *
36       * @return Returns the id of the thread.
37       */
38      int getThreadId();
39  
40      /**
41       * The name of the thread. Presence of a name is operating-system dependent and
42       * may include information (such as an index of running threads) that changes
43       * during execution.
44       *
45       * @return Returns the name of the task/thread.
46       */
47      default String getName() {
48          return "";
49      }
50  
51      /**
52       * Gets the execution state of the task/thread.
53       *
54       * @return Returns the execution state of the task/thread.
55       */
56      State getState();
57  
58      /**
59       * Gets cumulative CPU usage of this thread.
60       *
61       * @return The proportion of up time that the thread was executing in kernel or
62       *         user mode.
63       */
64      double getThreadCpuLoadCumulative();
65  
66      /**
67       * Gets CPU usage of this thread since a previous snapshot of the same thread,
68       * provided as a parameter.
69       *
70       * @param thread
71       *            An {@link OSThread} object containing statistics for this same
72       *            thread collected at a prior point in time. May be null.
73       *
74       * @return If the prior snapshot is for the same thread at a prior point in
75       *         time, the proportion of elapsed up time between the current thread
76       *         snapshot and the previous one that the thread was executing in kernel
77       *         or user mode. Returns cumulative load otherwise.
78       */
79      double getThreadCpuLoadBetweenTicks(OSThread thread);
80  
81      /**
82       * The owning process of this thread. For single-threaded processes, the owning
83       * process ID may be the same as the thread's ID.
84       *
85       * @return The owning process of this thread.
86       */
87      int getOwningProcessId();
88  
89      /**
90       * The memory address above which this thread can run.
91       *
92       * @return The start address.
93       */
94      default long getStartMemoryAddress() {
95          return 0L;
96      }
97  
98      /**
99       * A snapshot of the context switches the thread has done. Since the context
100      * switches could be voluntary and non-voluntary, this gives the sum of both.
101      * <p>
102      * Not available on AIX.
103      *
104      * @return sum of both voluntary and involuntary context switches.
105      */
106     default long getContextSwitches() {
107         return 0L;
108     }
109 
110     /**
111      * The number of minor (soft) faults the thread has made which have not required
112      * loading a memory page from disk. Sometimes called reclaims. Linux only.
113      *
114      * @return minor faults.
115      */
116     default long getMinorFaults() {
117         return 0L;
118     }
119 
120     /**
121      * The number of major (hard) faults the thread has made which have required
122      * loading a memory page from disk. Linux only.
123      *
124      * @return major faults.
125      */
126     default long getMajorFaults() {
127         return 0L;
128     }
129 
130     /**
131      * Kernel (privileged) time used by the thread.
132      *
133      * @return Returns the number of milliseconds the task/thread has executed in
134      *         kernel/system mode.
135      */
136     long getKernelTime();
137 
138     /**
139      * User time used by the thread.
140      *
141      * @return Returns the number of milliseconds the task/thread has executed in
142      *         user mode.
143      */
144     long getUserTime();
145 
146     /**
147      * Elapsed/up-time of the thread.
148      *
149      * @return Returns the number of milliseconds since the task/thread started.
150      */
151     long getUpTime();
152 
153     /**
154      * The start time of the thread.
155      *
156      * @return Returns the start time of the task/thread in number of milliseconds
157      *         since January 1, 1970.
158      */
159     long getStartTime();
160 
161     /**
162      * Priority of the thread, the meaning of which is dependent on the OS.
163      *
164      * @return priority.
165      */
166     int getPriority();
167 
168     /**
169      * Attempts to updates process attributes. Returns false if the update fails,
170      * which will occur if the process no longer exists. Not implemented for macOS,
171      * as thread ID is simply an index and not unique.
172      *
173      * @return {@code true} if the update was successful, false if the update
174      *         failed. In addition, on a failed update the thread state will be
175      *         changed to {@link State#INVALID}.
176      */
177     default boolean updateAttributes() {
178         return false;
179     }
180 }
