77      // Order the field is in /proc/pid/stat
78      enum ProcPidStat {
79          // The parsing implementation in ParseUtil requires these to be declared
80          // in increasing order
81          PPID(4), USER_TIME(14), KERNEL_TIME(15), PRIORITY(18), THREAD_COUNT(20), START_TIME(22), VSZ(23), RSS(24);
82  
83          private int order;
84  
85          public int getOrder() {
86              return this.order;
87          }
88  
89          ProcPidStat(int order) {
90              this.order = order;
91          }
92      }

96      static {
97          for (ProcPidStat stat : ProcPidStat.values()) {
98              PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder();
99          }

125     /**
126      * Correlate the youngest process start time in seconds with start time in
127      * jiffies
128      */
129     private static void init() {
130         // To correlate a process start time in seconds with the same process
131         // start time in jiffies. We prefer the youngest (or close to it) which
132         // minimizes its up time (etime)
133         // Timeline:
134         // BOOT|<----jiffies---->|<----etime---->|NOW
135         // BOOT|<------------uptime------------->|NOW
136 
137         // To avoid having to check all processes we can just pick the highest
138         // PID. This will either be the youngest one or at least big enough.
139 
140         // Get all the pid files (guaranteed to be digit-only filenames)
141         File[] pids = ProcUtil.getPidFiles();
142         // Sort descending "numerically"
143         Arrays.sort(pids, new Comparator<File>() {
144             @Override
145             public int compare(File f1, File f2) {
146                 return Integer.valueOf(f2.getName()).compareTo(Integer.valueOf(f1.getName()));
147             }
148         });
149 
150         // Iterate /proc/[pid]/stat checking the creation time (field 22,
151         // jiffies since boot). Since we're working on descending PIDs, we
152         // expect the first (higher PIDs) to be younger, but may have processes
153         // that ended since we collected the files. The first time we get a
154         // value we'll save it as the youngest and quit.
155         long youngestJiffies = 0L;
156         String youngestPid = "";
157         for (File pid : pids) {
158             List<String> stat = FileUtil.readFile(String.format("/proc/%s/stat", pid.getName()), false);
159             if (!stat.isEmpty()) {
160                 String[] split = ParseUtil.whitespaces.split(stat.get(0));
161                 if (split.length >= 22) {
162                     youngestJiffies = ParseUtil.parseLongOrDefault(split[21], 0L);
163                     youngestPid = pid.getName();
164                     break;
165                 }
166             }
167         }
168         LOG.debug("Youngest PID is {} with {} jiffies", youngestPid, youngestJiffies);
169         // Shouldn't happen but avoiding Division by zero
170         if (youngestJiffies == 0) {
171             LOG.error("Couldn't find any running processes, which is odd since we are in a running process. "
172                     + "Process time values are in jiffies, not milliseconds.");
173             return;
174         }
175 
176         float startTimeSecsSinceBoot = ProcUtil.getSystemUptimeFromProc();
177         bootTime = System.currentTimeMillis() - (long) (1000 * startTimeSecsSinceBoot);
178 
179         // This takes advantage of the fact that ps does all the heavy lifting
180         // of sorting out HZ internally.
181         String etime = ExecutingCommand.getFirstAnswer(String.format("ps -p %s -o etimes=", youngestPid));
182         // Since we picked the youngest process, it's safe to assume an
183         // etime close to 0 in case this command fails; the longer the system
184         // has been up, the less impact this assumption will have
185         if (!etime.isEmpty()) {
186             if (LOG.isDebugEnabled()) {
187                 LOG.debug("Etime is {} seconds", etime.trim());
188             }
189             startTimeSecsSinceBoot -= Float.parseFloat(etime.trim());
190         }
191         // By subtracting etime (secs) from uptime (secs) we get uptime (in
192         // secs) when the process was started. This correlates with startTime in
193         // jiffies for this process
194         LOG.debug("Start time in secs: {}", startTimeSecsSinceBoot);
195         if (startTimeSecsSinceBoot <= 0) {
196             LOG.warn("Couldn't calculate jiffies per second. "
197                     + "Process time values are in jiffies, not milliseconds.");
198             return;
199         }
200 
201         // divide jiffies (since boot) by seconds (since boot)
202         hz = (long) (youngestJiffies / startTimeSecsSinceBoot + 0.5f);
203         // reset to default if value is invalid
204         if (hz == 0)
205             hz = 1000L;
206     }

210             this.bitness = 64;
211         }

214     private static int getMemoryPageSize() {
215         try {
216             return Libc.INSTANCE.getpagesize();
217         } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
218             LOG.error("Failed to get the memory page size.", e);
219         }
220         // default to 4K if the above call fails
221         return 4096;
222     }

236     public OSProcess[] getProcesses(int limit, ProcessSort sort) {
237         List<OSProcess> procs = new ArrayList<>();
238         File[] pids = ProcUtil.getPidFiles();
239 
240         // now for each file (with digit name) get process info
241         for (File pid : pids) {
242             OSProcess proc = getProcess(ParseUtil.parseIntOrDefault(pid.getName(), 0));
243             if (proc != null) {
244                 procs.add(proc);
245             }
246         }
247         List<OSProcess> sorted = processSort(procs, limit, sort);
248         return sorted.toArray(new OSProcess[sorted.size()]);
249     }

254     @Override
255     public OSProcess getProcess(int pid) {
256         String path = "";
257         Pointer buf = new Memory(1024);
258         int size = Libc.INSTANCE.readlink(String.format("/proc/%d/exe", pid), buf, 1023);
259         if (size > 0) {
260             path = buf.getString(0).substring(0, size);
261         }
262         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format("/proc/%d/io", pid), ":");
263         long now = System.currentTimeMillis();
264         // See man proc for how to parse /proc/[pid]/stat
265         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
266         // A race condition may leave us with an empty string
267         if (stat.isEmpty()) {
268             return null;
269         }
270         OSProcess proc = new OSProcess();
271         // We can get name and status more easily from /proc/pid/status which we
272         // call later, so just get the numeric bits here
273         proc.setProcessID(pid);
274         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
275         proc.setParentProcessID((int) statArray[ProcPidStat.PPID.ordinal()]);
276         proc.setThreadCount((int) statArray[ProcPidStat.THREAD_COUNT.ordinal()]);
277         proc.setPriority((int) statArray[ProcPidStat.PRIORITY.ordinal()]);
278         proc.setVirtualSize(statArray[ProcPidStat.VSZ.ordinal()]);
279         proc.setResidentSetSize(statArray[ProcPidStat.RSS.ordinal()] * this.memoryPageSize);
280         proc.setKernelTime(statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / hz);
281         proc.setUserTime(statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / hz);
282         proc.setStartTime(bootTime + statArray[ProcPidStat.START_TIME.ordinal()] * 1000L / hz);
283         proc.setUpTime(now - proc.getStartTime());
284         // See man proc for how to parse /proc/[pid]/io
285         proc.setBytesRead(ParseUtil.parseLongOrDefault(MapUtil.getOrDefault(io, "read_bytes", ""), 0L));
286         proc.setBytesWritten(ParseUtil.parseLongOrDefault(MapUtil.getOrDefault(io, "write_bytes", ""), 0L));
287 
288         // gets the open files count
289         List<String> openFilesList = ExecutingCommand.runNative(String.format("ls -f /proc/%d/fd", pid));
290         proc.setOpenFiles(openFilesList.size() - 1L);
291 
292         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format("/proc/%d/status", pid), ":");
293         proc.setName(MapUtil.getOrDefault(status, "Name", ""));
294         proc.setPath(path);
295         switch (MapUtil.getOrDefault(status, "State", "U").charAt(0)) {
296         case 'R':
297             proc.setState(OSProcess.State.RUNNING);
298             break;
299         case 'S':
300             proc.setState(OSProcess.State.SLEEPING);
301             break;
302         case 'D':
303             proc.setState(OSProcess.State.WAITING);
304             break;
305         case 'Z':
306             proc.setState(OSProcess.State.ZOMBIE);
307             break;
308         case 'T':
309             proc.setState(OSProcess.State.STOPPED);
310             break;
311         default:
312             proc.setState(OSProcess.State.OTHER);
313             break;
314         }
315         proc.setUserID(ParseUtil.whitespaces.split(MapUtil.getOrDefault(status, "Uid", ""))[0]);
316         proc.setGroupID(ParseUtil.whitespaces.split(MapUtil.getOrDefault(status, "Gid", ""))[0]);
317         OSUser user = userGroupInfo.getUser(proc.getUserID());
318         if (user != null) {
319             proc.setUser(user.getUserName());
320         }
321         proc.setGroup(userGroupInfo.getGroupName(proc.getGroupID()));
322 
323         // THe /proc/pid/cmdline value is null-delimited
324         proc.setCommandLine(FileUtil.getStringFromFile(String.format("/proc/%d/cmdline", pid)));
325         try {
326             String cwdLink = String.format("/proc/%d/cwd", pid);
327             String cwd = new File(cwdLink).getCanonicalPath();
328             if (!cwd.equals(cwdLink)) {
329                 proc.setCurrentWorkingDirectory(cwd);
330             }
331         } catch (IOException e) {
332             LOG.trace("Couldn't find cwd for pid {}: {}", pid, e);
333         }
334         return proc;
335     }

340     @Override
341     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
342         List<OSProcess> procs = new ArrayList<>();
343         File[] procFiles = ProcUtil.getPidFiles();
344 
345         // now for each file (with digit name) get process info
346         for (File procFile : procFiles) {
347             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), 0);
348             if (parentPid == getParentPidFromProcFile(pid)) {
349                 OSProcess proc = getProcess(pid);
350                 if (proc != null) {
351                     procs.add(proc);
352                 }
353             }
354         }
355         List<OSProcess> sorted = processSort(procs, limit, sort);
356         return sorted.toArray(new OSProcess[sorted.size()]);
357     }

362         return (int) statArray[ProcPidStat.PPID.ordinal()];

407     private void setFamilyFromReleaseFiles() {
408         if (this.family == null) {
409             // There are two competing options for family/version information.
410             // Newer systems are adopting a standard /etc/os-release file:
411             // https://www.freedesktop.org/software/systemd/man/os-release.html
412             //
413             // Some systems are still using the lsb standard which parses a
414             // variety of /etc/*-release files and is most easily accessed via
415             // the commandline lsb_release -a, see here:
416             // http://linux.die.net/man/1/lsb_release
417             // In this case, the /etc/lsb-release file (if it exists) has
418             // optional overrides to the information in the /etc/distrib-release
419             // files, which show: "Distributor release x.x (Codename)"
420             //
421 
422             // Attempt to read /etc/os-release file.
423             if (readOsRelease()) {
424                 // If successful, we're done. this.family has been set and
425                 // possibly the versionID and codeName
426                 return;
427             }
428 
429             // Attempt to execute the `lsb_release` command
430             if (execLsbRelease()) {
431                 // If successful, we're done. this.family has been set and
432                 // possibly the versionID and codeName
433                 return;
434             }
435 
436             // The above two options should hopefully work on most
437             // distributions. If not, we keep having fun.
438             // Attempt to read /etc/lsb-release file
439             if (readLsbRelease()) {
440                 // If successful, we're done. this.family has been set and
441                 // possibly the versionID and codeName
442                 return;
443             }
444 
445             // If we're still looking, we search for any /etc/*-release (or
446             // similar) filename, for which the first line should be of the
447             // "Distributor release x.x (Codename)" format or possibly a
448             // "Distributor VERSION x.x (Codename)" format
449             String etcDistribRelease = getReleaseFilename();
450             if (readDistribRelease(etcDistribRelease)) {
451                 // If successful, we're done. this.family has been set and
452                 // possibly the versionID and codeName
453                 return;
454             }
455             // If we've gotten this far with no match, use the distrib-release
456             // filename (defaults will eventually give "Unknown")
457             this.family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
458                     .replace("version", "").replace("-", "").replace("_", ""));
459         }
460     }

462     /**
463      * Attempts to read /etc/os-release
464      *
465      * @return true if file successfully read and NAME= found
466      */
467     private boolean readOsRelease() {
468         if (new File("/etc/os-release").exists()) {
469             List<String> osRelease = FileUtil.readFile("/etc/os-release");
470             // Search for NAME=
471             for (String line : osRelease) {
472                 if (line.startsWith("VERSION=")) {
473                     LOG.debug("os-release: {}", line);
474                     // remove beginning and ending '"' characters, etc from
475                     // VERSION="14.04.4 LTS, Trusty Tahr" (Ubuntu style)
476                     // or VERSION="17 (Beefy Miracle)" (os-release doc style)
477                     line = line.replace("VERSION=", "").replaceAll("^\"|\"$", "").trim();
478                     String[] split = line.split("[()]");
479                     if (split.length <= 1) {
480                         // If no parentheses, check for Ubuntu's comma format
481                         split = line.split(", ");
482                     }
483                     if (split.length > 0) {
484                         this.versionId = split[0].trim();
485                     }
486                     if (split.length > 1) {
487                         this.codeName = split[1].trim();
488                     }
489                 } else if (line.startsWith("NAME=") && this.family == null) {
490                     LOG.debug("os-release: {}", line);
491                     // remove beginning and ending '"' characters, etc from
492                     // NAME="Ubuntu"
493                     this.family = line.replace("NAME=", "").replaceAll("^\"|\"$", "").trim();
494                 } else if (line.startsWith("VERSION_ID=") && this.versionId == null) {
495                     LOG.debug("os-release: {}", line);
496                     // remove beginning and ending '"' characters, etc from
497                     // VERSION_ID="14.04"
498                     this.versionId = line.replace("VERSION_ID=", "").replaceAll("^\"|\"$", "").trim();
499                 }
500             }
501         }
502         return this.family != null;
503     }

505     /**
506      * Attempts to execute `lsb_release -a`
507      *
508      * @return true if the command successfully executed and Distributor ID: or
509      *         Description: found
510      */
511     private boolean execLsbRelease() {
512         // If description is of the format Distrib release x.x (Codename)
513         // that is primary, otherwise use Distributor ID: which returns the
514         // distribution concatenated, e.g., RedHat instead of Red Hat
515         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
516             if (line.startsWith("Description:")) {
517                 LOG.debug("lsb_release -a: {}", line);
518                 line = line.replace("Description:", "").trim();
519                 if (line.contains(" release ")) {
520                     this.family = parseRelease(line, " release ");
521                 }
522             } else if (line.startsWith("Distributor ID:") && this.family == null) {
523                 LOG.debug("lsb_release -a: {}", line);
524                 this.family = line.replace("Distributor ID:", "").trim();
525             } else if (line.startsWith("Release:") && this.versionId == null) {
526                 LOG.debug("lsb_release -a: {}", line);
527                 this.versionId = line.replace("Release:", "").trim();
528             } else if (line.startsWith("Codename:") && this.codeName == null) {
529                 LOG.debug("lsb_release -a: {}", line);
530                 this.codeName = line.replace("Codename:", "").trim();
531             }
532         }
533         return this.family != null;
534     }

536     /**
537      * Attempts to read /etc/lsb-release
538      *
539      * @return true if file successfully read and DISTRIB_ID or
540      *         DISTRIB_DESCRIPTION found
541      */
542     private boolean readLsbRelease() {
543         if (new File("/etc/lsb-release").exists()) {
544             List<String> osRelease = FileUtil.readFile("/etc/lsb-release");
545             // Search for NAME=
546             for (String line : osRelease) {
547                 if (line.startsWith("DISTRIB_DESCRIPTION=")) {
548                     LOG.debug("lsb-release: {}", line);
549                     line = line.replace("DISTRIB_DESCRIPTION=", "").replaceAll("^\"|\"$", "").trim();
550                     if (line.contains(" release ")) {
551                         this.family = parseRelease(line, " release ");
552                     }
553                 } else if (line.startsWith("DISTRIB_ID=") && this.family == null) {
554                     LOG.debug("lsb-release: {}", line);
555                     this.family = line.replace("DISTRIB_ID=", "").replaceAll("^\"|\"$", "").trim();
556                 } else if (line.startsWith("DISTRIB_RELEASE=") && this.versionId == null) {
557                     LOG.debug("lsb-release: {}", line);
558                     this.versionId = line.replace("DISTRIB_RELEASE=", "").replaceAll("^\"|\"$", "").trim();
559                 } else if (line.startsWith("DISTRIB_CODENAME=") && this.codeName == null) {
560                     LOG.debug("lsb-release: {}", line);
561                     this.codeName = line.replace("DISTRIB_CODENAME=", "").replaceAll("^\"|\"$", "").trim();
562                 }
563             }
564         }
565         return this.family != null;
566     }

568     /**
569      * Attempts to read /etc/distrib-release (for some value of distrib)
570      *
571      * @return true if file successfully read and " release " or " VERSION "
572      *         found
573      */
574     private boolean readDistribRelease(String filename) {
575         if (new File(filename).exists()) {
576             List<String> osRelease = FileUtil.readFile(filename);
577             // Search for Distrib release x.x (Codename)
578             for (String line : osRelease) {
579                 LOG.debug("{}: {}", filename, line);
580                 if (line.contains(" release ")) {
581                     this.family = parseRelease(line, " release ");
582                     // If this parses properly we're done
583                     break;
584                 } else if (line.contains(" VERSION ")) {
585                     this.family = parseRelease(line, " VERSION ");
586                     // If this parses properly we're done
587                     break;
588                 }
589             }
590         }
591         return this.family != null;
592     }

594     /**
595      * Helper method to parse version description line style
596      *
597      * @param line
598      *            a String of the form "Distributor release x.x (Codename)"
599      * @param splitLine
600      *            A regex to split on, e.g. " release "
601      * @return the parsed family (versionID and codeName may have also been set)
602      */
603     private String parseRelease(String line, String splitLine) {
604         String[] split = line.split(splitLine);
605         String family = split[0].trim();
606         if (split.length > 1) {
607             split = split[1].split("[()]");
608             if (split.length > 0) {
609                 this.versionId = split[0].trim();
610             }
611             if (split.length > 1) {
612                 this.codeName = split[1].trim();
613             }
614         }
615         return family;
616     }

618     /**
619      * Looks for a collection of possible distrib-release filenames
620      *
621      * @return The first valid matching filename
622      */
623     protected static String getReleaseFilename() {
624         // Look for any /etc/*-release, *-version, and variants
625         File etc = new File("/etc");
626         // Find any *_input files in that path
627         File[] matchingFiles = etc.listFiles(new FileFilter() {
628             @Override
629             public boolean accept(File f) {
630                 return (f.getName().endsWith("-release") || f.getName().endsWith("-version")
631                         || f.getName().endsWith("_release") || f.getName().endsWith("_version"))
632                         && !(f.getName().endsWith("os-release") || f.getName().endsWith("lsb-release"));
633             }
634         });
635         if (matchingFiles != null && matchingFiles.length > 0) {
636             return matchingFiles[0].getPath();
637         }
638         if (new File("/etc/release").exists()) {
639             return "/etc/release";
640         }
641         // If all else fails, try this
642         return "/etc/issue";
643     }
