124     // Get a list of orders to pass to ParseUtil
125     private static final int[] PROC_PID_STAT_ORDERS = new int[ProcPidStat.values().length];
126     static {
127         for (ProcPidStat stat : ProcPidStat.values()) {
128             // The PROC_PID_STAT enum indices are 1-indexed.
129             // Subtract one to get a zero-based index
130             PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder() - 1;
131         }
132     }

134     // 2.6 Kernel has 44 elements, 3.3 has 47, and 3.5 has 52.
135     // Check /proc/self/stat to find its length
136     private static final int PROC_PID_STAT_LENGTH;
137     static {
138         String stat = FileUtil.getStringFromFile(ProcPath.SELF_STAT);
139         if (!stat.isEmpty() && stat.contains(")")) {
140             // add 3 to account for pid, process name in prarenthesis, and state
141             PROC_PID_STAT_LENGTH = ParseUtil.countStringToLongArray(stat, ' ') + 3;
142         } else {
143             // Default assuming recent kernel
144             PROC_PID_STAT_LENGTH = 52;
145         }
146     }

148     // Jiffies per second, used for process time counters.
149     private static final long USER_HZ = ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf CLK_TCK"),
150             100L);
151     // Boot time in MS.
152     private static final long BOOT_TIME;
153     static {
154         // Uptime is only in hundredths of seconds but we need thousandths.
155         // We can grab uptime twice and take average to reduce error, getting
156         // current time in between
157         double uptime = UpTime.getSystemUptimeSeconds();
158         long now = System.currentTimeMillis();
159         uptime += UpTime.getSystemUptimeSeconds();
160         // Uptime is now 2x seconds, so divide by 2, but
161         // we want milliseconds so multiply by 1000
162         // Ultimately multiply by 1000/2 = 500
163         BOOT_TIME = now - (long) (500d * uptime + 0.5);
164         // Cast/truncation is effectively rounding. Boot time could
165         // be +/- 5 ms due to System Uptime rounding to nearest 10ms.
166     }

258     private OSProcess getProcess(int pid, LinuxUserGroupInfo userGroupInfo, boolean slowFields) {
259         String path = "";
260         String procPidExe = String.format(ProcPath.PID_EXE, pid);
261         try {
262             Path link = Paths.get(procPidExe);
263             path = Files.readSymbolicLink(link).toString();
264             int index = path.indexOf(" (deleted)");
265             if (index != -1) {
266                 path = path.substring(0, index);
267             }
268         } catch (InvalidPathException | IOException | UnsupportedOperationException | SecurityException e) {
269             LOG.debug("Unable to open symbolic link {}", procPidExe);
270         }
271         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_IO, pid), ":");
272         // See man proc for how to parse /proc/[pid]/stat
273         long now = System.currentTimeMillis();
274         String stat = FileUtil.getStringFromFile(String.format(ProcPath.PID_STAT, pid));
275         // A race condition may leave us with an empty string
276         if (stat.isEmpty()) {
277             return null;
278         }
279         // We can get name and status more easily from /proc/pid/status which we
280         // call later, so just get the numeric bits here
281         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
282         // Fetch cached process if it exists
283         OSProcessml#OSProcess">OSProcess proc = new OSProcess(this);
284         proc.setProcessID(pid);
285         // The /proc/pid/cmdline value is null-delimited
286         proc.setCommandLine(FileUtil.getStringFromFile(String.format(ProcPath.PID_CMDLINE, pid)));
287         // BOOTTIME is in seconds and start time from proc/pid/stat is in jiffies.
288         // Combine units to jiffies and convert to millijiffies before hz division to
289         // avoid precision loss without having to cast
290         long startTime = (BOOT_TIME * USER_HZ + statArray[ProcPidStat.START_TIME.ordinal()]) * 1000L / USER_HZ;
291         // BOOT_TIME could be up to 500ms off and start time up to 5ms off. A process
292         // that has started within last 505ms could produce a future start time/negative
293         // up time, so insert a sanity check.
294         if (startTime >= now) {
295             startTime = now - 1;
296         }
297         proc.setStartTime(startTime);
298         proc.setParentProcessID((int) statArray[ProcPidStat.PPID.ordinal()]);
299         proc.setThreadCount((int) statArray[ProcPidStat.THREAD_COUNT.ordinal()]);
300         proc.setPriority((int) statArray[ProcPidStat.PRIORITY.ordinal()]);
301         proc.setVirtualSize(statArray[ProcPidStat.VSZ.ordinal()]);
302         proc.setResidentSetSize(statArray[ProcPidStat.RSS.ordinal()] * this.memoryPageSize);
303         proc.setKernelTime(statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / USER_HZ);
304         proc.setUserTime(statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / USER_HZ);
305         proc.setUpTime(now - proc.getStartTime());
306         // See man proc for how to parse /proc/[pid]/io
307         proc.setBytesRead(ParseUtil.parseLongOrDefault(io.getOrDefault("read_bytes", ""), 0L));
308         proc.setBytesWritten(ParseUtil.parseLongOrDefault(io.getOrDefault("write_bytes", ""), 0L));
309 
310         // gets the open files count
311         if (slowFields) {
312             List<String> openFilesList = ExecutingCommand.runNative(String.format(LS_F_PROC_PID_FD, pid));
313             proc.setOpenFiles(openFilesList.size() - 1L);
314 
315             // get 5th byte of file for 64-bit check
316             // https://en.wikipedia.org/wiki/Executable_and_Linkable_Format#File_header
317             byte[] buffer = new byte[5];
318             if (!path.isEmpty()) {
319                 try (InputStream is = new FileInputStream(path)) {
320                     if (is.read(buffer) == buffer.length) {
321                         proc.setBitness(buffer[4] == 1 ? 32 : 64);
322                     }
323                 } catch (IOException e) {
324                     LOG.warn("Failed to read process file: {}", path);
325                 }
326             }
327         }
328 
329         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_STATUS, pid), ":");
330         proc.setName(status.getOrDefault("Name", ""));
331         proc.setPath(path);
332         switch (status.getOrDefault("State", "U").charAt(0)) {
333         case 'R':
334             proc.setState(OSProcess.State.RUNNING);
335             break;
336         case 'S':
337             proc.setState(OSProcess.State.SLEEPING);
338             break;
339         case 'D':
340             proc.setState(OSProcess.State.WAITING);
341             break;
342         case 'Z':
343             proc.setState(OSProcess.State.ZOMBIE);
344             break;
345         case 'T':
346             proc.setState(OSProcess.State.STOPPED);
347             break;
348         default:
349             proc.setState(OSProcess.State.OTHER);
350             break;
351         }
352         proc.setUserID(ParseUtil.whitespaces.split(status.getOrDefault("Uid", ""))[0]);
353         proc.setGroupID(ParseUtil.whitespaces.split(status.getOrDefault("Gid", ""))[0]);
354         Pair<String, String> user = userGroupInfo.getUser(proc.getUserID());
355         proc.setUser(user.getB());
356         proc.setGroup(userGroupInfo.getGroupName(proc.getGroupID()));
357 
358         try {
359             String cwdLink = String.format(ProcPath.PID_CWD, pid);
360             String cwd = new File(cwdLink).getCanonicalPath();
361             if (!cwd.equals(cwdLink)) {
362                 proc.setCurrentWorkingDirectory(cwd);
363             }
364         } catch (IOException e) {
365             LOG.trace("Couldn't find cwd for pid {}: {}", pid, e.getMessage());
366         }
367         return proc;
368     }

384             }
385         }
386         List<OSProcess> sorted = processSort(procs, limit, sort);
387         return sorted.toArray(new OSProcess[0]);

390     private static int getParentPidFromProcFile(int pid) {
391         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
392         // A race condition may leave us with an empty string
393         if (stat.isEmpty()) {
394             return 0;
395         }
396         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
397         return (int) statArray[ProcPidStat.PPID.ordinal()];
398     }

400     @Override
401     public long getProcessAffinityMask(int processId) {
402         // Would prefer to use native sched_getaffinity call but variable sizing is
403         // kernel-dependent and requires C macros, so we use command line instead.
404         String mask = ExecutingCommand.getFirstAnswer("taskset -p " + processId);
405         // Output:
406         // pid 3283's current affinity mask: 3
407         // pid 9726's current affinity mask: f
408         String[] split = ParseUtil.whitespaces.split(mask);
409         try {
410             return new BigInteger(split[split.length - 1], 16).longValue();
411         } catch (NumberFormatException e) {
412             return 0;
413         }
414     }

456     private static Triplet<String, String, String> queryFamilyVersionCodenameFromReleaseFiles() {
457         Triplet<String, String, String> familyVersionCodename;
458         // There are two competing options for family/version information.
459         // Newer systems are adopting a standard /etc/os-release file:
460         // https://www.freedesktop.org/software/systemd/man/os-release.html
461         //
462         // Some systems are still using the lsb standard which parses a
463         // variety of /etc/*-release files and is most easily accessed via
464         // the commandline lsb_release -a, see here:
465         // https://linux.die.net/man/1/lsb_release
466         // In this case, the /etc/lsb-release file (if it exists) has
467         // optional overrides to the information in the /etc/distrib-release
468         // files, which show: "Distributor release x.x (Codename)"
469 
470         // Attempt to read /etc/system-release which has more details than
471         // os-release on (CentOS and Fedora)
472         if ((familyVersionCodename = readDistribRelease("/etc/system-release")) != null) {
473             // If successful, we're done. this.family has been set and
474             // possibly the versionID and codeName
475             return familyVersionCodename;
476         }
477 
478         // Attempt to read /etc/os-release file.
479         if ((familyVersionCodename = readOsRelease()) != null) {
480             // If successful, we're done. this.family has been set and
481             // possibly the versionID and codeName
482             return familyVersionCodename;
483         }
484 
485         // Attempt to execute the `lsb_release` command
486         if ((familyVersionCodename = execLsbRelease()) != null) {
487             // If successful, we're done. this.family has been set and
488             // possibly the versionID and codeName
489             return familyVersionCodename;
490         }
491 
492         // The above two options should hopefully work on most
493         // distributions. If not, we keep having fun.
494         // Attempt to read /etc/lsb-release file
495         if ((familyVersionCodename = readLsbRelease()) != null) {
496             // If successful, we're done. this.family has been set and
497             // possibly the versionID and codeName
498             return familyVersionCodename;
499         }
500 
501         // If we're still looking, we search for any /etc/*-release (or
502         // similar) filename, for which the first line should be of the
503         // "Distributor release x.x (Codename)" format or possibly a
504         // "Distributor VERSION x.x (Codename)" format
505         String etcDistribRelease = getReleaseFilename();
506         if ((familyVersionCodename = readDistribRelease(etcDistribRelease)) != null) {
507             // If successful, we're done. this.family has been set and
508             // possibly the versionID and codeName
509             return familyVersionCodename;
510         }
511         // If we've gotten this far with no match, use the distrib-release
512         // filename (defaults will eventually give "Unknown")
513         String family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
514                 .replace("version", "").replace("-", "").replace("_", ""));
515         return new Triplet<>(family, Constants.UNKNOWN, Constants.UNKNOWN);
516     }

518     /**
519      * Attempts to read /etc/os-release
520      *
521      * @return a triplet with the parsed family, versionID and codeName if file
522      *         successfully read and NAME= found, null otherwise
523      */
524     private static Triplet<String, String, String> readOsRelease() {
525         String family = null;
526         String versionId = Constants.UNKNOWN;
527         String codeName = Constants.UNKNOWN;
528         List<String> osRelease = FileUtil.readFile("/etc/os-release");
529         // Search for NAME=
530         for (String line : osRelease) {
531             if (line.startsWith("VERSION=")) {
532                 LOG.debug(OS_RELEASE_LOG, line);
533                 // remove beginning and ending '"' characters, etc from
534                 // VERSION="14.04.4 LTS, Trusty Tahr" (Ubuntu style)
535                 // or VERSION="17 (Beefy Miracle)" (os-release doc style)
536                 line = line.replace("VERSION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
537                 String[] split = line.split("[()]");
538                 if (split.length <= 1) {
539                     // If no parentheses, check for Ubuntu's comma format
540                     split = line.split(", ");
541                 }
542                 if (split.length > 0) {
543                     versionId = split[0].trim();
544                 }
545                 if (split.length > 1) {
546                     codeName = split[1].trim();
547                 }
548             } else if (line.startsWith("NAME=") && family == null) {
549                 LOG.debug(OS_RELEASE_LOG, line);
550                 // remove beginning and ending '"' characters, etc from
551                 // NAME="Ubuntu"
552                 family = line.replace("NAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
553             } else if (line.startsWith("VERSION_ID=") && versionId.equals(Constants.UNKNOWN)) {
554                 LOG.debug(OS_RELEASE_LOG, line);
555                 // remove beginning and ending '"' characters, etc from
556                 // VERSION_ID="14.04"
557                 versionId = line.replace("VERSION_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
558             }
559         }
560         return family == null ? null : new Triplet<>(family, versionId, codeName);
561     }

563     /**
564      * Attempts to execute `lsb_release -a`
565      *
566      * @return a triplet with the parsed family, versionID and codeName if the
567      *         command successfully executed and Distributor ID: or Description:
568      *         found, null otherwise
569      */
570     private static Triplet<String, String, String> execLsbRelease() {
571         String family = null;
572         String versionId = Constants.UNKNOWN;
573         String codeName = Constants.UNKNOWN;
574         // If description is of the format Distrib release x.x (Codename)
575         // that is primary, otherwise use Distributor ID: which returns the
576         // distribution concatenated, e.g., RedHat instead of Red Hat
577         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
578             if (line.startsWith("Description:")) {
579                 LOG.debug(LSB_RELEASE_A_LOG, line);
580                 line = line.replace("Description:", "").trim();
581                 if (line.contains(RELEASE_DELIM)) {
582                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
583                     family = triplet.getA();
584                     if (versionId.equals(Constants.UNKNOWN)) {
585                         versionId = triplet.getB();
586                     }
587                     if (codeName.equals(Constants.UNKNOWN)) {
588                         codeName = triplet.getC();
589                     }

671     /**
672      * Helper method to parse version description line style
673      *
674      * @param line
675      *            a String of the form "Distributor release x.x (Codename)"
676      * @param splitLine
677      *            A regex to split on, e.g. " release "
678      * @return a triplet with the parsed family, versionID and codeName
679      */
680     private static Triplet<String, String, String> parseRelease(String line, String splitLine) {
681         String[] split = line.split(splitLine);
682         String family = split[0].trim();
683         String versionId = Constants.UNKNOWN;
684         String codeName = Constants.UNKNOWN;
685         if (split.length > 1) {
686             split = split[1].split("[()]");
687             if (split.length > 0) {
688                 versionId = split[0].trim();
689             }
690             if (split.length > 1) {
691                 codeName = split[1].trim();
692             }
693         }
694         return new Triplet<>(family, versionId, codeName);
695     }
