5   package oshi.software.os.linux;
6   
7   import static org.slf4j.event.Level.DEBUG;
8   import static org.slf4j.event.Level.WARN;
9   import static oshi.ffm.ForeignFunctions.callInArenaLongOrDefault;
10  import static oshi.ffm.ForeignFunctions.callInArenaOrDefault;
11  
12  import java.lang.foreign.MemorySegment;
13  import java.lang.foreign.ValueLayout;
14  
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  import oshi.annotation.concurrent.ThreadSafe;
19  import oshi.ffm.platform.linux.LinuxLibcFunctions;
20  import oshi.software.common.os.linux.LinuxOSProcess;
21  import oshi.software.common.os.linux.LinuxOperatingSystem;
22  
23  /**
24   * FFM-based Linux OS process. Implements {@code getrlimit} and {@code getrusage} via FFM.
25   */
26  @ThreadSafe
27  public class LinuxOSProcessFFM extends LinuxOSProcess {
28  
29      private static final Logger LOG = LoggerFactory.getLogger(LinuxOSProcessFFM.class);
30  
31      public LinuxOSProcessFFM(int pid, LinuxOperatingSystem os) {
32          super(pid, os);
33      }
34  
35      @Override
36      protected long[] queryContextSwitches() {
37          return callInArenaOrDefault(arena -> {
38              MemorySegment rusage = arena.allocate(LinuxLibcFunctions.RUSAGE_SIZE);
39              if (0 == LinuxLibcFunctions.getrusage(LinuxLibcFunctions.RUSAGE_SELF, rusage)) {
40                  return new long[] { rusage.get(ValueLayout.JAVA_LONG, LinuxLibcFunctions.RUSAGE_NVCSW_OFFSET),
41                          rusage.get(ValueLayout.JAVA_LONG, LinuxLibcFunctions.RUSAGE_NIVCSW_OFFSET) };
42              }
43              return null;
44          }, LOG, DEBUG, "FFM getrusage failed", null);
45      }
46  
47      @Override
48      protected long queryRlimitSoft() {
49          long limit = callInArenaLongOrDefault(arena -> {
50              MemorySegment rlim = arena.allocate(LinuxLibcFunctions.RLIMIT_LAYOUT);
51              if (0 == LinuxLibcFunctions.getrlimit(LinuxLibcFunctions.RLIMIT_NOFILE, rlim)) {
52                  return LinuxLibcFunctions.rlimitCur(rlim);
53              }
54              return -1L;
55          }, LOG, WARN, "FFM getrlimit failed", -1L);
56          return limit < 0L ? getProcessOpenFileLimit(getProcessID(), 1) : limit;
57      }
58  
59      @Override
60      protected long queryRlimitHard() {
61          long limit = callInArenaLongOrDefault(arena -> {
62              MemorySegment rlim = arena.allocate(LinuxLibcFunctions.RLIMIT_LAYOUT);
63              if (0 == LinuxLibcFunctions.getrlimit(LinuxLibcFunctions.RLIMIT_NOFILE, rlim)) {
64                  return LinuxLibcFunctions.rlimitMax(rlim);
65              }
66              return -1L;
67          }, LOG, WARN, "FFM getrlimit failed", -1L);
68          return limit < 0L ? getProcessOpenFileLimit(getProcessID(), 2) : limit;
69      }
70  }
