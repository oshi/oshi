37  /**
38   * JNA-based Linux operating system implementation. Extends {@link LinuxOperatingSystem}, providing JNA implementations
39   * of process/thread ID queries and thread count via {@code sysinfo}.
40   */
41  @ThreadSafe
42  public class LinuxOperatingSystemJNA extends LinuxOperatingSystem {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(LinuxOperatingSystemJNA.class);
45  
46      /** This static field identifies if the udev library can be loaded. */
47      public static final boolean HAS_UDEV;
48      private static final long USER_HZ;
49      private static final long PAGE_SIZE;
50      /** This static field identifies if the gettid function is in the c library. */
51      public static final boolean HAS_GETTID;
52      /** This static field identifies if the syscall for gettid returns sane results. */
53      public static final boolean HAS_SYSCALL_GETTID;
54  
55      static {
56          long userHz = 100L;
57          long pageSz = 4096L;
58          boolean hasUdev = false;
59          boolean hasGettid = false;
60          boolean hasSyscallGettid = false;
61          try {
62              Map<Integer, Long> auxv = AuxvJNA.queryAuxv();
63              long hz = auxv.getOrDefault(Auxv.AT_CLKTCK, 0L);
64              userHz = hz > 0 ? hz
65                      : ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf CLK_TCK"), 100L);
66              long pagesz = auxv.getOrDefault(Auxv.AT_PAGESZ, 0L);
67              pageSz = pagesz > 0 ? pagesz
68                      : ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf PAGE_SIZE"), 4096L);
69              if (GlobalConfig.get(GlobalConfig.OSHI_OS_LINUX_ALLOWUDEV, true)) {
70                  try {
71                      @SuppressWarnings("unused")
72                      Udev lib = Udev.INSTANCE;
73                      hasUdev = true;
74                  } catch (UnsatisfiedLinkError e) {
75                      LOG.warn("Did not find udev library in operating system. Some features may not work.");
76                  }
77              } else {
78                  LOG.info("Loading of udev not allowed by configuration. Some features may not work.");
79              }
80  
81              try {
82                  LinuxLibc.INSTANCE.gettid();
83                  hasGettid = true;
84              } catch (UnsatisfiedLinkError e) {
85                  LOG.debug("Did not find gettid function in operating system. Using fallbacks.");
86              }
87  
88              hasSyscallGettid = hasGettid;
89              if (!hasGettid) {
90                  try {
91                      hasSyscallGettid = LinuxLibc.INSTANCE.syscall(LinuxLibc.SYS_GETTID).intValue() > 0;
92                  } catch (UnsatisfiedLinkError e) {
93                      LOG.debug("Did not find working syscall gettid function in operating system. Using procfs");
94                  }
95              }
96          } catch (NoClassDefFoundError e) {
97              LOG.error("Did not find JNA classes. Investigate incompatible version or missing native dll.");
98          }
99          HAS_UDEV = hasUdev;
100         HAS_GETTID = hasGettid;
101         HAS_SYSCALL_GETTID = hasSyscallGettid;
102         USER_HZ = userHz;
103         PAGE_SIZE = pageSz;
104     }
105 
106     @Override
107     public List<OSSession> getSessions() {
108         return USE_WHO_COMMAND ? oshi.util.driver.linux.Who.queryWho() : WhoJNA.queryUtxent();
109     }
110 
111     @Override
112     public FileSystem getFileSystem() {
113         return new LinuxFileSystemJNA();
114     }
115 
116     @Override
117     public NetworkParams getNetworkParams() {
118         return new LinuxNetworkParamsJNA();
119     }
120 
121     @Override
122     public OSProcess getProcess(int pid) {
123         OSProcess proc = createOSProcess(pid);
124         if (!proc.getState().equals(State.INVALID)) {
125             return proc;
126         }
127         return null;
128     }
129 
130     @Override
131     protected OSProcess createOSProcess(int pid) {
132         return new LinuxOSProcessJNA(pid, this);
133     }
134 
135     @Override
136     public int getProcessId() {
137         return LinuxLibc.INSTANCE.getpid();
138     }
139 
140     @Override
141     public int getThreadId() {
142         if (HAS_SYSCALL_GETTID) {
143             return HAS_GETTID ? LinuxLibc.INSTANCE.gettid()
144                     : LinuxLibc.INSTANCE.syscall(LinuxLibc.SYS_GETTID).intValue();
145         }
146         return ExceptionUtil.getIntOrDefault(
147                 () -> ParseUtil.parseIntOrDefault(
148                         Files.readSymbolicLink(new File(ProcPath.THREAD_SELF).toPath()).getFileName().toString(), 0),
149                 0);
150     }
151 
152     /**
153      * Gets Jiffies per second, useful for converting ticks to milliseconds and vice versa.
154      *
155      * @return Jiffies per second.
156      */
157     public static long hz() {
158         return USER_HZ;
159     }
160 
161     /**
162      * Gets Page Size, for converting memory stats from pages to bytes.
163      *
164      * @return Page Size in bytes.
165      */
166     public static long pageSize() {
167         return PAGE_SIZE;
168     }
169 
170     @Override
171     public long getHz() {
172         return USER_HZ;
173     }
174 
175     @Override
176     public long getPageSize() {
177         return PAGE_SIZE;
178     }
179 
180     @Override
181     public int getThreadCount() {
182         try (CloseableSysinfo info = new CloseableSysinfo()) {
183             if (0 != LibC.INSTANCE.sysinfo(info)) {
184                 LOG.error("Failed to get process thread count. Error code: {}", Native.getLastError());
185                 return 0;
186             }
187             return Short.toUnsignedInt(info.procs);
188         } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
189             LOG.error("Failed to get procs from sysinfo. {}", e.getMessage());
190         }
191         return 0;
192     }
