68      // Get a list of orders to pass to ParseUtil
69      private static final int[] PROC_PID_STAT_ORDERS = new int[ProcPidStat.values().length];
70      static {
71          for (ProcPidStat stat : ProcPidStat.values()) {
72              // The PROC_PID_STAT enum indices are 1-indexed.
73              // Subtract one to get a zero-based index
74              PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder() - 1;
75          }
76      }

151     @Override
152     public String getCurrentWorkingDirectory() {
153         try {
154             String cwdLink = String.format(ProcPath.PID_CWD, getProcessID());
155             String cwd = new File(cwdLink).getCanonicalPath();
156             if (!cwd.equals(cwdLink)) {
157                 return cwd;
158             }
159         } catch (IOException e) {
160             LOG.trace("Couldn't find cwd for pid {}: {}", getProcessID(), e.getMessage());
161         }
162         return "";
163     }

276     private int queryBitness() {
277         // get 5th byte of file for 64-bit check
278         // https://en.wikipedia.org/wiki/Executable_and_Linkable_Format#File_header
279         byte[] buffer = new byte[5];
280         if (!path.isEmpty()) {
281             try (InputStream is = new FileInputStream(path)) {
282                 if (is.read(buffer) == buffer.length) {
283                     return buffer[4] == 1 ? 32 : 64;
284                 }
285             } catch (IOException e) {
286                 LOG.warn("Failed to read process file: {}", path);
287             }
288         }
289         return 0;
290     }

292     @Override
293     public long getAffinityMask() {
294         // Would prefer to use native sched_getaffinity call but variable sizing is
295         // kernel-dependent and requires C macros, so we use command line instead.
296         String mask = ExecutingCommand.getFirstAnswer("taskset -p " + getProcessID());
297         // Output:
298         // pid 3283's current affinity mask: 3
299         // pid 9726's current affinity mask: f
300         String[] split = ParseUtil.whitespaces.split(mask);
301         try {
302             return new BigInteger(split[split.length - 1], 16).longValue();
303         } catch (NumberFormatException e) {
304             return 0;
305         }
306     }

308     @Override
309     public boolean updateAttributes() {
310         String procPidExe = String.format(ProcPath.PID_EXE, getProcessID());
311         try {
312             Path link = Paths.get(procPidExe);
313             this.path = Files.readSymbolicLink(link).toString();
314             // For some services the symbolic link process has terminated
315             int index = path.indexOf(" (deleted)");
316             if (index != -1) {
317                 path = path.substring(0, index);
318             }
319         } catch (InvalidPathException | IOException | UnsupportedOperationException | SecurityException e) {
320             LOG.debug("Unable to open symbolic link {}", procPidExe);
321         }
322         // Fetch all the values here
323         // check for terminated process race condition after last one.
324         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_IO, getProcessID()), ":");
325         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_STATUS, getProcessID()),
326                 ":");
327         String stat = FileUtil.getStringFromFile(String.format(ProcPath.PID_STAT, getProcessID()));
328         if (stat.isEmpty()) {
329             this.state = INVALID;
330             return false;
331         }
332         // If some details couldn't be read from ProcPath.PID_STATUS try reading it from
333         // ProcPath.PID_STAT
334         getMissingDetails(status, stat);
335 
336         long now = System.currentTimeMillis();
337 
338         // We can get name and status more easily from /proc/pid/status which we
339         // call later, so just get the numeric bits here
340         // See man proc for how to parse /proc/[pid]/stat
341         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS,
342                 ProcessStat.PROC_PID_STAT_LENGTH, ' ');
343 
344         // BOOTTIME is in seconds and start time from proc/pid/stat is in jiffies.
345         // Combine units to jiffies and convert to millijiffies before hz division to
346         // avoid precision loss without having to cast
347         this.startTime = (LinuxOperatingSystem.BOOTTIME * LinuxOperatingSystem.getHz()
348                 + statArray[ProcPidStat.START_TIME.ordinal()]) * 1000L / LinuxOperatingSystem.getHz();
349         // BOOT_TIME could be up to 500ms off and start time up to 5ms off. A process
350         // that has started within last 505ms could produce a future start time/negative
351         // up time, so insert a sanity check.
352         if (startTime >= now) {
353             startTime = now - 1;
354         }
355         this.parentProcessID = (int) statArray[ProcPidStat.PPID.ordinal()];
356         this.threadCount = (int) statArray[ProcPidStat.THREAD_COUNT.ordinal()];
357         this.priority = (int) statArray[ProcPidStat.PRIORITY.ordinal()];
358         this.virtualSize = statArray[ProcPidStat.VSZ.ordinal()];
359         this.residentSetSize = statArray[ProcPidStat.RSS.ordinal()] * PAGE_SIZE;
360         this.kernelTime = statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / LinuxOperatingSystem.getHz();
361         this.userTime = statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / LinuxOperatingSystem.getHz();
362         this.minorFaults = statArray[ProcPidStat.MINOR_FAULTS.ordinal()];
363         this.majorFaults = statArray[ProcPidStat.MAJOR_FAULTS.ordinal()];
364         long nonVoluntaryContextSwitches = ParseUtil.parseLongOrDefault(status.get("nonvoluntary_ctxt_switches"), 0L);
365         long voluntaryContextSwitches = ParseUtil.parseLongOrDefault(status.get("voluntary_ctxt_switches"), 0L);
366         this.contextSwitches = voluntaryContextSwitches + nonVoluntaryContextSwitches;
367 
368         this.upTime = now - startTime;
369 
370         // See man proc for how to parse /proc/[pid]/io
371         this.bytesRead = ParseUtil.parseLongOrDefault(io.getOrDefault("read_bytes", ""), 0L);
372         this.bytesWritten = ParseUtil.parseLongOrDefault(io.getOrDefault("write_bytes", ""), 0L);
373 
374         // Don't set open files or bitness or currentWorkingDirectory; fetch on demand.
375 
376         this.userID = ParseUtil.whitespaces.split(status.getOrDefault("Uid", ""))[0];
377         this.user = UserGroupInfo.getUser(userID);
378         this.groupID = ParseUtil.whitespaces.split(status.getOrDefault("Gid", ""))[0];
379         this.group = UserGroupInfo.getGroupName(groupID);
380         this.name = status.getOrDefault("Name", "");
381         this.state = ProcessStat.getState(status.getOrDefault("State", "U").charAt(0));
382         return true;
383     }

385     /**
386      * If some details couldn't be read from ProcPath.PID_STATUS try reading it from
387      * ProcPath.PID_STAT
388      *
389      * @param status
390      *            status map to fill.
391      * @param stat
392      *            string to read from.
393      */
394     private static void getMissingDetails(Map<String, String> status, String stat) {
395         if (status == null || stat == null) {
396             return;
397         }
398 
399         int nameStart = stat.indexOf('(');
400         int nameEnd = stat.indexOf(')');
401         if (Util.isBlank(status.get("Name")) && nameStart > 0 && nameStart < nameEnd) {
402             // remove leading and trailing parentheses
403             String statName = stat.substring(nameStart + 1, nameEnd);
404             status.put("Name", statName);
405         }
406 
407         // As per man, the next item after the name is the state
408         if (Util.isBlank(status.get("State")) && nameEnd > 0 && stat.length() > nameEnd + 2) {
409             String statState = String.valueOf(stat.charAt(nameEnd + 2));
410             status.put("State", statState);
411         }
412     }

414     /**
415      * Enum used to update attributes. The order field represents the 1-indexed
416      * numeric order of the stat in /proc/pid/stat per the man file.
417      */
418     private enum ProcPidStat {
419         // The parsing implementation in ParseUtil requires these to be declared
420         // in increasing order
421         PPID(4), MINOR_FAULTS(10), MAJOR_FAULTS(12), USER_TIME(14), KERNEL_TIME(15), PRIORITY(18), THREAD_COUNT(20),
422         START_TIME(22), VSZ(23), RSS(24);
423 
424         private final int order;
425 
426         public int getOrder() {
427             return this.order;
428         }
429 
430         ProcPidStat(int order) {
431             this.order = order;
432         }
433     }
