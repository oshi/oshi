17  package oshi.software.os.linux;

19  import java.io.IOException;
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;

26  import oshi.hardware.HardwareAbstractionLayer;

37  /**
38   * @author alessandro[at]perucchi[dot]org
39   */
40  
41  public class LinuxHardwareAbstractionLayer implements HardwareAbstractionLayer {
42  	private static final Logger LOG = LoggerFactory.getLogger(LinuxHardwareAbstractionLayer.class);
43  
44  	private static final String SEPARATOR = "\\s+:\\s";
45  
46  	private Processor[] _processors = null;
47  
48  	private Memory _memory = null;
49  
50  	@Override
51  	public Memory getMemory() {
52  		if (this._memory == null) {
53  			this._memory = new GlobalMemory();
54  		}
55  		return this._memory;
56  	}
57  
58  	@Override
59  	public Processor[] getProcessors() {
60  
61  		if (this._processors == null) {
62  			List<Processor> processors = new ArrayList<>();
63  			List<String> cpuInfo = null;
64  			try {
65  				cpuInfo = FileUtil.readFile("/proc/cpuinfo");
66  			} catch (IOException e) {
67  				LOG.error("Problem with /proc/cpuinfo: {}", e.getMessage());
68  				return null;
69  			}
70  			CentralProcessor cpu = null;
71  			int numCPU = 0;
72  			for (String toBeAnalyzed : cpuInfo) {
73  				if (toBeAnalyzed.equals("")) {
74  					if (cpu != null) {
75  						processors.add(cpu);
76  					}
77  					cpu = null;
78  					continue;
79  				}
80  				if (cpu == null) {
81  					cpu = new CentralProcessor(numCPU++);
82  				}
83  				if (toBeAnalyzed.startsWith("model name\t")) {
84  					cpu.setName(toBeAnalyzed.split(SEPARATOR)[1]);
85  					continue;
86  				}
87  				if (toBeAnalyzed.startsWith("flags\t")) {
88  					String[] flags = toBeAnalyzed.split(SEPARATOR)[1].split(" ");
89  					boolean found = false;
90  					for (String flag : flags) {
91  						if (flag.equalsIgnoreCase("LM")) {
92  							found = true;
93  							break;
94  						}
95  					}
96  					cpu.setCpu64(found);
97  					continue;
98  				}
99  				if (toBeAnalyzed.startsWith("cpu family\t")) {
100 					cpu.setFamily(toBeAnalyzed.split(SEPARATOR)[1]);
101 					continue;
102 				}
103 				if (toBeAnalyzed.startsWith("model\t")) {
104 					cpu.setModel(toBeAnalyzed.split(SEPARATOR)[1]);
105 					continue;
106 				}
107 				if (toBeAnalyzed.startsWith("stepping\t")) {
108 					cpu.setStepping(toBeAnalyzed.split(SEPARATOR)[1]);
109 					continue;
110 				}
111 				if (toBeAnalyzed.startsWith("vendor_id")) {
112 					cpu.setVendor(toBeAnalyzed.split(SEPARATOR)[1]);
113 					continue;
114 				}
115 			}
116 			if (cpu != null) {
117 				processors.add(cpu);
118 			}
119 			this._processors = processors.toArray(new Processor[0]);
120 		}
121 
122 		return this._processors;
123 	}
124 
125 	@Override
126 	public PowerSource[] getPowerSources() {
127 		return LinuxPowerSource.getPowerSources();
128 	}
129 
130 	@Override
131 	public OSFileStore[] getFileStores() {
132 		return LinuxFileSystem.getFileStores();
133 	}
134 
135 }
