93  	/**
94  	 * Vendor frequency (in Hz), eg. for processor named Intel(R) Core(TM)2 Duo
95  	 * CPU T7300 @ 2.00GHz the vendor frequency is 2000000000.
96  	 * 
97  	 * @return Processor frequency or -1 if unknown.
98  	 * 
99  	 * @author alessio.fachechi[at]gmail[dot]com
100 	 */
101 	public long getVendorFreq() {
102 		if (_freq == null) {
103 			Pattern pattern = Pattern.compile("@ (.*)$");
104 			Matcher matcher = pattern.matcher(getName());
105 
106 			if (matcher.find()) {
107 				String unit = matcher.group(1);
108 				_freq = ParseUtil.parseHertz(unit);
109 			} else {
110 				_freq = -1L;
111 			}
112 		}
113 
114 		return _freq.longValue();
115 	}

127 	/**
128 	 * Identifier, eg. x86 Family 6 Model 15 Stepping 10.
129 	 * 
130 	 * @return Processor identifier.
131 	 */
132 	public String getIdentifier() {
133 		if (_identifier == null) {
134 			StringBuilder sb = new StringBuilder();
135 			if (getVendor().contentEquals("GenuineIntel"))
136 				sb.append(isCpu64bit() ? "Intel64" : "x86");
137 			else
138 				sb.append(getVendor());
139 			sb.append(" Family ");
140 			sb.append(getFamily());
141 			sb.append(" Model ");
142 			sb.append(getModel());
143 			sb.append(" Stepping ");
144 			sb.append(getStepping());
145 			_identifier = sb.toString();
146 		}
147 		return _identifier;
148 	}

227 	@Deprecated
228 	public float getLoad() {
229 		long[] prevTicks = getCpuLoadTicks();
230 		try {
231 			Thread.sleep(1000);
232 		} catch (InterruptedException e) {
233 			// Awake, O sleeper
234 		}
235 		long[] ticks = getCpuLoadTicks();
236 		long total = 0;
237 		for (int i = 0; i < ticks.length; i++) {
238 			total += (ticks[i] - prevTicks[i]);
239 		}
240 		long idle = ticks[ticks.length - 1] - prevTicks[ticks.length - 1];
241 		if (total > 0 && idle >= 0)
242 			return 100f * (total - idle) / total;
243 		else
244 			return 0f;
245 	}

247 	public long[] getCpuLoadTicks() {
248 		// /proc/stat expected format
249 		// first line is overall user,nice,system,idle, etc.
250 		// cpu 3357 0 4313 1362393 ...
251 		// TODO: per-processor subsequent lines for cpu0, cpu1, etc.
252 		long[] ticks = new long[4];
253 		String tickStr = "";
254 		try {
255 			List<String> procStat = FileUtil.readFile("/proc/stat");
256 			if (!procStat.isEmpty())
257 				tickStr = procStat.get(0);
258 		} catch (IOException e) {
259 			System.err.println("Problem with: /proc/stat");
260 			System.err.println(e.getMessage());
261 			return ticks;
262 		}
263 		String[] tickArr = tickStr.split("\\s+");
264 		if (tickStr.length() < 5)
265 			return ticks;
266 		for (int i = 0; i < 4; i++) {
267 			ticks[i] = Long.parseLong(tickArr[i + 1]);
268 		}
269 		return ticks;
270 	}
