24  package oshi.software.os;
25  
26  import oshi.annotation.concurrent.Immutable;
27  
28  /**
29   * Operating system services are responsible for the management of platform
30   * resources, including the processor, memory, files, and input and output. They
31   * generally shield applications from the implementation details of the machine.
32   * <p>
33   * This class is provided for information purposes only. Interpretation of the
34   * meaning of services is platform-dependent.
35   */
36  @Immutable
37  public class OSService {
38  
39      private final String name;
40      private final int processID;
41      private final State state;
42  
43      /**
44       * Service Execution States
45       */
46      public enum State {
47          RUNNING, STOPPED, OTHER
48      }
49  
50      /**
51       * Instantiate a new {@link OSService}.
52       *
53       * @param name
54       *            The service name.
55       * @param processID
56       *            The process ID if running, or 0 if stopped.
57       * @param state
58       *            The service {@link State}.
59       */
60      public OSService(String name, int processID, State state) {
61          this.name = name;
62          this.processID = processID;
63          this.state = state;
64      }
65  
66      /**
67       * <p>
68       * Getter for the field <code>name</code>.
69       * </p>
70       *
71       * @return Returns the name of the service.
72       */
73      public String getName() {
74          return this.name;
75      }
76  
77      /**
78       * <p>
79       * Getter for the field <code>processID</code>.
80       * </p>
81       *
82       * @return Returns the processID.
83       */
84      public int getProcessID() {
85          return this.processID;
86      }
87  
88      /**
89       * <p>
90       * Getter for the field <code>state</code>.
91       * </p>
92       *
93       * @return Returns the state of the service.
94       */
95      public State getState() {
96          return this.state;
97      }
98  
99  }
