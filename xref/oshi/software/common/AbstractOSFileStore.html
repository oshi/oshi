5   package oshi.software.common;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.os.OSFileStore;
9   
10  /**
11   * Common implementations for OSFileStore
12   */
13  @ThreadSafe
14  public abstract class AbstractOSFileStore implements OSFileStore {
15  
16      private String name;
17      private String volume;
18      private String label;
19      private String mount;
20      private String options;
21      private String uuid;
22      private boolean local;
23  
24      protected AbstractOSFileStore(String name, String volume, String label, String mount, String options, String uuid,
25              boolean local) {
26          this.name = name;
27          this.volume = volume;
28          this.label = label;
29          this.mount = mount;
30          this.options = options;
31          this.uuid = uuid;
32          this.local = local;
33      }
34  
35      protected AbstractOSFileStore() {
36      }
37  
38      @Override
39      public String getName() {
40          return this.name;
41      }
42  
43      @Override
44      public String getVolume() {
45          return this.volume;
46      }
47  
48      @Override
49      public String getLabel() {
50          return this.label;
51      }
52  
53      @Override
54      public String getMount() {
55          return this.mount;
56      }
57  
58      @Override
59      public String getOptions() {
60          return options;
61      }
62  
63      @Override
64      public String getUUID() {
65          return this.uuid;
66      }
67  
68      @Override
69      public boolean isLocal() {
70          return this.local;
71      }
72  
73      @Override
74      public String toString() {
75          return "OSFileStore [name=" + getName() + ", volume=" + getVolume() + ", label=" + getLabel()
76                  + ", logicalVolume=" + getLogicalVolume() + ", mount=" + getMount() + ", description="
77                  + getDescription() + ", fsType=" + getType() + ", options=\"" + getOptions() + "\", uuid=" + getUUID()
78                  + ", isLocal=" + isLocal() + ", freeSpace=" + getFreeSpace() + ", usableSpace=" + getUsableSpace()
79                  + ", totalSpace=" + getTotalSpace() + ", freeInodes=" + getFreeInodes() + ", totalInodes="
80                  + getTotalInodes() + "]";
81      }
82  }
