5   package oshi.software.common.os.linux;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.common.AbstractOSFileStore;
9   import oshi.software.os.OSFileStore;
10  
11  /**
12   * OSFileStore implementation
13   */
14  @ThreadSafe
15  public class LinuxOSFileStore extends AbstractOSFileStore {
16  
17      private final LinuxFileSystem fs;
18  
19      /**
20       * Creates a LinuxOSFileStore.
21       *
22       * @param name          the name
23       * @param volume        the volume
24       * @param label         the label
25       * @param mount         the mount point
26       * @param options       the mount options
27       * @param uuid          the UUID
28       * @param local         whether local
29       * @param logicalVolume the logical volume
30       * @param description   the description
31       * @param fsType        the filesystem type
32       * @param freeSpace     free space in bytes
33       * @param usableSpace   usable space in bytes
34       * @param totalSpace    total space in bytes
35       * @param freeInodes    free inodes
36       * @param totalInodes   total inodes
37       * @param fs            the parent filesystem
38       */
39      public LinuxOSFileStore(String name, String volume, String label, String mount, String options, String uuid,
40              boolean local, String logicalVolume, String description, String fsType, long freeSpace, long usableSpace,
41              long totalSpace, long freeInodes, long totalInodes, LinuxFileSystem fs) {
42          super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
43                  usableSpace, totalSpace, freeInodes, totalInodes);
44          this.fs = fs;
45      }
46  
47      @Override
48      public boolean updateAttributes() {
49          // Fast path: query space/inode stats directly on the known mount point
50          long[] vfs = fs.queryStatvfs(getMount());
51          if (vfs != null) {
52              long ts = vfs[2];
53              long us = vfs[3];
54              long frs = vfs[4];
55              // If native methods failed use JVM methods
56              if (ts == 0L) {
57                  java.io.File f = new java.io.File(getMount());
58                  ts = f.getTotalSpace();
59                  us = f.getUsableSpace();
60                  frs = f.getFreeSpace();
61              }
62              updateSpaceAndInodes(frs, us, ts, vfs[1], vfs[0]);
63              return true;
64          }
65          // Fall back to full enumeration if the direct call failed
66          for (OSFileStore fileStore : fs.getFileStoreMatching(getName(), LinuxFileSystem.buildUuidMap(), isLocal())) {
67              if (getVolume().equals(fileStore.getVolume()) && getMount().equals(fileStore.getMount())) {
68                  updateFrom(fileStore);
69                  return true;
70              }
71          }
72          return false;
73      }
74  }
