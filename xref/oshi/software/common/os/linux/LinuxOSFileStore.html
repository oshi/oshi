5   package oshi.software.common.os.linux;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.common.AbstractOSFileStore;
9   import oshi.software.os.OSFileStore;
10  
11  /**
12   * OSFileStore implementation
13   */
14  @ThreadSafe
15  public class LinuxOSFileStore extends AbstractOSFileStore {
16  
17      private final LinuxFileSystem fs;
18      // True if this store was created for an NFS mount whose server was unreachable during
19      // enumeration. The flag is final, so it is permanent for the life of this object:
20      // updateAttributes() always re-enumerates (re-running the reachability probe) instead of
21      // taking the direct statvfs fast-path, even after the server recovers. To get an object
22      // that uses the fast-path again, re-query the full list via FileSystem.getFileStores().
23      private final boolean unreachable;
24  
25      /**
26       * Creates a LinuxOSFileStore.
27       *
28       * @param name          the name
29       * @param volume        the volume
30       * @param label         the label
31       * @param mount         the mount point
32       * @param options       the mount options
33       * @param uuid          the UUID
34       * @param local         whether local
35       * @param logicalVolume the logical volume
36       * @param description   the description
37       * @param fsType        the filesystem type
38       * @param freeSpace     free space in bytes
39       * @param usableSpace   usable space in bytes
40       * @param totalSpace    total space in bytes
41       * @param freeInodes    free inodes
42       * @param totalInodes   total inodes
43       * @param fs            the parent filesystem
44       */
45      public LinuxOSFileStore(String name, String volume, String label, String mount, String options, String uuid,
46              boolean local, String logicalVolume, String description, String fsType, long freeSpace, long usableSpace,
47              long totalSpace, long freeInodes, long totalInodes, LinuxFileSystem fs) {
48          this(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
49                  usableSpace, totalSpace, freeInodes, totalInodes, fs, false);
50      }
51  
52      LinuxOSFileStore(String name, String volume, String label, String mount, String options, String uuid, boolean local,
53              String logicalVolume, String description, String fsType, long freeSpace, long usableSpace, long totalSpace,
54              long freeInodes, long totalInodes, LinuxFileSystem fs, boolean unreachable) {
55          super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
56                  usableSpace, totalSpace, freeInodes, totalInodes);
57          this.fs = fs;
58          this.unreachable = unreachable;
59      }
60  
61      @Override
62      public boolean updateAttributes() {
63          if (unreachable) {
64              // Skip queryStatvfs — it would hang on a still-stale NFS mount. Delegate to full
65              // enumeration so the reachability guard in getFileStoreMatching() runs again; if the
66              // server is back the returned store will have correct metrics. This object keeps using
67              // this slower path on every refresh (see the unreachable field); re-query the full list
68              // to obtain a store that resumes the direct statvfs fast-path.
69              for (OSFileStore fileStore : fs.getFileStoreMatching(getName(), LinuxFileSystem.buildUuidMap(),
70                      isLocal())) {
71                  if (getVolume().equals(fileStore.getVolume()) && getMount().equals(fileStore.getMount())) {
72                      updateFrom(fileStore);
73                      return true;
74                  }
75              }
76              return false;
77          }
78          // Fast path: query space/inode stats directly on the known mount point
79          long[] vfs = fs.queryStatvfs(getMount());
80          if (vfs != null) {
81              long ts = vfs[2];
82              long us = vfs[3];
83              long frs = vfs[4];
84              // If native methods failed use JVM methods
85              if (ts == 0L) {
86                  java.io.File f = new java.io.File(getMount());
87                  ts = f.getTotalSpace();
88                  us = f.getUsableSpace();
89                  frs = f.getFreeSpace();
90              }
91              updateSpaceAndInodes(frs, us, ts, vfs[1], vfs[0]);
92              return true;
93          }
94          // Fall back to full enumeration if the direct call failed
95          for (OSFileStore fileStore : fs.getFileStoreMatching(getName(), LinuxFileSystem.buildUuidMap(), isLocal())) {
96              if (getVolume().equals(fileStore.getVolume()) && getMount().equals(fileStore.getMount())) {
97                  updateFrom(fileStore);
98                  return true;
99              }
100         }
101         return false;
102     }
103 }
