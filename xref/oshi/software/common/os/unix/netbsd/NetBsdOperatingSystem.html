25  /**
26   * NetBsd is a free and open-source Unix-like operating system descended from the Berkeley Software Distribution (BSD),
27   * which was based on Research Unix.
28   * <p>
29   * The cross-BSD-common pieces live in {@link BsdOperatingSystem}; NetBSD has no native-access split, so this single
30   * class also provides the {@code ps} process enumeration, the text-parsed boot time, the current-thread factory, and
31   * the sysctl version / file-system / network queries.
32   */
33  @ThreadSafe
34  public class NetBsdOperatingSystem extends BsdOperatingSystem {
35  
36      @Override
37      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
38          String family = BsdSysctlUtil.sysctl("kern.ostype", "NetBSD");
39          String version = BsdSysctlUtil.sysctl("kern.osrelease", "");
40          String versionInfo = BsdSysctlUtil.sysctl("kern.version", "");
41          String buildNumber = versionInfo.split(":")[0].replace(family, "").replace(version, "").trim();
42  
43          return new Pair<>(family, new OSVersionInfo(version, null, buildNumber));
44      }
45  
46      @Override
47      public FileSystem getFileSystem() {
48          return new NetBsdFileSystem();
49      }
50  
51      @Override
52      public InternetProtocolStats getInternetProtocolStats() {
53          return new NetBsdInternetProtocolStats();
54      }
55  
56      @Override
57      public NetworkParams getNetworkParams() {
58          return new NetBsdNetworkParams();
59      }

61      @Override
62      protected List<OSProcess> getProcessListFromPS(int pid) {
63          List<OSProcess> procs = new ArrayList<>();
64          // https://man.netbsd.org/ps#KEYWORDS
65          // missing are threadCount and kernelTime which is included in cputime
66          String psCommand = "ps -awwxo " + NetBsdOSProcess.PS_COMMAND_ARGS;
67          if (pid >= 0) {
68              psCommand += " -p " + pid;
69          }
70          List<String> procList = ExecutingCommand.runNative(psCommand);
71          if (procList.isEmpty() || procList.size() < 2) {
72              return procs;
73          }
74  
75          // remove header row
76          procList.remove(0);
77          // Fill list
78          for (String proc : procList) {
79              Map<BsdPsKeyword, String> psMap = ParseUtil.stringToEnumMap(BsdPsKeyword.class, NetBsdOSProcess.PS_KEYWORDS,
80                      proc.trim(), ' ');
81              // Check if last (thus all) value populated
82              if (psMap.containsKey(BsdPsKeyword.ARGS)) {
83                  procs.add(createProcess(pid < 0 ? ParseUtil.parseIntOrDefault(psMap.get(BsdPsKeyword.PID), 0) : pid,
84                          psMap));
85              }
86          }
87          return procs;
88      }
89  
90      /**
91       * Creates a process object from parsed {@code ps} output. Overridden by the JNA subclass to return a native-capable
92       * {@link NetBsdOSProcess} implementation.
93       *
94       * @param pid   the process ID
95       * @param psMap the parsed {@code ps} columns for the process
96       * @return a new {@link OSProcess}
97       */
98      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
99          return new NetBsdOSProcess(pid, psMap, this);
100     }
101 
102     @Override
103     public int getProcessId() {
104         // RuntimeMXBean name format is "pid@hostname"
105         String name = ManagementFactory.getRuntimeMXBean().getName();
106         return ParseUtil.parseIntOrDefault(name.split("@")[0], -1);
107     }
108 
109     @Override
110     // Thread.threadId() (the non-deprecated replacement) requires Java 19; this module compiles to Java 8.
111     @SuppressWarnings({ "deprecation", "java:S1874" })
112     public int getThreadId() {
113         return (int) Thread.currentThread().getId();
114     }
115 
116     @Override
117     public OSThread getCurrentThread() {
118         OSProcess proc = getCurrentProcess();
119         final int tid = getThreadId();
120         return proc.getThreadDetails().stream().filter(t -> t.getThreadId() == tid).findFirst()
121                 .orElse(new NetBsdOSThread(proc.getProcessID(), tid));
122     }
123 
124     @Override
125     protected long queryBootTime() {
126         // Boot time will be the first consecutive string of digits.
127         return ParseUtil.parseLongOrDefault(
128                 ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
129                 System.currentTimeMillis() / 1000);
130     }
