5   package oshi.software.common.os.unix.aix;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.common.AbstractNetworkParams;
9   import oshi.util.Constants;
10  import oshi.util.ExecutingCommand;
11  import oshi.util.ParseUtil;
12  
13  /**
14   * Abstract base for AIX NetworkParams. {@code netstat -rnf inet[6]} parsing is shared; only {@link #getHostName()}
15   * differs between JNA (gethostname via JNA) and FFM (gethostname via FFM).
16   */
17  @ThreadSafe
18  public abstract class AixNetworkParams extends AbstractNetworkParams {
19  
20      /** AIX hostname maximum length, including the trailing NUL. {@code HOST_NAME_MAX = 255}. */
21      protected static final int HOST_NAME_BUF_SIZE = 256;
22  
23      @Override
24      public String getIpv4DefaultGateway() {
25          return getDefaultGateway("netstat -rnf inet");
26      }
27  
28      @Override
29      public String getIpv6DefaultGateway() {
30          return getDefaultGateway("netstat -rnf inet6");
31      }
32  
33      private static String getDefaultGateway(String netstat) {
34          for (String line : ExecutingCommand.runNative(netstat)) {
35              String[] split = ParseUtil.whitespaces.split(line);
36              if (split.length > 7 && "default".equals(split[0])) {
37                  return split[1];
38              }
39          }
40          return Constants.UNKNOWN;
41      }
42  }
