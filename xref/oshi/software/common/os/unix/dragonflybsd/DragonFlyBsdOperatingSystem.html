7   import static oshi.software.os.OperatingSystem.ProcessFiltering.VALID_PROCESS;
8   
9   import java.util.List;
10  import java.util.Map;
11  import java.util.function.Predicate;
12  import java.util.stream.Collectors;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.software.common.os.unix.bsd.BsdOperatingSystem;
16  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
17  import oshi.software.os.OSProcess;
18  import oshi.software.os.OSSession;
19  import oshi.software.os.OSThread;
20  import oshi.util.ExecutingCommand;
21  import oshi.util.ParseUtil;
22  import oshi.util.tuples.Pair;

24  /**
25   * Abstract base shared by the DragonFly BSD OperatingSystem implementations (JNA and FFM). The cross-BSD-common pieces
26   * live in {@link BsdOperatingSystem}; this layer adds the DragonFly {@code ps} process enumeration (which filters out
27   * the kernel threads that report a negative PID) and the current-thread factory. The native bits (the
28   * {@link OSProcess}/{@code FileSystem}/{@code NetworkParams}/{@code InternetProtocolStats} factories,
29   * {@code getpid}/{@code lwp_gettid}, sysctl version and boot-time queries, and {@code who} sessions) are provided by
30   * the JNA and FFM subclasses.
31   */
32  @ThreadSafe
33  public abstract class DragonFlyBsdOperatingSystem extends BsdOperatingSystem {
34  
35      @Override
36      protected List<OSProcess> getProcessListFromPS(int pid) {
37          String psCommand = "ps -awwxo " + DragonFlyBsdOSProcess.PS_COMMAND_ARGS;
38          if (pid >= 0) {
39              psCommand += " -p " + pid;
40          }
41  
42          Predicate<Map<BsdPsKeyword, String>> hasKeywordCommand = psMap -> psMap.containsKey(BsdPsKeyword.COMMAND);
43          return ExecutingCommand.runNative(psCommand).stream().skip(1).parallel()
44                  .map(proc -> ParseUtil
45                          .stringToEnumMap(BsdPsKeyword.class, DragonFlyBsdOSProcess.PS_KEYWORDS, proc.trim(), ' '))
46                  .filter(hasKeywordCommand)
47                  .map(psMap -> createProcess(pid < 0 ? ParseUtil.parseIntOrDefault(psMap.get(BsdPsKeyword.PID), 0) : pid,
48                          psMap))
49                  // DragonFlyBSD kernel threads report PID -1; filter them out
50                  .filter(proc -> proc.getProcessID() > 0).filter(VALID_PROCESS).collect(Collectors.toList());
51      }
52  
53      @Override
54      public OSThread getCurrentThread() {
55          OSProcess proc = getCurrentProcess();
56          final int tid = getThreadId();
57          return proc.getThreadDetails().stream().filter(t -> t.getThreadId() == tid).findFirst()
58                  .orElse(new DragonFlyBsdOSThread(proc.getProcessID(), tid));
59      }
60  
61      @Override
62      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
63          return buildFamilyVersionInfo(querySysctl("kern.ostype", "DragonFly"), querySysctl("kern.osrelease", ""),
64                  querySysctl("kern.version", ""));
65      }
66  
67      @Override
68      public List<OSSession> getSessions() {
69          return USE_WHO_COMMAND ? super.getSessions() : queryWhoSessions();
70      }
71  
72      @Override
73      protected long queryBootTime() {
74          long bootSeconds = queryKernBoottimeSeconds();
75          return bootSeconds > 0 ? bootSeconds : queryBootTimeFromCommand();
76      }
77  
78      /**
79       * Creates a backend-specific {@link OSProcess} from a parsed {@code ps} row.
80       *
81       * @param pid   the process ID
82       * @param psMap the parsed {@code ps} columns for this process
83       * @return a new OSProcess
84       */
85      protected abstract OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap);
86  
87      /**
88       * Reads a string-valued sysctl by name.
89       *
90       * @param name the sysctl name (e.g. {@code "kern.ostype"})
91       * @param def  the default value if the sysctl can't be read
92       * @return the sysctl value, or {@code def}
93       */
94      protected abstract String querySysctl(String name, String def);
95  
96      /**
97       * Reads the login sessions natively via {@code getutxent}.
98       *
99       * @return the sessions
100      */
101     protected abstract List<OSSession> queryWhoSessions();
102 
103     /**
104      * Reads the boot time (seconds since the epoch) from the native {@code kern.boottime} sysctl.
105      *
106      * @return the boot time in seconds, or {@code 0} if the sysctl is unavailable
107      */
108     protected abstract long queryKernBoottimeSeconds();
109 }
