5   package oshi.software.common.os.unix.freebsd;
6   
7   import java.util.Arrays;
8   import java.util.Locale;
9   import java.util.stream.Collectors;
10  
11  import oshi.software.common.AbstractOSProcess;
12  
13  public abstract class FreeBsdOSProcess extends AbstractOSProcess {
14  
15      /**
16       * Columns requested from {@code ps -awwxo} when enumerating threads. Shared by FreeBsdOSProcess subclasses and
17       * FreeBsdOSThread so the column list and parsing enum stay in lockstep.
18       */
19      public enum PsThreadColumns {
20          TDNAME, LWP, STATE, ETIMES, SYSTIME, TIME, TDADDR, NIVCSW, NVCSW, MAJFLT, MINFLT, PRI;
21      }
22  
23      public static final String PS_THREAD_COLUMNS = Arrays.stream(PsThreadColumns.values()).map(Enum::name)
24              .map(name -> name.toLowerCase(Locale.ROOT)).collect(Collectors.joining(","));
25  
26      protected FreeBsdOSProcess(int pid) {
27          super(pid);
28      }
29  }
