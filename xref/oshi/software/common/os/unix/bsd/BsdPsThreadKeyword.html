7   /**
8    * The union of {@code ps} thread (per-LWP) output columns used by the BSD-family {@code OSThread} implementations
9    * (FreeBSD, OpenBSD, DragonFly, NetBSD). Each platform declares its own ordered subset (see {@code PS_THREAD_KEYWORDS}
10   * on the per-platform {@code OSProcess} base) and parses {@code ps} output positionally against that subset. The
11   * constant order here is not significant; only the per-platform subset order is.
12   */
13  public enum BsdPsThreadKeyword {
14      TDNAME, LWP, TID, LID, STATE, ETIME, ETIMES, CPUTIME, SYSTIME, TIME, TDADDR, NIVCSW, NVCSW, MAJFLT, MINFLT, PRI,
15      ARGS;
16  }
