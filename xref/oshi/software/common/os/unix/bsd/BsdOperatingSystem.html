30  /**
31   * Abstract base shared by the BSD-family OperatingSystem implementations (FreeBSD, OpenBSD, DragonFly, NetBSD). Holds
32   * the cross-BSD-identical pieces: the {@code Unix/BSD} manufacturer, the {@code uname}-based bitness, the
33   * process/child/descendant queries, the {@code ps -axo pid} process count, the {@code ps -axo nlwp} thread count, the
34   * {@code /etc/rc.d} services, and the memoized system uptime/boot time. Per-platform work (the {@code ps} process
35   * enumeration, the boot-time source, the current-thread factory, and the native version/file-system/network queries) is
36   * provided by the subclasses.
37   */
38  @ThreadSafe
39  public abstract class BsdOperatingSystem extends AbstractOperatingSystem {
40  
41      private static final Logger LOG = LoggerFactory.getLogger(BsdOperatingSystem.class);
42  
43      private final Supplier<Long> bootTime = memoize(this::queryBootTime);
44  
45      @Override
46      public String queryManufacturer() {
47          return "Unix/BSD";
48      }
49  
50      @Override
51      protected int queryBitness(int jvmBitness) {
52          if (jvmBitness < 64 && ExecutingCommand.getFirstAnswer("uname -m").indexOf("64") == -1) {
53              return jvmBitness;
54          }
55          return 64;
56      }
57  
58      @Override
59      public List<OSProcess> queryAllProcesses() {
60          return getProcessListFromPS(-1);
61      }
62  
63      @Override
64      public List<OSProcess> queryChildProcesses(int parentPid) {
65          List<OSProcess> allProcs = queryAllProcesses();
66          Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, false);
67          return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
68      }
69  
70      @Override
71      public List<OSProcess> queryDescendantProcesses(int parentPid) {
72          List<OSProcess> allProcs = queryAllProcesses();
73          Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, true);
74          return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
75      }
76  
77      @Override
78      public OSProcess getProcess(int pid) {
79          if (pid < 0) {
80              // A negative pid is the getProcessListFromPS "all processes" sentinel, not a real process
81              return null;
82          }
83          List<OSProcess> procs = getProcessListFromPS(pid);
84          if (procs.isEmpty()) {
85              return null;
86          }
87          return procs.get(0);
88      }
89  
90      @Override
91      public int getProcessCount() {
92          List<String> procList = ExecutingCommand.runNative("ps -axo pid");
93          if (!procList.isEmpty()) {
94              // Subtract 1 for header
95              return procList.size() - 1;
96          }
97          return 0;
98      }
99  
100     @Override
101     public int getThreadCount() {
102         // Sum nlwp (number of LWPs) across all processes; the header row parses to 0
103         int threads = 0;
104         for (String proc : ExecutingCommand.runNative("ps -axo nlwp")) {
105             threads += ParseUtil.parseIntOrDefault(proc.trim(), 0);
106         }
107         return threads;
108     }
109 
110     @Override
111     public long getSystemUptime() {
112         return System.currentTimeMillis() / 1000 - getSystemBootTime();
113     }
114 
115     @Override
116     public long getSystemBootTime() {
117         return bootTime.get();
118     }
119 
120     @Override
121     public List<OSService> getServices() {
122         // Get running services
123         List<OSService> services = new ArrayList<>();
124         Set<String> running = new HashSet<>();
125         for (OSProcess p : getChildProcesses(1, ProcessFiltering.ALL_PROCESSES, ProcessSorting.PID_ASC, 0)) {
126             OSService s = new OSService(p.getName(), p.getProcessID(), RUNNING);
127             services.add(s);
128             running.add(p.getName());
129         }
130         // Get Directories for stopped services
131         File dir = new File("/etc/rc.d");
132         File[] listFiles;
133         if (dir.exists() && dir.isDirectory() && (listFiles = dir.listFiles()) != null) {
134             for (File f : listFiles) {
135                 String name = f.getName();
136                 if (!running.contains(name)) {
137                     OSService s = new OSService(name, 0, STOPPED);
138                     services.add(s);
139                 }
140             }
141         } else {
142             LOG.error("Directory: /etc/rc.d does not exist");
143         }
144         return services;
145     }

147     /**
148      * Builds the family/version info from the raw {@code kern.ostype}/{@code kern.osrelease}/{@code kern.version}
149      * sysctl values. Shared by the FreeBSD, DragonFly, and OpenBSD subclasses, which differ only in how those three
150      * strings are fetched (string-named vs. MIB-based sysctl).
151      *
152      * @param family      the {@code kern.ostype} value (e.g. {@code "FreeBSD"})
153      * @param version     the {@code kern.osrelease} value
154      * @param versionInfo the {@code kern.version} value
155      * @return the family name paired with the parsed version info
156      */
157     protected static Pair<String, OSVersionInfo> buildFamilyVersionInfo(String family, String version,
158             String versionInfo) {
159         String buildNumber = versionInfo.split(":")[0].replace(family, "").replace(version, "").trim();
160         return new Pair<>(family, new OSVersionInfo(version, null, buildNumber));
161     }
162 
163     /**
164      * Parses the boot time (seconds since the epoch) from the {@code sysctl -n kern.boottime} command output. Used as a
165      * fallback by the BSD subclasses when the native {@code kern.boottime} sysctl is unavailable.
166      *
167      * @return the boot time in seconds, or the current time if it can't be parsed
168      */
169     protected static long queryBootTimeFromCommand() {
170         // Boot time will be the first consecutive string of digits.
171         return ParseUtil.parseLongOrDefault(
172                 ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
173                 System.currentTimeMillis() / 1000);
174     }
175 
176     /**
177      * Enumerates processes (or a single process when {@code pid >= 0}) via the platform's {@code ps} command.
178      *
179      * @param pid the process ID, or {@code -1} for all processes
180      * @return the process list
181      */
182     protected abstract List<OSProcess> getProcessListFromPS(int pid);
183 
184     /**
185      * Queries the system boot time in seconds since the epoch.
186      *
187      * @return the boot time in seconds
188      */
189     protected abstract long queryBootTime();
190 }
