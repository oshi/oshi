5   package oshi.software.common.os.mac;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.common.AbstractOSThread;
9   import oshi.software.os.OSProcess.State;
10  
11  /**
12   * OSThread implementation
13   */
14  @ThreadSafe
15  public class MacOSThread extends AbstractOSThread {
16  
17      /**
18       * Creates a MacOSThread with full parameters.
19       *
20       * @param pid        the process ID
21       * @param threadId   the thread ID
22       * @param state      the thread state
23       * @param kernelTime kernel time
24       * @param userTime   user time
25       * @param startTime  start time
26       * @param upTime     up time
27       * @param priority   the priority
28       */
29      public MacOSThread(int pid, int threadId, State state, long kernelTime, long userTime, long startTime, long upTime,
30              int priority) {
31          super(pid);
32          this.threadId = threadId;
33          this.state = state;
34          this.kernelTime = kernelTime;
35          this.userTime = userTime;
36          this.startTime = startTime;
37          this.upTime = upTime;
38          this.priority = priority;
39      }
40  
41      /**
42       * Creates a MacOSThread with just a process ID.
43       *
44       * @param processId the process ID
45       */
46      public MacOSThread(int processId) {
47          this(processId, 0, State.INVALID, 0L, 0L, 0L, 0L, 0);
48      }
49  
50  }
