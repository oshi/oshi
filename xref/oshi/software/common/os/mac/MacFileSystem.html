5   package oshi.software.common.os.mac;
6   
7   import java.nio.file.PathMatcher;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map;
11  import java.util.regex.Pattern;
12  
13  import oshi.annotation.concurrent.ThreadSafe;
14  import oshi.software.common.AbstractFileSystem;
15  import oshi.util.FileSystemUtil;
16  
17  /**
18   * Common constants and configuration for macOS file system implementations.
19   */
20  @ThreadSafe
21  public abstract class MacFileSystem extends AbstractFileSystem {
22  
23      /**
24       * Default constructor.
25       */
26      protected MacFileSystem() {
27      }
28  
29      /** Configuration key for path excludes. */
30      public static final String OSHI_MAC_FS_PATH_EXCLUDES = "oshi.os.mac.filesystem.path.excludes";
31      /** Configuration key for path includes. */
32      public static final String OSHI_MAC_FS_PATH_INCLUDES = "oshi.os.mac.filesystem.path.includes";
33      /** Configuration key for volume excludes. */
34      public static final String OSHI_MAC_FS_VOLUME_EXCLUDES = "oshi.os.mac.filesystem.volume.excludes";
35      /** Configuration key for volume includes. */
36      public static final String OSHI_MAC_FS_VOLUME_INCLUDES = "oshi.os.mac.filesystem.volume.includes";
37  
38      /** File system path exclude matchers. */
39      protected static final List<PathMatcher> FS_PATH_EXCLUDES = FileSystemUtil
40              .loadAndParseFileSystemConfig(OSHI_MAC_FS_PATH_EXCLUDES);
41      /** File system path include matchers. */
42      protected static final List<PathMatcher> FS_PATH_INCLUDES = FileSystemUtil
43              .loadAndParseFileSystemConfig(OSHI_MAC_FS_PATH_INCLUDES);
44      /** File system volume exclude matchers. */
45      protected static final List<PathMatcher> FS_VOLUME_EXCLUDES = FileSystemUtil
46              .loadAndParseFileSystemConfig(OSHI_MAC_FS_VOLUME_EXCLUDES);
47      /** File system volume include matchers. */
48      protected static final List<PathMatcher> FS_VOLUME_INCLUDES = FileSystemUtil
49              .loadAndParseFileSystemConfig(OSHI_MAC_FS_VOLUME_INCLUDES);
50  
51      /** Pattern matching local disk device paths. */
52      protected static final Pattern LOCAL_DISK = Pattern.compile("/dev/disk\\d+(s\\d+)?");
53  
54      /** Mount flag: read-only. */
55      protected static final int MNT_RDONLY = 0x00000001;
56      /** Mount flag: synchronous. */
57      protected static final int MNT_SYNCHRONOUS = 0x00000002;
58      /** Mount flag: no exec. */
59      protected static final int MNT_NOEXEC = 0x00000004;
60      /** Mount flag: no setuid. */
61      protected static final int MNT_NOSUID = 0x00000008;
62      /** Mount flag: no device access. */
63      protected static final int MNT_NODEV = 0x00000010;
64      /** Mount flag: union. */
65      protected static final int MNT_UNION = 0x00000020;
66      /** Mount flag: asynchronous. */
67      protected static final int MNT_ASYNC = 0x00000040;
68      /** Mount flag: content protection. */
69      protected static final int MNT_CPROTECT = 0x00000080;
70      /** Mount flag: exported. */
71      protected static final int MNT_EXPORTED = 0x00000100;
72      /** Mount flag: quarantined. */
73      protected static final int MNT_QUARANTINE = 0x00000400;
74      /** Mount flag: local. */
75      protected static final int MNT_LOCAL = 0x00001000;
76      /** Mount flag: quotas. */
77      protected static final int MNT_QUOTA = 0x00002000;
78      /** Mount flag: root filesystem. */
79      protected static final int MNT_ROOTFS = 0x00004000;
80      /** Mount flag: volfs. */
81      protected static final int MNT_DOVOLFS = 0x00008000;
82      /** Mount flag: don't browse. */
83      protected static final int MNT_DONTBROWSE = 0x00100000;
84      /** Mount flag: ignore ownership. */
85      protected static final int MNT_IGNORE_OWNERSHIP = 0x00200000;
86      /** Mount flag: automounted. */
87      protected static final int MNT_AUTOMOUNTED = 0x00400000;
88      /** Mount flag: journaled. */
89      protected static final int MNT_JOURNALED = 0x00800000;
90      /** Mount flag: no user extended attributes. */
91      protected static final int MNT_NOUSERXATTR = 0x01000000;
92      /** Mount flag: deferred writes. */
93      protected static final int MNT_DEFWRITE = 0x02000000;
94      /** Mount flag: multilabel MAC. */
95      protected static final int MNT_MULTILABEL = 0x04000000;
96      /** Mount flag: no access time. */
97      protected static final int MNT_NOATIME = 0x10000000;
98  
99      /** Map of mount flags to option strings. */
100     protected static final Map<Integer, String> OPTIONS_MAP = new HashMap<>();
101     static {
102         OPTIONS_MAP.put(MNT_SYNCHRONOUS, "synchronous");
103         OPTIONS_MAP.put(MNT_NOEXEC, "noexec");
104         OPTIONS_MAP.put(MNT_NOSUID, "nosuid");
105         OPTIONS_MAP.put(MNT_NODEV, "nodev");
106         OPTIONS_MAP.put(MNT_UNION, "union");
107         OPTIONS_MAP.put(MNT_ASYNC, "asynchronous");
108         OPTIONS_MAP.put(MNT_CPROTECT, "content-protection");
109         OPTIONS_MAP.put(MNT_EXPORTED, "exported");
110         OPTIONS_MAP.put(MNT_QUARANTINE, "quarantined");
111         OPTIONS_MAP.put(MNT_LOCAL, "local");
112         OPTIONS_MAP.put(MNT_QUOTA, "quotas");
113         OPTIONS_MAP.put(MNT_ROOTFS, "rootfs");
114         OPTIONS_MAP.put(MNT_DOVOLFS, "volfs");
115         OPTIONS_MAP.put(MNT_DONTBROWSE, "nobrowse");
116         OPTIONS_MAP.put(MNT_IGNORE_OWNERSHIP, "noowners");
117         OPTIONS_MAP.put(MNT_AUTOMOUNTED, "automounted");
118         OPTIONS_MAP.put(MNT_JOURNALED, "journaled");
119         OPTIONS_MAP.put(MNT_NOUSERXATTR, "nouserxattr");
120         OPTIONS_MAP.put(MNT_DEFWRITE, "defwrite");
121         OPTIONS_MAP.put(MNT_MULTILABEL, "multilabel");
122         OPTIONS_MAP.put(MNT_NOATIME, "noatime");
123     }
124 }
