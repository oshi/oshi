5   package oshi.benchmark;
6   
7   import java.util.concurrent.TimeUnit;
8   
9   import org.openjdk.jmh.annotations.Benchmark;
10  import org.openjdk.jmh.annotations.BenchmarkMode;
11  import org.openjdk.jmh.annotations.Fork;
12  import org.openjdk.jmh.annotations.Measurement;
13  import org.openjdk.jmh.annotations.Mode;
14  import org.openjdk.jmh.annotations.OutputTimeUnit;
15  import org.openjdk.jmh.annotations.Scope;
16  import org.openjdk.jmh.annotations.Setup;
17  import org.openjdk.jmh.annotations.State;
18  import org.openjdk.jmh.annotations.Warmup;
19  import org.openjdk.jmh.runner.Runner;
20  import org.openjdk.jmh.runner.RunnerException;
21  import org.openjdk.jmh.runner.options.Options;
22  import org.openjdk.jmh.runner.options.OptionsBuilder;
23  
24  import oshi.hardware.CentralProcessor;
25  import oshi.util.GlobalConfig;
26  
27  /**
28   * Side-by-side benchmarks of JNA vs FFM implementations of {@link CentralProcessor#getProcessorCpuLoadTicks()}.
29   */
30  @BenchmarkMode(Mode.AverageTime)
31  @OutputTimeUnit(TimeUnit.MILLISECONDS)
32  @State(Scope.Benchmark)
33  @Warmup(iterations = 3, time = 2)
34  @Measurement(iterations = 5, time = 2)
35  @Fork(value = 1, jvmArgsPrepend = "--enable-native-access=ALL-UNNAMED")
36  public class CpuTicksBenchmark {
37  
38      private CentralProcessor jnaCpu;
39      private CentralProcessor ffmCpu;
40  
41      /** Creates a new benchmark instance. Required by JMH for {@code @State} classes. */
42      public CpuTicksBenchmark() {
43      }
44  
45      /**
46       * Initializes JNA and FFM {@link CentralProcessor} instances with memoization disabled.
47       */
48      @Setup
49      public void setup() {
50          GlobalConfig.set(GlobalConfig.OSHI_UTIL_MEMOIZER_EXPIRATION, 0);
51          jnaCpu = new oshi.SystemInfo().getHardware().getProcessor();
52          ffmCpu = new oshi.ffm.SystemInfo().getHardware().getProcessor();
53      }
54  
55      /**
56       * Benchmarks the JNA implementation of {@link CentralProcessor#getProcessorCpuLoadTicks()}.
57       *
58       * @return per-processor tick counts
59       */
60      @Benchmark
61      public long[][] jna() {
62          return jnaCpu.getProcessorCpuLoadTicks();
63      }
64  
65      /**
66       * Benchmarks the FFM implementation of {@link CentralProcessor#getProcessorCpuLoadTicks()}.
67       *
68       * @return per-processor tick counts
69       */
70      @Benchmark
71      public long[][] ffm() {
72          return ffmCpu.getProcessorCpuLoadTicks();
73      }
74  
75      /**
76       * Standalone entry point for running this benchmark outside the fat jar.
77       *
78       * @param args command-line arguments (unused)
79       * @throws RunnerException if the benchmark fails
80       */
81      public static void main(String[] args) throws RunnerException {
82          Options opt = new OptionsBuilder().include(CpuTicksBenchmark.class.getSimpleName()).build();
83          new Runner(opt).run();
84      }
85  }
