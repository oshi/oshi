5   package oshi.benchmark;
6   
7   import java.util.List;
8   import java.util.concurrent.TimeUnit;
9   
10  import org.openjdk.jmh.annotations.Benchmark;
11  import org.openjdk.jmh.annotations.BenchmarkMode;
12  import org.openjdk.jmh.annotations.Fork;
13  import org.openjdk.jmh.annotations.Measurement;
14  import org.openjdk.jmh.annotations.Mode;
15  import org.openjdk.jmh.annotations.OutputTimeUnit;
16  import org.openjdk.jmh.annotations.Scope;
17  import org.openjdk.jmh.annotations.Setup;
18  import org.openjdk.jmh.annotations.State;
19  import org.openjdk.jmh.annotations.Warmup;
20  import org.openjdk.jmh.infra.Blackhole;
21  import org.openjdk.jmh.runner.Runner;
22  import org.openjdk.jmh.runner.RunnerException;
23  import org.openjdk.jmh.runner.options.Options;
24  import org.openjdk.jmh.runner.options.OptionsBuilder;
25  
26  import oshi.software.os.OSFileStore;
27  import oshi.software.os.OperatingSystem;
28  import oshi.util.GlobalConfig;
29  
30  /**
31   * Side-by-side benchmarks of JNA vs FFM implementations of {@link OSFileStore#updateAttributes()}.
32   */
33  @BenchmarkMode(Mode.AverageTime)
34  @OutputTimeUnit(TimeUnit.MICROSECONDS)
35  @State(Scope.Benchmark)
36  @Warmup(iterations = 3, time = 2)
37  @Measurement(iterations = 5, time = 2)
38  @Fork(value = 1, jvmArgsPrepend = "--enable-native-access=ALL-UNNAMED")
39  public class FileStoreBenchmark {
40  
41      private List<OSFileStore> jnaStores;
42      private List<OSFileStore> ffmStores;
43  
44      /** Creates a new benchmark instance. Required by JMH for {@code @State} classes. */
45      public FileStoreBenchmark() {
46      }
47  
48      /**
49       * Initializes JNA and FFM {@link OSFileStore} lists with memoization disabled.
50       */
51      @Setup
52      public void setup() {
53          GlobalConfig.set(GlobalConfig.OSHI_UTIL_MEMOIZER_EXPIRATION, 0);
54          OperatingSystem jnaOs = new oshi.SystemInfo().getOperatingSystem();
55          OperatingSystem ffmOs = new oshi.ffm.SystemInfo().getOperatingSystem();
56          jnaStores = jnaOs.getFileSystem().getFileStores();
57          ffmStores = ffmOs.getFileSystem().getFileStores();
58      }
59  
60      /**
61       * Benchmarks the JNA implementation of {@link OSFileStore#updateAttributes()}.
62       *
63       * @param bh JMH black hole to prevent dead-code elimination
64       */
65      @Benchmark
66      public void jna(Blackhole bh) {
67          for (OSFileStore store : jnaStores) {
68              bh.consume(store.updateAttributes());
69          }
70      }
71  
72      /**
73       * Benchmarks the FFM implementation of {@link OSFileStore#updateAttributes()}.
74       *
75       * @param bh JMH black hole to prevent dead-code elimination
76       */
77      @Benchmark
78      public void ffm(Blackhole bh) {
79          for (OSFileStore store : ffmStores) {
80              bh.consume(store.updateAttributes());
81          }
82      }
83  
84      /**
85       * Standalone entry point for running this benchmark outside the fat jar.
86       *
87       * @param args command-line arguments (unused)
88       * @throws RunnerException if the benchmark fails
89       */
90      public static void main(String[] args) throws RunnerException {
91          Options opt = new OptionsBuilder().include(FileStoreBenchmark.class.getSimpleName()).build();
92          new Runner(opt).run();
93      }
94  }
