17  package oshi;
18  
19  import oshi.hardware.HardwareAbstractionLayer;
20  import oshi.software.os.OperatingSystem;
21  import oshi.software.os.linux.LinuxHardwareAbstractionLayer;
22  import oshi.software.os.linux.LinuxOperatingSystem;
23  import oshi.software.os.mac.MacHardwareAbstractionLayer;
24  import oshi.software.os.mac.MacOperatingSystem;
25  import oshi.software.os.windows.WindowsHardwareAbstractionLayer;
26  import oshi.software.os.windows.WindowsOperatingSystem;
27  
28  import com.sun.jna.Platform;
29  
30  /**
31   * System information. This is the main entry point to Oshi. This object
32   * provides getters which instantiate the appropriate platform-specific
33   * implementations of {@link OperatingSystem} (software) and
34   * {@link HardwareAbstractionLayer} (hardware).
35   * 
36   * @author dblock[at]dblock[dot]org
37   */
38  public class SystemInfo {
39  	private OperatingSystem _os = null;
40  	private HardwareAbstractionLayer _hardware = null;
41  	private PlatformEnum currentPlatformEnum;
42  
43  	{
44  		if (Platform.isWindows())
45  			this.currentPlatformEnum = PlatformEnum.WINDOWS;
46  		else if (Platform.isLinux())
47  			this.currentPlatformEnum = PlatformEnum.LINUX;
48  		else if (Platform.isMac())
49  			this.currentPlatformEnum = PlatformEnum.MACOSX;
50  		else
51  			this.currentPlatformEnum = PlatformEnum.UNKNOWN;
52  	}
53  
54  	/**
55  	 * Creates a new instance of the appropriate platform-specific
56  	 * {@link OperatingSystem}.
57  	 * 
58  	 * @return A new instance of {@link OperatingSystem}.
59  	 */
60  	public OperatingSystem getOperatingSystem() {
61  		if (this._os == null) {
62  			switch (this.currentPlatformEnum) {
63  
64  			case WINDOWS:
65  				this._os = new WindowsOperatingSystem();
66  				break;
67  			case LINUX:
68  				this._os = new LinuxOperatingSystem();
69  				break;
70  			case MACOSX:
71  				this._os = new MacOperatingSystem();
72  				break;
73  			default:
74  				throw new RuntimeException("Operating system not supported: " + Platform.getOSType());
75  			}
76  		}
77  		return this._os;
78  	}
79  
80  	/**
81  	 * Creates a new instance of the appropriate platform-specific
82  	 * {@link HardwareAbstractionLayer}.
83  	 * 
84  	 * @return A new instance of {@link HardwareAbstractionLayer}.
85  	 */
86  	public HardwareAbstractionLayer getHardware() {
87  		if (this._hardware == null) {
88  			switch (this.currentPlatformEnum) {
89  
90  			case WINDOWS:
91  				this._hardware = new WindowsHardwareAbstractionLayer();
92  				break;
93  			case LINUX:
94  				this._hardware = new LinuxHardwareAbstractionLayer();
95  				break;
96  			case MACOSX:
97  				this._hardware = new MacHardwareAbstractionLayer();
98  				break;
99  			default:
100 				throw new RuntimeException("Operating system not supported: " + Platform.getOSType());
101 			}
102 		}
103 		return this._hardware;
104 	}
105 }
