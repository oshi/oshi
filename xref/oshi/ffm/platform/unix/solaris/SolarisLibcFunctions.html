25  /**
26   * FFM bindings for Solaris/illumos libc functions used by OSHI.
27   * <p>
28   * Solaris uses kstat as its primary kernel-statistics interface, so the libc surface here is small: thread/process IDs
29   * and {@code getrlimit}. The {@code RLIMIT_NOFILE} resource constant is {@code 5} on Solaris/illumos (not {@code 7} as
30   * on Linux — see <a href="https://illumos.org/man/2/getrlimit">illumos getrlimit(2)</a>).
31   */
32  public final class SolarisLibcFunctions extends ForeignFunctions {
33  
34      private SolarisLibcFunctions() {
35      }
36  
37      /** Layout of the C {@code size_t} type on 64-bit Solaris/illumos. */
38      public static final ValueLayout.OfLong SIZE_T = ValueLayout.JAVA_LONG;
39  
40      /**
41       * {@code getrlimit} resource: maximum number of open file descriptors. illumos/Solaris value (5). JNA's
42       * {@code Resource.RLIMIT_NOFILE} is the Linux value (7), which on Solaris corresponds to {@code RLIM_NLIMITS}
43       * (invalid); the JNA-side {@code oshi.jna.platform.unix.SolarisLibc#RLIMIT_NOFILE} shadows it with the correct
44       * value.
45       */
46      public static final int RLIMIT_NOFILE = 5;
47  
48      /** Layout of Solaris/illumos {@code struct rlimit}: two {@code rlim_t} (LP64 long) fields. */
49      public static final StructLayout RLIMIT_LAYOUT = MemoryLayout.structLayout(JAVA_LONG.withName("rlim_cur"),
50              JAVA_LONG.withName("rlim_max"));
51  
52      private static final VarHandle RLIMIT_CUR = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_cur"));
53      private static final VarHandle RLIMIT_MAX = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_max"));

154     // ---- utmpx (Solaris/illumos <utmpx.h>) ----
155 
156     /** utmpx entry type: session leader of a logged in user. */
157     public static final short LOGIN_PROCESS = 6;
158     /** utmpx entry type: normal process. */
159     public static final short USER_PROCESS = 7;
160 
161     static final int UTX_USERSIZE = 32;
162     static final int UTX_IDSIZE = 4;
163     static final int UTX_LINESIZE = 32;
164     static final int UTX_HOSTSIZE = 257;
165 
166     /**
167      * Layout of Solaris/illumos {@code struct utmpx} on LP64. Mirrors the JNA {@code SolarisLibc.SolarisUtmpx} struct.
168      *
169      * <pre>
170      *   char[32]  ut_user                      (32) @ 0
171      *   char[4]   ut_id                         (4) @ 32
172      *   char[32]  ut_line                      (32) @ 36
173      *   int       ut_pid                        (4) @ 68
174      *   short     ut_type                       (2) @ 72
175      *   short[2]  ut_exit {e_termination,e_exit}(4) @ 74
176      *   pad (align ut_tv to 8)                  (2) @ 78
177      *   long      ut_tv.tv_sec                   (8) @ 80
178      *   long      ut_tv.tv_usec                  (8) @ 88
179      *   int       ut_session                     (4) @ 96
180      *   int[5]    pad                           (20) @ 100
181      *   short     ut_syslen                      (2) @ 120
182      *   char[257] ut_host                      (257) @ 122
183      *   trailing pad (align struct to 8)         (5) @ 379
184      *   total = 384 bytes
185      * </pre>
186      */
187     public static final StructLayout UTMPX_LAYOUT = MemoryLayout.structLayout(
188             MemoryLayout.sequenceLayout(UTX_USERSIZE, JAVA_BYTE).withName("ut_user"),
189             MemoryLayout.sequenceLayout(UTX_IDSIZE, JAVA_BYTE).withName("ut_id"),
190             MemoryLayout.sequenceLayout(UTX_LINESIZE, JAVA_BYTE).withName("ut_line"), JAVA_INT.withName("ut_pid"),
191             JAVA_SHORT.withName("ut_type"), MemoryLayout.paddingLayout(6), JAVA_LONG.withName("ut_tv_sec"),
192             JAVA_LONG.withName("ut_tv_usec"), MemoryLayout.paddingLayout(26),
193             MemoryLayout.sequenceLayout(UTX_HOSTSIZE, JAVA_BYTE).withName("ut_host"), MemoryLayout.paddingLayout(5));
194 
195     private static final VarHandle UTMPX_TYPE = UTMPX_LAYOUT.varHandle(PathElement.groupElement("ut_type"));
196     private static final VarHandle UTMPX_TV_SEC = UTMPX_LAYOUT.varHandle(PathElement.groupElement("ut_tv_sec"));
197     private static final VarHandle UTMPX_TV_USEC = UTMPX_LAYOUT.varHandle(PathElement.groupElement("ut_tv_usec"));
198 
199     private static final long UTMPX_USER_OFFSET = UTMPX_LAYOUT.byteOffset(PathElement.groupElement("ut_user"));
200     private static final long UTMPX_LINE_OFFSET = UTMPX_LAYOUT.byteOffset(PathElement.groupElement("ut_line"));
201     private static final long UTMPX_HOST_OFFSET = UTMPX_LAYOUT.byteOffset(PathElement.groupElement("ut_host"));

212     // struct utmpx * getutxent(void);
213     private static final MethodHandle getutxent = LINKER.downcallHandle(LIBC.findOrThrow("getutxent"),
214             FunctionDescriptor.of(ADDRESS));
215 
216     /**
217      * Reads the next entry from the utmpx database.
218      *
219      * @return a pointer to the utmpx structure, or {@code null} if no more entries
220      * @throws Throwable on FFM invocation error
221      */
222     public static MemorySegment getutxent() throws Throwable {
223         MemorySegment result = (MemorySegment) getutxent.invokeExact();
224         return result.equals(MemorySegment.NULL) ? null : result;
225     }

232     public static void endutxent() throws Throwable {

276     /**
277      * Reads the login time from a utmpx segment as epoch milliseconds.
278      *
279      * @param ut segment populated by {@link #getutxent()} and reinterpreted to {@link #UTMPX_LAYOUT}
280      * @return login time in milliseconds since epoch
281      */
282     public static long utmpxLoginTime(MemorySegment ut) {
283         long tvSec = (long) UTMPX_TV_SEC.get(ut, 0L);
284         long tvUsec = (long) UTMPX_TV_USEC.get(ut, 0L);
285         return tvSec * 1000L + tvUsec / 1000L;
