5   package oshi.ffm.platform.unix.dragonflybsd;
6   
7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   
9   import java.lang.foreign.FunctionDescriptor;
10  import java.lang.foreign.SymbolLookup;
11  import java.lang.invoke.MethodHandle;
12  
13  import oshi.ffm.ForeignFunctions;
14  
15  /**
16   * FFM bindings for DragonFly BSD-specific libc functions. DragonFly shares most of its libc surface with FreeBSD, so
17   * shared functions ({@code sysctlbyname}, {@code sysctl}, {@code getpid}, {@code getrlimit}, etc.) are called via
18   * {@link oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions}. This class adds only the DragonFly-specific
19   * {@code lwp_gettid()} (FreeBSD uses {@code thr_self} instead).
20   */
21  public final class DragonFlyBsdLibcFunctions extends ForeignFunctions {
22  
23      private DragonFlyBsdLibcFunctions() {
24      }
25  
26      private static final SymbolLookup LIBC = LINKER.defaultLookup();
27  
28      // int lwp_gettid(void);
29      private static final MethodHandle lwp_gettid = LINKER.downcallHandle(LIBC.findOrThrow("lwp_gettid"),
30              FunctionDescriptor.of(JAVA_INT));
31  
32      /**
33       * Calls {@code lwp_gettid()} — returns the kernel thread ID of the calling LWP.
34       *
35       * @return the kernel thread ID, or -1 on error
36       * @throws Throwable on FFM invocation error
37       */
38      public static int lwp_gettid() throws Throwable {
39          return (int) lwp_gettid.invokeExact();
40      }
41  }
