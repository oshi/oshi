28  /**
29   * FFM binding to {@code dxgi.dll} for enumerating display adapters via COM vtable calls.
30   */
31  public final class DxgiFFM extends ForeignFunctions {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(DxgiFFM.class);
34  
35      private static final int DXGI_ERROR_NOT_FOUND = 0x887A0002;
36  
37      // DXGI_ADAPTER_DESC layout (x64):
38      // Description: WCHAR[128] = 256 bytes, VendorId/DeviceId/SubSysId/Revision: 4*4=16 bytes,
39      // DedicatedVideoMemory/DedicatedSystemMemory/SharedSystemMemory: 3*8=24 bytes (SIZE_T on x64),
40      // AdapterLuid: 8 bytes (DWORD LowPart + LONG HighPart)
41      private static final long DESC_SIZE = 304;
42      private static final long DESC_VENDOR_ID_OFFSET = 256;
43      private static final long DESC_DEVICE_ID_OFFSET = 260;
44      private static final long DESC_DEDICATED_VIDEO_MEMORY_OFFSET = 272;
45      private static final long DESC_LUID_LOW_OFFSET = 296;
46      private static final long DESC_LUID_HIGH_OFFSET = 300;
47  
48      // IID_IDXGIFactory {7B7166EC-21C7-44AE-B21A-C9AE321AE369}
49      private static final byte[] IID_IDXGI_FACTORY = { (byte) 0xEC, 0x66, 0x71, 0x7B, (byte) 0xC7, 0x21, (byte) 0xAE,
50              0x44, (byte) 0xB2, 0x1A, (byte) 0xC9, (byte) 0xAE, 0x32, 0x1A, (byte) 0xE3, 0x69 };
51  
52      private static final boolean AVAILABLE;
53      private static final MethodHandle CREATE_DXGI_FACTORY;
54  
55      static {
56          boolean available = false;
57          MethodHandle hCreate = null;
58          try {
59              SymbolLookup dxgi = SymbolLookup.libraryLookup("dxgi", Arena.global());
60              MemorySegment sym = dxgi.findOrThrow("CreateDXGIFactory");
61              hCreate = Linker.nativeLinker().downcallHandle(sym, FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
62              available = true;
63          } catch (Throwable t) {
64              LOG.debug("dxgi.dll not available via FFM: {}", t.getMessage());
65          }
66          AVAILABLE = available;
67          CREATE_DXGI_FACTORY = hCreate;
68      }
69  
70      private DxgiFFM() {
71      }
72  
73      /**
74       * Enumerates all DXGI display adapters.
75       *
76       * @return list of {@link DxgiAdapterInfo}, one per adapter; empty if DXGI is unavailable
77       */
78      public static List<DxgiAdapterInfo> queryAdapters() {
79          if (!AVAILABLE) {
80              return Collections.emptyList();
81          }
82          try (Arena arena = Arena.ofConfined()) {
83              // Write IID to memory
84              MemorySegment riid = arena.allocate(16);
85              riid.copyFrom(MemorySegment.ofArray(IID_IDXGI_FACTORY));
86  
87              // CreateDXGIFactory(riid, &ppFactory)
88              MemorySegment ppFactory = arena.allocate(ADDRESS);
89              int hr = (int) CREATE_DXGI_FACTORY.invokeExact(riid, ppFactory);
90              if (hr < 0) {
91                  LOG.debug("CreateDXGIFactory failed: 0x{}", Integer.toHexString(hr));
92                  return Collections.emptyList();
93              }
94  
95              MemorySegment factory = ppFactory.get(ADDRESS, 0).reinterpret(Long.MAX_VALUE, arena, null);
96              List<DxgiAdapterInfo> result = new ArrayList<>();
97              try {
98                  for (int i = 0;; i++) {
99                      MemorySegment ppAdapter = arena.allocate(ADDRESS);
100                     int enumHr = vtableCallEnumAdapters(factory, i, ppAdapter, arena);
101                     if (enumHr == DXGI_ERROR_NOT_FOUND) {
102                         break;
103                     }
104                     if (enumHr < 0) {
105                         LOG.debug("IDXGIFactory::EnumAdapters({}) failed: 0x{}", i, Integer.toHexString(enumHr));
106                         break;
107                     }
108 
109                     MemorySegment adapter = ppAdapter.get(ADDRESS, 0).reinterpret(Long.MAX_VALUE, arena, null);
110                     try {
111                         MemorySegment desc = arena.allocate(DESC_SIZE);
112                         int descHr = vtableCallGetDesc(adapter, desc, arena);
113                         if (descHr >= 0) {
114                             String name = readDescriptionWchars(desc);
115                             int vendorId = desc.get(JAVA_INT, DESC_VENDOR_ID_OFFSET);
116                             int deviceId = desc.get(JAVA_INT, DESC_DEVICE_ID_OFFSET);
117                             long vram = desc.get(JAVA_LONG, DESC_DEDICATED_VIDEO_MEMORY_OFFSET);
118                             int luidLow = desc.get(JAVA_INT, DESC_LUID_LOW_OFFSET);
119                             int luidHigh = desc.get(JAVA_INT, DESC_LUID_HIGH_OFFSET);
120                             result.add(new DxgiAdapterInfo(name, vendorId, deviceId, vram, luidLow, luidHigh));
121                             LOG.debug("DXGI adapter {}: '{}' vendor=0x{} device=0x{} vram={} luid=0x{}_0x{}", i, name,
122                                     Integer.toHexString(vendorId), Integer.toHexString(deviceId), vram,
123                                     Integer.toHexString(luidHigh), Integer.toHexString(luidLow));
124                         }
125                     } finally {
126                         vtableRelease(adapter, arena);
127                     }
128                 }
129             } finally {
130                 vtableRelease(factory, arena);
131             }
132             return Collections.unmodifiableList(result);
133         } catch (Throwable t) {
134             LOG.debug("DXGI enumeration failed: {}", t.getMessage());
135             return Collections.emptyList();
136         }
137     }
138 
139     // IDXGIFactory::EnumAdapters is vtable slot 7
140     private static int vtableCallEnumAdapters(MemorySegment factory, int index, MemorySegment ppAdapter, Arena arena)
141             throws Throwable {
142         MemorySegment vtable = factory.get(ADDRESS, 0).reinterpret(8 * 32L, arena, null);
143         MemorySegment fn = vtable.get(ADDRESS, 7 * ADDRESS.byteSize());
144         MethodHandle mh = Linker.nativeLinker().downcallHandle(fn,
145                 FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS));
146         return (int) mh.invokeExact(factory, index, ppAdapter);
147     }
148 
149     // IDXGIAdapter::GetDesc is vtable slot 8
150     private static int vtableCallGetDesc(MemorySegment adapter, MemorySegment desc, Arena arena) throws Throwable {
151         MemorySegment vtable = adapter.get(ADDRESS, 0).reinterpret(8 * 32L, arena, null);
152         MemorySegment fn = vtable.get(ADDRESS, 8 * ADDRESS.byteSize());
153         MethodHandle mh = Linker.nativeLinker().downcallHandle(fn, FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
154         return (int) mh.invokeExact(adapter, desc);
155     }
156 
157     // IUnknown::Release is vtable slot 2
158     private static void vtableRelease(MemorySegment obj, Arena arena) throws Throwable {
159         MemorySegment vtable = obj.get(ADDRESS, 0).reinterpret(8 * 32L, arena, null);
160         MemorySegment fn = vtable.get(ADDRESS, 2 * ADDRESS.byteSize());
161         MethodHandle mh = Linker.nativeLinker().downcallHandle(fn, FunctionDescriptor.of(JAVA_INT, ADDRESS));
162         mh.invokeExact(obj);
163     }
164 
165     private static String readDescriptionWchars(MemorySegment desc) {
166         // Description is WCHAR[128] at offset 0
167         int len = 0;
168         while (len < 128 && desc.get(JAVA_CHAR, len * 2L) != 0) {
169             len++;
170         }
171         char[] chars = new char[len];
172         for (int i = 0; i < len; i++) {
173             chars[i] = desc.get(JAVA_CHAR, i * 2L);
174         }
175         return new String(chars);
176     }
