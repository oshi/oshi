5   package oshi.ffm.platform.windows;
6   
7   import static java.lang.foreign.MemoryLayout.structLayout;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   import static java.lang.foreign.ValueLayout.JAVA_LONG;
10  import static java.lang.foreign.ValueLayout.JAVA_SHORT;
11  
12  import java.lang.foreign.MemoryLayout;
13  import java.lang.foreign.StructLayout;
14  
15  /**
16   * FFM struct layouts for Windows Performance Counter structures read from HKEY_PERFORMANCE_DATA.
17   * <p>
18   * The registry data is organized as:
19   *
20   * <pre>
21   * [ PERF_DATA_BLOCK ]                          -- data header with timestamps
22   *   [ PERF_OBJECT_TYPE ]                       -- object type header (e.g., Process, Thread)
23   *     [ PERF_COUNTER_DEFINITION ] × NumCounters -- counter metadata (offset, size, index)
24   *   [ PERF_INSTANCE_DEFINITION ] × NumInstances -- per-instance data:
25   *     [ instance name (wide string) ]
26   *     [ PERF_COUNTER_BLOCK ]                    -- raw counter values for this instance
27   *       [ counter data bytes ]
28   * </pre>
29   *
30   * @see <a href="https://learn.microsoft.com/en-us/windows/win32/perfctrs/performance-counters-structures">Performance
31   *      Counters Structures</a>
32   */
33  public interface WinPerfFFM {
34  
35      // PERF_DATA_BLOCK: describes the performance data block header
36      // https://learn.microsoft.com/en-us/windows/win32/api/winperf/ns-winperf-perf_data_block
37      //
38      // Fields:
39      // WCHAR[4] Signature (8 bytes)
40      // DWORD LittleEndian (4 bytes)
41      // DWORD Version (4 bytes)
42      // DWORD Revision (4 bytes)
43      // DWORD TotalByteLength (4 bytes)
44      // DWORD HeaderLength (4 bytes)
45      // DWORD NumObjectTypes (4 bytes)
46      // LONG DefaultObject (4 bytes)
47      // SYSTEMTIME SystemTime (16 bytes: 8 × WORD)
48      // LARGE_INTEGER PerfTime (8 bytes)
49      // LARGE_INTEGER PerfFreq (8 bytes)
50      // LARGE_INTEGER PerfTime100nSec (8 bytes)
51      // DWORD SystemNameLength (4 bytes)
52      // DWORD SystemNameOffset (4 bytes)
53      StructLayout PERF_DATA_BLOCK = structLayout(JAVA_SHORT.withName("Signature0"), JAVA_SHORT.withName("Signature1"),
54              JAVA_SHORT.withName("Signature2"), JAVA_SHORT.withName("Signature3"), JAVA_INT.withName("LittleEndian"),
55              JAVA_INT.withName("Version"), JAVA_INT.withName("Revision"), JAVA_INT.withName("TotalByteLength"),
56              JAVA_INT.withName("HeaderLength"), JAVA_INT.withName("NumObjectTypes"), JAVA_INT.withName("DefaultObject"),
57              // 4 bytes padding for 8-byte alignment of LARGE_INTEGER fields
58              MemoryLayout.paddingLayout(4),
59              // SYSTEMTIME: 8 × WORD (16 bytes) — we only need the struct size, not individual fields
60              MemoryLayout.paddingLayout(16).withName("SystemTime"), JAVA_LONG.withName("PerfTime"),
61              JAVA_LONG.withName("PerfFreq"), JAVA_LONG.withName("PerfTime100nSec"),
62              JAVA_INT.withName("SystemNameLength"), JAVA_INT.withName("SystemNameOffset"));
63  
64      long PERF_DATA_BLOCK_HeaderLength = PERF_DATA_BLOCK
65              .byteOffset(MemoryLayout.PathElement.groupElement("HeaderLength"));
66      long PERF_DATA_BLOCK_NumObjectTypes = PERF_DATA_BLOCK
67              .byteOffset(MemoryLayout.PathElement.groupElement("NumObjectTypes"));
68      long PERF_DATA_BLOCK_PerfTime100nSec = PERF_DATA_BLOCK
69              .byteOffset(MemoryLayout.PathElement.groupElement("PerfTime100nSec"));
70  
71      // PERF_OBJECT_TYPE: describes object-specific performance information
72      // https://learn.microsoft.com/en-us/windows/win32/api/winperf/ns-winperf-perf_object_type
73      //
74      // Fields:
75      // DWORD TotalByteLength (4 bytes)
76      // DWORD DefinitionLength (4 bytes)
77      // DWORD HeaderLength (4 bytes)
78      // DWORD ObjectNameTitleIndex (4 bytes)
79      // DWORD ObjectNameTitle (4 bytes) — always 32-bit
80      // DWORD ObjectHelpTitleIndex (4 bytes)
81      // DWORD ObjectHelpTitle (4 bytes) — always 32-bit
82      // DWORD DetailLevel (4 bytes)
83      // DWORD NumCounters (4 bytes)
84      // LONG DefaultCounter (4 bytes)
85      // LONG NumInstances (4 bytes)
86      // DWORD CodePage (4 bytes)
87      // LARGE_INTEGER PerfTime (8 bytes)
88      // LARGE_INTEGER PerfFreq (8 bytes)
89      StructLayout PERF_OBJECT_TYPE = structLayout(JAVA_INT.withName("TotalByteLength"),
90              JAVA_INT.withName("DefinitionLength"), JAVA_INT.withName("HeaderLength"),
91              JAVA_INT.withName("ObjectNameTitleIndex"), JAVA_INT.withName("ObjectNameTitle"),
92              JAVA_INT.withName("ObjectHelpTitleIndex"), JAVA_INT.withName("ObjectHelpTitle"),
93              JAVA_INT.withName("DetailLevel"), JAVA_INT.withName("NumCounters"), JAVA_INT.withName("DefaultCounter"),
94              JAVA_INT.withName("NumInstances"), JAVA_INT.withName("CodePage"), JAVA_LONG.withName("PerfTime"),
95              JAVA_LONG.withName("PerfFreq"));
96  
97      long PERF_OBJECT_TYPE_TotalByteLength = PERF_OBJECT_TYPE
98              .byteOffset(MemoryLayout.PathElement.groupElement("TotalByteLength"));
99      long PERF_OBJECT_TYPE_DefinitionLength = PERF_OBJECT_TYPE
100             .byteOffset(MemoryLayout.PathElement.groupElement("DefinitionLength"));
101     long PERF_OBJECT_TYPE_HeaderLength = PERF_OBJECT_TYPE
102             .byteOffset(MemoryLayout.PathElement.groupElement("HeaderLength"));
103     long PERF_OBJECT_TYPE_ObjectNameTitleIndex = PERF_OBJECT_TYPE
104             .byteOffset(MemoryLayout.PathElement.groupElement("ObjectNameTitleIndex"));
105     long PERF_OBJECT_TYPE_NumCounters = PERF_OBJECT_TYPE
106             .byteOffset(MemoryLayout.PathElement.groupElement("NumCounters"));
107     long PERF_OBJECT_TYPE_NumInstances = PERF_OBJECT_TYPE
108             .byteOffset(MemoryLayout.PathElement.groupElement("NumInstances"));
109 
110     // PERF_COUNTER_DEFINITION: describes a performance counter
111     // https://learn.microsoft.com/en-us/windows/win32/api/winperf/ns-winperf-perf_counter_definition
112     //
113     // Fields:
114     // DWORD ByteLength (4 bytes)
115     // DWORD CounterNameTitleIndex (4 bytes)
116     // DWORD CounterNameTitle (4 bytes) — always 32-bit
117     // DWORD CounterHelpTitleIndex (4 bytes)
118     // DWORD CounterHelpTitle (4 bytes) — always 32-bit
119     // LONG DefaultScale (4 bytes)
120     // DWORD DetailLevel (4 bytes)
121     // DWORD CounterType (4 bytes)
122     // DWORD CounterSize (4 bytes)
123     // DWORD CounterOffset (4 bytes)
124     StructLayout PERF_COUNTER_DEFINITION = structLayout(JAVA_INT.withName("ByteLength"),
125             JAVA_INT.withName("CounterNameTitleIndex"), JAVA_INT.withName("CounterNameTitle"),
126             JAVA_INT.withName("CounterHelpTitleIndex"), JAVA_INT.withName("CounterHelpTitle"),
127             JAVA_INT.withName("DefaultScale"), JAVA_INT.withName("DetailLevel"), JAVA_INT.withName("CounterType"),
128             JAVA_INT.withName("CounterSize"), JAVA_INT.withName("CounterOffset"));
129 
130     long PERF_COUNTER_DEF_ByteLength = PERF_COUNTER_DEFINITION
131             .byteOffset(MemoryLayout.PathElement.groupElement("ByteLength"));
132     long PERF_COUNTER_DEF_CounterNameTitleIndex = PERF_COUNTER_DEFINITION
133             .byteOffset(MemoryLayout.PathElement.groupElement("CounterNameTitleIndex"));
134     long PERF_COUNTER_DEF_CounterSize = PERF_COUNTER_DEFINITION
135             .byteOffset(MemoryLayout.PathElement.groupElement("CounterSize"));
136     long PERF_COUNTER_DEF_CounterOffset = PERF_COUNTER_DEFINITION
137             .byteOffset(MemoryLayout.PathElement.groupElement("CounterOffset"));
138 
139     // PERF_INSTANCE_DEFINITION: describes an instance of a performance object
140     // https://learn.microsoft.com/en-us/windows/win32/api/winperf/ns-winperf-perf_instance_definition
141     //
142     // Fields:
143     // DWORD ByteLength (4 bytes)
144     // DWORD ParentObjectTitleIndex (4 bytes)
145     // DWORD ParentObjectInstance (4 bytes)
146     // LONG UniqueID (4 bytes)
147     // DWORD NameOffset (4 bytes)
148     // DWORD NameLength (4 bytes)
149     StructLayout PERF_INSTANCE_DEFINITION = structLayout(JAVA_INT.withName("ByteLength"),
150             JAVA_INT.withName("ParentObjectTitleIndex"), JAVA_INT.withName("ParentObjectInstance"),
151             JAVA_INT.withName("UniqueID"), JAVA_INT.withName("NameOffset"), JAVA_INT.withName("NameLength"));
152 
153     long PERF_INSTANCE_DEF_ByteLength = PERF_INSTANCE_DEFINITION
154             .byteOffset(MemoryLayout.PathElement.groupElement("ByteLength"));
155     long PERF_INSTANCE_DEF_NameOffset = PERF_INSTANCE_DEFINITION
156             .byteOffset(MemoryLayout.PathElement.groupElement("NameOffset"));
157 
158     // PERF_COUNTER_BLOCK: header for the raw counter data block following each instance
159     // https://learn.microsoft.com/en-us/windows/win32/api/winperf/ns-winperf-perf_counter_block
160     //
161     // Fields:
162     // DWORD ByteLength (4 bytes)
163     // Followed by the actual counter data bytes at offsets defined by PERF_COUNTER_DEFINITION.CounterOffset
164     StructLayout PERF_COUNTER_BLOCK = structLayout(JAVA_INT.withName("ByteLength"));
165 
166     long PERF_COUNTER_BLOCK_ByteLength = 0L;
167 }
