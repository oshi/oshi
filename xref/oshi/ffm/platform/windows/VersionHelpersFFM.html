34  /**
35   * FFM port of the Windows SDK versionhelpers.h inline functions. Uses {@link Kernel32FFM#VerSetConditionMask} and
36   * {@link Kernel32FFM#VerifyVersionInfoW} to determine the current operating system version.
37   */
38  public final class VersionHelpersFFM {
39  
40      private static final Logger LOG = LoggerFactory.getLogger(VersionHelpersFFM.class);
41  
42      private static final int ERROR_OLD_WIN_VERSION = 0x47E;
43  
44      private VersionHelpersFFM() {
45      }
46  
47      /**
48       * Tests whether the current OS version matches, or is greater than, the provided version information.
49       *
50       * @param wMajorVersion     The major version to test
51       * @param wMinorVersion     The minor version to test
52       * @param wServicePackMajor The service pack to test
53       * @return true if the current OS version matches or is greater than the provided version
54       */
55      public static boolean IsWindowsVersionOrGreater(int wMajorVersion, int wMinorVersion, int wServicePackMajor) {
56          try (Arena arena = Arena.ofConfined()) {
57              MemorySegment osvi = arena.allocate(OSVERSIONINFOEX);
58              osvi.fill((byte) 0);
59              osvi.set(JAVA_INT, 0, (int) OSVERSIONINFOEX.byteSize()); // dwOSVersionInfoSize
60              osvi.set(JAVA_INT, OSVERSIONINFOEX_MAJOR_VERSION_OFFSET, wMajorVersion);
61              osvi.set(JAVA_INT, OSVERSIONINFOEX_MINOR_VERSION_OFFSET, wMinorVersion);
62              osvi.set(JAVA_SHORT, OSVERSIONINFOEX_SP_MAJOR_OFFSET, (short) wServicePackMajor);
63  
64              long dwlConditionMask = 0;
65              dwlConditionMask = Kernel32FFM.VerSetConditionMask(dwlConditionMask, VER_MAJORVERSION,
66                      (byte) VER_GREATER_EQUAL);
67              dwlConditionMask = Kernel32FFM.VerSetConditionMask(dwlConditionMask, VER_MINORVERSION,
68                      (byte) VER_GREATER_EQUAL);
69              dwlConditionMask = Kernel32FFM.VerSetConditionMask(dwlConditionMask, VER_SERVICEPACKMAJOR,
70                      (byte) VER_GREATER_EQUAL);
71  
72              return Kernel32FFM.VerifyVersionInfoW(osvi, VER_MAJORVERSION | VER_MINORVERSION | VER_SERVICEPACKMAJOR,
73                      dwlConditionMask);
74          }
75      }
76  
77      /**
78       * @return true if the current OS version matches, or is greater than, the Windows Vista version.
79       */
80      public static boolean IsWindowsVistaOrGreater() {
81          return IsWindowsVersionOrGreater((byte) (WIN32_WINNT_VISTA >>> 8), (byte) WIN32_WINNT_VISTA, 0);
82      }
83

112     /**
113      * @return true if the current OS is a Windows Server release.
114      */
115     public static boolean IsWindowsServer() {
116         try (Arena arena = Arena.ofConfined()) {
117             MemorySegment osvi = arena.allocate(OSVERSIONINFOEX);
118             osvi.fill((byte) 0);
119             osvi.set(JAVA_INT, 0, (int) OSVERSIONINFOEX.byteSize()); // dwOSVersionInfoSize
120             osvi.set(JAVA_BYTE, OSVERSIONINFOEX_PRODUCT_TYPE_OFFSET, VER_NT_WORKSTATION);
121 
122             long dwlConditionMask = Kernel32FFM.VerSetConditionMask(0, VER_PRODUCT_TYPE, (byte) VER_EQUAL);
123 
124             if (Kernel32FFM.VerifyVersionInfoW(osvi, VER_PRODUCT_TYPE, dwlConditionMask)) {
125                 // Product type IS VER_NT_WORKSTATION, so not a server
126                 return false;
127             }
128             // VerifyVersionInfoW returned false; check why
129             int error = Kernel32FFM.GetLastError().orElse(0);
130             if (error == ERROR_OLD_WIN_VERSION) {
131                 // Condition not met: product type is not VER_NT_WORKSTATION, so it is a server
132                 return true;
133             }
134             LOG.debug("VerifyVersionInfoW failed with unexpected error code: {}", error);
135             return false;
136         }
137     }
