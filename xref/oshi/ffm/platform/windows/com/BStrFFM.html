18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;

24  /**
25   * FFM utilities for handling BSTR (Basic String) - the string type used in COM automation.
26   * <p>
27   * A BSTR is a pointer to a null-terminated Unicode string preceded by a 4-byte length prefix. The pointer points to the
28   * first character, not the length prefix.
29   * </p>
30   */
31  public final class BStrFFM extends WindowsForeignFunctions {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(BStrFFM.class);
34  
35      private static final SymbolLookup OLEAUT32 = lib("OleAut32.dll");
36  
37      private BStrFFM() {
38      }
39  
40      // SysAllocString
41      private static final MethodHandle SysAllocString = downcall(OLEAUT32, "SysAllocString", ADDRESS, ADDRESS);
42  
43      /**
44       * Allocates a new BSTR and copies the specified string into it.
45       *
46       * @param arena the arena for string conversion
47       * @param str   the string to convert
48       * @return the BSTR pointer, or NULL if allocation failed
49       */
50      public static MemorySegment fromString(Arena arena, String str) {
51          if (str == null) {
52              return MemorySegment.NULL;
53          }
54          return getOrDefault(() -> {
55              MemorySegment wideStr = toWideString(arena, str);
56              return (MemorySegment) SysAllocString.invokeExact(wideStr);
57          }, MemorySegment.NULL, LOG, "BStrFFM.fromString failed");
58      }
59  
60      // SysFreeString
61      private static final MethodHandle SysFreeString = downcall(OLEAUT32, "SysFreeString", null, ADDRESS);
62  
63      /**
64       * Frees a BSTR allocated by SysAllocString.
65       *
66       * @param bstr the BSTR to free
67       */
68      public static void free(MemorySegment bstr) {
69          if (bstr == null || bstr.equals(MemorySegment.NULL)) {
70              return;
71          }
72          // Block lambda (not an expression lambda) so invokeExact() on this void handle is in a statement context and
73          // its return type is inferred as void; an expression lambda infers Object -> WrongMethodTypeException (#3422).
74          // NOSONAR on the next line so S1602 does not collapse the block back to the buggy expression form.
75          runOrLog(() -> { // NOSONAR java:S1602 - block form required; expression lambda miscompiles void invokeExact
76              SysFreeString.invokeExact(bstr);
77          }, LOG, "BStrFFM.free failed");
78      }
79  
80      // SysStringLen
81      private static final MethodHandle SysStringLen = downcall(OLEAUT32, "SysStringLen", JAVA_INT, ADDRESS);
82  
83      /**
84       * Returns the length of a BSTR in characters (not bytes).
85       *
86       * @param bstr the BSTR
87       * @return the length in characters, or 0 if null
88       */
89      public static int length(MemorySegment bstr) {
90          if (bstr == null || bstr.equals(MemorySegment.NULL)) {
91              return 0;
92          }
93          return getIntOrDefault(() -> (int) SysStringLen.invokeExact(bstr), 0, LOG, "BStrFFM.length failed");
94      }
95  
96      /**
97       * Converts a BSTR to a Java String.
98       *
99       * @param bstr  the BSTR pointer
100      * @param arena the arena for memory reinterpretation
101      * @return the Java string, or empty string if null
102      */
103     public static String toString(MemorySegment bstr, Arena arena) {
104         if (bstr == null || bstr.equals(MemorySegment.NULL)) {
105             return "";
106         }
107         int len = length(bstr);
108         if (len == 0) {
109             return "";
110         }
111         // Reinterpret the BSTR memory to read the characters
112         MemorySegment strSeg = bstr.reinterpret((long) len * 2 + 2, arena, null);
113         StringBuilder sb = new StringBuilder(len);
114         for (int i = 0; i < len; i++) {
115             char c = strSeg.get(JAVA_CHAR, (long) i * 2);
116             sb.append(c);
117         }
118         return sb.toString();
119     }
