5   package oshi.ffm.platform.windows;
6   
7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_CHAR;
9   import static java.lang.foreign.ValueLayout.JAVA_INT;
10  
11  import java.lang.foreign.Arena;
12  import java.lang.foreign.MemorySegment;
13  import java.lang.foreign.SymbolLookup;
14  import java.lang.invoke.MethodHandle;
15  import java.util.ArrayList;
16  import java.util.Collections;
17  import java.util.List;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * FFM bindings for Shell32 functions.
24   */
25  public final class Shell32FFM extends WindowsForeignFunctions {
26  
27      private static final Logger LOG = LoggerFactory.getLogger(Shell32FFM.class);
28  
29      private static final SymbolLookup SHELL32 = lib("Shell32");
30  
31      // LPWSTR* CommandLineToArgvW(LPCWSTR lpCmdLine, int* pNumArgs)
32      private static final MethodHandle CommandLineToArgvW = downcall(SHELL32, "CommandLineToArgvW", ADDRESS, ADDRESS,
33              ADDRESS);
34  
35      /**
36       * Parses a Unicode command line string and returns an array of pointers to the command line arguments.
37       *
38       * @param commandLine The command line string to parse
39       * @return List of command line arguments, or empty list if parsing fails
40       */
41      public static List<String> CommandLineToArgv(String commandLine) {
42          if (commandLine == null || commandLine.isEmpty()) {
43              return Collections.emptyList();
44          }
45  
46          try (Arena arena = Arena.ofConfined()) {
47              MemorySegment cmdLineSegment = toWideString(arena, commandLine);
48              MemorySegment numArgsSegment = arena.allocate(JAVA_INT);
49  
50              MemorySegment argvPtr = (MemorySegment) CommandLineToArgvW.invokeExact(cmdLineSegment, numArgsSegment);
51  
52              if (argvPtr == null || argvPtr.address() == 0) {
53                  return Collections.emptyList();
54              }
55  
56              try {
57                  int numArgs = numArgsSegment.get(JAVA_INT, 0);
58                  if (numArgs <= 0) {
59                      return Collections.emptyList();
60                  }
61  
62                  List<String> args = new ArrayList<>(numArgs);
63                  // Reinterpret the pointer array with proper size
64                  MemorySegment argvArray = argvPtr.reinterpret(numArgs * ADDRESS.byteSize());
65  
66                  for (int i = 0; i < numArgs; i++) {
67                      MemorySegment argPtr = argvArray.get(ADDRESS, i * ADDRESS.byteSize());
68                      if (argPtr != null && argPtr.address() != 0) {
69                          // Calculate string length by finding null terminator
70                          // Reinterpret with large size to scan for null
71                          MemorySegment scanSegment = argPtr.reinterpret(32768); // Max scan size
72                          int charCount = 0;
73                          while (charCount < 16384 && scanSegment.get(JAVA_CHAR, (long) charCount * 2) != 0) {
74                              charCount++;
75                          }
76                          // Reinterpret to exact size needed (include null terminator)
77                          MemorySegment argSegment = argPtr.reinterpret((long) (charCount + 1) * 2);
78                          args.add(readWideString(argSegment));
79                      }
80                  }
81  
82                  return args;
83              } finally {
84                  // Free the memory allocated by CommandLineToArgvW
85                  Kernel32FFM.LocalFree(argvPtr);
86              }
87          } catch (Throwable t) {
88              LOG.debug("Shell32FFM.CommandLineToArgv failed: {}", t.getMessage());
89              return Collections.emptyList();
90          }
91      }
92  }
