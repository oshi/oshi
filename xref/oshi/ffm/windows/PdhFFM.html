7   import java.lang.foreign.MemoryLayout;
8   import java.lang.foreign.MemorySegment;
9   import java.lang.foreign.StructLayout;
10  import java.lang.foreign.SymbolLookup;
11  import java.lang.invoke.MethodHandle;
12  
13  import static java.lang.foreign.MemoryLayout.structLayout;
14  import static java.lang.foreign.ValueLayout.ADDRESS;
15  import static java.lang.foreign.ValueLayout.JAVA_INT;
16  import static java.lang.foreign.ValueLayout.JAVA_LONG;
17  import static java.lang.foreign.ValueLayout.JAVA_DOUBLE;
18  
19  public class PdhFFM extends WindowsForeignFunctions {
20  
21      private static final SymbolLookup Pdh = lib("Pdh");
22  
23      public static final int PDH_FMT_LARGE = 0x00000400;
24      public static final int PDH_MORE_DATA = 0x800007D2;
25  
26      private static final MethodHandle PdhOpenQuery = downcall(Pdh, "PdhOpenQueryW", JAVA_INT, ADDRESS, ADDRESS,
27              ADDRESS);
28  
29      public static int PdhOpenQuery(MemorySegment dataSource, MemorySegment userData, MemorySegment query)
30              throws Throwable {
31          return (int) PdhOpenQuery.invokeExact(dataSource, userData, query);
32      }
33  
34      private static final MethodHandle PdhAddEnglishCounter = downcall(Pdh, "PdhAddEnglishCounterW", JAVA_INT, ADDRESS,
35              ADDRESS, ADDRESS, ADDRESS);
36  
37      public static int PdhAddEnglishCounter(MemorySegment query, MemorySegment counterPath, MemorySegment userData,
38              MemorySegment counter) throws Throwable {
39          return (int) PdhAddEnglishCounter.invokeExact(query, counterPath, userData, counter);
40      }
41  
42      private static final MethodHandle PdhCollectQueryData = downcall(Pdh, "PdhCollectQueryData", JAVA_INT, ADDRESS);
43  
44      public static int PdhCollectQueryData(MemorySegment query) throws Throwable {
45          return (int) PdhCollectQueryData.invokeExact(query);
46      }
47  
48      private static final MethodHandle PdhGetFormattedCounterArray = downcall(Pdh, "PdhGetFormattedCounterArrayW",
49              JAVA_INT, ADDRESS, JAVA_INT, ADDRESS, ADDRESS, ADDRESS);
50  
51      public static int PdhGetFormattedCounterArray(MemorySegment counter, int format, MemorySegment bufferSize,
52              MemorySegment itemCount, MemorySegment buffer) throws Throwable {
53          return (int) PdhGetFormattedCounterArray.invokeExact(counter, format, bufferSize, itemCount, buffer);
54      }
55  
56      private static final MethodHandle PdhCloseQuery = downcall(Pdh, "PdhCloseQuery", JAVA_INT, ADDRESS);
57  
58      public static int PdhCloseQuery(MemorySegment query) throws Throwable {
59          return (int) PdhCloseQuery.invokeExact(query);
60      }
61  
62      public static final MemoryLayout PDH_FMT_COUNTERVALUE_UNION = MemoryLayout.unionLayout(
63              JAVA_INT.withName("longValue"), JAVA_DOUBLE.withName("doubleValue"), JAVA_LONG.withName("largeValue"),
64              ADDRESS.withName("AnsiStringValue"), ADDRESS.withName("WideStringValue"));
65  
66      public static final StructLayout PDH_FMT_COUNTERVALUE_LAYOUT = MemoryLayout.structLayout(
67              JAVA_INT.withName("CStatus"), MemoryLayout.paddingLayout(4), // align union to 8 bytes
68              PDH_FMT_COUNTERVALUE_UNION.withName("Value"));
69  
70      public static final StructLayout PDH_FMT_COUNTERVALUE_ITEM_LAYOUT = structLayout(ADDRESS.withName("szName"),
71              PDH_FMT_COUNTERVALUE_LAYOUT.withName("FmtValue"));

73  }
