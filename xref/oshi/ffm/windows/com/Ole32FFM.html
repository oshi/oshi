72      // CoInitializeEx
73      private static final MethodHandle CoInitializeEx = downcall(OLE32, "CoInitializeEx", JAVA_INT, ADDRESS, JAVA_INT);
74  
75      /**
76       * Initializes the COM library for use by the calling thread.
77       *
78       * @param coInit the concurrency model and initialization options (COINIT_*)
79       * @return HRESULT: S_OK if successful, S_FALSE if already initialized, or error code
80       */
81      public static OptionalInt CoInitializeEx(int coInit) {
82          try {
83              int hr = (int) CoInitializeEx.invokeExact(NULL, coInit);
84              return OptionalInt.of(hr);
85          } catch (Throwable t) {
86              LOG.debug("Ole32FFM.CoInitializeEx failed", t);
87              return OptionalInt.empty();
88          }
89      }
90  
91      // CoUninitialize
92      private static final MethodHandle CoUninitialize = downcall(OLE32, "CoUninitialize", null);
93  
94      /**
95       * Closes the COM library on the current thread.
96       */
97      public static void CoUninitialize() {
98          try {
99              CoUninitialize.invokeExact();
100         } catch (Throwable t) {
101             LOG.debug("Ole32FFM.CoUninitialize failed", t);
102         }
103     }

105     // CoInitializeSecurity
106     private static final MethodHandle CoInitializeSecurity = downcall(OLE32, "CoInitializeSecurity", JAVA_INT, ADDRESS,
107             JAVA_INT, ADDRESS, ADDRESS, JAVA_INT, JAVA_INT, ADDRESS, JAVA_INT, ADDRESS);

109     /**
110      * Registers security and sets the default security values for the process.
111      *
112      * @param authnLevel   the default authentication level
113      * @param impLevel     the default impersonation level
114      * @param capabilities additional capabilities (EOAC_*)
115      * @return HRESULT: S_OK if successful, RPC_E_TOO_LATE if already called, or error code
116      */
117     public static OptionalInt CoInitializeSecurity(int authnLevel, int impLevel, int capabilities) {
118         try {
119             int hr = (int) CoInitializeSecurity.invokeExact(NULL, // pSecDesc
120                     -1, // cAuthSvc
121                     NULL, // asAuthSvc
122                     NULL, // pReserved1
123                     authnLevel, // dwAuthnLevel
124                     impLevel, // dwImpLevel
125                     NULL, // pAuthList
126                     capabilities, // dwCapabilities
127                     NULL // pReserved3
128             );
129             return OptionalInt.of(hr);
130         } catch (Throwable t) {
131             LOG.debug("Ole32FFM.CoInitializeSecurity failed", t);
132             return OptionalInt.empty();
133         }
134     }

136     // CoCreateInstance
137     private static final MethodHandle CoCreateInstance = downcall(OLE32, "CoCreateInstance", JAVA_INT, ADDRESS, ADDRESS,
138             JAVA_INT, ADDRESS, ADDRESS);
139 
140     /**
141      * Creates a single uninitialized object of the class associated with a specified CLSID.
142      *
143      * @param arena  the arena for memory allocation
144      * @param clsid  the CLSID of the object to create
145      * @param clsctx the context in which the code that manages the object will run (CLSCTX_*)
146      * @param iid    the IID of the interface to obtain
147      * @return the interface pointer, or NULL if failed
148      */
149     public static MemorySegment CoCreateInstance(Arena arena, MemorySegment clsid, int clsctx, MemorySegment iid) {
150         try {
151             MemorySegment ppv = arena.allocate(ADDRESS);
152             int hr = (int) CoCreateInstance.invokeExact(clsid, NULL, clsctx, iid, ppv);
153             if (hr != S_OK) {
154                 LOG.debug("CoCreateInstance failed with HRESULT: 0x{}", Integer.toHexString(hr));
155                 return NULL;
156             }
157             return ppv.get(ADDRESS, 0);
158         } catch (Throwable t) {
159             LOG.debug("Ole32FFM.CoCreateInstance failed", t);
160             return NULL;
161         }
162     }

166             JAVA_INT, JAVA_INT, ADDRESS, JAVA_INT, JAVA_INT, ADDRESS, JAVA_INT);

168     /**
169      * Sets the authentication information for a proxy.
170      *
171      * @param pProxy       the proxy to set authentication on
172      * @param authnSvc     the authentication service (use -1 for default)
173      * @param authzSvc     the authorization service (use -1 for default)
174      * @param authnLevel   the authentication level
175      * @param impLevel     the impersonation level
176      * @param capabilities additional capabilities
177      * @return HRESULT
178      */
179     public static OptionalInt CoSetProxyBlanket(MemorySegment pProxy, int authnSvc, int authzSvc, int authnLevel,
180             int impLevel, int capabilities) {
181         try {
182             int hr = (int) CoSetProxyBlanket.invokeExact(pProxy, authnSvc, // dwAuthnSvc
183                     authzSvc, // dwAuthzSvc
184                     NULL, // pServerPrincName
185                     authnLevel, // dwAuthnLevel
186                     impLevel, // dwImpLevel
187                     NULL, // pAuthInfo
188                     capabilities // dwCapabilities
189             );
190             return OptionalInt.of(hr);
191         } catch (Throwable t) {
192             LOG.debug("Ole32FFM.CoSetProxyBlanket failed", t);
193             return OptionalInt.empty();
194         }
195     }

197     /**
198      * Checks if an HRESULT indicates success (non-negative value).
199      *
200      * @param hr the HRESULT value
201      * @return true if successful
202      */
203     public static boolean succeeded(int hr) {
204         return hr >= 0;
205     }
206 
207     /**
208      * Checks if an HRESULT indicates failure (negative value).
209      *
210      * @param hr the HRESULT value
211      * @return true if failed
212      */
213     public static boolean failed(int hr) {
214         return hr < 0;
215     }
