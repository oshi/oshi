5   package oshi.ffm.windows.com;
6   
7   /**
8    * Unchecked exception thrown when a COM operation fails with an unexpected HRESULT.
9    */
10  public class FfmComException extends RuntimeException {
11  
12      private static final long serialVersionUID = 1L;
13  
14      private final int hresult;
15  
16      public FfmComException(String message, int hresult) {
17          super(message + " (HRESULT: 0x" + Integer.toHexString(hresult) + ")");
18          this.hresult = hresult;
19      }
20  
21      public int getHresult() {
22          return hresult;
23      }
24  }
