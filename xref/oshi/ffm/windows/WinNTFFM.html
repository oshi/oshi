5   package oshi.ffm.windows;
6   
7   import java.lang.foreign.MemoryLayout;
8   import java.lang.foreign.StructLayout;
9   
10  import static java.lang.foreign.MemoryLayout.structLayout;
11  import static java.lang.foreign.ValueLayout.JAVA_INT;
12  import static java.lang.foreign.ValueLayout.JAVA_SHORT;
13  import static java.lang.foreign.ValueLayout.JAVA_LONG;
14  
15  public interface WinNTFFM {
16  
17      int EVENTLOG_BACKWARDS_READ = 0x0008;
18      int EVENTLOG_SEQUENTIAL_READ = 0x0001;
19      int KEY_READ = 0x20019;
20      int KEY_WOW64_64KEY = 0x0100;
21      int KEY_WOW64_32KEY = 0x0200;
22      int REG_SZ = 1;
23      int REG_EXPAND_SZ = 2;
24      int REG_DWORD = 4;
25      int SE_PRIVILEGE_ENABLED = 0x00000002;
26      int TOKEN_QUERY = 0x0008;
27      int TOKEN_ADJUST_PRIVILEGES = 0x0020;
28      int TokenElevation = 20;
29  
30      StructLayout EVENTLOGRECORD = structLayout(JAVA_INT.withName("Length"), JAVA_INT.withName("Reserved"),
31              JAVA_INT.withName("RecordNumber"), JAVA_INT.withName("TimeGenerated"), JAVA_INT.withName("TimeWritten"),
32              JAVA_INT.withName("EventID"), JAVA_SHORT.withName("EventType"), JAVA_SHORT.withName("NumStrings"),
33              JAVA_SHORT.withName("EventCategory"), JAVA_SHORT.withName("ReservedFlags"), MemoryLayout.paddingLayout(4),
34              JAVA_INT.withName("ClosingRecordNumber"), JAVA_INT.withName("StringOffset"),
35              JAVA_INT.withName("UserSidLength"), JAVA_INT.withName("UserSidOffset"), JAVA_INT.withName("DataLength"),
36              JAVA_INT.withName("DataOffset"));
37  
38      StructLayout LUID = structLayout(JAVA_INT.withName("LowPart"), JAVA_INT.withName("HighPart"));
39  
40      StructLayout LUID_AND_ATTRIBUTES = structLayout(LUID.withName("Luid"), JAVA_INT.withName("Attributes"));
41  
42      StructLayout PERFORMANCE_INFORMATION = structLayout(JAVA_INT.withName("cb"), MemoryLayout.paddingLayout(4),
43              JAVA_LONG.withName("CommitTotal"), JAVA_LONG.withName("CommitLimit"), JAVA_LONG.withName("CommitPeak"),
44              JAVA_LONG.withName("PhysicalTotal"), JAVA_LONG.withName("PhysicalAvailable"),
45              JAVA_LONG.withName("SystemCache"), JAVA_LONG.withName("KernelTotal"), JAVA_LONG.withName("KernelPaged"),
46              JAVA_LONG.withName("KernelNonpaged"), JAVA_LONG.withName("PageSize"), JAVA_INT.withName("HandleCount"),
47              JAVA_INT.withName("ProcessCount"), JAVA_INT.withName("ThreadCount"), MemoryLayout.paddingLayout(4));
48  
49      StructLayout TOKEN_PRIVILEGES = structLayout(JAVA_INT.withName("PrivilegeCount"),
50              LUID_AND_ATTRIBUTES.withName("Privileges"));
51  
52      StructLayout TOKEN_ELEVATION = structLayout(JAVA_INT.withName("TokenIsElevated"));
53  
54      long OFFSET_TIME_GENERATED = EVENTLOGRECORD.byteOffset(MemoryLayout.PathElement.groupElement("TimeGenerated"));
55  
56      long OFFSET_EVENTID = EVENTLOGRECORD.byteOffset(MemoryLayout.PathElement.groupElement("EventID"));
57  
58      long OFFSET_LENGTH = EVENTLOGRECORD.byteOffset(MemoryLayout.PathElement.groupElement("Length"));
59  
60      long TOKEN_PRIVILEGES_PRIVILEGE_COUNT_OFFSET = TOKEN_PRIVILEGES
61              .byteOffset(MemoryLayout.PathElement.groupElement("PrivilegeCount"));
62  
63      long TOKEN_PRIVILEGES_LUID_OFFSET = TOKEN_PRIVILEGES.byteOffset(MemoryLayout.PathElement.groupElement("Privileges"),
64              MemoryLayout.PathElement.groupElement("Luid"));
65  
66      long TOKEN_PRIVILEGES_ATTRIBUTES_OFFSET = TOKEN_PRIVILEGES.byteOffset(
67              MemoryLayout.PathElement.groupElement("Privileges"), MemoryLayout.PathElement.groupElement("Attributes"));
68  
69  }
