5   package oshi.ffm.windows;
6   
7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   
10  import java.lang.foreign.MemorySegment;
11  import java.lang.foreign.SymbolLookup;
12  import java.lang.invoke.MethodHandle;
13  
14  public final class Wtsapi32FFM extends WindowsForeignFunctions {
15  
16      private static final SymbolLookup WTS = lib("Wtsapi32");
17  
18      public static final MemorySegment WTS_CURRENT_SERVER_HANDLE = MemorySegment.NULL;
19  
20      private static final MethodHandle WTSEnumerateSessionsW = downcall(WTS, "WTSEnumerateSessionsW", JAVA_INT, ADDRESS,
21              JAVA_INT, JAVA_INT, ADDRESS, ADDRESS);
22  
23      public static boolean WTSEnumerateSessions(MemorySegment hServer, int reserved, int version,
24              MemorySegment ppSessionInfo, MemorySegment pCount) throws Throwable {
25          return isSuccess((int) WTSEnumerateSessionsW.invokeExact(hServer, reserved, version, ppSessionInfo, pCount));
26      }
27  
28      private static final MethodHandle WTSQuerySessionInformationW = downcall(WTS, "WTSQuerySessionInformationW",
29              JAVA_INT, ADDRESS, JAVA_INT, JAVA_INT, ADDRESS, ADDRESS);
30  
31      public static boolean WTSQuerySessionInformation(MemorySegment hServer, int sessionId, int wtsInfoClass,
32              MemorySegment ppBuffer, MemorySegment pBytesReturned) throws Throwable {
33          return isSuccess((int) WTSQuerySessionInformationW.invokeExact(hServer, sessionId, wtsInfoClass, ppBuffer,
34                  pBytesReturned));
35      }
36  
37      private static final MethodHandle WTSFreeMemory = downcall(WTS, "WTSFreeMemory", null, ADDRESS);
38  
39      public static void WTSFreeMemory(MemorySegment pMemory) throws Throwable {
40          WTSFreeMemory.invokeExact(pMemory);
41      }
42  }
