5   package oshi.ffm;
6   
7   import static java.lang.foreign.ValueLayout.JAVA_BYTE;
8   
9   import java.lang.foreign.Arena;
10  import java.lang.foreign.FunctionDescriptor;
11  import java.lang.foreign.Linker;
12  import java.lang.foreign.MemoryLayout;
13  import java.lang.foreign.MemoryLayout.PathElement;
14  import java.lang.foreign.MemorySegment;
15  import java.lang.foreign.StructLayout;
16  import java.lang.foreign.SymbolLookup;
17  import java.lang.invoke.MethodHandle;
18  import java.lang.invoke.VarHandle;
19  
20  /**
21   * Base class providing utility methods for working with the Java Foreign Function and Memory (FFM) API.
22   * <p>
23   * Subclasses use these helpers to load native libraries and frameworks, create downcall handles, and read data from
24   * native memory segments.
25   */
26  public abstract class ForeignFunctions {
27  
28      /** The native linker for the current platform. */
29      protected static final Linker LINKER = Linker.nativeLinker();
30  
31      /** A shared auto arena used for library symbol lookups. */
32      protected static final Arena LIBRARY_ARENA = Arena.ofAuto();
33  
34      /** Symbol lookup for libraries already loaded into the current process. */
35      protected static final SymbolLookup SYMBOL_LOOKUP = SymbolLookup.loaderLookup();
36  
37      /** Not intended for instantiation. */
38      protected ForeignFunctions() {
39      }
40  
41      /**
42       * Lookup a native library by simple name, mapping it to the platform-specific filename (e.g. {@code "c"} →
43       * {@code "libc.so"} on Linux).
44       *
45       * @param libraryName the platform-independent library name
46       * @return the symbol lookup for the library
47       */
48      public static SymbolLookup libraryLookup(String libraryName) {
49          return SymbolLookup.libraryLookup(System.mapLibraryName(libraryName), LIBRARY_ARENA);
50      }
51  
52      /**
53       * Reinterpret a raw native pointer as a struct of the given layout, scoped to the provided arena.
54       *
55       * @param pointer the native pointer
56       * @param layout  the struct layout
57       * @param arena   the arena to scope the resulting segment to
58       * @return a memory segment over the struct, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
59       */
60      public static MemorySegment getStructFromNativePointer(MemorySegment pointer, StructLayout layout, Arena arena) {
61          if (pointer == null || pointer.equals(MemorySegment.NULL)) {
62              return null;
63          }
64          return MemorySegment.ofAddress(pointer.address()).reinterpret(layout.byteSize(), arena, null);
65      }
66  
67      /**
68       * Read a null-terminated UTF-8 string from a raw native pointer.
69       *
70       * @param pointer the native pointer
71       * @param arena   the arena to scope the reinterpreted segment to
72       * @return the Java string, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
73       */
74      public static String getStringFromNativePointer(MemorySegment pointer, Arena arena) {
75          if (pointer == null || pointer.equals(MemorySegment.NULL)) {
76              return null;
77          }
78          // More than enough space for 255 UTF characters
79          return MemorySegment.ofAddress(pointer.address()).reinterpret(1024, arena, null).getString(0);
80      }
81  
82      /**
83       * Copy {@code length} bytes from a raw native pointer into a Java byte array.
84       *
85       * @param pointer the native pointer
86       * @param length  the number of bytes to copy
87       * @param arena   the arena to scope the reinterpreted segment to
88       * @return the byte array, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
89       */
90      public static byte[] getByteArrayFromNativePointer(MemorySegment pointer, long length, Arena arena) {
91          if (pointer == null || pointer.equals(MemorySegment.NULL)) {
92              return null;
93          }
94          MemorySegment bytesSegment = MemorySegment.ofAddress(pointer.address()).reinterpret(length, arena, null);
95          byte[] result = new byte[(int) length];
96          MemorySegment.copy(bytesSegment, JAVA_BYTE, 0, result, 0, (int) length);
97          return result;
98      }
99  
100     /**
101      * Lookup a library by name in the global arena.
102      *
103      * @param name the library name
104      * @return the symbol lookup for the library
105      */
106     public static SymbolLookup lib(String name) {
107         return SymbolLookup.libraryLookup(name, Arena.global());
108     }
109 
110     /**
111      * Create a downcall handle for a symbol in a library.
112      *
113      * @param lib        the symbol lookup
114      * @param symbol     the symbol name
115      * @param resLayout  the return layout
116      * @param argLayouts the argument layouts
117      * @return the method handle
118      */
119     public static MethodHandle downcall(SymbolLookup lib, String symbol, MemoryLayout resLayout,
120             MemoryLayout... argLayouts) {
121         MemorySegment sym = lib.findOrThrow(symbol);
122         FunctionDescriptor fd = (resLayout == null) ? FunctionDescriptor.ofVoid(argLayouts)
123                 : FunctionDescriptor.of(resLayout, argLayouts);
124         return LINKER.downcallHandle(sym, fd);
125     }
126 
127     /** Linker option to capture {@code errno} after a native call. */
128     protected static final Linker.Option CAPTURE_CALL_STATE = Linker.Option.captureCallState("errno");
129 
130     /** Layout of the captured call state segment, containing {@code errno}. */
131     public static final StructLayout CAPTURED_STATE_LAYOUT = Linker.Option.captureStateLayout();
132 
133     /** Handle to read the {@code errno} field from a captured call state segment. */
134     protected static final VarHandle ERRNO_HANDLE = CAPTURED_STATE_LAYOUT.varHandle(PathElement.groupElement("errno"));
135 
136     /**
137      * Read the {@code errno} value from a captured call state segment.
138      *
139      * @param callState the memory segment returned by a call made with {@link #CAPTURE_CALL_STATE}
140      * @return the {@code errno} value
141      */
142     public static int getErrno(MemorySegment callState) {
143         return (int) ERRNO_HANDLE.get(callState, 0);
144     }
145 }
