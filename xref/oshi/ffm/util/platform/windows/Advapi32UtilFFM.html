68      /**
69       * Checks whether the current process is running with elevated privileges.
70       *
71       * @return true if the process is elevated, false otherwise
72       */
73      public static boolean isCurrentProcessElevated() {
74          try (Arena arena = Arena.ofConfined()) {
75  
76              Optional<MemorySegment> hProcessOpt = Kernel32FFM.GetCurrentProcess();
77              if (hProcessOpt.isEmpty()) {
78                  return false;
79              }
80              MemorySegment hProcess = hProcessOpt.get();
81  
82              MemorySegment hTokenPtr = arena.allocate(ADDRESS);
83  
84              if (!OpenProcessToken(hProcess, WinNTFFM.TOKEN_QUERY, hTokenPtr)) {
85                  return false;
86              }
87  
88              MemorySegment hToken = hTokenPtr.get(ADDRESS, 0);
89  
90              try {
91                  MemorySegment elevation = arena.allocate(WinNTFFM.TOKEN_ELEVATION);
92                  MemorySegment returnLength = arena.allocate(JAVA_INT);
93  
94                  boolean success = GetTokenInformation(hToken, WinNTFFM.TokenElevation, elevation,
95                          (int) WinNTFFM.TOKEN_ELEVATION.byteSize(), returnLength);
96  
97                  if (!success) {
98                      return false;
99                  }
100 
101                 int tokenIsElevated = elevation.get(JAVA_INT,
102                         WinNTFFM.TOKEN_ELEVATION.byteOffset(MemoryLayout.PathElement.groupElement("TokenIsElevated")));
103 
104                 return tokenIsElevated > 0;
105             } finally {
106                 Kernel32FFM.CloseHandle(hToken);
107             }
108         } catch (Throwable t) {
109             LOG.debug("Advapi32FFM.isCurrentProcessElevated failed", t);
110             return false;
111         }
112     }

114     /**
115      * Enumerates the subkey names under an open registry key.
116      *
117      * @param hKey handle to an open registry key
118      * @return array of subkey names
119      * @throws Throwable if the native call fails
120      */
121     public static String[] registryGetKeys(MemorySegment hKey) throws Throwable {
122         try (Arena arena = Arena.ofConfined()) {
123             MemorySegment lpcSubKeys = arena.allocate(JAVA_INT);
124             MemorySegment lpcMaxSubKeyLen = arena.allocate(JAVA_INT);
125 
126             int rc = RegQueryInfoKey(hKey, NULL, NULL, NULL, lpcSubKeys, lpcMaxSubKeyLen, NULL, NULL, NULL, NULL, NULL,
127                     NULL);
128 
129             if (rc != ERROR_SUCCESS) {
130                 throw new Win32Exception(rc);
131             }
132 
133             int subKeyCount = lpcSubKeys.get(JAVA_INT, 0);
134             int maxNameLen = lpcMaxSubKeyLen.get(JAVA_INT, 0);
135 
136             List<String> keys = new ArrayList<>(subKeyCount);
137             for (int i = 0; i < subKeyCount; i++) {
138                 MemorySegment nameBuf = arena.allocate((maxNameLen + 1) * 2);
139                 MemorySegment nameLen = arena.allocate(JAVA_INT);
140                 nameLen.set(JAVA_INT, 0, maxNameLen + 1);
141 
142                 rc = RegEnumKeyEx(hKey, i, nameBuf, nameLen, NULL, NULL, NULL, NULL);
143                 if (rc != ERROR_SUCCESS) {
144                     throw new Win32Exception(rc);
145                 }
146 
147                 keys.add(readWideString(nameBuf));
148             }
149 
150             return keys.toArray(String[]::new);
151         }
152     }

154     /**
155      * Opens a registry key and enumerates its subkey names.
156      *
157      * @param rootKey         the root key handle (e.g., HKEY_LOCAL_MACHINE)
158      * @param keyPath         the registry key path
159      * @param samDesiredExtra additional access flags to combine with KEY_READ
160      * @return array of subkey names
161      * @throws Throwable if the native call fails
162      */
163     public static String[] registryGetKeys(MemorySegment rootKey, String keyPath, int samDesiredExtra)
164             throws Throwable {
165         try (Arena arena = Arena.ofConfined()) {
166             MemorySegment phkResult = arena.allocate(ADDRESS);
167             int rc = RegOpenKeyEx(rootKey, toWideString(arena, keyPath), 0, KEY_READ | samDesiredExtra, phkResult);
168             if (rc != ERROR_SUCCESS) {
169                 throw new Win32Exception(rc);
170             }
171 
172             MemorySegment hKey = phkResult.get(ADDRESS, 0);
173             try {
174                 return registryGetKeys(hKey);
175             } finally {
176                 rc = RegCloseKey(hKey);
177                 if (rc != ERROR_SUCCESS) {
178                     LOG.warn("Failed to close registry key, error code: {}", rc);
179                 }
180             }
181         }
182     }

185      * Reads a REG_DWORD value from an open registry key.
186      *
187      * @param hKey      handle to an open registry key
188      * @param valueName the value name to read
189      * @return the DWORD value
190      * @throws Throwable if the native call fails
191      */
192     public static int registryGetDword(MemorySegment hKey, String valueName) throws Throwable {
193         try (Arena arena = Arena.ofConfined()) {
194             MemorySegment pData = arena.allocate(JAVA_INT);
195             MemorySegment lpType = arena.allocate(JAVA_INT);
196             MemorySegment lpcbData = arena.allocate(JAVA_INT);
197             lpcbData.set(JAVA_INT, 0, Integer.BYTES);
198 
199             int rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, pData, lpcbData);
200             checkSuccess(rc, ERROR_INSUFFICIENT_BUFFER);
201             return pData.get(JAVA_INT, 0);
202         }
203     }

270     /**
271      * Reads a registry value of any supported type from an open registry key.
272      *
273      * @param hKey      handle to an open registry key
274      * @param valueName the value name to read
275      * @return the value (Integer for REG_DWORD, Long for REG_QWORD, byte[] for REG_BINARY, String for
276      *         REG_SZ/REG_EXPAND_SZ), or null if unsupported
277      * @throws Throwable if the native call fails
278      */
279     public static Object registryGetValue(MemorySegment hKey, String valueName) throws Throwable {
280         try (Arena arena = Arena.ofConfined()) {
281             MemorySegment lpType = arena.allocate(JAVA_INT);
282             MemorySegment lpcbData = arena.allocate(JAVA_INT);
283 
284             int rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, MemorySegment.NULL, lpcbData);
285             checkSuccess(rc, ERROR_INSUFFICIENT_BUFFER);
286 
287             int type = lpType.get(JAVA_INT, 0);
288             int size = lpcbData.get(JAVA_INT, 0);
289 
290             return switch (type) {
291                 case REG_SZ, REG_EXPAND_SZ -> registryGetString(hKey, valueName, size);
292                 case REG_DWORD -> registryGetDword(hKey, valueName);
293                 case REG_QWORD -> registryGetQword(hKey, valueName);
294                 case REG_BINARY -> registryGetBinary(hKey, valueName, size);
295                 default -> {
296                     LOG.warn("Unsupported registry data type {} for {}", type, valueName);
297                     yield null;
298                 }
299             };
300         }
301     }

303     /**
304      * Reads a registry value from the given root key and path.
305      *
306      * @param rootKey   The root key handle (e.g., HKEY_LOCAL_MACHINE)
307      * @param keyPath   The registry key path
308      * @param valueName The value name to read
309      * @return The value object (Integer for REG_DWORD, Long for REG_QWORD, byte[] for REG_BINARY, String for
310      *         REG_SZ/REG_EXPAND_SZ), or null on failure
311      */
312     public static Object registryGetValue(MemorySegment rootKey, String keyPath, String valueName) {
313         return callInArenaOrDefault(arena -> {
314             MemorySegment phkResult = arena.allocate(ADDRESS);
315             int rc = RegOpenKeyEx(rootKey, toWideString(arena, keyPath), 0, KEY_READ, phkResult);
316             if (rc != ERROR_SUCCESS) {
317                 return null;
318             }
319             MemorySegment hKey = phkResult.get(ADDRESS, 0);
320             try {
321                 return registryGetValue(hKey, valueName);
322             } finally {
323                 int closeRc = RegCloseKey(hKey);
324                 if (closeRc != ERROR_SUCCESS) {
325                     LOG.debug("Failed to close registry key {}: error {}", keyPath, closeRc);
326                 }
327             }
328         }, LOG, TRACE, "Failed to read registry value", null);
329     }

331     /**
332      * Checks whether a registry value exists under the given root key and path.
333      *
334      * @param rootKey   The root key handle (e.g., HKEY_LOCAL_MACHINE)
335      * @param keyPath   The registry key path
336      * @param valueName The value name to check
337      * @return true if the value exists, false otherwise
338      */
339     public static boolean registryValueExists(MemorySegment rootKey, String keyPath, String valueName) {
340         return callInArenaBooleanOrDefault(arena -> {
341             MemorySegment phkResult = arena.allocate(ADDRESS);
342             int rc = RegOpenKeyEx(rootKey, toWideString(arena, keyPath), 0, KEY_READ, phkResult);
343             if (rc != ERROR_SUCCESS) {
344                 return false;
345             }
346             MemorySegment hKey = phkResult.get(ADDRESS, 0);
347             try {
348                 MemorySegment lpType = arena.allocate(JAVA_INT);
349                 rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, MemorySegment.NULL,
350                         arena.allocate(JAVA_INT));
351                 return rc == ERROR_SUCCESS || rc == ERROR_MORE_DATA || rc == ERROR_INSUFFICIENT_BUFFER;
352             } finally {
353                 int closeRc = RegCloseKey(hKey);
354                 if (closeRc != ERROR_SUCCESS) {
355                     LOG.debug("Failed to close registry key {}: error {}", keyPath, closeRc);
356                 }
357             }
358         }, LOG, TRACE, "Failed to check registry value exists", false);
359     }

361     /**
362      * Reads a REG_MULTI_SZ value from an open registry key.
363      *
364      * @param rootKey   The root key handle (e.g., HKEY_LOCAL_MACHINE)
365      * @param keyPath   The registry key path
366      * @param valueName The value name to read
367      * @return An array of strings from the multi-sz value, or an empty array on failure
368      */
369     public static String[] registryGetStringArray(MemorySegment rootKey, String keyPath, String valueName) {
370         try (Arena arena = Arena.ofConfined()) {
371             MemorySegment phkResult = arena.allocate(ADDRESS);
372             int rc = RegOpenKeyEx(rootKey, toWideString(arena, keyPath), 0, KEY_READ, phkResult);
373             if (rc != ERROR_SUCCESS) {
374                 throw new Win32Exception(rc);
375             }
376             MemorySegment hKey = phkResult.get(ADDRESS, 0);
377             try {
378                 MemorySegment lpType = arena.allocate(JAVA_INT);
379                 MemorySegment lpcbData = arena.allocate(JAVA_INT);
380 
381                 rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, MemorySegment.NULL, lpcbData);
382                 if (rc != ERROR_SUCCESS && rc != ERROR_INSUFFICIENT_BUFFER && rc != ERROR_MORE_DATA) {
383                     throw new Win32Exception(rc);
384                 }
385                 if (lpType.get(JAVA_INT, 0) != REG_MULTI_SZ) {
386                     throw new IllegalStateException(
387                             "Unexpected registry type " + lpType.get(JAVA_INT, 0) + ", expected REG_MULTI_SZ");
388                 }
389 
390                 int size = lpcbData.get(JAVA_INT, 0);
391                 MemorySegment data;
392                 rc = ERROR_MORE_DATA;
393                 for (int attempt = 0; attempt < 3 && rc != ERROR_SUCCESS; attempt++) {
394                     // Allocate extra for double-null terminator
395                     data = arena.allocate(size + 4L);
396                     data.fill((byte) 0);
397                     lpcbData.set(JAVA_INT, 0, size + 4);
398 
399                     rc = RegQueryValueEx(hKey, toWideString(arena, valueName), 0, lpType, data, lpcbData);
400                     if (rc == ERROR_MORE_DATA || rc == ERROR_INSUFFICIENT_BUFFER) {
401                         size = lpcbData.get(JAVA_INT, 0);
402                         continue;
403                     }
404                     if (rc != ERROR_SUCCESS) {
405                         throw new Win32Exception(rc);
406                     }
407 
408                     // Parse multi-sz: null-delimited wide strings, double-null terminated
409                     int bytesWritten = lpcbData.get(JAVA_INT, 0);
410                     List<String> result = new ArrayList<>();
411                     long offset = 0;
412                     while (offset < bytesWritten) {
413                         String s = readWideString(data.asSlice(offset, bytesWritten - offset));
414                         if (s.isEmpty()) {
415                             break;
416                         }
417                         result.add(s);
418                         offset += ((long) s.length() + 1) * 2; // chars + null, 2 bytes each
419                     }
420                     return result.toArray(new String[0]);
421                 }
422                 throw new Win32Exception(rc);
423             } finally {
424                 int closeRc = RegCloseKey(hKey);
425                 if (closeRc != ERROR_SUCCESS) {
426                     LOG.debug("Failed to close registry key {}\\\\{}: error {}", keyPath, valueName, closeRc);
427                 }
428             }
429         } catch (Throwable t) {
430             LOG.warn("Failed to read registry string array {}\\\\{}: {}", keyPath, valueName, t.getMessage());
431             return new String[0];
432         }
433     }

435     /**
436      * Queries the system boot time from the Windows Event Log.
437      *
438      * @return the boot time as a Unix epoch timestamp in seconds
439      */
440     public static long querySystemBootTime() {
441         String eventLog = SYSTEM_LOG.get();
442         try (Arena arena = Arena.ofConfined()) {
443             Optional<MemorySegment> hEventLogOpt = Advapi32FFM.OpenEventLog(arena, eventLog);
444             if (hEventLogOpt.isEmpty()) {
445                 LOG.warn("Unable to open system Event Log. Falling back to uptime.");
446                 return System.currentTimeMillis() / 1000L - Kernel32UtilFFM.querySystemUptime();
447             }
448 
449             MemorySegment hEventLog = hEventLogOpt.get();
450             try {
451                 int bufSize = 64 * 1024;
452                 MemorySegment buffer = arena.allocate(bufSize);
453                 MemorySegment bytesRead = arena.allocate(JAVA_INT);
454                 MemorySegment minBytesNeeded = arena.allocate(JAVA_INT);
455 
456                 long event6005Time = 0L;
457 
458                 long offsetEventId = WinNTFFM.OFFSET_EVENTID;
459                 long offsetTimeGenerated = WinNTFFM.OFFSET_TIME_GENERATED;
460                 long offsetLength = WinNTFFM.OFFSET_LENGTH;
461 
462                 while (Advapi32FFM.ReadEventLog(hEventLog,
463                         WinNTFFM.EVENTLOG_BACKWARDS_READ | WinNTFFM.EVENTLOG_SEQUENTIAL_READ, buffer, bufSize,
464                         bytesRead, minBytesNeeded)) {
465 
466                     int read = bytesRead.get(JAVA_INT, 0);
467                     int offset = 0;
468 
469                     while (offset < read) {
470                         MemorySegment eventRecord = buffer.asSlice(offset, WinNTFFM.EVENTLOGRECORD.byteSize());
471 
472                         int eventId = eventRecord.get(JAVA_INT, (int) offsetEventId);
473                         long timeGenerated = Integer
474                                 .toUnsignedLong(eventRecord.get(JAVA_INT, (int) offsetTimeGenerated));
475 
476                         if (eventId == 12) { // system boot
477                             return timeGenerated;
478                         } else if (eventId == 6005) { // event log startup
479                             if (event6005Time > 0) {
480                                 return event6005Time;
481                             }
482                             event6005Time = timeGenerated;
483                         }
484 
485                         // Advance to next record
486                         int length = eventRecord.get(JAVA_INT, (int) offsetLength);
487                         offset += length;
488                     }
489                 }
490 
491                 if (event6005Time > 0) {
492                     return event6005Time;
493                 }
494             } finally {
495                 // Close on every path, including a Throwable from a malformed record (matches the fallback contract).
496                 Advapi32FFM.CloseEventLog(hEventLog);
497             }
498         } catch (Throwable t) {
499             LOG.error("Exception while querying system boottime, fallback to boot time from uptime", t);
500         }
501         // Fallback: approximate boot time from uptime
502         return System.currentTimeMillis() / 1000L - Kernel32UtilFFM.querySystemUptime();
503     }

505     /**
506      * Queries and validates the configured system event log name.
507      *
508      * @return the event log name, or null if unavailable
509      */
510     public static String querySystemLog() {
511         String systemLog = GlobalConfig.get(GlobalConfig.OSHI_OS_WINDOWS_EVENTLOG, "System");
512         if (systemLog.isEmpty()) {
513             return null;
514         }
515 
516         try (Arena arena = Arena.ofConfined()) {
517             MemorySegment sourceName = toWideString(arena, systemLog);
518 
519             Optional<MemorySegment> hEventLog = Advapi32FFM.OpenEventLog(NULL, sourceName);
520             if (hEventLog.isEmpty()) {
521                 LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
522                         systemLog);
523                 return null;
524             }
525 
526             // Opened only to validate the configured log name; close it so the handle is not leaked.
527             Advapi32FFM.CloseEventLog(hEventLog.get());
528             return systemLog;
529         } catch (Throwable t) {
530             LOG.error("Exception while opening system event log", t);
531             return null;
532         }
