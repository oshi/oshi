51  /**
52   * Helper class to centralize the boilerplate portions of PDH counter setup using the FFM API.
53   * <p>
54   * This utility collects counter data with a single call to {@code PdhCollectQueryData}, which yields valid results for
55   * raw counters. OSHI queries raw counter values and computes rates in Java rather than using PDH formatted rate
56   * counters, which would require two samples separated by an interval.
57   */
58  public final class PerfDataUtilFFM {

60      private PerfDataUtilFFM() {
61      }

63      private static final Logger LOG = LoggerFactory.getLogger(PerfDataUtilFFM.class);

66      private static final Map<String, String> LOCALIZE_CACHE = new ConcurrentHashMap<>();

68      /**
69       * Query multiple PDH counter values in a single query, one per enum constant.
70       *
71       * @param <T>          An enum implementing {@link PdhCounterProperty}
72       * @param propertyEnum The enum class whose constants define the counters to query
73       * @param perfObject   The PDH object name (e.g., "Process")
74       * @return An {@link EnumMap} of values indexed by enum constant. May contain a subset of constants if individual
75       *         counters fail to add or read. Returns an empty map if the query itself fails to open or collect.
76       */
77      public static <T extends Enum<T> & PdhCounterProperty> Map<T, Long> queryCounters(Class<T> propertyEnum,
78              String perfObject) {
79          T[] props = propertyEnum.getEnumConstants();
80          EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
81  
82          try (Arena arena = Arena.ofConfined()) {
83              MemorySegment queryPtr = arena.allocate(ADDRESS);
84              checkSuccess(PdhOpenQuery(MemorySegment.NULL, MemorySegment.NULL, queryPtr));
85              MemorySegment query = null;
86  
87              try {
88                  query = queryPtr.get(ADDRESS, 0);
89  
90                  // Add all counters to the single query, skipping any that fail
91                  EnumMap<T, MemorySegment> counterHandles = new EnumMap<>(propertyEnum);
92                  EnumMap<T, Boolean> baseFlags = new EnumMap<>(propertyEnum);
93                  for (T prop : props) {
94                      String counterName = prop.getCounter();
95                      boolean isBase = PerfCounter.isBase(counterName);
96                      if (isBase) {
97                          counterName = PerfCounter.stripBaseSuffix(counterName);
98                      }
99                      String path = counterPath(perfObject, prop.getInstance(), counterName);
100                     try {
101                         counterHandles.put(prop, addEnglishCounter(arena, query, path));
102                         baseFlags.put(prop, isBase);
103                     } catch (Throwable t) {
104                         LOG.debug("Failed to add counter {}: {}", path, t.getMessage());
105                     }
106                 }
107                 if (counterHandles.isEmpty()) {
108                     return valueMap;
109                 }
110 
111                 // Collect, retrying the initial-sample PDH_NO_DATA race
112                 collectQueryDataWithRetry(query);
113 
114                 // Read all values, skipping any that fail
115                 for (var entry : counterHandles.entrySet()) {
116                     try {
117                         valueMap.put(entry.getKey(),
118                                 readRawCounterValue(arena, entry.getValue(), baseFlags.get(entry.getKey())));
119                     } catch (Throwable t) {
120                         LOG.debug("Failed to read counter for {}: {}", entry.getKey(), t.getMessage());
121                     }
122                 }
123 
124             } finally {
125                 if (query != null) {
126                     PdhCloseQuery(query);
127                 }
128             }
129 
130         } catch (Throwable t) {
131             LOG.debug("PDH queryCounters failed for {}: {}", perfObject, t.getMessage(), t);
132             return new EnumMap<>(propertyEnum);
133         }
134         return valueMap;
135     }

137     private static String counterPath(String object, String instance, String counter) {
138         StringBuilder path = new StringBuilder();
139         path.append('\\').append(object);
140         if (instance != null) {
141             path.append('(').append(instance).append(')');
142         }
143         path.append('\\').append(counter);
144         return path.toString();
145     }
146 
147     private static void collectQueryDataWithRetry(MemorySegment query) throws Throwable {
148         int ret = PdhCollectQueryData(query);
149         // Due to a race condition, the initial collection may fail with PDH_NO_DATA; retry with exponential backoff
150         // (matches the JNA PerfDataUtil backend).
151         int retries = 0;
152         while (ret == PDH_NO_DATA && retries++ < 3) {
153             Util.sleep(1L << retries);
154             ret = PdhCollectQueryData(query);
155         }
156         checkSuccess(ret);
157     }

159     private static MemorySegment addEnglishCounter(Arena arena, MemorySegment query, String path) throws Throwable {
160         MemorySegment counterPtr = arena.allocate(ADDRESS);
161         checkSuccess(PdhAddEnglishCounter(query, toWideString(arena, path), MemorySegment.NULL, counterPtr));
162         return counterPtr.get(ADDRESS, 0);
163     }

165     private static final long RAW_CSTATUS_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("CStatus"));
166     private static final long RAW_FIRST_VALUE_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("FirstValue"));
167     private static final long RAW_SECOND_VALUE_OFFSET = PDH_RAW_COUNTER_LAYOUT.byteOffset(groupElement("SecondValue"));
168 
169     private static long readRawCounterValue(Arena arena, MemorySegment counterHandle, boolean secondValue)
170             throws Throwable {
171         MemorySegment value = arena.allocate(PDH_RAW_COUNTER_LAYOUT);
172         checkSuccess(PdhGetRawCounterValue(counterHandle, MemorySegment.NULL, value));
173         int cStatus = value.get(JAVA_INT, RAW_CSTATUS_OFFSET);
174         if (cStatus != PDH_CSTATUS_VALID_DATA && cStatus != PDH_CSTATUS_NEW_DATA) {
175             throw new IllegalStateException(
176                     String.format(Locale.ROOT, "PDH raw counter CStatus invalid: 0x%08X", cStatus));
177         }
178         return value.get(JAVA_LONG, secondValue ? RAW_SECOND_VALUE_OFFSET : RAW_FIRST_VALUE_OFFSET);
179     }

181     /**
182      * Enumerate instances of a PDH performance object, filtering by the wildcard pattern from the first enum constant.
183      *
184      * @param perfObject     The PDH object name (e.g., "Process")
185      * @param instanceFilter The wildcard filter pattern (lowercase)
186      * @return A list of matching instance names, or an empty list on failure
187      */
188     static List<String> enumInstances(String perfObject, String instanceFilter) {
189         String localizedObject = localizeObjectName(perfObject);
190         return callInArenaOrDefault(arena -> {
191             MemorySegment objectName = toWideString(arena, localizedObject);
192             MemorySegment counterLen = arena.allocate(JAVA_INT);
193             MemorySegment instanceLen = arena.allocate(JAVA_INT);
194 
195             // First call: get required buffer sizes
196             counterLen.set(JAVA_INT, 0, 0);
197             instanceLen.set(JAVA_INT, 0, 0);
198             int result = PdhEnumObjectItemsW(MemorySegment.NULL, MemorySegment.NULL, objectName, MemorySegment.NULL,
199                     counterLen, MemorySegment.NULL, instanceLen, 100, 0);
200             if (result != ERROR_SUCCESS && result != PDH_MORE_DATA) {
201                 LOG.warn("Failed to enumerate instances for {}: 0x{}", perfObject, Integer.toHexString(result));
202                 return Collections.<String>emptyList();
203             }
204 
205             // Retry loop for race conditions (e.g., process/thread list changes)
206             MemorySegment counterBuf;
207             MemorySegment instanceBuf;
208             int retries = 0;
209             do {
210                 int cLen = counterLen.get(JAVA_INT, 0);
211                 int iLen = instanceLen.get(JAVA_INT, 0);
212                 counterBuf = cLen > 0 ? arena.allocate((long) cLen * 2) : MemorySegment.NULL;
213                 instanceBuf = iLen > 0 ? arena.allocate((long) iLen * 2) : MemorySegment.NULL;
214 
215                 result = PdhEnumObjectItemsW(MemorySegment.NULL, MemorySegment.NULL, objectName, counterBuf, counterLen,
216                         instanceBuf, instanceLen, 100, 0);
217                 if (result == PDH_MORE_DATA) {
218                     counterLen.set(JAVA_INT, 0, counterLen.get(JAVA_INT, 0) + 1024);
219                     instanceLen.set(JAVA_INT, 0, instanceLen.get(JAVA_INT, 0) + 1024);
220                 }
221             } while (result == PDH_MORE_DATA && ++retries < 5);
222 
223             if (result != ERROR_SUCCESS) {
224                 LOG.warn("Failed to enumerate instances for {}: 0x{}", localizedObject, Integer.toHexString(result));
225                 return Collections.<String>emptyList();
226             }
227 
228             List<String> instances = parseMultiSz(instanceBuf);
229             instances.removeIf(i -> !Util.wildcardMatch(i.toLowerCase(Locale.ROOT), instanceFilter));
230             return instances;
231         }, LOG, LogLevel.WARN, "Failed to enumerate instances for " + localizedObject, Collections.emptyList());
232     }

234     /**
235      * Parse a null-terminated double-null-terminated wide string list (multi-sz) into a list of strings.
236      *
237      * @param buf the memory segment containing the multi-sz wide string data
238      * @return a list of parsed strings
239      */
240     private static List<String> parseMultiSz(MemorySegment buf) {
241         List<String> result = new ArrayList<>();
242         if (buf == null || buf.equals(MemorySegment.NULL)) {
243             return result;
244         }
245         long offset = 0;
246         long size = buf.byteSize();
247         while (offset < size) {
248             StringBuilder sb = new StringBuilder();
249             while (offset < size) {
250                 char c = buf.get(JAVA_CHAR, offset);
251                 offset += 2;
252                 if (c == '\0') {
253                     break;
254                 }
255                 sb.append(c);
256             }
257             if (sb.isEmpty()) {
258                 break; // double null = end of list
259             }
260             result.add(sb.toString());
261         }
262         return result;
263     }

267      *
268      * @param <T>          An enum implementing {@link PdhCounterWildcardProperty}
269      * @param propertyEnum The enum class whose constants define the counters to query. The first constant defines the

271      * @param perfObject   The PDH object name (e.g., "Process")

277         return queryWildcardCounters(propertyEnum, perfObject, null);

280     /**
281      * Query wildcard PDH counter values for all instances matching the given filter or the default from the first enum
282      * constant.
283      *
284      * @param <T>          An enum implementing {@link PdhCounterWildcardProperty}
285      * @param propertyEnum The enum class whose constants define the counters to query. The first constant defines the
286      *                     default instance filter; remaining constants define counter names.
287      * @param perfObject   The PDH object name (e.g., "Process")
288      * @param customFilter A custom instance filter to use. If null, uses the first element of the property enum.
289      * @return A pair of (instances, valueMap) where valueMap is indexed by enum constant (excluding the first). Returns
290      *         empty list and empty map on failure.
291      */
292     public static <T extends Enum<T> & PdhCounterWildcardProperty> Pair<List<String>, Map<T, List<Long>>> queryWildcardCounters(
293             Class<T> propertyEnum, String perfObject, String customFilter) {
294 
295         T[] props = propertyEnum.getEnumConstants();
296         if (props.length < 2) {
297             throw new IllegalArgumentException("Enum " + propertyEnum.getName()
298                     + " must have at least two elements, an instance filter and a counter.");
299         }
300         String instanceFilter = Util.isBlank(customFilter) ? props[0].getCounter().toLowerCase(Locale.ROOT)
301                 : customFilter.toLowerCase(Locale.ROOT);
302 
303         List<String> instances = enumInstances(perfObject, instanceFilter);
304         EnumMap<T, List<Long>> valuesMap = new EnumMap<>(propertyEnum);
305         if (instances.isEmpty()) {
306             return new Pair<>(Collections.emptyList(), valuesMap);
307         }
308 
309         try (Arena arena = Arena.ofConfined()) {
310             MemorySegment queryPtr = arena.allocate(ADDRESS);
311             checkSuccess(PdhOpenQuery(MemorySegment.NULL, MemorySegment.NULL, queryPtr));
312             MemorySegment query = null;
313 
314             try {
315                 query = queryPtr.get(ADDRESS, 0);
316 
317                 // Add counters for each instance × property (skip first which is the filter)
318                 // counterHandles[propIndex - 1][instanceIndex]
319                 // Detect _Base suffix: strip it from the counter path and read SecondValue instead
320                 boolean[] isBase = new boolean[props.length - 1];
321                 @SuppressWarnings("unchecked")
322                 List<MemorySegment>[] counterHandles = new List[props.length - 1];
323                 for (int p = 1; p < props.length; p++) {
324                     String counterName = props[p].getCounter();
325                     if (PerfCounter.isBase(counterName)) {
326                         counterName = PerfCounter.stripBaseSuffix(counterName);
327                         isBase[p - 1] = true;
328                     }
329                     counterHandles[p - 1] = new ArrayList<>(instances.size());
330                     for (String instance : instances) {
331                         String path = counterPath(perfObject, instance, counterName);
332                         try {
333                             counterHandles[p - 1].add(addEnglishCounter(arena, query, path));
334                         } catch (Throwable t) {
335                             LOG.warn("Failed to add wildcard counter {}: {}", path, t.getMessage());
336                             return new Pair<>(Collections.emptyList(), new EnumMap<>(propertyEnum));
337                         }
338                     }
339                 }
340 
341                 // Collect, retrying the initial-sample PDH_NO_DATA race
342                 collectQueryDataWithRetry(query);
343 
344                 // Read values; use SecondValue for _Base counters
345                 for (int p = 1; p < props.length; p++) {
346                     List<Long> values = new ArrayList<>(instances.size());
347                     for (MemorySegment handle : counterHandles[p - 1]) {
348                         try {
349                             values.add(readRawCounterValue(arena, handle, isBase[p - 1]));
350                         } catch (Throwable t) {
351                             LOG.debug("Failed to read counter for {}: {}", props[p], t.getMessage());
352                             values.add(0L);
353                         }
354                     }
355                     valuesMap.put(props[p], values);
356                 }
357             } finally {
358                 if (query != null) {
359                     PdhCloseQuery(query);
360                 }
361             }
362         } catch (Throwable t) {
363             LOG.debug("PDH queryWildcardCounters failed for {}: {}", perfObject, t.getMessage(), t);
364             return new Pair<>(Collections.emptyList(), new EnumMap<>(propertyEnum));
365         }
366         return new Pair<>(instances, valuesMap);
367     }

369     /**
370      * Localize a PDH object name. {@code PdhEnumObjectItemsW} requires the localized name unlike
371      * {@code PdhAddEnglishCounterW} which handles English names natively.
372      *
373      * @param perfObject The English object name
374      * @return The localized name, or the original if localization fails
375      */
376     private static String localizeObjectName(String perfObject) {
377         return LOCALIZE_CACHE.computeIfAbsent(perfObject, PerfDataUtilFFM::localizeUsingPerfIndex);
378     }
379 
380     private static String localizeUsingPerfIndex(String perfObject) {
381         int index = lookupPerfIndexByEnglishName(perfObject);
382         if (index == 0) {
383             LOG.debug("Unable to find English counter index for {}", perfObject);
384             return perfObject;
385         }
386         String localized = lookupPerfNameByIndex(index);
387         if (localized.isEmpty()) {
388             return perfObject;
389         }
390         LOG.debug("Localized {} to {}", perfObject, localized);
391         return localized;
392     }

394     private static int lookupPerfIndexByEnglishName(String name) {
395         return HkeyPerformanceDataUtilFFM.getCounterIndex(name);
396     }

398     private static String lookupPerfNameByIndex(int index) {
399         return callInArenaOrDefault(arena -> {
400             MemorySegment bufSize = arena.allocate(JAVA_INT);
401             bufSize.set(JAVA_INT, 0, 0);
402             int result = PdhLookupPerfNameByIndexW(MemorySegment.NULL, index, MemorySegment.NULL, bufSize);
403             if (result != PDH_MORE_DATA && result != ERROR_SUCCESS) {
404                 return "";
405             }
406             int size = bufSize.get(JAVA_INT, 0);
407             if (size <= 0) {
408                 return "";
409             }
410             MemorySegment nameBuf = arena.allocate((long) size * 2);
411             result = PdhLookupPerfNameByIndexW(MemorySegment.NULL, index, nameBuf, bufSize);
412             if (result != ERROR_SUCCESS) {
413                 return "";
414             }
415             return readWideString(nameBuf);
416         }, LOG, LogLevel.DEBUG, "Failed to lookup perf name by index " + index, "");
417     }
