48      /** Layout for nvmlUtilization_t struct. */
49      public static final StructLayout UTILIZATION_LAYOUT = MemoryLayout.structLayout(JAVA_INT.withName("gpu"),
50              JAVA_INT.withName("memory"));

53      /** Layout for nvmlMemory_t struct. */

55              JAVA_LONG.withName("free"), JAVA_LONG.withName("used"));

57      // nvmlPciInfo_t: { char busIdLegacy[16]; unsigned int domain, bus, device, pciDeviceId, pciSubSystemId;
58      // char busId[32]; }
59      /** Layout for nvmlPciInfo_t struct. */
60      public static final StructLayout PCI_INFO_LAYOUT = MemoryLayout.structLayout(
61              MemoryLayout.sequenceLayout(16, JAVA_BYTE).withName("busIdLegacy"), JAVA_INT.withName("domain"),
62              JAVA_INT.withName("bus"), JAVA_INT.withName("device"), JAVA_INT.withName("pciDeviceId"),
63              JAVA_INT.withName("pciSubSystemId"), MemoryLayout.sequenceLayout(32, JAVA_BYTE).withName("busId"));

65      // Method handles
66      private static final MethodHandle nvmlInit_v2;
67      private static final MethodHandle nvmlShutdown;
68      private static final MethodHandle nvmlDeviceGetCount_v2;
69      private static final MethodHandle nvmlDeviceGetHandleByIndex_v2;
70      private static final MethodHandle nvmlDeviceGetName;
71      private static final MethodHandle nvmlDeviceGetPciInfo_v3;
72      private static final MethodHandle nvmlDeviceGetUtilizationRates;
73      private static final MethodHandle nvmlDeviceGetMemoryInfo;
74      private static final MethodHandle nvmlDeviceGetTemperature;
75      private static final MethodHandle nvmlDeviceGetPowerUsage;
76      private static final MethodHandle nvmlDeviceGetClockInfo;
77      private static final MethodHandle nvmlDeviceGetFanSpeed;
78  
79      private static final boolean AVAILABLE;
80  
81      static {
82          boolean available = false;
83          MethodHandle hInit = null;
84          MethodHandle hShutdown = null;
85          MethodHandle hGetCount = null;
86          MethodHandle hGetHandle = null;
87          MethodHandle hGetName = null;
88          MethodHandle hGetPci = null;
89          MethodHandle hGetUtil = null;
90          MethodHandle hGetMem = null;
91          MethodHandle hGetTemp = null;
92          MethodHandle hGetPower = null;
93          MethodHandle hGetClock = null;
94          MethodHandle hGetFan = null;
95          try {
96              String libName = System.getProperty("os.name", "").toLowerCase(java.util.Locale.ROOT).contains("win")
97                      ? "nvml"
98                      : "nvidia-ml";
99              SymbolLookup nvml = libraryLookup(libName);
100 
101             hInit = LINKER.downcallHandle(nvml.findOrThrow("nvmlInit_v2"), FunctionDescriptor.of(JAVA_INT));
102             hShutdown = LINKER.downcallHandle(nvml.findOrThrow("nvmlShutdown"), FunctionDescriptor.of(JAVA_INT));
103             hGetCount = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetCount_v2"),
104                     FunctionDescriptor.of(JAVA_INT, ADDRESS));
105             hGetHandle = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetHandleByIndex_v2"),
106                     FunctionDescriptor.of(JAVA_INT, JAVA_INT, ADDRESS));
107             hGetName = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetName"),
108                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS, JAVA_INT));
109             hGetPci = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetPciInfo_v3"),
110                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
111             hGetUtil = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetUtilizationRates"),
112                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
113             hGetMem = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetMemoryInfo"),
114                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
115             hGetTemp = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetTemperature"),
116                     FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS));
117             hGetPower = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetPowerUsage"),
118                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
119             hGetClock = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetClockInfo"),
120                     FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS));
121             hGetFan = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetFanSpeed"),
122                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
123             available = true;
124             LOG.debug("NVML library loaded via FFM");
125         } catch (Throwable t) {
126             LOG.debug("NVML library not available via FFM: {}", t.getMessage());
127         }
128         nvmlInit_v2 = hInit;
129         nvmlShutdown = hShutdown;
130         nvmlDeviceGetCount_v2 = hGetCount;
131         nvmlDeviceGetHandleByIndex_v2 = hGetHandle;
132         nvmlDeviceGetName = hGetName;
133         nvmlDeviceGetPciInfo_v3 = hGetPci;
134         nvmlDeviceGetUtilizationRates = hGetUtil;
135         nvmlDeviceGetMemoryInfo = hGetMem;
136         nvmlDeviceGetTemperature = hGetTemp;
137         nvmlDeviceGetPowerUsage = hGetPower;
138         nvmlDeviceGetClockInfo = hGetClock;
139         nvmlDeviceGetFanSpeed = hGetFan;
140         AVAILABLE = available;
141     }

282     /**
283      * Reads a null-terminated string from a byte sequence within a struct segment.
284      *
285      * @param seg    the struct segment
286      * @param offset byte offset of the char array field
287      * @param maxLen maximum length of the char array
288      * @return the string, or empty if null/empty
289      */
290     public static String readString(MemorySegment seg, long offset, int maxLen) {
291         for (int i = 0; i < maxLen; i++) {
292             if (seg.get(JAVA_BYTE, offset + i) == 0) {
293                 if (i == 0) {
294                     return "";
295                 }
296                 return seg.getString(offset);
297             }
298         }
299         byte[] bytes = new byte[maxLen];
300         MemorySegment.copy(seg, JAVA_BYTE, offset, bytes, 0, maxLen);
301         return new String(bytes, java.nio.charset.StandardCharsets.UTF_8).trim();
302     }
