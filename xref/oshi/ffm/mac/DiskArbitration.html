5   package oshi.ffm.mac;
6   
7   import static oshi.ffm.mac.DiskArbitrationFunctions.DADiskCopyDescription;
8   import static oshi.ffm.mac.DiskArbitrationFunctions.DADiskCreateFromBSDName;
9   import static oshi.ffm.mac.DiskArbitrationFunctions.DADiskCreateFromIOMedia;
10  import static oshi.ffm.mac.DiskArbitrationFunctions.DADiskGetBSDName;
11  import static oshi.ffm.mac.DiskArbitrationFunctions.DASessionCreate;
12  
13  import java.lang.foreign.Arena;
14  import java.lang.foreign.MemorySegment;
15  
16  import oshi.ffm.mac.CoreFoundation.CFAllocatorRef;
17  import oshi.ffm.mac.CoreFoundation.CFDictionaryRef;
18  import oshi.ffm.mac.CoreFoundation.CFTypeRef;
19  
20  /**
21   * Disk Arbitration is a low-level framework based on Core Foundation. The Disk Arbitration framework provides the
22   * ability to get various pieces of information about a volume.
23   */
24  public interface DiskArbitration {
25  
26      /**
27       * Type of a reference to DASession instances.
28       */
29      class DASessionRef extends CFTypeRef {
30          public DASessionRef(MemorySegment segment) {
31              super(segment);
32          }
33  
34          /**
35           * Creates a new session.
36           *
37           * @param allocator The allocator to use, or null for default
38           * @return A reference to a new DASession
39           */
40          public static DASessionRef create(CFAllocatorRef allocator) {
41              MemorySegment allocSeg = allocator != null ? allocator.segment() : MemorySegment.NULL;
42              MemorySegment sessionSeg = DASessionCreate(allocSeg);
43              return new DASessionRef(sessionSeg);
44          }
45      }
46  
47      /**
48       * Type of a reference to DADisk instances.
49       */
50      class DADiskRef extends CFTypeRef {
51          public DADiskRef(MemorySegment segment) {
52              super(segment);
53          }
54  
55          /**
56           * Creates a new disk object from a BSD device name.
57           *
58           * @param allocator The allocator to use, or null for default
59           * @param session   The DASession in which to contact Disk Arbitration
60           * @param bsdName   The BSD device name
61           * @return A reference to a new DADisk
62           */
63          public static DADiskRef createFromBSDName(CFAllocatorRef allocator, DASessionRef session, String bsdName) {
64              try (Arena arena = Arena.ofConfined()) {
65                  MemorySegment allocSeg = allocator != null ? allocator.segment() : MemorySegment.NULL;
66                  MemorySegment sessionSeg = session != null ? session.segment() : MemorySegment.NULL;
67                  MemorySegment bsdNameSeg = arena.allocateFrom(bsdName);
68  
69                  MemorySegment diskSeg = DADiskCreateFromBSDName(allocSeg, sessionSeg, bsdNameSeg);
70                  return new DADiskRef(diskSeg);
71              }
72          }
73  
74          /**
75           * Creates a new disk object from an IOMedia object.
76           *
77           * @param allocator The allocator to use, or null for default
78           * @param session   The DASession in which to contact Disk Arbitration
79           * @param media     The IOKit.IOObject media object
80           * @return A reference to a new DADisk
81           */
82          public static DADiskRef createFromIOMedia(CFAllocatorRef allocator, DASessionRef session,
83                  IOKit.IOObject media) {
84              MemorySegment allocSeg = allocator != null ? allocator.segment() : MemorySegment.NULL;
85              MemorySegment sessionSeg = session != null ? session.segment() : MemorySegment.NULL;
86              MemorySegment mediaSeg = media != null ? media.segment() : MemorySegment.NULL;
87  
88              MemorySegment diskSeg = DADiskCreateFromIOMedia(allocSeg, sessionSeg, mediaSeg);
89              return new DADiskRef(diskSeg);
90          }
91  
92          /**
93           * Obtains the Disk Arbitration description of the specified disk.
94           *
95           * @return The disk's Disk Arbitration description
96           */
97          public CFDictionaryRef copyDescription() {
98              if (isNull()) {
99                  return new CFDictionaryRef(MemorySegment.NULL);
100             }
101             MemorySegment dictSeg = DADiskCopyDescription(segment());
102             return new CFDictionaryRef(dictSeg);
103         }
104 
105         /**
106          * Obtains the BSD device name for the specified disk.
107          *
108          * @return The disk's BSD device name
109          */
110         public String getBSDName() {
111             if (isNull()) {
112                 return null;
113             }
114             MemorySegment nameSeg = DADiskGetBSDName(segment());
115             if (nameSeg.equals(MemorySegment.NULL)) {
116                 return null;
117             }
118             return nameSeg.getString(0);
119         }
120     }
121 }
