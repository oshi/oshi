5   package oshi.ffm.mac;
6   
7   import java.util.List;
8   
9   import java.lang.foreign.SymbolLookup;
10  
11  import oshi.ffm.ForeignFunctions;
12  
13  /**
14   * Base class for macOS FFM function bindings, adding framework lookup support on top of {@link ForeignFunctions}.
15   */
16  public abstract class MacForeignFunctions extends ForeignFunctions {
17  
18      /**
19       * Search paths for macOS frameworks, in priority order.
20       * <ul>
21       * <li>{@code /System/Library/Frameworks} - Reserved strictly for Apple's core OS frameworks (e.g. IOKit,
22       * CoreFoundation, AppKit).</li>
23       * <li>{@code /Library/Frameworks} - Standard location for third-party frameworks available to all users (e.g. SDL2,
24       * GStreamer, or custom drivers).</li>
25       * </ul>
26       */
27      protected static final List<String> FRAMEWORK_SEARCH_PATHS = List.of("/System/Library/Frameworks",
28              "/Library/Frameworks");
29  
30      /** Not intended for instantiation. */
31      protected MacForeignFunctions() {
32      }
33  
34      /**
35       * Lookup a macOS framework by simple name, e.g. {@code "IOKit"}. Searches {@link #FRAMEWORK_SEARCH_PATHS} in order.
36       *
37       * @param frameworkName the framework name without path or extension
38       * @return the symbol lookup for the framework
39       * @throws IllegalArgumentException if the framework is not found in any search path
40       */
41      protected static SymbolLookup frameworkLookup(String frameworkName) {
42          for (String base : FRAMEWORK_SEARCH_PATHS) {
43              try {
44                  return SymbolLookup.libraryLookup(base + "/" + frameworkName + ".framework/" + frameworkName,
45                          LIBRARY_ARENA);
46              } catch (IllegalArgumentException ignored) {
47                  // Not found in this directory, try the next one
48              }
49          }
50          throw new IllegalArgumentException("Framework not found: " + frameworkName);
51      }
52  }
