24  package oshi.driver.windows.registry;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  import com.sun.jna.platform.win32.Advapi32Util; // NOSONAR squid:S1191
33  import com.sun.jna.platform.win32.Advapi32Util.Account;
34  import com.sun.jna.platform.win32.Advapi32Util.InfoKey;
35  import com.sun.jna.platform.win32.Win32Exception;
36  import com.sun.jna.platform.win32.WinNT;
37  import com.sun.jna.platform.win32.WinReg;
38  import com.sun.jna.platform.win32.WinReg.HKEY;
39  
40  import oshi.annotation.concurrent.ThreadSafe;
41  import oshi.software.os.OSSession;
42  
43  /**
44   * Utility to read session data from HKEY_USERS
45   */
46  @ThreadSafe
47  public final class HkeyUserData {
48  
49      private static final String PATH_DELIMITER = "\\";
50      private static final String DEFAULT_DEVICE = "Console";
51      private static final String VOLATILE_ENV_SUBKEY = "Volatile Environment";
52      private static final String CLIENTNAME = "CLIENTNAME";
53      private static final String SESSIONNAME = "SESSIONNAME";
54  
55      private static final Logger LOG = LoggerFactory.getLogger(HkeyUserData.class);
56  
57      private HkeyUserData() {
58      }
59  
60      public static List<OSSession> queryUserSessions() {
61          List<OSSession> sessions = new ArrayList<>();
62          for (String sidKey : Advapi32Util.registryGetKeys(WinReg.HKEY_USERS)) {
63              if (!sidKey.startsWith(".") && !sidKey.endsWith("_Classes")) {
64                  try {
65                      Account a = Advapi32Util.getAccountBySid(sidKey);
66                      String name = a.name;
67                      String device = DEFAULT_DEVICE;
68                      String host = a.domain; // temporary default
69                      long loginTime = 0;
70                      String keyPath = sidKey + PATH_DELIMITER + VOLATILE_ENV_SUBKEY;
71                      if (Advapi32Util.registryKeyExists(WinReg.HKEY_USERS, keyPath)) {
72                          HKEY hKey = Advapi32Util.registryGetKey(WinReg.HKEY_USERS, keyPath, WinNT.KEY_READ).getValue();
73                          // InfoKey write time is user login time
74                          InfoKey info = Advapi32Util.registryQueryInfoKey(hKey, 0);
75                          loginTime = info.lpftLastWriteTime.toTime();
76                          for (String subKey : Advapi32Util.registryGetKeys(hKey)) {
77                              String subKeyPath = keyPath + PATH_DELIMITER + subKey;
78                              // Check for session and client name
79                              if (Advapi32Util.registryValueExists(WinReg.HKEY_USERS, subKeyPath, SESSIONNAME)) {
80                                  String session = Advapi32Util.registryGetStringValue(WinReg.HKEY_USERS, subKeyPath,
81                                          SESSIONNAME);
82                                  if (!session.isEmpty()) {
83                                      device = session;
84                                  }
85                              }
86                              if (Advapi32Util.registryValueExists(WinReg.HKEY_USERS, subKeyPath, CLIENTNAME)) {
87                                  String client = Advapi32Util.registryGetStringValue(WinReg.HKEY_USERS, subKeyPath,
88                                          CLIENTNAME);
89                                  if (!client.isEmpty() && !DEFAULT_DEVICE.equals(client)) {
90                                      host = client;
91                                  }
92                              }
93                          }
94                          Advapi32Util.registryCloseKey(hKey);
95                      }
96                      sessions.add(new OSSession(name, device, loginTime, host));
97                  } catch (Win32Exception ex) {
98                      LOG.warn("Error querying SID {} from registry: {}", sidKey, ex.getMessage());
99                  }
100             }
101         }
102         return sessions;
103     }
104 }
