204     @Override
205     public boolean updateAttributes() {
206         List<PowerSource> psArr = getPowerSources();
207         for (PowerSource ps : psArr) {
208             if (ps.getName().equals(this.name)) {
209                 this.name = ps.getName();
210                 this.deviceName = ps.getDeviceName();
211                 this.remainingCapacityPercent = ps.getRemainingCapacityPercent();
212                 this.timeRemainingEstimated = ps.getTimeRemainingEstimated();
213                 this.timeRemainingInstant = ps.getTimeRemainingInstant();
214                 this.powerUsageRate = ps.getPowerUsageRate();
215                 this.voltage = ps.getVoltage();
216                 this.amperage = ps.getAmperage();
217                 this.powerOnLine = ps.isPowerOnLine();
218                 this.charging = ps.isCharging();
219                 this.discharging = ps.isDischarging();
220                 this.capacityUnits = ps.getCapacityUnits();
221                 this.currentCapacity = ps.getCurrentCapacity();
222                 this.maxCapacity = ps.getMaxCapacity();
223                 this.designCapacity = ps.getDesignCapacity();
224                 this.cycleCount = ps.getCycleCount();
225                 this.chemistry = ps.getChemistry();
226                 this.manufactureDate = ps.getManufactureDate();
227                 this.manufacturer = ps.getManufacturer();
228                 this.serialNumber = ps.getSerialNumber();
229                 this.temperature = ps.getTemperature();
230                 return true;
231             }
232         }
233         // Didn't find this battery
234         return false;
235     }

237     private static List<PowerSource> getPowerSources() {
238         switch (SystemInfo.getCurrentPlatform()) {
239         case WINDOWS:
240             return WindowsPowerSource.getPowerSources();
241         case MACOS:
242             return MacPowerSource.getPowerSources();
243         case LINUX:
244             return LinuxPowerSource.getPowerSources();
245         case SOLARIS:
246             return SolarisPowerSource.getPowerSources();
247         case FREEBSD:
248             return FreeBsdPowerSource.getPowerSources();
249         case AIX:
250             return AixPowerSource.getPowerSources();
251         default:
252             throw new UnsupportedOperationException("Operating system not supported: " + Platform.getOSType());
253         }
254     }

256     @Override
257     public String toString() {
258         StringBuilder sb = new StringBuilder();
259         sb.append("Name: ").append(getName()).append(", ");
260         sb.append("Device Name: ").append(getDeviceName()).append(",\n ");
261         sb.append("RemainingCapacityPercent: ").append(getRemainingCapacityPercent() * 100).append("%, ");
262         sb.append("Time Remaining: ").append(formatTimeRemaining(getTimeRemainingEstimated())).append(", ");
263         sb.append("Time Remaining Instant: ").append(formatTimeRemaining(getTimeRemainingInstant())).append(",\n ");
264         sb.append("Power Usage Rate: ").append(getPowerUsageRate()).append("mW, ");
265         sb.append("Voltage: ");
266         if (getVoltage() > 0) {
267             sb.append(getVoltage()).append("V, ");
268         } else {
269             sb.append(Constants.UNKNOWN);
270         }
271         sb.append("Amperage: ").append(getAmperage()).append("mA,\n ");
272         sb.append("Power OnLine: ").append(isPowerOnLine()).append(", ");
273         sb.append("Charging: ").append(isCharging()).append(", ");
274         sb.append("Discharging: ").append(isDischarging()).append(",\n ");
275         sb.append("Capacity Units: ").append(getCapacityUnits()).append(", ");
276         sb.append("Current Capacity: ").append(getCurrentCapacity()).append(", ");
277         sb.append("Max Capacity: ").append(getMaxCapacity()).append(", ");
278         sb.append("Design Capacity: ").append(getDesignCapacity()).append(",\n ");
279         sb.append("Cycle Count: ").append(getCycleCount()).append(", ");
280         sb.append("Chemistry: ").append(getChemistry()).append(", ");
281         sb.append("Manufacture Date: ").append(getManufactureDate() != null ? getManufactureDate() : Constants.UNKNOWN)
282                 .append(", ");
283         sb.append("Manufacturer: ").append(getManufacturer()).append(",\n ");
284         sb.append("SerialNumber: ").append(getSerialNumber()).append(", ");
285         sb.append("Temperature: ");
286         if (getTemperature() > 0) {
287             sb.append(getTemperature()).append("Â°C");
288         } else {
289             sb.append(Constants.UNKNOWN);
290         }
291         return sb.toString();
292     }

294     /**
295      * Estimated time remaining on power source, formatted as HH:mm
296      *
297      * @param timeInSeconds
298      *            The time remaining, in seconds
299      * @return formatted String of time remaining
300      */
301     private static String formatTimeRemaining(double timeInSeconds) {
302         String formattedTimeRemaining;
303         if (timeInSeconds < -1.5) {
304             formattedTimeRemaining = "Charging";
305         } else if (timeInSeconds < 0) {
306             formattedTimeRemaining = "Unknown";
307         } else {
308             int hours = (int) (timeInSeconds / 3600);
309             int minutes = (int) (timeInSeconds % 3600 / 60);
310             formattedTimeRemaining = String.format("%d:%02d", hours, minutes);
311         }
312         return formattedTimeRemaining;
313     }
