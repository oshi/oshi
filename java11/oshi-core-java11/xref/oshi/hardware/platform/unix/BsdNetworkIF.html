24  package oshi.hardware.platform.unix;
25  
26  import java.net.NetworkInterface;
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import oshi.annotation.concurrent.ThreadSafe;
34  import oshi.hardware.NetworkIF;
35  import oshi.hardware.common.AbstractNetworkIF;
36  import oshi.util.ExecutingCommand;
37  import oshi.util.ParseUtil;
38  
39  /**
40   * BsdNetworkIF applicable to FreeBSD and OpenBSD.
41   */
42  @ThreadSafe
43  public final class BsdNetworkIF extends AbstractNetworkIF {
44  
45      private static final Logger LOG = LoggerFactory.getLogger(BsdNetworkIF.class);
46  
47      private long bytesRecv;
48      private long bytesSent;
49      private long packetsRecv;
50      private long packetsSent;
51      private long inErrors;
52      private long outErrors;
53      private long inDrops;
54      private long collisions;
55      private long timeStamp;
56  
57      public BsdNetworkIF(NetworkInterface netint) throws InstantiationException {
58          super(netint);
59          updateAttributes();
60      }
61  
62      /**
63       * Gets all network interfaces on this machine
64       *
65       * @param includeLocalInterfaces
66       *            include local interfaces in the result
67       * @return A list of {@link NetworkIF} objects representing the interfaces
68       */
69      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
70          List<NetworkIF> ifList = new ArrayList<>();
71          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
72              try {
73                  ifList.add(new BsdNetworkIF(ni));
74              } catch (InstantiationException e) {
75                  LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
76              }
77          }
78          return ifList;
79      }
80  
81      @Override
82      public long getBytesRecv() {
83          return this.bytesRecv;
84      }
85  
86      @Override
87      public long getBytesSent() {
88          return this.bytesSent;
89      }
90  
91      @Override
92      public long getPacketsRecv() {
93          return this.packetsRecv;
94      }
95  
96      @Override
97      public long getPacketsSent() {
98          return this.packetsSent;
99      }
100 
101     @Override
102     public long getInErrors() {
103         return this.inErrors;
104     }
105 
106     @Override
107     public long getOutErrors() {
108         return this.outErrors;
109     }
110 
111     @Override
112     public long getInDrops() {
113         return this.inDrops;
114     }
115 
116     @Override
117     public long getCollisions() {
118         return this.collisions;
119     }
120 
121     @Override
122     public long getSpeed() {
123         return 0;
124     }
125 
126     @Override
127     public long getTimeStamp() {
128         return this.timeStamp;
129     }
130 
131     @Override
132     public boolean updateAttributes() {
133         String stats = ExecutingCommand.getAnswerAt("netstat -bI " + getName(), 1);
134         this.timeStamp = System.currentTimeMillis();
135         String[] split = ParseUtil.whitespaces.split(stats);
136         if (split.length < 12) {
137             // No update
138             return false;
139         }
140         this.bytesSent = ParseUtil.parseUnsignedLongOrDefault(split[10], 0L);
141         this.bytesRecv = ParseUtil.parseUnsignedLongOrDefault(split[7], 0L);
142         this.packetsSent = ParseUtil.parseUnsignedLongOrDefault(split[8], 0L);
143         this.packetsRecv = ParseUtil.parseUnsignedLongOrDefault(split[4], 0L);
144         this.outErrors = ParseUtil.parseUnsignedLongOrDefault(split[9], 0L);
145         this.inErrors = ParseUtil.parseUnsignedLongOrDefault(split[5], 0L);
146         this.collisions = ParseUtil.parseUnsignedLongOrDefault(split[11], 0L);
147         this.inDrops = ParseUtil.parseUnsignedLongOrDefault(split[6], 0L);
148         return true;
149     }
150 }
