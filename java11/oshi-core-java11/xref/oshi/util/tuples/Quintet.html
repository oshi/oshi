24  package oshi.util.tuples;
25  
26  import oshi.annotation.concurrent.ThreadSafe;
27  
28  /**
29   * Convenience class for returning multiple objects from methods.
30   *
31   * @param <A>
32   *            Type of the first element
33   * @param <B>
34   *            Type of the second element
35   * @param <C>
36   *            Type of the third element
37   * @param <D>
38   *            Type of the fourth element
39   * @param <E>
40   *            Type of the fifth element
41   */
42  @ThreadSafe
43  public class Quintet<A, B, C, D, E> {
44  
45      private final A a;
46      private final B b;
47      private final C c;
48      private final D d;
49      private final E e;
50  
51      /**
52       * Create a quintet and store five objects.
53       *
54       * @param a
55       *            the first object to store
56       * @param b
57       *            the second object to store
58       * @param c
59       *            the third object to store
60       * @param d
61       *            the fourth object to store
62       * @param e
63       *            the fifth object to store
64       */
65      public Quintet(A a, B b, C c, D d, E e) {
66          this.a = a;
67          this.b = b;
68          this.c = c;
69          this.d = d;
70          this.e = e;
71      }
72  
73      /**
74       * Returns the first stored object.
75       *
76       * @return first object stored
77       */
78      public final A getA() {
79          return a;
80      }
81  
82      /**
83       * Returns the second stored object.
84       *
85       * @return second object stored
86       */
87      public final B getB() {
88          return b;
89      }
90  
91      /**
92       * Returns the third stored object.
93       *
94       * @return third object stored
95       */
96      public final C getC() {
97          return c;
98      }
99  
100     /**
101      * Returns the fourth stored object.
102      *
103      * @return fourth object stored
104      */
105     public final D getD() {
106         return d;
107     }
108 
109     /**
110      * Returns the fifth stored object.
111      *
112      * @return fifth object stored
113      */
114     public final E getE() {
115         return e;
116     }
117 }
