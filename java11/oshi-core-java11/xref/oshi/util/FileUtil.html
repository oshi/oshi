62      /**
63       * Read an entire file at one time. Intended primarily for Linux /proc
64       * filesystem to avoid recalculating file contents on iterative reads.
65       *
66       * @param filename
67       *            The file to read
68       * @return A list of Strings representing each line of the file, or an empty
69       *         list if file could not be read or is empty
70       */
71      public static List<String> readFile(String filename) {
72          return readFile(filename, true);
73      }
74  
75      /**
76       * Read an entire file at one time. Intended primarily for Linux /proc
77       * filesystem to avoid recalculating file contents on iterative reads.
78       *
79       * @param filename
80       *            The file to read
81       * @param reportError
82       *            Whether to log errors reading the file
83       * @return A list of Strings representing each line of the file, or an empty
84       *         list if file could not be read or is empty
85       */
86      public static List<String> readFile(String filename, boolean reportError) {
87          if (new File(filename).canRead()) {
88              if (LOG.isDebugEnabled()) {
89                  LOG.debug(READING_LOG, filename);
90              }
91              try {
92                  return Files.readAllLines(Paths.get(filename), StandardCharsets.UTF_8);
93              } catch (IOException e) {
94                  if (reportError) {
95                      LOG.error("Error reading file {}. {}", filename, e.getMessage());
96                  } else {
97                      LOG.debug("Error reading file {}. {}", filename, e.getMessage());
98                  }
99              }
100         } else if (reportError) {
101             LOG.warn("File not found or not readable: {}", filename);
102         }
103         return new ArrayList<>();
104     }

148     /**
149      * Read a file and return the long value contained therein. Intended primarily
150      * for Linux /sys filesystem
151      *
152      * @param filename
153      *            The file to read
154      * @return The value contained in the file, if any; otherwise zero
155      */
156     public static long getLongFromFile(String filename) {
157         if (LOG.isDebugEnabled()) {
158             LOG.debug(READING_LOG, filename);
159         }
160         List<String> read = FileUtil.readFile(filename, false);
161         if (!read.isEmpty()) {
162             if (LOG.isTraceEnabled()) {
163                 LOG.trace(READ_LOG, read.get(0));
164             }
165             return ParseUtil.parseLongOrDefault(read.get(0), 0L);
166         }
167         return 0L;
168     }

240     /**
241      * Read a file and return a map of string keys to string values contained
242      * therein. Intended primarily for Linux {@code /proc/[pid]} files to provide
243      * more detailed or accurate information not available in the API.
244      *
245      * @param filename
246      *            The file to read
247      * @param separator
248      *            Character(s) in each line of the file that separate the key and
249      *            the value.
250      *
251      * @return The map contained in the file, delimited by the separator, with the
252      *         value whitespace trimmed. If keys and values are not parsed, an empty
253      *         map is returned.
254      */
255     public static Map<String, String> getKeyValueMapFromFile(String filename, String separator) {
256         Map<String, String> map = new HashMap<>();
257         if (LOG.isDebugEnabled()) {
258             LOG.debug(READING_LOG, filename);
259         }
260         List<String> lines = FileUtil.readFile(filename, false);
261         for (String line : lines) {
262             String[] parts = line.split(separator);
263             if (parts.length == 2) {
264                 map.put(parts[0], parts[1].trim());
265             }
266         }
267         return map;
268     }

270     /**
271      * Read a configuration file from the sequence of context classloader, system
272      * classloader and classloader of the current class, and return its properties
273      *
274      * @param propsFilename
275      *            The filename
276      * @return A {@link Properties} object containing the properties.
277      */
278     public static Properties readPropertiesFromFilename(String propsFilename) {
279         Properties archProps = new Properties();
280         // Load the configuration file from at least one of multiple possible
281         // ClassLoaders, evaluated in order, eliminating duplicates
282         for (ClassLoader loader : Stream.of(Thread.currentThread().getContextClassLoader(),
283                 ClassLoader.getSystemClassLoader(), FileUtil.class.getClassLoader())
284                 .collect(Collectors.toCollection(LinkedHashSet::new))) {
285             if (readPropertiesFromClassLoader(propsFilename, archProps, loader)) {
286                 return archProps;
287             }
288         }
289         LOG.warn("Failed to load default configuration");
290         return archProps;
291     }
292 
293     private static boolean readPropertiesFromClassLoader(String propsFilename, Properties archProps,
294             ClassLoader loader) {
295         if (loader == null) {
296             return false;
297         }
298         // Load the configuration file from the classLoader
299         try {
300             List<URL> resources = Collections.list(loader.getResources(propsFilename));
301             if (resources.isEmpty()) {
302                 LOG.debug("No {} file found from ClassLoader {}", propsFilename, loader);
303                 return false;
304             }
305             if (resources.size() > 1) {
306                 LOG.warn("Configuration conflict: there is more than one {} file on the classpath", propsFilename);
307                 return true;
308             }
309 
310             try (InputStream in = resources.get(0).openStream()) {
311                 if (in != null) {
312                     archProps.load(in);
313                 }
314             }
315             return true;
316         } catch (IOException e) {
317             return false;
318         }
319     }

321     /**
322      * Reads the target of a symbolic link
323      *
324      * @param file
325      *            The file to read
326      * @return The symlink name, or null if the read failed
327      */
328     public static String readSymlinkTarget(File file) {
329         try {
330             return Files.readSymbolicLink(Paths.get(file.getAbsolutePath())).toString();
331         } catch (IOException e) {
332             return null;
333         }
