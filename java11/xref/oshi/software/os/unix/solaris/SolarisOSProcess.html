232     private int queryBitness() {
233         List<String> pflags = ExecutingCommand.runNative("pflags " + getProcessID());
234         for (String line : pflags) {
235             if (line.contains("data model")) {
236                 if (line.contains("LP32")) {
237                     return 32;
238                 } else if (line.contains("LP64")) {
239                     return 64;
240                 }
241             }
242         }
243         return 0;
244     }

246     @Override
247     public long getAffinityMask() {
248         long bitMask = 0L;
249         String cpuset = ExecutingCommand.getFirstAnswer("pbind -q " + getProcessID());
250         // Sample output:
251         // <empty string if no binding>
252         // pid 101048 strongly bound to processor(s) 0 1 2 3.
253         if (cpuset.isEmpty()) {
254             List<String> allProcs = ExecutingCommand.runNative("psrinfo");
255             for (String proc : allProcs) {
256                 String[] split = ParseUtil.whitespaces.split(proc);
257                 int bitToSet = ParseUtil.parseIntOrDefault(split[0], -1);
258                 if (bitToSet >= 0) {
259                     bitMask |= 1L << bitToSet;
260                 }
261             }
262             return bitMask;
263         } else if (cpuset.endsWith(".") && cpuset.contains("strongly bound to processor(s)")) {
264             String parse = cpuset.substring(0, cpuset.length() - 1);
265             String[] split = ParseUtil.whitespaces.split(parse);
266             for (int i = split.length - 1; i >= 0; i--) {
267                 int bitToSet = ParseUtil.parseIntOrDefault(split[i], -1);
268                 if (bitToSet >= 0) {
269                     bitMask |= 1L << bitToSet;
270                 } else {
271                     // Once we run into the word processor(s) we're done
272                     break;
273                 }
274             }
275         }
276         return bitMask;
277     }

279     @Override
280     public List<OSThread> getThreadDetails() {
281         List<OSThread> threads = new ArrayList<>();
282         List<String> threadList = ExecutingCommand.runNative("ps -o " + PS_THREAD_COLUMNS + " -p " + getProcessID());
283         if (threadList.size() > 1) {
284             // Get a map by lwpid of prstat output
285             List<String> prstatList = ExecutingCommand.runNative("prstat -L -v -p " + getProcessID() + " 1 1");
286             Map<String, String> prstatRowMap = new HashMap<>();
287             for (String s : prstatList) {
288                 String row = s.trim();
289                 // Last element is PROCESS/LWPID
290                 int idx = row.lastIndexOf('/');
291                 if (idx > 0) {
292                     prstatRowMap.put(row.substring(idx + 1), row);
293                 }
294             }
295             // remove header row and iterate thread list
296             threadList.remove(0);
297             for (String thread : threadList) {
298                 Map<PsThreadColumns, String> psMap = ParseUtil.stringToEnumMap(PsThreadColumns.class, thread.trim(),
299                         ' ');
300                 // Check if last (thus all) value populated
301                 if (psMap.containsKey(PsThreadColumns.PRI)) {
302                     String lwpStr = psMap.get(PsThreadColumns.LWP);
303                     Map<PrstatKeywords, String> prstatMap = ParseUtil.stringToEnumMap(PrstatKeywords.class,
304                             prstatRowMap.getOrDefault(lwpStr, ""), ' ');
305                     threads.add(new SolarisOSThread(getProcessID(), psMap, prstatMap));
306                 }
307             }
308         }
309         return threads;
310     }

312     @Override
313     public boolean updateAttributes() {
314         int pid = getProcessID();
315         List<String> procList = ExecutingCommand
316                 .runNative("ps -o " + SolarisOperatingSystem.PS_COMMAND_ARGS + " -p " + pid);
317         if (procList.size() > 1) {
318             Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, procList.get(1).trim(), ' ');
319             // Check if last (thus all) value populated
320             if (psMap.containsKey(PsKeywords.ARGS)) {
321                 String pidStr = psMap.get(PsKeywords.PID);
322                 List<String> prstatList = ExecutingCommand.runNative("prstat -v -p " + pidStr + " 1 1");
323                 String prstatRow = "";
324                 for (String s : prstatList) {
325                     String row = s.trim();
326                     if (row.startsWith(pidStr + " ")) {
327                         prstatRow = row;
328                         break;
329                     }
330                 }
331                 Map<PrstatKeywords, String> prstatMap = ParseUtil.stringToEnumMap(PrstatKeywords.class, prstatRow, ' ');
332                 return updateAttributes(psMap, prstatMap);
333             }
334         }
335         this.state = State.INVALID;
336         return false;
337     }
338 
339     private boolean updateAttributes(Map<PsKeywords, String> psMap, Map<PrstatKeywords, String> prstatMap) {
340         long now = System.currentTimeMillis();
341         this.state = getStateFromOutput(psMap.get(PsKeywords.S).charAt(0));
342         this.parentProcessID = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PPID), 0);
343         this.user = psMap.get(PsKeywords.USER);
344         this.userID = psMap.get(PsKeywords.UID);
345         this.group = psMap.get(PsKeywords.GROUP);
346         this.groupID = psMap.get(PsKeywords.GID);
347         this.threadCount = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.NLWP), 0);
348         this.priority = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PRI), 0);
349         // These are in KB, multiply
350         this.virtualSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.VSZ), 0) * 1024;
351         this.residentSetSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.RSS), 0) * 1024;
352         // Avoid divide by zero for processes up less than a second
353         long elapsedTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.ETIME), 0L);
354         this.upTime = elapsedTime < 1L ? 1L : elapsedTime;
355         this.startTime = now - this.upTime;
356         this.kernelTime = 0L;
357         this.userTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.TIME), 0L);
358         this.path = psMap.get(PsKeywords.COMM);
359         this.name = this.path.substring(this.path.lastIndexOf('/') + 1);
360         this.commandLineBackup = psMap.get(PsKeywords.ARGS);
361         if (prstatMap.containsKey(PrstatKeywords.ICX)) {
362             long nonVoluntaryContextSwitches = ParseUtil.parseLongOrDefault(prstatMap.get(PrstatKeywords.ICX), 0L);
363             long voluntaryContextSwitches = ParseUtil.parseLongOrDefault(prstatMap.get(PrstatKeywords.VCX), 0L);
364             this.contextSwitches = voluntaryContextSwitches + nonVoluntaryContextSwitches;
365         }
366         return true;
367     }

369     /***
370      * Returns Enum STATE for the state value obtained from status string of
371      * thread/process.
372      *
373      * @param stateValue
374      *            state value from the status string
375      * @return The state
376      */
377     static State getStateFromOutput(char stateValue) {
378         State state;
379         switch (stateValue) {
380         case 'O':
381             state = RUNNING;
382             break;
383         case 'S':
384             state = SLEEPING;
385             break;
386         case 'R':
387         case 'W':
388             state = WAITING;
389             break;
390         case 'Z':
391             state = ZOMBIE;
392             break;
393         case 'T':
394             state = STOPPED;
395             break;
396         default:
397             state = OTHER;
398             break;
399         }
400         return state;
401     }
