51  /**
52   * Network interfaces implementation.
53   */
54  @ThreadSafe
55  public abstract class AbstractNetworkIF implements NetworkIF {
56  
57      private static final Logger LOG = LoggerFactory.getLogger(AbstractNetworkIF.class);
58  
59      private static final String OSHI_VM_MAC_ADDR_PROPERTIES = "oshi.vmmacaddr.properties";
60  
61      private NetworkInterface networkInterface;
62      private String name;
63      private String displayName;
64      private int index;
65      private int mtu;
66      private String mac;
67      private String[] ipv4;
68      private Short[] subnetMasks;
69      private String[] ipv6;
70      private Short[] prefixLengths;
71  
72      private final Supplier<Properties> vmMacAddrProps = memoize(AbstractNetworkIF::queryVmMacAddrProps);
73  
74      /**
75       * Construct a {@link NetworkIF} object backed by the specified
76       * {@link NetworkInterface}.
77       *
78       * @param netint
79       *            The core java {@link NetworkInterface} backing this object.
80       * @throws InstantiationException
81       *             If a socket exception prevents access to the backing interface.
82       */
83      protected AbstractNetworkIF(NetworkInterface netint) throws InstantiationException {
84          this(netint, netint.getDisplayName());
85      }
86  
87      /**
88       * Construct a {@link NetworkIF} object backed by the specified
89       * {@link NetworkInterface}.
90       *
91       * @param netint
92       *            The core java {@link NetworkInterface} backing this object.
93       * @param displayName
94       *            A string to use for the display name in preference to the
95       *            {@link NetworkInterface} value.
96       * @throws InstantiationException
97       *             If a socket exception prevents access to the backing interface.
98       */
99      protected AbstractNetworkIF(NetworkInterface netint, String displayName) throws InstantiationException {
100         this.networkInterface = netint;
101         try {
102             this.name = networkInterface.getName();
103             this.displayName = displayName;
104             this.index = networkInterface.getIndex();
105             // Set MTU
106             this.mtu = networkInterface.getMTU();
107             // Set MAC
108             byte[] hwmac = networkInterface.getHardwareAddress();
109             if (hwmac != null) {
110                 List<String> octets = new ArrayList<>(6);
111                 for (byte b : hwmac) {
112                     octets.add(String.format("%02x", b));
113                 }
114                 this.mac = String.join(":", octets);
115             } else {
116                 this.mac = Constants.UNKNOWN;
117             }
118             // Set IP arrays
119             ArrayList<String> ipv4list = new ArrayList<>();
120             ArrayList<Short> subnetMaskList = new ArrayList<>();
121             ArrayList<String> ipv6list = new ArrayList<>();
122             ArrayList<Short> prefixLengthList = new ArrayList<>();
123 
124             for (InterfaceAddress interfaceAddress : networkInterface.getInterfaceAddresses()) {
125                 InetAddress address = interfaceAddress.getAddress();
126                 if (address.getHostAddress().length() > 0) {
127                     if (address.getHostAddress().contains(":")) {
128                         ipv6list.add(address.getHostAddress().split("%")[0]);
129                         prefixLengthList.add(interfaceAddress.getNetworkPrefixLength());
130                     } else {
131                         ipv4list.add(address.getHostAddress());
132                         subnetMaskList.add(interfaceAddress.getNetworkPrefixLength());
133                     }
134                 }
135             }
136 
137             this.ipv4 = ipv4list.toArray(new String[0]);
138             this.subnetMasks = subnetMaskList.toArray(new Short[0]);
139             this.ipv6 = ipv6list.toArray(new String[0]);
140             this.prefixLengths = prefixLengthList.toArray(new Short[0]);
141         } catch (SocketException e) {
142             throw new InstantiationException(e.getMessage());
143         }
144     }
145 
146     /**
147      * Returns network interfaces on this machine.
148      *
149      * @param includeLocalInterfaces
150      *            include local interfaces in the result
151      * @return A list of network interfaces
152      */
153     protected static List<NetworkInterface> getNetworkInterfaces(boolean includeLocalInterfaces) {
154         List<NetworkInterface> interfaces = getAllNetworkInterfaces();
155 
156         return includeLocalInterfaces ? interfaces
157                 : getAllNetworkInterfaces().stream().filter(networkInterface1 -> !isLocalInterface(networkInterface1))
158                         .collect(Collectors.toList());
159     }
160 
161     /**
162      * Returns all network interfaces.
163      *
164      * @return A list of network interfaces
165      */
166     private static List<NetworkInterface> getAllNetworkInterfaces() {
167         try {
168             Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
169             return interfaces == null ? Collections.emptyList() : Collections.list(interfaces);
170         } catch (SocketException ex) {
171             LOG.error("Socket exception when retrieving interfaces: {}", ex.getMessage());
172         }
173         return Collections.emptyList();
174     }
175 
176     private static boolean isLocalInterface(NetworkInterface networkInterface) {
177         try {
178             // getHardwareAddress also checks for loopback
179             return networkInterface.getHardwareAddress() == null;
180         } catch (SocketException e) {
181             LOG.error("Socket exception when retrieving interface information for {}: {}", networkInterface,
182                     e.getMessage());
183         }
184         return false;
185     }

237     @Override
238     public boolean isKnownVmMacAddr() {
239         String oui = getMacaddr().length() > 7 ? getMacaddr().substring(0, 8) : getMacaddr();
240         return this.vmMacAddrProps.get().containsKey(oui.toUpperCase());
241     }
242 
243     private static Properties queryVmMacAddrProps() {
244         return FileUtil.readPropertiesFromFilename(OSHI_VM_MAC_ADDR_PROPERTIES);
245     }

247     @Override
248     public String toString() {
249         StringBuilder sb = new StringBuilder();
250         sb.append("Name: ").append(getName());
251         if (!getName().equals(getDisplayName())) {
252             sb.append(" (").append(getDisplayName()).append(")");
253         }
254         if (!getIfAlias().isEmpty()) {
255             sb.append(" [IfAlias=").append(getIfAlias()).append("]");
256         }
257         sb.append("\n");
258         sb.append("  MAC Address: ").append(getMacaddr()).append("\n");
259         sb.append("  MTU: ").append(ParseUtil.unsignedIntToLong(getMTU())).append(", ").append("Speed: ")
260                 .append(getSpeed()).append("\n");
261         String[] ipv4withmask = getIPv4addr();
262         if (this.ipv4.length == this.subnetMasks.length) {
263             for (int i = 0; i < this.subnetMasks.length; i++) {
264                 ipv4withmask[i] += "/" + this.subnetMasks[i];
265             }
266         }
267         sb.append("  IPv4: ").append(Arrays.toString(ipv4withmask)).append("\n");
268         String[] ipv6withprefixlength = getIPv6addr();
269         if (this.ipv6.length == this.prefixLengths.length) {
270             for (int j = 0; j < this.prefixLengths.length; j++) {
271                 ipv6withprefixlength[j] += "/" + this.prefixLengths[j];
272             }
273         }
274         sb.append("  IPv6: ").append(Arrays.toString(ipv6withprefixlength)).append("\n");
275         sb.append("  Traffic: received ").append(getPacketsRecv()).append(" packets/")
276                 .append(FormatUtil.formatBytes(getBytesRecv())).append(" (" + getInErrors() + " err, ")
277                 .append(getInDrops() + " drop);");
278         sb.append(" transmitted ").append(getPacketsSent()).append(" packets/")
279                 .append(FormatUtil.formatBytes(getBytesSent())).append(" (" + getOutErrors() + " err, ")
280                 .append(getCollisions() + " coll);");
281         return sb.toString();
282     }
