45  /**
46   * Utility to handle WMI Queries. Designed to be extended with user-customized
47   * behavior.
48   */
49  @ThreadSafe
50  public class WmiQueryHandler {
51  
52      private static final Logger LOG = LoggerFactory.getLogger(WmiQueryHandler.class);
53  
54      private static int globalTimeout = GlobalConfig.get("oshi.util.wmi.timeout", -1);
55  
56      static {
57          if (globalTimeout == 0 || globalTimeout < -1) {
58              throw new GlobalConfig.PropertyException("oshi.util.wmi.timeout");
59          }
60      }
61  
62      // Timeout for WMI queries
63      protected int wmiTimeout = globalTimeout;
64  
65      // Cache failed wmi classes
66      protected final Set<String> failedWmiClassNames = new HashSet<>();
67  
68      // Preferred threading model
69      private int comThreading = Ole32.COINIT_MULTITHREADED;
70  
71      // Track initialization of Security
72      private boolean securityInitialized = false;
73  
74      private static final Class<?>[] EMPTY_CLASS_ARRAY = new Class<?>[0];
75      private static final Object[] EMPTY_OBJECT_ARRAY = new Object[0];
76  
77      // Factory to create this or a subclass
78      private static Class<? extends WmiQueryHandler> customClass = null;
79  
80      /**
81       * Factory method to create an instance of this class. To override this class,
82       * use {@link #setInstanceClass(Class)} to define a subclass which extends
83       * {@link oshi.util.platform.windows.WmiQueryHandler}.
84       *
85       * @return An instance of this class or a class defined by
86       *         {@link #setInstanceClass(Class)}
87       */
88      public static synchronized WmiQueryHandler createInstance() {
89          if (customClass == null) {
90              return new WmiQueryHandler();
91          }
92          try {
93              return customClass.getConstructor(EMPTY_CLASS_ARRAY).newInstance(EMPTY_OBJECT_ARRAY);
94          } catch (NoSuchMethodException | SecurityException e) {
95              LOG.error("Failed to find or access a no-arg constructor for {}", customClass);
96          } catch (InstantiationException | IllegalAccessException | IllegalArgumentException
97                  | InvocationTargetException e) {
98              LOG.error("Failed to create a new instance of {}", customClass);
99          }
100         return null;
101     }
102 
103     /**
104      * Define a subclass to be instantiated by {@link #createInstance()}. The class
105      * must extend {@link oshi.util.platform.windows.WmiQueryHandler}.
106      *
107      * @param instanceClass
108      *            The class to instantiate with {@link #createInstance()}.
109      */
110     public static synchronized void setInstanceClass(Class<? extends WmiQueryHandler> instanceClass) {
111         customClass = instanceClass;
112     }

114     /**
115      * Query WMI for values. Makes no assumptions on whether the user has previously
116      * initialized COM.
117      *
118      * @param <T>
119      *            WMI queries use an Enum to identify the fields to query, and use
120      *            the enum values as keys to retrieve the results.
121      * @param query
122      *            A WmiQuery object encapsulating the namespace, class, and
123      *            properties
124      * @return a WmiResult object containing the query results, wrapping an EnumMap
125      */
126     public <T extends Enum<T>> WbemcliUtil.WmiResult<T> queryWMI(WbemcliUtil.WmiQuery<T> query) {
127         return queryWMI(query, true);
128     }
129 
130     /**
131      * Query WMI for values.
132      *
133      * @param <T>
134      *            WMI queries use an Enum to identify the fields to query, and use
135      *            the enum values as keys to retrieve the results.
136      * @param query
137      *            A WmiQuery object encapsulating the namespace, class, and
138      *            properties
139      * @param initCom
140      *            Whether to initialize COM. If {@code true}, initializes COM before
141      *            the query and uninitializes after. If {@code false}, assumes the
142      *            user has initialized COM separately. This can improve WMI query
143      *            performance.
144      * @return a WmiResult object containing the query results, wrapping an EnumMap
145      */
146     public <T extends Enum<T>> WbemcliUtil.WmiResult<T> queryWMI(WbemcliUtil.WmiQuery<T> query, boolean initCom) {
147         WbemcliUtil.WmiResult<T> result = WbemcliUtil.INSTANCE.new WmiResult<>(query.getPropertyEnum());
148         if (failedWmiClassNames.contains(query.getWmiClassName())) {
149             return result;
150         }
151         boolean comInit = false;
152         try {
153             if (initCom) {
154                 comInit = initCOM();
155             }
156             result = query.execute(wmiTimeout);
157         } catch (COMException e) {
158             // Ignore any exceptions with OpenHardwareMonitor
159             if (!WmiUtil.OHM_NAMESPACE.equals(query.getNameSpace())) {
160                 final int hresult = e.getHresult() == null ? -1 : e.getHresult().intValue();
161                 switch (hresult) {
162                 case Wbemcli.WBEM_E_INVALID_NAMESPACE:
163                     LOG.warn("COM exception: Invalid Namespace {}", query.getNameSpace());
164                     break;
165                 case Wbemcli.WBEM_E_INVALID_CLASS:
166                     LOG.warn("COM exception: Invalid Class {}", query.getWmiClassName());
167                     break;
168                 case Wbemcli.WBEM_E_INVALID_QUERY:
169                     LOG.warn("COM exception: Invalid Query: {}", WmiUtil.queryToString(query));
170                     break;
171                 default:
172                     handleComException(query, e);
173                     break;
174                 }
175                 failedWmiClassNames.add(query.getWmiClassName());
176             }
177         } catch (TimeoutException e) {
178             LOG.warn("WMI query timed out after {} ms: {}", wmiTimeout, WmiUtil.queryToString(query));
179         }
180         if (comInit) {
181             unInitCOM();
182         }
183         return result;
184     }
185 
186     /**
187      * COM Exception handler. Logs a warning message.
188      *
189      * @param query
190      *            a {@link com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery}
191      *            object.
192      * @param ex
193      *            a {@link com.sun.jna.platform.win32.COM.COMException} object.
194      */
195     protected void handleComException(WbemcliUtil.WmiQuery<?> query, COMException ex) {
196         LOG.warn(
197                 "COM exception querying {}, which might not be on your system. Will not attempt to query it again. Error was {}: {}",
198                 query.getWmiClassName(), ex.getHresult().intValue(), ex.getMessage());
199     }

201     /**
202      * Initializes COM library and sets security to impersonate the local user
203      *
204      * @return True if COM was initialized and needs to be uninitialized, false
205      *         otherwise
206      */
207     public boolean initCOM() {
208         boolean comInit = false;
209         // Step 1: --------------------------------------------------
210         // Initialize COM. ------------------------------------------
211         comInit = initCOM(getComThreading());
212         if (!comInit) {
213             comInit = initCOM(switchComThreading());
214         }
215         // Step 2: --------------------------------------------------
216         // Set general COM security levels --------------------------
217         if (comInit && !isSecurityInitialized()) {
218             WinNT.HRESULT hres = Ole32.INSTANCE.CoInitializeSecurity(null, -1, null, null,
219                     Ole32.RPC_C_AUTHN_LEVEL_DEFAULT, Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE, null);
220             // If security already initialized we get RPC_E_TOO_LATE
221             // This can be safely ignored
222             if (COMUtils.FAILED(hres) && hres.intValue() != WinError.RPC_E_TOO_LATE) {
223                 Ole32.INSTANCE.CoUninitialize();
224                 throw new COMException("Failed to initialize security.", hres);
225             }
226             securityInitialized = true;
227         }
228         return comInit;
229     }
230 
231     /**
232      * Initializes COM with a specific threading model
233      *
234      * @param coInitThreading
235      *            The threading model
236      * @return True if COM was initialized and needs to be uninitialized, false
237      *         otherwise
238      */
239     protected boolean initCOM(int coInitThreading) {
240         WinNT.HRESULT hres = Ole32.INSTANCE.CoInitializeEx(null, coInitThreading);
241         switch (hres.intValue()) {
242         // Successful local initialization (S_OK) or was already initialized
243         // (S_FALSE) but still needs uninit
244         case COMUtils.S_OK:
245         case COMUtils.S_FALSE:
246             return true;
247         // COM was already initialized with a different threading model
248         case WinError.RPC_E_CHANGED_MODE:
249             return false;
250         // Any other results is impossible
251         default:
252             throw new COMException("Failed to initialize COM library.", hres);
253         }
254     }
255 
256     /**
257      * UnInitializes COM library. This should be called once for every successful
258      * call to initCOM.
259      */
260     public void unInitCOM() {
261         Ole32.INSTANCE.CoUninitialize();
262     }
263 
264     /**
265      * Returns the current threading model for COM initialization, as OSHI is
266      * required to match if an external program has COM initialized already.
267      *
268      * @return The current threading model
269      */
270     public int getComThreading() {
271         return comThreading;
272     }
273 
274     /**
275      * Switches the current threading model for COM initialization, as OSHI is
276      * required to match if an external program has COM initialized already.
277      *
278      * @return The new threading model after switching
279      */
280     public int switchComThreading() {
281         if (comThreading == Ole32.COINIT_APARTMENTTHREADED) {
282             comThreading = Ole32.COINIT_MULTITHREADED;
283         } else {
284             comThreading = Ole32.COINIT_APARTMENTTHREADED;
285         }
286         return comThreading;
287     }
288 
289     /**
290      * Security only needs to be initialized once. This boolean identifies whether
291      * that has happened.
292      *
293      * @return Returns the securityInitialized.
294      */
295     public boolean isSecurityInitialized() {
296         return securityInitialized;
297     }

299     /**
300      * Gets the current WMI timeout. WMI queries will fail if they take longer than
301      * this number of milliseconds. A value of -1 is infinite (no timeout).
302      *
303      * @return Returns the current value of wmiTimeout.
304      */
305     public int getWmiTimeout() {
306         return wmiTimeout;
307     }
308 
309     /**
310      * Sets the WMI timeout. WMI queries will fail if they take longer than this
311      * number of milliseconds.
312      *
313      * @param wmiTimeout
314      *            The wmiTimeout to set, in milliseconds. To disable timeouts, set
315      *            timeout as -1 (infinite).

318         this.wmiTimeout = wmiTimeout;
319     }
