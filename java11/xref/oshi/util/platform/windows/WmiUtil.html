56      /**
57       * Translate a WmiQuery to the actual query string
58       *
59       * @param <T>
60       *            WMI queries use an Enum to identify the fields to query, and use
61       *            the enum values as keys to retrieve the results.
62       * @param query
63       *            The WmiQuery object
64       * @return The string that is queried in WMI
65       */
66      public static <T extends Enum<T>> String queryToString(WmiQuery<T> query) {
67          T[] props = query.getPropertyEnum().getEnumConstants();
68          StringBuilder sb = new StringBuilder("SELECT ");
69          sb.append(props[0].name());
70          for (int i = 1; i < props.length; i++) {
71              sb.append(',').append(props[i].name());
72          }
73          sb.append(" FROM ").append(query.getWmiClassName());
74          return sb.toString();
75      }

77      /**
78       * Gets a String value from a WmiResult
79       *
80       * @param <T>
81       *            WMI queries use an Enum to identify the fields to query, and use
82       *            the enum values as keys to retrieve the results.
83       * @param result
84       *            The WmiResult from which to fetch the value
85       * @param property
86       *            The property (column) to fetch
87       * @param index
88       *            The index (row) to fetch
89       * @return The stored value if non-null, an empty-string otherwise
90       */
91      public static <T extends Enum<T>> String getString(WmiResult<T> result, T property, int index) {
92          if (result.getCIMType(property) == Wbemcli.CIM_STRING) {
93              return getStr(result, property, index);
94          }
95          throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "String",
96                  result.getCIMType(property), result.getVtType(property)));
97      }

99      /**
100      * Gets a Date value from a WmiResult as a String in ISO 8601 format
101      *
102      * @param <T>
103      *            WMI queries use an Enum to identify the fields to query, and use
104      *            the enum values as keys to retrieve the results.
105      * @param result
106      *            The WmiResult from which to fetch the value
107      * @param property
108      *            The property (column) to fetch
109      * @param index
110      *            The index (row) to fetch
111      * @return The stored value if non-null, an empty-string otherwise
112      */
113     public static <T extends Enum<T>> String getDateString(WmiResult<T> result, T property, int index) {
114         OffsetDateTime dateTime = getDateTime(result, property, index);
115         // Null result returns the Epoch
116         if (dateTime.equals(Constants.UNIX_EPOCH)) {
117             return "";
118         }
119         return dateTime.toLocalDate().toString();
120     }

122     /**
123      * Gets a DateTime value from a WmiResult as an OffsetDateTime
124      *
125      * @param <T>
126      *            WMI queries use an Enum to identify the fields to query, and use
127      *            the enum values as keys to retrieve the results.
128      * @param result
129      *            The WmiResult from which to fetch the value
130      * @param property
131      *            The property (column) to fetch
132      * @param index
133      *            The index (row) to fetch
134      * @return The stored value if non-null, otherwise the constant
135      *         {@link oshi.util.Constants#UNIX_EPOCH}
136      */
137     public static <T extends Enum<T>> OffsetDateTime getDateTime(WmiResult<T> result, T property, int index) {
138         if (result.getCIMType(property) == Wbemcli.CIM_DATETIME) {
139             return ParseUtil.parseCimDateTimeToOffset(getStr(result, property, index));
140         }
141         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "DateTime",
142                 result.getCIMType(property), result.getVtType(property)));
143     }

167     private static <T extends Enum<T>> String getStr(WmiResult<T> result, T property, int index) {
168         Object o = result.getValue(property, index);
169         if (o == null) {
170             return "";
171         } else if (result.getVtType(property) == Variant.VT_BSTR) {
172             return (String) o;
173         }
174         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "String-mapped",
175                 result.getCIMType(property), result.getVtType(property)));
176     }

178     /**
179      * Gets a Uint64 value from a WmiResult (parsing the String). Note that while
180      * the CIM type is unsigned, the return type is signed and the parsing will
181      * exclude any return values above Long.MAX_VALUE.
182      *
183      * @param <T>
184      *            WMI queries use an Enum to identify the fields to query, and use
185      *            the enum values as keys to retrieve the results.
186      * @param result
187      *            The WmiResult from which to fetch the value
188      * @param property
189      *            The property (column) to fetch
190      * @param index
191      *            The index (row) to fetch
192      * @return The stored value if non-null and parseable as a long, 0 otherwise
193      */
194     public static <T extends Enum<T>> long getUint64(WmiResult<T> result, T property, int index) {
195         Object o = result.getValue(property, index);
196         if (o == null) {
197             return 0L;
198         } else if (result.getCIMType(property) == Wbemcli.CIM_UINT64 && result.getVtType(property) == Variant.VT_BSTR) {
199             return ParseUtil.parseLongOrDefault((String) o, 0L);
200         }
201         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "UINT64",
202                 result.getCIMType(property), result.getVtType(property)));
203     }

229     /**
230      * Gets an UINT32 value from a WmiResult as a long, preserving the unsignedness.
231      *
232      * @param <T>
233      *            WMI queries use an Enum to identify the fields to query, and use
234      *            the enum values as keys to retrieve the results.
235      * @param result
236      *            The WmiResult from which to fetch the value
237      * @param property
238      *            The property (column) to fetch
239      * @param index
240      *            The index (row) to fetch
241      * @return The stored value if non-null, 0 otherwise
242      */
243     public static <T extends Enum<T>> long getUint32asLong(WmiResult<T> result, T property, int index) {
244         if (result.getCIMType(property) == Wbemcli.CIM_UINT32) {
245             return getInt(result, property, index) & 0xFFFFFFFFL;
246         }
247         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "UINT32",
248                 result.getCIMType(property), result.getVtType(property)));
249     }

299     private static <T extends Enum<T>> int getInt(WmiResult<T> result, T property, int index) {
300         Object o = result.getValue(property, index);
301         if (o == null) {
302             return 0;
303         } else if (result.getVtType(property) == Variant.VT_I4) {
304             return (int) o;
305         }
306         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "32-bit integer",
307                 result.getCIMType(property), result.getVtType(property)));
308     }

311      * Gets a Float value from a WmiResult
312      *
313      * @param <T>
314      *            WMI queries use an Enum to identify the fields to query, and use
315      *            the enum values as keys to retrieve the results.
316      * @param result
317      *            The WmiResult from which to fetch the value
318      * @param property
319      *            The property (column) to fetch
320      * @param index
321      *            The index (row) to fetch
322      * @return The stored value if non-null, 0 otherwise
323      */
324     public static <T extends Enum<T>> float getFloat(WmiResult<T> result, T property, int index) {
325         Object o = result.getValue(property, index);
326         if (o == null) {
327             return 0f;
328         } else if (result.getCIMType(property) == Wbemcli.CIM_REAL32 && result.getVtType(property) == Variant.VT_R4) {
329             return (float) o;
330         }
331         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "Float",
332                 result.getCIMType(property), result.getVtType(property)));
