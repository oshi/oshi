24  package oshi.driver.unix.aix.perfstat;
25  
26  import com.sun.jna.platform.unix.aix.Perfstat; // NOSONAR squid:S1191
27  import com.sun.jna.platform.unix.aix.Perfstat.perfstat_cpu_t;
28  import com.sun.jna.platform.unix.aix.Perfstat.perfstat_cpu_total_t;
29  import com.sun.jna.platform.unix.aix.Perfstat.perfstat_id_t;
30  
31  import oshi.annotation.concurrent.ThreadSafe;
32  
33  /**
34   * Utility to query performance stats for cpu
35   */
36  @ThreadSafe
37  public final class PerfstatCpu {
38  
39      private static final Perfstat PERF = Perfstat.INSTANCE;
40  
41      private PerfstatCpu() {
42      }
43  
44      /**
45       * Queries perfstat_cpu_total for total CPU usage statistics
46       *
47       * @return usage statistics
48       */
49      public static perfstat_cpu_total_t queryCpuTotal() {
50          perfstat_cpu_total_t cpu = new perfstat_cpu_total_t();
51          int ret = PERF.perfstat_cpu_total(null, cpu, cpu.size(), 1);
52          if (ret > 0) {
53              return cpu;
54          }
55          return new perfstat_cpu_total_t();
56      }
57  
58      /**
59       * Queries perfstat_cpu for per-CPU usage statistics
60       *
61       * @return an array of usage statistics
62       */
63      public static perfstat_cpu_t[] queryCpu() {
64          perfstat_cpu_t cpu = new perfstat_cpu_t();
65          // With null, null, ..., 0, returns total # of elements
66          int cputotal = PERF.perfstat_cpu(null, null, cpu.size(), 0);
67          if (cputotal > 0) {
68              perfstat_cpu_t[] statp = (perfstat_cpu_t[]) cpu.toArray(cputotal);
69              perfstat_id_t firstcpu = new perfstat_id_t(); // name is ""
70              int ret = PERF.perfstat_cpu(firstcpu, statp, cpu.size(), cputotal);
71              if (ret > 0) {
72                  return statp;
73              }
74          }
75          return new perfstat_cpu_t[0];
76      }
77  
78      /**
79       * Returns affinity mask from the number of CPU in the OS.
80       *
81       * @return affinity mask
82       */
83      public static long queryCpuAffinityMask() {
84          int cpus = queryCpuTotal().ncpus;
85          if (cpus < 63) {
86              return (1L << cpus) - 1;
87          }
88          return cpus == 63 ? Long.MAX_VALUE : -1L;
89      }
90  }
