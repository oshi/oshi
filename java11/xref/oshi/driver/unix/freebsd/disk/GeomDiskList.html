24  package oshi.driver.unix.freebsd.disk;
25  
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  
30  import oshi.annotation.concurrent.ThreadSafe;
31  import oshi.util.Constants;
32  import oshi.util.ExecutingCommand;
33  import oshi.util.ParseUtil;
34  import oshi.util.tuples.Triplet;
35  
36  /**
37   * Utility to query geom part list
38   */
39  @ThreadSafe
40  public final class GeomDiskList {
41  
42      private static final String GEOM_DISK_LIST = "geom disk list";
43  
44      private GeomDiskList() {
45      }
46  
47      /**
48       * Queries disk data using geom
49       *
50       * @return A map with disk name as the key and a Triplet of model, serial, and
51       *         size as the value
52       */
53      public static Map<String, Triplet<String, String, Long>> queryDisks() {
54          // Map of device name to disk, to be returned
55          Map<String, Triplet<String, String, Long>> diskMap = new HashMap<>();
56          // Parameters needed.
57          String diskName = null; // Non-null identifies a valid partition
58          String descr = Constants.UNKNOWN;
59          String ident = Constants.UNKNOWN;
60          long mediaSize = 0L;
61  
62          List<String> geom = ExecutingCommand.runNative(GEOM_DISK_LIST);
63          for (String line : geom) {
64              line = line.trim();
65              // Marks the DiskStore device
66              if (line.startsWith("Geom name:")) {
67                  // Save any previous disk in the map
68                  if (diskName != null) {
69                      diskMap.put(diskName, new Triplet<>(descr, ident, mediaSize));
70                      descr = Constants.UNKNOWN;
71                      ident = Constants.UNKNOWN;
72                      mediaSize = 0L;
73                  }
74                  // Now use new diskName
75                  diskName = line.substring(line.lastIndexOf(' ') + 1);
76              }
77              // If we don't have a valid store, don't bother parsing anything
78              if (diskName != null) {
79                  line = line.trim();
80                  if (line.startsWith("Mediasize:")) {
81                      String[] split = ParseUtil.whitespaces.split(line);
82                      if (split.length > 1) {
83                          mediaSize = ParseUtil.parseLongOrDefault(split[1], 0L);
84                      }
85                  }
86                  if (line.startsWith("descr:")) {
87                      descr = line.replace("descr:", "").trim();
88                  }
89                  if (line.startsWith("ident:")) {
90                      ident = line.replace("ident:", "").replace("(null)", "").trim();
91                  }
92              }
93          }
94          if (diskName != null) {
95              diskMap.put(diskName, new Triplet<>(descr, ident, mediaSize));
96          }
97          return diskMap;
98      }
99  }
