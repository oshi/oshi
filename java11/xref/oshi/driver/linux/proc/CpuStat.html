24  package oshi.driver.linux.proc;
25  
26  import java.util.List;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  import oshi.hardware.CentralProcessor.TickType;
30  import oshi.util.FileUtil;
31  import oshi.util.ParseUtil;
32  import oshi.util.platform.linux.ProcPath;
33  
34  /**
35   * Utility to read CPU statistics from {@code /proc/stat}
36   */
37  @ThreadSafe
38  public final class CpuStat {
39  
40      private CpuStat() {
41      }
42  
43      /**
44       * Gets the System CPU ticks array from {@code /proc/stat}
45       *
46       * @return Array of CPU ticks
47       */
48      public static long[] getSystemCpuLoadTicks() {
49          long[] ticks = new long[TickType.values().length];
50          // /proc/stat expected format
51          // first line is overall user,nice,system,idle,iowait,irq, etc.
52          // cpu 3357 0 4313 1362393 ...
53          String tickStr;
54          List<String> procStat = FileUtil.readFile(ProcPath.STAT);
55          if (!procStat.isEmpty()) {
56              tickStr = procStat.get(0);
57          } else {
58              return ticks;
59          }
60          // Split the line. Note the first (0) element is "cpu" so remaining
61          // elements are offset by 1 from the enum index
62          String[] tickArr = ParseUtil.whitespaces.split(tickStr);
63          if (tickArr.length <= TickType.IDLE.getIndex()) {
64              // If ticks don't at least go user/nice/system/idle, abort
65              return ticks;
66          }
67          // Note tickArr is offset by 1 because first element is "cpu"
68          for (int i = 0; i < TickType.values().length; i++) {
69              ticks[i] = ParseUtil.parseLongOrDefault(tickArr[i + 1], 0L);
70          }
71          // Ignore guest or guest_nice, they are included in user/nice
72          return ticks;
73      }
74  
75      /**
76       * Gets an arrya of Processor CPU ticks array from /proc/stat
77       *
78       * @param logicalProcessorCount
79       *            The number of logical processors, which corresponds to the number
80       *            of lines to read from the file.
81       * @return Array of CPU ticks for each processor
82       */
83      public static long[][] getProcessorCpuLoadTicks(int logicalProcessorCount) {
84          long[][] ticks = new long[logicalProcessorCount][TickType.values().length];
85          // /proc/stat expected format
86          // first line is overall user,nice,system,idle, etc.
87          // cpu 3357 0 4313 1362393 ...
88          // per-processor subsequent lines for cpu0, cpu1, etc.
89          int cpu = 0;
90          List<String> procStat = FileUtil.readFile(ProcPath.STAT);
91          for (String stat : procStat) {
92              if (stat.startsWith("cpu") && !stat.startsWith("cpu ")) {
93                  // Split the line. Note the first (0) element is "cpu" so
94                  // remaining
95                  // elements are offset by 1 from the enum index
96                  String[] tickArr = ParseUtil.whitespaces.split(stat);
97                  if (tickArr.length <= TickType.IDLE.getIndex()) {
98                      // If ticks don't at least go user/nice/system/idle, abort
99                      return ticks;
100                 }
101                 // Note tickArr is offset by 1
102                 for (int i = 0; i < TickType.values().length; i++) {
103                     ticks[cpu][i] = ParseUtil.parseLongOrDefault(tickArr[i + 1], 0L);
104                 }
105                 // Ignore guest or guest_nice, they are included in
106                 if (++cpu >= logicalProcessorCount) {
107                     break;
108                 }
109             }
110         }
111         return ticks;
112     }
113 
114     /**
115      * Gets the number of context switches from /proc/stat
116      *
117      * @return The number of context switches if available, -1 otherwise
118      */
119     public static long getContextSwitches() {
120         List<String> procStat = FileUtil.readFile(ProcPath.STAT);
121         for (String stat : procStat) {
122             if (stat.startsWith("ctxt ")) {
123                 String[] ctxtArr = ParseUtil.whitespaces.split(stat);
124                 if (ctxtArr.length == 2) {
125                     return ParseUtil.parseLongOrDefault(ctxtArr[1], 0);
126                 }
127             }
128         }
129         return 0L;
130     }
131 
132     /**
133      * Gets the number of interrupts from /proc/stat
134      *
135      * @return The number of interrupts if available, -1 otherwise
136      */
137     public static long getInterrupts() {
138         List<String> procStat = FileUtil.readFile(ProcPath.STAT);
139         for (String stat : procStat) {
140             if (stat.startsWith("intr ")) {
141                 String[] intrArr = ParseUtil.whitespaces.split(stat);
142                 if (intrArr.length > 2) {
143                     return ParseUtil.parseLongOrDefault(intrArr[1], 0);
144                 }
145             }
146         }
147         return 0L;
148     }
149 
150     /**
151      * Gets the boot time from /proc/stat
152      *
153      * @return The boot time if available, 0 otherwise
154      */
155     public static long getBootTime() {
156         // Boot time given by btime variable in /proc/stat.
157         List<String> procStat = FileUtil.readFile(ProcPath.STAT);
158         for (String stat : procStat) {
159             if (stat.startsWith("btime")) {
160                 String[] bTime = ParseUtil.whitespaces.split(stat);
161                 return ParseUtil.parseLongOrDefault(bTime[1], 0L);
162             }
163         }
164         return 0;
165     }
166 }
