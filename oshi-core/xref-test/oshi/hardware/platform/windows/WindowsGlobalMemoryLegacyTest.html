5   package oshi.hardware.platform.windows;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.hasSize;
9   import static org.hamcrest.Matchers.is;
10  
11  import java.util.ArrayList;
12  import java.util.Comparator;
13  import java.util.List;
14  
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledOnOs;
17  import org.junit.jupiter.api.condition.OS;
18  
19  import oshi.hardware.PhysicalMemory;
20  
21  /**
22   * Exercises the pre-Windows-10 physical-memory fallback path against the real system by forcing the
23   * {@code isWindows10OrGreater()} version gate. Only the version flag is overridden, so the genuine WMI query runs; a
24   * rotted legacy path surfaces as a divergence from the modern path rather than as a mere coverage number.
25   */
26  @EnabledOnOs(OS.WINDOWS)
27  class WindowsGlobalMemoryLegacyTest {
28  
29      // Stable ordering on fields that must agree between the two schemas, so the comparison can't flake on WMI row order
30      private static final Comparator<PhysicalMemory> BY_BANK = Comparator.comparing(PhysicalMemory::getBankLabel)
31              .thenComparing(PhysicalMemory::getSerialNumber).thenComparingLong(PhysicalMemory::getCapacity);
32  
33      /** A real {@link WindowsGlobalMemoryJNA} with only the Windows-version gate forced. */
34      private static final class ForcedVersion extends WindowsGlobalMemoryJNA {
35          private final boolean windows10OrGreater;
36  
37          ForcedVersion(boolean windows10OrGreater) {
38              this.windows10OrGreater = windows10OrGreater;
39          }
40  
41          @Override
42          protected boolean isWindows10OrGreater() {
43              return windows10OrGreater;
44          }
45      }
46  
47      /**
48       * The legacy (pre-Win10) and modern physical-memory queries hit the same {@code Win32_PhysicalMemory} class and
49       * differ only in how they resolve the memory-type string, so every other field must agree bank-for-bank. A broken
50       * legacy query (wrong field, wrong parse) shows up here as a mismatch. When the runner reports no banks both lists
51       * are empty and the legacy branch is still exercised.
52       */
53      @Test
54      void legacyPhysicalMemoryMatchesModernExceptType() {
55          List<PhysicalMemory> legacy = new ArrayList<>(new ForcedVersion(false).getPhysicalMemory());
56          List<PhysicalMemory> modern = new ArrayList<>(new ForcedVersion(true).getPhysicalMemory());
57          legacy.sort(BY_BANK);
58          modern.sort(BY_BANK);
59  
60          assertThat("legacy and modern must report the same bank count", legacy, hasSize(modern.size()));
61          for (int i = 0; i < modern.size(); i++) {
62              PhysicalMemory l = legacy.get(i);
63              PhysicalMemory m = modern.get(i);
64              assertThat(l.getBankLabel(), is(m.getBankLabel()));
65              assertThat(l.getCapacity(), is(m.getCapacity()));
66              assertThat(l.getClockSpeed(), is(m.getClockSpeed()));
67              assertThat(l.getManufacturer(), is(m.getManufacturer()));
68              assertThat(l.getPartNumber(), is(m.getPartNumber()));
69              assertThat(l.getSerialNumber(), is(m.getSerialNumber()));
70              // memoryType is intentionally not compared: legacy resolves Win32 MemoryType, modern SMBIOSMemoryType
71          }
72      }
73  }
