24  package oshi.hardware;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.both;
28  import static org.hamcrest.Matchers.containsString;
29  import static org.hamcrest.Matchers.greaterThan;
30  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
31  import static org.hamcrest.Matchers.is;
32  import static org.hamcrest.Matchers.lessThanOrEqualTo;
33  import static org.hamcrest.Matchers.notNullValue;
34  
35  import java.util.List;
36  
37  import org.junit.jupiter.api.BeforeAll;
38  import org.junit.jupiter.api.Test;
39  import org.junit.jupiter.api.TestInstance;
40  import org.junit.jupiter.api.TestInstance.Lifecycle;
41  
42  import oshi.SystemInfo;
43  
44  /**
45   * Test GlobalMemory
46   */
47  @TestInstance(Lifecycle.PER_CLASS)
48  class GlobalMemoryTest {
49      private GlobalMemory memory = null;
50  
51      @BeforeAll
52      void setUp() {
53          SystemInfo si = new SystemInfo();
54          HardwareAbstractionLayer hal = si.getHardware();
55          this.memory = hal.getMemory();
56      }
57  
58      @Test
59      void testGlobalMemory() {
60          assertThat("Memory shouldn't be null", memory, is(notNullValue()));
61          assertThat("Total memory should be greater than zero", memory.getTotal(), is(greaterThan(0L)));
62          assertThat("Available memory should be between 0 and total memory", memory.getAvailable(),
63                  is(both(greaterThanOrEqualTo(0L)).and(lessThanOrEqualTo(memory.getTotal()))));
64          assertThat("Memory page size should be greater than zero", memory.getPageSize(), is(greaterThan(0L)));
65          assertThat("Memory toString should contain the substring \"Available\"", memory.toString(),
66                  containsString("Available"));
67      }
68  
69      @Test
70      void testPhysicalMemory() {
71          List<PhysicalMemory> pm = memory.getPhysicalMemory();
72          for (PhysicalMemory m : pm) {
73              assertThat("Bank label shouldn't be null", m.getBankLabel(), is(notNullValue()));
74              assertThat("Capacity should be nonnegative", m.getCapacity(), is(greaterThanOrEqualTo(0L)));
75              assertThat("Speed should be nonnegative", m.getClockSpeed(), is(greaterThanOrEqualTo(0L)));
76              assertThat("Manufacturer shouldn't be null", m.getManufacturer(), is(notNullValue()));
77              assertThat("Memory type shouldn't be null", m.getMemoryType(), is(notNullValue()));
78          }
79      }
80  }
