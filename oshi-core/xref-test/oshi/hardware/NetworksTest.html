19  package oshi.hardware;
20  
21  import static org.junit.Assert.assertEquals;
22  import static org.junit.Assert.assertNotNull;
23  import static org.junit.Assert.assertTrue;
24  
25  import java.io.IOException;
26  
27  import org.junit.Test;
28  
29  import oshi.SystemInfo;
30  
31  /**
32   * Test Networks
33   */
34  public class NetworksTest {
35      /**
36       * Test network interfaces extraction.
37       *
38       * @throws IOException
39       *             Signals that an I/O exception has occurred.
40       */
41      @Test
42      public void testNetworkInterfaces() throws IOException {
43          long timeStamp = System.currentTimeMillis();
44          SystemInfo si = new SystemInfo();
45  
46          for (NetworkIF net : si.getHardware().getNetworkIFs()) {
47              assertNotNull(net.getNetworkInterface());
48              assertNotNull(net.getName());
49              assertNotNull(net.getDisplayName());
50              assertNotNull(net.getMacaddr());
51              assertNotNull(net.getIPv4addr());
52              assertNotNull(net.getIPv6addr());
53              assertTrue(net.getBytesRecv() >= 0);
54              assertTrue(net.getBytesSent() >= 0);
55              assertTrue(net.getPacketsRecv() >= 0);
56              assertTrue(net.getPacketsSent() >= 0);
57              assertTrue(net.getInErrors() >= 0);
58              assertTrue(net.getOutErrors() >= 0);
59              assertTrue(net.getSpeed() >= 0);
60              assertTrue(net.getMTU() >= 0);
61              assertTrue(net.getTimeStamp() > 0);
62  
63              net.setBytesRecv(10L);
64              net.setBytesSent(20L);
65              net.setPacketsRecv(30L);
66              net.setPacketsSent(40L);
67              net.setInErrors(60L);
68              net.setOutErrors(70L);
69              net.setSpeed(50L);
70              net.setTimeStamp(timeStamp);
71  
72              assertEquals(10L, net.getBytesRecv());
73              assertEquals(20L, net.getBytesSent());
74              assertEquals(30L, net.getPacketsRecv());
75              assertEquals(40L, net.getPacketsSent());
76              assertEquals(60L, net.getInErrors());
77              assertEquals(70L, net.getOutErrors());
78              assertEquals(50L, net.getSpeed());
79              assertEquals(timeStamp, net.getTimeStamp());
80  
81              net.updateNetworkStats();
82              assertTrue(net.getBytesRecv() >= 0);
83              assertTrue(net.getBytesSent() >= 0);
84              assertTrue(net.getPacketsRecv() >= 0);
85              assertTrue(net.getPacketsSent() >= 0);
86              assertTrue(net.getInErrors() >= 0);
87              assertTrue(net.getOutErrors() >= 0);
88              assertTrue(net.getSpeed() >= 0);
89              assertTrue(net.getTimeStamp() > 0);
90          }
91      }
92  }
