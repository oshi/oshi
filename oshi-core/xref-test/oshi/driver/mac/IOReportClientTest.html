5   package oshi.driver.mac;

7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.greaterThanOrEqualTo;
10  import static org.hamcrest.Matchers.is;
11  import static org.hamcrest.Matchers.notNullValue;
12  import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
13  
14  import org.junit.jupiter.api.Assumptions;
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledOnOs;
17  import org.junit.jupiter.api.condition.OS;
18  
19  import oshi.hardware.GpuTicks;
20  
21  @EnabledOnOs(OS.MAC)
22  class IOReportClientTest {
23  
24      @Test
25      void testCreateAndCloseDoesNotThrow() {
26          // create() returns null on Intel Mac or sandboxed CI; both are valid.
27          // When non-null, verify the client is usable (returns a non-null ticks snapshot)
28          // and that close() completes without throwing.
29          IOReportClient client = IOReportClient.create();
30          if (client != null) {
31              try {
32                  assertThat("sampleGpuTicks must not return null on a live client", client.sampleGpuTicks(),
33                          is(notNullValue()));
34              } finally {
35                  client.close();
36              }
37          }
38      }
39  
40      @Test
41      void testSampleGpuTicksNotNull() {
42          IOReportClient client = IOReportClient.create();
43          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
44          try {
45              assertThat("sampleGpuTicks should never return null", client.sampleGpuTicks(), is(notNullValue()));
46          } finally {
47              client.close();
48          }
49      }
50  
51      @Test
52      void testSampleGpuTicksNonNegative() {
53          IOReportClient client = IOReportClient.create();
54          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
55          try {
56              GpuTicks ticks = client.sampleGpuTicks();
57              assertThat("activeTicks should be non-negative", ticks.getActiveTicks(), is(greaterThanOrEqualTo(0L)));
58              assertThat("idleTicks should be non-negative", ticks.getIdleTicks(), is(greaterThanOrEqualTo(0L)));
59          } finally {
60              client.close();
61          }
62      }
63  
64      @Test
65      void testSampleGpuTicksTotalPositive() {
66          IOReportClient client = IOReportClient.create();
67          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
68          try {
69              GpuTicks ticks = client.sampleGpuTicks();
70              Assumptions.assumeTrue(ticks.getActiveTicks() + ticks.getIdleTicks() > 0,
71                      "Skipping: IOReport GPU ticks unavailable (no GPU or sandboxed CI environment)");
72              long total = ticks.getActiveTicks() + ticks.getIdleTicks();
73              assertThat("active + idle should be positive", total, is(greaterThan(0L)));
74          } finally {
75              client.close();
76          }
77      }
78  
79      @Test
80      void testSampleGpuTicksMonotonic() throws InterruptedException {
81          IOReportClient client = IOReportClient.create();
82          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
83          try {
84              GpuTicks first = client.sampleGpuTicks();
85              Thread.sleep(150);
86              GpuTicks second = client.sampleGpuTicks();
87              assertThat("activeTicks should be non-decreasing", second.getActiveTicks(),
88                      is(greaterThanOrEqualTo(first.getActiveTicks())));
89              assertThat("idleTicks should be non-decreasing", second.getIdleTicks(),
90                      is(greaterThanOrEqualTo(first.getIdleTicks())));
91          } finally {
92              client.close();
93          }
94      }
95  
96      @Test
97      void testSamplePowerWattsValidOrSentinel() throws InterruptedException {
98          IOReportClient client = IOReportClient.create();
99          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
100         try {
101             double first = client.samplePowerWatts();
102             assertThat("Power should be -1 (unavailable) or a non-negative wattage", first == -1d || first >= 0d,
103                     is(true));
104             Thread.sleep(100);
105             double second = client.samplePowerWatts();
106             assertThat("Power delta should be -1 (unavailable) or a non-negative wattage",
107                     second == -1d || second >= 0d, is(true));
108         } finally {
109             client.close();
110         }
111     }
112 
113     @Test
114     void testCloseIsIdempotent() {
115         IOReportClient client = IOReportClient.create();
116         Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
117         client.close();
118         assertDoesNotThrow(client::close);
119     }
120 
121     @Test
122     void testSamplingAfterCloseReturnsSentinels() {
123         IOReportClient client = IOReportClient.create();
124         Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
125         client.close();
126         GpuTicks ticks = client.sampleGpuTicks();
127         assertThat("Closed client should return activeTicks=0", ticks.getActiveTicks(), is(0L));
128         assertThat("Closed client should return idleTicks=0", ticks.getIdleTicks(), is(0L));
129         assertThat("Closed client power should return -1", client.samplePowerWatts(), is(-1d));
130         assertThat("Closed client utilization should return -1", client.sampleGpuUtilization(), is(-1d));
131     }
