5   package oshi.driver.mac;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.greaterThanOrEqualTo;
10  import static org.hamcrest.Matchers.is;
11  import static org.hamcrest.Matchers.notNullValue;
12  
13  import org.junit.jupiter.api.Assumptions;
14  import org.junit.jupiter.api.Test;
15  import org.junit.jupiter.api.condition.EnabledOnOs;
16  import org.junit.jupiter.api.condition.OS;
17  
18  import oshi.hardware.GpuTicks;
19  
20  @EnabledOnOs(OS.MAC)
21  class IOReportClientTest {
22  
23      @Test
24      void testCreateAndCloseDoesNotThrow() {
25          // create() returns null on Intel Mac or sandboxed CI; both are valid.
26          // When non-null, verify the client is usable (returns a non-null ticks snapshot)
27          // and that close() completes without throwing.
28          IOReportClient client = IOReportClient.create();
29          if (client != null) {
30              try {
31                  assertThat("sampleGpuTicks must not return null on a live client", client.sampleGpuTicks(),
32                          is(notNullValue()));
33              } finally {
34                  client.close();
35              }
36          }
37      }
38  
39      @Test
40      void testSampleGpuTicksNotNull() {
41          IOReportClient client = IOReportClient.create();
42          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
43          try {
44              assertThat("sampleGpuTicks should never return null", client.sampleGpuTicks(), is(notNullValue()));
45          } finally {
46              client.close();
47          }
48      }
49  
50      @Test
51      void testSampleGpuTicksNonNegative() {
52          IOReportClient client = IOReportClient.create();
53          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
54          try {
55              GpuTicks ticks = client.sampleGpuTicks();
56              assertThat("activeTicks should be non-negative", ticks.getActiveTicks(), is(greaterThanOrEqualTo(0L)));
57              assertThat("idleTicks should be non-negative", ticks.getIdleTicks(), is(greaterThanOrEqualTo(0L)));
58          } finally {
59              client.close();
60          }
61      }
62  
63      @Test
64      void testSampleGpuTicksTotalPositive() {
65          IOReportClient client = IOReportClient.create();
66          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
67          try {
68              GpuTicks ticks = client.sampleGpuTicks();
69              Assumptions.assumeTrue(ticks.getActiveTicks() + ticks.getIdleTicks() > 0,
70                      "Skipping: IOReport GPU ticks unavailable (no GPU or sandboxed CI environment)");
71              assertThat("active + idle should be positive", ticks.getActiveTicks() + ticks.getIdleTicks(),
72                      is(greaterThan(0L)));
73          } finally {
74              client.close();
75          }
76      }
77  
78      @Test
79      void testSampleGpuTicksMonotonic() throws InterruptedException {
80          IOReportClient client = IOReportClient.create();
81          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
82          try {
83              GpuTicks first = client.sampleGpuTicks();
84              Thread.sleep(150);
85              GpuTicks second = client.sampleGpuTicks();
86              assertThat("activeTicks should be non-decreasing", second.getActiveTicks(),
87                      is(greaterThanOrEqualTo(first.getActiveTicks())));
88              assertThat("idleTicks should be non-decreasing", second.getIdleTicks(),
89                      is(greaterThanOrEqualTo(first.getIdleTicks())));
90          } finally {
91              client.close();
92          }
93      }
94  
95      @Test
96      void testSamplePowerWattsValidOrSentinel() throws InterruptedException {
97          IOReportClient client = IOReportClient.create();
98          Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
99          try {
100             double first = client.samplePowerWatts();
101             assertThat("Power should be -1 (unavailable) or a non-negative wattage", first == -1d || first >= 0d,
102                     is(true));
103             Thread.sleep(100);
104             double second = client.samplePowerWatts();
105             assertThat("Power delta should be -1 (unavailable) or a non-negative wattage",
106                     second == -1d || second >= 0d, is(true));
107         } finally {
108             client.close();
109         }
110     }
111 
112     @Test
113     void testCloseIsIdempotent() {
114         IOReportClient client = IOReportClient.create();
115         Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
116         client.close();
117         client.close(); // must not throw
118     }
119 
120     @Test
121     void testSamplingAfterCloseReturnsSentinels() {
122         IOReportClient client = IOReportClient.create();
123         Assumptions.assumeTrue(client != null, "Skipping: IOReport unavailable");
124         client.close();
125         GpuTicks ticks = client.sampleGpuTicks();
126         assertThat("Closed client should return activeTicks=0", ticks.getActiveTicks(), is(0L));
127         assertThat("Closed client should return idleTicks=0", ticks.getIdleTicks(), is(0L));
128         assertThat("Closed client power should return -1", client.samplePowerWatts(), is(-1d));
129         assertThat("Closed client utilization should return -1", client.sampleGpuUtilization(), is(-1d));
130     }
