49  class WMIDriversTest {
50  
51      @Test
52      void testQueryWMIMSAcpi() {
53          if (Platform.isWindows()) {
54              assertThat("Failed MSAcpiThermalZoneTemperature.queryCurrentTemperature",
55                      MSAcpiThermalZoneTemperature.queryCurrentTemperature().getResultCount(),
56                      is(greaterThanOrEqualTo(0)));
57          }
58      }
59  
60      @Test
61      void testQueryWMIMSFT() {
62          if (Platform.isWindows()) {
63              WmiQueryHandler handler = WmiQueryHandler.createInstance();
64              assertThat(handler, is(notNullValue()));
65              boolean comInit = handler.initCOM();
66  
67              assertThat("Failed MSFTStorage.queryPhysicalDisks",
68                      MSFTStorage.queryPhysicalDisks(handler).getResultCount(), is(greaterThanOrEqualTo(0)));
69              assertThat("Failed MSFTStorage.queryStoragePoolPhysicalDisks",
70                      MSFTStorage.queryStoragePoolPhysicalDisks(handler).getResultCount(), is(greaterThanOrEqualTo(0)));
71              assertThat("Failed MSFTStorage.queryStoragePools", MSFTStorage.queryStoragePools(handler).getResultCount(),
72                      is(greaterThanOrEqualTo(0)));
73              assertThat("Failed MSFTStorage.queryVirtualDisks", MSFTStorage.queryVirtualDisks(handler).getResultCount(),
74                      is(greaterThanOrEqualTo(0)));
75  
76              if (comInit) {
77                  handler.unInitCOM();
78              }
79          }
80      }

82      // OhmHardware and OhmSensor excluded
83      // We'll never have OHM running during testing
84  
85      @Test
86      void testQueryWMIWin32() {
87          if (Platform.isWindows()) {
88              WmiQueryHandler handler = WmiQueryHandler.createInstance();
89              assertThat(handler, is(notNullValue()));
90              boolean comInit = handler.initCOM();
91  
92              assertThat("Failed Win32DiskDrive.queryDiskDrive", Win32DiskDrive.queryDiskDrive(handler).getResultCount(),
93                      is(greaterThanOrEqualTo(0)));
94  
95              assertThat("Failed Win32DiskDriveToDiskPartition.queryDriveToPartition",
96                      Win32DiskDriveToDiskPartition.queryDriveToPartition(handler).getResultCount(),
97                      is(greaterThanOrEqualTo(0)));
98  
99              assertThat("Failed Win32DiskPartition.queryPartition",
100                     Win32DiskPartition.queryPartition(handler).getResultCount(), is(greaterThanOrEqualTo(0)));
101 
102             assertThat("Failed Win32LogicalDiskToPartition.queryDiskToPartition",
103                     Win32LogicalDiskToPartition.queryDiskToPartition(handler).getResultCount(),
104                     is(greaterThanOrEqualTo(0)));
105 
106             if (comInit) {
107                 handler.unInitCOM();
108             }
109 
110             assertThat("Failed Win32BaseBoard.queryBaseboardInfo", Win32BaseBoard.queryBaseboardInfo().getResultCount(),
111                     is(greaterThan(0)));
112 
113             assertThat("Failed Win32Bios.queryBiosInfo", Win32Bios.queryBiosInfo().getResultCount(),
114                     is(greaterThan(0)));
115             assertThat("Failed Win32Bios.querySerialNumber", Win32Bios.querySerialNumber().getResultCount(),
116                     is(greaterThan(0)));
117 
118             assertThat("Failed Win32ComputerSystemProduct.queryIdentifyingNumberUUID",
119                     Win32ComputerSystemProduct.queryIdentifyingNumberUUID().getResultCount(), is(greaterThan(0)));
120 
121             assertThat("Failed Win32Fan.querySpeed", Win32Fan.querySpeed().getResultCount(),
122                     is(greaterThanOrEqualTo(0)));
123 
124             WmiResult<LogicalDiskProperty> ld = Win32LogicalDisk.queryLogicalDisk(null, false);
125             assertThat("Failed Win32LogicalDisk.queryLogicalDisk", ld.getResultCount(), is(greaterThan(0)));
126             assertThat("Failed Win32LogicalDisk.queryLogicalDisk", Win32LogicalDisk
127                     .queryLogicalDisk(WmiUtil.getString(ld, LogicalDiskProperty.NAME, 0), false).getResultCount(),
128                     is(greaterThan(0)));
129             assertThat("Failed Win32LogicalDisk.queryLogicalDisk",
130                     Win32LogicalDisk.queryLogicalDisk(null, true).getResultCount(), is(greaterThan(0)));
131 
132             assertThat("Failed Win32OperatingSystem.queryOsVersion",
133                     Win32OperatingSystem.queryOsVersion().getResultCount(), is(greaterThan(0)));
134 
135             if (VersionHelpers.IsWindows10OrGreater()) {
136                 assertThat("Failed Win32PhysicalMemory.queryphysicalMemory",
137                         Win32PhysicalMemory.queryphysicalMemory().getResultCount(), is(greaterThan(0)));
138             } else {
139                 assertThat("Failed Win32PhysicalMemory.queryphysicalMemoryWin8",
140                         Win32PhysicalMemory.queryphysicalMemoryWin8().getResultCount(), is(greaterThan(0)));
141             }
142 
143             WmiResult<CommandLineProperty> cl = Win32Process.queryCommandLines(null);
144             assertThat("Failed Win32Process.queryCommandLines", cl.getResultCount(), is(greaterThan(0)));
145             Set<Integer> clset = IntStream.range(0, cl.getResultCount())
146                     .map(i -> WmiUtil.getUint32(cl, CommandLineProperty.PROCESSID, i)).boxed()
147                     .collect(Collectors.toSet());
148             assertThat("Failed Win32Process.queryProcesses", Win32Process.queryProcesses(clset).getResultCount(),
149                     is(both(greaterThan(0)).and(lessThanOrEqualTo(clset.size()))));
150 
151             Win32ProcessCached cache = Win32ProcessCached.getInstance();
152             assertThat("Failed Win32ProcessCached.getCommandLine",
153                     cache.getCommandLine(WmiUtil.getUint32(cl, CommandLineProperty.PROCESSID, 0), 0L),
154                     is(notNullValue()));
155 
156             assertThat("Failed Win32Processor.queryBitness", Win32Processor.queryBitness().getResultCount(),
157                     is(greaterThan(0)));
158             assertThat("Failed Win32Processor.queryProcessorId", Win32Processor.queryProcessorId().getResultCount(),
159                     is(greaterThan(0)));
160             assertThat("Failed Win32Processor.queryVoltage", Win32Processor.queryVoltage().getResultCount(),
161                     is(greaterThan(0)));
162 
163             assertThat("Failed Win32VideoController.queryVideoController",
164                     Win32VideoController.queryVideoController().getResultCount(), is(greaterThanOrEqualTo(0)));
165         }
