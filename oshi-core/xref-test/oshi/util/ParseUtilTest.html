195         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 2, 4), Float.MIN_VALUE);

376         assertThat("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(456L));
377         assertThat("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", result[1],
378                 is(now));
379 
380         foo = String.format(Locale.ROOT, "The numbers are %d %d %d %d %s", 123, 456, 789, now,
381                 "709af748-5f8e-41b3-b73a-b440ef4406c8");
382         count = ParseUtil.countStringToLongArray(foo, ' ');
383         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
