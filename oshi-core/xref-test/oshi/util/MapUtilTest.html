19  package oshi.util;
20  
21  import static org.junit.Assert.assertEquals;
22  
23  import java.util.HashMap;
24  import java.util.List;
25  import java.util.Map;
26  
27  import org.junit.Test;
28  
29  /**
30   * Test map methods
31   */
32  public class MapUtilTest {
33  
34      @Test
35      public void testGetPutAndCompute() {
36          Map<String, String> testMap = new HashMap<>();
37  
38          testMap.put("a", "A");
39          assertEquals("A", MapUtil.getOrDefault(testMap, "a", "Z"));
40          assertEquals("Z", MapUtil.getOrDefault(testMap, "b", "Z"));
41  
42          assertEquals("A", MapUtil.putIfAbsent(testMap, "a", "Z"));
43          assertEquals("A", testMap.get("a"));
44          assertEquals(null, MapUtil.putIfAbsent(testMap, "b", "B"));
45          assertEquals("B", testMap.get("b"));
46  
47          Map<String, List<String>> testListMap = new HashMap<>();
48          assertEquals(0, MapUtil.createNewListIfAbsent(testListMap, "a").size());
49          testListMap.get("a").add("foo");
50          assertEquals(1, MapUtil.createNewListIfAbsent(testListMap, "a").size());
51          assertEquals("foo", MapUtil.createNewListIfAbsent(testListMap, "a").get(0));
52      }
53  }
