22  /**
23   * Tests for {@link ProcstatUtil}. Pure parsing methods are covered with sample {@code fstat} output; the integration
24   * test is gated to DragonFly BSD.
25   */
26  class ProcstatUtilTest {
27  
28      // DragonFly's fstat output positions the FD/type classifier ("wd", "root", "text", or a numeric fd) at the 5th
29      // whitespace-split token (index 4). The parser excludes "wd"/"root"/"text" classifiers when counting open files
30      // and uses the last token of the "wd" row as the working directory path. Sample rows here mimic that layout.
31  
32      @Test
33      void testParseCwdFromFstatFindsWdRow() {
34          List<String> fstat = Arrays.asList("USER     CMD       PID    NLWP   FD MOUNT INUM MODE NAME",
35                  "dan      bash      1234   1      wd /     45   drwx /home/dan",
36                  "dan      bash      1234   1      0  /dev  12   crw  /dev/tty");
37          assertThat(ProcstatUtil.parseCwdFromFstat(fstat), is("/home/dan"));
38      }
39  
40      @Test
41      void testParseCwdFromFstatNoWdRowReturnsEmpty() {
42          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
43                  "dan  bash 1234 1 0  /dev 12 crw /dev/tty");
44          assertThat(ProcstatUtil.parseCwdFromFstat(fstat), is(emptyString()));
45      }
46  
47      @Test
48      void testParseCwdFromFstatEmpty() {
49          assertThat(ProcstatUtil.parseCwdFromFstat(Collections.<String>emptyList()), is(emptyString()));
50      }
51  
52      @Test
53      void testParseOpenFilesExcludesWdRootTextAndHeader() {
54          // 5 rows total. The wd, root, and text rows are excluded; the header (token "FD" at index 4) is not excluded
55          // by the filter but is removed via the -1 adjustment at the end. With 2 countable rows minus the header
56          // adjustment, the result is 1.
57          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
58                  "dan  bash 1234 1 wd   /    45 drwx /home/dan", "dan  bash 1234 1 root /    1  drwx /",
59                  "dan  bash 1234 1 text /    99 -rxr /bin/bash", "dan  bash 1234 1 0    /dev 12 crw  /dev/tty",
60                  "dan  bash 1234 1 1    /dev 12 crw  /dev/tty");
61          assertThat(ProcstatUtil.parseOpenFiles(fstat), is(2L));
62      }
63  
64      @Test
65      void testParseOpenFilesNoCountableRowsReturnsZero() {
66          // Only header + excluded rows → no fd counted → returns 0 (per implementation: fd > 0 ? fd - 1 : 0)
67          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
68                  "dan  bash 1234 1 wd /home 45 drwx /home/dan");
69          // Header row counts (split[4]="FD" not in {wd,root,text}), then the wd row is excluded.
70          // Count=1 (header) → returns 1-1=0.
71          assertThat(ProcstatUtil.parseOpenFiles(fstat), is(0L));
72      }
73  
74      @Test
75      void testParseOpenFilesEmpty() {
76          assertThat(ProcstatUtil.parseOpenFiles(Collections.<String>emptyList()), is(0L));
77      }
78  
79      @Test
80      void testGetCwdMapNegativePidReturnsEmptyMap() {
81          // No live command is run for a negative pid; the map is empty.
82          assertThat(ProcstatUtil.getCwdMap(-1).size(), is(0));
83      }
84  
85      @Test
86      @EnabledIfSystemProperty(named = "os.name", matches = "(?i)dragonfly")
87      void testProcstatLive() {
88          int pid = new SystemInfo().getOperatingSystem().getProcessId();
89          assertThat("Open files must be nonnegative", ProcstatUtil.getOpenFiles(pid), is(greaterThanOrEqualTo(0L)));
90          assertThat("Cwd should be nonempty", ProcstatUtil.getCwd(pid), is(not(emptyString())));
91      }
92  }
