60       * Test read file.
61       */
62      @Test
63      void testReadFile() {
64          // Write to a temp file
65          Path multilineFile = null;
66          try {
67              multilineFile = Files.createTempFile("oshitest.multiline", null);
68              String s = "Line 1\nLine 2\nThe third line\nLine 4\nLine 5\n";
69              Files.write(multilineFile, s.getBytes(StandardCharsets.UTF_8));
70          } catch (IOException e) {
71              fail("IO Exception creating or writing to temporary multiline file.");
72          }
73  
74          // Try the new temp file
75          List<String> tempFileStrings = FileUtil.readFile(multilineFile.toString());
76          assertThat("Temp file line one mismatch", tempFileStrings.get(0), is("Line 1"));
77          List<String> matchingLines = tempFileStrings.stream().filter(s -> s.startsWith("Line "))
78                  .collect(Collectors.toList());
79          assertThat("Matching lines mismatch", matchingLines.size(), is(4));
80  
81          // Delete the temp file
82          try {
83              Files.deleteIfExists(multilineFile);
84          } catch (IOException e) {
85              fail("IO Exception deleting temporary multiline file.");
86          }
87  
88          // Try file not found on deleted file
89          assertThat("Deleted file should return empty", FileUtil.readFile(multilineFile.toString()), is(empty()));
90      }

95      @Test
96      void testGetFromFile() {
97          // Write to temp file
98          Path integerFile = null;
99          try {
100             integerFile = Files.createTempFile("oshitest.int", null);
101             Files.write(integerFile, "123\n".getBytes(StandardCharsets.UTF_8));
102         } catch (IOException e) {
103             fail("IO Exception creating or writing to temporary integer file.");
104         }
105         assertThat("unsigned long from int", FileUtil.getUnsignedLongFromFile(integerFile.toString()), is(123L));
106         assertThat("long from int", FileUtil.getLongFromFile(integerFile.toString()), is(123L));
107         assertThat("int from int", FileUtil.getIntFromFile(integerFile.toString()), is(123));
108         assertThat("string from int", FileUtil.getStringFromFile(integerFile.toString()), is("123"));
109 
110         // Delete the temp file
111         try {
112             Files.deleteIfExists(integerFile);
113         } catch (IOException e) {
114             fail("IO Exception deleting temporary integer file.");
115         }
116 
117         // Write to temp file
118         Path stringFile = null;
119         try {
120             stringFile = Files.createTempFile("oshitest.str", null);
121             Files.write(stringFile, "foo bar\n".getBytes(StandardCharsets.UTF_8));
122         } catch (IOException e) {
123             fail("IO Exception creating or writing to temporary string file.");
124         }
125 
126         assertThat("unsigned long from string", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
127         assertThat("long from string", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
128         assertThat("int from string", FileUtil.getIntFromFile(stringFile.toString()), is(0));
129         assertThat("string from string", FileUtil.getStringFromFile(stringFile.toString()), is("foo bar"));
130         // Delete the temp file
131         try {
132             Files.deleteIfExists(stringFile);
133         } catch (IOException e) {
134             fail("IO Exception deleting temporary string file.");
135         }
136 
137         // Try file not found on deleted file
138         assertThat("unsigned long from invalid", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
139         assertThat("long from invalid", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
140         assertThat("int from invalid", FileUtil.getIntFromFile(stringFile.toString()), is(0));
141         assertThat("string from invalid ", FileUtil.getStringFromFile(stringFile.toString()), is(emptyString()));
142     }

144     @Test
145     void testReadProcIo() {
146         Map<String, String> expected = new LinkedHashMap<>();
147         expected.put("rchar", "124788352");
148         expected.put("wchar", "124802481");
149         expected.put("syscr", "135");
150         expected.put("syscw", "1547");
151         expected.put("read_bytes", "40304640");
152         expected.put("write_bytes", "124780544");
153         expected.put("cancelled_write_bytes", "42");
154         // Write this to a temp file
155         Path procIoFile = null;
156         try {
157             procIoFile = Files.createTempFile("oshitest.procio", null);
158             for (Entry<String, String> e : expected.entrySet()) {
159                 String s = e.getKey() + ": " + e.getValue() + "\n";
160                 Files.write(procIoFile, s.getBytes(StandardCharsets.UTF_8), StandardOpenOption.APPEND);
161             }
162         } catch (IOException e) {
163             fail("IO Exception creating or writing to temporary procIo file.");
164         }
165         // Read into map
166         Map<String, String> actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
167         assertThat("procio size", actual, is(aMapWithSize(expected.size())));
168         for (Entry<String, String> entry : expected.entrySet()) {
169             assertThat("procio entry", actual, hasEntry(entry.getKey(), entry.getValue()));
170         }
171 
172         // Cleanup
173         try {
174             Files.deleteIfExists(procIoFile);
175         } catch (IOException e) {
176             fail("IO Exception deleting temporary procIo file.");
177         }
178 
179         // Test deleted file
180         actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
181         assertThat("procio size", actual, anEmptyMap());
182     }

192     @Test
193     void testReadBinaryFile() {
194         ByteBuffer buff = ByteBuffer.allocate(18 + Native.LONG_SIZE + Native.SIZE_T_SIZE);
195         buff.order(ByteOrder.nativeOrder());
196         buff.putLong(123L);
197         buff.putInt(45);
198         buff.putShort((short) 67);
199         buff.put((byte) 89);
200         if (Native.LONG_SIZE > 4) {
201             buff.putLong(10L);
202         } else {
203             buff.putInt(10);
204         }
205         if (Native.SIZE_T_SIZE > 4) {
206             buff.putLong(11L);
207         } else {
208             buff.putInt(11);
209         }
210         byte[] arr = new byte[] { 1, 2, 3 };
211         buff.put(arr);
212 
213         // Write to temp file
214         Path binaryFile = null;
215         try {
216             binaryFile = Files.createTempFile("oshitest.binary", null);
217             Files.write(binaryFile, buff.array());
218         } catch (IOException e) {
219             fail("IO Exception creating or writing to temporary binary file.");
220         }
221 
222         // Read from file
223         buff = FileUtil.readAllBytesAsBuffer(binaryFile.toString());
224         assertThat("Buffer size should match bytes written", buff.limit(),
225                 is(18 + Native.LONG_SIZE + Native.SIZE_T_SIZE));
226         assertThat("Long from buffer should match", FileUtil.readLongFromBuffer(buff), is(123L));
227         assertThat("Int from buffer should match", FileUtil.readIntFromBuffer(buff), is(45));
228         assertThat("Short from buffer should match", FileUtil.readShortFromBuffer(buff), is((short) 67));
229         assertThat("Byte from buffer should match", FileUtil.readByteFromBuffer(buff), is((byte) 89));
230         assertThat("NativeLong from buffer should match", FileUtil.readNativeLongFromBuffer(buff).longValue(), is(10L));
231         assertThat("SizeT from buffer should match", FileUtil.readSizeTFromBuffer(buff).longValue(), is(11L));
232         byte[] array = new byte[3];
233         FileUtil.readByteArrayFromBuffer(buff, array);
234         assertArrayEquals(arr, array, "Byte array from buffer should match");
235         // Test reads past end of file
236         assertThat("Long from buffer at limit should be 0", FileUtil.readLongFromBuffer(buff), is(0L));
237         assertThat("Int from buffer at limit should be 0", FileUtil.readIntFromBuffer(buff), is(0));
238         assertThat("Short from buffer at limit should be 0", FileUtil.readShortFromBuffer(buff), is((short) 0));
239         assertThat("Byte from buffer at limit should be 0", FileUtil.readByteFromBuffer(buff), is((byte) 0));
240         assertThat("NativeLong from buffer at limit should be 0", FileUtil.readNativeLongFromBuffer(buff).longValue(),
241                 is(0L));
242         assertThat("SizeT from buffer at limit should be 0", FileUtil.readSizeTFromBuffer(buff).longValue(), is(0L));
243         byte[] arr0 = new byte[] { 0, 0, 0 };
244         array = new byte[3];
245         FileUtil.readByteArrayFromBuffer(buff, array);
246         assertArrayEquals(arr0, array, "Byte array from buffer at limit should be all 0s");
247 
248         // Cleanup
249         try {
250             Files.deleteIfExists(binaryFile);
251         } catch (IOException e) {
252             fail("IO Exception deleting temporary procIo file.");
253         }
254     }
