21  import java.util.Arrays;
22  import java.util.List;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import oshi.hardware.CentralProcessor.TickType;
28  import oshi.hardware.Display;
29  import oshi.hardware.HWDiskStore;
30  import oshi.hardware.HardwareAbstractionLayer;
31  import oshi.hardware.NetworkIF;
32  import oshi.hardware.PowerSource;
33  import oshi.hardware.UsbDevice;
34  import oshi.software.os.FileSystem;
35  import oshi.software.os.OSFileStore;
36  import oshi.software.os.OSProcess;
37  import oshi.software.os.OperatingSystem;
38  import oshi.software.os.OperatingSystem.ProcessSort;
39  import oshi.util.FormatUtil;
40  import oshi.util.Util;
41  
42  /**
43   * The Class SystemInfoTest.
44   *
45   * @author dblock[at]dblock[dot]org
46   */
47  public class SystemInfoTest {
48  
49      /**
50       * The main method.
51       *
52       * @param args
53       *            the arguments
54       */
55      public static void main(String[] args) {
56          // Options: ERROR > WARN > INFO > DEBUG > TRACE
57          System.setProperty(org.slf4j.impl.SimpleLogger.DEFAULT_LOG_LEVEL_KEY, "INFO");
58          Logger LOG = LoggerFactory.getLogger(SystemInfoTest.class);
59  
60          LOG.info("Initializing System...");
61          SystemInfo si = new SystemInfo();
62  
63          // software
64          // software: operating system
65          OperatingSystem os = si.getOperatingSystem();
66          System.out.println(os);
67  
68          // hardware
69          HardwareAbstractionLayer hal = si.getHardware();
70  
71          // hardware: processors
72          System.out.println(hal.getProcessor());
73          System.out.println(" " + hal.getProcessor().getPhysicalProcessorCount() + " physical CPU(s)");
74          System.out.println(" " + hal.getProcessor().getLogicalProcessorCount() + " logical CPU(s)");
75  
76          System.out.println("Identifier: " + hal.getProcessor().getIdentifier());
77          System.out.println("Serial Num: " + hal.getProcessor().getSystemSerialNumber());
78  
79          // hardware: memory
80          LOG.info("Checking Memory...");
81          System.out.println("Memory: " + FormatUtil.formatBytes(hal.getMemory().getAvailable()) + "/"
82                  + FormatUtil.formatBytes(hal.getMemory().getTotal()));
83          System.out.println("Swap used: " + FormatUtil.formatBytes(hal.getMemory().getSwapUsed()) + "/"
84                  + FormatUtil.formatBytes(hal.getMemory().getSwapTotal()));
85          // uptime
86          LOG.info("Checking Uptime...");
87          System.out.println("Uptime: " + FormatUtil.formatElapsedSecs(hal.getProcessor().getSystemUptime()));
88  
89          // CPU
90          LOG.info("Checking CPU...");
91          long[] prevTicks = hal.getProcessor().getSystemCpuLoadTicks();
92          System.out.println("CPU, IOWait, and IRQ ticks @ 0 sec:" + Arrays.toString(prevTicks));
93          // Wait a second...
94          Util.sleep(1000);
95          long[] ticks = hal.getProcessor().getSystemCpuLoadTicks();
96          System.out.println("CPU, IOWait, and IRQ ticks @ 1 sec:" + Arrays.toString(ticks));
97          long user = ticks[TickType.USER.getIndex()] - prevTicks[TickType.USER.getIndex()];
98          long nice = ticks[TickType.NICE.getIndex()] - prevTicks[TickType.NICE.getIndex()];
99          long sys = ticks[TickType.SYSTEM.getIndex()] - prevTicks[TickType.SYSTEM.getIndex()];
100         long idle = ticks[TickType.IDLE.getIndex()] - prevTicks[TickType.IDLE.getIndex()];
101         long iowait = ticks[TickType.IOWAIT.getIndex()] - prevTicks[TickType.IOWAIT.getIndex()];
102         long irq = ticks[TickType.IRQ.getIndex()] - prevTicks[TickType.IRQ.getIndex()];
103         long softirq = ticks[TickType.SOFTIRQ.getIndex()] - prevTicks[TickType.SOFTIRQ.getIndex()];
104         long totalCpu = user + nice + sys + idle + iowait + irq + softirq;
105 
106         System.out.format(
107                 "User: %.1f%% Nice: %.1f%% System: %.1f%% Idle: %.1f%% IOwait: %.1f%% IRQ: %.1f%% SoftIRQ: %.1f%%%n",
108                 100d * user / totalCpu, 100d * nice / totalCpu, 100d * sys / totalCpu, 100d * idle / totalCpu,
109                 100d * iowait / totalCpu, 100d * irq / totalCpu, 100d * softirq / totalCpu);
110         System.out.format("CPU load: %.1f%% (counting ticks)%n",
111                 hal.getProcessor().getSystemCpuLoadBetweenTicks() * 100);
112         System.out.format("CPU load: %.1f%% (OS MXBean)%n", hal.getProcessor().getSystemCpuLoad() * 100);
113         double[] loadAverage = hal.getProcessor().getSystemLoadAverage(3);
114         System.out.println("CPU load averages:" + (loadAverage[0] < 0 ? " N/A" : String.format(" %.2f", loadAverage[0]))
115                 + (loadAverage[1] < 0 ? " N/A" : String.format(" %.2f", loadAverage[1]))
116                 + (loadAverage[2] < 0 ? " N/A" : String.format(" %.2f", loadAverage[2])));
117         // per core CPU
118         StringBuilder procCpu = new StringBuilder("CPU load per processor:");
119         double[] load = hal.getProcessor().getProcessorCpuLoadBetweenTicks();
120         for (int cpu = 0; cpu < load.length; cpu++) {
121             procCpu.append(String.format(" %.1f%%", load[cpu] * 100));
122         }
123         System.out.println(procCpu.toString());
124 
125         // Processes
126         System.out.println("Processes: " + os.getProcessCount() + ", Threads: " + os.getThreadCount());
127         // Sort by highest CPU
128         List<OSProcess> procs = Arrays.asList(os.getProcesses(5, ProcessSort.CPU));
129 
130         System.out.println("   PID  %CPU %MEM       VSZ       RSS Name");
131         for (int i = 0; i < procs.size() && i < 5; i++) {
132             OSProcess p = procs.get(i);
133             System.out.format(" %5d %5.1f %4.1f %9s %9s %s%n", p.getProcessID(),
134                     100d * (p.getKernelTime() + p.getUserTime()) / p.getUpTime(),
135                     100d * p.getResidentSetSize() / hal.getMemory().getTotal(),
136                     FormatUtil.formatBytes(p.getVirtualSize()), FormatUtil.formatBytes(p.getResidentSetSize()),
137                     p.getName());
138         }
139 
140         // hardware: sensors
141         LOG.info("Checking Sensors...");
142         System.out.println("Sensors:");
143         System.out.format(" CPU Temperature: %.1fÂ°C%n", hal.getSensors().getCpuTemperature());
144         System.out.println(" Fan Speeds:" + Arrays.toString(hal.getSensors().getFanSpeeds()));
145         System.out.format(" CPU Voltage: %.1fV%n", hal.getSensors().getCpuVoltage());
146 
147         // hardware: power
148         LOG.info("Checking Power sources...");
149         StringBuilder sb = new StringBuilder("Power: ");
150         if (hal.getPowerSources().length == 0) {
151             sb.append("Unknown");
152         } else {
153             double timeRemaining = hal.getPowerSources()[0].getTimeRemaining();
154             if (timeRemaining < -1d) {
155                 sb.append("Charging");
156             } else if (timeRemaining < 0d) {
157                 sb.append("Calculating time remaining");
158             } else {
159                 sb.append(String.format("%d:%02d remaining", (int) (timeRemaining / 3600),
160                         (int) (timeRemaining / 60) % 60));
161             }
162         }
163         for (PowerSource pSource : hal.getPowerSources()) {
164             sb.append(String.format("%n %s @ %.1f%%", pSource.getName(), pSource.getRemainingCapacity() * 100d));
165         }
166         System.out.println(sb.toString());
167 
168         // hardware: disks
169         LOG.info("Checking Disks...");
170         System.out.println("Disks:");
171 
172         HWDiskStore[] dskArray = hal.getDiskStores();
173         for (HWDiskStore dsk : dskArray) {
174             boolean readwrite = dsk.getReads() > 0 || dsk.getWrites() > 0;
175             System.out.format(" %s: (model: %s - S/N: %s) size: %s, reads: %s, writes: %s %n", dsk.getName(),
176                     dsk.getModel(), dsk.getSerial(),
177                     dsk.getSize() > 0 ? FormatUtil.formatBytesDecimal(dsk.getSize()) : "?",
178                     readwrite ? FormatUtil.formatBytes(dsk.getReads()) : "?",
179                     readwrite ? FormatUtil.formatBytes(dsk.getWrites()) : "?");
180         }
181 
182         // software: file system
183         LOG.info("Checking File System...");
184         System.out.println("File System:");
185 
186         FileSystem filesystem = os.getFileSystem();
187         System.out.format(" File Descriptors: %d/%d%n", filesystem.getOpenFileDescriptors(),
188                 filesystem.getMaxFileDescriptors());
189 
190         OSFileStore[] fsArray = filesystem.getFileStores();
191         for (OSFileStore fs : fsArray) {
192             long usable = fs.getUsableSpace();
193             long total = fs.getTotalSpace();
194             System.out.format(" %s (%s) [%s] %s of %s free (%.1f%%) is %s and is mounted at %s%n", fs.getName(),
195                     fs.getDescription().isEmpty() ? "file system" : fs.getDescription(), fs.getType(),
196                     FormatUtil.formatBytes(usable), FormatUtil.formatBytes(fs.getTotalSpace()), 100d * usable / total,
197                     fs.getVolume(), fs.getMount());
198         }
199 
200         // hardware: network interfaces
201         LOG.info("Checking Network interfaces...");
202         System.out.println("Network interfaces:");
203 
204         NetworkIF[] netArray = hal.getNetworkIFs();
205         for (NetworkIF net : netArray) {
206             System.out.format(" Name: %s (%s)%n", net.getName(), net.getDisplayName());
207             System.out.format("   MAC Address: %s %n", net.getMacaddr());
208             System.out.format("   MTU: %s, Speed: %s %n", net.getMTU(), FormatUtil.formatValue(net.getSpeed(), "bps"));
209             System.out.format("   IPv4: %s %n", Arrays.toString(net.getIPv4addr()));
210             System.out.format("   IPv6: %s %n", Arrays.toString(net.getIPv6addr()));
211             boolean hasData = net.getBytesRecv() > 0 || net.getBytesSent() > 0 || net.getPacketsRecv() > 0
212                     || net.getPacketsSent() > 0;
213             System.out.format("   Traffic: received %s/%s; transmitted %s/%s %n",
214                     hasData ? net.getPacketsRecv() + " packets" : "?",
215                     hasData ? FormatUtil.formatBytes(net.getBytesRecv()) : "?",
216                     hasData ? net.getPacketsSent() + " packets" : "?",
217                     hasData ? FormatUtil.formatBytes(net.getBytesSent()) : "?");
218         }
219 
220         // hardware: displays
221         LOG.info("Checking Displays...");
222         System.out.println("Displays:");
223         int i = 0;
224         for (Display display : hal.getDisplays()) {
225             System.out.println(" Display " + i + ":");
226             System.out.println(display.toString());
227             i++;
228         }
229 
230         // hardware: USB devices
231         LOG.info("Checking USB Devices...");
232         System.out.println("USB Devices:");
233         for (UsbDevice usbDevice : hal.getUsbDevices(true)) {
234             System.out.println(usbDevice.toString());
235         }
236     }
237 }
