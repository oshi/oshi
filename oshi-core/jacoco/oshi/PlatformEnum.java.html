package oshi;

/**
 * Enum of supported operating systems.
 *
 * @author alessandro[at]perucchi[dot]org
 */
public enum PlatformEnum {
    /**
     * Microsoft Windows
     */
    WINDOWS,
    /**
     * A flavor of Linux
     */
    LINUX,
    /**
     * macOS (OS X)
     */
    MACOSX,
    /**
     * Solaris (SunOS)
     */
    SOLARIS,
    /**
     * FreeBSD
     */
    FREEBSD,
    /**
     * OpenBSD, WindowsCE, or an unspecified system
     */
    UNKNOWN;
}
