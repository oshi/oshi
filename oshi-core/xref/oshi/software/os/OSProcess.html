24  package oshi.software.os;
25  
26  import java.util.List;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  import oshi.driver.windows.wmi.Win32ProcessCached;
30  
31  /**
32   * Represents a Process on the operating system, which may contain multiple
33   * threads.
34   */
35  @ThreadSafe
36  public interface OSProcess {
37  
38      /**
39       * <p>
40       * Getter for the field <code>name</code>.
41       * </p>
42       *
43       * @return Returns the name of the process.
44       */
45      String getName();
46  
47      /**
48       * <p>
49       * Getter for the field <code>path</code>.
50       * </p>
51       *
52       * @return Returns the full path of the executing process.
53       */
54      String getPath();
55  
56      /**
57       * <p>
58       * Getter for the field <code>commandLine</code>.
59       * </p>
60       *
61       * @return Returns the process command line. The format of this string is
62       *         platform-dependent and may require the end user to parse the result.
63       *         <p>
64       *         On Linux and macOS systems, the string is null-character-delimited,
65       *         to permit the end user to parse the executable and arguments if
66       *         desired. Further, the macOS variant may include environment variables
67       *         which the end user may wish to exclude from display.
68       *         <p>
69       *         On Solaris, the string is truncated to 80 characters.
70       *         <p>
71       *         On Windows, by default, performs a single WMI query for this process,
72       *         with some latency. If this method will be frequently called for
73       *         multiple processes, see the configuration file to enable a batch
74       *         query mode leveraging {@link Win32ProcessCached#getCommandLine} to
75       *         improve performance.
76       */
77      String getCommandLine();
78  
79      /**
80       * <p>
81       * Getter for the field <code>currentWorkingDirectory</code>.
82       * </p>
83       *
84       * @return Returns the process current working directory.
85       *
86       *         On Windows, this value is only populated for the current process.
87       */
88      String getCurrentWorkingDirectory();
89  
90      /**
91       * <p>
92       * Getter for the field <code>user</code>.
93       * </p>
94       *
95       * @return Returns the user name. On Windows systems, also returns the domain
96       *         prepended to the username.
97       */
98      String getUser();
99  
100     /**
101      * <p>
102      * Getter for the field <code>userID</code>.
103      * </p>
104      *
105      * @return Returns the userID. On Windows systems, returns the Security ID (SID)
106      */
107     String getUserID();
108 
109     /**
110      * <p>
111      * Getter for the field <code>group</code>.
112      * </p>
113      *
114      * @return Returns the group.
115      *
116      *         On Windows systems, populating this value for processes other than
117      *         the current user requires administrative privileges (and still may
118      *         fail for some system processes) and can incur significant latency.
119      *         When successful, returns a the default primary group with access to
120      *         this process, corresponding to the SID in {@link #getGroupID()}.
121      */
122     String getGroup();
123 
124     /**
125      * <p>
126      * Getter for the field <code>groupID</code>.
127      * </p>
128      *
129      * @return Returns the groupID.
130      *
131      *         On Windows systems, populating this value for processes other than
132      *         the current user requires administrative privileges (and still may
133      *         fail for some system processes) and can incur significant latency.
134      *         When successful, returns the default primary group SID with access to
135      *         this process, corresponding to the name in {@link #getGroup()}.
136      */
137     String getGroupID();
138 
139     /**
140      * <p>
141      * Getter for the field <code>state</code>.
142      * </p>
143      *
144      * @return Returns the execution state of the process.
145      */
146     State getState();
147 
148     /**
149      * <p>
150      * Getter for the field <code>processID</code>.
151      * </p>
152      *
153      * @return Returns the processID.
154      */
155     int getProcessID();
156 
157     /**
158      * <p>
159      * Getter for the field <code>parentProcessID</code>.
160      * </p>
161      *
162      * @return Returns the parentProcessID, if any; 0 otherwise.
163      */
164     int getParentProcessID();
165 
166     /**
167      * <p>
168      * Getter for the field <code>threadCount</code>.
169      * </p>
170      *
171      * @return Returns the number of threads in this process.
172      */
173     int getThreadCount();
174 
175     /**
176      * <p>
177      * Getter for the field <code>priority</code>.
178      * </p>
179      *
180      * @return Returns the priority of this process.
181      *
182      *         For Linux and Unix, priority is a value in the range -20 to 19 (20 on
183      *         some systems). The default priority is 0; lower priorities cause more
184      *         favorable scheduling.
185      *
186      *         For Windows, priority values can range from 0 (lowest priority) to 31
187      *         (highest priority).
188      *
189      *         macOS has 128 priority levels, ranging from 0 (lowest priority) to
190      *         127 (highest priority). They are divided into several major bands: 0
191      *         through 51 are the normal levels; the default priority is 31. 52
192      *         through 79 are the highest priority regular threads; 80 through 95
193      *         are for kernel mode threads; and 96 through 127 correspond to
194      *         real-time threads, which are treated differently than other threads
195      *         by the scheduler.
196      */
197     int getPriority();
198 
199     /**
200      * <p>
201      * Getter for the field <code>virtualSize</code>.
202      * </p>
203      *
204      * @return Returns the Virtual Memory Size (VSZ). It includes all memory that
205      *         the process can access, including memory that is swapped out and
206      *         memory that is from shared libraries.
207      */
208     long getVirtualSize();
209 
210     /**
211      * <p>
212      * Getter for the field <code>residentSetSize</code>.
213      * </p>
214      *
215      * @return Returns the Resident Set Size (RSS). On Windows, returns the Private
216      *         Working Set size. It is used to show how much memory is allocated to
217      *         that process and is in RAM. It does not include memory that is
218      *         swapped out. It does include memory from shared libraries as long as
219      *         the pages from those libraries are actually in memory. It does
220      *         include all stack and heap memory.
221      */
222     long getResidentSetSize();
223 
224     /**
225      * <p>
226      * Getter for the field <code>kernelTime</code>.
227      * </p>
228      *
229      * @return Returns the number of milliseconds the process has executed in
230      *         kernel/system mode.
231      */
232     long getKernelTime();
233 
234     /**
235      * <p>
236      * Getter for the field <code>userTime</code>.
237      * </p>
238      *
239      * @return Returns the number of milliseconds the process has executed in user
240      *         mode.
241      */
242     long getUserTime();
243 
244     /**
245      * <p>
246      * Getter for the field <code>upTime</code>.
247      * </p>
248      *
249      * @return Returns the number of milliseconds since the process started.
250      */
251     long getUpTime();
252 
253     /**
254      * <p>
255      * Getter for the field <code>startTime</code>.
256      * </p>
257      *
258      * @return Returns the start time of the process in number of milliseconds since
259      *         January 1, 1970.
260      */
261     long getStartTime();
262 
263     /**
264      * <p>
265      * Getter for the field <code>bytesRead</code>.
266      * </p>
267      *
268      * @return Returns the number of bytes the process has read from disk.
269      */
270     long getBytesRead();
271 
272     /**
273      * <p>
274      * Getter for the field <code>bytesWritten</code>.
275      * </p>
276      *
277      * @return Returns the number of bytes the process has written to disk.
278      */
279     long getBytesWritten();
280 
281     /**
282      * Sets the number of open file handles (or network connections) that belongs to
283      * the process
284      *
285      * On FreeBSD and Solaris, this value is only populated if information for a
286      * single process id is requested.
287      *
288      * @return open files or -1 if unknown or not supported
289      */
290     long getOpenFiles();
291 
292     /**
293      * Gets cumulative CPU usage of this process.
294      * <p>
295      * This calculation sums CPU ticks across all processors and may exceed 100% for
296      * multi-threaded processes. This is consistent with the cumulative CPU
297      * presented by the "top" command on Linux/Unix machines.
298      *
299      * @return The proportion of up time that the process was executing in kernel or
300      *         user mode.
301      */
302     double getProcessCpuLoadCumulative();
303 
304     /**
305      * Gets CPU usage of this process since a previous snapshot of the same process,
306      * provided as a parameter.
307      * <p>
308      * This calculation sums CPU ticks across all processors and may exceed 100% for
309      * multi-threaded processes. This is consistent with process usage calulations
310      * on Linux/Unix machines, but should be divided by the number of logical
311      * processors to match the value displayed by the Windows Task Manager.
312      * <p>
313      * The accuracy of this calculation is dependent on both the number of threads
314      * on which the process is executing, and the precision of the Operating
315      * System's tick counters. A polling interval of at least a few seconds is
316      * recommended.
317      *
318      * @param proc
319      *            An {@link OSProcess} object containing statistics for this same
320      *            process collected at a prior point in time. May be null.
321      *
322      * @return If the prior snapshot is for the same process at a prior point in
323      *         time, the proportion of elapsed up time between the current process
324      *         snapshot and the previous one that the process was executing in
325      *         kernel or user mode. Returns cumulative load otherwise.
326      */
327     double getProcessCpuLoadBetweenTicks(OSProcess proc);
328 
329     /**
330      * Attempts to get the bitness (32 or 64) of the process.
331      *
332      * @return The bitness, if able to be determined, 0 otherwise.
333      */
334     int getBitness();
335 
336     /**
337      * Retrieves the process affinity mask for this process.
338      * <p>
339      * On Windows systems with more than 64 processors, if the threads of the
340      * calling process are in a single processor group, returns the process affinity
341      * mask for that group (which may be zero if the specified process is running in
342      * a different group). If the calling process contains threads in multiple
343      * groups, returns zero.
344      * <p>
345      * Because macOS does not export interfaces that identify processors or control
346      * thread placement, explicit thread to processor binding is not supported and
347      * this method will return a bitmask of all logical processors.
348      * <p>
349      * If the Operating System fails to retrieve an affinity mask (e.g., the process
350      * has terminated), returns zero.
351      *
352      * @return a bit vector in which each bit represents the processors that a
353      *         process is allowed to run on.
354      */
355     long getAffinityMask();
356 
357     /**
358      * Attempts to updates process attributes. Returns false if the update fails,
359      * which will occur if the process no longer exists.
360      *
361      * @return {@code true} if the update was successful, false if the update
362      *         failed. In addition, on a failed update the process state will be
363      *         changed to {@link State#INVALID}.
364      */
365     boolean updateAttributes();
366 
367     /**
368      * Retrieves the threads of the process and their details.
369      * <p>
370      * The amount of returned information is operating-system dependent and may
371      * incur some latency.
372      *
373      * @return a list of threads
374      */
375     List<OSThread> getThreadDetails();
376 
377     /**
378      * The number of minor (soft) faults the process has made which have not
379      * required loading a memory page from disk. Sometimes called reclaims.
380      * <p>
381      * Not available on Solaris. On Windows, this includes the total of major and
382      * minor faults.
383      *
384      * @return minor page faults (reclaims).
385      */
386     long getMinorFaults();
387 
388     /**
389      * The number of major (hard) faults the process has made which have required
390      * loading a memory page from disk.
391      * <p>
392      * Not available on Solaris. Windows does not distinguish major and minor faults
393      * at the process level, so this value returns 0 and major faults are included
394      * in {@link #getMinorFaults()}.
395      *
396      * @return major page faults.
397      */
398     long getMajorFaults();
399 
400     /**
401      * Process Execution States
402      */
403     enum State {
404         /**
405          * Intermediate state in process creation
406          */
407         NEW,
408         /**
409          * Actively executing process
410          */
411         RUNNING,
412         /**
413          * Interruptible sleep state
414          */
415         SLEEPING,
416         /**
417          * Blocked, uninterruptible sleep state
418          */
419         WAITING,
420         /**
421          * Intermediate state in process termination
422          */
423         ZOMBIE,
424         /**
425          * Stopped by the user, such as for debugging
426          */
427         STOPPED,
428         /**
429          * Other or unknown states not defined
430          */
431         OTHER,
432         /**
433          * The state resulting if the process fails to update statistics, probably due
434          * to termination.
435          */
436         INVALID
437     }
438 }
