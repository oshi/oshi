24  package oshi.software.os;
25  
26  import java.util.List;
27  import java.util.Map;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.driver.windows.wmi.Win32ProcessCached;
31  import oshi.util.FileUtil;
32  import oshi.util.GlobalConfig;
33  
34  /**
35   * Represents a Process on the operating system, which may contain multiple
36   * threads.
37   */
38  @ThreadSafe
39  public interface OSProcess {
40  
41      /**
42       * Gets the name of the process, often the executable program.
43       *
44       * @return the name of the process.
45       */
46      String getName();
47  
48      /**
49       * Gets the full filesystem path of the executing process.
50       *
51       * @return the full path of the executing process.
52       */
53      String getPath();
54  
55      /**
56       * Gets the process command line used to start the process, including arguments
57       * if available to be determined. This method generally returns the same
58       * information as {@link #getArguments()} in a more user-readable format, and is
59       * more robust to non-elevated access.
60       * <p>
61       * The format of this string is platform-dependent, may be truncated, and may
62       * require the end user to parse the result. Users should generally prefer
63       * {@link #getArguments()} which already parses the results, and use this method
64       * as a backup.
65       * <p>
66       * On Linux and macOS systems, the string is null-character-delimited, to permit
67       * the end user to parse the executable and arguments if desired. This
68       * null-delimited behavior may change in future versions and should not be
69       * relied upon; use {@link #getArguments()} instead.
70       * <p>
71       * On AIX and Solaris, the string may be truncated to 80 characters if there was
72       * insufficient permission to read the process memory.
73       * <p>
74       * On Windows, attempts to retrieve the value from process memory, which
75       * requires that the process be owned by the same user as the executing process,
76       * or elevated permissions, and additionally requires the target process to have
77       * the same bitness (e.g., this will fail on a 32-bit process if queried by
78       * 64-bit and vice versa). If reading process memory fails, by default, performs
79       * a single WMI query for this process, with some latency. If this method will
80       * be frequently called for multiple processes, see the configuration file to
81       * enable a batch query mode leveraging
82       * {@link Win32ProcessCached#getCommandLine} to improve performance, or setting
83       * that parameter via {@link GlobalConfig#set(String, Object)} before
84       * instantiating any {@link OSProcess} object.
85       *
86       * @return the process command line.
87       */
88      String getCommandLine();
89  
90      /**
91       * Makes a best effort attempt to get a list of the the command-line arguments
92       * of the process. Returns the same information as {@link #getCommandLine()} but
93       * parsed to a list. May require elevated permissions or same-user ownership.
94       *
95       * @return A list of Strings representing the arguments. May return an empty
96       *         list if there was a failure (for example, because the process is
97       *         already dead or permission was denied).
98       */
99      List<String> getArguments();
100 
101     /**
102      * Makes a best effort attempt to obtain the environment variables of the
103      * process. May require elevated permissions or same-user ownership.
104      *
105      * @return A map representing the environment variables and their values. May
106      *         return an empty map if there was a failure (for example, because the
107      *         process is already dead or permission was denied).
108      */
109     Map<String, String> getEnvironmentVariables();
110 
111     /**
112      * Makes a best effort attempt to obtain the current working directory for the
113      * process.
114      *
115      * @return the process current working directory.
116      */
117     String getCurrentWorkingDirectory();
118 
119     /**
120      * Gets the user name of the process owner.
121      *
122      * @return the user name. On Windows systems, also returns the domain prepended
123      *         to the username.
124      */
125     String getUser();
126 
127     /**
128      * Gets the user id of the process owner.
129      *
130      * @return the userID. On Windows systems, returns the Security ID (SID)
131      */
132     String getUserID();
133 
134     /**
135      * Gets the group under which the process is executing.
136      * <p>
137      * On Windows systems, populating this value for processes other than the
138      * current user requires administrative privileges (and still may fail for some
139      * system processes) and can incur significant latency. When successful, returns
140      * a the default primary group with access to this process, corresponding to the
141      * SID in {@link #getGroupID()}.
142      *
143      * @return the group.
144      */
145     String getGroup();
146 
147     /**
148      * Gets the group id under which the process is executing.
149      * <p>
150      * On Windows systems, populating this value for processes other than the
151      * current user requires administrative privileges (and still may fail for some
152      * system processes) and can incur significant latency. When successful, returns
153      * the default primary group SID with access to this process, corresponding to
154      * the name in {@link #getGroup()}.
155      *
156      * @return the groupID.
157      */
158     String getGroupID();
159 
160     /**
161      * Gets the process state.
162      *
163      * @return the execution state of the process.
164      */
165     State getState();
166 
167     /**
168      * Gets the process ID.
169      * <p>
170      * While this is a 32-bit value, it is unsigned on Windows and in extremely rare
171      * circumstances may return a negative value.
172      *
173      * @return the processID.
174      */
175     int getProcessID();
176 
177     /**
178      * Gets the process ID of this process's parent.
179      *
180      * @return the parentProcessID, if any; 0 otherwise.
181      */
182     int getParentProcessID();
183 
184     /**
185      * Gets the number of threads being executed by this process. More information
186      * is available using {@link #getThreadDetails()}.
187      *
188      * @return the number of threads in this process.
189      */
190     int getThreadCount();
191 
192     /**
193      * Gets the priority of this process.
194      * <p>
195      * For Linux and Unix, priority is a value in the range -20 to 19 (20 on some
196      * systems). The default priority is 0; lower priorities cause more favorable
197      * scheduling.
198      * <p>
199      * For Windows, priority values can range from 0 (lowest priority) to 31
200      * (highest priority).
201      * <p>
202      * macOS has 128 priority levels, ranging from 0 (lowest priority) to 127
203      * (highest priority). They are divided into several major bands: 0 through 51
204      * are the normal levels; the default priority is 31. 52 through 79 are the
205      * highest priority regular threads; 80 through 95 are for kernel mode threads;
206      * and 96 through 127 correspond to real-time threads, which are treated
207      * differently than other threads by the scheduler.
208      *
209      * @return the priority of this process.
210      */
211     int getPriority();
212 
213     /**
214      * Gets the Virtual Memory Size (VSZ). Includes all memory that the process can
215      * access, including memory that is swapped out and memory that is from shared
216      * libraries.
217      *
218      * @return the Virtual Memory Size
219      */
220     long getVirtualSize();
221 
222     /**
223      * Gets the Resident Set Size (RSS). Used to show how much memory is allocated
224      * to that process and is in RAM. It does not include memory that is swapped
225      * out. It does include memory from shared libraries as long as the pages from
226      * those libraries are actually in memory. It does include all stack and heap
227      * memory.
228      * <p>
229      * On Windows, returns the Private Working Set size, which should match the
230      * "Memory" column in the Windows Task Manager.
231      * <p>
232      * On Linux, returns the RSS value from {@code /proc/[pid]/stat}, which may be
233      * inaccurate because of a kernel-internal scalability optimization. If accurate
234      * values are required, read {@code /proc/[pid]/smaps} using
235      * {@link FileUtil#getKeyValueMapFromFile(String, String)}.
236      *
237      * @return the Resident Set Size
238      */
239     long getResidentSetSize();
240 
241     /**
242      * Gets kernel/system (privileged) time used by the process.
243      *
244      * @return the number of milliseconds the process has executed in kernel/system
245      *         mode.
246      */
247     long getKernelTime();
248 
249     /**
250      * Gets user time used by the process.
251      *
252      * @return the number of milliseconds the process has executed in user mode.
253      */
254     long getUserTime();
255 
256     /**
257      * Gets up time / elapsed time since the process started.
258      *
259      * @return the number of milliseconds since the process started.
260      */
261     long getUpTime();
262 
263     /**
264      * Gets the process start time.
265      *
266      * @return the start time of the process in number of milliseconds since January
267      *         1, 1970 UTC.
268      */
269     long getStartTime();
270 
271     /**
272      * Gets the bytes read by the process.
273      *
274      * @return the number of bytes the process has read from disk.
275      */
276     long getBytesRead();
277 
278     /**
279      * Gets the bytes written by the process.
280      *
281      * @return the number of bytes the process has written to disk.
282      */
283     long getBytesWritten();
284 
285     /**
286      * Gets the number of open file handles (or network connections) that belongs to
287      * the process.
288      * <p>
289      * On FreeBSD and Solaris, this value is only populated if information for a
290      * single process id is requested.
291      *
292      * @return open files or -1 if unknown or not supported
293      */
294     long getOpenFiles();
295 
296     /**
297      * Gets cumulative CPU usage of this process.
298      * <p>
299      * This calculation sums CPU ticks across all processors and may exceed 100% for
300      * multi-threaded processes. This is consistent with the cumulative CPU
301      * presented by the "top" command on Linux/Unix machines.
302      *
303      * @return The proportion of up time that the process was executing in kernel or
304      *         user mode.
305      */
306     double getProcessCpuLoadCumulative();
307 
308     /**
309      * Gets CPU usage of this process since a previous snapshot of the same process,
310      * provided as a parameter.
311      * <p>
312      * This calculation sums CPU ticks across all processors and may exceed 100% for
313      * multi-threaded processes. This is consistent with process usage calulations
314      * on Linux/Unix machines, but should be divided by the number of logical
315      * processors to match the value displayed by the Windows Task Manager.
316      * <p>
317      * The accuracy of this calculation is dependent on both the number of threads
318      * on which the process is executing, and the precision of the Operating
319      * System's tick counters. A polling interval of at least a few seconds is
320      * recommended.
321      *
322      * @param proc
323      *            An {@link OSProcess} object containing statistics for this same
324      *            process collected at a prior point in time. May be null.
325      *
326      * @return If the prior snapshot is for the same process at a prior point in
327      *         time, the proportion of elapsed up time between the current process
328      *         snapshot and the previous one that the process was executing in
329      *         kernel or user mode. Returns cumulative load otherwise.
330      */
331     double getProcessCpuLoadBetweenTicks(OSProcess proc);
332 
333     /**
334      * Attempts to get the bitness (32 or 64) of the process.
335      *
336      * @return The bitness, if able to be determined, 0 otherwise.
337      */
338     int getBitness();
339 
340     /**
341      * Gets the process affinity mask for this process.
342      * <p>
343      * On Windows systems with more than 64 processors, if the threads of the
344      * calling process are in a single processor group, returns the process affinity
345      * mask for that group (which may be zero if the specified process is running in
346      * a different group). If the calling process contains threads in multiple
347      * groups, returns zero.
348      * <p>
349      * Because macOS does not export interfaces that identify processors or control
350      * thread placement, explicit thread to processor binding is not supported and
351      * this method will return a bitmask of all logical processors.
352      * <p>
353      * If the Operating System fails to retrieve an affinity mask (e.g., the process
354      * has terminated), returns zero.
355      *
356      * @return a bit vector in which each bit represents the processors that a
357      *         process is allowed to run on.
358      */
359     long getAffinityMask();
360 
361     /**
362      * Attempts to update process attributes. Returns false if the update fails,
363      * which will occur if the process no longer exists.
364      *
365      * @return {@code true} if the update was successful, false if the update
366      *         failed. In addition, on a failed update the process state will be
367      *         changed to {@link State#INVALID}.
368      */
369     boolean updateAttributes();
370 
371     /**
372      * Retrieves the threads of the process and their details.
373      * <p>
374      * The amount of returned information is operating-system dependent and may
375      * incur some latency.
376      *
377      * @return a list of threads
378      */
379     List<OSThread> getThreadDetails();
380 
381     /**
382      * Gets the number of minor (soft) faults the process has made which have not
383      * required loading a memory page from disk. Sometimes called reclaims.
384      * <p>
385      * Not available on Solaris. On Windows, this includes the total of major and
386      * minor faults.
387      *
388      * @return minor page faults (reclaims).
389      */
390     default long getMinorFaults() {
391         return 0L;
392     }
393 
394     /**
395      * Gets the number of major (hard) faults the process has made which have
396      * required loading a memory page from disk.
397      * <p>
398      * Not available on Solaris. Windows does not distinguish major and minor faults
399      * at the process level, so this value returns 0 and major faults are included
400      * in {@link #getMinorFaults()}.
401      *
402      * @return major page faults.
403      */
404     default long getMajorFaults() {
405         return 0L;
406     }
407 
408     /**
409      * A snapshot of the context switches the process has done. Since the context
410      * switches could be voluntary and non-voluntary, this gives the sum of both.
411      * <p>
412      * Not available on Windows. An approximation may be made by summing associated
413      * values from {@link OSThread#getContextSwitches()}.
414      * <p>
415      * Not available on AIX.
416      *
417      * @return sum of both voluntary and involuntary context switches if available,
418      *         0 otherwise.
419      */
420     default long getContextSwitches() {
421         return 0L;
422     }
423 
424     /**
425      * Process and Thread Execution States
426      */
427     enum State {
428         /**
429          * Intermediate state in process creation
430          */
431         NEW,
432         /**
433          * Actively executing process
434          */
435         RUNNING,
436         /**
437          * Interruptible sleep state
438          */
439         SLEEPING,
440         /**
441          * Blocked, uninterruptible sleep state
442          */
443         WAITING,
444         /**
445          * Intermediate state in process termination
446          */
447         ZOMBIE,
448         /**
449          * Stopped by the user, such as for debugging
450          */
451         STOPPED,
452         /**
453          * Other or unknown states not defined
454          */
455         OTHER,
456         /**
457          * The state resulting if the process fails to update statistics, probably due
458          * to termination.
459          */
460         INVALID,
461         /**
462          * Special case of waiting if the process has been intentionally suspended
463          * (Windows only)
464          */
465         SUSPENDED
466     }
