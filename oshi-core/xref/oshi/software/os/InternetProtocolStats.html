28  import java.util.Arrays;
29  import java.util.List;

31  import oshi.annotation.concurrent.Immutable;

34  /**
35   * Includes key statistics of TCP and UDP protocols
36   */
37  @ThreadSafe
38  public interface InternetProtocolStats {
39  
40      /**
41       * Get the TCP stats for IPv4 connections.
42       * <p>
43       * On macOS connection information requires elevated permissions. Without
44       * elevatd permissions, segment data is estimated.
45       *
46       * @return a {@link TcpStats} object encapsulating the stats.
47       */
48      TcpStats getTCPv4Stats();
49  
50      /**
51       * Get the TCP stats for IPv6 connections, if available. If not available
52       * separately, these may be 0 and included in IPv4 connections.
53       *
54       * @return a {@link TcpStats} object encapsulating the stats.
55       */
56      TcpStats getTCPv6Stats();
57  
58      /**
59       * Get the UDP stats for IPv4 datagrams.
60       *
61       * @return a {@link UdpStats} object encapsulating the stats.
62       */
63      UdpStats getUDPv4Stats();
64  
65      /**
66       * Get the UDP stats for IPv6 datagrams, if available. If not available
67       * separately, these may be 0 and included in IPv4 datagrams.
68       *
69       * @return a {@link UdpStats} object encapsulating the stats.
70       */
71      UdpStats getUDPv6Stats();
72  
73      /**
74       * Gets a list of TCP and UDP connections.
75       *
76       * @return A list of {@link IPConnection} objects for TCP and UDP connections.
77       */
78      List<IPConnection> getConnections();
79  
80      /**
81       * Encapsulates statistics associated with a TCP connection.
82       */
83      @Immutable
84      final class TcpStats {
85          private final long connectionsEstablished;
86          private final long connectionsActive;
87          private final long connectionsPassive;
88          private final long connectionFailures;
89          private final long connectionsReset;
90          private final long segmentsSent;
91          private final long segmentsReceived;
92          private final long segmentsRetransmitted;
93          private final long inErrors;
94          private final long outResets;
95  
96          public TcpStats(long connectionsEstablished, long connectionsActive, long connectionsPassive,
97                  long connectionFailures, long connectionsReset, long segmentsSent, long segmentsReceived,
98                  long segmentsRetransmitted, long inErrors, long outResets) {
99              this.connectionsEstablished = connectionsEstablished;
100             this.connectionsActive = connectionsActive;
101             this.connectionsPassive = connectionsPassive;
102             this.connectionFailures = connectionFailures;
103             this.connectionsReset = connectionsReset;
104             this.segmentsSent = segmentsSent;
105             this.segmentsReceived = segmentsReceived;
106             this.segmentsRetransmitted = segmentsRetransmitted;
107             this.inErrors = inErrors;
108             this.outResets = outResets;
109         }
110 
111         /**
112          * Connections Established is the number of TCP connections for which the
113          * current state is either ESTABLISHED or CLOSE-WAIT
114          *
115          * @return the connectionsEstablished
116          */
117         public long getConnectionsEstablished() {
118             return connectionsEstablished;
119         }
120 
121         /**
122          * Connections Active is the number of times TCP connections have made a direct
123          * transition to the SYN-SENT state from the CLOSED state. In other words, it
124          * shows a number of connections which are initiated by the local computer. The
125          * value is a cumulative total.
126          *
127          * @return the connectionsActive
128          */
129         public long getConnectionsActive() {
130             return connectionsActive;
131         }
132 
133         /**
134          * Connections Passive is the number of times TCP connections have made a direct
135          * transition to the SYN-RCVD state from the LISTEN state. In other words, it
136          * shows a number of connections to the local computer, which are initiated by
137          * remote computers. The value is a cumulative total.
138          *
139          * @return the connectionsPassive
140          */
141         public long getConnectionsPassive() {
142             return connectionsPassive;
143         }
144 
145         /**
146          * Connection Failures is the number of times TCP connections have made a direct
147          * transition to the CLOSED state from the SYN-SENT state or the SYN-RCVD state,
148          * plus the number of times TCP connections have made a direct transition to the
149          * LISTEN state from the SYN-RCVD state.
150          *
151          * @return the connectionFailures
152          */
153         public long getConnectionFailures() {
154             return connectionFailures;
155         }
156 
157         /**
158          * Connections Reset is the number of times TCP connections have made a direct
159          * transition to the CLOSED state from either the ESTABLISHED state or the
160          * CLOSE-WAIT state.
161          *
162          * @return the connectionsReset
163          */
164         public long getConnectionsReset() {
165             return connectionsReset;
166         }
167 
168         /**
169          * Segments Sent is the number of segments sent, including those on current
170          * connections, but excluding those containing only retransmitted bytes.
171          *
172          * @return the segmentsSent
173          */
174         public long getSegmentsSent() {
175             return segmentsSent;
176         }
177 
178         /**
179          * Segments Received is the number of segments received, including those
180          * received in error. This count includes segments received on currently
181          * established connections.
182          *
183          * @return the segmentsReceived
184          */
185         public long getSegmentsReceived() {
186             return segmentsReceived;
187         }
188 
189         /**
190          * Segments Retransmitted is the number of segments retransmitted, that is,
191          * segments transmitted containing one or more previously transmitted bytes.
192          *
193          * @return the segmentsRetransmitted
194          */
195         public long getSegmentsRetransmitted() {
196             return segmentsRetransmitted;
197         }
198 
199         /**
200          * The number of errors received.
201          *
202          * @return the inErrors
203          */
204         public long getInErrors() {
205             return inErrors;
206         }
207 
208         /**
209          * The number of segments transmitted with the reset flag set.
210          *
211          * @return the outResets
212          */
213         public long getOutResets() {
214             return outResets;
215         }
216 
217         @Override
218         public String toString() {
219             return "TcpStats [connectionsEstablished=" + connectionsEstablished + ", connectionsActive="
220                     + connectionsActive + ", connectionsPassive=" + connectionsPassive + ", connectionFailures="
221                     + connectionFailures + ", connectionsReset=" + connectionsReset + ", segmentsSent=" + segmentsSent
222                     + ", segmentsReceived=" + segmentsReceived + ", segmentsRetransmitted=" + segmentsRetransmitted
223                     + ", inErrors=" + inErrors + ", outResets=" + outResets + "]";
224         }
225     }

227     /**
228      * Encapsulates statistics associated with a UDP connection.
229      */
230     @Immutable
231     final class UdpStats {
232         private final long datagramsSent;
233         private final long datagramsReceived;
234         private final long datagramsNoPort;
235         private final long datagramsReceivedErrors;
236 
237         public UdpStats(long datagramsSent, long datagramsReceived, long datagramsNoPort,
238                 long datagramsReceivedErrors) {
239             this.datagramsSent = datagramsSent;
240             this.datagramsReceived = datagramsReceived;
241             this.datagramsNoPort = datagramsNoPort;
242             this.datagramsReceivedErrors = datagramsReceivedErrors;
243         }
244 
245         /**
246          * Datagrams Sent is the number of UDP datagrams sent from the entity.
247          *
248          * @return the datagramsSent
249          */
250         public long getDatagramsSent() {
251             return datagramsSent;
252         }
253 
254         /**
255          * Datagrams Received is the number of UDP datagrams delivered to UDP users.
256          *
257          * @return the datagramsReceived
258          */
259         public long getDatagramsReceived() {
260             return datagramsReceived;
261         }
262 
263         /**
264          * Datagrams No Port is the number of received UDP datagrams for which there was
265          * no application at the destination port.
266          *
267          * @return the datagramsNoPort
268          */
269         public long getDatagramsNoPort() {
270             return datagramsNoPort;
271         }
272 
273         /**
274          * Datagrams Received Errors is the number of received UDP datagrams that could
275          * not be delivered for reasons other than the lack of an application at the
276          * destination port.
277          *
278          * @return the datagramsReceivedErrors
279          */
280         public long getDatagramsReceivedErrors() {
281             return datagramsReceivedErrors;
282         }
283 
284         @Override
285         public String toString() {
286             return "UdpStats [datagramsSent=" + datagramsSent + ", datagramsReceived=" + datagramsReceived
287                     + ", datagramsNoPort=" + datagramsNoPort + ", datagramsReceivedErrors=" + datagramsReceivedErrors
288                     + "]";
289         }
290     }
291 
292     /**
293      * The TCP connection state as described in RFC 793.
294      */
295     enum TcpState {
296         UNKNOWN, CLOSED, LISTEN, SYN_SENT, SYN_RECV, ESTABLISHED, FIN_WAIT_1, FIN_WAIT_2, CLOSE_WAIT, CLOSING, LAST_ACK,
297         TIME_WAIT, NONE;
298     }

300     /**
301      * Encapsulates information associated with an IP connection.
302      */
303     @Immutable
304     final class IPConnection {
305         private final String type;
306         private final byte[] localAddress;
307         private final int localPort;
308         private final byte[] foreignAddress;
309         private final int foreignPort;
310         private final TcpState state;
311         private final int transmitQueue;
312         private final int receiveQueue;
313         private int owningProcessId;
314 
315         public IPConnection(String type, byte[] localAddress, int localPort, byte[] foreignAddress, int foreignPort,
316                 TcpState state, int transmitQueue, int receiveQueue, int owningProcessId) {
317             this.type = type;
318             this.localAddress = Arrays.copyOf(localAddress, localAddress.length);
319             this.localPort = localPort;
320             this.foreignAddress = Arrays.copyOf(foreignAddress, foreignAddress.length);
321             this.foreignPort = foreignPort;
322             this.state = state;
323             this.transmitQueue = transmitQueue;
324             this.receiveQueue = receiveQueue;
325             this.owningProcessId = owningProcessId;
326         }
327 
328         /**
329          * Returns the connection protocol type, e.g., tcp4, tcp6, udp4, udp6
330          *
331          * @return The protocol type
332          */
333         public String getType() {
334             return type;
335         }
336 
337         /**
338          * Gets the local address. For IPv4 addresses this is a 4-byte array. For IPv6
339          * addresses this is a 16-byte array.
340          * <p>
341          * On Unix operating systems, the 16-bit value may be truncated, giving only the
342          * high order bytes. IPv6 addresses ending in zeroes should be considered
343          * suspect.
344          *
345          * @return The local address, or an empty array if the listener can accept a
346          *         connection on any interface.
347          */
348         public byte[] getLocalAddress() {
349             return Arrays.copyOf(localAddress, localAddress.length);
350         }
351 
352         /**
353          * Gets the local port.
354          *
355          * @return The local port, or 0 if unknown, or any port.
356          */
357         public int getLocalPort() {
358             return localPort;
359         }
360 
361         /**
362          * Gets the foreign/remote address. For IPv4 addresses this is a 4-byte array.
363          * For IPv6 addresses this is a 16-byte array.
364          * <p>
365          * On Unix operating systems, this value may be truncated. IPv6 addresses ending
366          * in zeroes should be considered suspect.
367          *
368          * @return The foreign/remote address, or an empty array if unknown. An empty
369          *         array will also result if
370          */
371         public byte[] getForeignAddress() {
372             return Arrays.copyOf(foreignAddress, foreignAddress.length);
373         }
374 
375         /**
376          * Gets the foreign/remote port.
377          *
378          * @return The foreign/remote port, or 0 if unknown.
379          */
380         public int getForeignPort() {
381             return foreignPort;
382         }
383 
384         /**
385          * Gets the connection state (TCP connections only).
386          *
387          * @return The connection state if known or relevant, null otherwise.
388          */
389         public TcpState getState() {
390             return state;
391         }
392 
393         /**
394          * Gets the size of the transmit queue. Not available on Windows.
395          *
396          * @return The size of the transmit queue, or 0 if unknown.
397          */
398         public int getTransmitQueue() {
399             return transmitQueue;
400         }
401 
402         /**
403          * Gets the size of the receive queue. Not available on Windows.
404          *
405          * @return The size of the receive queue, or 0 if unknown.
406          */
407         public int getReceiveQueue() {
408             return receiveQueue;
409         }
410 
411         /**
412          * Gets the id of the process which holds this connection.
413          *
414          * @return The process id of the process which holds this connection if known,
415          *         -1 otherwise.
416          */
417         public int getowningProcessId() {
418             return owningProcessId;
419         }
420 
421         @Override
422         public String toString() {
423             String localIp = "*";
424             try {
425                 localIp = InetAddress.getByAddress(localAddress).toString();
426             } catch (UnknownHostException e) { // NOSONAR squid:S108
427             }
428             String foreignIp = "*";
429             try {
430                 foreignIp = InetAddress.getByAddress(foreignAddress).toString();
431             } catch (UnknownHostException e) { // NOSONAR squid:S108
432             }
433             return "IPConnection [type=" + type + ", localAddress=" + localIp + ", localPort=" + localPort
434                     + ", foreignAddress=" + foreignIp + ", foreignPort=" + foreignPort + ", state=" + state
435                     + ", transmitQueue=" + transmitQueue + ", receiveQueue=" + receiveQueue + ", owningProcessId="
436                     + owningProcessId + "]";
437         }
438     }
