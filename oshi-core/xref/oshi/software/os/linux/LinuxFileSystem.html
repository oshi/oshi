52  /**
53   * The Linux File System contains {@link oshi.software.os.OSFileStore}s which
54   * are a storage pool, device, partition, volume, concrete file system or other
55   * implementation specific means of file storage. In Linux, these are found in
56   * the /proc/mount filesystem, excluding temporary and kernel mounts.
57   */

63      public static final String OSHI_LINUX_FS_PATH_EXCLUDES = "oshi.os.linux.filesystem.path.excludes";
64      public static final String OSHI_LINUX_FS_PATH_INCLUDES = "oshi.os.linux.filesystem.path.includes";
65      public static final String OSHI_LINUX_FS_VOLUME_EXCLUDES = "oshi.os.linux.filesystem.volume.excludes";
66      public static final String OSHI_LINUX_FS_VOLUME_INCLUDES = "oshi.os.linux.filesystem.volume.includes";

69              .loadAndParseFileSystemConfig(OSHI_LINUX_FS_PATH_EXCLUDES);

77      private static final String UNICODE_SPACE = "\\040";

79      @Override
80      public List<OSFileStore> getFileStores(boolean localOnly) {
81          // Map of volume with device path as key
82          Map<String, String> volumeDeviceMap = new HashMap<>();
83          File devMapper = new File("/dev/mapper");
84          File[] volumes = devMapper.listFiles();
85          if (volumes != null) {
86              for (File volume : volumes) {
87                  try {
88                      volumeDeviceMap.put(volume.getCanonicalPath(), volume.getAbsolutePath());
89                  } catch (IOException e) {
90                      LOG.error("Couldn't get canonical path for {}. {}", volume.getName(), e.getMessage());
91                  }
92              }
93          }
94          // Map uuids with device path as key
95          Map<String, String> uuidMap = new HashMap<>();
96          File uuidDir = new File("/dev/disk/by-uuid");
97          File[] uuids = uuidDir.listFiles();
98          if (uuids != null) {
99              for (File uuid : uuids) {
100                 try {
101                     // Store UUID as value with path (e.g., /dev/sda1) and volumes as key
102                     String canonicalPath = uuid.getCanonicalPath();
103                     uuidMap.put(canonicalPath, uuid.getName().toLowerCase());
104                     if (volumeDeviceMap.containsKey(canonicalPath)) {
105                         uuidMap.put(volumeDeviceMap.get(canonicalPath), uuid.getName().toLowerCase());
106                     }
107                 } catch (IOException e) {
108                     LOG.error("Couldn't get canonical path for {}. {}", uuid.getName(), e.getMessage());
109                 }
110             }
111         }
112 
113         // List file systems
114         return getFileStoreMatching(null, uuidMap, localOnly);
115     }

117     // called from LinuxOSFileStore
118     static List<OSFileStore> getFileStoreMatching(String nameToMatch, Map<String, String> uuidMap) {
119         return getFileStoreMatching(nameToMatch, uuidMap, false);
120     }

122     private static List<OSFileStore> getFileStoreMatching(String nameToMatch, Map<String, String> uuidMap,
123             boolean localOnly) {
124         List<OSFileStore> fsList = new ArrayList<>();
125 
126         Map<String, String> labelMap = queryLabelMap();
127 
128         // Parse /proc/mounts to get fs types
129         List<String> mounts = FileUtil.readFile(ProcPath.MOUNTS);
130         for (String mount : mounts) {
131             String[] split = mount.split(" ");
132             // As reported in fstab(5) manpage, struct is:
133             // 1st field is volume name
134             // 2nd field is path with spaces escaped as \040
135             // 3rd field is fs type
136             // 4th field is mount options
137             // 5th field is used by dump(8) (ignored)
138             // 6th field is fsck order (ignored)
139             if (split.length < 6) {
140                 continue;
141             }
142 
143             // Exclude pseudo file systems
144             String volume = split[0].replace(UNICODE_SPACE, " ");
145             String name = volume;
146             String path = split[1].replace(UNICODE_SPACE, " ");
147             if (path.equals("/")) {
148                 name = "/";
149             }
150             String type = split[2];
151 
152             // Skip non-local drives if requested, and exclude pseudo file systems
153             if ((localOnly && NETWORK_FS_TYPES.contains(type))
154                     || !path.equals("/") && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path,
155                             volume, FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
156                 continue;
157             }
158 
159             String options = split[3];
160 
161             // If only updating for one name, skip others
162             if (nameToMatch != null && !nameToMatch.equals(name)) {
163                 continue;
164             }
165 
166             String uuid = uuidMap != null ? uuidMap.getOrDefault(split[0], "") : "";
167 
168             String description;
169             if (volume.startsWith("/dev")) {
170                 description = "Local Disk";
171             } else if (volume.equals("tmpfs")) {
172                 description = "Ram Disk";
173             } else if (NETWORK_FS_TYPES.contains(type)) {
174                 description = "Network Disk";
175             } else {
176                 description = "Mount Point";
177             }
178 
179             // Add in logical volume found at /dev/mapper, useful when linking
180             // file system with drive.
181             String logicalVolume = "";
182             String volumeMapperDirectory = "/dev/mapper/";
183             Path link = Paths.get(volume);
184             if (link.toFile().exists() && Files.isSymbolicLink(link)) {
185                 try {
186                     Path slink = Files.readSymbolicLink(link);
187                     Path full = Paths.get(volumeMapperDirectory + slink.toString());
188                     if (full.toFile().exists()) {
189                         logicalVolume = full.normalize().toString();
190                     }
191                 } catch (IOException e) {
192                     LOG.warn("Couldn't access symbolic path  {}. {}", link, e.getMessage());
193                 }
194             }
195 
196             long totalInodes = 0L;
197             long freeInodes = 0L;
198             long totalSpace = 0L;
199             long usableSpace = 0L;
200             long freeSpace = 0L;
201 
202             try {
203                 LibC.Statvfs vfsStat = new LibC.Statvfs();
204                 if (0 == LibC.INSTANCE.statvfs(path, vfsStat)) {
205                     totalInodes = vfsStat.f_files.longValue();
206                     freeInodes = vfsStat.f_ffree.longValue();
207                     // Per stavfs, these units are in fragments
208                     totalSpace = vfsStat.f_blocks.longValue() * vfsStat.f_frsize.longValue();
209                     usableSpace = vfsStat.f_bavail.longValue() * vfsStat.f_frsize.longValue();
210                     freeSpace = vfsStat.f_bfree.longValue() * vfsStat.f_frsize.longValue();
211                 } else {
212                     LOG.warn("Failed to get information to use statvfs. path: {}, Error code: {}", path,
213                             Native.getLastError());
214                 }
215             } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
216                 LOG.error("Failed to get file counts from statvfs. {}", e.getMessage());
217             }
218             // If native methods failed use JVM methods
219             if (totalSpace == 0L) {
220                 File tmpFile = new File(path);
221                 totalSpace = tmpFile.getTotalSpace();
222                 usableSpace = tmpFile.getUsableSpace();
223                 freeSpace = tmpFile.getFreeSpace();
224             }
225 
226             fsList.add(new LinuxOSFileStore(name, volume, labelMap.getOrDefault(path, name), path, options, uuid,
227                     logicalVolume, description, type, freeSpace, usableSpace, totalSpace, freeInodes, totalInodes));
228         }
229         return fsList;
230     }

232     private static Map<String, String> queryLabelMap() {
233         Map<String, String> labelMap = new HashMap<>();
234         for (String line : ExecutingCommand.runNative("lsblk -o mountpoint,label")) {
235             String[] split = ParseUtil.whitespaces.split(line, 2);
236             if (split.length == 2) {
237                 labelMap.put(split[0], split[1]);
238             }
239         }
240         return labelMap;
241     }

253     /**
254      * Returns a value from the Linux system file /proc/sys/fs/file-nr.
255      *
256      * @param index
257      *            The index of the value to retrieve. 0 returns the total allocated
258      *            file descriptors. 1 returns the number of used file descriptors
259      *            for kernel 2.4, or the number of unused file descriptors for
260      *            kernel 2.6. 2 returns the maximum number of file descriptors that
261      *            can be allocated.
262      * @return Corresponding file descriptor value from the Linux system file.
263      */
264     private static long getFileDescriptors(int index) {
265         String filename = ProcPath.SYS_FS_FILE_NR;
266         if (index < 0 || index > 2) {
267             throw new IllegalArgumentException("Index must be between 0 and 2.");
268         }
269         List<String> osDescriptors = FileUtil.readFile(filename);
270         if (!osDescriptors.isEmpty()) {
271             String[] splittedLine = osDescriptors.get(0).split("\\D+");
272             return ParseUtil.parseLongOrDefault(splittedLine[index], 0L);
273         }
274         return 0L;
275     }
