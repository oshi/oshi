35   * The Linux File System contains {@link oshi.software.os.OSFileStore}s which are a storage pool, device, partition,
36   * volume, concrete file system or other implementation specific means of file storage. In Linux, these are found in the
37   * /proc/mount filesystem, excluding temporary and kernel mounts.

60      @Override
61      public List<OSFileStore> getFileStores(boolean localOnly) {
62          // Map of volume with device path as key
63          Map<String, String> volumeDeviceMap = new HashMap<>();
64          File devMapper = new File("/dev/mapper");
65          File[] volumes = devMapper.listFiles();
66          if (volumes != null) {
67              for (File volume : volumes) {
68                  try {
69                      volumeDeviceMap.put(volume.getCanonicalPath(), volume.getAbsolutePath());
70                  } catch (IOException e) {
71                      LOG.error("Couldn't get canonical path for {}. {}", volume.getName(), e.getMessage());
72                  }
73              }
74          }
75          // Map uuids with device path as key
76          Map<String, String> uuidMap = new HashMap<>();
77          File uuidDir = new File("/dev/disk/by-uuid");
78          File[] uuids = uuidDir.listFiles();
79          if (uuids != null) {
80              for (File uuid : uuids) {
81                  try {
82                      // Store UUID as value with path (e.g., /dev/sda1) and volumes as key
83                      String canonicalPath = uuid.getCanonicalPath();
84                      uuidMap.put(canonicalPath, uuid.getName().toLowerCase(Locale.ROOT));
85                      if (volumeDeviceMap.containsKey(canonicalPath)) {
86                          uuidMap.put(volumeDeviceMap.get(canonicalPath), uuid.getName().toLowerCase(Locale.ROOT));
87                      }
88                  } catch (IOException e) {
89                      LOG.error("Couldn't get canonical path for {}. {}", uuid.getName(), e.getMessage());
90                  }
91              }
92          }
93  
94          // List file systems
95          return getFileStoreMatching(null, uuidMap, localOnly);
96      }

99      static List<OSFileStore> getFileStoreMatching(String nameToMatch, Map<String, String> uuidMap) {
100         return getFileStoreMatching(nameToMatch, uuidMap, false);
101     }

103     private static List<OSFileStore> getFileStoreMatching(String nameToMatch, Map<String, String> uuidMap,
104             boolean localOnly) {
105         List<OSFileStore> fsList = new ArrayList<>();
106 
107         Map<String, String> labelMap = queryLabelMap();
108 
109         // Parse /proc/mounts to get fs types
110         List<String> mounts = FileUtil.readFile(ProcPath.MOUNTS);
111         for (String mount : mounts) {
112             String[] split = mount.split(" ");
113             // As reported in fstab(5) manpage, struct is:
114             // 1st field is volume name
115             // 2nd field is path with spaces escaped as \040
116             // 3rd field is fs type
117             // 4th field is mount options
118             // 5th field is used by dump(8) (ignored)
119             // 6th field is fsck order (ignored)
120             if (split.length < 6) {
121                 continue;
122             }
123 
124             // Exclude pseudo file systems
125             String volume = split[0].replace(UNICODE_SPACE, " ");
126             String name = volume;
127             String path = split[1].replace(UNICODE_SPACE, " ");
128             if (path.equals("/")) {
129                 name = "/";
130             }
131             String type = split[2];
132 
133             // Skip non-local drives if requested, and exclude pseudo file systems
134             if ((localOnly && NETWORK_FS_TYPES.contains(type))
135                     || !path.equals("/") && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path,
136                             volume, FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
137                 continue;
138             }
139 
140             String options = split[3];
141 
142             // If only updating for one name, skip others
143             if (nameToMatch != null && !nameToMatch.equals(name)) {
144                 continue;
145             }
146 
147             String uuid = uuidMap != null ? uuidMap.getOrDefault(split[0], "") : "";
148 
149             String description;
150             if (volume.startsWith("/dev")) {
151                 description = "Local Disk";
152             } else if (volume.equals("tmpfs")) {
153                 description = "Ram Disk";
154             } else if (NETWORK_FS_TYPES.contains(type)) {
155                 description = "Network Disk";
156             } else {
157                 description = "Mount Point";
158             }
159 
160             // Add in logical volume found at /dev/mapper, useful when linking
161             // file system with drive.
162             String logicalVolume = "";
163             String volumeMapperDirectory = "/dev/mapper/";
164             Path link = Paths.get(volume);
165             if (link.toFile().exists() && Files.isSymbolicLink(link)) {
166                 try {
167                     Path slink = Files.readSymbolicLink(link);
168                     Path full = Paths.get(volumeMapperDirectory + slink.toString());
169                     if (full.toFile().exists()) {
170                         logicalVolume = full.normalize().toString();
171                     }
172                 } catch (IOException e) {
173                     LOG.warn("Couldn't access symbolic path  {}. {}", link, e.getMessage());
174                 }
175             }
176 
177             long totalInodes = 0L;
178             long freeInodes = 0L;
179             long totalSpace = 0L;
180             long usableSpace = 0L;
181             long freeSpace = 0L;
182 
183             try {
184                 LibC.Statvfs vfsStat = new LibC.Statvfs();
185                 if (0 == LibC.INSTANCE.statvfs(path, vfsStat)) {
186                     totalInodes = vfsStat.f_files.longValue();
187                     freeInodes = vfsStat.f_ffree.longValue();
188                     // Per stavfs, these units are in fragments
189                     totalSpace = vfsStat.f_blocks.longValue() * vfsStat.f_frsize.longValue();
190                     usableSpace = vfsStat.f_bavail.longValue() * vfsStat.f_frsize.longValue();
191                     freeSpace = vfsStat.f_bfree.longValue() * vfsStat.f_frsize.longValue();
192                 } else {
193                     LOG.warn("Failed to get information to use statvfs. path: {}, Error code: {}", path,
194                             Native.getLastError());
195                 }
196             } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
197                 LOG.error("Failed to get file counts from statvfs. {}", e.getMessage());
198             }
199             // If native methods failed use JVM methods
200             if (totalSpace == 0L) {
201                 File tmpFile = new File(path);
202                 totalSpace = tmpFile.getTotalSpace();
203                 usableSpace = tmpFile.getUsableSpace();
204                 freeSpace = tmpFile.getFreeSpace();
205             }
206 
207             fsList.add(new LinuxOSFileStore(name, volume, labelMap.getOrDefault(path, name), path, options, uuid,
208                     logicalVolume, description, type, freeSpace, usableSpace, totalSpace, freeInodes, totalInodes));
209         }
210         return fsList;
211     }

213     private static Map<String, String> queryLabelMap() {
214         Map<String, String> labelMap = new HashMap<>();
215         for (String line : ExecutingCommand.runNative("lsblk -o mountpoint,label")) {
216             String[] split = ParseUtil.whitespaces.split(line, 2);
217             if (split.length == 2) {
218                 labelMap.put(split[0], split[1]);
219             }
220         }
221         return labelMap;
222     }

239     /**
240      * Returns a value from the Linux system file /proc/sys/fs/file-nr.
241      *
242      * @param index The index of the value to retrieve. 0 returns the total allocated file descriptors. 1 returns the
243      *              number of used file descriptors for kernel 2.4, or the number of unused file descriptors for kernel
244      *              2.6. 2 returns the maximum number of file descriptors that can be allocated.
245      * @return Corresponding file descriptor value from the Linux system file.
246      */
247     private static long getFileDescriptors(int index) {
248         String filename = ProcPath.SYS_FS_FILE_NR;
249         if (index < 0 || index > 2) {
250             throw new IllegalArgumentException("Index must be between 0 and 2.");
251         }
252         List<String> osDescriptors = FileUtil.readFile(filename);
253         if (!osDescriptors.isEmpty()) {
254             String[] splittedLine = osDescriptors.get(0).split("\\D+");
255             return ParseUtil.parseLongOrDefault(splittedLine[index], 0L);
256         }
257         return 0L;
258     }
259 
260     private static long getFileDescriptorsPerProcess() {
261         return FileUtil.getLongFromFile(ProcPath.SYS_FS_FILE_MAX);
262     }
