67      public MacOperatingSystem() {
68          this.manufacturer = "Apple";
69          this.version = new MacOSVersionInfoEx();
70          this.family = ParseUtil.getFirstIntValue(this.version.getVersion()) == 10
71                  && ParseUtil.getNthIntValue(this.version.getVersion(), 2) >= 12 ? "macOS"
72                          : System.getProperty("os.name");
73          // Set max processes
74          this.maxProc = SysctlUtil.sysctl("kern.maxproc", 0x1000);
75          initBitness();
76      }
77  
78      private void initBitness() {
79          if (this.bitness < 64) {
80              if (getVersion().getOsxVersionNumber() > 7) {
81                  this.bitness = 64;
82              } else {
83                  this.bitness = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("getconf LONG_BIT"), 32);
84              }
85          }
86      }

99      @Override
100     public OSProcess[] getProcesses(int limit, ProcessSort sort) {
101         List<OSProcess> procs = new ArrayList<>();
102         int[] pids = new int[this.maxProc];
103         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids,
104                 pids.length * SystemB.INT_SIZE) / SystemB.INT_SIZE;
105         for (int i = 0; i < numberOfProcesses; i++) {
106             // Handle off-by-one bug in proc_listpids where the size returned
107             // is: SystemB.INT_SIZE * (pids + 1)
108             if (pids[i] == 0) {
109                 continue;
110             }
111 
112             OSProcess proc = getProcess(pids[i]);
113             if (proc != null) {
114                 procs.add(proc);
115             }
116         }
117         List<OSProcess> sorted = processSort(procs, limit, sort);
118         return sorted.toArray(new OSProcess[sorted.size()]);
119     }

124     @Override
125     public OSProcess getProcess(int pid) {
126         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
127         if (0 > SystemB.INSTANCE.proc_pidinfo(pid, SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo, taskAllInfo.size())) {
128             return null;
129         }
130         String name = null;
131         String path = "";
132         Pointer buf = new Memory(SystemB.PROC_PIDPATHINFO_MAXSIZE);
133         if (0 < SystemB.INSTANCE.proc_pidpath(pid, buf, SystemB.PROC_PIDPATHINFO_MAXSIZE)) {
134             path = buf.getString(0).trim();
135             // Overwrite name with last part of path
136             String[] pathSplit = path.split("/");
137             if (pathSplit.length > 0) {
138                 name = pathSplit[pathSplit.length - 1];
139             }
140         }
141         // If process is gone, return null
142         if (taskAllInfo.ptinfo.pti_threadnum < 1) {
143             return null;
144         }
145         if (name == null) {
146             // pbi_comm contains first 16 characters of name
147             // null terminated
148             for (int t = 0; t < taskAllInfo.pbsd.pbi_comm.length; t++) {
149                 if (taskAllInfo.pbsd.pbi_comm[t] == 0) {
150                     name = new String(taskAllInfo.pbsd.pbi_comm, 0, t);
151                     break;
152                 }
153             }
154         }
155         long bytesRead = 0;
156         long bytesWritten = 0;
157         if (getVersion().getOsxVersionNumber() >= 9) {
158             RUsageInfoV2 rUsageInfoV2 = new RUsageInfoV2();
159             if (0 == SystemB.INSTANCE.proc_pid_rusage(pid, SystemB.RUSAGE_INFO_V2, rUsageInfoV2)) {
160                 bytesRead = rUsageInfoV2.ri_diskio_bytesread;
161                 bytesWritten = rUsageInfoV2.ri_diskio_byteswritten;
162             }
163         }
164         long now = System.currentTimeMillis();
165         OSProcessml#OSProcess">OSProcess proc = new OSProcess();
166         proc.setName(name);
167         proc.setPath(path);
168         switch (taskAllInfo.pbsd.pbi_status) {
169         case SSLEEP:
170             proc.setState(OSProcess.State.SLEEPING);
171             break;
172         case SWAIT:
173             proc.setState(OSProcess.State.WAITING);
174             break;
175         case SRUN:
176             proc.setState(OSProcess.State.RUNNING);
177             break;
178         case SIDL:
179             proc.setState(OSProcess.State.NEW);
180             break;
181         case SZOMB:
182             proc.setState(OSProcess.State.ZOMBIE);
183             break;
184         case SSTOP:
185             proc.setState(OSProcess.State.STOPPED);
186             break;
187         default:
188             proc.setState(OSProcess.State.OTHER);
189             break;
190         }
191         proc.setProcessID(pid);
192         proc.setParentProcessID(taskAllInfo.pbsd.pbi_ppid);
193         proc.setUserID(Integer.toString(taskAllInfo.pbsd.pbi_uid));
194         Passwd user = SystemB.INSTANCE.getpwuid(taskAllInfo.pbsd.pbi_uid);
195         proc.setUser(user == null ? proc.getUserID() : user.pw_name);
196         proc.setGroupID(Integer.toString(taskAllInfo.pbsd.pbi_gid));
197         Group group = SystemB.INSTANCE.getgrgid(taskAllInfo.pbsd.pbi_gid);
198         proc.setGroup(group == null ? proc.getGroupID() : group.gr_name);
199         proc.setThreadCount(taskAllInfo.ptinfo.pti_threadnum);
200         proc.setPriority(taskAllInfo.ptinfo.pti_priority);
201         proc.setVirtualSize(taskAllInfo.ptinfo.pti_virtual_size);
202         proc.setResidentSetSize(taskAllInfo.ptinfo.pti_resident_size);
203         proc.setKernelTime(taskAllInfo.ptinfo.pti_total_system / 1000000L);
204         proc.setUserTime(taskAllInfo.ptinfo.pti_total_user / 1000000L);
205         proc.setStartTime(taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L);
206         proc.setUpTime(now - proc.getStartTime());
207         proc.setBytesRead(bytesRead);
208         proc.setBytesWritten(bytesWritten);
209         proc.setCommandLine(getCommandLine(pid));
210         // gets the open files count
211         proc.setOpenFiles(taskAllInfo.pbsd.pbi_nfiles);
212 
213         VnodePathInfo vpi = new VnodePathInfo();
214         if (0 < SystemB.INSTANCE.proc_pidinfo(pid, SystemB.PROC_PIDVNODEPATHINFO, 0, vpi, vpi.size())) {
215             int len = 0;
216             for (byte b : vpi.pvi_cdir.vip_path) {
217                 if (b == 0) {
218                     break;
219                 }
220                 len++;
221             }
222             proc.setCurrentWorkingDirectory(new String(vpi.pvi_cdir.vip_path, 0, len, StandardCharsets.US_ASCII));
223         }
224         return proc;
225     }

230     @Override
231     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
232         List<OSProcess> procs = new ArrayList<>();
233         int[] pids = new int[this.maxProc];
234         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids,
235                 pids.length * SystemB.INT_SIZE) / SystemB.INT_SIZE;
236         for (int i = 0; i < numberOfProcesses; i++) {
237             // Handle off-by-one bug in proc_listpids where the size returned
238             // is: SystemB.INT_SIZE * (pids + 1)
239             if (pids[i] == 0) {
240                 continue;
241             }
242             if (parentPid == getParentProcessPid(pids[i])) {
243                 OSProcess proc = getProcess(pids[i]);
244                 if (proc != null) {
245                     procs.add(proc);
246                 }
247             }
248         }
249         List<OSProcess> sorted = processSort(procs, limit, sort);
250         return sorted.toArray(new OSProcess[sorted.size()]);
251     }

253     private int getParentProcessPid(int pid) {
254         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
255         if (0 > SystemB.INSTANCE.proc_pidinfo(pid, SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo, taskAllInfo.size())) {
256             return 0;
257         }
258         return taskAllInfo.pbsd.pbi_ppid;
259     }

261     private String getCommandLine(int pid) {
262         // Get command line via sysctl
263         int[] mib = new int[3];
264         mib[0] = 1; // CTL_KERN
265         mib[1] = 49; // KERN_PROCARGS2
266         mib[2] = pid;
267         // Allocate memory for arguments
268         int argmax = SysctlUtil.sysctl("kern.argmax", 0);
269         Pointer procargs = new Memory(argmax);
270         IntByReference size = new IntByReference(argmax);
271         // Fetch arguments
272         if (0 != SystemB.INSTANCE.sysctl(mib, mib.length, procargs, size, null, 0)) {
273             LOG.error("Failed syctl call: kern.procargs2, Error code: {}", Native.getLastError());
274             return "";
275         }
276         // Procargs contains an int representing total # of args, followed by a
277         // null-terminated execpath string and then the arguments, each
278         // null-terminated (possible multiple consecutive nulls),
279         // The execpath string is also the first arg.
280         int nargs = procargs.getInt(0);
281         // Sanity check
282         if (nargs < 0 || nargs > 1024) {
283             LOG.error("Nonsensical number of process arguments for pid {}: {}", pid, nargs);
284             return "";
285         }
286         List<String> args = new ArrayList<>(nargs);
287         // Skip first int (containing value of nargs)
288         long offset = SystemB.INT_SIZE;
289         // Skip exec_command
290         offset += procargs.getString(offset).length();
291         // Iterate character by character using offset
292         // Build each arg and add to list
293         while (nargs-- > 0 && offset < size.getValue()) {
294             // Advance through additional nulls
295             while (procargs.getByte(offset) == 0) {
296                 if (++offset >= size.getValue()) {
297                     break;
298                 }
299             }
300             // Grab a string. This should go until the null terminator
301             String arg = procargs.getString(offset);
302             args.add(arg);
303             // Advance offset to next null
304             offset += arg.length();
305         }
306         // Return args null-delimited
307         return FormatUtil.join("\0", args);
308     }

334     /**
335      * {@inheritDoc}
336      */
337     @Override
338     public int getThreadCount() {
339         // Get current pids, then slightly pad in case new process starts while
340         // allocating array space
341         int[] pids = new int[getProcessCount() + 10];
342         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids, pids.length)
343                 / SystemB.INT_SIZE;
344         int numberOfThreads = 0;
345         ProcTaskInfo taskInfo = new ProcTaskInfo();
346         for (int i = 0; i < numberOfProcesses; i++) {
347             SystemB.INSTANCE.proc_pidinfo(pids[i], SystemB.PROC_PIDTASKINFO, 0, taskInfo, taskInfo.size());
348             numberOfThreads += taskInfo.pti_threadnum;
349         }
350         return numberOfThreads;
351     }
