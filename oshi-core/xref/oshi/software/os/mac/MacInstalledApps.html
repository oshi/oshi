26  public final class MacInstalledApps {
27  
28      private static final String COLON = ":";
29      private static final CoreFoundation CF = CoreFoundation.INSTANCE;
30  
31      private MacInstalledApps() {
32      }
33  
34      public static List<ApplicationInfo> queryInstalledApps() {
35          List<String> output = ExecutingCommand.runNative("system_profiler SPApplicationsDataType");
36          return parseMacAppInfo(output);
37      }
38  
39      private static List<ApplicationInfo> parseMacAppInfo(List<String> lines) {
40          List<ApplicationInfo> appInfoList = new ArrayList<>();
41          String appName = null;
42          Map<String, String> appDetails = null;
43          boolean collectingAppDetails = false;
44          String dateFormat = getLocaleDateTimeFormat(kCFDateFormatterShortStyle);
45  
46          for (String line : lines) {
47              line = line.trim();
48  
49              // Check for app name, ends with ":"
50              if (line.endsWith(COLON)) {
51                  // When app and appDetails are not empty then we reached the next app, add it to the list
52                  if (appName != null && !appDetails.isEmpty()) {
53                      appInfoList.add(createAppInfo(appName, appDetails, dateFormat));
54                  }
55  
56                  // store app name and proceed with collecting app details
57                  appName = line.substring(0, line.length() - 1);
58                  appDetails = new HashMap<>();
59                  collectingAppDetails = true;
60                  continue;
61              }
62  
63              // Process app details
64              if (collectingAppDetails && line.contains(COLON)) {
65                  int colonIndex = line.indexOf(COLON);
66                  String key = line.substring(0, colonIndex).trim();
67                  String value = line.substring(colonIndex + 1).trim();
68                  appDetails.put(key, value);
69              }
70          }
71  
72          return appInfoList;
73      }
74  
75      private static ApplicationInfo createAppInfo(String name, Map<String, String> details, String dateFormat) {
76          String obtainedFrom = ParseUtil.getValueOrUnknown(details, "Obtained from");
77          String signedBy = ParseUtil.getValueOrUnknown(details, "Signed by");
78          String vendor = (obtainedFrom.equals("Identified Developer")) ? signedBy : obtainedFrom;
79  
80          String lastModified = details.getOrDefault("Last Modified", Constants.UNKNOWN);
81          long lastModifiedEpoch = ParseUtil.parseDateToEpoch(lastModified, dateFormat);
82  
83          // Additional info map
84          Map<String, String> additionalInfo = new HashMap<>();
85          additionalInfo.put("Kind", ParseUtil.getValueOrUnknown(details, "Kind"));
86          additionalInfo.put("Location", ParseUtil.getValueOrUnknown(details, "Location"));
87          additionalInfo.put("Get Info String", ParseUtil.getValueOrUnknown(details, "Get Info String"));
88  
89          return new ApplicationInfo(name, ParseUtil.getValueOrUnknown(details, "Version"), vendor, lastModifiedEpoch,
90                  additionalInfo);
91      }
92  
93      private static String getLocaleDateTimeFormat(CFDateFormatterStyle style) {
94          CFIndex styleIndex = style.index();
95          CFLocale locale = CF.CFLocaleCopyCurrent();
96          try {
97              CFDateFormatter formatter = CF.CFDateFormatterCreate(null, locale, styleIndex, styleIndex);
98              if (formatter == null) {
99                  return "";
100             }
101             try {
102                 CFStringRef format = CF.CFDateFormatterGetFormat(formatter);
103                 return (format == null) ? "" : format.stringValue();
104             } finally {
105                 CF.CFRelease(formatter);
106             }
107         } finally {
108             CF.CFRelease(locale);
109         }
110     }
