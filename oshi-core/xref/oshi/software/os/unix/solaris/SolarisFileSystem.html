41  /**
42   * The Solaris File System contains {@link OSFileStore}s which are a storage
43   * pool, device, partition, volume, concrete file system or other implementation
44   * specific means of file storage. In Solaris, these are found in the
45   * /proc/mount filesystem, excluding temporary and kernel mounts.
46   */
47  public class SolarisFileSystem implements FileSystem {

51      // Solaris defines a set of virtual file systems
52      private final List<String> pseudofs = Arrays.asList(new String[] { //
53              "proc", // Proc file system
54              "devfs", // Dev temporary file system
55              "ctfs", // Contract file system
56              "objfs", // Object file system
57              "mntfs", // Mount file system
58              "sharefs", // Share file system
59              "lofs", // Library file system
60              "autofs" // Auto mounting fs
61              // "tmpfs", // Temporary file system
62              // NOTE: tmpfs is evaluated apart, because Solaris uses it for
63              // RAMdisks
64      });
65  
66      // System path mounted as tmpfs
67      private final List<String> tmpfsPaths = Arrays.asList(new String[] { "/system", "/tmp", "/dev/fd" });
68  
69      /**
70       * Checks if file path equals or starts with an element in the given list
71       *
72       * @param aList
73       *            A list of path prefixes
74       * @param charSeq
75       *            a path to check
76       * @return true if the charSeq exactly equals, or starts with the directory in
77       *         aList
78       */
79      private boolean listElementStartsWith(List<String> aList, String charSeq) {
80          for (String match : aList) {
81              if (charSeq.equals(match) || charSeq.startsWith(match + "/")) {
82                  return true;
83              }
84          }
85          return false;
86      }

88      /**
89       * Gets File System Information.
90       *
91       * @return An array of {@link OSFileStore} objects representing mounted volumes.
92       *         May return disconnected volumes with
93       *         {@link OSFileStore#getTotalSpace()} = 0.
94       */
95      @Override
96      public OSFileStore[] getFileStores() {
97          List<OSFileStore> fsList = getFileStoreMatching(null);
98  
99          return fsList.toArray(new OSFileStore[0]);
100     }

102     private List<OSFileStore> getFileStoreMatching(String nameToMatch) {
103         List<OSFileStore> fsList = new ArrayList<>();
104 
105         // Get inode usage data
106         Map<String, Long> inodeFreeMap = new HashMap<>();
107         Map<String, Long> inodeTotalMap = new HashMap<>();
108         String key = null;
109         String total = null;
110         String free = null;
111         for (String line : ExecutingCommand.runNative("df -g")) {
112             /*- Sample Output:
113             /                  (/dev/md/dsk/d0    ):         8192 block size          1024 frag size
114             41310292 total blocks   18193814 free blocks 17780712 available        2486848 total files
115              2293351 free files     22282240 filesys id
116                  ufs fstype       0x00000004 flag             255 filename length
117             */
118             if (line.startsWith("/")) {
119                 key = ParseUtil.whitespaces.split(line)[0];
120                 total = null;
121             } else if (line.contains("available") && line.contains("total files")) {
122                 total = ParseUtil.getTextBetweenStrings(line, "available", "total files").trim();
123             } else if (line.contains("free files")) {
124                 free = ParseUtil.getTextBetweenStrings(line, "", "free files").trim();
125                 if (key != null && total != null) {
126                     inodeFreeMap.put(key, ParseUtil.parseLongOrDefault(free, 0L));
127                     inodeTotalMap.put(key, ParseUtil.parseLongOrDefault(total, 0L));
128                     key = null;
129                 }
130             }
131         }
132 
133         // Get mount table
134         for (String fs : ExecutingCommand.runNative("cat /etc/mnttab")) {
135             String[] split = ParseUtil.whitespaces.split(fs);
136             if (split.length < 5) {
137                 continue;
138             }
139             // 1st field is volume name
140             // 2nd field is mount point
141             // 3rd field is fs type
142             // other fields ignored
143             String volume = split[0];
144             String path = split[1];
145             String type = split[2];
146 
147             // Exclude pseudo file systems
148             if (this.pseudofs.contains(type) || path.equals("/dev") || listElementStartsWith(this.tmpfsPaths, path)
149                     || volume.startsWith("rpool") && !path.equals("/")) {
150                 continue;
151             }
152 
153             String name = path.substring(path.lastIndexOf('/') + 1);
154             // Special case for /, pull last element of volume instead
155             if (name.isEmpty()) {
156                 name = volume.substring(volume.lastIndexOf('/') + 1);
157             }
158 
159             if (nameToMatch != null && !nameToMatch.equals(name)) {
160                 continue;
161             }
162             File f = new File(path);
163             long totalSpace = f.getTotalSpace();
164             long usableSpace = f.getUsableSpace();
165             long freeSpace = f.getFreeSpace();
166 
167             String description;
168             if (volume.startsWith("/dev") || path.equals("/")) {
169                 description = "Local Disk";
170             } else if (volume.equals("tmpfs")) {
171                 description = "Ram Disk";
172             } else if (type.startsWith("nfs") || type.equals("cifs")) {
173                 description = "Network Disk";
174             } else {
175                 description = "Mount Point";
176             }
177 
178             // Add to the list
179             OSFileStoreml#OSFileStore">OSFileStore osStore = new OSFileStore();
180             osStore.setName(name);
181             osStore.setVolume(volume);
182             osStore.setMount(path);
183             osStore.setDescription(description);
184             osStore.setType(type);
185             osStore.setUUID(""); // No UUID info on Solaris
186             osStore.setFreeSpace(freeSpace);
187             osStore.setUsableSpace(usableSpace);
188             osStore.setTotalSpace(totalSpace);
189             osStore.setFreeInodes(inodeFreeMap.containsKey(path) ? inodeFreeMap.get(path) : 0L);
190             osStore.setTotalInodes(inodeTotalMap.containsKey(path) ? inodeTotalMap.get(path) : 0L);
191             fsList.add(osStore);
192         }
193         return fsList;
194     }

196     @Override
197     public long getOpenFileDescriptors() {
198         Kstat ksp = KstatUtil.kstatLookup(null, -1, "file_cache");
199         // Set values
200         if (ksp != null && KstatUtil.kstatRead(ksp)) {
201             return KstatUtil.kstatDataLookupLong(ksp, "buf_inuse");
202         }
203         return 0L;
204     }
205 
206     @Override
207     public long getMaxFileDescriptors() {
208         Kstat ksp = KstatUtil.kstatLookup(null, -1, "file_cache");
209         // Set values
210         if (ksp != null && KstatUtil.kstatRead(ksp)) {
211             return KstatUtil.kstatDataLookupLong(ksp, "buf_max");
212         }
213         return 0L;
214     }

216     public static boolean updateFileStoreStats(OSFileStore osFileStore) {
217         for (OSFileStore fileStore : new SolarisFileSystem().getFileStoreMatching(osFileStore.getName())) {
218             if (osFileStore.getVolume().equals(fileStore.getVolume())
219                     && osFileStore.getMount().equals(fileStore.getMount())) {
220                 osFileStore.setLogicalVolume(fileStore.getLogicalVolume());
221                 osFileStore.setDescription(fileStore.getDescription());
222                 osFileStore.setType(fileStore.getType());
223                 osFileStore.setFreeSpace(fileStore.getFreeSpace());
224                 osFileStore.setUsableSpace(fileStore.getUsableSpace());
225                 osFileStore.setTotalSpace(fileStore.getTotalSpace());
226                 osFileStore.setFreeInodes(fileStore.getFreeInodes());
227                 osFileStore.setTotalInodes(fileStore.getTotalInodes());
228                 return true;
229             }
230         }
231         return false;
232     }
