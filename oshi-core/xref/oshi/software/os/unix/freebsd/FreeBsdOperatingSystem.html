38   * computers.

42  public class FreeBsdOperatingSystem extends AbstractOperatingSystem {
43      private static final long serialVersionUID = 1L;
44  
45      public FreeBsdOperatingSystem() {
46          this.manufacturer = "Unix/BSD";
47          this.family = BsdSysctlUtil.sysctl("kern.ostype", "FreeBSD");
48          this.version = new FreeBsdOSVersionInfoEx();
49          initBitness();
50      }
51  
52      private void initBitness() {
53          if (this.bitness < 64 && ExecutingCommand.getFirstAnswer("uname -m").indexOf("64") != -1) {
54              this.bitness = 64;
55          }
56      }

58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public FileSystem getFileSystem() {
63          return new FreeBsdFileSystem();
64      }
65  
66      /**
67       * {@inheritDoc}
68       */
69      @Override
70      public OSProcess[] getProcesses(int limit, ProcessSort sort) {
71          List<OSProcess> procs = getProcessListFromPS(
72                  "ps -awwxo state,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etimes,systime,time,comm,args", -1);
73          List<OSProcess> sorted = processSort(procs, limit, sort);
74          return sorted.toArray(new OSProcess[sorted.size()]);
75      }
76  
77      /**
78       * {@inheritDoc}
79       */
80      @Override
81      public OSProcess getProcess(int pid) {
82          List<OSProcess> procs = getProcessListFromPS(
83                  "ps -awwxo state,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etimes,systime,time,comm,args -p ", pid);
84          if (procs.isEmpty()) {
85              return null;
86          }
87          return procs.get(0);
88      }
89  
90      /**
91       * {@inheritDoc}
92       */
93      @Override
94      public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
95          List<OSProcess> procs = getProcessListFromPS(
96                  "ps -awwxo state,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etimes,systime,time,comm,args --ppid",
97                  parentPid);
98          List<OSProcess> sorted = processSort(procs, limit, sort);
99          return sorted.toArray(new OSProcess[sorted.size()]);
100     }

102     private List<OSProcess> getProcessListFromPS(String psCommand, int pid) {
103         Map<Integer, String> cwdMap = LsofUtil.getCwdMap(pid);
104         List<OSProcess> procs = new ArrayList<>();
105         List<String> procList = ExecutingCommand.runNative(psCommand + (pid < 0 ? "" : pid));
106         if (procList.isEmpty() || procList.size() < 2) {
107             return procs;
108         }
109         // remove header row
110         procList.remove(0);
111         // Fill list
112         for (String proc : procList) {
113             String[] split = ParseUtil.whitespaces.split(proc.trim(), 16);
114             // Elements should match ps command order
115             if (split.length < 16) {
116                 continue;
117             }
118             long now = System.currentTimeMillis();
119             OSProcess fproc = new OSProcess();
120             switch (split[0].charAt(0)) {
121             case 'R':
122                 fproc.setState(OSProcess.State.RUNNING);
123                 break;
124             case 'I':
125             case 'S':
126                 fproc.setState(OSProcess.State.SLEEPING);
127                 break;
128             case 'D':
129             case 'L':
130             case 'U':
131                 fproc.setState(OSProcess.State.WAITING);
132                 break;
133             case 'Z':
134                 fproc.setState(OSProcess.State.ZOMBIE);
135                 break;
136             case 'T':
137                 fproc.setState(OSProcess.State.STOPPED);
138                 break;
139             default:
140                 fproc.setState(OSProcess.State.OTHER);
141                 break;
142             }
143             fproc.setProcessID(ParseUtil.parseIntOrDefault(split[1], 0));
144             fproc.setParentProcessID(ParseUtil.parseIntOrDefault(split[2], 0));
145             fproc.setUser(split[3]);
146             fproc.setUserID(split[4]);
147             fproc.setGroup(split[5]);
148             fproc.setGroupID(split[6]);
149             fproc.setThreadCount(ParseUtil.parseIntOrDefault(split[7], 0));
150             fproc.setPriority(ParseUtil.parseIntOrDefault(split[8], 0));
151             // These are in KB, multiply
152             fproc.setVirtualSize(ParseUtil.parseLongOrDefault(split[9], 0) * 1024);
153             fproc.setResidentSetSize(ParseUtil.parseLongOrDefault(split[10], 0) * 1024);
154             // Avoid divide by zero for processes up less than a second
155             long elapsedTime = ParseUtil.parseDHMSOrDefault(split[11], 0L);
156             fproc.setUpTime(elapsedTime < 1L ? 1L : elapsedTime);
157             fproc.setStartTime(now - fproc.getUpTime());
158             fproc.setKernelTime(ParseUtil.parseDHMSOrDefault(split[12], 0L));
159             fproc.setUserTime(ParseUtil.parseDHMSOrDefault(split[13], 0L) - fproc.getKernelTime());
160             fproc.setPath(split[14]);
161             fproc.setName(fproc.getPath().substring(fproc.getPath().lastIndexOf('/') + 1));
162             fproc.setCommandLine(split[15]);
163             fproc.setCurrentWorkingDirectory(MapUtil.getOrDefault(cwdMap, fproc.getProcessID(), ""));
164             // gets the open files count -- only do for single-PID requests
165             if (pid >= 0) {
166                 List<String> openFilesList = ExecutingCommand.runNative(String.format("lsof -p %d", pid));
167                 fproc.setOpenFiles(openFilesList.size() - 1L);
168             }
169             procs.add(fproc);
170         }
171         return procs;
172     }

174     /**
175      * {@inheritDoc}
176      */
177     @Override
178     public int getProcessId() {
179         return Libc.INSTANCE.getpid();
180     }
181 
182     /**
183      * {@inheritDoc}
184      */
185     @Override
186     public int getProcessCount() {
187         List<String> procList = ExecutingCommand.runNative("ps -axo pid");
188         if (!procList.isEmpty()) {
189             // Subtract 1 for header
190             return procList.size() - 1;
191         }
192         return 0;
193     }
194 
195     /**
196      * {@inheritDoc}
197      */
198     @Override
199     public int getThreadCount() {
200         int threads = 0;
201         for (String proc : ExecutingCommand.runNative("ps -axo nlwp")) {
202             threads += ParseUtil.parseIntOrDefault(proc.trim(), 0);
203         }
204         return threads;
205     }

212         return new FreeBsdNetworkParams();
