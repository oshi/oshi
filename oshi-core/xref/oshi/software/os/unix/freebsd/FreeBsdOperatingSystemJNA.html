28  /**
29   * JNA-backed FreeBSD operating system.
30   */
31  @ThreadSafe
32  public class FreeBsdOperatingSystemJNA extends FreeBsdOperatingSystem {
33  
34      @Override
35      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
36          String family = BsdSysctlUtil.sysctl("kern.ostype", "FreeBSD");
37  
38          String version = BsdSysctlUtil.sysctl("kern.osrelease", "");
39          String versionInfo = BsdSysctlUtil.sysctl("kern.version", "");
40          String buildNumber = versionInfo.split(":")[0].replace(family, "").replace(version, "").trim();
41  
42          return new Pair<>(family, new OSVersionInfo(version, null, buildNumber));
43      }
44  
45      @Override
46      public FileSystem getFileSystem() {
47          return new FreeBsdFileSystemJNA();
48      }
49  
50      @Override
51      public InternetProtocolStats getInternetProtocolStats() {
52          return new FreeBsdInternetProtocolStatsJNA();
53      }
54  
55      @Override
56      public NetworkParams getNetworkParams() {
57          return new FreeBsdNetworkParamsJNA();
58      }
59  
60      @Override
61      public List<OSSession> getSessions() {
62          return USE_WHO_COMMAND ? super.getSessions() : Who.queryUtxent();
63      }
64  
65      @Override
66      public int getProcessId() {
67          return FreeBsdLibc.INSTANCE.getpid();
68      }
69  
70      @Override
71      public int getThreadId() {
72          NativeLongByReference pTid = new NativeLongByReference();
73          if (FreeBsdLibc.INSTANCE.thr_self(pTid) < 0) {
74              return 0;
75          }
76          return pTid.getValue().intValue();
77      }
78  
79      @Override
80      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
81          return new FreeBsdOSProcessJNA(pid, psMap, this);
82      }
83  
84      @Override
85      protected long queryBootTime() {
86          Timeval tv = new Timeval();
87          if (!BsdSysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
88              // Usually this works. If it doesn't, fall back to text parsing.
89              // Boot time will be the first consecutive string of digits.
90              return ParseUtil.parseLongOrDefault(
91                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
92                      System.currentTimeMillis() / 1000);
93          }
94          // tv now points to a 128-bit timeval structure for boot time.
95          // First 8 bytes are seconds, second 8 bytes are microseconds (we ignore)
96          return tv.tv_sec;
97      }
98  }
