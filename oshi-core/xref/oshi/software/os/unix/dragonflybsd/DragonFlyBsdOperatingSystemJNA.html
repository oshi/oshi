5   package oshi.software.os.unix.dragonflybsd;
6   
7   import java.util.List;
8   import java.util.Map;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.driver.unix.freebsd.Who;
12  import oshi.jna.platform.unix.DragonFlyBsdLibc;
13  import oshi.jna.platform.unix.FreeBsdLibc;
14  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
15  import oshi.software.common.os.unix.dragonflybsd.DragonFlyBsdOperatingSystem;
16  import oshi.software.os.FileSystem;
17  import oshi.software.os.InternetProtocolStats;
18  import oshi.software.os.NetworkParams;
19  import oshi.software.os.OSProcess;
20  import oshi.software.os.OSSession;
21  import oshi.software.os.unix.freebsd.FreeBsdFileSystemJNA;
22  import oshi.software.os.unix.freebsd.FreeBsdInternetProtocolStatsJNA;
23  import oshi.software.os.unix.freebsd.FreeBsdNetworkParamsJNA;
24  import oshi.util.platform.unix.freebsd.BsdSysctlUtil;
25  
26  /**
27   * JNA-backed DragonFly BSD operating system.
28   */
29  @ThreadSafe
30  public class DragonFlyBsdOperatingSystemJNA extends DragonFlyBsdOperatingSystem {
31  
32      @Override
33      protected String querySysctl(String name, String def) {
34          return BsdSysctlUtil.sysctl(name, def);
35      }
36  
37      @Override
38      public FileSystem getFileSystem() {
39          return new FreeBsdFileSystemJNA();
40      }
41  
42      @Override
43      public InternetProtocolStats getInternetProtocolStats() {
44          return new FreeBsdInternetProtocolStatsJNA();
45      }
46  
47      @Override
48      public NetworkParams getNetworkParams() {
49          return new FreeBsdNetworkParamsJNA();
50      }
51  
52      @Override
53      protected List<OSSession> queryWhoSessions() {
54          return Who.queryUtxent();
55      }
56  
57      @Override
58      public int getProcessId() {
59          return DragonFlyBsdLibc.INSTANCE.getpid();
60      }
61  
62      @Override
63      public int getThreadId() {
64          int tid = DragonFlyBsdLibc.INSTANCE.lwp_gettid();
65          return tid < 0 ? 0 : tid;
66      }
67  
68      @Override
69      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
70          return new DragonFlyBsdOSProcessJNA(pid, psMap, this);
71      }
72  
73      @Override
74      protected long queryKernBoottimeSeconds() {
75          FreeBsdLibc.Timeval tv = new FreeBsdLibc.Timeval();
76          if (!BsdSysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
77              return 0L;
78          }
79          return tv.tv_sec;
80      }
81  }
