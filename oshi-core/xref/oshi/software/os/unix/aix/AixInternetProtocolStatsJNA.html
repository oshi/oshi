5   package oshi.software.os.unix.aix;
6   
7   import static oshi.util.Memoizer.defaultExpiration;
8   import static oshi.util.Memoizer.memoize;
9   
10  import java.util.function.Supplier;
11  
12  import com.sun.jna.Native;
13  import com.sun.jna.platform.unix.aix.Perfstat.perfstat_protocol_t;
14  
15  import oshi.annotation.concurrent.ThreadSafe;
16  import oshi.driver.unix.aix.perfstat.PerfstatProtocolJNA;
17  import oshi.software.common.os.unix.aix.AixInternetProtocolStats;
18  
19  /**
20   * JNA-backed AIX InternetProtocolStats.
21   */
22  @ThreadSafe
23  public final class AixInternetProtocolStatsJNA extends AixInternetProtocolStats {
24  
25      private final Supplier<perfstat_protocol_t[]> ipstats = memoize(PerfstatProtocolJNA::queryProtocols,
26              defaultExpiration());
27  
28      @Override
29      protected TcpStats queryTcpStats() {
30          for (perfstat_protocol_t stat : ipstats.get()) {
31              if ("tcp".equals(Native.toString(stat.name))) {
32                  return new TcpStats(stat.u.tcp.established, stat.u.tcp.initiated, stat.u.tcp.accepted,
33                          stat.u.tcp.dropped, stat.u.tcp.dropped, stat.u.tcp.opackets, stat.u.tcp.ipackets, 0L,
34                          stat.u.tcp.ierrors, 0L);
35              }
36          }
37          return null;
38      }
39  
40      @Override
41      protected UdpStats queryUdpStats() {
42          for (perfstat_protocol_t stat : ipstats.get()) {
43              if ("udp".equals(Native.toString(stat.name))) {
44                  return new UdpStats(stat.u.udp.opackets, stat.u.udp.ipackets, stat.u.udp.no_socket, stat.u.udp.ierrors);
45              }
46          }
47          return null;
48      }
49  }
