24  package oshi.software.os.unix.aix;
25  
26  import static com.sun.jna.platform.unix.LibCAPI.HOST_NAME_MAX; // NOSONAR squid:S1191
27  
28  import com.sun.jna.Native;
29  
30  import oshi.annotation.concurrent.ThreadSafe;
31  import oshi.jna.platform.unix.AixLibc;
32  import oshi.software.common.AbstractNetworkParams;
33  import oshi.util.Constants;
34  import oshi.util.ExecutingCommand;
35  import oshi.util.ParseUtil;
36  
37  /**
38   * AixNetworkParams class.
39   */
40  @ThreadSafe
41  final class AixNetworkParams extends AbstractNetworkParams {
42  
43      private static final AixLibc LIBC = AixLibc.INSTANCE;
44  
45      @Override
46      public String getHostName() {
47          byte[] hostnameBuffer = new byte[HOST_NAME_MAX + 1];
48          if (0 != LIBC.gethostname(hostnameBuffer, hostnameBuffer.length)) {
49              return super.getHostName();
50          }
51          return Native.toString(hostnameBuffer);
52      }
53  
54      @Override
55      public String getIpv4DefaultGateway() {
56          return getDefaultGateway("netstat -rnf inet");
57      }
58  
59      @Override
60      public String getIpv6DefaultGateway() {
61          return getDefaultGateway("netstat -rnf inet6");
62      }
63  
64      private static String getDefaultGateway(String netstat) {
65          /*-
66          $ netstat -rnf inet
67          Routing tables
68          Destination        Gateway           Flags   Refs     Use  If   Exp  Groups
69          *
70          Route Tree for Protocol Family 2 (Internet):
71          default            192.168.10.1      UG        9    873816 en0      -      -
72          127/8              127.0.0.1         U         9    839480 lo0      -      -
73          192.168.10.0       192.168.10.80     UHSb      0         0 en0      -      -   =>
74          192.168.10/24      192.168.10.80     U         3    394820 en0      -      -
75          192.168.10.80      127.0.0.1         UGHS      0         7 lo0      -      -
76          192.168.10.255     192.168.10.80     UHSb      2      7466 en0      -      -
77          */
78          for (String line : ExecutingCommand.runNative(netstat)) {
79              String[] split = ParseUtil.whitespaces.split(line);
80              if (split.length > 7 && "default".equals(split[0])) {
81                  return split[1];
82              }
83          }
84          return Constants.UNKNOWN;
85      }
86  }
