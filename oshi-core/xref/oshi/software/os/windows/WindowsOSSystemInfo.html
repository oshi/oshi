24  package oshi.software.os.windows;
25  
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import com.sun.jna.platform.win32.Kernel32; // NOSONAR squid:S1191
30  import com.sun.jna.platform.win32.WinBase.SYSTEM_INFO;
31  import com.sun.jna.platform.win32.WinNT.HANDLE;
32  import com.sun.jna.ptr.IntByReference;
33  
34  /**
35   * Windows OS native system information.
36   */
37  public class WindowsOSSystemInfo {
38      private static final Logger LOG = LoggerFactory.getLogger(WindowsOSSystemInfo.class);
39  
40      // Populated during call to init
41      private SYSTEM_INFO systemInfo = null;
42  
43      /**
44       * Constructor for WindowsOSSystemInfo.
45       */
46      public WindowsOSSystemInfo() {
47          init();
48      }
49  
50      /**
51       * Constructor for WindowsOSSystemInfo.
52       *
53       * @param si
54       *            a {@link com.sun.jna.platform.win32.WinBase.SYSTEM_INFO} object.
55       */
56      public WindowsOSSystemInfo(SYSTEM_INFO si) {
57          this.systemInfo = si;
58      }
59  
60      private void init() {
61          SYSTEM_INFO si = new SYSTEM_INFO();
62          Kernel32.INSTANCE.GetSystemInfo(si);
63  
64          try {
65              IntByReference isWow64 = new IntByReference();
66              // This returns a pseudo handle, currently (HANDLE)-1, that is
67              // interpreted as the current process handle. The pseudo handle need
68              // not be closed when it is no longer needed. Calling the
69              // CloseHandle function with a pseudo handle has no effect.
70              HANDLE hProcess = Kernel32.INSTANCE.GetCurrentProcess();
71              if (Kernel32.INSTANCE.IsWow64Process(hProcess, isWow64) && isWow64.getValue() > 0) {
72                  // Populates the class variable with information
73                  Kernel32.INSTANCE.GetNativeSystemInfo(si);
74              }
75          } catch (UnsatisfiedLinkError e) {
76              // no WOW64 support
77              LOG.trace("No WOW64 support: {}", e.getMessage());
78          }
79  
80          this.systemInfo = si;
81          LOG.debug("Initialized OSNativeSystemInfo");
82      }
83  
84      /**
85       * Number of processors.
86       *
87       * @return Integer.
88       */
89      public int getNumberOfProcessors() {
90          return this.systemInfo.dwNumberOfProcessors.intValue();
91      }
92  }
