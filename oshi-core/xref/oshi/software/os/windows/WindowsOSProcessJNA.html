70      @Override
71      public long getAffinityMask() {
72          final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
73          if (pHandle != null) {
74              try (CloseableULONGptrByReference processAffinity = new CloseableULONGptrByReference();
75                      CloseableULONGptrByReference systemAffinity = new CloseableULONGptrByReference()) {
76                  if (Kernel32.INSTANCE.GetProcessAffinityMask(pHandle, processAffinity, systemAffinity)) {
77                      return Pointer.nativeValue(processAffinity.getValue().toPointer());
78                  }
79              } finally {
80                  Kernel32.INSTANCE.CloseHandle(pHandle);
81              }
82          }
83          return 0L;
84      }

86      @Override
87      public boolean updateAttributes() {
88          Set<Integer> pids = Collections.singleton(this.getProcessID());
89          // Get data from the registry if possible
90          Map<Integer, ProcessPerfCounterBlock> pcbMap = ProcessPerformanceDataJNA.buildProcessMapFromRegistry(pids);
91          // otherwise performance counters with WMI backup
92          if (pcbMap == null || pcbMap.isEmpty()) {
93              pcbMap = ProcessPerformanceDataJNA.buildProcessMapFromPerfCounters(pids);
94          }
95          ProcessPerfCounterBlock pcb = pcbMap == null ? null : pcbMap.get(this.getProcessID());
96          if (USE_PROCSTATE_SUSPENDED) {
97              // Populate name from pcb before querying threads, since the fallback path uses getName()
98              if (pcb != null) {
99                  setName(pcb.getName());
100             }
101             setTcb(queryMatchingThreads(pids));
102         }
103         Map<Integer, WtsInfo> wts = ProcessWtsData.queryProcessWtsMap(pids);
104         return updateAttributes(pcb, wts == null ? null : wts.get(this.getProcessID()));
105     }

107     @Override
108     protected boolean updateAttributes(ProcessPerfCounterBlock pcb, WtsInfo wts) {
109         if (!super.updateAttributes(pcb, wts)) {
110             return false;
111         }
112 
113         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
114         if (pHandle != null) {
115             try {
116                 // Test for 32-bit process on 64-bit windows
117                 if (IS_VISTA_OR_GREATER && getBitness() == 64) {
118                     try (CloseableIntByReference wow64 = new CloseableIntByReference()) {
119                         if (Kernel32.INSTANCE.IsWow64Process(pHandle, wow64) && wow64.getValue() > 0) {
120                             setBitness(32);
121                         }
122                     }
123                 }
124                 try { // EXECUTABLEPATH
125                     if (IS_WINDOWS7_OR_GREATER) {
126                         setPath(Kernel32Util.QueryFullProcessImageName(pHandle, 0));
127                     }
128                 } catch (Win32Exception e) {
129                     // Best-effort path lookup; leave process state and path untouched on failure
130                 }
131             } finally {
132                 Kernel32.INSTANCE.CloseHandle(pHandle);
133             }
134         }
135 
136         return !getState().equals(State.INVALID);
137     }

144     @Override
145     protected Map<Integer, ThreadPerfCounterBlock> queryMatchingThreads(Set<Integer> pids) {
146         Map<Integer, ThreadPerfCounterBlock> threads = ThreadPerformanceDataJNA.buildThreadMapFromRegistry(pids);
147         if (threads == null || threads.isEmpty()) {
148             threads = ThreadPerformanceDataJNA.buildThreadMapFromPerfCounters(pids, this.getName(), -1);
149         }
150         return threads;
151     }

153     @Override
154     protected String queryCommandLine() {
155         String cwdCmd = getCwdCmdEnv().getB();
156         if (!cwdCmd.isEmpty()) {
157             return cwdCmd;
158         }
159         if (USE_BATCH_COMMANDLINE) {
160             return Win32ProcessCachedJNA.getInstance().getCommandLine(getProcessID(), getStartTime());
161         }
162         WmiResult<CommandLineProperty> commandLineProcs = Win32ProcessJNA
163                 .queryCommandLines(Collections.singleton(getProcessID()));
164         if (commandLineProcs.getResultCount() > 0) {
165             return WmiUtil.getString(commandLineProcs, CommandLineProperty.COMMANDLINE, 0);
166         }
167         return "";
168     }

189     /**
190      * Opens this process's access token and resolves an account (name and SID) from it. Shared by the user- and
191      * group-info queries, which differ only in how the account is extracted from the token.
192      *
193      * @param accountFromToken extracts the {@link Account} from the process token (e.g. token user or primary group)
194      * @param infoType         a short label ({@code "user"}/{@code "group"}) for log messages
195      * @return a pair of (account name, SID string), or {@link #defaultPair()} if the token can't be read
196      */
197     private Pair<String, String> queryTokenAccount(Function<HANDLE, Account> accountFromToken, String infoType) {
198         Pair<String, String> pair = null;
199         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
200         if (pHandle != null) {
201             try {
202                 try (CloseableHANDLEByReference phToken = new CloseableHANDLEByReference()) {
203                     try {
204                         if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_QUERY, phToken)) {
205                             Account account = accountFromToken.apply(phToken.getValue());
206                             pair = new Pair<>(account.name, account.sidString);
207                         } else {
208                             int error = Kernel32.INSTANCE.GetLastError();
209                             // Access denied errors are common. Fail silently.
210                             if (error != WinError.ERROR_ACCESS_DENIED) {
211                                 LOG.error("Failed to get process token for process {}: {}", getProcessID(), error);
212                             }
213                         }
214                     } catch (Win32Exception e) {
215                         LOG.warn("Failed to query {} info for process {} ({}): {}", infoType, getProcessID(), getName(),
216                                 e.getMessage());
217                     } finally {
218                         final HANDLE token = phToken.getValue();
219                         if (token != null) {
220                             Kernel32.INSTANCE.CloseHandle(token);
221                         }
222                     }
223                 }
224             } finally {
225                 // Close pHandle even if constructing the CloseableHANDLEByReference above fails
226                 Kernel32.INSTANCE.CloseHandle(pHandle);
227             }
228         }
229         return pair != null ? pair : defaultPair();
230     }

232     @Override
233     protected Triplet<String, String, Map<String, String>> queryCwdCommandlineEnvironment() {
234         HANDLE h = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION | WinNT.PROCESS_VM_READ, false,
235                 getProcessID());
236         if (h != null) {
237             try {
238                 // Can't check 32-bit procs from a 64-bit one
239                 if (WindowsOperatingSystemJNA.isX86() == WindowsOperatingSystemJNA.isWow(h)) {
240                     try (CloseableIntByReference nRead = new CloseableIntByReference()) {
241                         // Start by getting the address of the PEB
242                         NtDll.PROCESS_BASIC_INFORMATION pbi = new NtDll.PROCESS_BASIC_INFORMATION();
243                         int ret = NtDll.INSTANCE.NtQueryInformationProcess(h, NtDll.PROCESS_BASIC_INFORMATION,
244                                 pbi.getPointer(), pbi.size(), nRead);
245                         if (ret != 0) {
246                             return defaultCwdCommandlineEnvironment();
247                         }
248                         pbi.read();
249 
250                         // Now fetch the PEB
251                         NtDll.PEB peb = new NtDll.PEB();
252                         Kernel32.INSTANCE.ReadProcessMemory(h, pbi.PebBaseAddress, peb.getPointer(), peb.size(), nRead);
253                         if (nRead.getValue() == 0) {
254                             return defaultCwdCommandlineEnvironment();
255                         }
256                         peb.read();
257 
258                         // Now fetch the Process Parameters structure containing our data
259                         NtDll.RTL_USER_PROCESS_PARAMETERS upp = new NtDll.RTL_USER_PROCESS_PARAMETERS();
260                         Kernel32.INSTANCE.ReadProcessMemory(h, peb.ProcessParameters, upp.getPointer(), upp.size(),
261                                 nRead);
262                         if (nRead.getValue() == 0) {
263                             return defaultCwdCommandlineEnvironment();
264                         }
265                         upp.read();
266 
267                         // Get CWD and Command Line strings here
268                         String cwd = readUnicodeString(h, upp.CurrentDirectory.DosPath);
269                         String cl = readUnicodeString(h, upp.CommandLine);
270 
271                         // Fetch the Environment Strings
272                         int envSize = upp.EnvironmentSize.intValue();
273                         if (envSize > 0) {
274                             try (Memory buffer = new Memory(envSize)) {
275                                 Kernel32.INSTANCE.ReadProcessMemory(h, upp.Environment, buffer, envSize, nRead);
276                                 if (nRead.getValue() > 0) {
277                                     char[] env = buffer.getCharArray(0, envSize / 2);
278                                     Map<String, String> envMap = ParseUtil.parseCharArrayToStringMap(env);
279                                     // First entry in Environment is "=::=::\"
280                                     envMap.remove("");
281                                     return new Triplet<>(cwd, cl, Collections.unmodifiableMap(envMap));
282                                 }
283                             }
284                         }
285                         return new Triplet<>(cwd, cl, Collections.emptyMap());
286                     }
287                 }
288             } finally {
289                 Kernel32.INSTANCE.CloseHandle(h);
290             }
291         }
292         return defaultCwdCommandlineEnvironment();
293     }

295     private static String readUnicodeString(HANDLE h, UNICODE_STRING s) {
296         if (s.Length > 0) {
297             // Add space for null terminator
298             try (Memory m = new Memory(s.Length + 2L); CloseableIntByReference nRead = new CloseableIntByReference()) {
299                 m.clear(); // really only need null in last 2 bytes but this is easier
300                 Kernel32.INSTANCE.ReadProcessMemory(h, s.Buffer, m, s.Length, nRead);
301                 if (nRead.getValue() > 0) {
302                     return m.getWideString(0);
303                 }
304             }
305         }
306         return "";
307     }
