126     private static final long MAX_WINDOWS_HANDLES;
127     static {
128         // Determine whether 32-bit or 64-bit handle limit, although both are
129         // essentially infinite for practical purposes. See
130         // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
131         if (System.getenv("ProgramFiles(x86)") == null) {
132             MAX_WINDOWS_HANDLES = 16_777_216L - 32_768L;
133         } else {
134             MAX_WINDOWS_HANDLES = 16_777_216L - 65_536L;
135         }

148     @Override
149     public OSFileStore[] getFileStores(boolean localOnly) {
150         // Create list to hold results
151         ArrayList<OSFileStore> result;
152 
153         // Begin with all the local volumes
154         result = getLocalVolumes(null);
155 
156         // Build a map of existing mount point to OSFileStore
157         Map<String, OSFileStore> volumeMap = new HashMap<>();
158         for (OSFileStore volume : result) {
159             volumeMap.put(volume.getMount(), volume);
160         }
161 
162         // Iterate through volumes in WMI and update description (if it exists)
163         // or add new if it doesn't (expected for network drives)
164         for (OSFileStore wmiVolume : getWmiVolumes(null, localOnly)) {
165             if (volumeMap.containsKey(wmiVolume.getMount())) {
166                 // If the volume is already in our list, update the name field
167                 // using WMI's more verbose name
168                 volumeMap.get(wmiVolume.getMount()).setName(wmiVolume.getName());
169             } else if (!localOnly) {
170                 // Otherwise add the new volume in its entirety
171                 result.add(wmiVolume);
172             }
173         }
174         return result.toArray(new OSFileStore[0]);
175     }

177     /**
178      * Private method for getting all mounted local drives.
179      *
180      * @param nameToMatch
181      *            an optional string to filter match, null otherwise
182      * @return A list of {@link OSFileStore} objects representing all local mounted
183      *         volumes
184      */
185     private static ArrayList<OSFileStore> getLocalVolumes(String nameToMatch) {
186         ArrayList<OSFileStore> fs;
187         String volume;
188         String strFsType;
189         String strName;
190         String strMount;
191         WinNT.HANDLE hVol;
192         WinNT.LARGE_INTEGER userFreeBytes;
193         WinNT.LARGE_INTEGER totalBytes;
194         WinNT.LARGE_INTEGER systemFreeBytes;
195         boolean retVal;
196         char[] aVolume;
197         char[] fstype;
198         char[] name;
199         char[] mount;
200         IntByReference pFlags;
201 
202         fs = new ArrayList<>();
203         aVolume = new char[BUFSIZE];
204 
205         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
206         if (hVol == WinBase.INVALID_HANDLE_VALUE) {
207             return fs;
208         }
209 
210         while (true) {
211             fstype = new char[16];
212             name = new char[BUFSIZE];
213             mount = new char[BUFSIZE];
214             pFlags = new IntByReference();
215 
216             userFreeBytes = new WinNT.LARGE_INTEGER(0L);
217             totalBytes = new WinNT.LARGE_INTEGER(0L);
218             systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
219 
220             volume = new String(aVolume).trim();
221             Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, pFlags, fstype, 16);
222             final int flags = pFlags.getValue();
223             Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
224 
225             strMount = new String(mount).trim();
226             if (!strMount.isEmpty()) {
227                 strName = new String(name).trim();
228                 strFsType = new String(fstype).trim();
229 
230                 StringBuilder options = new StringBuilder((FILE_READ_ONLY_VOLUME & flags) == 0 ? "rw" : "ro");
231                 String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
232                         .map(Map.Entry::getValue).collect(Collectors.joining(","));
233                 if (!moreOptions.isEmpty()) {
234                     options.append(',').append(moreOptions);
235                 }
236                 String osName = String.format("%s (%s)", strName, strMount);
237                 if (nameToMatch == null || nameToMatch.equals(osName)) {
238                     Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
239                     // Parse uuid from volume name
240                     String uuid = ParseUtil.parseUuidOrDefault(volume, "");
241 
242                     // Volume is mounted
243                     OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
244                     osStore.setName(osName);
245                     osStore.setVolume(volume);
246                     osStore.setMount(strMount);
247                     osStore.setDescription(getDriveType(strMount));
248                     osStore.setType(strFsType);
249                     osStore.setOptions(options.toString());
250                     osStore.setUUID(uuid);
251                     osStore.setFreeSpace(systemFreeBytes.getValue());
252                     osStore.setUsableSpace(userFreeBytes.getValue());
253                     osStore.setTotalSpace(totalBytes.getValue());
254                     System.out.println(osStore.toString());
255                     fs.add(osStore);
256                 }
257             }
258             retVal = Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE);
259             if (!retVal) {
260                 Kernel32.INSTANCE.FindVolumeClose(hVol);
261                 break;
262             }
263         }
264 
265         return fs;
266     }

268     /**
269      * Private method for getting logical drives listed in WMI.
270      *
271      * @param nameToMatch
272      *            an optional string to filter match, null otherwise
273      * @param localOnly
274      *            Whether to only search local drives
275      * @return A list of {@link OSFileStore} objects representing all network
276      *         mounted volumes
277      */
278     private static List<OSFileStore> getWmiVolumes(String nameToMatch, boolean localOnly) {
279         long free;
280         long total;
281         List<OSFileStore> fs = new ArrayList<>();
282 
283         StringBuilder wmiClassName = new StringBuilder("Win32_LogicalDisk");
284         boolean where = false;
285         if (localOnly) {
286             wmiClassName.append(" WHERE DriveType != 4");
287             where = true;
288         }
289         if (nameToMatch != null) {
290             wmiClassName.append(where ? " WHERE" : " AND").append(" Name=\"").append(nameToMatch).append('\"');
291         }
292         WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
293         WmiQuery<LogicalDiskProperty> logicalDiskQuery = new WmiQuery<>(wmiClassName.toString(),
294                 LogicalDiskProperty.class);
295         WmiResult<LogicalDiskProperty> drives = wmiQueryHandler.queryWMI(logicalDiskQuery);
296 
297         for (int i = 0; i < drives.getResultCount(); i++) {
298             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
299             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
300             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
301             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
302             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
303             String volume;
304             if (type != 4) {
305                 char[] chrVolume = new char[BUFSIZE];
306                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
307                 volume = new String(chrVolume).trim();
308             } else {
309                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
310                 String[] split = volume.split("\\\\");
311                 if (split.length > 1 && split[split.length - 1].length() > 0) {
312                     description = split[split.length - 1];
313                 }
314             }
315 
316             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
317             osStore.setName(String.format("%s (%s)", description, name));
318             osStore.setVolume(volume);
319             osStore.setMount(name + "\\");
320             osStore.setDescription(getDriveType(name));
321             osStore.setType(WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i));
322             osStore.setUUID("");
323             osStore.setFreeSpace(free); // no separate field, assume same
324             osStore.setUsableSpace(free);
325             osStore.setTotalSpace(total);
326             fs.add(osStore);
327         }
328 
329         return fs;
330     }

332     /**
333      * Private method for getting mounted drive type.
334      *
335      * @param drive
336      *            Mounted drive
337      * @return A drive type description
338      */
339     private static String getDriveType(String drive) {
340         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
341         case 2:
342             return "Removable drive";
343         case 3:
344             return "Fixed drive";
345         case 4:
346             return "Network drive";
347         case 5:
348             return "CD-ROM";
349         case 6:
350             return "RAM drive";
351         default:
352             return "Unknown drive type";
353         }
354     }

356     /** {@inheritDoc} */
357     @Override
358     public long getOpenFileDescriptors() {
359         Map<HandleCountProperty, List<Long>> valueListMap = this.handlePerfCounters.queryValuesWildcard();
360         List<Long> valueList = valueListMap.get(HandleCountProperty.HANDLECOUNT);
361         long descriptors = 0L;
362         if (valueList != null) {
363             for (int i = 0; i < valueList.size(); i++) {
364                 descriptors += valueList.get(i);
365             }
366         }
367         return descriptors;
368     }

376     /**
377      * <p>
378      * updateFileStoreStats.
379      * </p>
380      *
381      * @param osFileStore
382      *            a {@link oshi.software.os.OSFileStore} object.
383      * @return a boolean.
384      */
385     public static boolean updateFileStoreStats(OSFileStore osFileStore) {
386         // Check if we have the volume locally
387         List<OSFileStore> volumes = getLocalVolumes(osFileStore.getName());
388         if (volumes.isEmpty()) {
389             // Not locally, search WMI
390             volumes = getWmiVolumes(osFileStore.getName(), false);
391         }
392         for (OSFileStore fileStore : volumes) {
393             if (osFileStore.getVolume().equals(fileStore.getVolume())
394                     && osFileStore.getMount().equals(fileStore.getMount())) {
395                 osFileStore.setLogicalVolume(fileStore.getLogicalVolume());
396                 osFileStore.setDescription(fileStore.getDescription());
397                 osFileStore.setType(fileStore.getType());
398                 osFileStore.setFreeSpace(fileStore.getFreeSpace());
399                 osFileStore.setUsableSpace(fileStore.getUsableSpace());
400                 osFileStore.setTotalSpace(fileStore.getTotalSpace());
401                 osFileStore.setFreeInodes(fileStore.getFreeInodes());
402                 osFileStore.setTotalInodes(fileStore.getTotalInodes());
403                 return true;
404             }
405         }
406         return false;
407     }
