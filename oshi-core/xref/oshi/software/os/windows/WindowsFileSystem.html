95      private static final long MAX_WINDOWS_HANDLES;
96      static {
97          // Determine whether 32-bit or 64-bit handle limit, although both are
98          // essentially infinite for practical purposes. See
99          // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
100         if (System.getenv("ProgramFiles(x86)") == null) {
101             MAX_WINDOWS_HANDLES = 16_777_216L - 32_768L;
102         } else {
103             MAX_WINDOWS_HANDLES = 16_777_216L - 65_536L;
104         }

112     /**
113      * Gets File System Information.
114      *
115      * @return An array of {@link OSFileStore} objects representing mounted volumes.
116      *         May return disconnected volumes with
117      *         {@link OSFileStore#getTotalSpace()} = 0.
118      */
119     @Override
120     public OSFileStore[] getFileStores() {
121         // Create list to hold results
122         ArrayList<OSFileStore> result;
123 
124         // Begin with all the local volumes
125         result = getLocalVolumes(null);
126 
127         // Build a map of existing mount point to OSFileStore
128         Map<String, OSFileStore> volumeMap = new HashMap<>();
129         for (OSFileStore volume : result) {
130             volumeMap.put(volume.getMount(), volume);
131         }
132 
133         // Iterate through volumes in WMI and update description (if it exists)
134         // or add new if it doesn't (expected for network drives)
135         for (OSFileStore wmiVolume : getWmiVolumes(null)) {
136             if (volumeMap.containsKey(wmiVolume.getMount())) {
137                 // If the volume is already in our list, update the name field
138                 // using WMI's more verbose name
139                 volumeMap.get(wmiVolume.getMount()).setName(wmiVolume.getName());
140             } else {
141                 // Otherwise add the new volume in its entirety
142                 result.add(wmiVolume);
143             }
144         }
145         return result.toArray(new OSFileStore[0]);
146     }

148     /**
149      * Private method for getting all mounted local drives.
150      * 
151      * @param nameToMatch
152      *            an optional string to filter match, null otherwise
153      * @return A list of {@link OSFileStore} objects representing all local mounted
154      *         volumes
155      */
156     private ArrayList<OSFileStore> getLocalVolumes(String nameToMatch) {
157         ArrayList<OSFileStore> fs;
158         String volume;
159         String strFsType;
160         String strName;
161         String strMount;
162         WinNT.HANDLE hVol;
163         WinNT.LARGE_INTEGER userFreeBytes;
164         WinNT.LARGE_INTEGER totalBytes;
165         WinNT.LARGE_INTEGER systemFreeBytes;
166         boolean retVal;
167         char[] aVolume;
168         char[] fstype;
169         char[] name;
170         char[] mount;
171 
172         fs = new ArrayList<>();
173         aVolume = new char[BUFSIZE];
174 
175         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
176         if (hVol == WinNT.INVALID_HANDLE_VALUE) {
177             return fs;
178         }
179 
180         while (true) {
181             fstype = new char[16];
182             name = new char[BUFSIZE];
183             mount = new char[BUFSIZE];
184 
185             userFreeBytes = new WinNT.LARGE_INTEGER(0L);
186             totalBytes = new WinNT.LARGE_INTEGER(0L);
187             systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
188 
189             volume = new String(aVolume).trim();
190             Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, null, fstype, 16);
191             Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
192 
193             strMount = new String(mount).trim();
194             strName = new String(name).trim();
195             strFsType = new String(fstype).trim();
196             String osName = String.format("%s (%s)", strName, strMount);
197             if (nameToMatch == null || nameToMatch.equals(osName)) {
198                 Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
199                 // Parse uuid from volume name
200                 String uuid = ParseUtil.parseUuidOrDefault(volume, "");
201 
202                 if (!strMount.isEmpty()) {
203                     // Volume is mounted
204                     OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
205                     osStore.setName(osName);
206                     osStore.setVolume(volume);
207                     osStore.setMount(strMount);
208                     osStore.setDescription(getDriveType(strMount));
209                     osStore.setType(strFsType);
210                     osStore.setUUID(uuid);
211                     osStore.setFreeSpace(systemFreeBytes.getValue());
212                     osStore.setUsableSpace(userFreeBytes.getValue());
213                     osStore.setTotalSpace(totalBytes.getValue());
214                     fs.add(osStore);
215                 }
216             }
217             retVal = Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE);
218             if (!retVal) {
219                 Kernel32.INSTANCE.FindVolumeClose(hVol);
220                 break;
221             }
222         }
223 
224         return fs;
225     }

227     /**
228      * Private method for getting logical drives listed in WMI.
229      * 
230      * @param nameToMatch
231      *            an optional string to filter match, null otherwise
232      * @return A list of {@link OSFileStore} objects representing all network
233      *         mounted volumes
234      */
235     private List<OSFileStore> getWmiVolumes(String nameToMatch) {
236         long free;
237         long total;
238         List<OSFileStore> fs = new ArrayList<>();
239 
240         String wmiClassName = this.logicalDiskQuery.getWmiClassName();
241         if (nameToMatch != null) {
242             this.logicalDiskQuery.setWmiClassName(wmiClassName + " WHERE Name=\"" + nameToMatch + "\"");
243         }
244         WmiResult<LogicalDiskProperty> drives = wmiQueryHandler.queryWMI(this.logicalDiskQuery);
245         if (nameToMatch != null) {
246             this.logicalDiskQuery.setWmiClassName(wmiClassName);
247         }
248 
249         for (int i = 0; i < drives.getResultCount(); i++) {
250             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
251             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
252             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
253             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
254             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
255             String volume;
256             if (type != 4) {
257                 char[] chrVolume = new char[BUFSIZE];
258                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
259                 volume = new String(chrVolume).trim();
260             } else {
261                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
262                 String[] split = volume.split("\\\\");
263                 if (split.length > 1 && split[split.length - 1].length() > 0) {
264                     description = split[split.length - 1];
265                 }
266             }
267 
268             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
269             osStore.setName(String.format("%s (%s)", description, name));
270             osStore.setVolume(volume);
271             osStore.setMount(name + "\\");
272             osStore.setDescription(getDriveType(name));
273             osStore.setType(WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i));
274             osStore.setUUID("");
275             osStore.setFreeSpace(free); // no separate field, assume same
276             osStore.setUsableSpace(free);
277             osStore.setTotalSpace(total);
278             fs.add(osStore);
279         }
280 
281         return fs;
282     }

284     /**
285      * Private method for getting mounted drive type.
286      *
287      * @param drive
288      *            Mounted drive
289      * @return A drive type description
290      */
291     private String getDriveType(String drive) {
292         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
293         case 2:
294             return "Removable drive";
295         case 3:
296             return "Fixed drive";
297         case 4:
298             return "Network drive";
299         case 5:
300             return "CD-ROM";
301         case 6:
302             return "RAM drive";
303         default:
304             return "Unknown drive type";
305         }
306     }

308     @Override
309     public long getOpenFileDescriptors() {
310         Map<HandleCountProperty, List<Long>> valueListMap = this.handlePerfCounters.queryValuesWildcard();
311         List<Long> valueList = valueListMap.get(HandleCountProperty.HANDLECOUNT);
312         long descriptors = 0L;
313         if (valueList != null) {
314             for (int i = 0; i < valueList.size(); i++) {
315                 descriptors += valueList.get(i);
316             }
317         }
318         return descriptors;
319     }

326     public static boolean updateFileStoreStats(OSFileStore osFileStore) {
327         WindowsFileSystemSystem.html#WindowsFileSystem">WindowsFileSystem wfs = new WindowsFileSystem();
328         // Check if we have the volume locally
329         List<OSFileStore> volumes = wfs.getLocalVolumes(osFileStore.getName());
330         if (volumes.isEmpty()) {
331             // Not locally, search WMI
332             volumes = wfs.getWmiVolumes(osFileStore.getName());
333         }
334         for (OSFileStore fileStore : volumes) {
335             if (osFileStore.getVolume().equals(fileStore.getVolume())
336                     && osFileStore.getMount().equals(fileStore.getMount())) {
337                 osFileStore.setLogicalVolume(fileStore.getLogicalVolume());
338                 osFileStore.setDescription(fileStore.getDescription());
339                 osFileStore.setType(fileStore.getType());
340                 osFileStore.setFreeSpace(fileStore.getFreeSpace());
341                 osFileStore.setUsableSpace(fileStore.getUsableSpace());
342                 osFileStore.setTotalSpace(fileStore.getTotalSpace());
343                 osFileStore.setFreeInodes(fileStore.getFreeInodes());
344                 osFileStore.setTotalInodes(fileStore.getTotalInodes());
345                 return true;
346             }
347         }
348         return false;
