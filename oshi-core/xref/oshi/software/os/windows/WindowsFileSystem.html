44  /**
45   * The Windows File System contains {@link OSFileStore}s which are a storage
46   * pool, device, partition, volume, concrete file system or other implementation
47   * specific means of file storage. In Windows, these are represented by a drive
48   * letter, e.g., "A:\" and "C:\"
49   *
50   * @author enrico[dot]bianchi[at]gmail[dot]com
51   */

56      private static final int BUFSIZE = 255;

58      private static final int SEM_FAILCRITICALERRORS = 0x0001;

60      enum LogicalDiskProperty {
61          DESCRIPTION, DRIVETYPE, FILESYSTEM, FREESPACE, NAME, PROVIDERNAME, SIZE;
62      }

64      private final transient WmiQuery<LogicalDiskProperty> LOGICAL_DISK_QUERY = new WmiQuery<>("Win32_LogicalDisk",
65              LogicalDiskProperty.class);

67      /*
68       * For handle counts
69       */
70      enum HandleCountProperty {
71          HANDLECOUNT;
72      }
73  
74      // Only one of these will be used
75      private transient PerfCounter handleCountCounter = null;
76      private transient WmiQuery<HandleCountProperty> handleCountQuery = null;
77  
78      private static final long MAX_WINDOWS_HANDLES;
79      static {
80          // Determine whether 32-bit or 64-bit handle limit, although both are
81          // essentially infinite for practical purposes. See
82          // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
83          if (System.getenv("ProgramFiles(x86)") == null) {
84              MAX_WINDOWS_HANDLES = 16_777_216L - 32_768L;
85          } else {
86              MAX_WINDOWS_HANDLES = 16_777_216L - 65_536L;
87          }
88      }

90      public WindowsFileSystem() {
91          // Set error mode to fail rather than prompt for FLoppy/CD-Rom
92          Kernel32.INSTANCE.SetErrorMode(SEM_FAILCRITICALERRORS);
93          initPdhCounters();
94      }
95  
96      private void initPdhCounters() {
97          this.handleCountCounter = PerfDataUtil.createCounter("Process", "_Total", "Handle Count");
98          if (!PerfDataUtil.addCounterToQuery(this.handleCountCounter)) {
99              this.handleCountCounter = null;
100             this.handleCountQuery = new WmiQuery<>("Win32_Process", HandleCountProperty.class);
101         }
102     }

104     /**
105      * Gets File System Information.
106      *
107      * @return An array of {@link OSFileStore} objects representing mounted
108      *         volumes. May return disconnected volumes with
109      *         {@link OSFileStore#getTotalSpace()} = 0.
110      */
111     @Override
112     public OSFileStore[] getFileStores() {
113         // Create list to hold results
114         ArrayList<OSFileStore> result;
115 
116         // Begin with all the local volumes
117         result = getLocalVolumes();
118 
119         // Build a map of existing mount point to OSFileStore
120         Map<String, OSFileStore> volumeMap = new HashMap<>();
121         for (OSFileStore volume : result) {
122             volumeMap.put(volume.getMount(), volume);
123         }
124 
125         // Iterate through volumes in WMI and update description (if it exists)
126         // or add new if it doesn't (expected for network drives)
127         for (OSFileStore wmiVolume : getWmiVolumes()) {
128             if (volumeMap.containsKey(wmiVolume.getMount())) {
129                 // If the volume is already in our list, update the name field
130                 // using WMI's more verbose name
131                 volumeMap.get(wmiVolume.getMount()).setName(wmiVolume.getName());
132             } else {
133                 // Otherwise add the new volume in its entirety
134                 result.add(wmiVolume);
135             }
136         }
137         return result.toArray(new OSFileStore[result.size()]);
138     }

140     /**
141      * Private method for getting all mounted local drives.
142      *
143      * @return A list of {@link OSFileStore} objects representing all local
144      *         mounted volumes
145      */
146     private ArrayList<OSFileStore> getLocalVolumes() {
147         ArrayList<OSFileStore> fs;
148         String volume;
149         String strFsType;
150         String strName;
151         String strMount;
152         WinNT.HANDLE hVol;
153         WinNT.LARGE_INTEGER userFreeBytes;
154         WinNT.LARGE_INTEGER totalBytes;
155         WinNT.LARGE_INTEGER systemFreeBytes;
156         boolean retVal;
157         char[] aVolume;
158         char[] fstype;
159         char[] name;
160         char[] mount;
161 
162         fs = new ArrayList<>();
163         aVolume = new char[BUFSIZE];
164 
165         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
166         if (hVol == WinNT.INVALID_HANDLE_VALUE) {
167             return fs;
168         }
169 
170         while (true) {
171             fstype = new char[16];
172             name = new char[BUFSIZE];
173             mount = new char[BUFSIZE];
174 
175             userFreeBytes = new WinNT.LARGE_INTEGER(0L);
176             totalBytes = new WinNT.LARGE_INTEGER(0L);
177             systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
178 
179             volume = new String(aVolume).trim();
180             Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, null, fstype, 16);
181             Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
182             Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
183 
184             strMount = new String(mount).trim();
185             strName = new String(name).trim();
186             strFsType = new String(fstype).trim();
187             // Parse uuid from volume name
188             String uuid = ParseUtil.parseUuidOrDefault(volume, "");
189 
190             if (!strMount.isEmpty()) {
191                 // Volume is mounted
192                 OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
193                 osStore.setName(String.format("%s (%s)", strName, strMount));
194                 osStore.setVolume(volume);
195                 osStore.setMount(strMount);
196                 osStore.setDescription(getDriveType(strMount));
197                 osStore.setType(strFsType);
198                 osStore.setUUID(uuid);
199                 osStore.setUsableSpace(systemFreeBytes.getValue());
200                 osStore.setTotalSpace(totalBytes.getValue());
201                 fs.add(osStore);
202             }
203             retVal = Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE);
204             if (!retVal) {
205                 Kernel32.INSTANCE.FindVolumeClose(hVol);
206                 break;
207             }
208         }
209 
210         return fs;
211     }

213     /**
214      * Private method for getting logical drives listed in WMI.
215      *
216      * @return A list of {@link OSFileStore} objects representing all network
217      *         mounted volumes
218      */
219     private List<OSFileStore> getWmiVolumes() {
220         long free;
221         long total;
222         List<OSFileStore> fs = new ArrayList<>();
223 
224         WmiResult<LogicalDiskProperty> drives = WmiQueryHandler.getInstance().queryWMI(this.LOGICAL_DISK_QUERY);
225 
226         for (int i = 0; i < drives.getResultCount(); i++) {
227             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
228             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
229             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
230             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
231             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
232             String volume;
233             if (type != 4) {
234                 char[] chrVolume = new char[BUFSIZE];
235                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
236                 volume = new String(chrVolume).trim();
237             } else {
238                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
239                 String[] split = volume.split("\\\\");
240                 if (split.length > 1 && split[split.length - 1].length() > 0) {
241                     description = split[split.length - 1];
242                 }
243             }
244 
245             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
246             osStore.setName(String.format("%s (%s)", description, name));
247             osStore.setVolume(volume);
248             osStore.setMount(name + "\\");
249             osStore.setDescription(getDriveType(name));
250             osStore.setType(WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i));
251             osStore.setUUID("");
252             osStore.setUsableSpace(free);
253             osStore.setTotalSpace(total);
254             fs.add(osStore);
255         }
256 
257         return fs;
258     }

260     /**
261      * Private method for getting mounted drive type.
262      *
263      * @param drive
264      *            Mounted drive
265      * @return A drive type description
266      */
267     private String getDriveType(String drive) {
268         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
269         case 2:
270             return "Removable drive";
271         case 3:
272             return "Fixed drive";
273         case 4:
274             return "Network drive";
275         case 5:
276             return "CD-ROM";
277         case 6:
278             return "RAM drive";
279         default:
280             return "Unknown drive type";
281         }
282     }

284     @Override
285     public long getOpenFileDescriptors() {
286         // Try PDH if counter exists
287         if (this.handleCountCounter != null) {
288             PerfDataUtil.updateQuery(this.handleCountCounter);
289             return PerfDataUtil.queryCounter(this.handleCountCounter);
290         }
291         // Use WMI instead
292         WmiResult<HandleCountProperty> result = WmiQueryHandler.getInstance().queryWMI(this.handleCountQuery);
293         long descriptors = 0L;
294         for (int i = 0; i < result.getResultCount(); i++) {
295             descriptors += WmiUtil.getUint32(result, HandleCountProperty.HANDLECOUNT, i);
296         }
297         return descriptors;
298     }

300     @Override
301     public long getMaxFileDescriptors() {
302         return MAX_WINDOWS_HANDLES;
