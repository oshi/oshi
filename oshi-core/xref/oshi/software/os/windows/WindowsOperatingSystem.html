145     @Override
146     public FamilyVersionInfo queryFamilyVersionInfo() {
147         WmiResult<OSVersionProperty> versionInfo = new Win32OperatingSystem().queryOsVersion();
148         if (versionInfo.getResultCount() < 1) {
149             return new FamilyVersionInfo("Windows", new OSVersionInfo(System.getProperty("os.version"), null, null));
150         }
151         // Guaranteed that versionInfo is not null and lists non-empty
152         // before calling the parse*() methods
153         int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
154         String buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
155         String version = parseVersion(versionInfo, suiteMask, buildNumber);
156         String codeName = parseCodeName(suiteMask);
157         return new FamilyVersionInfo("Windows", new OSVersionInfo(version, codeName, buildNumber));
158     }

160     private static String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask, String buildNumber) {
161 
162         // Initialize a default, sane value
163         String version = System.getProperty("os.version");
164 
165         // Version is major.minor.build. Parse the version string for
166         // major/minor and get the build number separately
167         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
168         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
169         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
170 
171         // see
172         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
173         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
174                 0) == WinNT.VER_NT_WORKSTATION;
175         switch (major) {
176         case 10:
177             if (minor == 0) {
178                 if (ntWorkstation) {
179                     version = "10";
180                 } else {
181                     // Build numbers greater than 17762 is Server 2019 for OS
182                     // Version 10.0
183                     version = (ParseUtil.parseLongOrDefault(buildNumber, 0L) > 17762) ? "Server 2019" : "Server 2016";
184                 }
185             }
186             break;
187         case 6:
188             if (minor == 3) {
189                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
190             } else if (minor == 2) {
191                 version = ntWorkstation ? "8" : "Server 2012";
192             } else if (minor == 1) {
193                 version = ntWorkstation ? "7" : "Server 2008 R2";
194             } else if (minor == 0) {
195                 version = ntWorkstation ? "Vista" : "Server 2008";
196             }
197             break;
198         case 5:
199             if (minor == 2) {
200                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
201                     version = "Home Server";
202                 } else if (ntWorkstation) {
203                     version = "XP"; // 64 bits
204                 } else {
205                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
206                             : "Server 2003 R2";
207                 }
208             } else if (minor == 1) {
209                 version = "XP"; // 32 bits
210             } else if (minor == 0) {
211                 version = "2000";
212             }
213             break;
214         default:
215             break;
216         }
217 
218         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
219         if (!sp.isEmpty() && !"unknown".equals(sp)) {
220             version = version + " " + sp.replace("Service Pack ", "SP");
221         }
222 
223         return version;
224     }

226     /**
227      * Gets suites available on the system and return as a codename
228      *
229      * @param suiteMask
230      *
231      * @return Suites
232      */
233     private static String parseCodeName(int suiteMask) {
234         List<String> suites = new ArrayList<>();
235         if ((suiteMask & 0x00000002) != 0) {
236             suites.add("Enterprise");
237         }
238         if ((suiteMask & 0x00000004) != 0) {
239             suites.add("BackOffice");
240         }
241         if ((suiteMask & 0x00000008) != 0) {
242             suites.add("Communication Server");
243         }
244         if ((suiteMask & 0x00000080) != 0) {
245             suites.add("Datacenter");
246         }
247         if ((suiteMask & 0x00000200) != 0) {
248             suites.add("Home");
249         }
250         if ((suiteMask & 0x00000400) != 0) {
251             suites.add("Web Server");
252         }
253         if ((suiteMask & 0x00002000) != 0) {
254             suites.add("Storage Server");
255         }
256         if ((suiteMask & 0x00004000) != 0) {
257             suites.add("Compute Cluster");
258         }
259         // 0x8000, Home Server, is included in main version name
260         return String.join(",", suites);
261     }

301     @Override
302     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
303         Set<Integer> childPids = new HashSet<>();
304         // Get processes from ToolHelp API for parent PID
305         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
306         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
307         try {
308             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
309                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
310                     childPids.add(processEntry.th32ProcessID.intValue());
311                 }
312             }
313         } finally {
314             Kernel32.INSTANCE.CloseHandle(snapshot);
315         }
316         List<OSProcess> procList = getProcesses(childPids);
317         List<OSProcess> sorted = processSort(procList, limit, sort);
318         return sorted.toArray(new OSProcess[0]);
319     }

327     /**
328      * Private method to do the heavy lifting for all the getProcess functions.
329      *
330      * @param pids
331      *            A collection of pids to query. If null, the entire process list
332      *            will be queried.
333      * @param slowFields
334      *            Whether to include fields that incur processor latency
335      * @return A corresponding list of processes
336      */
337     private List<OSProcess> processMapToList(Collection<Integer> pids, boolean slowFields) {
338         // Get data from the registry if possible, otherwise performance counters with
339         // WMI backup
340         Map<Integer, OSProcess> processMap = (HKEY_PERFORMANCE_DATA != null)
341                 ? HKEY_PERFORMANCE_DATA.buildProcessMapFromRegistry(this, pids)
342                 : buildProcessMapFromPerfCounters(pids);
343 
344         // define here to avoid object repeated creation overhead later
345         List<String> groupList = new ArrayList<>();
346         List<String> groupIDList = new ArrayList<>();
347         int myPid = getProcessId();
348 
349         // Structure we'll fill from native memory pointer for Vista+
350         Pointer pProcessInfo = null;
351         WTS_PROCESS_INFO_EX[] processInfo = null;
352         IntByReference pCount = new IntByReference(0);
353 
354         // WMI result we'll use for pre-Vista
355         WmiResult<ProcessXPProperty> processWmiResult = null;
356 
357         // Get processes from WTS (post-XP)
358         if (IS_WINDOWS7_OR_GREATER) {
359             final PointerByReference ppProcessInfo = new PointerByReference();
360             if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
361                     new IntByReference(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1), Wtsapi32.WTS_ANY_SESSION, ppProcessInfo,
362                     pCount)) {
363                 LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
364                 return new ArrayList<>(0);
365             }
366             // extract the pointed-to pointer and create array
367             pProcessInfo = ppProcessInfo.getValue();
368             final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
369             processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
370         } else {
371             // Pre-Vista we can't use WTSEnumerateProcessesEx so we'll grab the
372             // same info from WMI and fake the array
373             processWmiResult = new Win32Process().queryProcesses(pids);
374         }
375 
376         // Store a subset of processes in a list to later return.
377         List<OSProcess> processList = new ArrayList<>();
378 
379         int procCount = IS_WINDOWS7_OR_GREATER ? processInfo.length : processWmiResult.getResultCount();
380         for (int i = 0; i < procCount; i++) {
381             int pid = IS_WINDOWS7_OR_GREATER ? processInfo[i].ProcessId
382                     : WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i);
383             OSProcess proc = null;
384             // If the cache is empty, there was a problem with
385             // filling the cache using performance information.
386             if (processMap.isEmpty()) {
387                 if (pids != null && !pids.contains(pid)) {
388                     continue;
389                 }
390                 proc = new OSProcess(this);
391                 proc.setProcessID(pid);
392                 proc.setName(IS_WINDOWS7_OR_GREATER ? processInfo[i].pProcessName
393                         : WmiUtil.getString(processWmiResult, ProcessXPProperty.NAME, i));
394             } else {
395                 proc = processMap.get(pid);
396                 if (proc == null || pids != null && !pids.contains(pid)) {
397                     continue;
398                 }
399             }
400             // For my own process, set CWD
401             if (pid == myPid) {
402                 String cwd = new File(".").getAbsolutePath();
403                 // trim off trailing "."
404                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
405             }
406 
407             if (IS_WINDOWS7_OR_GREATER) {
408                 WTS_PROCESS_INFO_EX procInfo = processInfo[i];
409                 proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
410                 proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
411                 proc.setThreadCount(procInfo.NumberOfThreads);
412                 proc.setVirtualSize(procInfo.PagefileUsage & 0xffff_ffffL);
413                 proc.setOpenFiles(procInfo.HandleCount);
414             } else {
415                 proc.setKernelTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10000L);
416                 proc.setUserTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10000L);
417                 proc.setThreadCount(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i));
418                 // WMI Pagefile usage is in KB
419                 proc.setVirtualSize(1024
420                         * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffff_ffffL));
421                 proc.setOpenFiles(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i));
422             }
423 
424             // Get a handle to the process for various extended info. Only gets
425             // current user unless running as administrator
426             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
427                     proc.getProcessID());
428             if (pHandle != null) {
429                 proc.setBitness(this.getBitness());
430                 // Only test for 32-bit process on 64-bit windows
431                 if (IS_VISTA_OR_GREATER && this.getBitness() == 64) {
432                     IntByReference wow64 = new IntByReference(0);
433                     if (Kernel32.INSTANCE.IsWow64Process(pHandle, wow64)) {
434                         proc.setBitness(wow64.getValue() > 0 ? 32 : 64);
435                     }
436                 }
437                 // Full path
438                 final HANDLEByReference phToken = new HANDLEByReference();
439                 try {// EXECUTABLEPATH
440                     proc.setPath(IS_WINDOWS7_OR_GREATER ? Kernel32Util.QueryFullProcessImageName(pHandle, 0)
441                             : WmiUtil.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i));
442                     if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY,
443                             phToken)) {
444                         Account account = Advapi32Util.getTokenAccount(phToken.getValue());
445                         proc.setUser(account.name);
446                         proc.setUserID(account.sidString);
447                         // Fetching group information incurs ~10ms per process.
448                         if (slowFields) {
449                             Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
450                             // get groups
451                             groupList.clear();
452                             groupIDList.clear();
453                             for (Account a : accounts) {
454                                 groupList.add(a.name);
455                                 groupIDList.add(a.sidString);
456                             }
457                             proc.setGroup(String.join(",", groupList));
458                             proc.setGroupID(String.join(",", groupIDList));
459                         }
460                     } else {
461                         int error = Kernel32.INSTANCE.GetLastError();
462                         // Access denied errors are common. Fail silently.
463                         if (error != WinError.ERROR_ACCESS_DENIED) {
464                             LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
465                                     Kernel32.INSTANCE.GetLastError());
466                         }
467                     }
468                 } catch (Win32Exception e) {
469                     handleWin32ExceptionOnGetProcessInfo(proc, e);
470                 } finally {
471                     final HANDLE token = phToken.getValue();
472                     if (token != null) {
473                         Kernel32.INSTANCE.CloseHandle(token);
474                     }
475                 }
476                 Kernel32.INSTANCE.CloseHandle(pHandle);
477             }
478 
479             // There is no easy way to get ExecutuionState for a process.
480             // The WMI value is null. It's possible to get thread Execution
481             // State and possibly roll up.
482             proc.setState(OSProcess.State.RUNNING);
483 
484             // Initialize default
485             proc.setCommandLine("");
486 
487             processList.add(proc);
488         }
489         // Clean up memory allocated in C (only Vista+ but null pointer
490         // effectively tests)
491         if (pProcessInfo != null && !Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1, pProcessInfo,
492                 pCount.getValue())) {
493             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
494             return new ArrayList<>(0);
495         }
496 
497         // Command Line only accessible via WMI.
498         if (slowFields) {
499             Set<Integer> pidsToQuery = new HashSet<>();
500             if (pids != null) {
501                 for (OSProcess process : processList) {
502                     pidsToQuery.add(process.getProcessID());
503                 }
504             }
505             WmiResult<CommandLineProperty> commandLineProcs = new Win32Process().queryCommandLines(pidsToQuery);
506             for (int p = 0; p < commandLineProcs.getResultCount(); p++) {
507                 int pid = WmiUtil.getUint32(commandLineProcs, CommandLineProperty.PROCESSID, p);
508                 if (processMap.containsKey(pid)) {
509                     OSProcess proc = processMap.get(pid);
510                     proc.setCommandLine(WmiUtil.getString(commandLineProcs, CommandLineProperty.COMMANDLINE, p));
511                 }
512             }
513         }
514         return processList;
515     }

518         LOG.warn("Failed to set path or get user/group on PID {}. It may have terminated. {}", proc.getProcessID(),
519                 ex.getMessage());
520     }

522     private Map<Integer, OSProcess> buildProcessMapFromPerfCounters(Collection<Integer> pids) {
523         Map<Integer, OSProcess> processMap = new HashMap<>();
524         Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> instanceValues = new ProcessInformation()
525                 .queryProcessCounters();
526         long now = System.currentTimeMillis(); // 1970 epoch
527         List<String> instances = instanceValues.getA();
528         Map<ProcessPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
529         List<Long> pidList = valueMap.get(ProcessPerformanceProperty.PROCESSID);
530         List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.PARENTPROCESSID);
531         List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.PRIORITY);
532         List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.READTRANSFERCOUNT);
533         List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.WRITETRANSFERCOUNT);
534         List<Long> workingSetSizeList = valueMap.get(ProcessPerformanceProperty.PRIVATEPAGECOUNT);
535         List<Long> creationTimeList = valueMap.get(ProcessPerformanceProperty.CREATIONDATE);
536 
537         for (int inst = 0; inst < instances.size(); inst++) {
538             int pid = pidList.get(inst).intValue();
539             if (pids == null || pids.contains(pid)) {
540                 OSProcessml#OSProcess">OSProcess proc = new OSProcess(this);
541                 processMap.put(pid, proc);
542 
543                 proc.setProcessID(pid);
544                 proc.setName(instances.get(inst));
545                 proc.setParentProcessID(ppidList.get(inst).intValue());
546                 proc.setPriority(priorityList.get(inst).intValue());
547                 // if creation time value is less than current millis, it's in 1970 epoch,
548                 // otherwise it's 1601 epoch and we must convert
549                 long ctime = creationTimeList.get(inst);
550                 if (ctime > now) {
551                     ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
552                 }
553                 proc.setUpTime(now - ctime);
554                 proc.setStartTime(ctime);

622     private static long querySystemBootTime() {
623         String eventLog = systemLog.get();
624         if (eventLog != null) {
625             try {
626                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
627                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
628                 // Startup" is enabled we may not see event 12, so also check for most recent ID
629                 // 6005 (Event log startup) as a reasonably close backup.
630                 long event6005Time = 0L;
631                 while (iter.hasNext()) {
632                     EventLogRecord record = iter.next();
633                     if (record.getStatusCode() == 12) {
634                         // Event 12 is system boot. We want this value unless we find two 6005 events
635                         // first (may occur with Fast Boot)
636                         return record.getRecord().TimeGenerated.longValue();
637                     } else if (record.getStatusCode() == 6005) {
638                         // If we already found one, this means we've found a second one without finding
639                         // an event 12. Return the latest one.
640                         if (event6005Time > 0) {
641                             return event6005Time;
642                         }
643                         // First 6005; tentatively assign
644                         event6005Time = record.getRecord().TimeGenerated.longValue();
645                     }
646                 }
647                 // Only one 6005 found, return
648                 if (event6005Time > 0) {
649                     return event6005Time;
650                 }
651             } catch (Win32Exception e) {
652                 LOG.warn("Can't open event log \"{}\".", eventLog);
653             }
654         }
655         // If we get this far, event log reading has failed, either from no log or no
656         // startup times. Subtract up time from current time as a reasonable proxy.
657         return System.currentTimeMillis() / 1000L - querySystemUptime();
658     }

666      * Enables debug privileges for this process, required for OpenProcess() to get
667      * processes other than the current user
668      */
669     private static void enableDebugPrivilege() {
670         HANDLEByReference hToken = new HANDLEByReference();
671         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
672                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
673         if (!success) {
674             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
675             return;
676         }
677         WinNT.LUID luid = new WinNT.LUID();
678         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
679         if (!success) {
680             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
681             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
682             return;
683         }
684         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
685         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
686         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
687         if (!success) {
688             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
689         }
690         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
691     }

739     private static class HkeyPerformanceData {
740         /*
741          * Grow as needed but persist
742          */
743         private int perfDataBufferSize = 8192;
744         /*
745          * Process counter index in integer and string form
746          */
747         private int processIndex; // 6
748         private String processIndexStr; // "6"
749         /*
750          * Registry counter data byte offsets
751          */
752         private int priorityBaseOffset; // 92
753         private int elapsedTimeOffset; // 96
754         private int idProcessOffset; // 104
755         private int creatingProcessIdOffset; // 108
756         private int ioReadOffset; // 160
757         private int ioWriteOffset; // 168
758         private int workingSetPrivateOffset; // 192
759 
760         private HkeyPerformanceData() throws InstantiationException {
761             // Get the title indices
762             int priorityBaseIndex = 0;
763             int elapsedTimeIndex = 0;
764             int idProcessIndex = 0;
765             int creatingProcessIdIndex = 0;
766             int ioReadIndex = 0;
767             int ioWriteIndex = 0;
768             int workingSetPrivateIndex = 0;
769 
770             try {
771                 final String ENGLISH_COUNTER_KEY = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Perflib\\009";
772                 final String ENGLISH_COUNTER_VALUE = "Counter";
773 
774                 // Look up list of english names and ids
775                 String[] counters = Advapi32Util.registryGetStringArray(WinReg.HKEY_LOCAL_MACHINE, ENGLISH_COUNTER_KEY,
776                         ENGLISH_COUNTER_VALUE);
777                 // Array contains alternating index/name pairs
778                 // {"1", "1847", "2", "System", "4", "Memory", ... }
779                 // Get position of name in the array (odd index), return parsed value of
780                 // previous even index
781                 for (int i = 1; i < counters.length; i += 2) {
782                     if (counters[i].equals("Process")) {
783                         this.processIndex = Integer.parseInt(counters[i - 1]);
784                     } else if (counters[i].equals("Priority Base")) {
785                         priorityBaseIndex = Integer.parseInt(counters[i - 1]);
786                     } else if (counters[i].equals("Elapsed Time")) {
787                         elapsedTimeIndex = Integer.parseInt(counters[i - 1]);
788                     } else if (counters[i].equals("ID Process")) {
789                         idProcessIndex = Integer.parseInt(counters[i - 1]);
790                     } else if (counters[i].equals("Creating Process ID")) {
791                         creatingProcessIdIndex = Integer.parseInt(counters[i - 1]);
792                     } else if (counters[i].equals("IO Read Bytes/sec")) {
793                         ioReadIndex = Integer.parseInt(counters[i - 1]);
794                     } else if (counters[i].equals("IO Write Bytes/sec")) {
795                         ioWriteIndex = Integer.parseInt(counters[i - 1]);
796                     } else if (counters[i].equals("Working Set - Private")) {
797                         workingSetPrivateIndex = Integer.parseInt(counters[i - 1]);
798                     }
799                 }
800             } catch (NumberFormatException e) {
801                 // Unexpected but handle anyway
802                 throw new InstantiationException("Failed to parse counter index/name array.");
803             } catch (Win32Exception e) {
804                 throw new InstantiationException("Unable to locate English counter names in registry Perflib 009.");
805             }
806             // If any of the indices are 0, we failed
807             if (this.processIndex == 0 || priorityBaseIndex == 0 || elapsedTimeIndex == 0 || idProcessIndex == 0
808                     || creatingProcessIdIndex == 0 || ioReadIndex == 0 || ioWriteIndex == 0
809                     || workingSetPrivateIndex == 0) {
810                 throw new InstantiationException("Failed to parse counter index/name array.");
811             }
812             this.processIndexStr = Integer.toString(this.processIndex);
813 
814             // now load the Process registry to match up the offsets
815             // Sequentially increase the buffer until everything fits.
816             // Save this buffer size for later use
817             IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
818             Pointer pPerfData = new Memory(this.perfDataBufferSize);
819             int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
820                     pPerfData, lpcbData);
821             if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
822                 throw new InstantiationException("Error " + ret + " reading HKEY_PERFORMANCE_DATA from the registry.");
823             }
824             while (ret == WinError.ERROR_MORE_DATA) {
825                 this.perfDataBufferSize += 4096;
826                 lpcbData.setValue(this.perfDataBufferSize);
827                 pPerfData = new Memory(this.perfDataBufferSize);
828                 ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
829                         pPerfData, lpcbData);
830             }
831 
832             PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
833 
834             // See format at
835             // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
836             // [ ] Object Type
837             // [ ][ ][ ] Multiple counter definitions
838             // Then multiple:
839             // [ ] Instance Definition
840             // [ ] Instance name
841             // [ ] Counter Block
842             // [ ][ ][ ] Counter data for each definition above
843 
844             long perfObjectOffset = perfData.HeaderLength;
845 
846             // Iterate object types. For Process should only be one here
847             for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
848                 PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
849                 // Identify where counter definitions start
850                 long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
851                 // If this isn't the Process object, ignore
852                 if (perfObject.ObjectNameTitleIndex == this.processIndex) {
853                     for (int counter = 0; counter < perfObject.NumCounters; counter++) {
854                         PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
855                                 pPerfData.share(perfCounterOffset));
856                         if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
857                             this.priorityBaseOffset = perfCounter.CounterOffset;
858                         } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
859                             this.elapsedTimeOffset = perfCounter.CounterOffset;
860                         } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
861                             this.creatingProcessIdOffset = perfCounter.CounterOffset;
862                         } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
863                             this.idProcessOffset = perfCounter.CounterOffset;
864                         } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
865                             this.ioReadOffset = perfCounter.CounterOffset;
866                         } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
867                             this.ioWriteOffset = perfCounter.CounterOffset;
868                         } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
869                             this.workingSetPrivateOffset = perfCounter.CounterOffset;
870                         }
871                         // Increment for next Counter
872                         perfCounterOffset += perfCounter.ByteLength;
873                     }
874                     // We're done, break the loop
875                     break;
876                 }
877                 // Increment for next object (should never need this)
878                 perfObjectOffset += perfObject.TotalByteLength;
879             }
880         }

882         private Map<Integer, OSProcess> buildProcessMapFromRegistry(OperatingSystem os, Collection<Integer> pids) {
883             Map<Integer, OSProcess> processMap = new HashMap<>();
884             // Grab the PERF_DATA_BLOCK from the registry.
885             // Sequentially increase the buffer until everything fits.
886             IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
887             Pointer pPerfData = new Memory(this.perfDataBufferSize);
888             int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
889                     pPerfData, lpcbData);
890             if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
891                 LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
892                 return processMap;
893             }
894             while (ret == WinError.ERROR_MORE_DATA) {
895                 this.perfDataBufferSize += 4096;
896                 lpcbData.setValue(this.perfDataBufferSize);
897                 pPerfData = new Memory(this.perfDataBufferSize);
898                 ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
899                         pPerfData, lpcbData);
900             }
901 
902             PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
903             long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
904             long now = System.currentTimeMillis(); // 1970 epoch
905 
906             // See format at
907             // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
908             // [ ] Object Type
909             // [ ][ ][ ] Multiple counter definitions
910             // Then multiple:
911             // [ ] Instance Definition
912             // [ ] Instance name
913             // [ ] Counter Block
914             // [ ][ ][ ] Counter data for each definition above
915             long perfObjectOffset = perfData.HeaderLength;
916 
917             // Iterate object types. For Process should only be one here
918             for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
919                 PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
920                 // If this isn't the Process object, ignore
921                 if (perfObject.ObjectNameTitleIndex == this.processIndex) {
922                     // Skip over counter definitions
923                     // There will be many of these, this points to the first one
924                     long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
925 
926                     // We need this for every process, initialize outside loop to
927                     // save overhead
928                     PERF_COUNTER_BLOCK perfCounterBlock = null;
929                     // Iterate instances.
930                     // The last instance is _Total so subtract 1 from max
931                     for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
932                         PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
933                                 pPerfData.share(perfInstanceOffset));
934                         long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
935 
936                         int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
937                         if (pids == null || pids.contains(pid)) {
938                             OSProcessml#OSProcess">OSProcess proc = new OSProcess(os);
939                             processMap.put(pid, proc);
940 
941                             proc.setProcessID(pid);
942                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
943                             long upTime = (perfTime100nSec
944                                     - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset)) / 10_000L;
945                             proc.setUpTime(upTime < 1L ? 1L : upTime);
946                             proc.setStartTime(now - upTime);
947                             proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
948                             proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
949                             proc.setResidentSetSize(
950                                     pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
951                             proc.setParentProcessID(
952                                     pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
953                             proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
954                         }
955 
956                         // Increment to next instance
957                         perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
958                         perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
959                     }
960                     // We've found the process object and are done, no need to look at any other
961                     // objects (shouldn't be any). Break the loop
962                     break;
963                 }
964                 // Increment for next object (should never need this)
965                 perfObjectOffset += perfObject.TotalByteLength;
966             }
967             return processMap;
968         }
