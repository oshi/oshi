153     @Override
154     public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
155         String version = System.getProperty("os.name");
156         if (version.startsWith("Windows ")) {
157             version = version.substring(8);
158         }
159 
160         String sp = null;
161         int suiteMask = 0;
162         String buildNumber = null;
163         WmiResult<OSVersionProperty> versionInfo = Win32OperatingSystem.queryOsVersion();
164         if (versionInfo.getResultCount() > 0) {
165             sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
166             if (!sp.isEmpty() && !Constants.UNKNOWN.equals(sp)) {
167                 version = version + " " + sp.replace("Service Pack ", "SP");
168             }
169             suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
170             buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
171         }
172         String codeName = parseCodeName(suiteMask);
173         return new Pair<>("Windows", new OSVersionInfo(version, codeName, buildNumber));
174     }

176     /**
177      * Gets suites available on the system and return as a codename
178      *
179      * @param suiteMask
180      *            The suite mask bitmask
181      *
182      * @return Suites
183      */
184     private static String parseCodeName(int suiteMask) {
185         List<String> suites = new ArrayList<>();
186         if ((suiteMask & 0x00000002) != 0) {
187             suites.add("Enterprise");
188         }
189         if ((suiteMask & 0x00000004) != 0) {
190             suites.add("BackOffice");
191         }
192         if ((suiteMask & 0x00000008) != 0) {
193             suites.add("Communications Server");
194         }
195         if ((suiteMask & 0x00000080) != 0) {
196             suites.add("Datacenter");
197         }
198         if ((suiteMask & 0x00000200) != 0) {
199             suites.add("Home");
200         }
201         if ((suiteMask & 0x00000400) != 0) {
202             suites.add("Web Server");
203         }
204         if ((suiteMask & 0x00002000) != 0) {
205             suites.add("Storage Server");
206         }
207         if ((suiteMask & 0x00004000) != 0) {
208             suites.add("Compute Cluster");
209         }
210         if ((suiteMask & 0x00008000) != 0) {
211             suites.add("Home Server");
212         }
213         return String.join(",", suites);
214     }

218         if (jvmBitness < 64 && System.getenv("ProgramFiles(x86)") != null && IS_VISTA_OR_GREATER) {
219             WmiResult<BitnessProperty> bitnessMap = Win32Processor.queryBitness();
220             if (bitnessMap.getResultCount() > 0) {

222             }
223         }
224         return jvmBitness;

227     @Override
228     public boolean isElevated() {
229         HANDLEByReference hToken = new HANDLEByReference();
230         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(), WinNT.TOKEN_QUERY,
231                 hToken);
232         if (!success) {
233             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
234             return false;
235         }
236         try {
237             TOKEN_ELEVATION elevation = new TOKEN_ELEVATION();
238             if (Advapi32.INSTANCE.GetTokenInformation(hToken.getValue(), TOKENELEVATION, elevation, elevation.size(),
239                     new IntByReference())) {
240                 return elevation.TokenIsElevated > 0;
241             }
242         } finally {
243             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
244         }
245         return false;
246     }

288     private static Map<Integer, Integer> getParentPidsFromSnapshot() {
289         Map<Integer, Integer> parentPidMap = new HashMap<>();
290         // Get processes from ToolHelp API for parent PID
291         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
292         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
293         try {
294             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
295                 parentPidMap.put(processEntry.th32ProcessID.intValue(), processEntry.th32ParentProcessID.intValue());
296             }
297         } finally {
298             Kernel32.INSTANCE.CloseHandle(snapshot);
299         }
300         return parentPidMap;
301     }

309     private List<OSProcess> processMapToList(Collection<Integer> pids) {
310         // Get data from the registry if possible
311         Map<Integer, ProcessPerformanceData.PerfCounterBlock> processMap = processMapFromRegistry.get();
312         // otherwise performance counters with WMI backup
313         if (processMap == null || processMap.isEmpty()) {
314             processMap = (pids == null) ? processMapFromPerfCounters.get()
315                     : ProcessPerformanceData.buildProcessMapFromPerfCounters(pids);
316         }
317         Map<Integer, ThreadPerformanceData.PerfCounterBlock> threadMap = null;
318         if (USE_PROCSTATE_SUSPENDED) {
319             // Get data from the registry if possible
320             threadMap = threadMapFromRegistry.get();
321             // otherwise performance counters with WMI backup
322             if (threadMap == null || threadMap.isEmpty()) {
323                 threadMap = (pids == null) ? threadMapFromPerfCounters.get()
324                         : ThreadPerformanceData.buildThreadMapFromPerfCounters(pids);
325             }
326         }
327 
328         Map<Integer, WtsInfo> processWtsMap = ProcessWtsData.queryProcessWtsMap(pids);
329 
330         Set<Integer> mapKeys = new HashSet<>(processWtsMap.keySet());
331         mapKeys.retainAll(processMap.keySet());
332 
333         List<OSProcess> processList = new ArrayList<>();
334         for (Integer pid : mapKeys) {
335             processList.add(new WindowsOSProcess(pid, this, processMap, processWtsMap, threadMap));
336         }
337         return processList;
338     }

386     private static long querySystemUptime() {
387         // Uptime is in seconds so divide milliseconds
388         // GetTickCount64 requires Vista (6.0) or later
389         if (IS_VISTA_OR_GREATER) {
390             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
391         } else {
392             // 32 bit rolls over at ~ 49 days
393             return Kernel32.INSTANCE.GetTickCount() / 1000L;
394         }
395     }

402     private static long querySystemBootTime() {
403         String eventLog = systemLog.get();
404         if (eventLog != null) {
405             try {
406                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
407                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
408                 // Startup" is enabled we may not see event 12, so also check for most recent ID
409                 // 6005 (Event log startup) as a reasonably close backup.
410                 long event6005Time = 0L;
411                 while (iter.hasNext()) {
412                     EventLogRecord record = iter.next();
413                     if (record.getStatusCode() == 12) {
414                         // Event 12 is system boot. We want this value unless we find two 6005 events
415                         // first (may occur with Fast Boot)
416                         return record.getRecord().TimeGenerated.longValue();
417                     } else if (record.getStatusCode() == 6005) {
418                         // If we already found one, this means we've found a second one without finding
419                         // an event 12. Return the latest one.
420                         if (event6005Time > 0) {
421                             return event6005Time;
422                         }
423                         // First 6005; tentatively assign
424                         event6005Time = record.getRecord().TimeGenerated.longValue();
425                     }
426                 }
427                 // Only one 6005 found, return
428                 if (event6005Time > 0) {
429                     return event6005Time;
430                 }
431             } catch (Win32Exception e) {
432                 LOG.warn("Can't open event log \"{}\".", eventLog);
433             }
434         }
435         // If we get this far, event log reading has failed, either from no log or no
436         // startup times. Subtract up time from current time as a reasonable proxy.
437         return System.currentTimeMillis() / 1000L - querySystemUptime();
438     }

445     /**
446      * Attempts to enable debug privileges for this process, required for
447      * OpenProcess() to get processes other than the current user. Requires elevated
448      * permissions.
449      *
450      * @return {@code true} if debug privileges were successfully enabled.
451      */
452     private static boolean enableDebugPrivilege() {
453         HANDLEByReference hToken = new HANDLEByReference();
454         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
455                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
456         if (!success) {
457             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
458             return false;
459         }
460         try {
461             WinNT.LUID luid = new WinNT.LUID();
462             success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
463             if (!success) {
464                 LOG.error("LookupPrivilegeValue failed. Error: {}", Native.getLastError());
465                 return false;
466             }
467             WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
468             tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
469             success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
470             int err = Native.getLastError();
471             if (!success) {
472                 LOG.error("AdjustTokenPrivileges failed. Error: {}", err);
473                 return false;
474             } else if (err == WinError.ERROR_NOT_ALL_ASSIGNED) {
475                 LOG.debug("Debug privileges not enabled.");
476                 return false;
477             }
478         } finally {
479             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
480         }
481         return true;
482     }

484     @Override
485     public List<OSService> getServices() {
486         try (W32ServiceManager sm = new W32ServiceManager()) {
487             sm.open(Winsvc.SC_MANAGER_ENUMERATE_SERVICE);
488             Winsvc.ENUM_SERVICE_STATUS_PROCESS[] services = sm.enumServicesStatusExProcess(WinNT.SERVICE_WIN32,
489                     Winsvc.SERVICE_STATE_ALL, null);
490             List<OSService> svcArray = new ArrayList<>();
491             for (Winsvc.ENUM_SERVICE_STATUS_PROCESS service : services) {
492                 State state;
493                 switch (service.ServiceStatusProcess.dwCurrentState) {
494                 case 1:
495                     state = STOPPED;
496                     break;
497                 case 4:
498                     state = RUNNING;
499                     break;
500                 default:
501                     state = OTHER;
502                     break;
503                 }
504                 svcArray.add(new OSService(service.lpDisplayName, service.ServiceStatusProcess.dwProcessId, state));
505             }
506             return svcArray;
507         } catch (com.sun.jna.platform.win32.Win32Exception ex) {
508             LOG.error("Win32Exception: {}", ex.getMessage());
509             return Collections.emptyList();
510         }
511     }

513     private static String querySystemLog() {
514         String systemLog = GlobalConfig.get("oshi.os.windows.eventlog", "System");
515         if (systemLog.isEmpty()) {
516             // Use faster boot time approximation
517             return null;
518         }
519         // Check whether it works
520         HANDLE h = Advapi32.INSTANCE.OpenEventLog(null, systemLog);
521         if (h == null) {
522             LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
523                     systemLog);
524             return null;
525         }
526         return systemLog;
527     }

562     /**
563      * Is the specified process x86 or x86-compatibility mode?
564      *
565      * @param h
566      *            The handle to the processs to check
567      * @return true if the process is 32-bit
568      */
569     static boolean isWow(HANDLE h) {
570         if (X86) {
571             return true;
572         }
573         IntByReference isWow = new IntByReference();
574         Kernel32.INSTANCE.IsWow64Process(h, isWow);
575         return isWow.getValue() != 0;
576     }
577 
578     private static boolean isCurrentWow() {
579         if (X86) {
580             return true;
581         }
582         HANDLE h = Kernel32.INSTANCE.GetCurrentProcess();
583         if (h != null) {
584             try {
585                 return isWow(h);
586             } finally {
587                 Kernel32.INSTANCE.CloseHandle(h);
588             }
589         }
590         return false;
591     }
