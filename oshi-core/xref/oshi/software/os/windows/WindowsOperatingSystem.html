141     private void initRegistry() {
142         // Get the title indices
143         int priorityBaseIndex;
144         int elapsedTimeIndex;
145         int idProcessIndex;
146         int creatingProcessIdIndex;
147         int ioReadIndex;
148         int ioWriteIndex;
149         int workingSetPrivateIndex;
150         try {
151             this.processIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Process");
152             priorityBaseIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Priority Base");
153             elapsedTimeIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Elapsed Time");
154             idProcessIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("ID Process");
155             creatingProcessIdIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Creating Process ID");
156             ioReadIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("IO Read Bytes/sec");
157             ioWriteIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("IO Write Bytes/sec");
158             workingSetPrivateIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Working Set - Private");
159         } catch (Win32Exception e) {
160             LOG.error("Unable to locate English counter names in registry Perflib 009. Assuming English counters.");
161             DWORDByReference index = new DWORDByReference();
162             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Process", index);
163             this.processIndex = index.getValue().intValue();
164             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Priority Base", index);
165             priorityBaseIndex = index.getValue().intValue();
166             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Elapsed Time", index);
167             elapsedTimeIndex = index.getValue().intValue();
168             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "ID Process", index);
169             idProcessIndex = index.getValue().intValue();
170             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Creating Process ID", index);
171             creatingProcessIdIndex = index.getValue().intValue();
172             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Read Bytes/sec", index);
173             ioReadIndex = index.getValue().intValue();
174             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Write Bytes/sec", index);
175             ioWriteIndex = index.getValue().intValue();
176             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Working Set - Private", index);
177             workingSetPrivateIndex = index.getValue().intValue();
178         }
179         this.processIndexStr = Integer.toString(this.processIndex);
180 
181         // now load the Process registry to match up the offsets
182         // Sequentially increase the buffer until everything fits.
183         // Save this buffer size for later use
184         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
185         Pointer pPerfData = new Memory(this.perfDataBufferSize);
186         int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
187                 pPerfData, lpcbData);
188         if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
189             LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
190             return;
191         }
192         while (ret == WinError.ERROR_MORE_DATA) {
193             this.perfDataBufferSize += 4096;
194             lpcbData.setValue(this.perfDataBufferSize);
195             pPerfData = new Memory(this.perfDataBufferSize);
196             ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
197                     pPerfData, lpcbData);
198         }
199 
200         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
201 
202         // See format at
203         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
204         // [ ] Object Type
205         // [ ][ ][ ] Multiple counter definitions
206         // Then multiple:
207         // [ ] Instance Definition
208         // [ ] Instance name
209         // [ ] Counter Block
210         // [ ][ ][ ] Counter data for each definition above
211 
212         long perfObjectOffset = perfData.HeaderLength;
213 
214         // Iterate object types. For Process should only be one here
215         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
216             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
217             // Identify where counter definitions start
218             long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
219             // If this isn't the Process object, ignore
220             if (perfObject.ObjectNameTitleIndex == this.processIndex) {
221                 for (int counter = 0; counter < perfObject.NumCounters; counter++) {
222                     PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
223                             pPerfData.share(perfCounterOffset));
224                     if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
225                         this.priorityBaseOffset = perfCounter.CounterOffset;
226                     } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
227                         this.elapsedTimeOffset = perfCounter.CounterOffset;
228                     } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
229                         this.creatingProcessIdOffset = perfCounter.CounterOffset;
230                     } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
231                         this.idProcessOffset = perfCounter.CounterOffset;
232                     } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
233                         this.ioReadOffset = perfCounter.CounterOffset;
234                     } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
235                         this.ioWriteOffset = perfCounter.CounterOffset;
236                     } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
237                         this.workingSetPrivateOffset = perfCounter.CounterOffset;
238                     }
239                     // Increment for next Counter
240                     perfCounterOffset += perfCounter.ByteLength;
241                 }
242                 // We're done, break the loop
243                 break;
244             }
245             // Increment for next object (should never need this)
246             perfObjectOffset += perfObject.TotalByteLength;
247         }
248     }

250     private void initBitness() {
251         // If bitness is 64 we are 64 bit.
252         // If 32 test if we are on 64-bit OS
253         if (this.bitness < 64) {
254             // Try the easy way
255             if (System.getenv("ProgramFiles(x86)") != null) {
256                 this.bitness = 64;
257             } else {
258                 WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<>("Win32_Processor", BitnessProperty.class);
259                 WmiResult<BitnessProperty> bitnessMap = WmiQueryHandler.getInstance().queryWMI(bitnessQuery);
260                 if (bitnessMap.getResultCount() > 0) {
261                     this.bitness = WmiUtil.getUint16(bitnessMap, BitnessProperty.ADDRESSWIDTH, 0);
262                 }
263             }
264         }

288     @Override
289     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
290         Set<Integer> childPids = new HashSet<>();
291         // Get processes from ToolHelp API for parent PID
292         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
293         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
294         try {
295             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
296                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
297                     childPids.add(processEntry.th32ProcessID.intValue());
298                 }
299             }
300         } finally {
301             Kernel32.INSTANCE.CloseHandle(snapshot);
302         }
303         List<OSProcess> procList = getProcesses(childPids);
304         List<OSProcess> sorted = processSort(procList, limit, sort);
305         return sorted.toArray(new OSProcess[sorted.size()]);
306     }

331     /**
332      * Private method to do the heavy lifting for all the getProcess functions.
333      *
334      * @param pids
335      *            A collection of pids to query. If null, the entire process
336      *            list will be queried.
337      * @param slowFields
338      *            Whether to include fields that incur processor latency
339      * @return A corresponding list of processes
340      */
341     private List<OSProcess> processMapToList(Collection<Integer> pids, boolean slowFields) {
342         // Get data from the registry to update cache
343         updateProcessMapFromRegistry(pids);
344 
345         // define here to avoid object repeated creation overhead later
346         List<String> groupList = new ArrayList<>();
347         List<String> groupIDList = new ArrayList<>();
348         int myPid = getProcessId();
349 
350         // Structure we'll fill from native memory pointer for Vista+
351         Pointer pProcessInfo = null;
352         WTS_PROCESS_INFO_EX[] processInfo = null;
353         IntByReference pCount = new IntByReference(0);
354 
355         // WMI result we'll use for pre-Vista
356         WmiResult<ProcessXPProperty> processWmiResult = null;
357 
358         // Get processes from WTS (post-XP)
359         if (IS_VISTA_OR_GREATER) {
360             final PointerByReference ppProcessInfo = new PointerByReference();
361             if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
362                     new IntByReference(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1), Wtsapi32.WTS_ANY_SESSION, ppProcessInfo,
363                     pCount)) {
364                 LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
365                 return new ArrayList<>(0);
366             }
367             // extract the pointed-to pointer and create array
368             pProcessInfo = ppProcessInfo.getValue();
369             final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
370             processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
371         } else {
372             // Pre-Vista we can't use WTSEnumerateProcessesEx so we'll grab the
373             // same info from WMI and fake the array
374             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
375             if (pids != null) {
376                 boolean first = true;
377                 for (Integer pid : pids) {
378                     if (first) {
379                         sb.append(" WHERE ProcessID=");
380                         first = false;
381                     } else {
382                         sb.append(" OR ProcessID=");
383                     }
384                     sb.append(pid);
385                 }
386             }
387             PROCESS_QUERY_XP.setWmiClassName(sb.toString());
388             processWmiResult = WmiQueryHandler.getInstance().queryWMI(PROCESS_QUERY_XP);
389         }
390 
391         // Store a subset of processes in a list to later return.
392         List<OSProcess> processList = new ArrayList<>();
393 
394         int procCount = IS_VISTA_OR_GREATER ? processInfo.length : processWmiResult.getResultCount();
395         for (int i = 0; i < procCount; i++) {
396 
397             // Skip if only updating a subset of pids, or if not in cache.
398             // (Cache should have just been updated from registry so this will
399             // only occur in a race condition for a just-started process.)
400             // However, when the cache is empty, there was a problem with
401             // filling
402             // the cache using performance information. When this happens, we
403             // ignore
404             // the cache completely.
405 
406             int pid = IS_VISTA_OR_GREATER ? processInfo[i].ProcessId
407                     : WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i);
408             OSProcess proc = null;
409             if (this.processMap.isEmpty()) {
410                 if (pids != null && !pids.contains(pid)) {
411                     continue;
412                 }
413                 proc = new OSProcess();
414                 proc.setProcessID(pid);
415                 proc.setName(IS_VISTA_OR_GREATER ? processInfo[i].pProcessName
416                         : WmiUtil.getString(processWmiResult, ProcessXPProperty.NAME, i));
417             } else {
418                 proc = this.processMap.get(pid);
419                 if (proc == null || pids != null && !pids.contains(pid)) {
420                     continue;
421                 }
422             }
423             // For my own process, set CWD
424             if (pid == myPid) {
425                 String cwd = new File(".").getAbsolutePath();
426                 // trim off trailing "."
427                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
428             }
429 
430             if (IS_VISTA_OR_GREATER) {
431                 WTS_PROCESS_INFO_EX procInfo = processInfo[i];
432                 proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
433                 proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
434                 proc.setThreadCount(procInfo.NumberOfThreads);
435                 proc.setVirtualSize(procInfo.PagefileUsage & 0xffff_ffffL);
436                 proc.setOpenFiles(procInfo.HandleCount);
437             } else {
438                 proc.setKernelTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10000L);
439                 proc.setUserTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10000L);
440                 proc.setThreadCount(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i));
441                 // WMI Pagefile usage is in KB
442                 proc.setVirtualSize(1024
443                         * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffff_ffffL));
444                 proc.setOpenFiles(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i));
445             }
446 
447             // Get a handle to the process for various extended info. Only gets
448             // current user unless running as administrator
449             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
450                     proc.getProcessID());
451             if (pHandle != null) {
452                 // Full path
453                 final HANDLEByReference phToken = new HANDLEByReference();
454                 try {// EXECUTABLEPATH
455                     proc.setPath(IS_VISTA_OR_GREATER ? Kernel32Util.QueryFullProcessImageName(pHandle, 0)
456                             : WmiUtil.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i));
457                     if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY,
458                             phToken)) {
459                         Account account = Advapi32Util.getTokenAccount(phToken.getValue());
460                         proc.setUser(account.name);
461                         proc.setUserID(account.sidString);
462                         // Fetching group information incurs ~10ms per process.
463                         if (slowFields) {
464                             Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
465                             // get groups
466                             groupList.clear();
467                             groupIDList.clear();
468                             for (Account a : accounts) {
469                                 groupList.add(a.name);
470                                 groupIDList.add(a.sidString);
471                             }
472                             proc.setGroup(FormatUtil.join(",", groupList));
473                             proc.setGroupID(FormatUtil.join(",", groupIDList));
474                         }
475                     } else {
476                         int error = Kernel32.INSTANCE.GetLastError();
477                         // Access denied errors are common. Fail silently.
478                         if (error != WinError.ERROR_ACCESS_DENIED) {
479                             LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
480                                     Kernel32.INSTANCE.GetLastError());
481                         }
482                     }
483                 } catch (Win32Exception e) {
484                     handleWin32ExceptionOnGetProcessInfo(proc, e);
485                 } finally {
486                     final HANDLE token = phToken.getValue();
487                     if (token != null) {
488                         Kernel32.INSTANCE.CloseHandle(token);
489                     }
490                 }
491             }
492             Kernel32.INSTANCE.CloseHandle(pHandle);
493 
494             // There is no easy way to get ExecutuionState for a process.
495             // The WMI value is null. It's possible to get thread Execution
496             // State and possibly roll up.
497             proc.setState(OSProcess.State.RUNNING);
498 
499             processList.add(proc);
500         }
501         // Clean up memory allocated in C (only Vista+ but null pointer
502         // effectively tests)
503         if (pProcessInfo != null && !Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1, pProcessInfo,
504                 pCount.getValue()))
505 
506         {
507             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
508             return new ArrayList<>(0);
509         }
510 
511         // Command Line only accessible via WMI.
512         // Utilize cache to only update new processes
513         Set<Integer> emptyCommandLines = new HashSet<>();
514         for (OSProcess cachedProcess : processList) {
515             // If the process in the cache has an empty command line.
516             if (cachedProcess.getCommandLine().isEmpty()) {
517                 emptyCommandLines.add(cachedProcess.getProcessID());
518             }
519         }
520         if (!emptyCommandLines.isEmpty()) {
521             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
522             boolean first = true;
523             for (Integer pid : emptyCommandLines) {
524                 if (first) {
525                     sb.append(" WHERE ProcessID=");
526                     first = false;
527                 } else {
528                     sb.append(" OR ProcessID=");
529                 }
530                 sb.append(pid);
531             }
532             PROCESS_QUERY.setWmiClassName(sb.toString());
533             WmiResult<ProcessProperty> commandLineProcs = WmiQueryHandler.getInstance().queryWMI(PROCESS_QUERY);
534 
535             for (int p = 0; p < commandLineProcs.getResultCount(); p++) {
536                 int pid = WmiUtil.getUint32(commandLineProcs, ProcessProperty.PROCESSID, p);
537                 // This should always be true because emptyCommandLines was
538                 // built from a subset of the cache, but just in case, protect
539                 // against dereferencing null
540                 if (this.processMap.containsKey(pid)) {
541                     OSProcess proc = this.processMap.get(pid);
542                     proc.setCommandLine(WmiUtil.getString(commandLineProcs, ProcessProperty.COMMANDLINE, p));
543                 }
544             }
545         }
546 
547         return processList;
548     }

550     protected void handleWin32ExceptionOnGetProcessInfo(OSProcess proc, Win32Exception ex) {
551         LOG.warn("Failed to set path or get user/group on PID {}. It may have terminated. {}", proc.getProcessID(),
552                 ex.getMessage());
553     }

555     private void updateProcessMapFromRegistry(Collection<Integer> pids) {
556         List<Integer> pidsToKeep = new ArrayList<>();
557 
558         // Grab the PERF_DATA_BLOCK from the registry.
559         // Sequentially increase the buffer until everything fits.
560         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
561         Pointer pPerfData = new Memory(this.perfDataBufferSize);
562         int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
563                 pPerfData, lpcbData);
564         if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
565             LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
566             return;
567         }
568         while (ret == WinError.ERROR_MORE_DATA) {
569             this.perfDataBufferSize += 4096;
570             lpcbData.setValue(this.perfDataBufferSize);
571             pPerfData = new Memory(this.perfDataBufferSize);
572             ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
573                     pPerfData, lpcbData);
574         }
575 
576         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
577         long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
578         long now = System.currentTimeMillis(); // 1970 epoch
579 
580         // See format at
581         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
582         // [ ] Object Type
583         // [ ][ ][ ] Multiple counter definitions
584         // Then multiple:
585         // [ ] Instance Definition
586         // [ ] Instance name
587         // [ ] Counter Block
588         // [ ][ ][ ] Counter data for each definition above
589 
590         long perfObjectOffset = perfData.HeaderLength;
591 
592         // Iterate object types. For Process should only be one here
593         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
594             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
595             // If this isn't the Process object, ignore
596             if (perfObject.ObjectNameTitleIndex == this.processIndex) {
597                 // Skip over counter definitions
598                 // There will be many of these, this points to the first one
599                 long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
600 
601                 // We need this for every process, initialize outside loop to
602                 // save overhead
603                 PERF_COUNTER_BLOCK perfCounterBlock = null;
604                 // Iterate instances.
605                 // The last instance is _Total so subtract 1 from max
606                 for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
607                     PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
608                             pPerfData.share(perfInstanceOffset));
609                     long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
610 
611                     int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
612                     if (pids == null || pids.contains(pid)) {
613                         pidsToKeep.add(pid);
614 
615                         // If process exists fetch from cache
616                         OSProcess proc = null;
617                         if (this.processMap.containsKey(pid)) {
618                             proc = this.processMap.get(pid);
619                         }
620                         // If not in cache or if start time differs too much,
621                         // create new process to add/replace cache value
622                         long upTime = (perfTime100nSec
623                                 - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset)) / 10_000L;
624                         long startTime = now - upTime;
625                         if (proc == null || Math.abs(startTime - proc.getStartTime()) > 200) {
626                             proc = new OSProcess();
627                             proc.setProcessID(pid);
628                             proc.setStartTime(startTime);
629                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
630                             // Adds or replaces previous
631                             this.processMap.put(pid, proc);
632                         }
633 
634                         // Update stats
635                         proc.setUpTime(upTime < 1L ? 1L : upTime);
636                         proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
637                         proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
638                         proc.setResidentSetSize(
639                                 pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
640                         proc.setParentProcessID(
641                                 pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
642                         proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
643                     }
644 
645                     // Increment to next instance
646                     perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
647                     perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
648                 }
649                 // We've found the process object and are done, no need to look
650                 // at any other objects (shouldn't be any). Break the loop
651                 break;
652             }
653             // Increment for next object (should never need this)
654             perfObjectOffset += perfObject.TotalByteLength;
655         }
656         // If this was a full update, delete any pid we didn't find from the
657         // cache.
658         if (pids == null) {
659             for (Integer pid : new HashSet<>(this.processMap.keySet())) {
660                 if (!pidsToKeep.contains(pid)) {
661                     this.processMap.remove(pid);
662                 }
663             }
664         }
665     }

709     /**
710      * Enables debug privileges for this process, required for OpenProcess() to
711      * get processes other than the current user
712      */
713     private static void enableDebugPrivilege() {
714         HANDLEByReference hToken = new HANDLEByReference();
715         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
716                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
717         if (!success) {
718             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
719             return;
720         }
721         WinNT.LUID luid = new WinNT.LUID();
722         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
723         if (!success) {
724             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
725             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
726             return;
727         }
728         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
729         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
730         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
731         if (!success) {
732             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
733         }
734         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
735     }
