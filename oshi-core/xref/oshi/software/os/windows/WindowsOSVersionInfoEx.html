44  /**
45   * <p>
46   * WindowsOSVersionInfoEx class.
47   * </p>
48   *
49   * @deprecated Use {@link OperatingSystem.OSVersionInfo}
50   */
51  @Deprecated
52  public class WindowsOSVersionInfoEx extends AbstractOSVersionInfoEx {
53  
54      private static final Logger LOG = LoggerFactory.getLogger(WindowsOSVersionInfoEx.class);
55  
56      /**
57       * <p>
58       * Constructor for WindowsOSVersionInfoEx.
59       * </p>
60       */
61      public WindowsOSVersionInfoEx() {
62          // Populate a key-value map from WMI
63          WmiResult<OSVersionProperty> versionInfo = new Win32OperatingSystem().queryOsVersion();
64          if (versionInfo.getResultCount() < 1) {
65              handleNoVersionInfo();
66          } else {
67              // Guaranteed that versionInfo is not null and lists non-empty
68              // before calling the parse*() methods
69              int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
70              setBuildNumber(WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0));
71              setVersion(parseVersion(versionInfo, suiteMask));
72              setCodeName(parseCodeName(suiteMask));
73              LOG.debug("Initialized OSVersionInfoEx");
74          }
75      }
76  
77      /**
78       * <p>
79       * handleNoVersionInfo.
80       * </p>
81       */
82      protected void handleNoVersionInfo() {
83          LOG.warn("No version data available.");
84          setVersion(System.getProperty("os.version"));
85          setCodeName("");
86          setBuildNumber("");
87      }
88  
89      /**
90       * Gets the operating system version
91       *
92       * @param suiteMask
93       *
94       * @return Version
95       */
96      private String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask) {
97  
98          // Initialize a default, sane value
99          String version = System.getProperty("os.version");
100 
101         // Version is major.minor.build. Parse the version string for
102         // major/minor and get the build number separately
103         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
104         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
105         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
106 
107         // see
108         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
109         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
110                 0) == WinNT.VER_NT_WORKSTATION;
111         switch (major) {
112         case 10:
113             if (minor == 0) {
114                 if (ntWorkstation) {
115                     version = "10";
116                 } else {
117                     // Build numbers greater than 17762 is Server 2019 for OS
118                     // Version 10.0
119                     version = (ParseUtil.parseLongOrDefault(getBuildNumber(), 0L) > 17762) ? "Server 2019"
120                             : "Server 2016";
121                 }
122             }
123             break;
124         case 6:
125             if (minor == 3) {
126                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
127             } else if (minor == 2) {
128                 version = ntWorkstation ? "8" : "Server 2012";
129             } else if (minor == 1) {
130                 version = ntWorkstation ? "7" : "Server 2008 R2";
131             } else if (minor == 0) {
132                 version = ntWorkstation ? "Vista" : "Server 2008";
133             }
134             break;
135         case 5:
136             if (minor == 2) {
137                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
138                     version = "Home Server";
139                 } else if (ntWorkstation) {
140                     version = "XP"; // 64 bits
141                 } else {
142                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
143                             : "Server 2003 R2";
144                 }
145             } else if (minor == 1) {
146                 version = "XP"; // 32 bits
147             } else if (minor == 0) {
148                 version = "2000";
149             }
150             break;
151         default:
152             break;
153         }
154 
155         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
156         if (!sp.isEmpty() && !"unknown".equals(sp)) {
157             version = version + " " + sp.replace("Service Pack ", "SP");
158         }
159 
160         return version;
161     }
162 
163     /**
164      * Gets suites available on the system and return as a codename
165      *
166      * @param suiteMask
167      *
168      * @return Suites
169      */
170     private String parseCodeName(int suiteMask) {
171         List<String> suites = new ArrayList<>();
172         if ((suiteMask & 0x00000002) != 0) {
173             suites.add("Enterprise");
174         }
175         if ((suiteMask & 0x00000004) != 0) {
176             suites.add("BackOffice");
177         }
178         if ((suiteMask & 0x00000008) != 0) {
179             suites.add("Communication Server");
180         }
181         if ((suiteMask & 0x00000080) != 0) {
182             suites.add("Datacenter");
183         }
184         if ((suiteMask & 0x00000200) != 0) {
185             suites.add("Home");
186         }
187         if ((suiteMask & 0x00000400) != 0) {
188             suites.add("Web Server");
189         }
190         if ((suiteMask & 0x00002000) != 0) {
191             suites.add("Storage Server");
192         }
193         if ((suiteMask & 0x00004000) != 0) {
194             suites.add("Compute Cluster");
195         }
196         // 0x8000, Home Server, is included in main version name
197         return String.join(",", suites);
198     }
