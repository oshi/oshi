127     public WindowsOSProcess(int pid, WindowsOperatingSystem os,
128             Map<Integer, ProcessPerformanceData.PerfCounterBlock> processMap, Map<Integer, WtsInfo> processWtsMap,
129             Map<Integer, ThreadPerformanceData.PerfCounterBlock> threadMap) {
130         super(pid);
131         // Save a copy of OS creating this object for later use
132         this.os = os;
133         // Initially set to match OS bitness. If 64 will check later for 32-bit process
134         this.bitness = os.getBitness();
135         updateAttributes(processMap.get(pid), processWtsMap.get(pid), threadMap);
136     }

258     @Override
259     public long getAffinityMask() {
260         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
261         if (pHandle != null) {
262             try {
263                 ULONG_PTRByReference processAffinity = new ULONG_PTRByReference();
264                 ULONG_PTRByReference systemAffinity = new ULONG_PTRByReference();
265                 if (Kernel32.INSTANCE.GetProcessAffinityMask(pHandle, processAffinity, systemAffinity)) {
266                     return Pointer.nativeValue(processAffinity.getValue().toPointer());
267                 }
268             } finally {
269                 Kernel32.INSTANCE.CloseHandle(pHandle);
270             }
271             Kernel32.INSTANCE.CloseHandle(pHandle);
272         }
273         return 0L;
274     }

281     @Override
282     public List<OSThread> getThreadDetails() {
283         // Get data from the registry if possible
284         Map<Integer, ThreadPerformanceData.PerfCounterBlock> threads = ThreadPerformanceData
285                 .buildThreadMapFromRegistry(Collections.singleton(getProcessID()));
286         // otherwise performance counters with WMI backup
287         if (threads != null) {
288             threads = ThreadPerformanceData.buildThreadMapFromPerfCounters(Collections.singleton(this.getProcessID()));
289         }
290         if (threads == null) {
291             return Collections.emptyList();
292         }
293         return threads.entrySet().stream()
294                 .map(entry -> new WindowsOSThread(getProcessID(), entry.getKey(), this.name, entry.getValue()))
295                 .collect(Collectors.toList());
296     }

298     @Override
299     public boolean updateAttributes() {
300         Set<Integer> pids = Collections.singleton(this.getProcessID());
301         // Get data from the registry if possible
302         Map<Integer, ProcessPerformanceData.PerfCounterBlock> pcb = ProcessPerformanceData
303                 .buildProcessMapFromRegistry(null);
304         // otherwise performance counters with WMI backup
305         if (pcb == null) {
306             pcb = ProcessPerformanceData.buildProcessMapFromPerfCounters(pids);
307         }
308         Map<Integer, ThreadPerformanceData.PerfCounterBlock> tcb = null;
309         if (USE_PROCSTATE_SUSPENDED) {
310             tcb = ThreadPerformanceData.buildThreadMapFromRegistry(null);
311             // otherwise performance counters with WMI backup
312             if (tcb == null) {
313                 tcb = ThreadPerformanceData.buildThreadMapFromPerfCounters(null);
314             }
315         }
316         Map<Integer, WtsInfo> wts = ProcessWtsData.queryProcessWtsMap(pids);
317         return updateAttributes(pcb.get(this.getProcessID()), wts.get(this.getProcessID()), tcb);
318     }

320     private boolean updateAttributes(ProcessPerformanceData.PerfCounterBlock pcb, WtsInfo wts,
321             Map<Integer, ThreadPerformanceData.PerfCounterBlock> threadMap) {
322         this.name = pcb.getName();
323         this.path = wts.getPath(); // Empty string for Win7+
324         this.parentProcessID = pcb.getParentProcessID();
325         this.threadCount = wts.getThreadCount();
326         this.priority = pcb.getPriority();
327         this.virtualSize = wts.getVirtualSize();
328         this.residentSetSize = pcb.getResidentSetSize();
329         this.kernelTime = wts.getKernelTime();
330         this.userTime = wts.getUserTime();
331         this.startTime = pcb.getStartTime();
332         this.upTime = pcb.getUpTime();
333         this.bytesRead = pcb.getBytesRead();
334         this.bytesWritten = pcb.getBytesWritten();
335         this.openFiles = wts.getOpenFiles();
336         this.pageFaults = pcb.getPageFaults();
337 
338         // There are only 3 possible Process states on Windows: RUNNING, SUSPENDED, or
339         // UNKNOWN. Processes are considered running unless all of their threads are
340         // SUSPENDED.
341         this.state = RUNNING;
342         if (threadMap != null) {
343             // If user hasn't enabled this in properties, we ignore
344             int pid = this.getProcessID();
345             // If any thread is NOT suspended, set running
346             for (ThreadPerformanceData.PerfCounterBlock tcb : threadMap.values()) {
347                 if (tcb.getOwningProcessID() == pid) {
348                     if (tcb.getThreadWaitReason() == 5) {
349                         this.state = SUSPENDED;
350                     } else {
351                         this.state = RUNNING;
352                         break;
353                     }
354                 }
355             }
356         }
357 
358         // Get a handle to the process for various extended info. Only gets
359         // current user unless running as administrator
360         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
361         if (pHandle != null) {
362             try {
363                 // Test for 32-bit process on 64-bit windows
364                 if (IS_VISTA_OR_GREATER && this.bitness == 64) {
365                     IntByReference wow64 = new IntByReference(0);
366                     if (Kernel32.INSTANCE.IsWow64Process(pHandle, wow64) && wow64.getValue() > 0) {
367                         this.bitness = 32;
368                     }
369                 }
370                 // Full path
371                 final HANDLEByReference phToken = new HANDLEByReference();
372                 try { // EXECUTABLEPATH
373                     if (IS_WINDOWS7_OR_GREATER) {
374                         this.path = Kernel32Util.QueryFullProcessImageName(pHandle, 0);
375                     }
376                 } catch (Win32Exception e) {
377                     this.state = INVALID;
378                 } finally {
379                     final HANDLE token = phToken.getValue();
380                     if (token != null) {
381                         Kernel32.INSTANCE.CloseHandle(token);
382                     }
383                 }
384             } finally {
385                 Kernel32.INSTANCE.CloseHandle(pHandle);
386             }
387         }
388 
389         return !this.state.equals(INVALID);
390     }

392     private String queryCommandLine() {
393         // Try to fetch from process memory
394         if (!cwdCmdEnv.get().getB().isEmpty()) {
395             return cwdCmdEnv.get().getB();
396         }
397         // If using batch mode fetch from WMI Cache
398         if (USE_BATCH_COMMANDLINE) {
399             return Win32ProcessCached.getInstance().getCommandLine(getProcessID(), getStartTime());
400         }
401         // If no cache enabled, query line by line
402         WmiResult<CommandLineProperty> commandLineProcs = Win32Process
403                 .queryCommandLines(Collections.singleton(getProcessID()));
404         if (commandLineProcs.getResultCount() > 0) {
405             return WmiUtil.getString(commandLineProcs, CommandLineProperty.COMMANDLINE, 0);
406         }
407         return "";
408     }

418     private String queryCwd() {
419         // Try to fetch from process memory
420         if (!cwdCmdEnv.get().getA().isEmpty()) {
421             return cwdCmdEnv.get().getA();
422         }
423         // For executing process, set CWD
424         if (getProcessID() == this.os.getProcessId()) {
425             String cwd = new File(".").getAbsolutePath();
426             // trim off trailing "."
427             if (!cwd.isEmpty()) {
428                 return cwd.substring(0, cwd.length() - 1);
429             }
430         }
431         return "";
432     }

434     private Pair<String, String> queryUserInfo() {
435         Pair<String, String> pair = null;
436         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
437         if (pHandle != null) {
438             final HANDLEByReference phToken = new HANDLEByReference();
439             try {
440                 if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY, phToken)) {
441                     Account account = Advapi32Util.getTokenAccount(phToken.getValue());
442                     pair = new Pair<>(account.name, account.sidString);
443                 } else {
444                     int error = Kernel32.INSTANCE.GetLastError();
445                     // Access denied errors are common. Fail silently.
446                     if (error != WinError.ERROR_ACCESS_DENIED) {
447                         LOG.error("Failed to get process token for process {}: {}", getProcessID(),
448                                 Kernel32.INSTANCE.GetLastError());
449                     }
450                 }
451             } catch (Win32Exception e) {
452                 LOG.warn("Failed to query user info for process {} ({}): {}", getProcessID(), getName(),
453                         e.getMessage());
454             } finally {
455                 final HANDLE token = phToken.getValue();
456                 if (token != null) {
457                     Kernel32.INSTANCE.CloseHandle(token);
458                 }
459                 Kernel32.INSTANCE.CloseHandle(pHandle);
460             }
461         }
462         if (pair == null) {
463             return new Pair<>(Constants.UNKNOWN, Constants.UNKNOWN);
464         }
465         return pair;
466     }

468     private Pair<String, String> queryGroupInfo() {
469         Pair<String, String> pair = null;
470         final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false, getProcessID());
471         if (pHandle != null) {
472             final HANDLEByReference phToken = new HANDLEByReference();
473             if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY, phToken)) {
474                 Account account = Advapi32Util.getTokenPrimaryGroup(phToken.getValue());
475                 pair = new Pair<>(account.name, account.sidString);
476             } else {
477                 int error = Kernel32.INSTANCE.GetLastError();
478                 // Access denied errors are common. Fail silently.
479                 if (error != WinError.ERROR_ACCESS_DENIED) {
480                     LOG.error("Failed to get process token for process {}: {}", getProcessID(),
481                             Kernel32.INSTANCE.GetLastError());
482                 }
483             }
484             final HANDLE token = phToken.getValue();
485             if (token != null) {
486                 Kernel32.INSTANCE.CloseHandle(token);
487             }
488             Kernel32.INSTANCE.CloseHandle(pHandle);
489         }
490         if (pair == null) {
491             return new Pair<>(Constants.UNKNOWN, Constants.UNKNOWN);
492         }
493         return pair;
494     }

496     private Triplet<String, String, Map<String, String>> queryCwdCommandlineEnvironment() {
497         // Get the process handle
498         HANDLE h = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION | WinNT.PROCESS_VM_READ, false,
499                 getProcessID());
500         if (h != null) {
501             try {
502                 // Can't check 32-bit procs from a 64-bit one
503                 if (WindowsOperatingSystem.isX86() == WindowsOperatingSystem.isWow(h)) {
504 
505                     IntByReference nRead = new IntByReference();
506 
507                     // Start by getting the address of the PEB
508                     NtDll.PROCESS_BASIC_INFORMATION pbi = new NtDll.PROCESS_BASIC_INFORMATION();
509                     int ret = NtDll.INSTANCE.NtQueryInformationProcess(h, NtDll.PROCESS_BASIC_INFORMATION,
510                             pbi.getPointer(), pbi.size(), nRead);
511                     if (ret != 0) {
512                         return defaultCwdCommandlineEnvironment();
513                     }
514                     pbi.read();
515 
516                     // Now fetch the PEB
517                     NtDll.PEB peb = new NtDll.PEB();
518                     Kernel32.INSTANCE.ReadProcessMemory(h, pbi.PebBaseAddress, peb.getPointer(), peb.size(), nRead);
519                     if (nRead.getValue() == 0) {
520                         return defaultCwdCommandlineEnvironment();
521                     }
522                     peb.read();
523 
524                     // Now fetch the Process Parameters structure containing our data
525                     NtDll.RTL_USER_PROCESS_PARAMETERS upp = new NtDll.RTL_USER_PROCESS_PARAMETERS();
526                     Kernel32.INSTANCE.ReadProcessMemory(h, peb.ProcessParameters, upp.getPointer(), upp.size(), nRead);
527                     if (nRead.getValue() == 0) {
528                         return defaultCwdCommandlineEnvironment();
529                     }
530                     upp.read();
531 
532                     // Get CWD and Command Line strings here
533                     String cwd = readUnicodeString(h, upp.CurrentDirectory.DosPath);
534                     String cl = readUnicodeString(h, upp.CommandLine);
535 
536                     // Fetch the Environment Strings
537                     int envSize = upp.EnvironmentSize.intValue();
538                     if (envSize > 0) {
539                         Memory buffer = new Memory(envSize);
540                         Kernel32.INSTANCE.ReadProcessMemory(h, upp.Environment, buffer, envSize, nRead);
541                         if (nRead.getValue() > 0) {
542                             char[] env = buffer.getCharArray(0, envSize / 2);
543                             Map<String, String> envMap = ParseUtil.parseCharArrayToStringMap(env);
544                             // First entry in Environment is "=::=::\"
545                             envMap.remove("");
546                             return new Triplet<>(cwd, cl, Collections.unmodifiableMap(envMap));
547                         }
548                     }
549                     return new Triplet<>(cwd, cl, Collections.emptyMap());
550                 }
551             } finally {
552                 Kernel32.INSTANCE.CloseHandle(h);
553             }
554         }
555         return defaultCwdCommandlineEnvironment();
556     }
557 
558     private static Triplet<String, String, Map<String, String>> defaultCwdCommandlineEnvironment() {
559         return new Triplet<>("", "", Collections.emptyMap());
560     }
561 
562     private static String readUnicodeString(HANDLE h, UNICODE_STRING s) {
563         IntByReference nRead = new IntByReference();
564         if (s.Length > 0) {
565             // Add space for null terminator
566             Memory m = new Memory(s.Length + 2L);
567             m.clear(); // really only need null in last 2 bytes but this is easier
568             Kernel32.INSTANCE.ReadProcessMemory(h, s.Buffer, m, s.Length, nRead);
569             if (nRead.getValue() > 0) {
570                 return m.getWideString(0);
571             }
572         }
573         return "";
574     }
