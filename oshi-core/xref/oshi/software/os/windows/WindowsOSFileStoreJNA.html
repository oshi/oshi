5   package oshi.software.os.windows;

7   import java.util.List;
8   
9   import com.sun.jna.platform.win32.Kernel32;
10  import com.sun.jna.platform.win32.WinNT;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.software.common.os.windows.WindowsOSFileStore;
14  import oshi.software.os.OSFileStore;
15  
16  /**
17   * JNA-based Windows OSFileStore implementation.
18   */
19  @ThreadSafe
20  public class WindowsOSFileStoreJNA extends WindowsOSFileStore {
21  
22      public WindowsOSFileStoreJNA(String name, String volume, String label, String mount, String options, String uuid,
23              boolean local, String logicalVolume, String description, String fsType, long freeSpace, long usableSpace,
24              long totalSpace, long freeInodes, long totalInodes) {
25          super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
26                  usableSpace, totalSpace, freeInodes, totalInodes);
27      }
28  
29      @Override
30      public boolean updateAttributes() {
31          // Fast path: call GetDiskFreeSpaceEx directly on the known volume
32          WinNT.LARGE_INTEGER userFreeBytes = new WinNT.LARGE_INTEGER(0L);
33          WinNT.LARGE_INTEGER totalBytes = new WinNT.LARGE_INTEGER(0L);
34          WinNT.LARGE_INTEGER systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
35          if (Kernel32.INSTANCE.GetDiskFreeSpaceEx(getVolume(), userFreeBytes, totalBytes, systemFreeBytes)) {
36              updateSpace(systemFreeBytes.getValue(), userFreeBytes.getValue(), totalBytes.getValue());
37              return true;
38          }
39          // Fall back to full enumeration
40          List<OSFileStore> volumes;
41          if (isLocal()) {
42              volumes = WindowsFileSystemJNA.getLocalVolumes(getVolume());
43          } else {
44              String nameToMatch = getMount().length() < 2 ? null : getMount().substring(0, 2);
45              volumes = WindowsFileSystemJNA.getWmiVolumes(nameToMatch, false);
46          }
47          for (OSFileStore fileStore : volumes) {
48              if (getVolume().equals(fileStore.getVolume()) && getMount().equals(fileStore.getMount())) {
49                  updateFrom(fileStore);
50                  return true;
51              }
52          }
53          return false;
54      }
55  }
