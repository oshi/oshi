24  package oshi.software.common;
25  
26  import java.util.List;
27  
28  import oshi.driver.unix.NetStat;
29  import oshi.software.os.InternetProtocolStats;
30  
31  /**
32   * Common implementations for IP Stats
33   */
34  public abstract class AbstractInternetProtocolStats implements InternetProtocolStats {
35  
36      public AbstractInternetProtocolStats() {
37          super();
38      }
39  
40      @Override
41      public TcpStats getTCPv6Stats() {
42          // Default when OS doesn't have separate TCPv6 stats
43          return new TcpStats(0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L);
44      }
45  
46      @Override
47      public UdpStats getUDPv6Stats() {
48          // Default when OS doesn't have separate UDPv6 stats
49          return new UdpStats(0L, 0L, 0L, 0L);
50      }
51  
52      @Override
53      public List<IPConnection> getConnections() {
54          return NetStat.queryNetstat();
55      }
56  }
