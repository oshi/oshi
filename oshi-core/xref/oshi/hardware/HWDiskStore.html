24  /**
25   * A storage mechanism where data are recorded by various electronic, magnetic,
26   * optical, or mechanical changes to a surface layer of one or more rotating
27   * disks or or flash storage such as a removable or solid state drive. In
28   * constrast to a File System, defining the way an Operating system uses the
29   * storage, the Disk Store represents the hardware which a FileSystem uses for
30   * its File Stores.
31   *
32   * @author enrico[dot]bianchi[at]gmail[dot]com
33   */
34  public class HWDiskStore implements Serializable, Comparable<HWDiskStore> {
35

41      private long size;
42      private long reads;
43      private long readBytes;
44      private long writes;
45      private long writeBytes;
46      private long transferTime;
47      private HWPartition[] partitions;
48      private long timeStamp;

50      /**
51       * Create an object with empty/default values
52       */
53      public HWDiskStore() {
54          this("", "", "", 0L, 0L, 0L, 0L, 0L, 0L, new HWPartition[0], 0L);
55      }

57      /**
58       * Create an object with all values
59       *
60       * @param name
61       *            Name of the disk (e.g., /dev/disk1)
62       * @param model
63       *            Model of the disk
64       * @param serial
65       *            Disk serial number, if available
66       * @param size
67       *            Disk capacity in bytes
68       * @param reads
69       *            Number of reads from the disk
70       * @param readBytes
71       *            Number of bytes read from the disk
72       * @param writes
73       *            Number of writes to the disk
74       * @param writeBytes
75       *            Number of bytes written to the disk
76       * @param transferTime
77       *            milliseconds spent reading or writing to the disk
78       * @param partitions
79       *            Partitions on this disk
80       * @param timeStamp
81       *            milliseconds since the epoch
82       */
83      public HWDiskStore(String name, String model, String serial, long size, long reads, long readBytes, long writes,
84              long writeBytes, long transferTime, HWPartition[] partitions, long timeStamp) {
85          setName(name);
86          setModel(model);
87          setSerial(serial);
88          setSize(size);
89          setReads(reads);
90          setReadBytes(readBytes);
91          setWrites(writes);
92          setWriteBytes(writeBytes);
93          setTransferTime(transferTime);
94          setPartitions(partitions);
95          setTimeStamp(timeStamp);
96      }

272     /**
273      * {@inheritDoc}
274      */
275     @Override
276     public int hashCode() {
277         final int prime = 31;
278         int result = 1;
279         result = prime * result + (this.model == null ? 0 : this.model.hashCode());
280         result = prime * result + (this.name == null ? 0 : this.name.hashCode());
281         result = prime * result + Arrays.hashCode(this.partitions);
282         result = prime * result + (this.serial == null ? 0 : this.serial.hashCode());
283         result = prime * result + (int) (this.size ^ this.size >>> 32);
284         return result;
285     }

287     /**
288      * {@inheritDoc}
289      */
290     @Override
291     public boolean equals(Object obj) {
292         if (this == obj) {
293             return true;
294         }
295         if (obj == null) {
296             return false;
297         }
298         if (!(obj instanceof HWDiskStore)) {
299             return false;
300         }
301         HWDiskStore other = (HWDiskStore) obj;
302         if (this.model == null) {
303             if (other.model != null) {
304                 return false;
305             }
306         } else if (!this.model.equals(other.model)) {
307             return false;
308         }
309         if (this.name == null) {
310             if (other.name != null) {
311                 return false;
312             }
313         } else if (!this.name.equals(other.name)) {
314             return false;
315         }
316         if (!Arrays.equals(this.partitions, other.partitions)) {
317             return false;
318         }
319         if (this.serial == null) {
320             if (other.serial != null) {
321                 return false;
322             }
323         } else if (!this.serial.equals(other.serial)) {
324             return false;
325         }
326         return this.size == other.size;
327     }
