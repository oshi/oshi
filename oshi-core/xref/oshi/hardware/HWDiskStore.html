99      /**
100      * Make a best effort to update all the statistics about the drive without
101      * needing to recreate the drive list. This method provides for more
102      * frequent periodic updates of drive statistics. It will not detect if a
103      * removable drive has been removed and replaced by a different drive in
104      * between method calls.
105      *
106      * @return True if the update was (probably) successful, false if the disk
107      *         was not found
108      */
109     public boolean updateAtrributes() {
110         switch (SystemInfo.getCurrentPlatformEnum()) {
111         case WINDOWS:
112             return WindowsDisks.updateDiskStats(this);
113         case LINUX:
114             return LinuxDisks.updateDiskStats(this);
115         case MACOSX:
116             return MacDisks.updateDiskStats(this);
117         case SOLARIS:
118             return SolarisDisks.updateDiskStats(this);
119         case FREEBSD:
120             return FreeBsdDisks.updateDiskStats(this);
121         default:
122             LOG.error("Unsupported platform. No update performed.");
123             break;
124         }
125         return false;
126     }
