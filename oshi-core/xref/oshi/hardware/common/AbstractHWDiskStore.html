24  package oshi.hardware.common;
25  
26  import oshi.annotation.concurrent.ThreadSafe;
27  import oshi.hardware.HWDiskStore;
28  import oshi.util.FormatUtil;
29  
30  /**
31   * Common methods for platform HWDiskStore classes
32   */
33  @ThreadSafe
34  public abstract class AbstractHWDiskStore implements HWDiskStore {
35  
36      private final String name;
37      private final String model;
38      private final String serial;
39      private final long size;
40  
41      protected AbstractHWDiskStore(String name, String model, String serial, long size) {
42          this.name = name;
43          this.model = model;
44          this.serial = serial;
45          this.size = size;
46      }
47  
48      @Override
49      public String getName() {
50          return this.name;
51      }
52  
53      @Override
54      public String getModel() {
55          return this.model;
56      }
57  
58      @Override
59      public String getSerial() {
60          return this.serial;
61      }
62  
63      @Override
64      public long getSize() {
65          return this.size;
66      }
67  
68      @Override
69      public String toString() {
70          boolean readwrite = getReads() > 0 || getWrites() > 0;
71          StringBuilder sb = new StringBuilder();
72          sb.append(getName()).append(": ");
73          sb.append("(model: ").append(getModel());
74          sb.append(" - S/N: ").append(getSerial()).append(") ");
75          sb.append("size: ").append(getSize() > 0 ? FormatUtil.formatBytesDecimal(getSize()) : "?").append(", ");
76          sb.append("reads: ").append(readwrite ? getReads() : "?");
77          sb.append(" (").append(readwrite ? FormatUtil.formatBytes(getReadBytes()) : "?").append("), ");
78          sb.append("writes: ").append(readwrite ? getWrites() : "?");
79          sb.append(" (").append(readwrite ? FormatUtil.formatBytes(getWriteBytes()) : "?").append("), ");
80          sb.append("xfer: ").append(readwrite ? getTransferTime() : "?");
81          return sb.toString();
82      }
83  }
