24  package oshi.hardware.common;
25  
26  import oshi.annotation.concurrent.Immutable;
27  import oshi.hardware.SoundCard;
28  
29  /**
30   * An abstract Sound Card
31   */
32  @Immutable
33  public abstract class AbstractSoundCard implements SoundCard {
34  
35      private String kernelVersion;
36      private String name;
37      private String codec;
38  
39      /**
40       * Abstract Sound Card Constructor
41       *
42       * @param kernelVersion
43       *            The version
44       * @param name
45       *            The name
46       * @param codec
47       *            The codec
48       */
49      protected AbstractSoundCard(String kernelVersion, String name, String codec) {
50          this.kernelVersion = kernelVersion;
51          this.name = name;
52          this.codec = codec;
53      }
54  
55      @Override
56      public String getDriverVersion() {
57          return this.kernelVersion;
58      }
59  
60      @Override
61      public String getName() {
62          return this.name;
63      }
64  
65      @Override
66      public String getCodec() {
67          return this.codec;
68      }
69  
70      @Override
71      public String toString() {
72          StringBuilder builder = new StringBuilder();
73          builder.append("SoundCard@");
74          builder.append(Integer.toHexString(hashCode()));
75          builder.append(" [name=");
76          builder.append(this.name);
77          builder.append(", kernelVersion=");
78          builder.append(this.kernelVersion);
79          builder.append(", codec=");
80          builder.append(this.codec);
81          builder.append(']');
82          return builder.toString();
83      }
84  
85  }
