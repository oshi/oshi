5   package oshi.hardware.platform.unix.freebsd;
6   
7   import com.sun.jna.Memory;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.platform.unix.freebsd.FreeBsdCentralProcessor;
11  import oshi.jna.platform.unix.FreeBsdLibc;
12  import oshi.util.platform.unix.freebsd.BsdSysctlUtil;
13  
14  /**
15   * A CPU
16   */
17  @ThreadSafe
18  public class FreeBsdCentralProcessorJNA extends FreeBsdCentralProcessor {
19  
20      @Override
21      protected String sysctlString(String name, String def) {
22          return BsdSysctlUtil.sysctl(name, def);
23      }
24  
25      @Override
26      protected long sysctlLong(String name, long def) {
27          return BsdSysctlUtil.sysctl(name, def);
28      }
29  
30      @Override
31      protected int sysctlInt(String name, int def) {
32          return BsdSysctlUtil.sysctl(name, def);
33      }
34  
35      @Override
36      protected long[] queryCpTimes(String name) {
37          try (Memory p = BsdSysctlUtil.sysctl(name)) {
38              if (p == null) {
39                  return null;
40              }
41              int count = (int) (p.size() / Long.BYTES);
42              long[] result = new long[count];
43              for (int i = 0; i < count; i++) {
44                  result[i] = p.getLong((long) i * Long.BYTES);
45              }
46              return result;
47          }
48      }
49  
50      @Override
51      protected int getloadavgNative(double[] loadavg, int nelem) {
52          return FreeBsdLibc.INSTANCE.getloadavg(loadavg, nelem);
53      }
54  }
