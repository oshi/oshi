27  /**
28   * FreeBSD hard disk implementation.
29   */
30  @ThreadSafe
31  public final class FreeBsdHWDiskStoreJNA extends FreeBsdHWDiskStore {
32  
33      private FreeBsdHWDiskStoreJNA(String name, String model, String serial, long size) {
34          super(name, model, serial, size);
35      }
36  
37      @Override
38      public boolean updateAttributes() {
39          List<String> output = ExecutingCommand.runNative("iostat -Ix " + getName());
40          long now = System.currentTimeMillis();
41          boolean diskFound = false;
42          for (String line : output) {
43              String[] split = ParseUtil.whitespaces.split(line);
44              if (split.length < 7 || !split[0].equals(getName())) {
45                  continue;
46              }
47              diskFound = true;
48              setReads((long) ParseUtil.parseDoubleOrDefault(split[1], 0d));
49              setWrites((long) ParseUtil.parseDoubleOrDefault(split[2], 0d));
50              // In KB
51              setReadBytes((long) (ParseUtil.parseDoubleOrDefault(split[3], 0d) * 1024));
52              setWriteBytes((long) (ParseUtil.parseDoubleOrDefault(split[4], 0d) * 1024));
53              // # transactions
54              setCurrentQueueLength(ParseUtil.parseLongOrDefault(split[5], 0L));
55              // In seconds, multiply for ms
56              setTransferTime((long) (ParseUtil.parseDoubleOrDefault(split[6], 0d) * 1000));
57              setTimeStamp(now);
58          }
59          return diskFound;
60      }
61  
62      /**
63       * Gets the disks on this machine
64       *
65       * @return a list of {@link HWDiskStore} objects representing the disks
66       */
67      public static List<HWDiskStore> getDisks() {
68          // Result
69          List<HWDiskStore> diskList = new ArrayList<>();
70  
71          // Get map of disk names to partitions
72          Map<String, List<HWPartition>> partitionMap = GeomPartList.queryPartitions();
73  
74          // Get map of disk names to disk info
75          Map<String, Triplet<String, String, Long>> diskInfoMap = GeomDiskList.queryDisks();
76  
77          // Get list of disks from sysctl
78          List<String> devices = Arrays.asList(ParseUtil.whitespaces.split(BsdSysctlUtil.sysctl("kern.disks", "")));
79  
80          // Run iostat -Ix to enumerate disks by name and get kb r/w
81          List<String> iostat = ExecutingCommand.runNative("iostat -Ix");
82          long now = System.currentTimeMillis();
83          for (String line : iostat) {
84              String[] split = ParseUtil.whitespaces.split(line);
85              if (split.length > 6 && devices.contains(split[0])) {
86                  Triplet<String, String, Long> storeInfo = diskInfoMap.get(split[0]);
87                  FreeBsdHWDiskStoreJNA store = (storeInfo == null)
88                          ? new FreeBsdHWDiskStoreJNA(split[0], Constants.UNKNOWN, Constants.UNKNOWN, 0L)
89                          : new FreeBsdHWDiskStoreJNA(split[0], storeInfo.getA(), storeInfo.getB(), storeInfo.getC());
90                  store.setReads((long) ParseUtil.parseDoubleOrDefault(split[1], 0d));
91                  store.setWrites((long) ParseUtil.parseDoubleOrDefault(split[2], 0d));
92                  // In KB
93                  store.setReadBytes((long) (ParseUtil.parseDoubleOrDefault(split[3], 0d) * 1024));
94                  store.setWriteBytes((long) (ParseUtil.parseDoubleOrDefault(split[4], 0d) * 1024));
95                  // # transactions
96                  store.setCurrentQueueLength(ParseUtil.parseLongOrDefault(split[5], 0L));
97                  // In seconds, multiply for ms
98                  store.setTransferTime((long) (ParseUtil.parseDoubleOrDefault(split[6], 0d) * 1000));
99                  store.setPartitionList(Collections
100                         .unmodifiableList(partitionMap.getOrDefault(split[0], Collections.emptyList()).stream()
101                                 .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList())));
102                 store.setTimeStamp(now);
103                 diskList.add(store);
104             }
105         }
106         return diskList;
107     }
