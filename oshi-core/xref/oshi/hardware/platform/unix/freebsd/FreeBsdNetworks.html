19  package oshi.hardware.platform.unix.freebsd;
20  
21  import oshi.hardware.NetworkIF;
22  import oshi.hardware.common.AbstractNetworks;
23  import oshi.util.ExecutingCommand;
24  import oshi.util.ParseUtil;
25  
26  /**
27   * @author widdis[at]gmail[dot]com
28   */
29  public class FreeBsdNetworks extends AbstractNetworks {
30  
31      private static final long serialVersionUID = 1L;
32  
33      /**
34       * Updates interface network statistics on the given interface. Statistics
35       * include packets and bytes sent and received, and interface speed.
36       *
37       * @param netIF
38       *            The interface on which to update statistics
39       */
40      public static void updateNetworkStats(NetworkIF netIF) {
41          String stats = ExecutingCommand.getAnswerAt("netstat -bI " + netIF.getName(), 1);
42          netIF.setTimeStamp(System.currentTimeMillis());
43          String[] split = ParseUtil.whitespaces.split(stats);
44          if (split.length < 12) {
45              // No update
46              return;
47          }
48          netIF.setBytesSent(ParseUtil.parseUnsignedLongOrDefault(split[10], 0L));
49          netIF.setBytesRecv(ParseUtil.parseUnsignedLongOrDefault(split[7], 0L));
50          netIF.setPacketsSent(ParseUtil.parseUnsignedLongOrDefault(split[8], 0L));
51          netIF.setPacketsRecv(ParseUtil.parseUnsignedLongOrDefault(split[4], 0L));
52          netIF.setOutErrors(ParseUtil.parseUnsignedLongOrDefault(split[9], 0L));
53          netIF.setInErrors(ParseUtil.parseUnsignedLongOrDefault(split[5], 0L));
54      }
55  }
