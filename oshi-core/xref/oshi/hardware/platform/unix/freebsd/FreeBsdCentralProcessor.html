58      private static final Pattern CPUMASK = Pattern.compile(".*<cpu\\s.*mask=\"(?:0x)?(\\p{XDigit}+)\".*>.*</cpu>.*");
59  
60      private static final Pattern CPUINFO = Pattern
61              .compile("Origin=\"([^\"]*)\".*Id=(\\S+).*Family=(\\S+).*Model=(\\S+).*Stepping=(\\S+).*");
62      private static final Pattern CPUINFO2 = Pattern.compile("Features=(\\S+)<.*");
63  
64      private static final long BOOTTIME;
65      static {
66          Timeval tv = new Timeval();
67          if (!BsdSysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
68              // Usually this works. If it doesn't, fall back to text parsing.
69              // Boot time will be the first consecutive string of digits.
70              BOOTTIME = ParseUtil.parseLongOrDefault(
71                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
72                      System.currentTimeMillis() / 1000);
73          } else {
74              // tv now points to a 128-bit timeval structure for boot time.
75              // First 8 bytes are seconds, second 8 bytes are microseconds
76              // (we ignore)
77              BOOTTIME = tv.tv_sec;
78          }
79      }

82       * Create a Processor
83       */
84      public FreeBsdCentralProcessor() {
85          super();
86          // Initialize class variables
87          initVars();
88          // Initialize tick arrays
89          initTicks();

91          LOG.debug("Initialized Processor");
92      }

94      private void initVars() {
95          setName(BsdSysctlUtil.sysctl("hw.model", ""));
96          // This is apparently the only reliable source for this stuff on
97          // FreeBSD...
98          long processorID = 0L;
99          List<String> cpuInfo = FileUtil.readFile("/var/run/dmesg.boot");
100         for (String line : cpuInfo) {
101             line = line.trim();
102             // Prefer hw.model to this one
103             if (line.startsWith("CPU:") && getName().isEmpty()) {
104                 setName(line.replace("CPU:", "").trim());
105             } else if (line.startsWith("Origin=")) {
106                 Matcher m = CPUINFO.matcher(line);
107                 if (m.matches()) {
108                     setVendor(m.group(1));
109                     processorID |= Long.decode(m.group(2));
110                     setFamily(Integer.decode(m.group(3)).toString());
111                     setModel(Integer.decode(m.group(4)).toString());
112                     setStepping(Integer.decode(m.group(5)).toString());
113                 }
114             } else if (line.startsWith("Features=")) {
115                 Matcher m = CPUINFO2.matcher(line);
116                 if (m.matches()) {
117                     processorID |= Long.decode(m.group(1)) << 32;
118                 }
119                 // No further interest in this file
120                 break;
121             }
122         }
123         setCpu64(ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64"));
124         setProcessorID(getProcessorID(processorID));
125     }

127     /**
128      * Updates logical and physical processor/package counts
129      */
130     @Override
131     protected void calculateProcessorCounts() {
132         String[] topology = BsdSysctlUtil.sysctl("kern.sched.topology_spec", "").split("\\n|\\r");
133         int physMask = 0;
134         int virtMask = 0;
135         int lastMask = 0;
136         int physPackage = 0;
137         for (String topo : topology) {
138             if (topo.contains("<cpu")) {
139                 // Find <cpu> tag and extract bits
140                 Matcher m = CPUMASK.matcher(topo);
141                 if (m.matches()) {
142                     // Add this processor mask to cpus. Regex guarantees parsing
143                     lastMask = Integer.parseInt(m.group(1), 16);
144                     physMask |= lastMask;
145                     virtMask |= lastMask;
146                 }
147             } else if (topo.contains("<flags>")
148                     && (topo.contains("HTT") || topo.contains("SMT") || topo.contains("THREAD"))) {
149                 // These are virtual cpus, remove processor mask from physical
150                 physMask &= ~lastMask;
151             } else if (topo.contains("<group level=\"2\"")) {
152                 // This is a physical package
153                 physPackage++;
154             }
155         }
156 
157         this.logicalProcessorCount = Integer.bitCount(virtMask);
158         if (this.logicalProcessorCount < 1) {
159             LOG.error("Couldn't find logical processor count. Assuming 1.");
160             this.logicalProcessorCount = 1;
161         }
162         this.physicalProcessorCount = Integer.bitCount(physMask);
163         if (this.physicalProcessorCount < 1) {
164             LOG.error("Couldn't find physical processor count. Assuming 1.");
165             this.physicalProcessorCount = 1;
166         }
167         this.physicalPackageCount = physPackage;
168         if (this.physicalPackageCount < 1) {
169             LOG.error("Couldn't find physical package count. Assuming 1.");
170             this.physicalPackageCount = 1;
171         }
172     }

174     /**
175      * {@inheritDoc}
176      */
177     @Override
178     public synchronized long[] getSystemCpuLoadTicks() {
179         long[] ticks = new long[TickType.values().length];
180         CpTime cpTime = new CpTime();
181         BsdSysctlUtil.sysctl("kern.cp_time", cpTime);
182         ticks[TickType.USER.getIndex()] = cpTime.cpu_ticks[Libc.CP_USER];
183         ticks[TickType.NICE.getIndex()] = cpTime.cpu_ticks[Libc.CP_NICE];
184         ticks[TickType.SYSTEM.getIndex()] = cpTime.cpu_ticks[Libc.CP_SYS];
185         ticks[TickType.IRQ.getIndex()] = cpTime.cpu_ticks[Libc.CP_INTR];
186         ticks[TickType.IDLE.getIndex()] = cpTime.cpu_ticks[Libc.CP_IDLE];
187         return ticks;
188     }

190     /**
191      * {@inheritDoc}
192      */
193     @Override
194     public double[] getSystemLoadAverage(int nelem) {
195         if (nelem < 1 || nelem > 3) {
196             throw new IllegalArgumentException("Must include from one to three elements.");
197         }
198         double[] average = new double[nelem];
199         int retval = Libc.INSTANCE.getloadavg(average, nelem);
200         if (retval < nelem) {
201             for (int i = Math.max(retval, 0); i < average.length; i++) {
202                 average[i] = -1d;
203             }
204         }
205         return average;
206     }

208     /**
209      * {@inheritDoc}
210      */
211     @Override
212     public long[][] getProcessorCpuLoadTicks() {
213         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
214 
215         // Allocate memory for array of CPTime
216         long size = new CpTime().size();
217         long arraySize = size * this.logicalProcessorCount;
218         Pointer p = new Memory(arraySize);
219         String name = "kern.cp_times";
220         // Fetch
221         if (0 != Libc.INSTANCE.sysctlbyname(name, p, new IntByReference((int) arraySize), null, 0)) {
222             LOG.error("Failed syctl call: {}, Error code: {}", name, Native.getLastError());
223             return ticks;
224         }
225         // p now points to the data; need to copy each element
226         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
227             ticks[cpu][TickType.USER.getIndex()] = p.getLong(size * cpu + Libc.CP_USER * Libc.UINT64_SIZE); // lgtm
228             ticks[cpu][TickType.NICE.getIndex()] = p.getLong(size * cpu + Libc.CP_NICE * Libc.UINT64_SIZE); // lgtm
229             ticks[cpu][TickType.SYSTEM.getIndex()] = p.getLong(size * cpu + Libc.CP_SYS * Libc.UINT64_SIZE); // lgtm
230             ticks[cpu][TickType.IRQ.getIndex()] = p.getLong(size * cpu + Libc.CP_INTR * Libc.UINT64_SIZE); // lgtm
231             ticks[cpu][TickType.IDLE.getIndex()] = p.getLong(size * cpu + Libc.CP_IDLE * Libc.UINT64_SIZE); // lgtm
232         }
233         return ticks;
234     }

241         return System.currentTimeMillis() / 1000 - BOOTTIME;
242     }

244     /**
245      * Fetches the ProcessorID from dmidecode (if possible with root
246      * permissions), otherwise uses the values from /var/run/dmesg.boot
247      *
248      * @param processorID
249      * @return The ProcessorID string
250      */
251     private String getProcessorID(long processorID) {
252         boolean procInfo = false;
253         String marker = "Processor Information";
254         for (String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
255             if (!procInfo && checkLine.contains(marker)) {
256                 marker = "ID:";
257                 procInfo = true;
258             } else if (procInfo && checkLine.contains(marker)) {
259                 return checkLine.split(marker)[1].trim();
260             }
261         }
262         // If we've gotten this far, dmidecode failed. Used the passed-in values
263         return String.format("%016X", processorID);
264     }

266     /**
267      * {@inheritDoc}
268      */
269     @Override
270     public long getContextSwitches() {
271         String name = "vm.stats.sys.v_swtch";
272         IntByReference size = new IntByReference(Libc.INT_SIZE);
273         Pointer p = new Memory(size.getValue());
274         if (0 != Libc.INSTANCE.sysctlbyname(name, p, size, null, 0)) {
275             return -1;
276         }
277         return ParseUtil.unsignedIntToLong(p.getInt(0));
278     }
