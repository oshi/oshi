24  package oshi.hardware.platform.unix.openbsd;
25  
26  import static oshi.util.Memoizer.memoize;
27  
28  import java.util.List;
29  import java.util.function.Supplier;
30  
31  import oshi.annotation.concurrent.Immutable;
32  import oshi.hardware.common.AbstractFirmware;
33  import oshi.util.Constants;
34  import oshi.util.ExecutingCommand;
35  import oshi.util.ParseUtil;
36  import oshi.util.Util;
37  import oshi.util.tuples.Triplet;
38  
39  /**
40   * OpenBSD Firmware implementation
41   */
42  @Immutable
43  public class OpenBsdFirmware extends AbstractFirmware {
44      private final Supplier<Triplet<String, String, String>> manufVersRelease = memoize(OpenBsdFirmware::readDmesg);
45  
46      @Override
47      public String getManufacturer() {
48          return manufVersRelease.get().getA();
49      }
50  
51      @Override
52      public String getVersion() {
53          return manufVersRelease.get().getB();
54      }
55  
56      @Override
57      public String getReleaseDate() {
58          return manufVersRelease.get().getC();
59      }
60  
61      private static Triplet<String, String, String> readDmesg() {
62          String version = null;
63          String vendor = null;
64          String releaseDate = "";
65  
66          List<String> dmesg = ExecutingCommand.runNative("dmesg");
67          for (String line : dmesg) {
68              // bios0 at mainbus0: SMBIOS rev. 2.7 @ 0xdcc0e000 (67 entries)
69              // bios0: vendor LENOVO version "GLET90WW (2.44 )" date 09/13/2017
70              // bios0: LENOVO 20AWA08J00
71              if (line.startsWith("bios0: vendor")) {
72                  version = ParseUtil.getStringBetween(line, '"');
73                  releaseDate = ParseUtil.parseMmDdYyyyToYyyyMmDD(ParseUtil.parseLastString(line));
74                  vendor = line.split("vendor")[1].trim();
75              }
76          }
77          return new Triplet<>(Util.isBlank(vendor) ? Constants.UNKNOWN : vendor,
78                  Util.isBlank(version) ? Constants.UNKNOWN : version,
79                  Util.isBlank(releaseDate) ? Constants.UNKNOWN : releaseDate);
80      }
81  }
