109     @Override
110     public boolean updateAttributes() {
111         this.timeStamp = System.currentTimeMillis();
112         if (HAS_KSTAT2) {
113             // Use Kstat2 implementation
114             return updateAttributes2();
115         }
116         try (KstatChain kc = KstatUtil.openChain()) {
117             Kstat ksp = kc.lookup(null, 0, getName());
118             if (ksp != null && kc.read(ksp)) {
119                 KstatIO data = new KstatIO(ksp.ks_data);
120                 this.reads = data.reads;
121                 this.writes = data.writes;
122                 this.readBytes = data.nread;
123                 this.writeBytes = data.nwritten;
124                 this.currentQueueLength = (long) data.wcnt + data.rcnt;
125                 // rtime and snaptime are nanoseconds, convert to millis
126                 this.transferTime = data.rtime / 1_000_000L;
127                 this.timeStamp = ksp.ks_snaptime / 1_000_000L;
128                 return true;
129             }
130         }
131         return false;
132     }
133 
134     private boolean updateAttributes2() {
135         String fullName = getName();
136         String alpha = fullName;
137         String numeric = "";
138         for (int c = 0; c < fullName.length(); c++) {
139             if (fullName.charAt(c) >= '0' && fullName.charAt(c) <= '9') {
140                 alpha = fullName.substring(0, c);
141                 numeric = fullName.substring(c);
142                 break;
143             }
144         }
145         // Try device style notation
146         Object[] results = KstatUtil.queryKstat2("kstat:/disk/" + alpha + "/" + getName() + "/0", "reads", "writes",
147                 "nread", "nwritten", "wcnt", "rcnt", "rtime", "snaptime");
148         // If failure try io notation
149         if (results[results.length - 1] == null) {
150             results = KstatUtil.queryKstat2("kstat:/disk/" + alpha + "/" + numeric + "/io", "reads", "writes", "nread",
151                     "nwritten", "wcnt", "rcnt", "rtime", "snaptime");
152         }
153         if (results[results.length - 1] == null) {
154             return false;
155         }
156         this.reads = results[0] == null ? 0L : (long) results[0];
157         this.writes = results[1] == null ? 0L : (long) results[1];
158         this.readBytes = results[2] == null ? 0L : (long) results[2];
159         this.writeBytes = results[3] == null ? 0L : (long) results[3];
160         this.currentQueueLength = results[4] == null ? 0L : (long) results[4];
161         this.currentQueueLength += results[5] == null ? 0L : (long) results[5];
162         // rtime and snaptime are nanoseconds, convert to millis
163         this.transferTime = results[6] == null ? 0L : (long) results[6] / 1_000_000L;
164         this.timeStamp = (long) results[7] / 1_000_000L;
165         return true;
166     }

168     /**
169      * Gets the disks on this machine
170      *
171      * @return a list of {@link HWDiskStore} objects representing the disks
172      */
173     public static List<HWDiskStore> getDisks() {
174         // Create map to correlate disk name with block device mount point for
175         // later use in partition info
176         Map<String, String> deviceMap = Iostat.queryPartitionToMountMap();
177 
178         // Create map to correlate disk name with block device mount point for
179         // later use in partition info. Run lshal, if available, to get block device
180         // major (we'll use partition # for minor)
181         Map<String, Integer> majorMap = Lshal.queryDiskToMajorMap();
182 
183         // Create map of model, vendor, product, serial, size
184         // We'll use Model if available, otherwise Vendor+Product
185         Map<String, Quintet<String, String, String, String, Long>> deviceStringMap = Iostat
186                 .queryDeviceStrings(deviceMap.keySet());
187 
188         List<HWDiskStore> storeList = new ArrayList<>();
189         for (Entry<String, Quintet<String, String, String, String, Long>> entry : deviceStringMap.entrySet()) {
190             String storeName = entry.getKey();
191             Quintet<String, String, String, String, Long> val = entry.getValue();
192             storeList.add(createStore(storeName, val.getA(), val.getB(), val.getC(), val.getD(), val.getE(),
193                     deviceMap.getOrDefault(storeName, ""), majorMap.getOrDefault(storeName, 0)));
194         }
195 
196         return storeList;
197     }
198 
199     private static SolarisHWDiskStore createStore(String diskName, String model, String vendor, String product,
200             String serial, long size, String mount, int major) {
201         SolarisHWDiskStore store = new SolarisHWDiskStore(diskName,
202                 model.isEmpty() ? (vendor + " " + product).trim() : model, serial, size);
203         store.partitionList = Collections.unmodifiableList(Prtvtoc.queryPartitions(mount, major).stream()
204                 .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList()));
205         store.updateAttributes();
206         return store;
207     }
