19  package oshi.hardware.platform.unix.solaris;
20  
21  import java.util.ArrayList;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.Map.Entry;
25  
26  import oshi.hardware.HWDiskStore;
27  import oshi.hardware.common.AbstractDisks;
28  import oshi.util.ExecutingCommand;
29  import oshi.util.ParseUtil;
30  
31  /**
32   * Solaris hard disk implementation.
33   *
34   * @author widdis[at]gmail[dot]com
35   */
36  public class SolarisDisks extends AbstractDisks {
37  
38      private static final long serialVersionUID = 1L;
39  
40      @Override
41      public HWDiskStore[] getDisks() {
42          // Create map indexed by device name for multiple command reference
43          Map<String, HWDiskStore> diskMap = new HashMap<>();
44  
45          // First, run iostat -er to enumerate disks by name. Sample output:
46          ArrayList<String> disks = ExecutingCommand.runNative("iostat -er");
47          for (String line : disks) {
48              // The -r switch enables comma delimited for easy parsing!
49              String[] split = line.split(",");
50              if (split.length < 5 || split[0].equals("device")) {
51                  continue;
52              }
53              HWDiskStore store = new HWDiskStore();
54              store.setName(split[0]);
55              diskMap.put(split[0], store);
56          }
57  
58          // Next, run iostat -Er to get model, etc.
59          disks = ExecutingCommand.runNative("iostat -Er");
60          // We'll use Model if available, otherwise Vendor+Product
61          String disk = "";
62          String model = "";
63          String vendor = "";
64          String product = "";
65          String serial = "";
66          long size = 0;
67          for (String line : disks) {
68              // The -r switch enables comma delimited for easy parsing!
69              // No guarantees on which line the results appear so we'll nest
70              // a loop iterating on the comma splits
71              String[] split = line.split(",");
72              for (String keyValue : split) {
73                  keyValue = keyValue.trim();
74                  // If entry is tne name of a disk, this is beginning of new
75                  // output for that disk.
76                  if (diskMap.keySet().contains(keyValue)) {
77                      // First, if we have existing output from previous,
78                      // update
79                      if (!disk.isEmpty()) {
80                          updateStore(diskMap.get(disk), model, vendor, product, serial, size);
81                      }
82                      // Reset values for next iteration
83                      disk = keyValue;
84                      model = "";
85                      vendor = "";
86                      product = "";
87                      serial = "";
88                      size = 0L;
89                      continue;
90                  }
91                  // Otherwise update variables
92                  if (keyValue.startsWith("Model:")) {
93                      model = keyValue.replace("Model:", "").trim();
94                  } else if (keyValue.startsWith("Serial No:")) {
95                      serial = keyValue.replace("Serial No:", "").trim();
96                  } else if (keyValue.startsWith("Vendor:")) {
97                      vendor = keyValue.replace("Vendor:", "").trim();
98                  } else if (keyValue.startsWith("Product:")) {
99                      product = keyValue.replace("Product:", "").trim();
100                 } else if (keyValue.startsWith("Size:")) {
101                     // Size: 1.23GB <1227563008 bytes>
102                     String[] bytes = keyValue.split("<");
103                     if (bytes.length > 1) {
104                         bytes = bytes[1].split("\\s+");
105                         size = ParseUtil.parseLongOrDefault(bytes[0], 0L);
106                     }
107                 }
108             }
109             // At end of output update last entry
110             if (!disk.isEmpty()) {
111                 updateStore(diskMap.get(disk), model, vendor, product, serial, size);
112             }
113         }
114 
115         // Finally use kstat to get reads/writes
116         // simultaneously populate result array
117         HWDiskStore[] results = new HWDiskStore[diskMap.keySet().size()];
118         int index = 0;
119         for (Entry<String, HWDiskStore> entry : diskMap.entrySet()) {
120             ArrayList<String> stats = ExecutingCommand.runNative("kstat -p ::" + entry.getKey());
121             for (String line : stats) {
122                 String[] split = line.split("\\s+");
123                 if (split.length < 2) {
124                     continue;
125                 }
126                 if (split[0].endsWith(":reads")) {
127                     entry.getValue().setReads(ParseUtil.parseLongOrDefault(split[1], 0L));
128                 } else if (split[0].endsWith(":writes")) {
129                     entry.getValue().setWrites(ParseUtil.parseLongOrDefault(split[1], 0L));
130                 } else if (split[0].endsWith(":nread")) {
131                     entry.getValue().setReadBytes(ParseUtil.parseLongOrDefault(split[1], 0L));
132                 } else if (split[0].endsWith(":nwritten")) {
133                     entry.getValue().setWriteBytes(ParseUtil.parseLongOrDefault(split[1], 0L));
134                 } else if (split[0].endsWith(":rtime")) {
135                     entry.getValue().setTransferTime((long) (ParseUtil.parseDoubleOrDefault(split[1], 0d) * 1000));
136                 }
137             }
138             results[index++] = entry.getValue();
139         }
140 
141         return results;
142     }
143 
144     /**
145      * Updates the HWDiskStore. If model name is nonempty it is used, otherwise
146      * vendor+product are used for model
147      * 
148      * @param store
149      *            A HWDiskStore
150      * @param model
151      *            model name, or empty string if none
152      * @param vendor
153      *            vendor name, or empty string if none
154      * @param product
155      *            product nmae, or empty string if none
156      * @param serial
157      *            serial number, or empty string if none
158      * @param size
159      *            size of the drive in bytes
160      */
161     private void updateStore(HWDiskStore store, String model, String vendor, String product, String serial, long size) {
162         store.setModel(model.isEmpty() ? (vendor + " " + product).trim() : model);
163         store.setSerial(serial);
164         store.setSize(size);
165     }
166 }
