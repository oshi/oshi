36  /**
37   * Solaris hard disk implementation.
38   *
39   * @author widdis[at]gmail[dot]com
40   */
41  public class SolarisDisks implements Disks {
42  
43      private static final long serialVersionUID = 1L;
44  
45      @Override
46      public HWDiskStore[] getDisks() {
47          // Create map indexed by device name for multiple command reference
48          Map<String, HWDiskStore> diskMap = new HashMap<>();
49          // First, run iostat -er to enumerate disks by name. Sample output:
50          // errors
51          // device,s/w,h/w,trn,tot
52          // cmdk0,0,0,0,0
53          // sd0,0,0,0
54          List<String> disks = ExecutingCommand.runNative("iostat -er");
55  
56          // Create map to correlate disk name with block device mount point for
57          // later use in partition info
58          Map<String, String> deviceMap = new HashMap<>();
59          // Also run iostat -ern to get the same list by mount point. Sample
60          // output:
61          // errors
62          // s/w,h/w,trn,tot,device
63          // 0,0,0,0,c1d0
64          // 0,0,0,0,c1t1d0
65          List<String> mountpoints = ExecutingCommand.runNative("iostat -ern");
66          String disk;
67          for (int i = 0; i < disks.size() && i < mountpoints.size(); i++) {
68              // Map disk
69              disk = disks.get(i);
70              String[] diskSplit = disk.split(",");
71              if (diskSplit.length < 5 || "device".equals(diskSplit[0])) {
72                  continue;
73              }
74              HWDiskStore store = new HWDiskStore();
75              store.setName(diskSplit[0]);
76              diskMap.put(diskSplit[0], store);
77              // Map mount
78              String mount = mountpoints.get(i);
79              String[] mountSplit = mount.split(",");
80              if (mountSplit.length < 5 || "device".equals(mountSplit[4])) {
81                  continue;
82              }
83              deviceMap.put(diskSplit[0], mountSplit[4]);
84          }
85  
86          // Create map to correlate disk name with blick device mount point for
87          // later use in partition info
88          Map<String, Integer> majorMap = new HashMap<>();
89          // Run lshal, if available, to get block device major (we'll use
90          // partition # for minor)
91          List<String> lshal = ExecutingCommand.runNative("lshal");
92          disk = "";
93          for (String line : lshal) {
94              if (line.startsWith("udi ")) {
95                  String udi = ParseUtil.getSingleQuoteStringValue(line);
96                  disk = udi.substring(udi.lastIndexOf('/') + 1);
97                  continue;
98              }
99              line = line.trim();
100             if (line.startsWith("block.major")) {
101                 majorMap.put(disk, ParseUtil.getFirstIntValue(line));
102             }
103         }
104 
105         // Next, run iostat -Er to get model, etc.
106         disks = ExecutingCommand.runNative("iostat -Er");
107         // We'll use Model if available, otherwise Vendor+Product
108         disk = "";
109         String model = "";
110         String vendor = "";
111         String product = "";
112         String serial = "";
113         long size = 0;
114         for (String line : disks) {
115             // The -r switch enables comma delimited for easy parsing!
116             // No guarantees on which line the results appear so we'll nest
117             // a loop iterating on the comma splits
118             String[] split = line.split(",");
119             for (String keyValue : split) {
120                 keyValue = keyValue.trim();
121                 // If entry is tne name of a disk, this is beginning of new
122                 // output for that disk.
123                 if (diskMap.keySet().contains(keyValue)) {
124                     // First, if we have existing output from previous,
125                     // update
126                     if (!disk.isEmpty()) {
127                         updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
128                                 majorMap.getOrDefault(disk, 0));
129                     }
130                     // Reset values for next iteration
131                     disk = keyValue;
132                     model = "";
133                     vendor = "";
134                     product = "";
135                     serial = "";
136                     size = 0L;
137                     continue;
138                 }
139                 // Otherwise update variables
140                 if (keyValue.startsWith("Model:")) {
141                     model = keyValue.replace("Model:", "").trim();
142                 } else if (keyValue.startsWith("Serial No:")) {
143                     serial = keyValue.replace("Serial No:", "").trim();
144                 } else if (keyValue.startsWith("Vendor:")) {
145                     vendor = keyValue.replace("Vendor:", "").trim();
146                 } else if (keyValue.startsWith("Product:")) {
147                     product = keyValue.replace("Product:", "").trim();
148                 } else if (keyValue.startsWith("Size:")) {
149                     // Size: 1.23GB <1227563008 bytes>
150                     String[] bytes = keyValue.split("<");
151                     if (bytes.length > 1) {
152                         bytes = bytes[1].split("\\s+");
153                         size = ParseUtil.parseLongOrDefault(bytes[0], 0L);
154                     }
155                 }
156             }
157             // At end of output update last entry
158             if (!disk.isEmpty()) {
159                 updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
160                         majorMap.getOrDefault(disk, 0));
161             }
162         }
163 
164         // Finally use kstat to get reads/writes
165         // simultaneously populate result array
166         HWDiskStore[] results = new HWDiskStore[diskMap.keySet().size()];
167         int index = 0;
168         for (Entry<String, HWDiskStore> entry : diskMap.entrySet()) {
169             Kstat ksp = KstatUtil.kstatLookup(null, 0, entry.getKey());
170             if (ksp != null && KstatUtil.kstatRead(ksp)) {
171                 KstatIO data = new KstatIO(ksp.ks_data);
172                 entry.getValue().setReads(data.reads);
173                 entry.getValue().setWrites(data.writes);
174                 entry.getValue().setReadBytes(data.nread);
175                 entry.getValue().setWriteBytes(data.nwritten);
176                 // rtime and snaptime are nanoseconds, convert to millis
177                 entry.getValue().setTransferTime(data.rtime / 1000000L);
178                 entry.getValue().setTimeStamp(ksp.ks_snaptime / 1000000L);
179             }
180             results[index++] = entry.getValue();
181         }
182 
183         return results;
184     }
185 
186     /**
187      * Updates the HWDiskStore. If model name is nonempty it is used, otherwise
188      * vendor+product are used for model
189      *
190      * @param store
191      *            A HWDiskStore
192      * @param model
193      *            model name, or empty string if none
194      * @param vendor
195      *            vendor name, or empty string if none
196      * @param product
197      *            product nmae, or empty string if none
198      * @param serial
199      *            serial number, or empty string if none
200      * @param size
201      *            size of the drive in bytes
202      * @param mount
203      *            The mount point of this store, used to fetch partition info
204      * @param major
205      *            The major device number for the partition
206      */
207     private void updateStore(HWDiskStore store, String model, String vendor, String product, String serial, long size,
208             String mount, int major) {
209         store.setModel(model.isEmpty() ? (vendor + " " + product).trim() : model);
210         store.setSerial(serial);
211         store.setSize(size);
212 
213         // Temporary list to hold partitions
214         List<HWPartition> partList = new ArrayList<>();
215 
216         // Now grab prtvotc output for partitions
217         // This requires sudo permissions; will result in "permission denied
218         // otherwise" in which case we return empty partition list
219         List<String> prtvotc = ExecutingCommand.runNative("prtvtoc /dev/dsk/" + mount);
220         // Sample output - see man prtvtoc
221         if (prtvotc.size() > 1) {
222             int bytesPerSector = 0;
223             String[] split;
224             // We have a result, parse partition table
225             for (String line : prtvotc) {
226                 // If line starts with asterisk we ignore except for the one
227                 // specifying bytes per sector
228                 if (line.startsWith("*")) {
229                     if (line.endsWith("bytes/sector")) {
230                         split = line.split("\\s+");
231                         if (split.length > 0) {
232                             bytesPerSector = ParseUtil.parseIntOrDefault(split[1], 0);
233                         }
234                     }
235                     continue;
236                 }
237                 // If bytes/sector is still 0, these are not real partitions so
238                 // ignore.
239                 if (bytesPerSector == 0) {
240                     continue;
241                 }
242                 // Lines without asterisk have 6 or 7 whitespace-split values
243                 // representing (last field optional):
244                 // Partition Tag Flags Sector Count Sector Mount
245                 split = line.trim().split("\\s+");
246                 // Partition 2 is always the whole disk so we ignore it
247                 if (split.length < 6 || "2".equals(split[0])) {
248                     continue;
249                 }
250                 HWPartition partition = new HWPartition();
251                 // First field is partition number
252                 partition.setIdentification(mount + "s" + split[0]);
253                 partition.setMajor(major);
254                 partition.setMinor(ParseUtil.parseIntOrDefault(split[0], 0));
255                 // Second field is tag. Parse:
256                 switch (ParseUtil.parseIntOrDefault(split[1], 0)) {
257                 case 0x01:
258                 case 0x18:
259                     partition.setName("boot");
260                     break;
261                 case 0x02:
262                     partition.setName("root");
263                     break;
264                 case 0x03:
265                     partition.setName("swap");
266                     break;
267                 case 0x04:
268                     partition.setName("usr");
269                     break;
270                 case 0x05:
271                     partition.setName("backup");
272                     break;
273                 case 0x06:
274                     partition.setName("stand");
275                     break;
276                 case 0x07:
277                     partition.setName("var");
278                     break;
279                 case 0x08:
280                     partition.setName("home");
281                     break;
282                 case 0x09:
283                     partition.setName("altsctr");
284                     break;
285                 case 0x0a:
286                     partition.setName("cache");
287                     break;
288                 case 0x0b:
289                     partition.setName("reserved");
290                     break;
291                 case 0x0c:
292                     partition.setName("system");
293                     break;
294                 case 0x0e:
295                     partition.setName("public region");
296                     break;
297                 case 0x0f:
298                     partition.setName("private region");
299                     break;
300                 default:
301                     partition.setName("unknown");
302                     break;
303                 }
304                 // Third field is flags.
305                 // First character writable, second is mountable
306                 switch (split[2]) {
307                 case "00":
308                     partition.setType("wm");
309                     break;
310                 case "10":
311                     partition.setType("rm");
312                     break;
313                 case "01":
314                     partition.setType("wu");
315                     break;
316                 default:
317                     partition.setType("ru");
318                     break;
319                 }
320                 // Fifth field is sector count
321                 partition.setSize(bytesPerSector * ParseUtil.parseLongOrDefault(split[4], 0L));
322                 // Seventh field (if present) is mount point
323                 if (split.length > 6) {
324                     partition.setMountPoint(split[6]);
325                 }
326                 partList.add(partition);
327             }
328             store.setPartitions(partList.toArray(new HWPartition[partList.size()]));
329         }
330     }
