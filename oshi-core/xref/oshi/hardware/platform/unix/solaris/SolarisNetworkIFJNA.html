20  /**
21   * JNA-backed Solaris NetworkIF. Uses Kstat2 where available, falling back to the legacy {@code kstat} chain.
22   */
23  @ThreadSafe
24  public final class SolarisNetworkIFJNA extends SolarisNetworkIF {
25  
26      public SolarisNetworkIFJNA(NetworkInterface netint) throws InstantiationException {
27          super(netint);
28          updateAttributes();
29      }
30  
31      /**
32       * Gets all network interfaces on this machine.
33       *
34       * @param includeLocalInterfaces include local interfaces in the result
35       * @return a list of {@link NetworkIF} objects representing the interfaces
36       */
37      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
38          return getNetworks(includeLocalInterfaces, SolarisNetworkIFJNA::new);
39      }
40  
41      @Override
42      protected IfStats queryStats() {
43          if (HAS_KSTAT2) {
44              return queryStats2();
45          }
46          try (KstatChain kc = KstatUtil.openChain()) {
47              Kstat ksp = kc.lookup("link", -1, getName());
48              if (ksp == null) { // Solaris 10 compatibility
49                  ksp = kc.lookup(null, -1, getName());
50              }
51              if (ksp != null && kc.read(ksp)) {
52                  IfStats stats = new IfStats();
53                  stats.bytesSent = KstatUtil.dataLookupLong(ksp, "obytes64");
54                  stats.bytesRecv = KstatUtil.dataLookupLong(ksp, "rbytes64");
55                  stats.packetsSent = KstatUtil.dataLookupLong(ksp, "opackets64");
56                  stats.packetsRecv = KstatUtil.dataLookupLong(ksp, "ipackets64");
57                  stats.outErrors = KstatUtil.dataLookupLong(ksp, "oerrors");
58                  stats.inErrors = KstatUtil.dataLookupLong(ksp, "ierrors");
59                  stats.collisions = KstatUtil.dataLookupLong(ksp, "collisions");
60                  stats.inDrops = KstatUtil.dataLookupLong(ksp, "dl_idrops");
61                  stats.speed = KstatUtil.dataLookupLong(ksp, "ifspeed");
62                  // Snap time in ns; convert to ms
63                  stats.timeStamp = ksp.ks_snaptime / 1_000_000L;
64                  return stats;
65              }
66          }
67          return null;
68      }
69  
70      private IfStats queryStats2() {
71          Object[] results = KstatUtil.queryKstat2("kstat:/net/link/" + getName() + "/0", "obytes64", "rbytes64",
72                  "opackets64", "ipackets64", "oerrors", "ierrors", "collisions", "dl_idrops", "ifspeed", "snaptime");
73          if (results[results.length - 1] == null) {
74              return null;
75          }
76          IfStats stats = new IfStats();
77          stats.bytesSent = results[0] == null ? 0L : (long) results[0];
78          stats.bytesRecv = results[1] == null ? 0L : (long) results[1];
79          stats.packetsSent = results[2] == null ? 0L : (long) results[2];
80          stats.packetsRecv = results[3] == null ? 0L : (long) results[3];
81          stats.outErrors = results[4] == null ? 0L : (long) results[4];
82          stats.inErrors = results[5] == null ? 0L : (long) results[5];
83          stats.collisions = results[6] == null ? 0L : (long) results[6];
84          stats.inDrops = results[7] == null ? 0L : (long) results[7];
85          stats.speed = results[8] == null ? 0L : (long) results[8];
86          // Snap time in ns; convert to ms
87          stats.timeStamp = (long) results[9] / 1_000_000L;
88          return stats;
89      }
