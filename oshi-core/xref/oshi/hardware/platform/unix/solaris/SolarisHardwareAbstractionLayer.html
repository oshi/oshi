24  package oshi.hardware.platform.unix.solaris;
25  
26  import java.util.List;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  import oshi.hardware.CentralProcessor;
30  import oshi.hardware.ComputerSystem;
31  import oshi.hardware.Display;
32  import oshi.hardware.GlobalMemory;
33  import oshi.hardware.GraphicsCard;
34  import oshi.hardware.HWDiskStore;
35  import oshi.hardware.NetworkIF;
36  import oshi.hardware.PowerSource;
37  import oshi.hardware.Sensors;
38  import oshi.hardware.SoundCard;
39  import oshi.hardware.UsbDevice;
40  import oshi.hardware.common.AbstractHardwareAbstractionLayer;
41  import oshi.hardware.platform.unix.UnixDisplay;
42  
43  /**
44   * SolarisHardwareAbstractionLayer class.
45   */
46  @ThreadSafe
47  public final class SolarisHardwareAbstractionLayer extends AbstractHardwareAbstractionLayer {
48  
49      @Override
50      public ComputerSystem createComputerSystem() {
51          return new SolarisComputerSystem();
52      }
53  
54      @Override
55      public GlobalMemory createMemory() {
56          return new SolarisGlobalMemory();
57      }
58  
59      @Override
60      public CentralProcessor createProcessor() {
61          return new SolarisCentralProcessor();
62      }
63  
64      @Override
65      public Sensors createSensors() {
66          return new SolarisSensors();
67      }
68  
69      @Override
70      public List<PowerSource> getPowerSources() {
71          return SolarisPowerSource.getPowerSources();
72      }
73  
74      @Override
75      public List<HWDiskStore> getDiskStores() {
76          return SolarisHWDiskStore.getDisks();
77      }
78  
79      @Override
80      public List<Display> getDisplays() {
81          return UnixDisplay.getDisplays();
82      }
83  
84      @Override
85      public List<NetworkIF> getNetworkIFs(boolean includeLocalInterfaces) {
86          return SolarisNetworkIF.getNetworks(includeLocalInterfaces);
87      }
88  
89      @Override
90      public List<UsbDevice> getUsbDevices(boolean tree) {
91          return SolarisUsbDevice.getUsbDevices(tree);
92      }
93  
94      @Override
95      public List<SoundCard> getSoundCards() {
96          return SolarisSoundCard.getSoundCards();
97      }
98  
99      @Override
100     public List<GraphicsCard> getGraphicsCards() {
101         return SolarisGraphicsCard.getGraphicsCards();
102     }
103 }
