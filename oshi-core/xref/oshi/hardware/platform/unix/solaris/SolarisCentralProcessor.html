61      @Override
62      protected ProcessorIdentifier queryProcessorId() {
63          boolean cpu64bit = "64".equals(ExecutingCommand.getFirstAnswer("isainfo -b").trim());
64          if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
65              // Use Kstat2 implementation
66              return queryProcessorId2(cpu64bit);
67          }
68          String cpuVendor = "";
69          String cpuName = "";
70          String cpuFamily = "";
71          String cpuModel = "";
72          String cpuStepping = "";
73          long cpuFreq = 0L;
74  
75          // Get first result
76          try (KstatChain kc = KstatUtil.openChain()) {
77              Kstat ksp = KstatChain.lookup(CPU_INFO, -1, null);
78              // Set values
79              if (ksp != null && KstatChain.read(ksp)) {
80                  cpuVendor = KstatUtil.dataLookupString(ksp, "vendor_id");
81                  cpuName = KstatUtil.dataLookupString(ksp, "brand");
82                  cpuFamily = KstatUtil.dataLookupString(ksp, "family");
83                  cpuModel = KstatUtil.dataLookupString(ksp, "model");
84                  cpuStepping = KstatUtil.dataLookupString(ksp, "stepping");
85                  cpuFreq = KstatUtil.dataLookupLong(ksp, "clock_MHz") * 1_000_000L;
86              }
87          }
88          String processorID = getProcessorID(cpuStepping, cpuModel, cpuFamily);
89  
90          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
91                  cpuFreq);
92      }

94      private static ProcessorIdentifier queryProcessorId2(boolean cpu64bit) {
95          Object[] results = KstatUtil.queryKstat2(KSTAT_SYSTEM_CPU + "0" + INFO, "vendor_id", "brand", "family", "model",
96                  "stepping", "clock_MHz");
97  
98          String cpuVendor = results[0] == null ? "" : (String) results[0];
99          String cpuName = results[1] == null ? "" : (String) results[1];
100         String cpuFamily = results[2] == null ? "" : results[2].toString();
101         String cpuModel = results[3] == null ? "" : results[3].toString();
102         String cpuStepping = results[4] == null ? "" : results[4].toString();
103         long cpuFreq = results[5] == null ? 0L : (long) results[5];
104 
105         String processorID = getProcessorID(cpuStepping, cpuModel, cpuFamily);
106         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
107                 cpuFreq);
108     }

110     @Override
111     protected Pair<List<LogicalProcessor>, List<PhysicalProcessor>> initProcessorCounts() {
112         Map<Integer, Integer> numaNodeMap = mapNumaNodes();
113         if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
114             // Use Kstat2 implementation
115             return new Pair<>(initProcessorCounts2(numaNodeMap), null);
116         }
117         List<LogicalProcessor> logProcs = new ArrayList<>();
118         try (KstatChain kc = KstatUtil.openChain()) {
119             List<Kstat> kstats = KstatChain.lookupAll(CPU_INFO, -1, null);
120 
121             for (Kstat ksp : kstats) {
122                 if (ksp != null && KstatChain.read(ksp)) {
123                     int procId = logProcs.size(); // 0-indexed
124                     String chipId = KstatUtil.dataLookupString(ksp, "chip_id");
125                     String coreId = KstatUtil.dataLookupString(ksp, "core_id");
126                     LogicalProcessor logProc = new LogicalProcessor(procId, ParseUtil.parseIntOrDefault(coreId, 0),
127                             ParseUtil.parseIntOrDefault(chipId, 0), numaNodeMap.getOrDefault(procId, 0));
128                     logProcs.add(logProc);
129                 }
130             }
131         }
132         if (logProcs.isEmpty()) {
133             logProcs.add(new LogicalProcessor(0, 0, 0));
134         }
135         Map<Integer, String> dmesg = new HashMap<>();
136         // Jan 9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: Intel(r) Celeron(r)
137         // CPU J3455 @ 1.50GHz
138         // but NOT: Jan 9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: x86 (chipid
139         // 0x0 GenuineIntel 506C9 family 6 model 92 step 9 clock 1500 MHz)
140         Pattern p = Pattern.compile(".* cpu(\\\\d+): ((ARM|AMD|Intel).+)");
141         for (String s : ExecutingCommand.runNative("dmesg")) {
142             Matcher m = p.matcher(s);
143             if (m.matches()) {
144                 int coreId = ParseUtil.parseIntOrDefault(m.group(1), 0);
145                 dmesg.put(coreId, m.group(2).trim());
146             }
147         }
148         if (dmesg.isEmpty()) {
149             return new Pair<>(logProcs, null);
150         }
151         return new Pair<>(logProcs, createProcListFromDmesg(logProcs, dmesg));
152     }

154     private static List<LogicalProcessor> initProcessorCounts2(Map<Integer, Integer> numaNodeMap) {
155         List<LogicalProcessor> logProcs = new ArrayList<>();
156 
157         List<Object[]> results = KstatUtil.queryKstat2List(KSTAT_SYSTEM_CPU, INFO, "chip_id", "core_id");
158         for (Object[] result : results) {
159             int procId = logProcs.size();
160             long chipId = result[0] == null ? 0L : (long) result[0];
161             long coreId = result[1] == null ? 0L : (long) result[1];
162             LogicalProcessor logProc = new LogicalProcessor(procId, (int) coreId, (int) chipId,
163                     numaNodeMap.getOrDefault(procId, 0));
164             logProcs.add(logProc);
165         }
166         if (logProcs.isEmpty()) {
167             logProcs.add(new LogicalProcessor(0, 0, 0));
168         }
169         return logProcs;
170     }

172     private static Map<Integer, Integer> mapNumaNodes() {
173         // Get numa node info from lgrpinfo
174         Map<Integer, Integer> numaNodeMap = new HashMap<>();
175         int lgroup = 0;
176         for (String line : ExecutingCommand.runNative("lgrpinfo -c leaves")) {
177             // Format:
178             // lgroup 0 (root):
179             // CPUs: 0 1
180             // CPUs: 0-7
181             // CPUs: 0-3 6 7 12 13
182             // CPU: 0
183             // CPU: 1
184             if (line.startsWith("lgroup")) {
185                 lgroup = ParseUtil.getFirstIntValue(line);
186             } else if (line.contains("CPUs:") || line.contains("CPU:")) {
187                 for (Integer cpu : ParseUtil.parseHyphenatedIntList(line.split(":")[1])) {
188                     numaNodeMap.put(cpu, lgroup);
189                 }
190             }
191         }
192         return numaNodeMap;
193     }

195     @Override
196     public long[] querySystemCpuLoadTicks() {
197         long[] ticks = new long[TickType.values().length];
198         // Average processor ticks
199         long[][] procTicks = getProcessorCpuLoadTicks();
200         for (int i = 0; i < ticks.length; i++) {
201             for (long[] procTick : procTicks) {
202                 ticks[i] += procTick[i];
203             }
204             ticks[i] /= procTicks.length;
205         }
206         return ticks;
207     }

209     @Override
210     public long[] queryCurrentFreq() {
211         if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
212             // Use Kstat2 implementation
213             return queryCurrentFreq2(getLogicalProcessorCount());
214         }
215         long[] freqs = new long[getLogicalProcessorCount()];
216         Arrays.fill(freqs, -1);
217         try (KstatChain kc = KstatUtil.openChain()) {
218             for (int i = 0; i < freqs.length; i++) {
219                 for (Kstat ksp : KstatChain.lookupAll(CPU_INFO, i, null)) {
220                     if (KstatChain.read(ksp)) {
221                         freqs[i] = KstatUtil.dataLookupLong(ksp, "current_clock_Hz");
222                     }
223                 }
224             }
225         }
226         return freqs;
227     }

229     private static long[] queryCurrentFreq2(int processorCount) {
230         long[] freqs = new long[processorCount];
231         Arrays.fill(freqs, -1);
232 
233         List<Object[]> results = KstatUtil.queryKstat2List(KSTAT_SYSTEM_CPU, INFO, "current_clock_Hz");
234         int cpu = -1;
235         for (Object[] result : results) {
236             if (++cpu >= freqs.length) {
237                 break;
238             }
239             freqs[cpu] = result[0] == null ? -1L : (long) result[0];
240         }
241         return freqs;
242     }

244     @Override
245     public long queryMaxFreq() {
246         if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
247             // Use Kstat2 implementation
248             return queryMaxFreq2();
249         }
250         long max = -1L;
251         try (KstatChain kc = KstatUtil.openChain()) {
252             for (Kstat ksp : KstatChain.lookupAll(CPU_INFO, 0, null)) {
253                 if (KstatChain.read(ksp)) {
254                     String suppFreq = KstatUtil.dataLookupString(ksp, "supported_frequencies_Hz");
255                     if (!suppFreq.isEmpty()) {
256                         for (String s : suppFreq.split(":")) {
257                             long freq = ParseUtil.parseLongOrDefault(s, -1L);
258                             if (max < freq) {
259                                 max = freq;
260                             }
261                         }
262                     }
263                 }
264             }
265         }
266         return max;
267     }

269     private static long queryMaxFreq2() {
270         long max = -1L;
271         List<Object[]> results = KstatUtil.queryKstat2List(KSTAT_PM_CPU, PSTATE, "supported_frequencies");
272         for (Object[] result : results) {
273             for (long freq : result[0] == null ? new long[0] : (long[]) result[0]) {
274                 if (freq > max) {
275                     max = freq;
276                 }
277             }
278         }
279         return max;
280     }

282     @Override
283     public double[] getSystemLoadAverage(int nelem) {
284         if (nelem < 1 || nelem > 3) {
285             throw new IllegalArgumentException("Must include from one to three elements.");
286         }
287         double[] average = new double[nelem];
288         int retval = SolarisLibc.INSTANCE.getloadavg(average, nelem);
289         if (retval < nelem) {
290             for (int i = Math.max(retval, 0); i < average.length; i++) {
291                 average[i] = -1d;
292             }
293         }
294         return average;
295     }

297     @Override
298     public long[][] queryProcessorCpuLoadTicks() {
299         if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
300             // Use Kstat2 implementation
301             return queryProcessorCpuLoadTicks2(getLogicalProcessorCount());
302         }
303         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
304         int cpu = -1;
305         try (KstatChain kc = KstatUtil.openChain()) {
306             for (Kstat ksp : KstatChain.lookupAll("cpu", -1, "sys")) {
307                 // This is a new CPU
308                 if (++cpu >= ticks.length) {
309                     // Shouldn't happen
310                     break;
311                 }
312                 if (KstatChain.read(ksp)) {
313                     ticks[cpu][TickType.IDLE.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_idle");
314                     ticks[cpu][TickType.SYSTEM.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_kernel");
315                     ticks[cpu][TickType.USER.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_user");
316                 }
317             }
318         }
319         return ticks;
320     }

322     private static long[][] queryProcessorCpuLoadTicks2(int processorCount) {
323         long[][] ticks = new long[processorCount][TickType.values().length];
324         List<Object[]> results = KstatUtil.queryKstat2List(KSTAT_SYSTEM_CPU, SYS, "cpu_ticks_idle", "cpu_ticks_kernel",
325                 "cpu_ticks_user");
326         int cpu = -1;
327         for (Object[] result : results) {
328             if (++cpu >= ticks.length) {
329                 break;
330             }
331             ticks[cpu][TickType.IDLE.getIndex()] = result[0] == null ? 0L : (long) result[0];
332             ticks[cpu][TickType.SYSTEM.getIndex()] = result[1] == null ? 0L : (long) result[1];
333             ticks[cpu][TickType.USER.getIndex()] = result[2] == null ? 0L : (long) result[2];
334         }
335         return ticks;
336     }

338     /**
339      * Fetches the ProcessorID by encoding the stepping, model, family, and feature
340      * flags.
341      *
342      * @param stepping
343      *            The stepping
344      * @param model
345      *            The model
346      * @param family
347      *            The family
348      * @return The Processor ID string
349      */
350     private static String getProcessorID(String stepping, String model, String family) {
351         List<String> isainfo = ExecutingCommand.runNative("isainfo -v");
352         StringBuilder flags = new StringBuilder();
353         for (String line : isainfo) {
354             if (line.startsWith("32-bit")) {
355                 break;
356             } else if (!line.startsWith("64-bit")) {
357                 flags.append(' ').append(line.trim());
358             }
359         }
360         return createProcessorID(stepping, model, family, ParseUtil.whitespaces.split(flags.toString().toLowerCase()));
361     }

363     @Override
364     public long queryContextSwitches() {
365         if (SolarisOperatingSystem.IS_11_4_OR_HIGHER) {
366             // Use Kstat2 implementation
367             return queryContextSwitches2();
368         }
369         long swtch = 0;
370         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/pswitch\\\\|inv_swtch/");
371         for (String s : kstat) {
372             swtch += ParseUtil.parseLastLong(s, 0L);
373         }
374         return swtch;
375     }

378         long swtch = 0;
379         List<Object[]> results = KstatUtil.queryKstat2List(KSTAT_SYSTEM_CPU, SYS, "pswitch", "inv_swtch");
380         for (Object[] result : results) {
381             swtch += result[0] == null ? 0L : (long) result[0];
382             swtch += result[1] == null ? 0L : (long) result[1];
383         }
384         return swtch;
