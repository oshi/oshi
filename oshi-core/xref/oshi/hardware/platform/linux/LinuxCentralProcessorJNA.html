27  /**
28   * JNA-based Linux central processor implementation. Extends {@link LinuxCentralProcessor}, supplying the udev CPU
29   * enumeration and native library calls.
30   */
31  @ThreadSafe
32  final class LinuxCentralProcessorJNA extends LinuxCentralProcessor {
33  
34      private static final Logger LOG = LoggerFactory.getLogger(LinuxCentralProcessorJNA.class);
35  
36      LinuxCentralProcessorJNA() {
37          super(LinuxOperatingSystemJNA.hz());
38      }
39  
40      @Override
41      protected List<String> enumerateCpuSyspathsViaUdev() {
42          if (!HAS_UDEV) {
43              return null;
44          }
45          UdevContext udev = Udev.INSTANCE.udev_new();
46          if (udev == null) {
47              return null;
48          }
49          try {
50              List<String> syspaths = new ArrayList<>();
51              UdevEnumerate enumerate = udev.enumerateNew();
52              if (enumerate == null) {
53                  return null;
54              }
55              try {
56                  enumerate.addMatchSubsystem("cpu");
57                  enumerate.scanDevices();
58                  for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
59                      syspaths.add(entry.getName());
60                  }
61              } finally {
62                  enumerate.unref();
63              }
64              return syspaths;
65          } catch (RuntimeException e) {
66              // Any udev failure falls back to the shared sysfs scan
67              LOG.warn("Error enumerating CPUs via udev, falling back to sysfs", e);
68              return null;
69          } finally {
70              udev.unref();
71          }
72      }
73  
74      @Override
75      protected long queryHwcap() {
76          return AuxvJNA.queryAuxv().getOrDefault(Auxv.AT_HWCAP, 0L);
77      }
78  
79      @Override
80      protected int getloadavgNative(double[] loadavg, int nelem) {
81          return LinuxLibc.INSTANCE.getloadavg(loadavg, nelem);
82      }
83  }
