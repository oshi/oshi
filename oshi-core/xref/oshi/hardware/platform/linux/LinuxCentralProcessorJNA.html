43      @Override
44      protected Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyWithUdev() {
45          if (!HAS_UDEV) {
46              return readTopologyFromSysfs();
47          }
48          UdevContext udev = Udev.INSTANCE.udev_new();
49          if (udev == null) {
50              return readTopologyFromSysfs();
51          }
52          try {
53              return readTopologyFromUdev(udev);
54          } finally {
55              udev.unref();
56          }
57      }

59      @Override
60      protected boolean queryCurrentFreqFromUdev(long[] freqs) {
61          if (!HAS_UDEV) {
62              return queryCurrentFreqFromSysfs(freqs);
63          }
64          UdevContext udev = Udev.INSTANCE.udev_new();
65          if (udev == null) {
66              return queryCurrentFreqFromSysfs(freqs);
67          }
68          try {
69              return readCurrentFreqFromUdev(udev, freqs);
70          } finally {
71              udev.unref();
72          }
73      }

91      private static Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyFromUdev(
92              UdevContext udev) {
93          List<LogicalProcessor> logProcs = new ArrayList<>();
94          Set<ProcessorCache> caches = new HashSet<>();
95          Map<Integer, Integer> coreEfficiencyMap = new HashMap<>();
96          Map<Integer, String> modAliasMap = new HashMap<>();
97          UdevEnumerate enumerate = udev.enumerateNew();
98          try {
99              enumerate.addMatchSubsystem("cpu");
100             enumerate.scanDevices();
101             for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
102                 String syspath = entry.getName(); // /sys/devices/system/cpu/cpuX
103                 UdevDevice device = udev.deviceNewFromSyspath(syspath);
104                 String modAlias = null;
105                 if (device != null) {
106                     try {
107                         modAlias = device.getPropertyValue("MODALIAS");
108                     } finally {
109                         device.unref();
110                     }
111                 }
112                 logProcs.add(getLogicalProcessorFromSyspath(syspath, caches, modAlias, coreEfficiencyMap, modAliasMap));
113             }
114         } finally {
115             enumerate.unref();
116         }
117         return new Quartet<>(logProcs, orderedProcCaches(caches), coreEfficiencyMap, modAliasMap);
118     }

120     private static boolean readCurrentFreqFromUdev(UdevContext udev, long[] freqs) {
121         long max = 0L;
122         UdevEnumerate enumerate = udev.enumerateNew();
123         try {
124             enumerate.addMatchSubsystem("cpu");
125             enumerate.scanDevices();
126             for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
127                 String syspath = entry.getName();
128                 int cpu = ParseUtil.getFirstIntValue(syspath);
129                 if (cpu >= 0 && cpu < freqs.length) {
130                     freqs[cpu] = FileUtil.getLongFromFile(syspath + "/cpufreq/scaling_cur_freq");
131                     if (freqs[cpu] == 0) {
132                         freqs[cpu] = FileUtil.getLongFromFile(syspath + "/cpufreq/cpuinfo_cur_freq");
133                     }
134                     if (max < freqs[cpu]) {
135                         max = freqs[cpu];
136                     }
137                 }
138             }
139         } finally {
140             enumerate.unref();
141         }
142         if (max > 0L) {
143             for (int i = 0; i < freqs.length; i++) {
144                 freqs[i] *= 1000L;
145             }
146             return true;
147         }
148         return false;
149     }

153         return AuxvJNA.queryAuxv().getOrDefault(Auxv.AT_HWCAP, 0L);
154     }

156     @Override
157     public double[] getSystemLoadAverage(int nelem) {
158         if (nelem < 1 || nelem > 3) {
159             throw new IllegalArgumentException("Must include from one to three elements.");
160         }
161         double[] average = new double[nelem];
162         int retval = LinuxLibc.INSTANCE.getloadavg(average, nelem);
163         if (retval < 0) {
164             return super.getSystemLoadAverage(nelem);
165         }
166         for (int i = retval; i < nelem; i++) {
167             average[i] = -1d;
168         }
169         return average;
170     }

172     private static long readMaxFreqFromUdev(UdevContext udev) {
173         UdevEnumerate enumerate = udev.enumerateNew();
174         try {
175             enumerate.addMatchSubsystem("cpu");
176             enumerate.scanDevices();
177             UdevListEntry entry = enumerate.getListEntry();
178             if (entry != null) {
179                 String syspath = entry.getName();
180                 return queryMaxFreqFromCpuFreqPath(syspath.substring(0, syspath.lastIndexOf('/')) + "/cpufreq");
181             }
182         } finally {
183             enumerate.unref();
184         }
185         return -1L;
186     }
