97          UDEV_STAT_LENGTH = statLength;

162     private static List<HWDiskStore> getDisks(LinuxHWDiskStore storeToUpdate) {
163         LinuxHWDiskStore store = null;
164         List<HWDiskStore> result = new ArrayList<>();
165 
166         Map<String, String> mountsMap = readMountsMap();
167 
168         UdevContext udev = Udev.INSTANCE.udev_new();
169         try {
170             UdevEnumerate enumerate = udev.enumerateNew();
171             try {
172                 enumerate.addMatchSubsystem(BLOCK);
173                 enumerate.scanDevices();
174                 for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
175                     String syspath = entry.getName();
176                     UdevDevice device = udev.deviceNewFromSyspath(syspath);
177                     if (device != null) {
178                         try {
179                             // devnode is what we use as name, like /dev/sda
180                             String devnode = device.getDevnode();
181                             // Ignore loopback and ram disks; do nothing
182                             if (devnode != null && !devnode.startsWith("/dev/loop")
183                                     && !devnode.startsWith("/dev/ram")) {
184                                 if (DISK.equals(device.getDevtype())) {
185                                     // Null model and serial in virtual environments
186                                     String devModel = device.getPropertyValue(ID_MODEL);
187                                     String devSerial = device.getPropertyValue(ID_SERIAL_SHORT);
188                                     long devSize = ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
189                                             * SECTORSIZE;
190                                     if (devnode.startsWith("/dev/dm")) {
191                                         devModel = LOGICAL_VOLUME_GROUP;
192                                         devSerial = device.getPropertyValue(DM_UUID);
193                                         store = new LinuxHWDiskStore(devnode, devModel,
194                                                 devSerial == null ? Constants.UNKNOWN : devSerial, devSize);
195                                         String vgName = device.getPropertyValue(DM_VG_NAME);
196                                         String lvName = device.getPropertyValue(DM_LV_NAME);
197                                         store.partitionList.add(new HWPartition(
198                                                 getPartitionNameForDmDevice(vgName, lvName), device.getSysname(),
199                                                 device.getPropertyValue(ID_FS_TYPE) == null ? PARTITION
200                                                         : device.getPropertyValue(ID_FS_TYPE),
201                                                 device.getPropertyValue(ID_FS_UUID) == null ? ""
202                                                         : device.getPropertyValue(ID_FS_UUID),
203                                                 ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
204                                                         * SECTORSIZE,
205                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MAJOR), 0),
206                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MINOR), 0),
207                                                 getMountPointOfDmDevice(vgName, lvName)));
208                                     } else {
209                                         store = new LinuxHWDiskStore(devnode,
210                                                 devModel == null ? Constants.UNKNOWN : devModel,
211                                                 devSerial == null ? Constants.UNKNOWN : devSerial, devSize);
212                                     }
213                                     if (storeToUpdate == null) {
214                                         // If getting all stores, add to the list with stats
215                                         computeDiskStats(store, device.getSysattrValue(STAT));
216                                         result.add(store);
217                                     } else if (store.getName().equals(storeToUpdate.getName())
218                                             && store.getModel().equals(storeToUpdate.getModel())
219                                             && store.getSerial().equals(storeToUpdate.getSerial())
220                                             && store.getSize() == storeToUpdate.getSize()) {
221                                         // If we are only updating a single disk, the name, model, serial, and size are
222                                         // sufficient to test if this is a match. Add the (old) object, release handle
223                                         // and return.
224                                         computeDiskStats(storeToUpdate, device.getSysattrValue(STAT));
225                                         result.add(storeToUpdate);
226                                         break;
227                                     }
228                                 } else if (storeToUpdate == null && store != null // only add if getting new list
229                                         && PARTITION.equals(device.getDevtype())) {
230                                     // udev_device_get_parent_*() does not take a reference on the returned device,
231                                     // it is automatically unref'd with the parent
232                                     UdevDevice parent = device.getParentWithSubsystemDevtype(BLOCK, DISK);
233                                     if (parent != null && store.getName().equals(parent.getDevnode())) {
234                                         // `store` should still point to the parent HWDiskStore this partition is
235                                         // attached to. If not, it's an error, so skip.
236                                         String name = device.getDevnode();
237                                         store.partitionList.add(new HWPartition(name, device.getSysname(),
238                                                 device.getPropertyValue(ID_FS_TYPE) == null ? PARTITION
239                                                         : device.getPropertyValue(ID_FS_TYPE),
240                                                 device.getPropertyValue(ID_FS_UUID) == null ? ""
241                                                         : device.getPropertyValue(ID_FS_UUID),
242                                                 ParseUtil.parseLongOrDefault(device.getSysattrValue(SIZE), 0L)
243                                                         * SECTORSIZE,
244                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MAJOR), 0),
245                                                 ParseUtil.parseIntOrDefault(device.getPropertyValue(MINOR), 0),
246                                                 mountsMap.getOrDefault(name,
247                                                         getDependentNamesFromHoldersDirectory(device.getSysname()))));
248                                     }
249                                 }
250                             }
251                         } finally {
252                             device.unref();
253                         }
254                     }
255                 }
256             } finally {
257                 enumerate.unref();
258             }
259         } finally {
260             udev.unref();
261         }
262         // Iterate the list and make the partitions unmodifiable
263         for (HWDiskStore hwds : result) {
264             ((LinuxHWDiskStore) hwds).partitionList = Collections.unmodifiableList(hwds.getPartitions().stream()
265                     .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList()));
266         }
267         return result;
268     }

270     @Override
271     public boolean updateAttributes() {
272         // If this returns non-empty (the same store, but updated) then we were
273         // successful in the update
274         return !getDisks(this).isEmpty();
275     }

277     private static Map<String, String> readMountsMap() {
278         Map<String, String> mountsMap = new HashMap<>();
279         List<String> mounts = FileUtil.readFile(ProcPath.MOUNTS);
280         for (String mount : mounts) {
281             String[] split = ParseUtil.whitespaces.split(mount);
282             if (split.length < 2 || !split[0].startsWith(DEV_LOCATION)) {
283                 continue;
284             }
285             mountsMap.put(split[0], split[1]);
286         }
287         return mountsMap;
288     }

290     private static void computeDiskStats(LinuxHWDiskStore store, String devstat) {
291         long[] devstatArray = ParseUtil.parseStringToLongArray(devstat, UDEV_STAT_ORDERS, UDEV_STAT_LENGTH, ' ');
292         store.timeStamp = System.currentTimeMillis();
293 
294         // Reads and writes are converted in bytes
295         store.reads = devstatArray[UdevStat.READS.ordinal()];
296         store.readBytes = devstatArray[UdevStat.READ_BYTES.ordinal()] * SECTORSIZE;
297         store.writes = devstatArray[UdevStat.WRITES.ordinal()];
298         store.writeBytes = devstatArray[UdevStat.WRITE_BYTES.ordinal()] * SECTORSIZE;
299         store.currentQueueLength = devstatArray[UdevStat.QUEUE_LENGTH.ordinal()];
300         store.transferTime = devstatArray[UdevStat.ACTIVE_MS.ordinal()];
301     }

303     private static String getPartitionNameForDmDevice(String vgName, String lvName) {
304         return new StringBuilder().append(DEV_LOCATION).append(vgName).append('/').append(lvName).toString();
305     }
306 
307     private static String getMountPointOfDmDevice(String vgName, String lvName) {
308         return new StringBuilder().append(DEV_MAPPER).append(vgName).append('-').append(lvName).toString();
309     }
310 
311     private static String getDependentNamesFromHoldersDirectory(String sysPath) {
312         File holdersDir = new File(sysPath + "/holders");
313         File[] holders = holdersDir.listFiles();
314         if (holders != null) {
315             return Arrays.stream(holders).map(File::getName).collect(Collectors.joining(" "));
316         }
317         return "";
318     }

320     // Order the field is in udev stats
321     enum UdevStat {
322         // The parsing implementation in ParseUtil requires these to be declared
323         // in increasing order. Use 0-ordered index here
324         READS(0), READ_BYTES(2), WRITES(4), WRITE_BYTES(6), QUEUE_LENGTH(8), ACTIVE_MS(9);

326         private int order;

328         public int getOrder() {
329             return this.order;
330         }

332         UdevStat(int order) {
333             this.order = order;
334         }
