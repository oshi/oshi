72      private static String queryIfModel(NetworkInterface netint) {
73          String name = netint.getName();
74          UdevContext udev = Udev.INSTANCE.udev_new();
75          if (udev != null) {
76              try {
77                  UdevDevice device = udev.deviceNewFromSyspath("/sys/class/net/" + name);
78                  if (device != null) {
79                      try {
80                          String devVendor = device.getPropertyValue("ID_VENDOR_FROM_DATABASE");
81                          String devModel = device.getPropertyValue("ID_MODEL_FROM_DATABASE");
82                          if (!Util.isBlank(devModel)) {
83                              if (!Util.isBlank(devVendor)) {
84                                  return devVendor + " " + devModel;
85                              }
86                              return devModel;
87                          }
88                      } finally {
89                          device.unref();
90                      }
91                  }
92              } finally {
93                  udev.unref();
94              }
95          }
96          return name;
97      }

99      /**
100      * Gets network interfaces on this machine
101      *
102      * @param includeLocalInterfaces
103      *            include local interfaces in the result
104      * @return A list of {@link NetworkIF} objects representing the interfaces
105      */
106     public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
107         List<NetworkIF> ifList = new ArrayList<>();
108         for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
109             try {
110                 ifList.add(new LinuxNetworkIF(ni));
111             } catch (InstantiationException e) {
112                 LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
113             }
114         }
115         return ifList;
116     }

188     @Override
189     public boolean updateAttributes() {
190         try {
191             File ifDir = new File(String.format("/sys/class/net/%s/statistics", getName()));
192             if (!ifDir.isDirectory()) {
193                 return false;
194             }
195         } catch (SecurityException e) {
196             return false;
197         }
198         String ifTypePath = String.format("/sys/class/net/%s/type", getName());
199         String carrierPath = String.format("/sys/class/net/%s/carrier", getName());
200         String txBytesPath = String.format("/sys/class/net/%s/statistics/tx_bytes", getName());
201         String rxBytesPath = String.format("/sys/class/net/%s/statistics/rx_bytes", getName());
202         String txPacketsPath = String.format("/sys/class/net/%s/statistics/tx_packets", getName());
203         String rxPacketsPath = String.format("/sys/class/net/%s/statistics/rx_packets", getName());
204         String txErrorsPath = String.format("/sys/class/net/%s/statistics/tx_errors", getName());
205         String rxErrorsPath = String.format("/sys/class/net/%s/statistics/rx_errors", getName());
206         String collisionsPath = String.format("/sys/class/net/%s/statistics/collisions", getName());
207         String rxDropsPath = String.format("/sys/class/net/%s/statistics/rx_dropped", getName());
208         String ifSpeed = String.format("/sys/class/net/%s/speed", getName());
209         String ifAliasPath = String.format("/sys/class/net/%s/ifalias", getName());
210         String ifOperStatusPath = String.format("/sys/class/net/%s/operstate", getName());
211 
212         this.timeStamp = System.currentTimeMillis();
213         this.ifType = FileUtil.getIntFromFile(ifTypePath);
214         this.connectorPresent = FileUtil.getIntFromFile(carrierPath) > 0;
215         this.bytesSent = FileUtil.getUnsignedLongFromFile(txBytesPath);
216         this.bytesRecv = FileUtil.getUnsignedLongFromFile(rxBytesPath);
217         this.packetsSent = FileUtil.getUnsignedLongFromFile(txPacketsPath);
218         this.packetsRecv = FileUtil.getUnsignedLongFromFile(rxPacketsPath);
219         this.outErrors = FileUtil.getUnsignedLongFromFile(txErrorsPath);
220         this.inErrors = FileUtil.getUnsignedLongFromFile(rxErrorsPath);
221         this.collisions = FileUtil.getUnsignedLongFromFile(collisionsPath);
222         this.inDrops = FileUtil.getUnsignedLongFromFile(rxDropsPath);
223         long speedMiB = FileUtil.getUnsignedLongFromFile(ifSpeed);
224         // speed may be -1 from file.
225         this.speed = speedMiB < 0 ? 0 : speedMiB << 20;
226         this.ifAlias = FileUtil.getStringFromFile(ifAliasPath);
227         this.ifOperStatus = parseIfOperStatus(FileUtil.getStringFromFile(ifOperStatusPath));
228 
229         return true;
230     }
231 
232     private static IfOperStatus parseIfOperStatus(String operState) {
233         switch (operState) {
234         case "up":
235             return IfOperStatus.UP;
236         case "down":
237             return IfOperStatus.DOWN;
238         case "testing":
239             return IfOperStatus.TESTING;
240         case "dormant":
241             return IfOperStatus.DORMANT;
242         case "notpresent":
243             return IfOperStatus.NOT_PRESENT;
244         case "lowerlayerdown":
245             return IfOperStatus.LOWER_LAYER_DOWN;
246         case "unknown":
247         default:
248             return IfOperStatus.UNKNOWN;
249         }
250     }
