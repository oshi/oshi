24  package oshi.hardware.platform.linux;
25  
26  import java.io.File;
27  import java.util.ArrayList;
28  import java.util.List;
29  import java.util.Map;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  import oshi.annotation.concurrent.Immutable;
35  import oshi.hardware.SoundCard;
36  import oshi.hardware.common.AbstractSoundCard;
37  import oshi.util.FileUtil;
38  import oshi.util.platform.linux.ProcPath;
39  
40  /**
41   * Sound card data obtained via /proc/asound directory
42   */
43  @Immutable
44  final class LinuxSoundCard extends AbstractSoundCard {
45  
46      private static final Logger LOG = LoggerFactory.getLogger(LinuxSoundCard.class);
47  
48      private static final String CARD_FOLDER = "card";
49      private static final String CARDS_FILE = "cards";
50      private static final String ID_FILE = "id";
51  
52      /**
53       * Constructor for LinuxSoundCard.
54       *
55       * @param kernelVersion
56       *            The version
57       * @param name
58       *            The name
59       * @param codec
60       *            The codec
61       */
62      LinuxSoundCard(String kernelVersion, String name, String codec) {
63          super(kernelVersion, name, codec);
64      }
65  
66      /**
67       * Method to find all the card folders contained in the <b>asound</b> folder
68       * denoting the cards currently contained in our machine.
69       *
70       * @return : A list of files starting with 'card'
71       */
72      private static List<File> getCardFolders() {
73          File cardsDirectory = new File(ProcPath.ASOUND);
74          List<File> cardFolders = new ArrayList<>();
75          File[] allContents = cardsDirectory.listFiles();
76          if (allContents != null) {
77              for (File card : allContents) {
78                  if (card.getName().startsWith(CARD_FOLDER) && card.isDirectory()) {
79                      cardFolders.add(card);
80                  }
81              }
82          } else {
83              LOG.warn("No Audio Cards Found");
84          }
85          return cardFolders;
86      }
87  
88      /**
89       * Reads the 'version' file in the asound folder that contains the complete name
90       * of the ALSA driver. Reads all the lines of the file and retrieves the first
91       * line.
92       *
93       * @return The complete name of the ALSA driver currently residing in our
94       *         machine
95       */
96      private static String getSoundCardVersion() {
97          String driverVersion = FileUtil.getStringFromFile(ProcPath.ASOUND + "version");
98          return driverVersion.isEmpty() ? "not available" : driverVersion;
99      }
100 
101     /**
102      * Retrieves the codec of the sound card contained in the <b>codec</b> file. The
103      * name of the codec is always the first line of that file. <br>
104      * <b>Working</b> <br>
105      * This converts the codec file into key value pairs using the {@link FileUtil}
106      * class and then returns the value of the <b>Codec</b> key.
107      *
108      * @param cardDir
109      *            The sound card directory
110      * @return The name of the codec
111      */
112     private static String getCardCodec(File cardDir) {
113         String cardCodec = "";
114         File[] cardFiles = cardDir.listFiles();
115         if (cardFiles != null) {
116             for (File file : cardFiles) {
117                 if (file.getName().startsWith("codec")) {
118                     if (!file.isDirectory()) {
119                         cardCodec = FileUtil.getKeyValueMapFromFile(file.getPath(), ":").get("Codec");
120                     } else {
121                         // on various centos environments, this is a subdirectory
122                         // each file is usually named something like
123                         // codec#0-0
124                         // example : ac97#0-0
125                         File[] codecs = file.listFiles();
126                         if (codecs != null) {
127                             for (File codec : codecs) {
128                                 if (!codec.isDirectory() && codec.getName().contains("#")) {
129                                     cardCodec = codec.getName().substring(0, codec.getName().indexOf('#'));
130                                     break;
131                                 }
132                             }
133                         }
134                     }
135                 }
136             }
137         }
138         return cardCodec;
139     }
140 
141     /**
142      * Retrieves the name of the sound card by :
143      * <ol>
144      * <li>Reading the <b>id</b> file and comparing each id with the card id present
145      * in the <b>cards</b> file</li>
146      * <li>If the id and the card name matches , then it assigns that name to
147      * {@literal cardName}</li>
148      * </ol>
149      *
150      * @param file
151      *            The sound card File.
152      * @return The name of the sound card.
153      */
154     private static String getCardName(File file) {
155         String cardName = "Not Found..";
156         Map<String, String> cardNamePairs = FileUtil.getKeyValueMapFromFile(ProcPath.ASOUND + "/" + CARDS_FILE, ":");
157         String cardId = FileUtil.getStringFromFile(file.getPath() + "/" + ID_FILE);
158         for (Map.Entry<String, String> entry : cardNamePairs.entrySet()) {
159             if (entry.getKey().contains(cardId)) {
160                 cardName = entry.getValue();
161                 return cardName;
162             }
163         }
164         return cardName;
165     }
166 
167     /**
168      * public method used by
169      * {@link oshi.hardware.common.AbstractHardwareAbstractionLayer} to access the
170      * sound cards.
171      *
172      * @return List of {@link oshi.hardware.platform.linux.LinuxSoundCard} objects.
173      */
174     public static List<SoundCard> getSoundCards() {
175         List<SoundCard> soundCards = new ArrayList<>();
176         for (File cardFile : getCardFolders()) {
177             soundCards.add(new LinuxSoundCard(getSoundCardVersion(), getCardName(cardFile), getCardCodec(cardFile)));
178         }
179         return soundCards;
180     }
181 }
