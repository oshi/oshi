72      /**
73       * Gets Display Information
74       *
75       * @return An array of Display objects representing monitors, etc.
76       */
77      public static List<Display> getDisplays() {
78          List<Display> displays = new ArrayList<>();
79          // Intel: real EDID exposed under IODisplayConnect (returns nothing on Apple Silicon).
80          displays.addAll(getDisplaysFromService("IODisplayConnect", "IODisplayEDID", "IOService"));
81          // Apple Silicon external monitors: same stripped EDID as Intel path, just different service.
82          displays.addAll(getDisplaysFromService("IOPortTransportStateDisplayPort", "EDID", null));
83          // Apple Silicon built-in panel: no real EDID exposed, synthesize from DisplayAttributes.
84          displays.addAll(getAppleSiliconBuiltInDisplay());
85          return displays;
86      }

88      /**
89       * Helper method to get displays from a specific IOKit service
90       *
91       * @param serviceName    The IOKit service name to search for
92       * @param edidKeyName    The key name for the EDID property
93       * @param childEntryName The name of the child entry to search in, or null to search directly in the service
94       * @return List of Display objects found using this service
95       */
96      private static List<Display> getDisplaysFromService(String serviceName, String edidKeyName, String childEntryName) {
97          List<Display> displays = new ArrayList<>();
98  
99          IOIterator serviceIterator = IOKitUtil.getMatchingServices(serviceName);
100         if (serviceIterator != null) {
101             CFStringRef cfEdid = CFStringRef.createCFString(edidKeyName);
102             IORegistryEntry sdService = serviceIterator.next();
103 
104             while (sdService != null) {
105                 IORegistryEntry propertySource = null;
106 
107                 try {
108                     propertySource = childEntryName == null ? sdService : sdService.getChildEntry(childEntryName);
109                     if (propertySource != null) {
110                         CFTypeRef edidRaw = propertySource.createCFProperty(cfEdid);
111                         if (edidRaw != null) {
112                             CFDataRef edid = new CFDataRef(edidRaw.getPointer());
113                             try {
114                                 // EDID is a byte array of 128 bytes (or more)
115                                 int length = edid.getLength();
116                                 Pointer p = edid.getBytePtr();
117                                 if (length > 0) {
118                                     displays.add(new MacDisplayJNA(p.getByteArray(0, length)));
119                                 }
120                             } finally {
121                                 edid.release();
122                             }
123                         }
124                         if (childEntryName != null) {
125                             propertySource.release();
126                         }
127                     }
128                 } finally {
129                     sdService.release();
130                     sdService = serviceIterator.next();
131                 }
132             }
133             serviceIterator.release();
134             cfEdid.release();
135         }
136         return displays;
137     }

139     /**
140      * Discovers the Apple Silicon built-in display by matching the stable {@code IOMobileFramebuffer} base class. The
141      * leaf class name varies by macOS version ({@code IOMobileFramebufferShim} on current releases, {@code AppleCLCD2}
142      * on older ones), so matching the base class covers all generations. External monitors are skipped here (they are
143      * already enumerated via {@code IOPortTransportStateDisplayPort} with their real EDID); only the built-in panel,
144      * which has no physical EDID EPROM, is synthesized from {@code DisplayAttributes}.
145      *
146      * @return A list containing the built-in display, or empty if not found
147      */
148     private static List<Display> getAppleSiliconBuiltInDisplay() {
149         List<Display> displays = new ArrayList<>();
150         IOIterator iter = IOKitUtil.getMatchingServices("IOMobileFramebuffer");
151         if (iter == null) {
152             return displays;
153         }
154         CFStringRef cfExternal = CFStringRef.createCFString("external");
155         CFStringRef cfAttrs = CFStringRef.createCFString("DisplayAttributes");
156         try {
157             IORegistryEntry fb = iter.next();
158             while (fb != null) {
159                 try {
160                     addBuiltInDisplay(fb, cfExternal, cfAttrs, displays);
161                 } finally {
162                     fb.release();
163                 }
164                 fb = iter.next();
165             }
166         } finally {
167             iter.release();
168             cfExternal.release();
169             cfAttrs.release();
170         }
171         return displays;
172     }

174     // Synthesizes a display for the built-in panel from its DisplayAttributes dictionary. External framebuffer nodes
175     // (marked with "external" = true) are skipped, as are idle pipes with no DisplayAttributes.
176     private static void addBuiltInDisplay(IORegistryEntry fb, CFStringRef cfExternal, CFStringRef cfAttrs,
177             List<Display> displays) {
178         // Skip external monitors — they are already enumerated via IOPortTransportStateDisplayPort.
179         CFTypeRef externalRef = fb.createCFProperty(cfExternal);
180         if (externalRef != null) {
181             try {
182                 if (new CFBooleanRef(externalRef.getPointer()).booleanValue()) {
183                     return;
184                 }
185             } finally {
186                 externalRef.release();
187             }
188         }
189         // Synthesize from DisplayAttributes, read from the node or (fallback) its IODeviceTree parent.
190         CFTypeRef attrsRaw = fb.createCFProperty(cfAttrs);
191         if (attrsRaw == null) {
192             IORegistryEntry parent = fb.getParentEntry("IODeviceTree");
193             if (parent != null) {
194                 try {
195                     attrsRaw = parent.createCFProperty(cfAttrs);
196                 } finally {
197                     parent.release();
198                 }
199             }
200         }
201         if (attrsRaw == null) {
202             return;
203         }
204         try {
205             DisplayInfo info = synthesize(fb, new CFDictionaryRef(attrsRaw.getPointer()));
206             if (info != null) {
207                 displays.add(new MacDisplayJNA(info));
208             }
209         } finally {
210             attrsRaw.release();
211         }
212     }

214     // Maps an Apple Silicon DisplayAttributes dictionary onto a synthetic DisplayInfo via EdidUtil, enriched with
215     // native resolution and device name from the framebuffer node and CoreGraphics.
216     private static DisplayInfo synthesize(IORegistryEntry fb, CFDictionaryRef attrs) {
217         CFDictionaryRef product = cfDictGetDictionary(attrs, "ProductAttributes");
218         if (product == null) {
219             return null;
220         }
221         Long legacyMfg = cfDictGetLong(product, "LegacyManufacturerID");
222         Long week = cfDictGetLong(product, "WeekOfManufacture");
223         Long year = cfDictGetLong(product, "YearOfManufacture");
224         String model = cfDictGetString(product, "ProductName");
225         String serial = cfDictGetString(product, "AlphanumericSerialNumber");
226         // Native pixel resolution from the framebuffer node.
227         Long displayWidth = cfRegistryEntryGetLong(fb, "DisplayWidth");
228         Long displayHeight = cfRegistryEntryGetLong(fb, "DisplayHeight");
229         // Device tree name for fallback model name.
230         String fallbackName = null;
231         String ioNameMatched = cfRegistryEntryGetString(fb, "IONameMatched");
232         if (ioNameMatched != null) {
233             String shortName = ioNameMatched.contains(",") ? ioNameMatched.substring(0, ioNameMatched.indexOf(','))
234                     : ioNameMatched;
235             fallbackName = shortName + " (Built-in Display)";
236         }
237         // CoreGraphics properties: model number, serial number, physical size, and localized name.
238         Integer cgModel = null;
239         Integer cgSerial = null;
240         Double widthMm = null;
241         Double heightMm = null;
242         String displayName = null;
243         int builtInId = findBuiltInDisplayId();
244         if (builtInId >= 0) {
245             try {
246                 CoreGraphics cg = CoreGraphics.INSTANCE;
247                 cgModel = cg.CGDisplayModelNumber(builtInId);
248                 cgSerial = cg.CGDisplaySerialNumber(builtInId);
249                 CoreGraphicsExt.CGSizeByValue size = CoreGraphicsExt.INSTANCE.CGDisplayScreenSize(builtInId);
250                 widthMm = size.width;
251                 heightMm = size.height;
252             } catch (Exception e) {
253                 LOG.debug("Failed to get built-in display CoreGraphics properties", e);
254             }
255             displayName = getLocalizedDisplayName(builtInId);
256         }
257         return EdidUtil.synthesizeDisplayInfo(legacyMfg, cgModel, cgSerial, week == null ? null : week.intValue(),
258                 year == null ? null : year.intValue(), model, serial, displayWidth, displayHeight, fallbackName,
259                 widthMm, heightMm, displayName);
260     }

262     // Returns the CGDirectDisplayID of the built-in display, or -1 if not found.
263     private static int findBuiltInDisplayId() {
264         CoreGraphics cg = CoreGraphics.INSTANCE;
265         IntByReference count = new IntByReference();
266         if (cg.CGGetActiveDisplayList(0, null, count) != 0 || count.getValue() == 0) {
267             return -1;
268         }
269         int[] displayIds = new int[count.getValue()];
270         if (cg.CGGetActiveDisplayList(displayIds.length, displayIds, count) != 0) {
271             return -1;
272         }
273         for (int id : displayIds) {
274             if (cg.CGDisplayIsBuiltin(id) != 0) {
275                 return id;
276             }
277         }
278         return -1;
279     }

281     // Returns the NSScreen.localizedName for the given CGDirectDisplayID, or null.
282     private static String getLocalizedDisplayName(int targetDisplayId) {
283         return ExceptionUtil.getOrDefault(() -> {
284             ObjCRuntime objc = ObjCRuntime.INSTANCE;
285             // Autorelease pool is thread-local; concurrent callers each get their own pool
286             Pointer pool = objc.objc_autoreleasePoolPush();
287             try {
288                 return queryLocalizedDisplayName(objc, targetDisplayId);
289             } finally {
290                 objc.objc_autoreleasePoolPop(pool);
291             }
292         }, null, LOG, "Failed to get localized display name: {}");
293     }

295     private static String queryLocalizedDisplayName(ObjCRuntime objc, int targetDisplayId) {
296         Pointer nsScreenClass = objc.objc_getClass("NSScreen");
297         if (nsScreenClass == null) {
298             return null;
299         }
300         Pointer selScreens = objc.sel_registerName("screens");
301         Pointer selCount = objc.sel_registerName("count");
302         Pointer selObjectAt = objc.sel_registerName("objectAtIndex:");
303         Pointer selDeviceDesc = objc.sel_registerName("deviceDescription");
304         Pointer selLocalizedName = objc.sel_registerName("localizedName");
305 
306         Pointer screensArray = objc.objc_msgSend(nsScreenClass, selScreens);
307         if (screensArray == null) {
308             return null;
309         }
310         long count = Pointer.nativeValue(objc.objc_msgSend(screensArray, selCount));
311         CFStringRef cfKey = CFStringRef.createCFString("NSScreenNumber");
312         try {
313             for (long i = 0; i < count; i++) {
314                 Pointer screen = objc.objc_msgSend(screensArray, selObjectAt, i);
315                 if (screen == null) {
316                     continue;
317                 }
318                 Pointer deviceDesc = objc.objc_msgSend(screen, selDeviceDesc);
319                 if (deviceDesc == null) {
320                     continue;
321                 }
322                 Pointer cfNum = CF.CFDictionaryGetValue(new CFDictionaryRef(deviceDesc), cfKey);
323                 if (cfNum == null) {
324                     continue;
325                 }
326                 LongByReference outId = new LongByReference();
327                 if (CF.CFNumberGetValue(new CFNumberRef(cfNum), K_CF_NUMBER_SINT64, outId) != 0
328                         && (int) outId.getValue() == targetDisplayId) {
329                     Pointer nsName = objc.objc_msgSend(screen, selLocalizedName);
330                     if (nsName != null) {
331                         return new CFStringRef(nsName).stringValue();
332                     }
333                 }
334             }
335         } finally {
336             cfKey.release();
337         }
338         return null;
339     }

341     // Reads a long value from an IORegistryEntry property.
342     private static Long cfRegistryEntryGetLong(IORegistryEntry entry, String key) {
343         CFStringRef k = CFStringRef.createCFString(key);
344         try {
345             CFTypeRef ref = entry.createCFProperty(k);
346             if (ref == null) {
347                 return null;
348             }
349             try {
350                 CFNumberRef num = new CFNumberRef(ref.getPointer());
351                 LongByReference out = new LongByReference();
352                 CF.CFNumberGetValue(num, K_CF_NUMBER_SINT64, out);
353                 return out.getValue();
354             } finally {
355                 ref.release();
356             }
357         } finally {
358             k.release();
359         }
360     }

383     private static CFDictionaryRef cfDictGetDictionary(CFDictionaryRef dict, String key) {
384         CFStringRef k = CFStringRef.createCFString(key);
385         try {
386             Pointer v = CF.CFDictionaryGetValue(dict, k);
387             return v == null ? null : new CFDictionaryRef(v);
388         } finally {
389             k.release();
390         }
391     }
