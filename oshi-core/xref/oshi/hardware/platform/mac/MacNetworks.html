44  public class MacNetworks extends AbstractNetworks {

46      private static final Logger LOG = LoggerFactory.getLogger(MacNetworks.class);
47  
48      private static final int CTL_NET = 4;
49      private static final int PF_ROUTE = 17;
50      private static final int NET_RT_IFLIST2 = 6;
51      private static final int RTM_IFINFO2 = 0x12;
52  
53      /**
54       * Class to encapsulate IF data for method return
55       */
56      private static class IFdata {
57          private long oPackets;
58          private long iPackets;
59          private long oBytes;
60          private long iBytes;
61          private long oErrors;
62          private long iErrors;
63          private long speed;
64          private long timeStamp;
65  
66          IFdata(long oPackets, long iPackets, // NOSONAR squid:S00107
67                  long oBytes, long iBytes, long oErrors, long iErrors, long speed, long timeStamp) {
68              this.oPackets = oPackets;
69              this.iPackets = iPackets;
70              this.oBytes = oBytes;
71              this.iBytes = iBytes;
72              this.oErrors = oErrors;
73              this.iErrors = iErrors;
74              this.speed = speed;
75              this.timeStamp = timeStamp;
76          }
77      }
78  
79      /**
80       * Map all network interfaces. Ported from source code of "netstat -ir". See
81       * http://opensource.apple.com/source/network_cmds/network_cmds-457/
82       * netstat.tproj/if.c
83       * 
84       * @param index
85       *            The network interface index to query
86       * @return an {@link IFData} object encapsulating the stats if the index
87       *         matches, or null if there was no match or an error reading the stats
88       */
89      private static synchronized IFdata queryIFdata(int index) {
90          // Get buffer of all interface information
91          int[] mib = { CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST2, 0 };
92          IntByReference len = new IntByReference();
93          if (0 != SystemB.INSTANCE.sysctl(mib, 6, null, len, null, 0)) {
94              LOG.error("Didn't get buffer length for IFLIST2");
95              return null;
96          }
97          Pointer buf = new Memory(len.getValue());
98          if (0 != SystemB.INSTANCE.sysctl(mib, 6, buf, len, null, 0)) {
99              LOG.error("Didn't get buffer for IFLIST2");
100             return null;
101         }
102 
103         // Iterate offset from buf's pointer up to limit of buf
104         int lim = len.getValue();
105         int next = 0;
106         while (next < lim) {
107             // Get pointer to current native part of buf
108             Pointer p = new Pointer(Pointer.nativeValue(buf) + next);
109             // Cast pointer to if_msghdr
110             IFmsgHdr ifm = new IFmsgHdr(p);
111             ifm.read();
112             // Advance next
113             next += ifm.ifm_msglen;
114             // Skip messages which are not the right format
115             if (ifm.ifm_type != RTM_IFINFO2) {
116                 continue;
117             }
118             // Cast pointer to if_msghdr2
119             IFmsgHdr2 if2m = new IFmsgHdr2(p);
120             if2m.read();
121 
122             if (if2m.ifm_index == index) {
123                 return new IFdata(if2m.ifm_data.ifi_opackets, if2m.ifm_data.ifi_ipackets, if2m.ifm_data.ifi_obytes,
124                         if2m.ifm_data.ifi_ibytes, if2m.ifm_data.ifi_oerrors, if2m.ifm_data.ifi_ierrors,
125                         if2m.ifm_data.ifi_baudrate, System.currentTimeMillis());
126             }
127         }
128         // If we get here we didn't find the index
129         return null;
130     }
131 
132     /**
133      * Updates interface network statistics on the given interface. Statistics
134      * include packets and bytes sent and received, and interface speed.
135      *
136      * @param netIF
137      *            The interface on which to update statistics
138      */
139     public static synchronized void updateNetworkStats(NetworkIF netIF) {
140         // Update data
141         IFdata ifData = queryIFdata(netIF.queryNetworkInterface().getIndex());
142         if (ifData != null) {
143             netIF.setBytesSent(ifData.oBytes);
144             netIF.setBytesRecv(ifData.iBytes);
145             netIF.setPacketsSent(ifData.oPackets);
146             netIF.setPacketsRecv(ifData.iPackets);
147             netIF.setOutErrors(ifData.oErrors);
148             netIF.setInErrors(ifData.iErrors);
149             netIF.setSpeed(ifData.speed);
150             netIF.setTimeStamp(ifData.timeStamp);
151         }
152     }
