40  public class MacNetworks extends AbstractNetworks {
41  
42      private static final long serialVersionUID = 1L;
43  
44      private static final Logger LOG = LoggerFactory.getLogger(MacNetworks.class);
45  
46      private static int CTL_NET = 4;
47      private static int PF_ROUTE = 17;
48      private static int NET_RT_IFLIST2 = 6;
49      private static int RTM_IFINFO2 = 0x12;
50  
51      /**
52       * Class to store a subset of IF data in the ifMap
53       */
54      private static class IFdata {
55          IFdata(long oPackets, long iPackets, long oBytes, long iBytes, long speed) {
56              this.oPackets = oPackets;
57              this.iPackets = iPackets;
58              this.oBytes = oBytes;
59              this.iBytes = iBytes;
60              this.speed = speed;
61          }
62  
63          long oPackets;
64          long iPackets;
65          long oBytes;
66          long iBytes;
67          long speed;
68      }
69  
70      /**
71       * Update map no more frequently than 200ms.
72       * 
73       * Key is the index of the IF.
74       */
75      private static Map<Integer, IFdata> ifMap = new HashMap<>();
76      private static long lastIFmapTime = 0L;
77  
78      /**
79       * Map all network interfaces. Ported from source code of "netstat -ir". See
80       * http://opensource.apple.com/source/network_cmds/network_cmds-457/
81       * netstat.tproj/if.c
82       */
83      private static synchronized void mapIFs() {
84          long now = System.currentTimeMillis();
85          if (now - lastIFmapTime < 200L) {
86              // Polled too recently; do nothing
87              return;
88          }
89          lastIFmapTime = now;
90  
91          // Get buffer of all interface information
92          int mib[] = { CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST2, 0 };
93          IntByReference len = new IntByReference();
94          if (0 != SystemB.INSTANCE.sysctl(mib, 6, null, len, null, 0)) {
95              LOG.error("Didn't get buffer length for IFLIST2");
96              return;
97          }
98          Pointer buf = new Memory(len.getValue());
99          if (0 != SystemB.INSTANCE.sysctl(mib, 6, buf, len, null, 0)) {
100             LOG.error("Didn't get buffer for IFLIST2");
101             return;
102         }
103 
104         // CLear out old hashmap
105         ifMap.clear();
106 
107         // Iterate offset from buf's pointer up to limit of buf
108         int lim = len.getValue();
109         for (int next = 0; next < lim;) {
110             // Get pointer to current native part of buf
111             Pointer p = new Pointer(Pointer.nativeValue(buf) + next);
112             // Cast pointer to if_msghdr
113             IFmsgHdr ifm = new IFmsgHdr(p);
114             ifm.read();
115             // Advance next
116             next += ifm.ifm_msglen;
117             // Skip messages which are not the right format
118             if (ifm.ifm_type != RTM_IFINFO2) {
119                 continue;
120             }
121             // Cast pointer to if_msghdr2
122             IFmsgHdr2 if2m = new IFmsgHdr2(p);
123             if2m.read();
124 
125             // Grab data to place in map using index as the key
126             ifMap.put(Integer.valueOf(if2m.ifm_index),
127                     new IFdata(if2m.ifm_data.ifi_opackets, if2m.ifm_data.ifi_ipackets, if2m.ifm_data.ifi_obytes,
128                             if2m.ifm_data.ifi_ibytes, if2m.ifm_data.ifi_baudrate));
129         }
130     }
131 
132     /**
133      * Updates interface network statistics on the given interface. Statistics
134      * include packets and bytes sent and received, and interface speed.
135      * 
136      * @param netIF
137      *            The interface on which to update statistics
138      */
139     public static synchronized void updateNetworkStats(NetworkIF netIF) {
140         // Update data
141         mapIFs();
142 
143         Integer index = Integer.valueOf(netIF.getNetworkInterface().getIndex());
144         IFdata ifData = ifMap.get(index);
145         if (ifData != null) {
146             // Retrieve values from map and store
147             netIF.setBytesSent(ifData.oBytes);
148             netIF.setBytesRecv(ifData.iBytes);
149             netIF.setPacketsSent(ifData.oPackets);
150             netIF.setPacketsRecv(ifData.iPackets);
151             netIF.setSpeed(ifData.speed);
152         }
153     }
