28  /**
29   * Hardware data obtained by system_profiler
30   *
31   * @author SchiTho1 [at] Securiton AG
32   * @author widdis [at] gmail [dot] com
33   */
34  final class MacComputerSystem extends AbstractComputerSystem {
35  
36      private static final long serialVersionUID = 1L;
37  
38      private static final String APPLE = "Apple Inc.";
39  
40      MacComputerSystem() {
41          init();
42      }
43  
44      private void init() {
45  
46          setManufacturer(APPLE);
47  
48          // $ system_profiler SPHardwareDataType
49          // Hardware:
50          //
51          // Hardware Overview:
52          //
53          // Model Name: MacBook Pro
54          // Model Identifier: MacBookPro8,2
55          // Processor Name: Intel Core i7
56          // Processor Speed: 2.3 GHz
57          // Number of Processors: 1
58          // Total Number of Cores: 4
59          // L2 Cache (per Core): 256 KB
60          // L3 Cache: 8 MB
61          // Memory: 16 GB
62          // Boot ROM Version: MBP81.0047.B2C
63          // SMC Version (system): 1.69f4
64          // Serial Number (system): C02FH4XYCB71
65          // Hardware UUID: D92CE829-65AD-58FA-9C32-88968791B7BD
66          // Sudden Motion Sensor:
67          // State: Enabled
68  
69          String modelName = "";
70          final String modelNameMarker = "Model Name:";
71          String modelIdentifier = "";
72          final String modelIdMarker = "Model Identifier:";
73          String serialNumberSystem = "";
74          final String serialNumMarker = "Serial Number (system):";
75          String smcVersion = "";
76          final String smcMarker = "SMC Version (system):";
77          String bootRomVersion = "";
78          final String bootRomMarker = "Boot ROM Version:";
79  
80          final MacFirmwareare.html#MacFirmware">MacFirmware firmware = new MacFirmware();
81          firmware.setManufacturer(APPLE);
82          firmware.setName("EFI");
83  
84          final MacBaseboardrd.html#MacBaseboard">MacBaseboard baseboard = new MacBaseboard();
85          baseboard.setManufacturer(APPLE);
86          baseboard.setModel("SMC");
87  
88          // Populate name and ID
89          for (final String checkLine : ExecutingCommand.runNative("system_profiler SPHardwareDataType")) {
90              if (checkLine.contains(modelNameMarker)) {
91                  modelName = checkLine.split(modelNameMarker)[1].trim();
92              }
93              if (checkLine.contains(modelIdMarker)) {
94                  modelIdentifier = checkLine.split(modelIdMarker)[1].trim();
95              }
96              if (checkLine.contains(bootRomMarker)) {
97                  bootRomVersion = checkLine.split(bootRomMarker)[1].trim();
98              }
99              if (checkLine.contains(smcMarker)) {
100                 smcVersion = checkLine.split(Pattern.quote(smcMarker))[1].trim();
101             }
102             if (checkLine.contains(serialNumMarker)) {
103                 serialNumberSystem = checkLine.split(Pattern.quote(serialNumMarker))[1].trim();
104             }
105         }
106         // Use name (id) if both available; else either one
107         if (!modelName.isEmpty()) {
108             if (!modelIdentifier.isEmpty()) {
109                 setModel(modelName + " (" + modelIdentifier + ")");
110             } else {
111                 setModel(modelName);
112             }
113         } else {
114             if (!modelIdentifier.isEmpty()) {
115                 setModel(modelIdentifier);
116             }
117         }
118         if (serialNumberSystem.isEmpty()) {
119             serialNumberSystem = getSystemSerialNumber();
120         }
121         setSerialNumber(serialNumberSystem);
122         baseboard.setSerialNumber(serialNumberSystem);
123         if (!smcVersion.isEmpty()) {
124             baseboard.setVersion(smcVersion);
125         }
126         if (bootRomVersion != null && !bootRomVersion.isEmpty()) {
127             firmware.setVersion(bootRomVersion);
128         }
129 
130         setFirmware(firmware);
131         setBaseboard(baseboard);
132     }
133 
134     private String getSystemSerialNumber() {
135         String serialNumber = null;
136         int service = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
137         if (service != 0) {
138             // Fetch the serial number
139             serialNumber = IOKitUtil.getIORegistryStringProperty(service, "IOPlatformSerialNumber");
140             IOKit.INSTANCE.IOObjectRelease(service);
141         }
142         if (serialNumber == null) {
143             serialNumber = "unknown";
144         }
145         return serialNumber;
146     }
