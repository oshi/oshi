7   import java.nio.charset.StandardCharsets;
8   
9   import com.sun.jna.Native;
10  import com.sun.jna.platform.mac.IOKit.IORegistryEntry;
11  import com.sun.jna.platform.mac.IOKitUtil;
12  
13  import oshi.annotation.concurrent.Immutable;
14  import oshi.hardware.Baseboard;
15  import oshi.hardware.Firmware;
16  import oshi.hardware.common.platform.mac.MacComputerSystem;
17  import oshi.util.Constants;
18  import oshi.util.Util;
19  import oshi.util.tuples.Quartet;
20  
21  /**
22   * Hardware data obtained from ioreg using JNA.
23   */
24  @Immutable
25  final class MacComputerSystemJNA extends MacComputerSystem {
26  
27      @Override
28      public Firmware createFirmware() {
29          return new MacFirmwareJNA();
30      }
31  
32      @Override
33      public Baseboard createBaseboard() {
34          return new MacBaseboardJNA();
35      }
36  
37      @Override
38      protected Quartet<String, String, String, String> platformExpert() {
39          String manufacturer = null;
40          String model = null;
41          String serialNumber = null;
42          String uuid = null;
43          IORegistryEntry platformExpert = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
44          if (platformExpert != null) {
45              byte[] data = platformExpert.getByteArrayProperty("manufacturer");
46              if (data != null) {
47                  manufacturer = Native.toString(data, StandardCharsets.UTF_8);
48              }
49              data = platformExpert.getByteArrayProperty("model");
50              if (data != null) {
51                  model = Native.toString(data, StandardCharsets.UTF_8);
52              }
53              serialNumber = platformExpert.getStringProperty("IOPlatformSerialNumber");
54              uuid = platformExpert.getStringProperty("IOPlatformUUID");
55              platformExpert.release();
56          }
57          return new Quartet<>(Util.isBlank(manufacturer) ? "Apple Inc." : manufacturer,
58                  Util.isBlank(model) ? Constants.UNKNOWN : model,
59                  Util.isBlank(serialNumber) ? Constants.UNKNOWN : serialNumber,
60                  Util.isBlank(uuid) ? Constants.UNKNOWN : uuid);
61      }
62  }
