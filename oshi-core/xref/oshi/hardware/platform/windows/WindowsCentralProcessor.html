87      @Override
88      protected ProcessorIdentifier queryProcessorId() {
89          String cpuVendor = "";
90          String cpuName = "";
91          String cpuIdentifier = "";
92          String cpuFamily = "";
93          String cpuModel = "";
94          String cpuStepping = "";
95          long cpuVendorFreq = 0L;
96          String processorID;
97          boolean cpu64bit = false;
98  
99          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
100         String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
101         if (processorIds.length > 0) {
102             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
103             cpuVendor = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
104                     "VendorIdentifier");
105             cpuName = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
106                     "ProcessorNameString");
107             cpuIdentifier = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
108                     "Identifier");
109             try {
110                 cpuVendorFreq = Advapi32Util.registryGetIntValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "~MHz")
111                         * 1_000_000L;
112             } catch (Win32Exception e) {
113                 // Leave as 0, parse the identifier as backup
114             }
115         }
116         if (!cpuIdentifier.isEmpty()) {
117             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
118             cpuModel = parseIdentifier(cpuIdentifier, "Model");
119             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
120         }
121         try (CloseableSystemInfo sysinfo = new CloseableSystemInfo()) {
122             Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
123             int processorArchitecture = sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue();
124             if (processorArchitecture == 9 // PROCESSOR_ARCHITECTURE_AMD64
125                     || processorArchitecture == 12 // PROCESSOR_ARCHITECTURE_ARM64
126                     || processorArchitecture == 6) { // PROCESSOR_ARCHITECTURE_IA64
127                 cpu64bit = true;
128             }
129         }
130         WmiResult<ProcessorIdProperty> processorId = Win32Processor.queryProcessorId();
131         if (processorId.getResultCount() > 0) {
132             processorID = WmiUtil.getString(processorId, ProcessorIdProperty.PROCESSORID, 0);
133         } else {
134             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
135                     cpu64bit ? new String[] { "ia64" } : new String[0]);
136         }
137         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
138                 cpuVendorFreq);
139     }

141     /**
142      * Parses identifier string
143      *
144      * @param identifier the full identifier string
145      * @param key        the key to retrieve
146      * @return the string following id
147      */
148     private static String parseIdentifier(String identifier, String key) {
149         String[] idSplit = ParseUtil.whitespaces.split(identifier);
150         boolean found = false;
151         for (String s : idSplit) {
152             // If key string found, return next value
153             if (found) {
154                 return s;
155             }
156             found = s.equals(key);
157         }
158         // If key string not found, return empty string
159         return "";
160     }

162     @Override
163     protected Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> initProcessorCounts() {
164         if (VersionHelpers.IsWindows7OrGreater()) {
165             Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> procs = LogicalProcessorInformation
166                     .getLogicalProcessorInformationEx();
167             // Save numaNode,Processor lookup for future PerfCounter instance lookup
168             // The processor number is based on the Processor Group, so we keep a separate
169             // index by NUMA node.
170             int curNode = -1;
171             int procNum = 0;
172             // 0-indexed list of all lps for array lookup
173             int lp = 0;
174             this.numaNodeProcToLogicalProcMap = new HashMap<>();
175             for (LogicalProcessor logProc : procs.getA()) {
176                 int node = logProc.getNumaNode();
177                 // This list is grouped by NUMA node so a change in node will reset this counter
178                 if (node != curNode) {
179                     curNode = node;
180                     procNum = 0;
181                 }
182                 numaNodeProcToLogicalProcMap.put(String.format("%d,%d", logProc.getNumaNode(), procNum++), lp++);
183             }
184             return procs;
185         } else {
186             return LogicalProcessorInformation.getLogicalProcessorInformation();
187         }
188     }

190     @Override
191     public long[] querySystemCpuLoadTicks() {
192         // To get load in processor group scenario, we need perfmon counters, but the
193         // _Total instance is an average rather than total (scaled) number of ticks
194         // which matches GetSystemTimes() results. We can just query the per-processor
195         // ticks and add them up. Calling the get() method gains the benefit of
196         // synchronizing this output with the memoized result of per-processor ticks as
197         // well.
198         long[] ticks = new long[TickType.values().length];
199         // Sum processor ticks
200         long[][] procTicks = getProcessorCpuLoadTicks();
201         for (int i = 0; i < ticks.length; i++) {
202             for (long[] procTick : procTicks) {
203                 ticks[i] += procTick[i];
204             }
205         }
206         return ticks;
207     }

209     @Override
210     public long[] queryCurrentFreq() {
211         if (VersionHelpers.IsWindows7OrGreater()) {
212             Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> instanceValuePair = ProcessorInformation
213                     .queryFrequencyCounters();
214             List<String> instances = instanceValuePair.getA();
215             Map<ProcessorFrequencyProperty, List<Long>> valueMap = instanceValuePair.getB();
216             List<Long> percentMaxList = valueMap.get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
217             if (!instances.isEmpty()) {
218                 long maxFreq = this.getMaxFreq();
219                 long[] freqs = new long[getLogicalProcessorCount()];
220                 for (String instance : instances) {
221                     int cpu = instance.contains(",") ? numaNodeProcToLogicalProcMap.getOrDefault(instance, 0)
222                             : ParseUtil.parseIntOrDefault(instance, 0);
223                     if (cpu >= getLogicalProcessorCount()) {
224                         continue;
225                     }
226                     freqs[cpu] = percentMaxList.get(cpu) * maxFreq / 100L;
227                 }
228                 return freqs;
229             }
230         }
231         // If <Win7 or anything failed in PDH/WMI, use the native call
232         return queryNTPower(2); // Current is field index 2
233     }

235     @Override
236     public long queryMaxFreq() {
237         long[] freqs = queryNTPower(1); // Max is field index 1
238         return Arrays.stream(freqs).max().orElse(-1L);
239     }

241     /**
242      * Call CallNTPowerInformation for Processor information and return an array of the specified index
243      *
244      * @param fieldIndex The field, in order as defined in the {@link PowrProf#PROCESSOR_INFORMATION} structure.
245      * @return The array of values.
246      */
247     private long[] queryNTPower(int fieldIndex) {
248         ProcessorPowerInformation ppi = new ProcessorPowerInformation();
249         ProcessorPowerInformation[] ppiArray = (ProcessorPowerInformation[]) ppi.toArray(getLogicalProcessorCount());
250         long[] freqs = new long[getLogicalProcessorCount()];
251         if (0 != PowrProf.INSTANCE.CallNtPowerInformation(POWER_INFORMATION_LEVEL.ProcessorInformation, null, 0,
252                 ppiArray[0].getPointer(), ppi.size() * ppiArray.length)) {
253             LOG.error("Unable to get Processor Information");
254             Arrays.fill(freqs, -1L);
255             return freqs;
256         }
257         for (int i = 0; i < freqs.length; i++) {
258             if (fieldIndex == 1) { // Max
259                 freqs[i] = ppiArray[i].maxMhz * 1_000_000L;
260             } else if (fieldIndex == 2) { // Current
261                 freqs[i] = ppiArray[i].currentMhz * 1_000_000L;
262             } else {
263                 freqs[i] = -1L;
264             }
265         }
266         return freqs;
267     }

277     @Override
278     public long[][] queryProcessorCpuLoadTicks() {
279         // These are used in all cases
280         List<String> instances;
281         List<Long> systemList;
282         List<Long> userList;
283         List<Long> irqList;
284         List<Long> softIrqList;
285         List<Long> idleList;
286         // These are only used with USE_CPU_UTILITY
287         List<Long> baseList = null;
288         List<Long> systemUtility = null;
289         List<Long> processorUtility = null;
290         List<Long> processorUtilityBase = null;
291         List<Long> initSystemList = null;
292         List<Long> initUserList = null;
293         List<Long> initBase = null;
294         List<Long> initSystemUtility = null;
295         List<Long> initProcessorUtility = null;
296         List<Long> initProcessorUtilityBase = null;
297         if (USE_CPU_UTILITY) {
298             Pair<List<String>, Map<ProcessorUtilityTickCountProperty, List<Long>>> instanceValuePair = processorUtilityCounters
299                     .get();
300             instances = instanceValuePair.getA();
301             Map<ProcessorUtilityTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
302             systemList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
303             userList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
304             irqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTINTERRUPTTIME);
305             softIrqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTDPCTIME);
306             // % Processor Time is actually Idle time
307             idleList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORTIME);
308             baseList = valueMap.get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
309             // Utility ticks, if configured
310             systemUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
311             processorUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
312             processorUtilityBase = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
313 
314             initSystemList = initialUtilityCounters.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
315             initUserList = initialUtilityCounters.get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
316             initBase = initialUtilityCounters.get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
317             // Utility ticks, if configured
318             initSystemUtility = initialUtilityCounters.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
319             initProcessorUtility = initialUtilityCounters
320                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
321             initProcessorUtilityBase = initialUtilityCounters
322                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
323         } else {
324             Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = ProcessorInformation
325                     .queryProcessorCounters();
326             instances = instanceValuePair.getA();
327             Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
328             systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
329             userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
330             irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
331             softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
332             // % Processor Time is actually Idle time
333             idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
334         }
335 
336         int ncpu = getLogicalProcessorCount();
337         long[][] ticks = new long[ncpu][TickType.values().length];
338         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
339                 || idleList == null
340                 || (USE_CPU_UTILITY && (baseList == null || systemUtility == null || processorUtility == null
341                         || processorUtilityBase == null || initSystemList == null || initUserList == null
342                         || initBase == null || initSystemUtility == null || initProcessorUtility == null
343                         || initProcessorUtilityBase == null))) {
344             return ticks;
345         }
346         for (String instance : instances) {
347             int cpu = instance.contains(",") ? numaNodeProcToLogicalProcMap.getOrDefault(instance, 0)
348                     : ParseUtil.parseIntOrDefault(instance, 0);
349             if (cpu >= ncpu) {
350                 continue;
351             }
352             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
353             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
354             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
355             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
356             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
357 
358             // If users want Task Manager output we have to do some math to get there
359             if (USE_CPU_UTILITY) {
360                 // We have two new capacity numbers, processor (all but idle) and system
361                 // (included in processor). To further complicate matters, these are in percent
362                 // units so must be divided by 100.
363 
364                 // The 100NS elapsed time counter is a constant multiple of the capacity base
365                 // counter. By enforcing a memoized pause we'll either have zero elapsed time or
366                 // sufficient delay to determine this offset reliably once and not have to
367                 // recalculate it
368 
369                 // Get elapsed time in 100NS
370                 long deltaT = baseList.get(cpu) - initBase.get(cpu);
371                 if (deltaT > 0) {
372                     // Get elapsed utility base
373                     long deltaBase = processorUtilityBase.get(cpu) - initProcessorUtilityBase.get(cpu);
374                     // The ratio of elapsed clock to elapsed utility base is an integer constant.
375                     // We can calculate a conversion factor to ensure a consistent application of
376                     // the correction. Since Utility is in percent, this is actually 100x the true
377                     // multiplier but is at the level where the integer calculation is precise
378                     long multiplier = lazilyCalculateMultiplier(deltaBase, deltaT);
379 
380                     // 0 multiplier means we just re-initialized ticks
381                     if (multiplier > 0) {
382                         // Get utility delta
383                         long deltaProc = processorUtility.get(cpu) - initProcessorUtility.get(cpu);
384                         long deltaSys = systemUtility.get(cpu) - initSystemUtility.get(cpu);
385 
386                         // Calculate new target ticks
387                         // Correct for the 100x multiplier at the end
388                         long newUser = initUserList.get(cpu) + multiplier * (deltaProc - deltaSys) / 100;
389                         long newSystem = initSystemList.get(cpu) + multiplier * deltaSys / 100;
390 
391                         // Adjust user to new, saving the delta
392                         long delta = newUser - ticks[cpu][TickType.USER.getIndex()];
393                         ticks[cpu][TickType.USER.getIndex()] = newUser;
394                         // Do the same for system
395                         delta += newSystem - ticks[cpu][TickType.SYSTEM.getIndex()];
396                         ticks[cpu][TickType.SYSTEM.getIndex()] = newSystem;
397                         // Subtract delta from idle
398                         ticks[cpu][TickType.IDLE.getIndex()] -= delta;
399                     }
400                 }
401             }
402 
403             // Decrement IRQ from system to avoid double counting in the total array
404             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
405                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
406 
407             // Raw value is cumulative 100NS-ticks
408             // Divide by 10_000 to get milliseconds
409             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
410             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
411             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
412             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
413             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
414         }
415         // Skipping nice and IOWait, they'll stay 0
416         return ticks;
417     }

419     /**
420      * Lazily calculate the capacity tick multiplier once.
421      *
422      * @param deltaBase The difference in base ticks.
423      * @param deltaT    The difference in elapsed 100NS time
424      * @return The ratio of elapsed time to base ticks
425      */
426     private synchronized long lazilyCalculateMultiplier(long deltaBase, long deltaT) {
427         if (utilityBaseMultiplier == null) {
428             // If too much time has elapsed from class instantiation, re-initialize the
429             // ticks and return without calculating. Approx 7 minutes for 100NS counter to
430             // exceed max unsigned int.
431             if (deltaT >> 32 > 0) {
432                 initialUtilityCounters = processorUtilityCounters.get().getB();
433                 return 0L;
434             }
435             // Base counter wraps approximately every 115 minutes
436             // If deltaBase is nonpositive assume it has wrapped
437             if (deltaBase <= 0) {
438                 deltaBase += 1L << 32;
439             }
440             long multiplier = Math.round((double) deltaT / deltaBase);
441             // If not enough time has elapsed, return the value this one time but don't
442             // persist. 5000 ms = 50 million 100NS ticks
443             if (deltaT < 50_000_000L) {
444                 return multiplier;
445             }
446             utilityBaseMultiplier = multiplier;
447         }
448         return utilityBaseMultiplier;
449     }
