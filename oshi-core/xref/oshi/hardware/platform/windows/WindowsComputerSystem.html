47  /**
48   * Hardware data obtained from WMI.
49   */
50  @Immutable
51  final class WindowsComputerSystem extends AbstractComputerSystem {
52  
53      private final Supplier<Pair<String, String>> manufacturerModel = memoize(
54              WindowsComputerSystem::queryManufacturerModel);
55      private final Supplier<Pair<String, String>> serialNumberUUID = memoize(
56              WindowsComputerSystem::querySystemSerialNumberUUID);
57  
58      @Override
59      public String getManufacturer() {
60          return manufacturerModel.get().getA();
61      }
62  
63      @Override
64      public String getModel() {
65          return manufacturerModel.get().getB();
66      }
67  
68      @Override
69      public String getSerialNumber() {
70          return serialNumberUUID.get().getA();
71      }
72  
73      @Override
74      public String getHardwareUUID() {
75          return serialNumberUUID.get().getB();
76      }
77  
78      @Override
79      public Firmware createFirmware() {
80          return new WindowsFirmware();
81      }
82  
83      @Override
84      public Baseboard createBaseboard() {
85          return new WindowsBaseboard();
86      }
87  
88      private static Pair<String, String> queryManufacturerModel() {
89          String manufacturer = null;
90          String model = null;
91          WmiResult<ComputerSystemProperty> win32ComputerSystem = Win32ComputerSystem.queryComputerSystem();
92          if (win32ComputerSystem.getResultCount() > 0) {
93              manufacturer = WmiUtil.getString(win32ComputerSystem, ComputerSystemProperty.MANUFACTURER, 0);
94              model = WmiUtil.getString(win32ComputerSystem, ComputerSystemProperty.MODEL, 0);
95          }
96          return new Pair<>(Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer,
97                  Util.isBlank(model) ? Constants.UNKNOWN : model);
98      }
99  
100     private static Pair<String, String> querySystemSerialNumberUUID() {
101         String serialNumber = null;
102         String uuid = null;
103         WmiResult<ComputerSystemProductProperty> win32ComputerSystemProduct = Win32ComputerSystemProduct
104                 .queryIdentifyingNumberUUID();
105         if (win32ComputerSystemProduct.getResultCount() > 0) {
106             serialNumber = WmiUtil.getString(win32ComputerSystemProduct,
107                     ComputerSystemProductProperty.IDENTIFYINGNUMBER, 0);
108             uuid = WmiUtil.getString(win32ComputerSystemProduct, ComputerSystemProductProperty.UUID, 0);
109         }
110         if (Util.isBlank(serialNumber)) {
111             serialNumber = querySerialFromBios();
112         }
113         if (Util.isBlank(serialNumber)) {
114             serialNumber = Constants.UNKNOWN;
115         }
116         if (Util.isBlank(uuid)) {
117             uuid = Constants.UNKNOWN;
118         }
119         return new Pair<>(serialNumber, uuid);
120     }
121 
122     private static String querySerialFromBios() {
123         WmiResult<BiosSerialProperty> serialNum = Win32Bios.querySerialNumber();
124         if (serialNum.getResultCount() > 0) {
125             return WmiUtil.getString(serialNum, BiosSerialProperty.SERIALNUMBER, 0);
126         }
127         return null;
128     }
129 }
