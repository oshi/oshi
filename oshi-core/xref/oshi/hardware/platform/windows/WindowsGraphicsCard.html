24  /**
25   * Graphics Card obtained from WMI
26   */
27  @Immutable
28  final class WindowsGraphicsCard extends AbstractGraphicsCard {
29  
30      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
31  
32      /**
33       * Constructor for WindowsGraphicsCard
34       *
35       * @param name        The name
36       * @param deviceId    The device ID
37       * @param vendor      The vendor
38       * @param versionInfo The version info
39       * @param vram        The VRAM
40       */
41      WindowsGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
42          super(name, deviceId, vendor, versionInfo, vram);
43      }
44  
45      /**
46       * public method used by {@link oshi.hardware.common.AbstractHardwareAbstractionLayer} to access the graphics cards.
47       *
48       * @return List of {@link oshi.hardware.platform.windows.WindowsGraphicsCard} objects.
49       */
50      public static List<GraphicsCard> getGraphicsCards() {
51          List<GraphicsCard> cardList = new ArrayList<>();
52          if (IS_VISTA_OR_GREATER) {
53              WmiResult<VideoControllerProperty> cards = Win32VideoController.queryVideoController();
54              for (int index = 0; index < cards.getResultCount(); index++) {
55                  String name = WmiUtil.getString(cards, VideoControllerProperty.NAME, index);
56                  Triplet<String, String, String> idPair = ParseUtil.parseDeviceIdToVendorProductSerial(
57                          WmiUtil.getString(cards, VideoControllerProperty.PNPDEVICEID, index));
58                  String deviceId = idPair == null ? Constants.UNKNOWN : idPair.getB();
59                  String vendor = WmiUtil.getString(cards, VideoControllerProperty.ADAPTERCOMPATIBILITY, index);
60                  if (idPair != null) {
61                      if (Util.isBlank(vendor)) {
62                          deviceId = idPair.getA();
63                      } else {
64                          vendor = vendor + " (" + idPair.getA() + ")";
65                      }
66                  }
67                  String versionInfo = WmiUtil.getString(cards, VideoControllerProperty.DRIVERVERSION, index);
68                  if (!Util.isBlank(versionInfo)) {
69                      versionInfo = "DriverVersion=" + versionInfo;
70                  } else {
71                      versionInfo = Constants.UNKNOWN;
72                  }
73                  long vram = WmiUtil.getUint32asLong(cards, VideoControllerProperty.ADAPTERRAM, index);
74                  cardList.add(new WindowsGraphicsCard(Util.isBlank(name) ? Constants.UNKNOWN : name, deviceId,
75                          Util.isBlank(vendor) ? Constants.UNKNOWN : vendor, versionInfo, vram));
76              }
77          }
78          return cardList;
79      }
80  }
