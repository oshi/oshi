53      enum USBControllerProperty {
54          PNPDEVICEID;
55      }

57      private static List<String> controllerDeviceIdList = new ArrayList<>();
58      static {
59          // One time lookup of USB Controller PnP Device IDs which don't change
60          WmiQuery<USBControllerProperty> usbControllerQuery = WbemcliUtil.createQuery("Win32_USBController",
61                  USBControllerProperty.class);
62          WmiResult<USBControllerProperty> usbController = WmiUtil.queryWMI(usbControllerQuery);
63          for (int i = 0; i < usbController.getResultCount(); i++) {
64              controllerDeviceIdList.add(WmiUtil.getString(usbController, USBControllerProperty.PNPDEVICEID, i));
65          }
66      }

84      private static final Pattern VENDOR_PRODUCT_ID = Pattern
85              .compile(".*(?:VID|VEN)_(\\p{XDigit}{4})&(?:PID|DEV)_(\\p{XDigit}{4}).*");

88       * Map to store information using PNPDeviceID as the key
89       */
90      private static Map<String, WindowsUsbDevice> usbDeviceCache = new HashMap<>();

93       * Sets to track USB devices in cache

96      private static Set<String> devicesToRemove = new HashSet<>();

103     public WindowsUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
104             UsbDevice[] connectedDevices) {

108     /**
109      * {@inheritDoc}
110      */
111     public static UsbDevice[] getUsbDevices(boolean tree) {
112         UsbDevice[] devices = getUsbDevices();
113         if (tree) {
114             return devices;
115         }
116         List<UsbDevice> deviceList = new ArrayList<>();
117         // Top level is controllers; they won't be added to the list, but all
118         // their connected devices will be
119         for (UsbDevice device : devices) {
120             addDevicesToList(deviceList, device.getConnectedDevices());
121         }
122         return deviceList.toArray(new UsbDevice[deviceList.size()]);
123     }
124 
125     private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
126         for (UsbDevice device : connectedDevices) {
127             deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
128                     device.getProductId(), device.getSerialNumber(), new UsbDevice[0]));
129             addDevicesToList(deviceList, device.getConnectedDevices());
130         }
131     }

133     private static UsbDevice[] getUsbDevices() {
134         // The controllerDeviceIDList contains the PnPDeviceIDs of the
135         // controllers.
136 
137         // The deviceCache stores UsbDevices that we have already returned
138         // Navigate the device tree to see what's present: remove from cache if
139         // no longer present and add to cache if not present
140 
141         // Clear recursive ID map and set up sets
142         deviceTreeMap.clear();
143         // Add any seen in tree that aren't in cache
144         devicesToAdd.clear();
145         // Also remove from cache if not seen in tree
146         devicesToRemove = new HashSet<>(usbDeviceCache.keySet());
147 
148         List<WindowsUsbDevice> controllerDevices = new ArrayList<>();
149         for (String controllerDeviceId : controllerDeviceIdList) {
150             putChildrenInDeviceTree(controllerDeviceId, 0);
151             updateDeviceCache();
152             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(controllerDeviceId, "0000", "0000");
153             if (deviceAndChildren != null) {
154                 controllerDevices.add(deviceAndChildren);
155             }
156         }
157         return controllerDevices.toArray(new WindowsUsbDevice[controllerDevices.size()]);
158 
159     }

161     private static void updateDeviceCache() {
162 
163         // Remove devices no longer in tree
164         for (String deviceID : devicesToRemove) {
165             usbDeviceCache.remove(deviceID);
166             LOG.debug("Removing {} from USB device cache.", deviceID);
167         }
168         devicesToRemove.clear();
169         // Create list to add
170         if (!devicesToAdd.isEmpty()) {
171             StringBuilder sb = new StringBuilder();
172             boolean first = true;
173             for (String deviceID : devicesToAdd) {
174                 if (first) {
175                     sb.append(" WHERE (PnPDeviceID=\"");
176                     first = false;
177                 } else {
178                     sb.append(" OR (PnPDeviceID=\"");
179                 }
180                 sb.append(deviceID).append("\")");
181             }
182             String whereClause = sb.toString();
183             // Query Win32_PnPEntity to populate the maps
184             PNPENTITY_QUERY.setWmiClassName(PNPENTITY_BASE_CLASS + whereClause);
185             WmiResult<PnPEntityProperty> pnpEntity = WmiUtil.queryWMI(PNPENTITY_QUERY);
186             for (int i = 0; i < pnpEntity.getResultCount(); i++) {
187                 String pnpDeviceID = WmiUtil.getString(pnpEntity, PnPEntityProperty.PNPDEVICEID, i);
188                 String name = WmiUtil.getString(pnpEntity, PnPEntityProperty.NAME, i);
189                 String vendor = WmiUtil.getString(pnpEntity, PnPEntityProperty.MANUFACTURER, i);
190                 WindowsUsbDevice device = new WindowsUsbDevice(name, vendor, null, null, "", new WindowsUsbDevice[0]);
191                 usbDeviceCache.put(pnpDeviceID, device);
192                 LOG.debug("Adding {} to USB device cache.", pnpDeviceID);
193             }
194             // Get serial # for disk drives or other physical media
195             DISKDRIVE_QUERY.setWmiClassName(DISKDRIVE_BASE_CLASS + whereClause);
196             WmiResult<DiskDriveProperty> serialNumber = WmiUtil.queryWMI(DISKDRIVE_QUERY);
197             for (int i = 0; i < serialNumber.getResultCount(); i++) {
198                 String pnpDeviceID = WmiUtil.getString(serialNumber, DiskDriveProperty.PNPDEVICEID, i);
199                 if (usbDeviceCache.containsKey(pnpDeviceID)) {
200                     WindowsUsbDevice device = usbDeviceCache.get(pnpDeviceID);
201                     device.serialNumber = ParseUtil
202                             .hexStringToString(WmiUtil.getString(serialNumber, DiskDriveProperty.SERIALNUMBER, i));
203                 }
204             }
205         }
206     }

208     /**
209      * Navigates the Device Tree to place all children PNPDeviceIDs into the map
210      * for the specified deviceID. Recursively adds children's children, etc.
211      *
212      * @param deviceId
213      *            The device to add respective children to the map
214      * @param deviceInstance
215      *            The device instance (devnode handle), if known. If set to 0,
216      *            the code will search for a match.
217      */
218     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance) {
219         // Track deviceIDs to add/remove from cache
220         devicesToRemove.remove(deviceId);
221         if (!usbDeviceCache.containsKey(deviceId)) {
222             devicesToAdd.add(deviceId);
223         }
224         // If no devInst provided, find it
225         int devInst = deviceInstance;
226         if (devInst == 0) {
227             IntByReference pdnDevInst = new IntByReference();
228             Cfgmgr32.INSTANCE.CM_Locate_DevNode(pdnDevInst, deviceId, 0);
229             devInst = pdnDevInst.getValue();
230         }
231         // Now iterate the children. Call CM_Get_Child to get first child
232         IntByReference child = new IntByReference();
233         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
234             // Add first child to a list
235             List<String> childList = new ArrayList<>();
236             String childId = Cfgmgr32Util.CM_Get_Device_ID(child.getValue());
237             childList.add(childId);
238             deviceTreeMap.put(deviceId, childList);
239             putChildrenInDeviceTree(childId, child.getValue());
240             // Find any other children
241             IntByReference sibling = new IntByReference();
242             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
243                 // Add to the list
244                 String siblingId = Cfgmgr32Util.CM_Get_Device_ID(sibling.getValue());
245                 deviceTreeMap.get(deviceId).add(siblingId);
246                 putChildrenInDeviceTree(siblingId, sibling.getValue());
247                 // Make this sibling the new child to find other siblings
248                 child = sibling;
249             }
250         }
251     }

253     /**
254      * Recursively creates WindowsUsbDevices by fetching information from maps
255      * to populate fields
256      *
257      * @param hubDeviceId
258      *            The PNPdeviceID of this device.
259      * @param vid
260      *            The default (parent) vendor ID
261      * @param pid
262      *            The default (parent) product ID
263      * @return A WindowsUsbDevice corresponding to this deviceID, or null if
264      *         unable to find
265      */
266     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid) {
267         String vendorId = vid;
268         String productId = pid;
269         Matcher m = VENDOR_PRODUCT_ID.matcher(hubDeviceId);
270         if (m.matches()) {
271             vendorId = m.group(1).toLowerCase();
272             productId = m.group(2).toLowerCase();
273         }
274         List<String> pnpDeviceIds = MapUtil.getOrDefault(deviceTreeMap, hubDeviceId, new ArrayList<String>());
275         List<WindowsUsbDevice> usbDevices = new ArrayList<>();
276         for (String pnpDeviceId : pnpDeviceIds) {
277             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(pnpDeviceId, vendorId, productId);
278             if (deviceAndChildren != null) {
279                 usbDevices.add(deviceAndChildren);
280             }
281         }
282         Collections.sort(usbDevices);
283         if (usbDeviceCache.containsKey(hubDeviceId)) {
284             WindowsUsbDevice device = usbDeviceCache.get(hubDeviceId);
285             if (device.name.isEmpty()) {
286                 device.name = vendorId + ":" + productId;
287             }
288             device.vendorId = vendorId;
289             device.productId = productId;
290             device.connectedDevices = usbDevices.toArray(new WindowsUsbDevice[usbDevices.size()]);
291             return device;
292         }
293         return null;
294     }
