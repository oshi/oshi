72      /**
73       * {@inheritDoc}
74       *
75       * @param tree
76       *            a boolean.
77       * @return an array of {@link oshi.hardware.UsbDevice} objects.
78       */
79      public static UsbDevice[] getUsbDevices(boolean tree) {
80          UsbDevice[] devices = getUsbDevices();
81          if (tree) {
82              return devices;
83          }
84          List<UsbDevice> deviceList = new ArrayList<>();
85          // Top level is controllers; they won't be added to the list, but all
86          // their connected devices will be
87          for (UsbDevice device : devices) {
88              addDevicesToList(deviceList, device.getConnectedDevices());
89          }
90          return deviceList.toArray(new UsbDevice[0]);
91      }

93      private static UsbDevice[] getUsbDevices() {
94          // Map to build the recursive tree structure
95          Map<String, List<String>> deviceTreeMap = new HashMap<>();
96          // Track devices seen in the process
97          Set<String> devicesSeen = new HashSet<>();
98  
99          // Navigate the device tree to track what devices are present
100         List<WindowsUsbDevice> controllerDevices = new ArrayList<>();
101         List<String> controllerDeviceIdList = getControllerDeviceIdList();
102         for (String controllerDeviceId : controllerDeviceIdList) {
103             putChildrenInDeviceTree(controllerDeviceId, 0, deviceTreeMap, devicesSeen);
104         }
105         // Map to store information using PNPDeviceID as the key.
106         Map<String, WindowsUsbDevice> usbDeviceCache = populateDeviceCache(devicesSeen);
107         // recursively build results
108         for (String controllerDeviceId : controllerDeviceIdList) {
109             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(controllerDeviceId, "0000", "0000", deviceTreeMap,
110                     usbDeviceCache);
111             if (deviceAndChildren != null) {
112                 controllerDevices.add(deviceAndChildren);
113             }
114         }
115         return controllerDevices.toArray(new WindowsUsbDevice[0]);
116     }

118     private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
119         for (UsbDevice device : connectedDevices) {
120             deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
121                     device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(), new UsbDevice[0]));
122             addDevicesToList(deviceList, device.getConnectedDevices());
123         }
124     }

126     private static Map<String, WindowsUsbDevice> populateDeviceCache(Set<String> devicesToAdd) {
127         Map<String, WindowsUsbDevice> usbDeviceCache = new HashMap<>();
128         // Add devices not in the tree
129         if (!devicesToAdd.isEmpty()) {
130             StringBuilder sb = new StringBuilder();
131             boolean first = true;
132             for (String deviceID : devicesToAdd) {
133                 if (first) {
134                     sb.append(" WHERE (PnPDeviceID=\"");
135                     first = false;
136                 } else {
137                     sb.append(" OR (PnPDeviceID=\"");
138                 }
139                 sb.append(deviceID).append("\")");
140             }
141             String whereClause = sb.toString();
142             // Query Win32_PnPEntity to populate the maps
143             WmiResult<PnPEntityProperty> pnpEntity = new Win32PnPEntity().queryDeviceId(whereClause);
144             for (int i = 0; i < pnpEntity.getResultCount(); i++) {
145                 String pnpDeviceID = WmiUtil.getString(pnpEntity, PnPEntityProperty.PNPDEVICEID, i);
146                 String name = WmiUtil.getString(pnpEntity, PnPEntityProperty.NAME, i);
147                 String vendor = WmiUtil.getString(pnpEntity, PnPEntityProperty.MANUFACTURER, i);
148                 WindowsUsbDeviceUsbDevice.html#WindowsUsbDevice">WindowsUsbDevice device = new WindowsUsbDevice(name, vendor, null, null, "", pnpDeviceID,
149                         new WindowsUsbDevice[0]);
150                 usbDeviceCache.put(pnpDeviceID, device);
151                 LOG.debug("Adding {} to USB device cache.", pnpDeviceID);
152             }
153             // Get serial # for disk drives or other physical media
154             WmiResult<DeviceIdProperty> serialNumbers = new Win32DiskDrive().queryDiskDriveId(whereClause);
155             for (int i = 0; i < serialNumbers.getResultCount(); i++) {
156                 String pnpDeviceID = WmiUtil.getString(serialNumbers, DeviceIdProperty.PNPDEVICEID, i);
157                 if (usbDeviceCache.containsKey(pnpDeviceID)) {
158                     WindowsUsbDevice device = usbDeviceCache.get(pnpDeviceID);
159                     device.serialNumber = ParseUtil
160                             .hexStringToString(WmiUtil.getString(serialNumbers, DeviceIdProperty.SERIALNUMBER, i));
161                 }
162             }
163         }
164         return usbDeviceCache;
165     }

167     /**
168      * Navigates the Device Tree to place all children PNPDeviceIDs into the map for
169      * the specified deviceID. Recursively adds children's children, etc.
170      *
171      * @param deviceId
172      *            The device to add respective children to the map
173      * @param deviceInstance
174      *            The device instance (devnode handle), if known. If set to 0, the
175      *            code will search for a match.
176      * @param deviceTreeMap
177      *            The overall device tree map that starts at the controllers
178      * @param devicesSeen
179      *            Devices we've seen so we can add/remove from cache later
180      */
181     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance,
182             Map<String, List<String>> deviceTreeMap, Set<String> devicesSeen) {
183         devicesSeen.add(deviceId);
184         // If no devInst provided, find it
185         int devInst = deviceInstance;
186         if (devInst == 0) {
187             IntByReference pdnDevInst = new IntByReference();
188             Cfgmgr32.INSTANCE.CM_Locate_DevNode(pdnDevInst, deviceId, 0);
189             devInst = pdnDevInst.getValue();
190         }
191         // Now iterate the children. Call CM_Get_Child to get first child
192         IntByReference child = new IntByReference();
193         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
194             // Add first child to a list
195             List<String> childList = new ArrayList<>();
196             String childId = Cfgmgr32Util.CM_Get_Device_ID(child.getValue());
197             childList.add(childId);
198             deviceTreeMap.put(deviceId, childList);
199             putChildrenInDeviceTree(childId, child.getValue(), deviceTreeMap, devicesSeen);
200             // Find any other children
201             IntByReference sibling = new IntByReference();
202             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
203                 // Add to the list
204                 String siblingId = Cfgmgr32Util.CM_Get_Device_ID(sibling.getValue());
205                 deviceTreeMap.get(deviceId).add(siblingId);
206                 putChildrenInDeviceTree(siblingId, sibling.getValue(), deviceTreeMap, devicesSeen);
207                 // Make this sibling the new child to find other siblings
208                 child = sibling;
209             }
210         }
211     }

213     /**
214      * Recursively creates WindowsUsbDevices by fetching information from maps to
215      * populate fields
216      *
217      * @param hubDeviceId
218      *            The PNPdeviceID of this device.
219      * @param vid
220      *            The default (parent) vendor ID
221      * @param pid
222      *            The default (parent) product ID
223      * @param deviceTreeMap
224      * @param usbDeviceCache
225      * @return A WindowsUsbDevice corresponding to this deviceID, or null if unable
226      *         to find
227      */
228     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid,
229             Map<String, List<String>> deviceTreeMap, Map<String, WindowsUsbDevice> usbDeviceCache) {
230         String vendorId = vid;
231         String productId = pid;
232         Matcher m = VENDOR_PRODUCT_ID.matcher(hubDeviceId);
233         if (m.matches()) {
234             vendorId = m.group(1).toLowerCase();
235             productId = m.group(2).toLowerCase();
236         }
237         List<String> pnpDeviceIds = deviceTreeMap.getOrDefault(hubDeviceId, new ArrayList<String>());
238         List<WindowsUsbDevice> usbDevices = new ArrayList<>();
239         for (String pnpDeviceId : pnpDeviceIds) {
240             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(pnpDeviceId, vendorId, productId, deviceTreeMap,
241                     usbDeviceCache);
242             if (deviceAndChildren != null) {
243                 usbDevices.add(deviceAndChildren);
244             }
245         }
246         Collections.sort(usbDevices);
247         if (usbDeviceCache.containsKey(hubDeviceId)) {
248             WindowsUsbDevice device = usbDeviceCache.get(hubDeviceId);
249             if (device.name.isEmpty()) {
250                 device.name = vendorId + ":" + productId;
251             }
252             device.vendorId = vendorId;
253             device.productId = productId;
254             device.connectedDevices = usbDevices.toArray(new WindowsUsbDevice[0]);
255             return device;
256         }
257         return null;
258     }

260     /**
261      * Queries the USB Controller list
262      *
263      * @return A list of Strings of USB Controller PNPDeviceIDs
264      */
265     private static List<String> getControllerDeviceIdList() {
266         List<String> controllerDeviceIdsList = new ArrayList<>();
267         // One time lookup of USB Controller PnP Device IDs which don't
268         // change
269         WmiResult<USBControllerProperty> usbController = new Win32USBController().queryUSBControllers();
270         for (int i = 0; i < usbController.getResultCount(); i++) {
271             controllerDeviceIdsList.add(WmiUtil.getString(usbController, USBControllerProperty.PNPDEVICEID, i));
272         }
273         return controllerDeviceIdsList;
274     }
