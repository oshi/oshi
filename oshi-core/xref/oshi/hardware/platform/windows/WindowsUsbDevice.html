89      public static UsbDevice[] getUsbDevices(boolean tree) {
90          UsbDevice[] devices = getUsbDevices();
91          if (tree) {
92              return devices;
93          }
94          List<UsbDevice> deviceList = new ArrayList<>();
95          // Top level is controllers; they won't be added to the list, but all
96          // their connected devices will be
97          for (UsbDevice device : devices) {
98              addDevicesToList(deviceList, device.getConnectedDevices());
99          }
100         return deviceList.toArray(new UsbDevice[0]);
101     }

103     private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
104         for (UsbDevice device : connectedDevices) {
105             deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
106                     device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(), new UsbDevice[0]));
107             addDevicesToList(deviceList, device.getConnectedDevices());
108         }
109     }

111     private static UsbDevice[] getUsbDevices() {
112         // Map to build the recursive tree structure
113         Map<String, List<String>> deviceTreeMap = new HashMap<>();
114         // Track devices seen in the process
115         Set<String> devicesSeen = new HashSet<>();
116 
117         // Navigate the device tree to track what devices are present
118         List<WindowsUsbDevice> controllerDevices = new ArrayList<>();
119         WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
120         List<String> controllerDeviceIdList = getControllerDeviceIdList(wmiQueryHandler);
121         for (String controllerDeviceId : controllerDeviceIdList) {
122             putChildrenInDeviceTree(controllerDeviceId, 0, deviceTreeMap, devicesSeen);
123         }
124         // Map to store information using PNPDeviceID as the key.
125         Map<String, WindowsUsbDevice> usbDeviceCache = populateDeviceCache(devicesSeen, wmiQueryHandler);
126         // recursively build results
127         for (String controllerDeviceId : controllerDeviceIdList) {
128             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(controllerDeviceId, "0000", "0000", deviceTreeMap,
129                     usbDeviceCache);
130             if (deviceAndChildren != null) {
131                 controllerDevices.add(deviceAndChildren);
132             }
133         }
134         return controllerDevices.toArray(new WindowsUsbDevice[0]);
135     }

137     private static Map<String, WindowsUsbDevice> populateDeviceCache(Set<String> devicesToAdd,
138             WmiQueryHandler wmiQueryHandler) {
139         Map<String, WindowsUsbDevice> usbDeviceCache = new HashMap<>();
140         // Add devices not in the tree
141         if (!devicesToAdd.isEmpty()) {
142             StringBuilder sb = new StringBuilder();
143             boolean first = true;
144             for (String deviceID : devicesToAdd) {
145                 if (first) {
146                     sb.append(" WHERE (PnPDeviceID=\"");
147                     first = false;
148                 } else {
149                     sb.append(" OR (PnPDeviceID=\"");
150                 }
151                 sb.append(deviceID).append("\")");
152             }
153             String whereClause = sb.toString();
154             // Query Win32_PnPEntity to populate the maps
155             PNPENTITY_QUERY.setWmiClassName(PNPENTITY_BASE_CLASS + whereClause);
156             WmiResult<PnPEntityProperty> pnpEntity = wmiQueryHandler.queryWMI(PNPENTITY_QUERY);
157             for (int i = 0; i < pnpEntity.getResultCount(); i++) {
158                 String pnpDeviceID = WmiUtil.getString(pnpEntity, PnPEntityProperty.PNPDEVICEID, i);
159                 String name = WmiUtil.getString(pnpEntity, PnPEntityProperty.NAME, i);
160                 String vendor = WmiUtil.getString(pnpEntity, PnPEntityProperty.MANUFACTURER, i);
161                 WindowsUsbDeviceUsbDevice.html#WindowsUsbDevice">WindowsUsbDevice device = new WindowsUsbDevice(name, vendor, null, null, "", pnpDeviceID,
162                         new WindowsUsbDevice[0]);
163                 usbDeviceCache.put(pnpDeviceID, device);
164                 LOG.debug("Adding {} to USB device cache.", pnpDeviceID);
165             }
166             // Get serial # for disk drives or other physical media
167             DISKDRIVE_QUERY.setWmiClassName(DISKDRIVE_BASE_CLASS + whereClause);
168             WmiResult<DiskDriveProperty> serialNumber = wmiQueryHandler.queryWMI(DISKDRIVE_QUERY);
169             for (int i = 0; i < serialNumber.getResultCount(); i++) {
170                 String pnpDeviceID = WmiUtil.getString(serialNumber, DiskDriveProperty.PNPDEVICEID, i);
171                 if (usbDeviceCache.containsKey(pnpDeviceID)) {
172                     WindowsUsbDevice device = usbDeviceCache.get(pnpDeviceID);
173                     device.serialNumber = ParseUtil
174                             .hexStringToString(WmiUtil.getString(serialNumber, DiskDriveProperty.SERIALNUMBER, i));
175                 }
176             }
177         }
178         return usbDeviceCache;
179     }

181     /**
182      * Navigates the Device Tree to place all children PNPDeviceIDs into the map
183      * for the specified deviceID. Recursively adds children's children, etc.
184      * 
185      * @param deviceId
186      *            The device to add respective children to the map
187      * @param deviceInstance
188      *            The device instance (devnode handle), if known. If set to 0,
189      *            the code will search for a match.
190      * @param deviceTreeMap
191      *            The overall device tree map that starts at the controllers
192      * @param devicesSeen
193      *            Devices we've seen so we can add/remove from cache later
194      */
195     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance,
196             Map<String, List<String>> deviceTreeMap, Set<String> devicesSeen) {
197         devicesSeen.add(deviceId);
198         // If no devInst provided, find it
199         int devInst = deviceInstance;
200         if (devInst == 0) {
201             IntByReference pdnDevInst = new IntByReference();
202             Cfgmgr32.INSTANCE.CM_Locate_DevNode(pdnDevInst, deviceId, 0);
203             devInst = pdnDevInst.getValue();
204         }
205         // Now iterate the children. Call CM_Get_Child to get first child
206         IntByReference child = new IntByReference();
207         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
208             // Add first child to a list
209             List<String> childList = new ArrayList<>();
210             String childId = Cfgmgr32Util.CM_Get_Device_ID(child.getValue());
211             childList.add(childId);
212             deviceTreeMap.put(deviceId, childList);
213             putChildrenInDeviceTree(childId, child.getValue(), deviceTreeMap, devicesSeen);
214             // Find any other children
215             IntByReference sibling = new IntByReference();
216             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
217                 // Add to the list
218                 String siblingId = Cfgmgr32Util.CM_Get_Device_ID(sibling.getValue());
219                 deviceTreeMap.get(deviceId).add(siblingId);
220                 putChildrenInDeviceTree(siblingId, sibling.getValue(), deviceTreeMap, devicesSeen);
221                 // Make this sibling the new child to find other siblings
222                 child = sibling;
223             }
224         }
225     }

227     /**
228      * Recursively creates WindowsUsbDevices by fetching information from maps
229      * to populate fields
230      *
231      * @param hubDeviceId
232      *            The PNPdeviceID of this device.
233      * @param vid
234      *            The default (parent) vendor ID
235      * @param pid
236      *            The default (parent) product ID
237      * @param deviceTreeMap
238      * @param usbDeviceCache
239      * @return A WindowsUsbDevice corresponding to this deviceID, or null if
240      *         unable to find
241      */
242     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid,
243             Map<String, List<String>> deviceTreeMap, Map<String, WindowsUsbDevice> usbDeviceCache) {
244         String vendorId = vid;
245         String productId = pid;
246         Matcher m = VENDOR_PRODUCT_ID.matcher(hubDeviceId);
247         if (m.matches()) {
248             vendorId = m.group(1).toLowerCase();
249             productId = m.group(2).toLowerCase();
250         }
251         List<String> pnpDeviceIds = deviceTreeMap.getOrDefault(hubDeviceId, new ArrayList<String>());
252         List<WindowsUsbDevice> usbDevices = new ArrayList<>();
253         for (String pnpDeviceId : pnpDeviceIds) {
254             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(pnpDeviceId, vendorId, productId, deviceTreeMap,
255                     usbDeviceCache);
256             if (deviceAndChildren != null) {
257                 usbDevices.add(deviceAndChildren);
258             }
259         }
260         Collections.sort(usbDevices);
261         if (usbDeviceCache.containsKey(hubDeviceId)) {
262             WindowsUsbDevice device = usbDeviceCache.get(hubDeviceId);
263             if (device.name.isEmpty()) {
264                 device.name = vendorId + ":" + productId;
265             }
266             device.vendorId = vendorId;
267             device.productId = productId;
268             device.connectedDevices = usbDevices.toArray(new WindowsUsbDevice[0]);
269             return device;
270         }
271         return null;
272     }

274     /**
275      * Queries the USB Controller list, which doesn't change so we only need to
276      * query it once
277      * 
278      * @param wmiQueryHandler
279      * 
280      * @return A list of Strings of USB Controller PNPDeviceIDs
281      */
282     private static List<String> getControllerDeviceIdList(WmiQueryHandler wmiQueryHandler) {
283         if (controllerDeviceIds == null) {
284             controllerDeviceIds = new ArrayList<>();
285             // One time lookup of USB Controller PnP Device IDs which don't
286             // change
287             WmiQuery<USBControllerProperty> usbControllerQuery = new WmiQuery<>("Win32_USBController",
288                     USBControllerProperty.class);
289             WmiResult<USBControllerProperty> usbController = wmiQueryHandler.queryWMI(usbControllerQuery);
290             for (int i = 0; i < usbController.getResultCount(); i++) {
291                 controllerDeviceIds.add(WmiUtil.getString(usbController, USBControllerProperty.PNPDEVICEID, i));
292             }
293         }
294         return controllerDeviceIds;
295     }
