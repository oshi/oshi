122     /**
123      * <p>
124      * updateDiskStats.
125      * </p>
126      *
127      * @param diskStore
128      *            a {@link oshi.hardware.HWDiskStore} object.
129      * @return a boolean.
130      */
131     public static boolean updateDiskStats(HWDiskStore diskStore) {
132         String index = null;
133         HWPartition[] partitions = diskStore.getPartitions();
134         if (partitions.length > 0) {
135             // If a partition exists on this drive, the major property
136             // corresponds to the disk index, so use it.
137             index = Integer.toString(partitions[0].getMajor());
138         } else if (diskStore.getName().startsWith(PHYSICALDRIVE_PREFIX)) {
139             // If no partition exists, Windows reliably uses a name to match the
140             // disk index. That said, the skeptical person might wonder why a
141             // disk has read/write statistics without a partition, and wonder
142             // why this branch is even relevant as an option. The author of this
143             // comment does not have an answer for this valid question.
144             index = diskStore.getName().substring(PHYSICALDRIVE_PREFIX.length(), diskStore.getName().length());
145         } else {
146             // The author of this comment cannot fathom a circumstance in which
147             // the code reaches this point, but just in case it does, here's the
148             // correct response. If you get this log warning, the circumstances
149             // would be of great interest to the project's maintainers.
150             LOG.warn("Couldn't match index for {}", diskStore.getName());
151             return false;
152         }
153         DiskStats stats = queryReadWriteStats(index);
154         if (stats.readMap.containsKey(index)) {
155             diskStore.setReads(stats.readMap.getOrDefault(index, 0L));
156             diskStore.setReadBytes(stats.readByteMap.getOrDefault(index, 0L));
157             diskStore.setWrites(stats.writeMap.getOrDefault(index, 0L));
158             diskStore.setWriteBytes(stats.writeByteMap.getOrDefault(index, 0L));
159             diskStore.setCurrentQueueLength(stats.queueLengthMap.getOrDefault(index, 0L));
160             diskStore.setTransferTime(stats.timeStamp-stats.idleTimeMap.getOrDefault(index, stats.timeStamp));
161             diskStore.setTimeStamp(stats.timeStamp);
162             return true;
163         } else {
164             return false;
165         }

169     /** {@inheritDoc} */
170     @Override
171     public HWDiskStore[] getDisks() {
172         List<HWDiskStore> result;
173         result = new ArrayList<>();
174         DiskStats stats = queryReadWriteStats(null);
175         PartitionMaps maps = queryPartitionMaps();
176 
177         WmiResult<DiskDriveProperty> vals = wmiQueryHandler.queryWMI(diskDriveQuery);
178 
179         for (int i = 0; i < vals.getResultCount(); i++) {
180             HWDiskStorel#HWDiskStore">HWDiskStore ds = new HWDiskStore();
181             ds.setName(WmiUtil.getString(vals, DiskDriveProperty.NAME, i));
182             ds.setModel(String.format("%s %s", WmiUtil.getString(vals, DiskDriveProperty.MODEL, i),
183                     WmiUtil.getString(vals, DiskDriveProperty.MANUFACTURER, i)).trim());
184             // Most vendors store serial # as a hex string; convert
185             ds.setSerial(ParseUtil.hexStringToString(WmiUtil.getString(vals, DiskDriveProperty.SERIALNUMBER, i)));
186             String index = Integer.toString(WmiUtil.getUint32(vals, DiskDriveProperty.INDEX, i));
187             ds.setReads(stats.readMap.getOrDefault(index, 0L));
188             ds.setReadBytes(stats.readByteMap.getOrDefault(index, 0L));
189             ds.setWrites(stats.writeMap.getOrDefault(index, 0L));
190             ds.setWriteBytes(stats.writeByteMap.getOrDefault(index, 0L));
191             ds.setCurrentQueueLength(stats.queueLengthMap.getOrDefault(index, 0L));
192             ds.setTransferTime(stats.timeStamp-stats.idleTimeMap.getOrDefault(index, stats.timeStamp));
193             ds.setTimeStamp(stats.timeStamp);
194             ds.setSize(WmiUtil.getUint64(vals, DiskDriveProperty.SIZE, i));
195             // Get partitions
196             List<HWPartition> partitions = new ArrayList<>();
197             List<String> partList = maps.driveToPartitionMap.get(ds.getName());
198             if (partList != null && !partList.isEmpty()) {
199                 for (String part : partList) {
200                     if (maps.partitionMap.containsKey(part)) {
201                         partitions.add(maps.partitionMap.get(part));
202                     }
203                 }
204             }
205             ds.setPartitions(partitions.toArray(new HWPartition[0]));
206             // Add to list
207             result.add(ds);
208         }
209         return result.toArray(new HWDiskStore[0]);
210     }

212     /**
213      * Populates the maps for the specified index. If the index is null, populates
214      * all the maps
215      *
216      * @param index
217      *            The index to populate/update maps for
218      */
219     private static DiskStats queryReadWriteStats(String index) {
220         // Create object to hold and return results
221         DiskStats stats = new DiskStats();
222 
223         Map<PhysicalDiskProperty, List<Long>> valueMap = physicalDiskPerfCounters.queryValuesWildcard();
224         stats.timeStamp = System.currentTimeMillis();
225         List<String> instances = physicalDiskPerfCounters.getInstancesFromLastQuery();
226         List<Long> readList = valueMap.get(PhysicalDiskProperty.DISKREADSPERSEC);
227         List<Long> readByteList = valueMap.get(PhysicalDiskProperty.DISKREADBYTESPERSEC);
228         List<Long> writeList = valueMap.get(PhysicalDiskProperty.DISKWRITESPERSEC);
229         List<Long> writeByteList = valueMap.get(PhysicalDiskProperty.DISKWRITEBYTESPERSEC);
230         List<Long> queueLengthList = valueMap.get(PhysicalDiskProperty.CURRENTDISKQUEUELENGTH);
231         List<Long> idleTimeList = valueMap.get(PhysicalDiskProperty.PERCENTIDLETIME);
232 
233         if (instances.isEmpty() || readList == null || readByteList == null || writeList == null
234                 || writeByteList == null || queueLengthList == null || idleTimeList == null) {
235             return stats;
236         }
237         for (int i = 0; i < instances.size(); i++) {
238             String name = getIndexFromName(instances.get(i));
239             // If index arg passed, only update passed arg
240             if (index != null && !index.equals(name)) {
241                 continue;
242             }
243             stats.readMap.put(name, readList.get(i));
244             stats.readByteMap.put(name, readByteList.get(i));
245             stats.writeMap.put(name, writeList.get(i));
246             stats.writeByteMap.put(name, writeByteList.get(i));
247             stats.queueLengthMap.put(name, queueLengthList.get(i));
248             stats.idleTimeMap.put(name, idleTimeList.get(i) / 10_000L);
249         }
250         return stats;
251     }

253     private PartitionMaps queryPartitionMaps() {
254         // Create object to hold and return results
255         PartitionMaps maps = new PartitionMaps();
256 
257         // For Regexp matching DeviceIDs
258         Matcher mAnt;
259         Matcher mDep;
260 
261         // Map drives to partitions
262         WmiResult<DriveToPartitionProperty> drivePartitionMap = wmiQueryHandler.queryWMI(driveToPartitionQuery);
263         for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
264             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.ANTECEDENT, i));
265             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.DEPENDENT, i));
266             if (mAnt.matches() && mDep.matches()) {
267                 maps.driveToPartitionMap
268                         .computeIfAbsent(mAnt.group(1).replaceAll("\\\\\\\\", "\\\\"), x -> new ArrayList<>())
269                         .add(mDep.group(1));
270             }
271         }
272 
273         // Map partitions to logical disks
274         WmiResult<DriveToPartitionProperty> diskPartitionMap = wmiQueryHandler.queryWMI(diskToParitionQuery);
275         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
276             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DriveToPartitionProperty.ANTECEDENT, i));
277             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DriveToPartitionProperty.DEPENDENT, i));
278             if (mAnt.matches() && mDep.matches()) {
279                 maps.partitionToLogicalDriveMap.put(mAnt.group(1), mDep.group(1) + "\\");
280             }
281         }
282 
283         // Next, get all partitions and create objects
284         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = wmiQueryHandler.queryWMI(partitionQuery);
285         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
286             String deviceID = WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DEVICEID, i);
287             String logicalDrive = maps.partitionToLogicalDriveMap.getOrDefault(deviceID, "");
288             String uuid = "";
289             if (!logicalDrive.isEmpty()) {
290                 // Get matching volume for UUID
291                 char[] volumeChr = new char[BUFSIZE];
292                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive, volumeChr, BUFSIZE);
293                 uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
294             }
295             maps.partitionMap.put(deviceID,
296                     new HWPartition(WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.NAME, i),
297                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.TYPE, i),
298                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DESCRIPTION, i), uuid,
299                             WmiUtil.getUint64(hwPartitionQueryMap, DiskPartitionProperty.SIZE, i),
300                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.DISKINDEX, i),
301                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.INDEX, i), logicalDrive));
302         }
303         return maps;
304     }

306     /**
307      * Parse a drive name like "0 C:" to just the index "0"
308      *
309      * @param s
310      *            A drive name to parse
311      * @return The first space-delimited value
312      */
313     private static String getIndexFromName(String s) {
314         if (s.isEmpty()) {
315             return s;
316         }
317         return s.split("\\s")[0];
318     }
