26  /**
27   * Windows Bluetooth device enumeration via the Bluetooth API (bthprops.cpl).
28   */
29  @Immutable
30  public final class WindowsBluetoothDeviceJNA extends AbstractBluetoothDevice {
31  
32      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
33  
34      private WindowsBluetoothDeviceJNA(String name, String address, String majorDeviceClass, boolean connected,
35              boolean paired, int batteryLevel, String adapterName) {
36          super(name, address, majorDeviceClass, connected, paired, batteryLevel, adapterName);
37      }
38  
39      /**
40       * Gets Bluetooth devices known to the system.
41       *
42       * @return a list of {@link BluetoothDevice} objects
43       */
44      public static List<BluetoothDevice> getBluetoothDevices() {
45          if (!IS_VISTA_OR_GREATER) {
46              return Collections.emptyList();
47          }
48          List<BluetoothDevice> devices = new ArrayList<>();
49          BLUETOOTH_FIND_RADIO_PARAMS radioParams = new BLUETOOTH_FIND_RADIO_PARAMS();
50          PointerByReference phRadio = new PointerByReference();
51  
52          HANDLE hFindRadio = BluetoothApis.INSTANCE.BluetoothFindFirstRadio(radioParams, phRadio);
53          if (hFindRadio == null) {
54              return Collections.emptyList();
55          }
56  
57          try {
58              do {
59                  HANDLE hRadio = new HANDLE(phRadio.getValue());
60                  try {
61                      String adapterName = getRadioName(hRadio);
62                      queryDevicesForRadio(hRadio, adapterName, devices);
63                  } finally {
64                      Kernel32.INSTANCE.CloseHandle(hRadio);
65                  }
66              } while (BluetoothApis.INSTANCE.BluetoothFindNextRadio(hFindRadio, phRadio));
67          } finally {
68              BluetoothApis.INSTANCE.BluetoothFindRadioClose(hFindRadio);
69          }
70  
71          return Collections.unmodifiableList(devices);
72      }
73  
74      private static String getRadioName(HANDLE hRadio) {
75          BLUETOOTH_RADIO_INFO radioInfo = new BLUETOOTH_RADIO_INFO();
76          if (BluetoothApis.INSTANCE.BluetoothGetRadioInfo(hRadio, radioInfo) == 0) {
77              return new String(radioInfo.szName).trim();
78          }
79          return "";
80      }
81  
82      private static void queryDevicesForRadio(HANDLE hRadio, String adapterName, List<BluetoothDevice> devices) {
83          BLUETOOTH_DEVICE_SEARCH_PARAMS searchParams = new BLUETOOTH_DEVICE_SEARCH_PARAMS();
84          searchParams.fReturnAuthenticated = true;
85          searchParams.fReturnRemembered = true;
86          searchParams.fReturnConnected = true;
87          searchParams.fReturnUnknown = false;
88          searchParams.fIssueInquiry = false;
89          searchParams.cTimeoutMultiplier = 0;
90          searchParams.hRadio = hRadio;
91  
92          BLUETOOTH_DEVICE_INFO deviceInfo = new BLUETOOTH_DEVICE_INFO();
93          HANDLE hFind = BluetoothApis.INSTANCE.BluetoothFindFirstDevice(searchParams, deviceInfo);
94          if (hFind == null) {
95              return;
96          }
97  
98          try {
99              do {
100                 devices.add(parseDeviceInfo(deviceInfo, adapterName));
101                 deviceInfo = new BLUETOOTH_DEVICE_INFO();
102             } while (BluetoothApis.INSTANCE.BluetoothFindNextDevice(hFind, deviceInfo));
103         } finally {
104             BluetoothApis.INSTANCE.BluetoothFindDeviceClose(hFind);
105         }
106     }
107 
108     private static WindowsBluetoothDeviceJNA parseDeviceInfo(BLUETOOTH_DEVICE_INFO info, String adapterName) {
109         String name = new String(info.szName).trim();
110         String address = FormatUtil.formatMacAddress(info.Address.getAddress());
111         String majorClass = AbstractBluetoothDevice.parseMajorDeviceClass(info.ulClassofDevice);
112         boolean connected = info.fConnected;
113         boolean paired = info.fRemembered;
114         return new WindowsBluetoothDeviceJNA(name, address, majorClass, connected, paired, -1, adapterName);
115     }
