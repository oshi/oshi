24  package oshi.hardware;
25  
26  /**
27   * The VirtuallMemory class tracks information about the use of a computer's
28   * virtual memory (swap file) which temporarily moves rarely accessed
29   * information to a disk or other storage device.
30   */
31  public interface VirtualMemory {
32  
33      /**
34       * The current size of the paging/swap file(s), in bytes. If the paging/swap
35       * file can be extended, this is a soft limit.
36       *
37       * @return Total swap in bytes.
38       */
39      long getSwapTotal();
40  
41      /**
42       * The current memory committed to the paging/swap file(s), in bytes
43       *
44       * @return Swap used in bytes
45       */
46      long getSwapUsed();
47  
48      /**
49       * Number of pages read from paging/swap file(s) to resolve hard page faults.
50       * (Hard page faults occur when a process requires code or data that is not in
51       * its working set or elsewhere in physical memory, and must be retrieved from
52       * disk.) This property was designed as a primary indicator of the kinds of
53       * faults that cause system-wide delays. It includes pages retrieved to satisfy
54       * faults in the file system cache (usually requested by applications) and in
55       * non-cached mapped memory files.
56       *
57       * @return Pages swapped in
58       */
59      long getSwapPagesIn();
60  
61      /**
62       * Number of pages written to paging/swap file(s) to free up space in physical
63       * memory. Pages are written back to disk only if they are changed in physical
64       * memory, so they are likely to hold data, not code. A high rate of pages
65       * output might indicate a memory shortage. The operating system writes more
66       * pages back to disk to free up space when physical memory is in short supply.
67       *
68       * @return Pages swapped out
69       */
70      long getSwapPagesOut();
71  }
