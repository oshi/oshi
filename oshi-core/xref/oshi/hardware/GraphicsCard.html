5   package oshi.hardware;
6   
7   import oshi.annotation.concurrent.Immutable;
8   
9   /**
10   * Represents a graphics card (GPU) installed in the system.
11   *
12   * <p>
13   * Identity fields ({@link #getName()}, {@link #getDeviceId()}, {@link #getVendor()}, {@link #getVersionInfo()},
14   * {@link #getVRam()}) are static and safe to read at any time.
15   *
16   * <p>
17   * Live metrics (utilization, temperature, power, clocks, fan speed, VRAM used) are accessed through a {@link GpuStats}
18   * session obtained via {@link #createStatsSession()}. For a single snapshot, use try-with-resources:
19   *
20   * <pre>{@code
21   * try (GpuStats stats = card.createStatsSession()) {
22   *     double temp = stats.getTemperature();
23   *     double power = stats.getPowerDraw();
24   * }
25   * }</pre>
26   *
27   * <p>
28   * For repeated polling, hold the session open across iterations to preserve internal delta state. See {@link GpuStats}
29   * for a full polling example and details on which metrics require priming.
30   */
31  @Immutable
32  public interface GraphicsCard {
33  
34      /**
35       * Retrieves the full name of the card.
36       *
37       * @return The name of the card.
38       */
39      String getName();
40  
41      /**
42       * Retrieves the card's Device ID
43       *
44       * @return The Device ID of the card
45       */
46      String getDeviceId();
47  
48      /**
49       * Retrieves the card's manufacturer/vendor
50       *
51       * @return The vendor of the card as human-readable text if possible, or the Vendor ID (VID) otherwise
52       */
53      String getVendor();
54  
55      /**
56       * Retrieves a list of version/revision data from the card. Users may need to further parse this list to identify
57       * specific GPU capabilities.
58       *
59       * @return A comma-delimited list of version/revision data
60       */
61      String getVersionInfo();
62  
63      /**
64       * Retrieves the Video RAM (VRAM) available on the GPU
65       *
66       * @return Total number of bytes.
67       */
68      long getVRam();
69  
70      /**
71       * Opens a new {@link GpuStats} session for sampling dynamic GPU metrics. The caller is responsible for closing the
72       * session when done, preferably via try-with-resources.
73       *
74       * <p>
75       * Never returns null and never throws. Platforms that do not support a native session return a no-op instance whose
76       * metric methods return sentinel values.
77       *
78       * @return a new, open {@link GpuStats} session
79       */
80      GpuStats createStatsSession();
81  }
