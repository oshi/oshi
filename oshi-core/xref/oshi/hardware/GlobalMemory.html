24  package oshi.hardware;
25  
26  import java.util.List;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  
30  /**
31   * The GlobalMemory class tracks information about the use of a computer's
32   * physical memory (RAM) as well as any available virtual memory.
33   */
34  @ThreadSafe
35  public interface GlobalMemory {
36      /**
37       * The amount of actual physical memory, in bytes.
38       *
39       * @return Total number of bytes.
40       */
41      long getTotal();
42  
43      /**
44       * The amount of physical memory currently available, in bytes.
45       *
46       * @return Available number of bytes.
47       */
48      long getAvailable();
49  
50      /**
51       * The number of bytes in a memory page
52       *
53       * @return Page size in bytes.
54       */
55      long getPageSize();
56  
57      /**
58       * Virtual memory, such as a swap file.
59       *
60       * @return A VirtualMemory object.
61       */
62      VirtualMemory getVirtualMemory();
63  
64      /**
65       * Physical memory, such as banks of memory.
66       * <p>
67       * On Linux, requires elevated permissions. On FreeBSD and Solaris, requires
68       * installation of dmidecode.
69       *
70       * @return A list of PhysicalMemory objects.
71       */
72      List<PhysicalMemory> getPhysicalMemory();
73  }
