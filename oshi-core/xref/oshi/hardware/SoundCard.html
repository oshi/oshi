24  package oshi.hardware;
25  
26  import oshi.annotation.concurrent.Immutable;
27  
28  /**
29   * SoundCard interface.
30   */
31  @Immutable
32  public interface SoundCard {
33  
34      /**
35       * Retrieves the driver version currently in use in machine
36       *
37       * @return The current and complete name of the driver version
38       */
39      String getDriverVersion();
40  
41      /**
42       * Retrieves the full name of the card.
43       *
44       * @return The name of the card.
45       */
46      String getName();
47  
48      /**
49       * Retrieves the codec of the Sound card
50       *
51       * @return The name of the codec of the sound card
52       */
53      String getCodec();
54  
55  }
