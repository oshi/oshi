24  package oshi.hardware;
25  
26  import java.time.LocalDate;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  
30  /**
31   * The Power Source is one or more batteries with some capacity, and some state
32   * of charge/discharge
33   */
34  @ThreadSafe
35  public interface PowerSource {
36      /**
37       * Units of Battery Capacity
38       */
39      enum CapacityUnits {
40          /**
41           * MilliWattHours (mWh).
42           */
43          MWH,
44  
45          /**
46           * MilliAmpHours (mAh). Should be multiplied by voltage to convert to mWh.
47           */
48          MAH,
49  
50          /**
51           * Relative units. The specific units are not defined. The ratio of current/max
52           * capacity still represents state of charge and the ratio of max/design
53           * capacity still represents state of health.
54           */
55          RELATIVE;
56      }
57  
58      /**
59       * Name of the power source at the Operating System level.
60       *
61       * @return The power source name, as reported by the operating system.
62       */
63      String getName();
64  
65      /**
66       * Name of the power source at the device level.
67       *
68       * @return The power source name, as reported by the device itself.
69       */
70      String getDeviceName();
71  
72      /**
73       * Estimated remaining capacity as a fraction of max capacity.
74       * <p>
75       * This is an estimated/smoothed value which should correspond to the Operating
76       * System's "percent power" display, and may not directly correspond to the
77       * ratio of {@link #getCurrentCapacity()} to {@link #getMaxCapacity()}.
78       *
79       * @return A value between 0.0 (fully drained) and 1.0 (fully charged)
80       */
81      double getRemainingCapacityPercent();
82  
83      /**
84       * Estimated time remaining on the power source, in seconds, as reported by the
85       * operating system.
86       * <p>
87       * This is an estimated/smoothed value which should correspond to the Operating
88       * System's "battery time remaining" display, and will react slowly to changes
89       * in power consumption.
90       *
91       * @return If positive, seconds remaining. If negative, -1.0 (calculating) or
92       *         -2.0 (unlimited)
93       */
94      double getTimeRemainingEstimated();
95  
96      /**
97       * Estimated time remaining on the power source, in seconds, as reported by the
98       * battery. If the battery is charging, this value may represent time remaining
99       * to fully charge the battery.
100      * <p>
101      * Note that this value is not very accurate on some battery systems. The value
102      * may vary widely depending on present power usage, which could be affected by
103      * disk activity and other factors. This value will often be a higher value than
104      * {@link #getTimeRemainingEstimated()}.
105      *
106      * @return Seconds remaining to fully discharge or fully charge the battery.
107      */
108     double getTimeRemainingInstant();
109 
110     /**
111      * Power Usage Rate of the battery, in milliWatts (mW).
112      *
113      * @return If positive, the charge rate. If negative, the discharge rate.
114      */
115     double getPowerUsageRate();
116 
117     /**
118      * Voltage of the battery, in Volts.
119      *
120      * @return the battery voltage, or -1 if unknown.
121      */
122     double getVoltage();
123 
124     /**
125      * Amperage of the battery, in milliAmperes (mA).
126      *
127      * @return the battery amperage. If positive, charging the battery. If negative,
128      *         discharging the battery.
129      */
130     double getAmperage();
131 
132     /**
133      * Reports whether the device is plugged in to an external power source.
134      *
135      * @return {@code true} if plugged in, {@code false} otherwise.
136      */
137     boolean isPowerOnLine();
138 
139     /**
140      * Reports whether the battery is charging.
141      *
142      * @return {@code true} if the battery is charging, {@code false} otherwise.
143      */
144     boolean isCharging();
145 
146     /**
147      * Reports whether the battery is discharging.
148      *
149      * @return {@code true} if the battery is discharging, {@code false} otherwise.
150      */
151     boolean isDischarging();
152 
153     /**
154      * Reports =the units of {@link #getCurrentCapacity()},
155      * {@link #getMaxCapacity()}, and {@link #getDesignCapacity()}
156      *
157      * @return The units of battery capacity.
158      */
159     CapacityUnits getCapacityUnits();
160 
161     /**
162      * The current (remaining) capacity of the battery.
163      *
164      * @return The current capacity. Units are defined by
165      *         {@link #getCapacityUnits()}.
166      */
167     int getCurrentCapacity();
168 
169     /**
170      * The maximum capacity of the battery. When compared to design capacity,
171      * permits a measure of battery state of health. It is possible for max capacity
172      * to exceed design capacity.
173      *
174      * @return The maximum capacity. Units are defined by
175      *         {@link #getCapacityUnits()}.
176      */
177     int getMaxCapacity();
178 
179     /**
180      * The design (original) capacity of the battery. When compared to maximum
181      * capacity, permits a measure of battery state of health. It is possible for
182      * max capacity to exceed design capacity.
183      *
184      * @return The design capacity. Units are defined by
185      *         {@link #getCapacityUnits()}.
186      */
187     int getDesignCapacity();
188 
189     /**
190      * The cycle count of the battery, if known.
191      *
192      * @return The cycle count of the battery, or -1 if unknown.
193      */
194     int getCycleCount();
195 
196     /**
197      * The battery chemistry (e.g., Lithium Ion).
198      *
199      * @return the battery chemistry.
200      */
201     String getChemistry();
202 
203     /**
204      * The battery's date of manufacture.
205      * <p>
206      * Some battery manufacturers encode the manufacture date in the serial number.
207      * Parsing this value is operating system and battery manufacturer dependent,
208      * and is left to the user.
209      *
210      * @return the manufacture date, if available. May be {@code null}.
211      */
212     LocalDate getManufactureDate();
213 
214     /**
215      * The name of the battery's manufacturer.
216      *
217      * @return the manufacturer name.
218      */
219     String getManufacturer();
220 
221     /**
222      * The battery's serial number.
223      * <p>
224      * Some battery manufacturers encode the manufacture date in the serial number.
225      * Parsing this value is operating system and battery manufacturer dependent,
226      * and is left to the user.
227      *
228      * @return the serial number.
229      */
230     String getSerialNumber();
231 
232     /**
233      * The battery's temperature, in degrees Celsius.
234      *
235      * @return the battery's temperature, or 0 if uknown.
236      */
237     double getTemperature();
238 
239     /**
240      * Updates statistics on this battery.
241      *
242      * @return {@code true} if the update was successful. If {@code false} the
243      *         battery statistics are unchanged.
244      */
245     boolean updateAttributes();
246 }
