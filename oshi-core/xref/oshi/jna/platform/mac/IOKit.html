70      /**
71       * Holds the return value of SMC version query.
72       */
73      class SMCKeyDataVers extends Structure {
74          public byte major;
75          public byte minor;
76          public byte build;
77          public byte[] reserved = new byte[1];
78          public short release;
79  
80          @Override
81          protected List<String> getFieldOrder() {
82              return Arrays.asList(new String[] { "major", "minor", "build", "reserved", "release" });
83          }
84      }
85  
86      /**
87       * Holds the return value of SMC pLimit query.
88       */
89      class SMCKeyDataPLimitData extends Structure {
90          public short version;
91          public short length;
92          public int cpuPLimit;
93          public int gpuPLimit;
94          public int memPLimit;
95  
96          @Override
97          protected List<String> getFieldOrder() {
98              return Arrays.asList(new String[] { "version", "length", "cpuPLimit", "gpuPLimit", "memPLimit" });
99          }
100     }
101 
102     /**
103      * Holds the return value of SMC KeyInfo query.
104      */
105     class SMCKeyDataKeyInfo extends Structure {
106         public int dataSize;
107         public int dataType;
108         public byte dataAttributes;
109 
110         @Override
111         protected List<String> getFieldOrder() {
112             return Arrays.asList(new String[] { "dataSize", "dataType", "dataAttributes" });
113         }
114     }
115 
116     /**
117      * Holds the return value of SMC query.
118      */
119     class SMCKeyData extends Structure {
120         public int key;
121         public SMCKeyDataVers vers;
122         public SMCKeyDataPLimitData pLimitData;
123         public SMCKeyDataKeyInfo keyInfo;
124         public byte result;
125         public byte status;
126         public byte data8;
127         public int data32;
128         public byte[] bytes = new byte[32];
129 
130         @Override
131         protected List<String> getFieldOrder() {
132             return Arrays.asList(new String[] { "key", "vers", "pLimitData", "keyInfo", "result", "status", "data8",
133                     "data32", "bytes" });
134         }
135     }
136 
137     /**
138      * Holds an SMC value
139      */
140     class SMCVal extends Structure {
141         public byte[] key = new byte[5];
142         public int dataSize;
143         public byte[] dataType = new byte[5];
144         public byte[] bytes = new byte[32];
145 
146         @Override
147         protected List<String> getFieldOrder() {
148             return Arrays.asList(new String[] { "key", "dataSize", "dataType", "bytes" });
149         }
150     }
