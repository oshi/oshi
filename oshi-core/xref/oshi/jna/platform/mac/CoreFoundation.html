34  /**
35   * CoreFoundation framework for power supply stats. This class should be
36   * considered non-API as it may be removed if/when its code is incorporated into
37   * the JNA project.
38   *
39   * @author widdis[at]gmail[dot]com
40   */
41  public interface CoreFoundation extends Library {
42      CoreFoundation INSTANCE = Native.load("CoreFoundation", CoreFoundation.class);
43  
44      int UTF_8 = 0x08000100;
45  
46      int CFArrayGetCount(CFArrayRef array);
47  
48      /*
49       * Decorator class types for PointerType to better match ported code
50       */
51      class CFTypeRef extends PointerType {
52      }
53  
54      class CFNumberRef extends PointerType {
55      }
56  
57      class CFBooleanRef extends PointerType {
58      }
59  
60      class CFArrayRef extends PointerType {
61      }
62  
63      class CFDictionaryRef extends PointerType {
64      }
65  
66      class CFMutableDictionaryRef extends CFDictionaryRef {
67      }
68  
69      class CFAllocatorRef extends PointerType {
70      }
71  
72      class CFStringRef extends PointerType {
73          /**
74           * Creates a new CFString from the given Java string.
75           *
76           * @param s
77           *            A string
78           * @return A reference to a CFString representing s
79           */
80          public static CFStringRef toCFString(String s) {
81              final char[] chars = s.toCharArray();
82              int length = chars.length;
83              return CoreFoundation.INSTANCE.CFStringCreateWithCharacters(null, chars, new NativeLong(length));
84          }
85      }
86  
87      /*
88       * References are owned if created by functions including "Create" or "Copy"
89       * and must be released with CFRelease to avoid leaking references
90       */
91  
92      CFStringRef CFStringCreateWithCharacters(Object object, char[] chars, NativeLong length);
93  
94      CFMutableDictionaryRef CFDictionaryCreateMutable(CFAllocatorRef allocator, int capacity, Pointer keyCallBacks,
95              Pointer valueCallBacks);
96  
97      void CFRelease(PointerType blob);
98  
99      /*
100      * References are not owned if created by functions using "Get". Use
101      * CFRetain if object retention is required, and then CFRelease later. Do
102      * not use CFRelease if you do not own.
103      */
104 
105     void CFDictionarySetValue(CFMutableDictionaryRef dict, PointerType key, PointerType value);
106 
107     Pointer CFDictionaryGetValue(CFDictionaryRef dictionary, CFStringRef key);
108 
109     boolean CFDictionaryGetValueIfPresent(CFDictionaryRef dictionary, CFStringRef key, PointerType value);
110 
111     boolean CFStringGetCString(Pointer cfString, Pointer bufferToFill, long maxSize, int encoding);
112 
113     boolean CFBooleanGetValue(Pointer booleanRef);
