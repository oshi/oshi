81      class CFStringRef extends PointerType {
82          /**
83           * Creates a new CFString from the given Java string.
84           *
85           * @param s
86           *            A string
87           * @return A reference to a CFString representing s
88           */
89          public static CFStringRef toCFString(String s) {
90              final char[] chars = s.toCharArray();
91              int length = chars.length;
92              return CoreFoundation.INSTANCE.CFStringCreateWithCharacters(null, chars, new NativeLong(length));
93          }
94      }
