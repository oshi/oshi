24  package oshi.jna.platform.mac;
25  
26  import com.sun.jna.ptr.IntByReference;
27  import com.sun.jna.ptr.PointerByReference;
28  
29  import oshi.jna.platform.mac.CoreFoundation.CFDictionaryRef;
30  import oshi.jna.platform.mac.CoreFoundation.CFMutableDictionaryRef;
31  import oshi.jna.platform.mac.IOKit.IOIterator;
32  import oshi.jna.platform.mac.IOKit.IORegistryEntry;
33  import oshi.jna.platform.mac.IOKit.IOService;
34  
35  /**
36   * Provides utilities for IOKit.
37   */
38  public class IOKitUtil {
39      private static final IOKit IO = IOKit.INSTANCE;
40      private static final oshi.jna.platform.mac.SystemB SYS = oshi.jna.platform.mac.SystemB.INSTANCE;
41  
42      private IOKitUtil() {
43      }
44  
45      /**
46       * Gets a pointer to the Mach Master Port.
47       *
48       * @return The master port.
49       *         <p>
50       *         Multiple calls to {@link #getMasterPort} will not result in leaking
51       *         ports (each call to {@link IOKit#IOMasterPort} adds another send
52       *         right to the port) but it is considered good programming practice to
53       *         deallocate the port when you are finished with it, using
54       *         {@link oshi.jna.platform.mac.SystemB#mach_port_deallocate}.
55       */
56      public static int getMasterPort() {
57          IntByReference port = new IntByReference();
58          IO.IOMasterPort(0, port);
59          return port.getValue();
60      }
61  
62      /**
63       * Gets the IO Registry root.
64       *
65       * @return a handle to the IORoot. Callers should release when finished, using
66       *         {@link IOKit#IOObjectRelease}.
67       */
68      public static IORegistryEntry getRoot() {
69          int masterPort = getMasterPort();
70          IORegistryEntry root = IO.IORegistryGetRootEntry(masterPort);
71          SYS.mach_port_deallocate(SYS.mach_task_self(), masterPort);
72          return root;
73      }
74  
75      /**
76       * Opens a the first IOService matching a service name.
77       *
78       * @param serviceName
79       *            The service name to match
80       * @return a handle to an IOService if successful, {@code null} if failed.
81       *         Callers should release when finished, using
82       *         {@link IOKit#IOObjectRelease}.
83       */
84      public static IOService getMatchingService(String serviceName) {
85          CFMutableDictionaryRef dict = IO.IOServiceMatching(serviceName);
86          if (dict != null) {
87              return getMatchingService(dict);
88          }
89          return null;
90      }
91  
92      /**
93       * Opens a the first IOService matching a dictionary.
94       *
95       * @param matchingDictionary
96       *            The dictionary to match. This method will consume a reference to
97       *            the dictionary.
98       * @return a handle to an IOService if successful, {@code null} if failed.
99       *         Callers should release when finished, using
100      *         {@link IOKit#IOObjectRelease}.
101      */
102     public static IOService getMatchingService(CFDictionaryRef matchingDictionary) {
103         int masterPort = getMasterPort();
104         IOService service = IO.IOServiceGetMatchingService(masterPort, matchingDictionary);
105         SYS.mach_port_deallocate(SYS.mach_task_self(), masterPort);
106         return service;
107     }
108 
109     /**
110      * Convenience method to get IOService objects matching a service name.
111      *
112      * @param serviceName
113      *            The service name to match
114      * @return a handle to an IOIterator if successful, {@code null} if failed.
115      *         Callers should release when finished, using
116      *         {@link IOKit#IOObjectRelease}.
117      */
118     public static IOIterator getMatchingServices(String serviceName) {
119         CFMutableDictionaryRef dict = IO.IOServiceMatching(serviceName);
120         if (dict != null) {
121             return getMatchingServices(dict);
122         }
123         return null;
124     }
125 
126     /**
127      * Convenience method to get IOService objects matching a dictionary.
128      *
129      * @param matchingDictionary
130      *            The dictionary to match. This method will consume a reference to
131      *            the dictionary.
132      * @return a handle to an IOIterator if successful, {@code null} if failed.
133      *         Callers should release when finished, using
134      *         {@link IOKit#IOObjectRelease}.
135      */
136     public static IOIterator getMatchingServices(CFDictionaryRef matchingDictionary) {
137         int masterPort = getMasterPort();
138         PointerByReference serviceIterator = new PointerByReference();
139         int result = IO.IOServiceGetMatchingServices(masterPort, matchingDictionary, serviceIterator);
140         SYS.mach_port_deallocate(SYS.mach_task_self(), masterPort);
141         if (result == 0 && serviceIterator.getValue() != null) {
142             return new IOIterator(serviceIterator.getValue());
143         }
144         return null;
145     }
146 
147     /**
148      * Convenience method to get the IO dictionary matching a bsd name.
149      *
150      * @param bsdName
151      *            The bsd name of the registry entry
152      * @return The dictionary ref if successful, {@code null} if failed. Callers
153      *         should release when finished, using {@link IOKit#IOObjectRelease}.
154      */
155     public static CFMutableDictionaryRef getBSDNameMatchingDict(String bsdName) {
156         int masterPort = getMasterPort();
157         CFMutableDictionaryRef result = IO.IOBSDNameMatching(masterPort, 0, bsdName);
158         SYS.mach_port_deallocate(SYS.mach_task_self(), masterPort);
159         return result;
160     }
161 }
