5   package oshi.jna.platform.mac;
6   
7   import com.sun.jna.Library;
8   import com.sun.jna.Native;
9   import com.sun.jna.Pointer;
10  
11  /**
12   * Objective-C runtime bindings for dispatching messages to AppKit classes via {@code objc_msgSend}. This class should
13   * be considered non-API as it may be removed if/when its code is incorporated into the JNA project.
14   * <p>
15   * On ARM64, {@code objc_msgSend} must be called with a typed signature matching the selector's actual parameter types.
16   * Variadic {@code Object...} dispatch does not work reliably. Each overload below matches a specific argument shape.
17   */
18  public interface ObjCRuntime extends Library {
19  
20      ObjCRuntime INSTANCE = Native.load("objc", ObjCRuntime.class);
21  
22      Pointer objc_getClass(String className);
23  
24      Pointer sel_registerName(String selectorName);
25  
26      Pointer objc_msgSend(Pointer receiver, Pointer selector);
27  
28      Pointer objc_msgSend(Pointer receiver, Pointer selector, long index);
29  }
