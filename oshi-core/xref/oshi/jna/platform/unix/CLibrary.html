26  import com.sun.jna.Library;
27  import com.sun.jna.Pointer;
28  import com.sun.jna.Structure;
29  import com.sun.jna.Structure.FieldOrder;
30  import com.sun.jna.platform.unix.LibCAPI;
31  import com.sun.jna.ptr.PointerByReference;

33  /**
34   * C library with code common to all *nix-based operating systems. This class
35   * should be considered non-API as it may be removed if/when its code is
36   * incorporated into the JNA project.
37   */
38  public interface CLibrary extends LibCAPI, Library {
39  
40      int AI_CANONNAME = 2;
41  
42      int UT_LINESIZE = 32;
43      int UT_NAMESIZE = 32;
44      int UT_HOSTSIZE = 256;
45      int LOGIN_PROCESS = 6; // Session leader of a logged in user.
46      int USER_PROCESS = 7; // Normal process.
47  
48      @FieldOrder({ "sa_family", "sa_data" })
49      class Sockaddr extends Structure {
50          public short sa_family;
51          public byte[] sa_data = new byte[14];
52  
53          public static class ByReference extends Sockaddr implements Structure.ByReference {
54          }
55      }
56  
57      @FieldOrder({ "ai_flags", "ai_family", "ai_socktype", "ai_protocol", "ai_addrlen", "ai_addr", "ai_canonname",
58              "ai_next" })
59      class Addrinfo extends Structure {
60          public int ai_flags;
61          public int ai_family;
62          public int ai_socktype;
63          public int ai_protocol;
64          public int ai_addrlen;
65          public Sockaddr.ByReference ai_addr;
66          public String ai_canonname;
67          public ByReference ai_next;
68  
69          public static class ByReference extends Addrinfo implements Structure.ByReference {
70          }
71  
72          public Addrinfo() {
73          }
74  
75          public Addrinfo(Pointer p) {
76              super(p);
77              read();
78          }
79      }
80  
81      /*
82       * Between macOS and FreeBSD there are multiple versions of some tcp/udp/ip
83       * stats structures. Since we only need a few of the hundreds of fields, we can
84       * improve performance by selectively reading the ints from the appropriate
85       * offsets, which are consistent across the structure. These classes include the
86       * common fields and offsets.
87       */
88  
89      class BsdTcpstat {
90          public int tcps_connattempt; // 0
91          public int tcps_accepts; // 4
92          public int tcps_drops; // 12
93          public int tcps_conndrops; // 16
94          public int tcps_sndpack; // 64
95          public int tcps_sndrexmitpack; // 72
96          public int tcps_rcvpack; // 104
97          public int tcps_rcvbadsum; // 112
98          public int tcps_rcvbadoff; // 116
99          public int tcps_rcvmemdrop; // 120
100         public int tcps_rcvshort; // 124
101     }
102 
103     class BsdUdpstat {
104         public int udps_ipackets; // 0
105         public int udps_hdrops; // 4
106         public int udps_badsum; // 8
107         public int udps_badlen; // 12
108         public int udps_opackets; // 36
109         public int udps_noportmcast; // 48
110         public int udps_rcv6_swcsum; // 64
111         public int udps_snd6_swcsum; // 89
112     }
113 
114     class BsdIpstat {
115         public int ips_total; // 0
116         public int ips_badsum; // 4
117         public int ips_tooshort; // 8
118         public int ips_toosmall; // 12
119         public int ips_badhlen; // 16
120         public int ips_badlen; // 20
121         public int ips_delivered; // 56
122     }
123 
124     class BsdIp6stat {
125         public long ip6s_total; // 0
126         public long ip6s_localout; // 88
127     }
128 
129     /**
130      * Returns the process ID of the calling process. The ID is guaranteed to be
131      * unique and is useful for constructing temporary file names.
132      *
133      * @return the process ID of the calling process.
134      */
135     int getpid();
136 
137     /**
138      * Given node and service, which identify an Internet host and a service,
139      * getaddrinfo() returns one or more addrinfo structures, each of which contains
140      * an Internet address that can be specified in a call to bind(2) or connect(2).
141      *
142      * @param node
143      *            a numerical network address or a network hostname, whose network
144      *            addresses are looked up and resolved.
145      * @param service
146      *            sets the port in each returned address structure.
147      * @param hints
148      *            specifies criteria for selecting the socket address structures
149      *            returned in the list pointed to by res.
150      * @param res
151      *            returned address structure
152      * @return 0 on success; sets errno on failure
153      */
154     int getaddrinfo(String node, String service, Addrinfo hints, PointerByReference res);
155 
156     /**
157      * Frees the memory that was allocated for the dynamically allocated linked list
158      * res.
159      *
160      * @param res
161      *            Pointer to linked list returned by getaddrinfo
162      */
163     void freeaddrinfo(Pointer res);
164 
165     /**
166      * Translates getaddrinfo error codes to a human readable string, suitable for
167      * error reporting.
168      *
169      * @param e
170      *            Error code from getaddrinfo
171      * @return A human-readable version of the error code
172      */
173     String gai_strerror(int e);
174 
175     /**
176      * Rewinds the file pointer to the beginning of the utmp file. It is generally a
177      * good idea to call it before any of the other functions.
178      */
179     void setutxent();
180 
181     /**
182      * Closes the utmp file. It should be called when the user code is done
183      * accessing the file with the other functions.
184      */
185     void endutxent();
186 }
