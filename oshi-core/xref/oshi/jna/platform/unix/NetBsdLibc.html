5   package oshi.jna.platform.unix;
6   
7   import com.sun.jna.Native;
8   import com.sun.jna.Structure;
9   import com.sun.jna.Structure.FieldOrder;
10  
11  /**
12   * C library. This class should be considered non-API as it may be removed if/when its code is incorporated into the JNA
13   * project.
14   */
15  public interface NetBsdLibc extends CLibrary {
16      NetBsdLibc INSTANCE = BsdLibcLoader.loadLibc(NetBsdLibc.class);
17  
18      int CTL_KERN = 1; // "high kernel": proc, limits
19      int CTL_VM = 2; // virtual memory
20      int CTL_HW = 6; // generic cpu/io
21      int CTL_MACHDEP = 7; // machine dependent
22  
23      int KERN_OSTYPE = 1; // string: system version
24      int KERN_OSRELEASE = 2; // string: system release
25      int KERN_OSREV = 3; // int: system revision
26      int KERN_VERSION = 4; // string: compile time info
27      int KERN_MAXPROC = 6; // int: max processes
28      int KERN_ARGMAX = 8; // int: max arguments to exec
29      int KERN_CP_TIME = 51; // struct: cp_time
30  
31      int VM_UVMEXP2 = 5; // struct uvmexp_sysctl
32  
33      int HW_MACHINE = 1; // string: machine class
34      int HW_MODEL = 2; // string: specific machine model
35      int HW_NCPU = 3; // int: number of cpus
36      int HW_PAGESIZE = 7; // int: software page size
37      int HW_MACHINE_ARCH = 11; // string: machine architecture
38      int HW_NCPUONLINE = 16; // number of cpus being used
39  
40      /*
41       * CPU state indices
42       */
43      int CPUSTATES = 5;
44      int CP_USER = 0;
45      int CP_NICE = 1;
46      int CP_SYS = 2;
47      int CP_INTR = 3;
48      int CP_IDLE = 4;
49  
50      int UINT64_SIZE = Native.getNativeSize(long.class);
51      int INT_SIZE = Native.getNativeSize(int.class);
52  
53      /**
54       * Return type for BSD sysctl kern.boottime
55       */
56      @FieldOrder({ "tv_sec", "tv_usec" })
57      class Timeval extends Structure {
58          public long tv_sec; // seconds
59          public long tv_usec; // microseconds
60      }
61  
62      /**
63       * NetBSD's {@code RLIMIT_NOFILE} from {@code <sys/resource.h>} is 8, not 7 like Linux. JNA's inherited
64       * {@code Resource.RLIMIT_NOFILE} from the platform module is the Linux value, so calling
65       * {@code getrlimit(Resource.RLIMIT_NOFILE, …)} on NetBSD would query the wrong resource.
66       */
67      int RLIMIT_NOFILE = 8;
68  
69      /**
70       * Returns the LWP (lightweight process/thread) ID of the calling thread.
71       *
72       * @return the LWP ID of the calling thread.
73       */
74      int _lwp_self();
75  }
