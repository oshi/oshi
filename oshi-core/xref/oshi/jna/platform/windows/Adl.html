15  /**
16   * JNA bindings for the AMD Display Library (ADL) on Windows. This class should be considered non-API as it may be
17   * removed if/when its code is incorporated into the JNA project.
18   */
19  public interface Adl {
20  
21      int ADL_OK = 0;
22      int ADL_OVERDRIVE_VERSION_N = 8;
23      int ADL_FAN_SPEED_MODE_PERCENT = 1;
24      int ADL_OVERDRIVE_TEMPERATURE_EDGE = 1;
25  
26      /** ADL malloc callback: allocates memory and returns a pointer. Uses __stdcall per ADL SDK. */
27      interface AdlMallocCallback extends StdCallLibrary.StdCallCallback {
28          Pointer invoke(int size);
29      }
30  
31      interface AdlLibrary extends Library {
32          int ADL2_Main_Control_Create(AdlMallocCallback callback, int iEnumConnectedAdapters,
33                  PointerByReference context);
34  
35          int ADL2_Main_Control_Destroy(Pointer context);
36  
37          int ADL2_Adapter_NumberOfAdapters_Get(Pointer context, IntByReference numAdapters);
38  
39          int ADL2_Adapter_AdapterInfo_Get(Pointer context, AdapterInfo[] info, int inputSize);
40  
41          int ADL2_Overdrive_Caps(Pointer context, int iAdapterIndex, IntByReference iSupported, IntByReference iEnabled,
42                  IntByReference iVersion);
43  
44          int ADL2_OverdriveN_Temperature_Get(Pointer context, int iAdapterIndex, int iTemperatureType,
45                  IntByReference iTemperature);
46  
47          int ADL2_OverdriveN_PerformanceStatus_Get(Pointer context, int iAdapterIndex,
48                  ADLODNPerformanceStatus perfStatus);
49  
50          int ADL2_OverdriveN_FanControl_Get(Pointer context, int iAdapterIndex, ADLODNFanControl fanControl);
51  
52          int ADL2_Overdrive6_CurrentPower_Get(Pointer context, int iAdapterIndex, int iPowerType,
53                  IntByReference lpCurrentValue);
54      }
55  
56      @FieldOrder({ "iSize", "iAdapterIndex", "strUDID", "iBusNumber", "iDeviceNumber", "iFunctionNumber", "iVendorID",
57              "strAdapterName", "strDisplayName", "iPresent", "iExist", "strDriverPath", "strDriverPathExt",
58              "strPNPString", "iOSDisplayIndex" })
59      class AdapterInfo extends Structure {
60          public int iSize;
61          public int iAdapterIndex;
62          public byte[] strUDID = new byte[256];
63          public int iBusNumber;
64          public int iDeviceNumber;
65          public int iFunctionNumber;
66          public int iVendorID;
67          public byte[] strAdapterName = new byte[256];
68          public byte[] strDisplayName = new byte[256];
69          public int iPresent;
70          public int iExist;
71          public byte[] strDriverPath = new byte[256];
72          public byte[] strDriverPathExt = new byte[256];
73          public byte[] strPNPString = new byte[256];
74          public int iOSDisplayIndex;
75      }
76  
77      @FieldOrder({ "iCoreClock", "iMemoryClock", "iDCEFClock", "iGFXClock", "iUVDClock", "iVCEClock",
78              "iGPUActivityPercent", "iCurrentCorePerformanceLevel", "iCurrentMemoryPerformanceLevel",
79              "iCurrentDCEFPerformanceLevel", "iCurrentGFXPerformanceLevel", "iUVDPerformanceLevel",
80              "iVCEPerformanceLevel", "iCurrentBusSpeed", "iCurrentBusLanes", "iMaximumBusLanes", "iVDDC", "iVDDCI" })
81      class ADLODNPerformanceStatus extends Structure {
82          public int iCoreClock;
83          public int iMemoryClock;
84          public int iDCEFClock;
85          public int iGFXClock;
86          public int iUVDClock;
87          public int iVCEClock;
88          public int iGPUActivityPercent;
89          public int iCurrentCorePerformanceLevel;
90          public int iCurrentMemoryPerformanceLevel;
91          public int iCurrentDCEFPerformanceLevel;
92          public int iCurrentGFXPerformanceLevel;
93          public int iUVDPerformanceLevel;
94          public int iVCEPerformanceLevel;
95          public int iCurrentBusSpeed;
96          public int iCurrentBusLanes;
97          public int iMaximumBusLanes;
98          public int iVDDC;
99          public int iVDDCI;
100     }

103             "iTargetTemperature", "iMinPerformanceClock", "iMinFanLimit" })
104     class ADLODNFanControl extends Structure {
105         public int iMode;
106         public int iFanControlMode;
107         public int iCurrentFanSpeedMode;
108         public int iCurrentFanSpeed;
109         public int iTargetFanSpeed;
110         public int iTargetTemperature;
111         public int iMinPerformanceClock;
112         public int iMinFanLimit;
113     }
