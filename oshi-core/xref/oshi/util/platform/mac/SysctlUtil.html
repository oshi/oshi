19  package oshi.util.platform.mac;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import com.sun.jna.Memory;
25  import com.sun.jna.Native;
26  import com.sun.jna.Pointer;
27  import com.sun.jna.Structure;
28  import com.sun.jna.ptr.IntByReference;
29  
30  import oshi.jna.platform.mac.SystemB;
31  
32  /**
33   * Provides access to sysctl calls on OS X
34   *
35   * @author widdis[at]gmail[dot]com
36   */
37  public class SysctlUtil {
38      private static final Logger LOG = LoggerFactory.getLogger(SysctlUtil.class);
39  
40      private static final String SYSCTL_FAIL = "Failed syctl call: {}, Error code: {}";
41  
42      private SysctlUtil() {
43      }
44  
45      /**
46       * Executes a sysctl call with an int result
47       *
48       * @param name
49       *            name of the sysctl
50       * @param def
51       *            default int value
52       * @return The int result of the call if successful; the default otherwise
53       */
54      public static int sysctl(String name, int def) {
55          IntByReference size = new IntByReference(SystemB.INT_SIZE);
56          Pointer p = new Memory(size.getValue());
57          if (0 != SystemB.INSTANCE.sysctlbyname(name, p, size, null, 0)) {
58              LOG.error("Failed sysctl call: {}, Error code: {}", name, Native.getLastError());
59              return def;
60          }
61          return p.getInt(0);
62      }
63  
64      /**
65       * Executes a sysctl call with a long result
66       *
67       * @param name
68       *            name of the sysctl
69       * @param def
70       *            default long value
71       * @return The long result of the call if successful; the default otherwise
72       */
73      public static long sysctl(String name, long def) {
74          IntByReference size = new IntByReference(SystemB.UINT64_SIZE);
75          Pointer p = new Memory(size.getValue());
76          if (0 != SystemB.INSTANCE.sysctlbyname(name, p, size, null, 0)) {
77              LOG.error(SYSCTL_FAIL, name, Native.getLastError());
78              return def;
79          }
80          return p.getLong(0);
81      }
82  
83      /**
84       * Executes a sysctl call with a String result
85       *
86       * @param name
87       *            name of the sysctl
88       * @param def
89       *            default String value
90       * @return The String result of the call if successful; the default
91       *         otherwise
92       */
93      public static String sysctl(String name, String def) {
94          // Call first time with null pointer to get value of size
95          IntByReference size = new IntByReference();
96          if (0 != SystemB.INSTANCE.sysctlbyname(name, null, size, null, 0)) {
97              LOG.error(SYSCTL_FAIL, name, Native.getLastError());
98              return def;
99          }
100         // Add 1 to size for null terminated string
101         Pointer p = new Memory(size.getValue() + 1);
102         if (0 != SystemB.INSTANCE.sysctlbyname(name, p, size, null, 0)) {
103             LOG.error(SYSCTL_FAIL, name, Native.getLastError());
104             return def;
105         }
106         return p.getString(0);
107     }
108 
109     /**
110      * Executes a sysctl call with a Structure result
111      *
112      * @param name
113      *            name of the sysctl
114      * @param struct
115      *            structure for the result
116      * @return True if structure is successfuly populated, false otherwise
117      */
118     public static boolean sysctl(String name, Structure struct) {
119         if (0 != SystemB.INSTANCE.sysctlbyname(name, struct.getPointer(), new IntByReference(struct.size()), null, 0)) {
120             LOG.error(SYSCTL_FAIL, name, Native.getLastError());
121             return false;
122         }
123         struct.read();
124         return true;
125     }
126 }
