42  public class KstatUtil {
43      private static final Logger LOG = LoggerFactory.getLogger(KstatUtil.class);

45      // All active use of this variable must be synchronized
46      private static final KstatCtl kc = LibKstat.INSTANCE.kstat_open();
47  
48      static {
49          Runtime.getRuntime().addShutdownHook(new Thread() {
50              @Override
51              public void run() {
52                  LibKstat.INSTANCE.kstat_close(kc);
53              }
54          });
55      }

57      private KstatUtil() {
58      }

60      /**
61       * Convenience method for kstat_data_lookup() with String return values.
62       * Searches the kstat's data section for the record with the specified name.
63       * This operation is valid only for kstat types which have named data
64       * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
65       * have named data records.
66       *
67       * @param ksp
68       *            The kstat to search
69       * @param name
70       *            The key for the name-value pair, or name of the timer as
71       *            applicable
72       * @return The value as a String.
73       */
74      public static synchronized String kstatDataLookupString(Kstat ksp, String name) {
75          if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
76              throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
77          }
78          Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
79          if (p == null) {
80              LOG.error("Failed lo lookup kstat value for key {}", name);
81              return "";
82          }
83          KstatNamed data = new KstatNamed(p);
84          switch (data.data_type) {
85          case LibKstat.KSTAT_DATA_CHAR:
86              return new String(data.value.charc).trim();
87          case LibKstat.KSTAT_DATA_INT32:
88              return Integer.toString(data.value.i32);
89          case LibKstat.KSTAT_DATA_UINT32:
90              return FormatUtil.toUnsignedString(data.value.ui32);
91          case LibKstat.KSTAT_DATA_INT64:
92              return Long.toString(data.value.i64);
93          case LibKstat.KSTAT_DATA_UINT64:
94              return FormatUtil.toUnsignedString(data.value.ui64);
95          case LibKstat.KSTAT_DATA_STRING:
96              return data.value.str.addr.getString(0);
97          default:
98              LOG.error("Unimplemented kstat data type {}", data.data_type);
99              return "";
100         }
101     }

103     /**
104      * Convenience method for kstat_data_lookup() with numeric return values.
105      * Searches the kstat's data section for the record with the specified name.
106      * This operation is valid only for kstat types which have named data
107      * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
108      * have named data records.
109      *
110      * @param ksp
111      *            The kstat to search
112      * @param name
113      *            The key for the name-value pair, or name of the timer as
114      *            applicable
115      * @return The value as a long. If the data type is a character or string
116      *         type, returns 0 and logs an error.
117      */
118     public static synchronized long kstatDataLookupLong(Kstat ksp, String name) {
119         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
120             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
121         }
122         Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
123         if (p == null) {
124             if (LOG.isErrorEnabled()) {
125                 LOG.error("Failed lo lookup kstat value on {}:{}:{} for key {}", new String(ksp.ks_module).trim(),
126                         ksp.ks_instance, new String(ksp.ks_name).trim(), name);
127             }
128             return 0L;
129         }
130         KstatNamed data = new KstatNamed(p);
131         switch (data.data_type) {
132         case LibKstat.KSTAT_DATA_INT32:
133             return data.value.i32;
134         case LibKstat.KSTAT_DATA_UINT32:
135             return FormatUtil.getUnsignedInt(data.value.ui32);
136         case LibKstat.KSTAT_DATA_INT64:
137             return data.value.i64;
138         case LibKstat.KSTAT_DATA_UINT64:
139             return data.value.ui64;
140         default:
141             LOG.error("Unimplemented or non-numeric kstat data type {}", data.data_type);
142             return 0L;
143         }
144     }

146     /**
147      * Convenience method for kstat_read() which gets data from the kernel for
148      * the kstat pointed to by ksp. ksp.ks_data is automatically allocated (or
149      * reallocated) to be large enough to hold all of the data. ksp.ks_ndata is
150      * set to the number of data fields, ksp.ks_data_size is set to the total
151      * size of the data, and ksp.ks_snaptime is set to the high-resolution time
152      * at which the data snapshot was taken.
153      *
154      * @param ksp
155      *            The kstat from which to retrieve data
156      * @return True if successful; false otherwise
157      */
158     public static synchronized boolean kstatRead(Kstat ksp) {
159         int retry = 0;
160         while (0 > LibKstat.INSTANCE.kstat_read(kc, ksp, null)) {
161             if (LibKstat.EAGAIN != Native.getLastError() || 5 <= ++retry) {
162                 if (LOG.isErrorEnabled()) {
163                     LOG.error("Failed to read kstat {}:{}:{}", new String(ksp.ks_module).trim(), ksp.ks_instance,
164                             new String(ksp.ks_name).trim());
165                 }
166                 return false;
167             }
168             Util.sleep(8 << retry);
169         }
170         return true;
171     }

173     /**
174      * Convenience method for kstat_lookup(). Traverses the kstat chain,
175      * searching for a kstat with the same ks_module, ks_instance, and ks_name
176      * fields; this triplet uniquely identifies a kstat. If ks_module is NULL,
177      * ks_instance is -1, or ks_name is NULL, then those fields will be ignored
178      * in the search.
179      *
180      * @param module
181      *            The module, or null to ignore
182      * @param instance
183      *            The instance, or -1 to ignore
184      * @param name
185      *            The name, or null to ignore
186      * @return The first match of the requested Kstat structure if found, or
187      *         null
188      */
189     public static synchronized Kstat kstatLookup(String module, int instance, String name) {
190         int ret = LibKstat.INSTANCE.kstat_chain_update(kc);
191         if (ret < 0) {
192             LOG.error("Failed to update kstat chain");
193             return null;
194         }
195         return LibKstat.INSTANCE.kstat_lookup(kc, module, instance, name);
196     }

198     /**
199      * Convenience method for kstat_lookup(). Traverses the kstat chain,
200      * searching for all kstats with the same ks_module, ks_instance, and
201      * ks_name fields; this triplet uniquely identifies a kstat. If ks_module is
202      * NULL, ks_instance is -1, or ks_name is NULL, then those fields will be
203      * ignored in the search.
204      *
205      * @param module
206      *            The module, or null to ignore
207      * @param instance
208      *            The instance, or -1 to ignore
209      * @param name
210      *            The name, or null to ignore
211      * @return All matches of the requested Kstat structure if found, or an
212      *         empty list otherwise
213      */
214     public static synchronized List<Kstat> kstatLookupAll(String module, int instance, String name) {
215         List<Kstat> kstats = new ArrayList<>();
216         int ret = LibKstat.INSTANCE.kstat_chain_update(kc);
217         if (ret < 0) {
218             LOG.error("Failed to update kstat chain");
219             return kstats;
220         }
221         for (Kstat ksp = LibKstat.INSTANCE.kstat_lookup(kc, module, instance, name); ksp != null; ksp = ksp.next()) {
222             if ((module == null || module.equals(new String(ksp.ks_module).trim()))
223                     && (instance < 0 || instance == ksp.ks_instance)
224                     && (name == null || name.equals(new String(ksp.ks_name).trim()))) {
225                 kstats.add(ksp);
226             }
227         }
228         return kstats;
229     }
