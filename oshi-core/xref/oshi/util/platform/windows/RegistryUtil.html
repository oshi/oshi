8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;

12  import com.sun.jna.platform.win32.Advapi32Util;
13  import com.sun.jna.platform.win32.Win32Exception;
14  import com.sun.jna.platform.win32.WinReg;
15  import com.sun.jna.platform.win32.WinReg.HKEY;
16  
17  import oshi.driver.common.windows.registry.RegistryValueUtil;
18  
19  /**
20   * Utility for reading values from the Windows Registry.
21   */
22  public final class RegistryUtil {
23  
24      private static final Logger LOG = LoggerFactory.getLogger(RegistryUtil.class);
25  
26      private RegistryUtil() {
27      }
28  
29      /**
30       * Returns a registry value as a Long. (with access flag) Currently supports String and Integer
31       *
32       * @param root       the registry root
33       * @param path       the registry path
34       * @param key        the registry key
35       * @param accessFlag the access flag
36       * @return the registry value as a long
37       */
38      public static long getLongValue(HKEY root, String path, String key, int accessFlag) {
39          Object val = getRegistryValueOrNull(root, path, key, accessFlag);
40          return RegistryValueUtil.registryValueToLong(val);
41      }
42  
43      /**
44       * Gets a registry value or null if not found
45       *
46       * @param root       the registry root
47       * @param path       the registry path
48       * @param key        the registry key
49       * @param accessFlag the access flag
50       * @return the registry value or null
51       */
52      public static Object getRegistryValueOrNull(HKEY root, String path, String key, int accessFlag) {
53          try {
54              // registryGetValues opens the key with KEY_READ | accessFlag, so the WOW64 access flag (e.g.
55              // KEY_WOW64_32KEY/KEY_WOW64_64KEY) is honored when reading the value. Reading the value separately via
56              // registryGetValue(root, path, key) re-opened the key with the default view, ignoring accessFlag.
57              Map<String, Object> values = Advapi32Util.registryGetValues(root, path, accessFlag);
58              // A null key requests the (Default) value, which registryGetValues stores under the empty name.
59              String valueName = key == null ? "" : key;
60              Object value = values.get(valueName);
61              if (value == null) {
62                  // Registry value names are case-insensitive; fall back to a case-insensitive match.
63                  for (Map.Entry<String, Object> entry : values.entrySet()) {
64                      if (entry.getKey().equalsIgnoreCase(valueName)) {
65                          return entry.getValue();
66                      }
67                  }
68              }
69              return value;
70          } catch (Win32Exception e) {
71              LOG.trace("Unable to access {} with flag {}: {}", path, accessFlag, e.getMessage());
72          }
73          return null;
74      }
75  
76      /**
77       * Returns a registry value as a String. (with access flag) Currently supports String, Integer, and Binary
78       *
79       * @param root       the registry root
80       * @param path       the registry path
81       * @param key        the registry key
82       * @param accessFlag the access flag
83       * @return the registry value as a string
84       */
85      public static String getStringValue(WinReg.HKEY root, String path, String key, int accessFlag) {
86          Object val = getRegistryValueOrNull(root, path, key, accessFlag);
87          return RegistryValueUtil.registryValueToString(val);
88      }
89  
90  }
