63      // Cache failed wmi classes
64      private static Set<String> failedWmiClassNames = new HashSet<>();
65      // Not a built in manespace, failed connections are normal and don't need
66      // error logging
67      public static final String OHM_NAMESPACE = "ROOT\\OpenHardwareMonitor";

76       * Private constructor so this class can't be instantiated from the outside.
77       * Also initializes COM and sets up hooks to uninit if necessary.
78       */
79      private WmiUtil() {
80          // Initialize COM
81          initCOM();
82  
83          // Set up hook to uninit on shutdown
84          Runtime.getRuntime().addShutdownHook(new Thread() {
85              @Override
86              public void run() {
87                  unInitCOM();
88              }
89          });
90      }

92      /**
93       * Determine if WMI has the requested namespace. Some namespaces only exist
94       * on newer versions of Windows.
95       *
96       * @param namespace
97       *            The namespace to test
98       * @return true if the namespace exists, false otherwise
99       */
100     public static boolean hasNamespace(String namespace) {
101         if (hasNamespaceCache.contains(namespace)) {
102             return true;
103         } else if (hasNotNamespaceCache.contains(namespace)) {
104             return false;
105         }
106         if (WbemcliUtil.hasNamespace(namespace)) {
107             hasNamespaceCache.add(namespace);
108             return true;
109         }
110         hasNotNamespaceCache.add(namespace);
111         return false;
112     }

114     /**
115      * Query WMI for values, with no timeout.
116      *
117      * @param <T>
118      *            The properties enum
119      * @param query
120      *            A WmiQuery object encapsulating the namespace, class, and
121      *            properties
122      * @return a WmiResult object containing the query results, wrapping an
123      *         EnumMap
124      */
125     public static <T extends Enum<T>> WmiResult<T> queryWMI(WmiQuery<T> query) {
126 
127         WmiResult<T> result = WbemcliUtil.INSTANCE.new WmiResult<>(query.getPropertyEnum());
128         if (failedWmiClassNames.contains(query.getWmiClassName())) {
129             return result;
130         }
131         try {
132             // Initialize COM if not already done. Needed if COM was previously
133             // initialized externally but is no longer initialized.
134             if (!isComInitialized()) {
135                 initCOM();
136             }
137 
138             result = query.execute(wmiTimeout);
139         } catch (COMException e) {
140             // Ignore any exceptions with OpenHardwareMonitor
141             if (!OHM_NAMESPACE.equals(query.getNameSpace())) {
142                 final int hresult = e.getHresult() == null ? -1 : e.getHresult().intValue();
143                 switch (hresult) {
144                 case Wbemcli.WBEM_E_INVALID_NAMESPACE:
145                     LOG.warn("COM exception: Invalid Namespace {}", query.getNameSpace());
146                     break;
147                 case Wbemcli.WBEM_E_INVALID_CLASS:
148                     LOG.warn("COM exception: Invalid Class {}", query.getWmiClassName());
149                     break;
150                 case Wbemcli.WBEM_E_INVALID_QUERY:
151                     LOG.warn("COM exception: Invalid Query: {}", queryToString(query));
152                     break;
153                 default:
154                     LOG.warn(
155                             "COM exception querying {}, which might not be on your system. Will not attempt to query it again. Error was: {}:",
156                             query.getWmiClassName(), e.getMessage());
157                 }
158                 failedWmiClassNames.add(query.getWmiClassName());
159             }
160         } catch (TimeoutException e) {
161             LOG.error("WMI query timed out after {} ms: {}", wmiTimeout, queryToString(query));
162         }
163         return result;
164     }

166     /**
167      * Translate a WmiQuery to the actual query string
168      * 
169      * @param <T>
170      *            The properties enum
171      * @param query
172      *            The WmiQuery object
173      * @return The string that is queried in WMI
174      */
175     public static <T extends Enum<T>> String queryToString(WmiQuery<T> query) {
176         T[] props = query.getPropertyEnum().getEnumConstants();
177         StringBuilder sb = new StringBuilder("SELECT ");
178         sb.append(props[0].name());
179         for (int i = 1; i < props.length; i++) {
180             sb.append(',').append(props[i].name());
181         }
182         sb.append(" FROM ").append(query.getWmiClassName());
183         return sb.toString();
184     }

186     /**
187      * Gets a String value from a WmiResult
188      *
189      * @param <T>
190      *            The enum type containing the property keys
191      * @param result
192      *            The WmiResult from which to fetch the value
193      * @param property
194      *            The property (column) to fetch
195      * @param index
196      *            The index (row) to fetch
197      * @return The stored value if non-null, an empty-string otherwise
198      */
199     public static <T extends Enum<T>> String getString(WmiResult<T> result, T property, int index) {
200         if (result.getCIMType(property) == Wbemcli.CIM_STRING) {
201             return getStr(result, property, index);
202         }
203         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "String",
204                 result.getCIMType(property), result.getVtType(property)));
205     }

208      * Gets a Date value from a WmiResult as a String
209      *
210      * @param <T>
211      *            The enum type containing the property keys
212      * @param result
213      *            The WmiResult from which to fetch the value
214      * @param property
215      *            The property (column) to fetch
216      * @param index
217      *            The index (row) to fetch
218      * @return The stored value if non-null, an empty-string otherwise
219      */
220     public static <T extends Enum<T>> String getDateString(WmiResult<T> result, T property, int index) {
221         if (result.getCIMType(property) == Wbemcli.CIM_DATETIME) {
222             String date = getStr(result, property, index);
223             return date.substring(0, 4) + '-' + date.substring(4, 6) + '-' + date.substring(6, 8);
224         }
225         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "DateTime",
226                 result.getCIMType(property), result.getVtType(property)));
227     }

250     private static <T extends Enum<T>> String getStr(WmiResult<T> result, T property, int index) {
251         Object o = result.getValue(property, index);
252         if (o == null) {
253             return "";
254         } else if (result.getVtType(property) == Variant.VT_BSTR) {
255             return (String) o;
256         }
257         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "String-mapped",
258                 result.getCIMType(property), result.getVtType(property)));
259     }

261     /**
262      * Gets a Uint64 value from a WmiResult (parsing the String). Note that
263      * while the CIM type is unsigned, the return type is signed and the parsing
264      * will exclude any return values above Long.MAX_VALUE.
265      *
266      * @param <T>
267      *            The enum type containing the property keys
268      * @param result
269      *            The WmiResult from which to fetch the value
270      * @param property
271      *            The property (column) to fetch
272      * @param index
273      *            The index (row) to fetch
274      * @return The stored value if non-null and parseable as a long, 0 otherwise
275      */
276     public static <T extends Enum<T>> long getUint64(WmiResult<T> result, T property, int index) {
277         Object o = result.getValue(property, index);
278         if (o == null) {
279             return 0L;
280         } else if (result.getCIMType(property) == Wbemcli.CIM_UINT64 && result.getVtType(property) == Variant.VT_BSTR) {
281             return ParseUtil.parseLongOrDefault((String) o, 0L);
282         }
283         throw new ClassCastException(String.format(CLASS_CAST_MSG, property.name(), "UINT64",
284                 result.getCIMType(property), result.getVtType(property)));
285     }

413     /**
414      * Initializes COM library and sets security to impersonate the local user
415      */
416     public static void initCOM() {
417         HRESULT hres = null;
418         // Step 1: --------------------------------------------------
419         // Initialize COM. ------------------------------------------
420         if (!isComInitialized()) {
421             hres = Ole32.INSTANCE.CoInitializeEx(null, Ole32.COINIT_MULTITHREADED);
422             switch (hres.intValue()) {
423             // Successful local initialization
424             case COMUtils.S_OK:
425                 comInitialized = true;
426                 break;
427             // COM was already initialized
428             case COMUtils.S_FALSE:
429             case WinError.RPC_E_CHANGED_MODE:
430                 break;
431             // Any other results is an error
432             default:
433                 throw new COMException("Failed to initialize COM library.", hres);
434             }
435         }
436         // Step 2: --------------------------------------------------
437         // Set general COM security levels --------------------------
438         if (!isSecurityInitialized()) {
439             hres = Ole32.INSTANCE.CoInitializeSecurity(null, -1, null, null, Ole32.RPC_C_AUTHN_LEVEL_DEFAULT,
440                     Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE, null);
441             // If security already initialized we get RPC_E_TOO_LATE
442             // This can be safely ignored
443             if (COMUtils.FAILED(hres) && hres.intValue() != WinError.RPC_E_TOO_LATE) {
444                 Ole32.INSTANCE.CoUninitialize();
445                 throw new COMException("Failed to initialize security.", hres);
446             }
447             securityInitialized = true;
448         }
449     }

451     /**
452      * UnInitializes COM library if it was initialized by the {@link #initCOM()}
453      * method. Otherwise, does nothing.
454      */
455     public static void unInitCOM() {
456         if (isComInitialized()) {
457             Ole32.INSTANCE.CoUninitialize();
458             comInitialized = false;
459         }
460     }
