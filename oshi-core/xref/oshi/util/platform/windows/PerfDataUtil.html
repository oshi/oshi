32  /**
33   * Helper class to centralize the boilerplate portions of PDH counter setup and allow applications to easily add, query,
34   * and remove counters.
35   */
36  @ThreadSafe
37  public final class PerfDataUtil {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(PerfDataUtil.class);
40  
41      private static final DWORD_PTR PZERO = new DWORD_PTR(0);
42      private static final DWORDByReference PDH_FMT_RAW = new DWORDByReference(new DWORD(Pdh.PDH_FMT_RAW));
43      private static final Pdh PDH = Pdh.INSTANCE;
44  
45      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
46  
47      private PerfDataUtil() {
48      }
49  
50      /**
51       * Update a query and get the timestamp
52       *
53       * @param query The query to update all counters in
54       * @return The update timestamp of the first counter in the query
55       */
56      public static long updateQueryTimestamp(HANDLEByReference query) {
57          try (CloseableLONGLONGByReference pllTimeStamp = new CloseableLONGLONGByReference()) {
58              int ret = IS_VISTA_OR_GREATER ? PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp)
59                      : PDH.PdhCollectQueryData(query.getValue());
60              // Due to race condition, initial update may fail with PDH_NO_DATA.
61              int retries = 0;
62              while (ret == PdhMsg.PDH_NO_DATA && retries++ < 3) {
63                  // Exponential fallback.
64                  Util.sleep(1 << retries);
65                  ret = IS_VISTA_OR_GREATER ? PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp)
66                          : PDH.PdhCollectQueryData(query.getValue());
67              }
68              if (ret == WinError.ERROR_SUCCESS) {
69                  // Perf Counter timestamp is in local time
70                  return IS_VISTA_OR_GREATER ? ParseUtil.filetimeToUtcMs(pllTimeStamp.getValue().longValue(), true)
71                          : System.currentTimeMillis();
72              }
73              return handleError(ret, WARN, "Failed to update counter.", 0L);
74          }
75      }
76  
77      /**
78       * Open a pdh query
79       *
80       * @param q pointer to the query
81       * @return true if successful
82       */
83      public static boolean openQuery(HANDLEByReference q) {
84          int ret = PDH.PdhOpenQuery(null, PZERO, q);
85          if (ret == WinError.ERROR_SUCCESS) {
86              return true;
87          }
88          return handleError(ret, ERROR, "Failed to open PDH Query.", false);
89      }
90  
91      /**
92       * Close a pdh query
93       *
94       * @param q pointer to the query
95       * @return true if successful
96       */
97      public static boolean closeQuery(HANDLEByReference q) {
98          return WinError.ERROR_SUCCESS == PDH.PdhCloseQuery(q.getValue());
99      }
100 
101     /**
102      * Get value of pdh counter
103      *
104      * @param counter The counter to get the value of
105      * @return long value of the counter, or negative value representing an error code
106      */
107     public static long queryCounter(HANDLEByReference counter) {
108         try (CloseablePdhRawCounter counterValue = new CloseablePdhRawCounter()) {
109             int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
110             if (ret == WinError.ERROR_SUCCESS) {
111                 return counterValue.FirstValue;
112             }
113             return handleError(ret, WARN, "Failed to get counter.", (long) ret);
114         }
115     }
116 
117     /**
118      * Get value of pdh counter's second value (base counters)
119      *
120      * @param counter The counter to get the value of
121      * @return long value of the counter's second value, or negative value representing an error code
122      */
123     public static long querySecondCounter(HANDLEByReference counter) {
124         try (CloseablePdhRawCounter counterValue = new CloseablePdhRawCounter()) {
125             int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
126             if (ret == WinError.ERROR_SUCCESS) {
127                 return counterValue.SecondValue;
128             }
129             return handleError(ret, WARN, "Failed to get counter.", (long) ret);
130         }
131     }
132 
133     /**
134      * Adds a pdh counter to a query
135      *
136      * @param query Pointer to the query to add the counter
137      * @param path  String name of the PerfMon counter. For Vista+, must be in English. Must localize this path for
138      *              pre-Vista.
139      * @param p     Pointer to the counter
140      * @return true if successful
141      */
142     public static boolean addCounter(HANDLEByReference query, String path, HANDLEByReference p) {
143         int ret = IS_VISTA_OR_GREATER ? PDH.PdhAddEnglishCounter(query.getValue(), path, PZERO, p)
144                 : PDH.PdhAddCounter(query.getValue(), path, PZERO, p);
145         if (ret == WinError.ERROR_SUCCESS) {
146             return true;
147         }
148         return handleError(ret, WARN, "Failed to add PDH Counter: " + path + ",", false);
149     }
150 
151     /**
152      * Remove a pdh counter
153      *
154      * @param p pointer to the counter
155      * @return true if successful
156      */
157     public static boolean removeCounter(HANDLEByReference p) {
158         return WinError.ERROR_SUCCESS == PDH.PdhRemoveCounter(p.getValue());
159     }
160 
161     static <T> T handleError(int ret, Level level, String message, T defaultValue) {
162         if (LOG.isEnabledForLevel(level)) {
163             LOG.atLevel(level).log("{} Error code: {}", message,
164                     String.format(Locale.ROOT, FormatUtil.formatError(ret)));
165         }
166         return defaultValue;
167     }
