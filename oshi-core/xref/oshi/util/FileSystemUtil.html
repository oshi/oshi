24  package oshi.util;
25  
26  import java.nio.file.FileSystem;
27  import java.nio.file.FileSystems;
28  import java.nio.file.Path;
29  import java.nio.file.PathMatcher;
30  import java.nio.file.Paths;
31  import java.util.ArrayList;
32  import java.util.List;
33  
34  import oshi.annotation.concurrent.ThreadSafe;
35  
36  /**
37   * Utility class for common filesystem code
38   */
39  @ThreadSafe
40  public final class FileSystemUtil {
41  
42      private static final String GLOB_PREFIX = "glob:";
43      private static final String REGEX_PREFIX = "regex:";
44  
45      private FileSystemUtil() {
46      }
47  
48      /**
49       * Evaluates if file store (identified by {@code path} and {@code volume})
50       * should be excluded or not based on configuration
51       * {@code pathIncludes, pathExcludes, volumeIncludes, volumeExcludes}.
52       *
53       * Inclusion has priority over exclusion. If no exclusion/inclusion pattern is
54       * specified, then filestore is not excluded.
55       *
56       * @param path
57       *            Mountpoint of filestore.
58       * @param volume
59       *            Filestore volume.
60       * @param pathIncludes
61       *            List of patterns for path inclusions.
62       * @param pathExcludes
63       *            List of patterns for path exclusions.
64       * @param volumeIncludes
65       *            List of patterns for volume inclusions.
66       * @param volumeExcludes
67       *            List of patterns for volume exclusions.
68       * @return {@code true} if file store should be excluded or {@code false}
69       *         otherwise.
70       */
71      public static boolean isFileStoreExcluded(String path, String volume, List<PathMatcher> pathIncludes,
72              List<PathMatcher> pathExcludes, List<PathMatcher> volumeIncludes, List<PathMatcher> volumeExcludes) {
73          Path p = Paths.get(path);
74          Path v = Paths.get(volume);
75          if (matches(p, pathIncludes) || matches(v, volumeIncludes)) {
76              return false;
77          }
78          return matches(p, pathExcludes) || matches(v, volumeExcludes);
79      }
80  
81      /**
82       * Load from config and parse file system include/exclude line.
83       *
84       * @param configPropertyName
85       *            The config property containing the line to be parsed.
86       * @return List of PathMatchers to be used to match filestore volume and path.
87       */
88      public static List<PathMatcher> loadAndParseFileSystemConfig(String configPropertyName) {
89          String config = GlobalConfig.get(configPropertyName, "");
90          return parseFileSystemConfig(config);
91      }
92  
93      /**
94       * Parse file system include/exclude line.
95       *
96       * @param config
97       *            The config line to be parsed.
98       * @return List of PathMatchers to be used to match filestore volume and path.
99       */
100     public static List<PathMatcher> parseFileSystemConfig(String config) {
101         FileSystem fs = FileSystems.getDefault(); // can't be closed
102         List<PathMatcher> patterns = new ArrayList<>();
103         for (String item : config.split(",")) {
104             if (item.length() > 0) {
105                 // Using glob: prefix as the defult unless user has specified glob or regex. See
106                 // https://docs.oracle.com/javase/8/docs/api/java/nio/file/FileSystem.html#getPathMatcher-java.lang.String-
107                 if (!(item.startsWith(GLOB_PREFIX) || item.startsWith(REGEX_PREFIX))) {
108                     item = GLOB_PREFIX + item;
109                 }
110                 patterns.add(fs.getPathMatcher(item));
111             }
112         }
113         return patterns;
114     }
115 
116     /**
117      * Checks if {@code text} matches any of @param patterns}.
118      *
119      * @param text
120      *            The text to be matched.
121      * @param patterns
122      *            List of patterns.
123      * @return {@code true} if given text matches at least one glob pattern or
124      *         {@code false} otherwise.
125      * @see <a href="https://en.wikipedia.org/wiki/Glob_(programming)">Wikipedia -
126      *      glob (programming)</a>
127      */
128     public static boolean matches(Path text, List<PathMatcher> patterns) {
129         for (PathMatcher pattern : patterns) {
130             if (pattern.matches(text)) {
131                 return true;
132             }
133         }
134         return false;
135     }
136 
137 }
