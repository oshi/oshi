5   package oshi.util.gpu;
6   
7   import java.util.List;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.driver.common.windows.gpu.DxgiAdapterInfo;
11  import oshi.driver.common.windows.gpu.DxgiUtil;
12  import oshi.jna.platform.windows.Dxgi;
13  
14  /**
15   * Utility methods for DXGI adapter enumeration and matching on Windows.
16   */
17  @ThreadSafe
18  public final class DxgiUtilJNA {
19  
20      private DxgiUtilJNA() {
21      }
22  
23      /**
24       * Enumerates all DXGI display adapters and returns their identity and dedicated video memory.
25       *
26       * @return list of {@link DxgiAdapterInfo}, one per adapter; empty if DXGI is unavailable
27       */
28      public static List<DxgiAdapterInfo> queryAdapters() {
29          return Dxgi.queryAdapters();
30      }
31  
32      /**
33       * Finds the best-matching DXGI adapter for a given vendor ID, device ID, and adapter name.
34       *
35       * @param adapters    list from {@link #queryAdapters()}
36       * @param vendorId    PCI vendor ID parsed from the registry key (0 if unknown)
37       * @param deviceId    PCI device ID parsed from the registry key (0 if unknown)
38       * @param adapterName adapter name from the registry {@code DriverDesc} value
39       * @return best-matching adapter, or {@code null}
40       */
41      public static DxgiAdapterInfo findMatch(List<DxgiAdapterInfo> adapters, int vendorId, int deviceId,
42              String adapterName) {
43          return DxgiUtil.findMatch(adapters, vendorId, deviceId, adapterName);
44      }
45  
46      /**
47       * Converts a registry value to a VRAM size in bytes.
48       *
49       * @param value the registry value object
50       * @return the VRAM size in bytes, or 0 if the value type is unrecognised
51       */
52      public static long registryValueToVram(Object value) {
53          return DxgiUtil.registryValueToVram(value);
54      }
55  
56      /**
57       * Normalizes an adapter name for fuzzy matching.
58       *
59       * @param name the raw adapter name, may be {@code null}
60       * @return normalized name, never {@code null}
61       */
62      public static String normalizeName(String name) {
63          return DxgiUtil.normalizeName(name);
64      }
65  }
