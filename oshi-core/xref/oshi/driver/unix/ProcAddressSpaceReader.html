5   package oshi.driver.unix;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   
10  import com.sun.jna.Memory;
11  import com.sun.jna.NativeLong;
12  import com.sun.jna.platform.unix.LibCAPI.size_t;
13  import com.sun.jna.platform.unix.LibCAPI.ssize_t;
14  
15  import oshi.annotation.concurrent.NotThreadSafe;
16  import oshi.jna.platform.unix.CLibrary;
17  
18  /**
19   * Reads pointer-sized values and NUL-terminated strings out of a process's {@code /proc/<pid>/as} address space via
20   * libc {@code pread}, buffering one page at a time. Shared by the Solaris and AIX JNA {@code PsInfo} drivers, which
21   * read a process's argument and environment vectors this way.
22   * <p>
23   * Not thread-safe: each instance owns a single file descriptor and a reusable page buffer, so confine it to one thread
24   * and {@link #close()} it when done (it is {@link AutoCloseable}).
25   */
26  @NotThreadSafe
27  public final class ProcAddressSpaceReader implements AutoCloseable {
28  
29      private static final Logger LOG = LoggerFactory.getLogger(ProcAddressSpaceReader.class);
30  
31      private final CLibrary libc;
32      private final int fd;
33      private final long pageSize;
34      private final Memory buffer;
35      private final size_t bufSize;
36      private long bufStart;
37      private boolean bufValid;
38  
39      private ProcAddressSpaceReader(CLibrary libc, int fd, long pageSize) {
40          this.libc = libc;
41          this.fd = fd;
42          this.pageSize = pageSize;
43          this.buffer = new Memory(pageSize * 2);
44          this.bufSize = new size_t(this.buffer.size());
45          this.bufStart = 0L;
46          this.bufValid = false;
47      }
48  
49      /**
50       * Opens {@code /proc/<pid>/as} for reading.
51       *
52       * @param libc     the platform C library providing {@code open}/{@code pread}/{@code close}
53       * @param pid      the process ID
54       * @param pageSize the memory page size in bytes
55       * @return a reader, or {@code null} if the address space could not be opened (e.g. insufficient permission)
56       */
57      public static ProcAddressSpaceReader open(CLibrary libc, int pid, long pageSize) {
58          String procas = "/proc/" + pid + "/as";
59          int fd = libc.open(procas, 0);
60          if (fd < 0) {
61              LOG.trace("No permission to read file: {} ", procas);
62              return null;
63          }
64          return new ProcAddressSpaceReader(libc, fd, pageSize);
65      }
66  
67      /**
68       * Reads a pointer-sized value at {@code addr}.
69       *
70       * @param addr      the address to read
71       * @param increment the pointer size in bytes ({@code 8} for a 64-bit process, otherwise a 32-bit value)
72       * @return the pointer value, or {@code 0} if the page could not be read
73       */
74      public long readPointer(long addr, long increment) {
75          if (!ensurePage(addr)) {
76              return 0L;
77          }
78          return decodePointer(this.buffer, addr - this.bufStart, increment);
79      }
80  
81      /**
82       * Reads a NUL-terminated string at {@code addr}.
83       *
84       * @param addr the address to read
85       * @return the string, or an empty string if the page could not be read
86       */
87      public String readString(long addr) {
88          if (!ensurePage(addr)) {
89              return "";
90          }
91          return this.buffer.getString(addr - this.bufStart);
92      }
93  
94      /**
95       * Ensures the page containing {@code addr} is loaded in the buffer, reading it via {@code pread} if it is not
96       * already present.
97       *
98       * @param addr the address whose page must be buffered
99       * @return {@code true} if the page is available, {@code false} on read failure
100      */
101     private boolean ensurePage(long addr) {
102         // Reuse the buffer only if it holds a valid page covering addr
103         if (this.bufValid && addr >= this.bufStart && addr - this.bufStart <= this.pageSize) {
104             return true;
105         }
106         // pread overwrites the buffer, so invalidate the cached page before reading: a short read would otherwise
107         // leave bufStart pointing at a page the buffer no longer fully holds.
108         this.bufValid = false;
109         long newStart = Math.floorDiv(addr, this.pageSize) * this.pageSize;
110         ssize_t result = this.libc.pread(this.fd, this.buffer, this.bufSize, new NativeLong(newStart));
111         // May return less than asked but should be at least a full page
112         if (result.longValue() < this.pageSize) {
113             LOG.debug("Failed to read page from address space: {} bytes read", result.longValue());
114             return false;
115         }
116         this.bufStart = newStart;
117         this.bufValid = true;
118         return true;
119     }
120 
121     /**
122      * Decodes a pointer-sized value from a buffer. For 32-bit pointers the value is read unsigned so high addresses
123      * ({@code >= 0x80000000}) are not sign-extended.
124      *
125      * @param buffer    the buffer to read from
126      * @param offset    the offset within the buffer
127      * @param increment the pointer size in bytes ({@code 8} for 64-bit, otherwise 32-bit)
128      * @return the decoded value
129      */
130     static long decodePointer(Memory buffer, long offset, long increment) {
131         return increment == 8 ? buffer.getLong(offset) : Integer.toUnsignedLong(buffer.getInt(offset));
132     }
133 
134     @Override
135     public void close() {
136         this.libc.close(this.fd);
137         this.buffer.close();
138     }
139 }
