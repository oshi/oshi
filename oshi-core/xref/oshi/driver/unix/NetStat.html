49      /**
50       * Query netstat to obtain number of established TCP connections
51       *
52       * @return A pair with number of established IPv4 and IPv6 connections
53       */
54      public static Pair<Long, Long> queryTcpnetstat() {
55          long tcp4 = 0L;
56          long tcp6 = 0L;
57          List<String> activeConns = ExecutingCommand.runNative("netstat -n -p tcp");
58          for (String s : activeConns) {
59              if (s.endsWith("ESTABLISHED")) {
60                  if (s.startsWith("tcp4")) {
61                      tcp4++;
62                  } else if (s.startsWith("tcp6")) {
63                      tcp6++;
64                  }
65              }
66          }
67          return new Pair<>(tcp4, tcp6);
68      }
69  
70      /**
71       * Query netstat to all TCP and UDP connections
72       *
73       * @return A list of TCP and UDP connections
74       */
75      public static List<IPConnection> queryNetstat() {
76          List<IPConnection> connections = new ArrayList<>();
77          List<String> activeConns = ExecutingCommand.runNative("netstat -n");
78          for (String s : activeConns) {
79              String[] split = null;
80              if (s.startsWith("tcp") || s.startsWith("udp")) {
81                  split = ParseUtil.whitespaces.split(s);
82                  if (split.length >= 5) {
83                      String state = (split.length == 6) ? split[5] : null;
84                      String type = split[0];
85                      Pair<byte[], Integer> local = parseIP(split[3]);
86                      Pair<byte[], Integer> foreign = parseIP(split[4]);
87                      connections.add(new IPConnection(type, local.getA(), local.getB(), foreign.getA(), foreign.getB(),
88                              state == null ? TcpState.NONE : TcpState.valueOf(state),
89                              ParseUtil.parseIntOrDefault(split[2], 0), ParseUtil.parseIntOrDefault(split[1], 0), -1));
90                  }
91              }
92          }
93          return connections;
94      }
95  
96      private static Pair<byte[], Integer> parseIP(String s) {
97          // 73.169.134.6.9599 to 73.169.134.6 port 9599
98          // or
99          // 2001:558:600a:a5.123 to 2001:558:600a:a5 port 123
100         int portPos = s.lastIndexOf('.');
101         if (portPos > 0 && s.length() > portPos) {
102             int port = ParseUtil.parseIntOrDefault(s.substring(portPos + 1), 0);
103             String ip = s.substring(0, portPos);
104             try {
105                 // Try to parse existing IP
106                 return new Pair<>(InetAddress.getByName(ip).getAddress(), port);
107             } catch (UnknownHostException e) {
108                 try {
109                     // Try again with trailing ::
110                     if (ip.endsWith(":") && ip.contains("::")) {
111                         ip = ip + "0";
112                     } else if (ip.endsWith(":") || ip.contains("::")) {
113                         ip = ip + ":0";
114                     } else {
115                         ip = ip + "::0";
116                     }
117                     return new Pair<>(InetAddress.getByName(ip).getAddress(), port);
118                 } catch (UnknownHostException e2) {
119                     return new Pair<>(new byte[0], port);
120                 }
121             }
122         }
123         return new Pair<>(new byte[0], 0);
124     }

126     /**
127      * Gets TCP stats via {@code netstat -s}. Used for Linux and OpenBSD formats
128      *
129      * @param netstatStr
130      *            The command string
131      * @return The statistics
132      */
133     public static TcpStats queryTcpStats(String netstatStr) {
134         long connectionsEstablished = 0;
135         long connectionsActive = 0;
136         long connectionsPassive = 0;
137         long connectionFailures = 0;
138         long connectionsReset = 0;
139         long segmentsSent = 0;
140         long segmentsReceived = 0;
141         long segmentsRetransmitted = 0;
142         long inErrors = 0;
143         long outResets = 0;
144         List<String> netstat = ExecutingCommand.runNative(netstatStr);
145         for (String s : netstat) {
146             String[] split = s.trim().split(" ", 2);
147             if (split.length == 2) {
148                 switch (split[1]) {
149                 case "connections established":
150                 case "connection established (including accepts)":
151                 case "connections established (including accepts)":
152                     connectionsEstablished = ParseUtil.parseLongOrDefault(split[0], 0L);
153                     break;
154                 case "active connection openings":
155                     connectionsActive = ParseUtil.parseLongOrDefault(split[0], 0L);
156                     break;
157                 case "passive connection openings":
158                     connectionsPassive = ParseUtil.parseLongOrDefault(split[0], 0L);
159                     break;
160                 case "failed connection attempts":
161                 case "bad connection attempts":
162                     connectionFailures = ParseUtil.parseLongOrDefault(split[0], 0L);
163                     break;
164                 case "connection resets received":
165                 case "dropped due to RST":
166                     connectionsReset = ParseUtil.parseLongOrDefault(split[0], 0L);
167                     break;
168                 case "segments sent out":
169                 case "packet sent":
170                 case "packets sent":
171                     segmentsSent = ParseUtil.parseLongOrDefault(split[0], 0L);
172                     break;
173                 case "segments received":
174                 case "packet received":
175                 case "packets received":
176                     segmentsReceived = ParseUtil.parseLongOrDefault(split[0], 0L);
177                     break;
178                 case "segments retransmitted":
179                     segmentsRetransmitted = ParseUtil.parseLongOrDefault(split[0], 0L);
180                     break;
181                 case "bad segments received":
182                 case "discarded for bad checksum":
183                 case "discarded for bad checksums":
184                 case "discarded for bad header offset field":
185                 case "discarded for bad header offset fields":
186                 case "discarded because packet too short":
187                 case "discarded for missing IPsec protection":
188                     inErrors += ParseUtil.parseLongOrDefault(split[0], 0L);
189                     break;
190                 case "resets sent":
191                     outResets = ParseUtil.parseLongOrDefault(split[0], 0L);
192                     break;
193                 default:
194                     // handle special case variable strings
195                     if (split[1].contains("retransmitted") && split[1].contains("data packet")) {
196                         segmentsRetransmitted += ParseUtil.parseLongOrDefault(split[0], 0L);
197                     }
198                     break;
199                 }
200 
201             }
202 
203         }
204         return new TcpStats(connectionsEstablished, connectionsActive, connectionsPassive, connectionFailures,
205                 connectionsReset, segmentsSent, segmentsReceived, segmentsRetransmitted, inErrors, outResets);
206     }

208     /**
209      * Gets UDP stats via {@code netstat -s}. Used for Linux and OpenBSD formats
210      *
211      * @param netstatStr
212      *            The command string
213      * @return The statistics
214      */
215     public static UdpStats queryUdpStats(String netstatStr) {
216         long datagramsSent = 0;
217         long datagramsReceived = 0;
218         long datagramsNoPort = 0;
219         long datagramsReceivedErrors = 0;
220         List<String> netstat = ExecutingCommand.runNative(netstatStr);
221         for (String s : netstat) {
222             String[] split = s.trim().split(" ", 2);
223             if (split.length == 2) {
224                 switch (split[1]) {
225                 case "packets sent":
226                 case "datagram output":
227                 case "datagrams output":
228                     datagramsSent = ParseUtil.parseLongOrDefault(split[0], 0L);
229                     break;
230                 case "packets received":
231                 case "datagram received":
232                 case "datagrams received":
233                     datagramsReceived = ParseUtil.parseLongOrDefault(split[0], 0L);
234                     break;
235                 case "packets to unknown port received":
236                 case "dropped due to no socket":
237                 case "broadcast/multicast datagram dropped due to no socket":
238                 case "broadcast/multicast datagrams dropped due to no socket":
239                     datagramsNoPort += ParseUtil.parseLongOrDefault(split[0], 0L);
240                     break;
241                 case "packet receive errors":
242                 case "with incomplete header":
243                 case "with bad data length field":
244                 case "with bad checksum":
245                 case "woth no checksum":
246                     datagramsReceivedErrors += ParseUtil.parseLongOrDefault(split[0], 0L);
247                     break;
248                 default:
249                     break;
250                 }
251             }
252         }
253         return new UdpStats(datagramsSent, datagramsReceived, datagramsNoPort, datagramsReceivedErrors);
254     }
