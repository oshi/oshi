5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery;
10  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.driver.common.windows.wmi.Win32Processor;
14  import oshi.driver.common.windows.wmi.Win32Processor.BitnessProperty;
15  import oshi.driver.common.windows.wmi.Win32Processor.ProcessorIdProperty;
16  import oshi.driver.common.windows.wmi.Win32Processor.VoltProperty;
17  import oshi.util.platform.windows.WmiQueryHandler;
18  
19  /**
20   * Utility to query WMI class {@code Win32_Processor} using JNA.
21   */
22  @ThreadSafe
23  public final class Win32ProcessorJNA extends Win32Processor {
24  
25      private Win32ProcessorJNA() {
26      }
27  
28      /**
29       * Returns processor voltage.
30       *
31       * @return Current voltage of the processor. If the eighth bit is set, bits 0-6 contain the voltage multiplied by
32       *         10. If the eighth bit is not set, then the bit setting in VoltageCaps represents the voltage value.
33       */
34      public static WmiResult<VoltProperty> queryVoltage() {
35          WmiQuery<VoltProperty> voltQuery = new WmiQuery<>(WIN32_PROCESSOR, VoltProperty.class);
36          return Objects
37                  .requireNonNull(WmiQueryHandler.createInstance(), "WmiQueryHandler.createInstance() returned null")
38                  .queryWMI(voltQuery);
39      }
40  
41      /**
42       * Returns processor ID.
43       *
44       * @return Processor information that describes the processor features.
45       */
46      public static WmiResult<ProcessorIdProperty> queryProcessorId() {
47          WmiQuery<ProcessorIdProperty> idQuery = new WmiQuery<>(WIN32_PROCESSOR, ProcessorIdProperty.class);
48          return Objects
49                  .requireNonNull(WmiQueryHandler.createInstance(), "WmiQueryHandler.createInstance() returned null")
50                  .queryWMI(idQuery);
51      }
52  
53      /**
54       * Returns address width.
55       *
56       * @return On a 32-bit operating system, the value is 32 and on a 64-bit operating system it is 64.
57       */
58      public static WmiResult<BitnessProperty> queryBitness() {
59          WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<>(WIN32_PROCESSOR, BitnessProperty.class);
60          return Objects
61                  .requireNonNull(WmiQueryHandler.createInstance(), "WmiQueryHandler.createInstance() returned null")
62                  .queryWMI(bitnessQuery);
63      }
64  }
