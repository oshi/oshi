5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery;
10  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.driver.common.windows.wmi.MSAcpiThermalZoneTemperature;
14  import oshi.driver.common.windows.wmi.MSAcpiThermalZoneTemperature.TemperatureProperty;
15  import oshi.util.platform.windows.WmiQueryHandler;
16  
17  /**
18   * Utility to query WMI class {@code MSAcpi_ThermalZoneTemperature} using JNA.
19   */
20  @ThreadSafe
21  public final class MSAcpiThermalZoneTemperatureJNA extends MSAcpiThermalZoneTemperature {
22  
23      private MSAcpiThermalZoneTemperatureJNA() {
24      }
25  
26      /**
27       * Queries the current temperature.
28       *
29       * @return Temperature at thermal zone in tenths of degrees Kelvin.
30       */
31      public static WmiResult<TemperatureProperty> queryCurrentTemperature() {
32          WmiQuery<TemperatureProperty> curTempQuery = new WmiQuery<>(WMI_NAMESPACE, MS_ACPI_THERMAL_ZONE_TEMPERATURE,
33                  TemperatureProperty.class);
34          return Objects
35                  .requireNonNull(WmiQueryHandler.createInstance(), "WmiQueryHandler.createInstance() returned null")
36                  .queryWMI(curTempQuery);
37      }
38  }
