56   * Utility to read HKEY_PERFORMANCE_DATA information.
57   */

59  public final class HkeyPerformanceDataUtil {

61      private static final Logger LOG = LoggerFactory.getLogger(HkeyPerformanceDataUtil.class);

63      /*
64       * Do a one-time lookup of the HKEY_PERFORMANCE_TEXT counter indices and store
65       * in a map for efficient lookups on-demand.
66       */
67      private static final String HKEY_PERFORMANCE_TEXT = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Perflib\\009";
68      private static final String COUNTER = "Counter";
69      private static final Map<String, Integer> COUNTER_INDEX_MAP = mapCounterIndicesFromRegistry();
70  
71      private HkeyPerformanceDataUtil() {
72      }

74      /**
75       * Reads and parses a block of performance data from the registry.
76       *
77       * @param <T>
78       *            PDH Counters use an Enum to identify the fields to query in either
79       *            the counter or WMI backup, and use the enum values as keys to
80       *            retrieve the results.
81       * @param objectName
82       *            The counter object for which to fetch data
83       * @param counterEnum
84       *            Which counters to return data for
85       * @return A triplet containing the results. The first element maps the input
86       *         enum to the counter values where the first enum will contain the
87       *         instance name as a {@link String}, and the remaining values will
88       *         either be {@link Long}, {@link Integer}, or {@code null} depending on
89       *         whether the specified enum counter was present and the size of the
90       *         counter value. The second element is a timestamp in 100nSec
91       *         increments (Windows 1601 Epoch) while the third element is a
92       *         timestamp in milliseconds since the 1970 Epoch.
93       */
94      public static <T extends Enum<T> & PdhCounterWildcardProperty> Triplet<List<Map<T, Object>>, Long, Long> readPerfDataFromRegistry(
95              String objectName, Class<T> counterEnum) {
96          // Load indices
97          // e.g., call with "Process" and ProcessPerformanceProperty.class
98          Pair<Integer, EnumMap<T, Integer>> indices = getCounterIndices(objectName, counterEnum);
99          if (indices == null) {
100             return null;
101         }
102         // The above test checks validity of objectName as an index but it could still
103         // fail to read
104         Memory pPerfData = readPerfDataBuffer(objectName);
105         if (pPerfData == null) {
106             return null;
107         }
108         // Buffer is now successfully populated.
109         // See format at
110         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
111 
112         // Start with a data header (PERF_DATA_BLOCK)
113         // Then iterate one or more objects
114         // Each object contains
115         // [ ] Object Type header (PERF_OBJECT_TYPE)
116         // [ ][ ][ ] Multiple counter definitions (PERF_COUNTER_DEFINITION)
117         // Then after object(s), multiple:
118         // [ ] Instance Definition
119         // [ ] Instance name
120         // [ ] Counter Block
121         // [ ][ ][ ] Counter data for each definition above
122 
123         // Store timestamp
124         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
125         long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
126         long now = FILETIME.filetimeToDate((int) (perfTime100nSec >> 32), (int) (perfTime100nSec & 0xffffffffL))
127                 .getTime(); // 1970
128 
129         // Iterate object types.
130         long perfObjectOffset = perfData.HeaderLength;
131         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
132             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
133             // Some counters will require multiple objects so we iterate until we find the
134             // right one. e.g. Process (230) is by iteself but Thread (232) has Process
135             // object first
136             if (perfObject.ObjectNameTitleIndex == COUNTER_INDEX_MAP.get(objectName).intValue()) {
137                 // We found a matching object.
138 
139                 // Counter definitions start after the object header
140                 long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
141                 // Iterate counter definitions and fill maps with counter offsets and sizes
142                 Map<Integer, Integer> counterOffsetMap = new HashMap<>();
143                 Map<Integer, Integer> counterSizeMap = new HashMap<>();
144                 for (int counter = 0; counter < perfObject.NumCounters; counter++) {
145                     PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
146                             pPerfData.share(perfCounterOffset));
147                     counterOffsetMap.put(perfCounter.CounterNameTitleIndex, perfCounter.CounterOffset);
148                     counterSizeMap.put(perfCounter.CounterNameTitleIndex, perfCounter.CounterSize);
149                     // Increment for next Counter
150                     perfCounterOffset += perfCounter.ByteLength;
151                 }
152 
153                 // Instances start after all the object definitions. The DefinitionLength
154                 // includes both the header and all the definitions.
155                 long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
156 
157                 // Iterate instances and fill map
158                 List<Map<T, Object>> counterMaps = new ArrayList<>(perfObject.NumInstances);
159                 for (int inst = 0; inst < perfObject.NumInstances; inst++) {
160                     PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
161                             pPerfData.share(perfInstanceOffset));
162                     long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
163                     // Populate the enumMap
164                     Map<T, Object> counterMap = new EnumMap<>(counterEnum);
165                     T[] counterKeys = counterEnum.getEnumConstants();
166                     // First enum index is the name, ignore the counter text which is used for other
167                     // purposes
168                     counterMap.put(counterKeys[0],
169                             pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
170                     for (int i = 1; i < counterKeys.length; i++) {
171                         T key = counterKeys[i];
172                         int keyIndex = COUNTER_INDEX_MAP.get(key.getCounter());
173                         // All entries in size map have corresponding entry in offset map
174                         int size = counterSizeMap.getOrDefault(keyIndex, 0);
175                         // Currently, only DWORDs (4 bytes) and ULONGLONGs (8 bytes) are used to provide
176                         // counter values.
177                         if (size == 4) {
178                             counterMap.put(key,
179                                     pPerfData.getInt(perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
180                         } else if (size == 8) {
181                             counterMap.put(key,
182                                     pPerfData.getLong(perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
183                         } else {
184                             // If counter defined in enum isn't in registry, fail
185                             return null;
186                         }
187                     }
188                     counterMaps.add(counterMap);
189 
190                     // counters at perfCounterBlockOffset + appropriate offset per enum
191                     // use pPerfData.getInt or getLong as determined by counter size
192                     // Currently, only DWORDs (4 bytes) and ULONGLONGs (8 bytes) are used to provide
193                     // counter values.
194 
195                     // Increment to next instance
196                     perfInstanceOffset = perfCounterBlockOffset
197                             + new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset)).ByteLength;
198                 }
199                 // We've found the necessary object and are done, no need to look at any other
200                 // objects (shouldn't be any). Return results
201                 return new Triplet<>(counterMaps, perfTime100nSec, now);
202             }
203             // Increment for next object
204             perfObjectOffset += perfObject.TotalByteLength;
205         }
206         // Failed, return null
207         return null;
208     }

210     /**
211      * Looks up the counter index values for the given counter object and the enum
212      * of counter names.
213      *
214      * @param <T>
215      *            An enum containing the counters, whose class is passed as
216      *            {@code counterEnum}
217      * @param objectName
218      *            The counter object to look up the index for
219      * @param counterEnum
220      *            The {@link Enum} containing counters to look up the indices for.
221      *            The first Enum value will be ignored.
222      * @return A {@link Pair} containing the index of the counter object as the
223      *         first element, and an {@link EnumMap} mapping counter enum values to
224      *         their index as the second element, if the lookup is successful; null
225      *         otherwise.
226      */
227     private static <T extends Enum<T> & PdhCounterWildcardProperty> Pair<Integer, EnumMap<T, Integer>> getCounterIndices(
228             String objectName, Class<T> counterEnum) {
229         if (!COUNTER_INDEX_MAP.containsKey(objectName)) {
230             LOG.debug("Couldn't find counter index of {}.", objectName);
231             return null;
232         }
233         int counterIndex = COUNTER_INDEX_MAP.get(objectName);
234         T[] enumConstants = counterEnum.getEnumConstants();
235         EnumMap<T, Integer> indexMap = new EnumMap<>(counterEnum);
236         // Start iterating at 1 because first Enum value defines the name/instance and
237         // is not a counter name
238         for (int i = 1; i < enumConstants.length; i++) {
239             T key = enumConstants[i];
240             String counterName = key.getCounter();
241             if (!COUNTER_INDEX_MAP.containsKey(counterName)) {
242                 LOG.debug("Couldn't find counter index of {}.", counterName);
243                 return null;
244             }
245             indexMap.put(key, COUNTER_INDEX_MAP.get(counterName));
246         }
247         // We have all the pieces! Return them.
248         return new Pair<>(counterIndex, indexMap);
249     }

251     /**
252      * Read the performance data for a counter object from the registry.
253      *
254      * @param objectName
255      *            The counter object for which to fetch data. It is the user's
256      *            responsibility to ensure this key exists in
257      *            {@link #COUNTER_INDEX_MAP}.
258      * @return A buffer containing the data if successful, null otherwise.
259      */
260     private static Memory readPerfDataBuffer(String objectName) {
261         // Need this index as a string
262         String objectIndexStr = Integer.toString(COUNTER_INDEX_MAP.get(objectName));
263 
264         // Now load the data from the regsitry.
265         int bufferSize = 4096;
266         IntByReference lpcbData = new IntByReference(bufferSize);
267         Memory pPerfData = new Memory(bufferSize);
268         int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, objectIndexStr, 0, null, pPerfData,
269                 lpcbData);
270         if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
271             LOG.error("Error reading performance data from registry for {}.", objectName);
272             return null;
273         }
274         // Grow buffer as needed to fit the data
275         while (ret == WinError.ERROR_MORE_DATA) {
276             bufferSize += 4096;
277             lpcbData.setValue(bufferSize);
278             pPerfData = new Memory(bufferSize);
279             ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, objectIndexStr, 0, null, pPerfData,
280                     lpcbData);
281         }
282         return pPerfData;
283     }

285     /*
286      * Registry entries subordinate to HKEY_PERFORMANCE_TEXT key reference the text
287      * strings that describe counters in US English. Not supported in Windows 2000.
288      *
289      * With the "Counter" value, the resulting array contains alternating index/name
290      * pairs "1", "1847", "2", "System", "4", "Memory", ...
291      *
292      * These pairs are translated to a map for later lookup.
293      *
294      * @return An unmodifiable map containing counter name strings as keys and
295      * indices as integer values if the key is read successfully; an empty map
296      * otherwise.
297      */
298     private static Map<String, Integer> mapCounterIndicesFromRegistry() {
299         HashMap<String, Integer> indexMap = new HashMap<>();
300         try {
301             String[] counterText = Advapi32Util.registryGetStringArray(WinReg.HKEY_LOCAL_MACHINE, HKEY_PERFORMANCE_TEXT,
302                     COUNTER);
303             for (int i = 1; i < counterText.length; i += 2) {
304                 indexMap.putIfAbsent(counterText[i], Integer.parseInt(counterText[i - 1]));
305             }
306         } catch (Win32Exception we) {
307             LOG.error(
308                     "Unable to locate English counter names in registry Perflib 009. Counters may need to be rebuilt: ",
309                     we);
310         } catch (NumberFormatException nfe) {
311             // Unexpected to ever get this, but handling it anyway
312             LOG.error("Unable to parse English counter names in registry Perflib 009.");
313         }
314         return Collections.unmodifiableMap(indexMap);
315     }
