37   * Utility to read HKEY_PERFORMANCE_DATA information.
38   */

42      private static final Logger LOG = LoggerFactory.getLogger(HkeyPerformanceDataUtilJNA.class);
43  
44      private static final Map<String, Integer> COUNTER_INDEX_MAP = mapCounterIndicesFromRegistry();
45  
46      private static int maxPerfBufferSize = 16384;
47  
48      private HkeyPerformanceDataUtilJNA() {
49      }

51      /**
52       * Reads and parses a block of performance data from the registry.
53       *
54       * @param <T>         PDH Counters use an Enum to identify the fields to query in either the counter or WMI backup,
55       *                    and use the enum values as keys to retrieve the results.
56       * @param objectName  The counter object for which to fetch data
57       * @param counterEnum Which counters to return data for
58       * @return A triplet containing the results. The first element maps the input enum to the counter values where the
59       *         first enum will contain the instance name as a {@link String}, and the remaining values will either be
60       *         {@link Long}, {@link Integer}, or {@code null} depending on whether the specified enum counter was
61       *         present and the size of the counter value. The second element is a timestamp in 100nSec increments
62       *         (Windows 1601 Epoch) while the third element is a timestamp in milliseconds since the 1970 Epoch.
63       */
64      public static <T extends Enum<T> & PdhCounterWildcardProperty> Triplet<List<Map<T, Object>>, Long, Long> readPerfDataFromRegistry(
65              String objectName, Class<T> counterEnum) {
66          // Load indices
67          Pair<Integer, EnumMap<T, Integer>> indices = getCounterIndices(objectName, counterEnum, COUNTER_INDEX_MAP);
68          if (indices == null) {
69              return null;
70          }
71          int objectIndex = indices.getA();
72          EnumMap<T, Integer> enumIndexMap = indices.getB();
73          // The above test checks validity of objectName as an index but it could still
74          // fail to read
75          try (Memory pPerfData = readPerfDataBuffer(objectName)) {
76              if (pPerfData == null) {
77                  return null;
78              }
79              // Buffer is now successfully populated.
80  
81              // Store timestamp
82              PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
83              long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
84              long now = ParseUtil.filetimeToUtcMs(perfTime100nSec, false); // 1970
85  
86              // Iterate object types.
87              long perfObjectOffset = perfData.HeaderLength;
88              for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
89                  PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
90                  if (perfObject.ObjectNameTitleIndex == objectIndex) {
91                      // We found a matching object.
92  
93                      // Counter definitions start after the object header
94                      long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
95                      // Iterate counter definitions and fill maps with counter offsets and sizes
96                      Map<Integer, Integer> counterOffsetMap = new HashMap<>();
97                      Map<Integer, Integer> counterSizeMap = new HashMap<>();
98                      for (int counter = 0; counter < perfObject.NumCounters; counter++) {
99                          PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
100                                 pPerfData.share(perfCounterOffset));
101                         counterOffsetMap.put(perfCounter.CounterNameTitleIndex, perfCounter.CounterOffset);
102                         counterSizeMap.put(perfCounter.CounterNameTitleIndex, perfCounter.CounterSize);
103                         // Increment for next Counter
104                         perfCounterOffset += perfCounter.ByteLength;
105                     }
106 
107                     // Instances start after all the object definitions.
108                     long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
109 
110                     // Iterate instances and fill map
111                     T[] counterKeys = counterEnum.getEnumConstants();
112                     List<Map<T, Object>> counterMaps = new ArrayList<>(perfObject.NumInstances);
113                     for (int inst = 0; inst < perfObject.NumInstances; inst++) {
114                         PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
115                                 pPerfData.share(perfInstanceOffset));
116                         long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
117                         // Populate the enumMap
118                         Map<T, Object> counterMap = new EnumMap<>(counterEnum);
119                         counterMap.put(counterKeys[0],
120                                 pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
121                         for (int i = 1; i < counterKeys.length; i++) {
122                             T key = counterKeys[i];
123                             int keyIndex = enumIndexMap.get(key);
124                             int size = counterSizeMap.getOrDefault(keyIndex, 0);
125                             if (size == 4) {
126                                 counterMap.put(key,
127                                         pPerfData.getInt(perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
128                             } else if (size == 8) {
129                                 counterMap.put(key,
130                                         pPerfData.getLong(perfCounterBlockOffset + counterOffsetMap.get(keyIndex)));
131                             } else {
132                                 return null;
133                             }
134                         }
135                         counterMaps.add(counterMap);
136 
137                         // Increment to next instance
138                         perfInstanceOffset = perfCounterBlockOffset
139                                 + new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset)).ByteLength;
140                     }
141                     return new Triplet<>(counterMaps, perfTime100nSec, now);
142                 }
143                 // Increment for next object
144                 perfObjectOffset += perfObject.TotalByteLength;
145             }
146         }
147         return null;
148     }

150     /**
151      * Read the performance data for a counter object from the registry.
152      *
153      * @param objectName The counter object for which to fetch data.
154      * @return A buffer containing the data if successful, null otherwise.
155      */
156     private static synchronized Memory readPerfDataBuffer(String objectName) {
157         String objectIndexStr = Integer.toString(COUNTER_INDEX_MAP.get(objectName));
158 
159         try (CloseableIntByReference lpcbData = new CloseableIntByReference(maxPerfBufferSize)) {
160             Memory pPerfData = new Memory(maxPerfBufferSize);
161             int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, objectIndexStr, 0, null,
162                     pPerfData, lpcbData);
163             if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
164                 LOG.error("Error reading performance data from registry for {}.", objectName);
165                 pPerfData.close();
166                 return null;
167             }
168             while (ret == WinError.ERROR_MORE_DATA) {
169                 maxPerfBufferSize += 8192;
170                 lpcbData.setValue(maxPerfBufferSize);
171                 pPerfData.close();
172                 pPerfData = new Memory(maxPerfBufferSize);
173                 ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, objectIndexStr, 0, null,
174                         pPerfData, lpcbData);
175             }
176             // The buffer-grow retry can exit on a hard error rather than ERROR_SUCCESS; return null instead of an
177             // unvalidated buffer. Matches HkeyPerformanceDataUtilFFM.
178             if (ret != WinError.ERROR_SUCCESS) {
179                 LOG.error("Error reading performance data from registry for {}.", objectName);
180                 pPerfData.close();
181                 return null;
182             }
183             return pPerfData;
184         }
185     }

187     private static Map<String, Integer> mapCounterIndicesFromRegistry() {
188         try {
189             String[] counterText = Advapi32Util.registryGetStringArray(WinReg.HKEY_LOCAL_MACHINE, HKEY_PERFORMANCE_TEXT,
190                     COUNTER);
191             return buildCounterIndexMap(counterText);
192         } catch (Win32Exception we) {
193             LOG.error(
194                     "Unable to locate English counter names in registry Perflib 009. Counters may need to be rebuilt: ",
195                     we);
196         }
197         return buildCounterIndexMap(null);
198     }
