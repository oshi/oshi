5   package oshi.driver.linux.proc;
6   
7   import java.nio.ByteBuffer;
8   import java.util.HashMap;
9   import java.util.Map;
10  
11  import oshi.annotation.concurrent.ThreadSafe;
12  import oshi.jna.util.FileUtilJNA;
13  import oshi.util.FileUtil;
14  import oshi.util.driver.linux.proc.Auxv;
15  import oshi.util.linux.ProcPath;
16  
17  /**
18   * JNA-based utility to read the auxiliary vector from {@code /proc/self/auxv}.
19   */
20  @ThreadSafe
21  public final class AuxvJNA {
22  
23      private AuxvJNA() {
24      }
25  
26      /**
27       * Retrieve the auxiliary vector for the current process
28       *
29       * @return A map of auxiliary vector keys to their respective values
30       * @see <a href= "https://github.com/torvalds/linux/blob/v3.19/include/uapi/linux/auxvec.h">auxvec.h</a>
31       */
32      public static Map<Integer, Long> queryAuxv() {
33          ByteBuffer buff = FileUtil.readAllBytesAsBuffer(ProcPath.AUXV);
34          Map<Integer, Long> auxvMap = new HashMap<>();
35          int key;
36          do {
37              key = FileUtilJNA.readNativeLongFromBuffer(buff).intValue();
38              if (key != Auxv.AT_NULL) {
39                  auxvMap.put(key, FileUtilJNA.readNativeLongFromBuffer(buff).longValue());
40              }
41          } while (key != Auxv.AT_NULL);
42          return auxvMap;
43      }
44  }
