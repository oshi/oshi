79      /**
80       * Open a connection to SMC.
81       *
82       * @return The io_connect_t port (nonzero) if successful, 0 if failure. Caller must close with {@link #smcClose}.
83       */
84      public static int smcOpen() {
85          IOService smcService = IOKitUtilFFM.getMatchingService("AppleSMC");
86          if (smcService == null) {
87              LOG.error("Unable to locate AppleSMC service");
88              return 0;
89          }
90          try (Arena arena = Arena.ofConfined()) {
91              MemorySegment connPtr = arena.allocate(JAVA_INT);
92              int task = mach_task_self();
93              int result = IOServiceOpen(smcService.segment(), task, 0, connPtr);
94              if (result == 0) {
95                  int conn = connPtr.get(JAVA_INT, 0);
96                  if (conn == 0) {
97                      LOG.error("IOServiceOpen returned null connect handle");
98                      return 0;
99                  }
100                 return conn;
101             }
102             String hex = String.format(Locale.ROOT, "0x%08x", result);
103             LOG.error("Unable to open connection to AppleSMC service. Error: {}", hex);
104             return 0;
105         } catch (Throwable e) {
106             LOG.error("Exception opening SMC connection", e);
107             return 0;
108         } finally {
109             smcService.release();
110         }
111     }

113     /**
114      * Close connection to SMC.
115      *
116      * @param conn The io_connect_t port returned by {@link #smcOpen}
117      * @return 0 if successful, nonzero if failure
118      */
119     public static int smcClose(int conn) {
120         try {
121             return IOServiceClose(conn);
122         } catch (Throwable e) {
123             return -1;
124         }
125     }

127     /**
128      * Get a value from SMC which is in a floating point datatype (SP78, FPE2, FLT)
129      *
130      * @param conn The connection
131      * @param key  The key to retrieve
132      * @return Double representing the value
133      */
134     public static double smcGetFloat(int conn, String key) {
135         try (Arena arena = Arena.ofConfined()) {
136             MemorySegment val = arena.allocate(SMC_VAL);
137             int result = smcReadKey(conn, key, val, arena);
138             if (result == 0) {
139                 int dataSize = val.get(JAVA_INT, SMC_VAL.byteOffset(SMC_VAL_DATA_SIZE));
140                 if (dataSize > 0) {
141                     byte[] dataType = readByteArray(val, SMC_VAL.byteOffset(SMC_VAL_DATA_TYPE), 5);
142                     byte[] bytes = readByteArray(val, SMC_VAL.byteOffset(SMC_VAL_BYTES), dataSize);
143                     if (Arrays.equals(dataType, DATATYPE_SP78) && dataSize == 2) {
144                         return bytes[0] + (bytes[1] & 0xFF) / 256d;
145                     } else if (Arrays.equals(dataType, DATATYPE_FPE2) && dataSize == 2) {
146                         return ParseUtil.byteArrayToFloat(bytes, dataSize, 2);
147                     } else if (Arrays.equals(dataType, DATATYPE_FLT) && dataSize == 4) {
148                         return ByteBuffer.wrap(bytes).order(ByteOrder.LITTLE_ENDIAN).getFloat();
149                     }
150                 }
151             }
152         } catch (Throwable e) {
153             LOG.warn("Failed to read SMC float key {}: {}", key, e.getMessage());
154         }
155         return 0d;
156     }

158     /**
159      * Get the first positive value from a list of SMC keys.
160      *
161      * @param conn The connection
162      * @param keys The keys to try in order
163      * @return The first value greater than 0, or 0 if all keys fail
164      */
165     public static double smcGetFirstFloat(int conn, String... keys) {
166         for (String key : keys) {
167             double val = smcGetFloat(conn, key);
168             if (val > 0d) {
169                 return val;
170             }
171         }
172         return 0d;
173     }

175     /**
176      * Get a 64-bit integer value from SMC
177      *
178      * @param conn The connection
179      * @param key  The key to retrieve
180      * @return Long representing the value
181      */
182     public static long smcGetLong(int conn, String key) {
183         try (Arena arena = Arena.ofConfined()) {
184             MemorySegment val = arena.allocate(SMC_VAL);
185             int result = smcReadKey(conn, key, val, arena);
186             if (result == 0) {
187                 int dataSize = val.get(JAVA_INT, SMC_VAL.byteOffset(SMC_VAL_DATA_SIZE));
188                 byte[] bytes = readByteArray(val, SMC_VAL.byteOffset(SMC_VAL_BYTES), dataSize);
189                 return ParseUtil.byteArrayToLong(bytes, dataSize);
190             }
191         } catch (Throwable e) {
192             LOG.warn("Failed to read SMC long key {}: {}", key, e.getMessage());
193         }
194         return 0L;
195     }

197     private static int smcReadKey(int conn, String key, MemorySegment val, Arena arena) throws Throwable {
198         MemorySegment input = arena.allocate(SMC_KEY_DATA);
199         MemorySegment output = arena.allocate(SMC_KEY_DATA);
200 
201         input.set(JAVA_INT, SMC_KEY_DATA.byteOffset(SMC_KEY), (int) ParseUtil.strToLong(key, 4));
202 
203         int result = smcGetKeyInfo(conn, input, output, arena);
204         if (result == 0) {
205             int dataSize = output.get(JAVA_INT, KEY_INFO_DATA_SIZE_OFFSET);
206             int dataType = output.get(JAVA_INT, KEY_INFO_DATA_TYPE_OFFSET);
207 
208             val.set(JAVA_INT, SMC_VAL.byteOffset(SMC_VAL_DATA_SIZE), dataSize);
209             byte[] typeBytes = ParseUtil.longToByteArray(dataType, 4, 5);
210             writeByteArray(val, SMC_VAL.byteOffset(SMC_VAL_DATA_TYPE), typeBytes);
211 
212             // Reset input and re-populate for the actual read
213             input.fill((byte) 0);
214             input.set(JAVA_INT, SMC_KEY_DATA.byteOffset(SMC_KEY), (int) ParseUtil.strToLong(key, 4));
215             input.set(JAVA_INT, KEY_INFO_DATA_SIZE_OFFSET, dataSize);
216             input.set(JAVA_BYTE, SMC_KEY_DATA.byteOffset(SMC_DATA8), SMC_CMD_READ_BYTES);
217 
218             result = smcCall(conn, KERNEL_INDEX_SMC, input, output);
219             if (result == 0) {
220                 byte[] outBytes = readByteArray(output, SMC_KEY_DATA.byteOffset(SMC_BYTES), 32);
221                 writeByteArray(val, SMC_VAL.byteOffset(SMC_VAL_BYTES), outBytes);
222             }
223         }
224         return result;
225     }

227     private static int smcGetKeyInfo(int conn, MemorySegment input, MemorySegment output, Arena arena)
228             throws Throwable {
229         int key = input.get(JAVA_INT, SMC_KEY_DATA.byteOffset(SMC_KEY));
230         int[] cached = KEY_INFO_CACHE.get(key);
231         if (cached != null) {
232             output.set(JAVA_INT, KEY_INFO_DATA_SIZE_OFFSET, cached[0]);
233             output.set(JAVA_INT, KEY_INFO_DATA_TYPE_OFFSET, cached[1]);
234             output.set(JAVA_BYTE, KEY_INFO_DATA_ATTR_OFFSET, (byte) cached[2]);
235             return 0;
236         }
237         input.set(JAVA_BYTE, SMC_KEY_DATA.byteOffset(SMC_DATA8), SMC_CMD_READ_KEYINFO);
238         int result = smcCall(conn, KERNEL_INDEX_SMC, input, output);
239         if (result == 0) {
240             int dataSize = output.get(JAVA_INT, KEY_INFO_DATA_SIZE_OFFSET);
241             int dataType = output.get(JAVA_INT, KEY_INFO_DATA_TYPE_OFFSET);
242             byte dataAttr = output.get(JAVA_BYTE, KEY_INFO_DATA_ATTR_OFFSET);
243             KEY_INFO_CACHE.put(key, new int[] { dataSize, dataType, dataAttr & 0xFF });
244         }
245         return result;
246     }

248     private static int smcCall(int conn, int index, MemorySegment input, MemorySegment output) throws Throwable {
249         try (Arena arena = Arena.ofConfined()) {
250             MemorySegment outputSize = arena.allocate(JAVA_LONG);
251             outputSize.set(JAVA_LONG, 0, SMC_KEY_DATA.byteSize());
252             return IOConnectCallStructMethod(conn, index, input, SMC_KEY_DATA.byteSize(), output, outputSize);
253         }
254     }

256     private static byte[] readByteArray(MemorySegment seg, long offset, int length) {
257         byte[] result = new byte[length];
258         for (int i = 0; i < length; i++) {
259             result[i] = seg.get(JAVA_BYTE, offset + i);
260         }
261         return result;
262     }
263 
264     private static void writeByteArray(MemorySegment seg, long offset, byte[] data) {
265         for (int i = 0; i < data.length; i++) {
266             seg.set(JAVA_BYTE, offset + i, data[i]);
267         }
268     }
