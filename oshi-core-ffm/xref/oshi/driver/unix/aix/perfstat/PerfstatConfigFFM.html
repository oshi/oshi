25  /**
26   * FFM-backed driver for {@code perfstat_partition_config}, mirroring
27   * {@code oshi.driver.unix.aix.perfstat.PerfstatConfigJNA}.
28   */
29  @ThreadSafe
30  public final class PerfstatConfigFFM {
31  
32      private static final Logger LOG = LoggerFactory.getLogger(PerfstatConfigFFM.class);
33  
34      private PerfstatConfigFFM() {
35      }
36  
37      /** POJO mirror of the {@code perfstat_partition_config_t} fields OSHI consumes. */
38      public static final class PartitionConfig {
39          public int conf;
40          public String machineID = "";
41          public double processorMHz;
42          public String OSBuild = "";
43          public int smtthreads;
44          public long vcpusMax;
45      }
46  
47      /**
48       * Queries {@code perfstat_partition_config}.
49       *
50       * @return populated {@link PartitionConfig}, or an empty instance on error
51       */
52      public static PartitionConfig queryConfig() {
53          return ForeignFunctions.callInArenaOrDefault(arena -> {
54              PartitionConfig result = new PartitionConfig();
55              MemorySegment buf = arena.allocate(PERFSTAT_PARTITION_CONFIG_T_SIZE);
56              int ret = perfstat_partition_config(MemorySegment.NULL, buf, PERFSTAT_PARTITION_CONFIG_T_SIZE, 1);
57              if (ret > 0) {
58                  result.conf = configConf(buf);
59                  result.machineID = configMachineID(buf);
60                  result.processorMHz = configProcessorMHz(buf);
61                  result.OSBuild = configOSBuild(buf);
62                  result.smtthreads = configSmtthreads(buf);
63                  result.vcpusMax = configVcpusMax(buf);
64              }
65              return result;
66          }, LOG, LogLevel.TRACE, "Failed to query partition config", new PartitionConfig());
67      }
68  }
