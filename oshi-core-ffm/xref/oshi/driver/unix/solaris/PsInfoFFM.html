5   package oshi.driver.unix.solaris;
6   
7   import java.util.ArrayList;
8   import java.util.LinkedHashMap;
9   import java.util.List;
10  import java.util.Map;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.util.ExecutingCommand;
14  import oshi.util.tuples.Pair;
15  
16  /**
17   * FFM-backed reader of a Solaris process's argument list and environment.
18   * <p>
19   * The {@code psinfo}/{@code lwpsinfo}/{@code usage} structures are parsed by the shared
20   * {@link oshi.driver.common.unix.solaris.PsInfo} driver. The arg/env scan that the JNA driver performs via
21   * {@code pread} on {@code /proc/<pid>/as} is replaced here with {@code pargs -e}, which is simpler and avoids needing a
22   * libc binding for {@code open}/{@code close}/{@code pread}. The functional result is the same (the same argv/envp the
23   * kernel exposed), at the cost of forking a child process per query.
24   */
25  @ThreadSafe
26  public final class PsInfoFFM {
27  
28      private PsInfoFFM() {
29      }
30  
31      /**
32       * Reads the argument list and environment via {@code pargs -e}.
33       *
34       * @param pid the process ID
35       * @return a pair of (arg list, env map). Either may be empty if {@code pargs} fails or has insufficient permission.
36       */
37      public static Pair<List<String>, Map<String, String>> queryArgsEnv(int pid) {
38          List<String> args = new ArrayList<>();
39          Map<String, String> env = new LinkedHashMap<>();
40          // `pargs -ae <pid>` emits one line per arg/env entry:
41          // argv[0]: /bin/foo
42          // envp[0]: FOO=bar
43          List<String> lines = ExecutingCommand.runNative("pargs -ae " + pid);
44          for (String line : lines) {
45              String s = line.trim();
46              if (s.startsWith("argv[")) {
47                  int colon = s.indexOf(": ");
48                  if (colon > 0 && colon + 2 < s.length()) {
49                      args.add(s.substring(colon + 2));
50                  }
51              } else if (s.startsWith("envp[")) {
52                  int colon = s.indexOf(": ");
53                  if (colon > 0 && colon + 2 < s.length()) {
54                      String entry = s.substring(colon + 2);
55                      int eq = entry.indexOf('=');
56                      if (eq > 0) {
57                          env.put(entry.substring(0, eq), entry.substring(eq + 1));
58                      }
59                  }
60              }
61          }
62          return new Pair<>(args, env);
63      }
64  }
