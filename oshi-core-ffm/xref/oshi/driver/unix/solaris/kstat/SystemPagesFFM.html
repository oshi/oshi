7   import java.lang.foreign.MemorySegment;

9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM;
11  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM.KstatChain;
12  import oshi.util.tuples.Pair;

14  /**
15   * FFM equivalent of {@code oshi.driver.unix.solaris.kstat.SystemPages}. On illumos (where {@code HAS_KSTAT2} is
16   * {@code false}), this only exercises the legacy {@link KstatUtilFFM} path.
17   */
18  @ThreadSafe
19  public final class SystemPagesFFM {
20  
21      private SystemPagesFFM() {
22      }
23  
24      /**
25       * Queries the {@code system_pages} kstat and returns available and physical memory in pages.
26       *
27       * @return a pair (available, total) in pages; multiply by page size for bytes.
28       */
29      public static Pair<Long, Long> queryAvailableTotal() {
30          // The JNA SolarisOperatingSystemJNA prefers Kstat2 on Solaris 11.4+; the FFM Kstat2 binding is a
31          // stub today, so this method always falls through to LibKstat regardless of HAS_KSTAT2. Lower
32          // fidelity is acceptable until a downstream consumer needs Solaris 11.4-specific kstat2 data.
33          long memAvailable = 0L;
34          long memTotal = 0L;
35          try (KstatChain kc = KstatUtilFFM.openChain()) {
36              MemorySegment ksp = kc.lookup(null, -1, "system_pages");
37              if (ksp.address() != 0L && kc.read(ksp)) {
38                  memAvailable = KstatUtilFFM.dataLookupLong(ksp, "availrmem");
39                  memTotal = KstatUtilFFM.dataLookupLong(ksp, "physmem");
40              }
41          }
42          return new Pair<>(memAvailable, memTotal);
43      }
44  }
