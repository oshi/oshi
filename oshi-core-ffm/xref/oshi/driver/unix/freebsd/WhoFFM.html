28  /**
29   * FFM-based utility to query logged in users on FreeBSD via {@code getutxent}.
30   */
31  @ThreadSafe
32  public final class WhoFFM {
33  
34      private WhoFFM() {
35      }
36  
37      /**
38       * Walks the utmpx database and returns the active user sessions.
39       *
40       * @return a list of logged in user sessions
41       */
42      public static synchronized List<OSSession> queryUtxent() {
43          List<OSSession> whoList = new ArrayList<>();
44          try (Arena arena = Arena.ofConfined()) {
45              setutxent();
46              try {
47                  MemorySegment utPtr;
48                  while ((utPtr = getutxent()) != null) {
49                      MemorySegment ut = utPtr.reinterpret(UTMPX_LAYOUT.byteSize(), arena, null);
50                      short type = utmpxType(ut);
51                      if (type == USER_PROCESS || type == LOGIN_PROCESS) {
52                          String user = utmpxUser(ut);
53                          String device = utmpxLine(ut);
54                          String host = utmpxHost(ut);
55                          long loginTime = utmpxLoginTime(ut);
56                          if (!isSessionValid(user, device, loginTime)) {
57                              return oshi.util.driver.unix.Who.queryWho();
58                          }
59                          whoList.add(new OSSession(user, device, loginTime, host));
60                      }
61                  }
62              } finally {
63                  endutxent();
64              }
65          } catch (Throwable e) {
66              return oshi.util.driver.unix.Who.queryWho();
67          }
68          return whoList;
69      }
70  }
