8   import static oshi.ffm.platform.mac.MacSystem.F_MNTONNAME;
9   import static oshi.ffm.platform.mac.MacSystem.MNT_NOWAIT;
10  import static oshi.ffm.platform.mac.MacSystem.STATFS;
11  import static oshi.ffm.platform.mac.MacSystemFunctions.getfsstat64;
12  
13  import java.lang.foreign.Arena;
14  import java.lang.foreign.MemorySegment;
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  import oshi.annotation.concurrent.ThreadSafe;
19  
20  /**
21   * Utility to query fsstat using FFM.
22   */
23  @ThreadSafe
24  public final class FsstatFFM {
25  
26      private FsstatFFM() {
27      }
28  
29      /**
30       * Query fsstat to map partitions to mount points.
31       *
32       * @return A map with partitions as the key and mount points as the value
33       */
34      public static Map<String, String> queryPartitionToMountMap() {
35          Map<String, String> mountPointMap = new HashMap<>();
36          try (Arena arena = Arena.ofConfined()) {
37              int numfs = getfsstat64(MemorySegment.NULL, 0, 0);
38              if (numfs <= 0) {
39                  return mountPointMap;
40              }
41              long statfsSize = STATFS.byteSize();
42              MemorySegment statfsBuffer = arena.allocate(statfsSize * numfs);
43              numfs = getfsstat64(statfsBuffer, (int) statfsBuffer.byteSize(), MNT_NOWAIT);
44              if (numfs <= 0) {
45                  return mountPointMap;
46              }
47              for (int f = 0; f < numfs; f++) {
48                  MemorySegment statfs = statfsBuffer.asSlice(f * statfsSize, statfsSize);
49                  String mntFrom = statfs.getString(STATFS.byteOffset(F_MNTFROMNAME));
50                  String mntOn = statfs.getString(STATFS.byteOffset(F_MNTONNAME));
51                  mountPointMap.put(mntFrom.replace("/dev/", ""), mntOn);
52              }
53          } catch (Throwable _) {
54              // Fall through with empty map
55          }
56          return mountPointMap;
57      }
58  }
