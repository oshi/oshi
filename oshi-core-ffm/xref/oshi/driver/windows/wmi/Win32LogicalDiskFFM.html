5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.driver.common.windows.wmi.Win32LogicalDisk;
11  import oshi.driver.common.windows.wmi.Win32LogicalDisk.LogicalDiskProperty;
12  import oshi.ffm.util.platform.windows.WbemcliUtilFFM.WmiQuery;
13  import oshi.ffm.util.platform.windows.WbemcliUtilFFM.WmiResult;
14  import oshi.ffm.util.platform.windows.WmiQueryHandlerFFM;
15  
16  /**
17   * Utility to query WMI class {@code Win32_LogicalDisk} using FFM.
18   */
19  @ThreadSafe
20  public final class Win32LogicalDiskFFM extends Win32LogicalDisk {
21  
22      private Win32LogicalDiskFFM() {
23      }
24  
25      /**
26       * Queries logical disk information
27       *
28       * @param nameToMatch an optional string to filter match, null otherwise
29       * @param localOnly   Whether to only search local drives
30       * @return Logical Disk Information
31       */
32      public static WmiResult<LogicalDiskProperty> queryLogicalDisk(String nameToMatch, boolean localOnly) {
33          WmiQuery<LogicalDiskProperty> logicalDiskQuery = new WmiQuery<>(
34                  buildWmiClassNameWithWhere(nameToMatch, localOnly), LogicalDiskProperty.class);
35          return Objects.requireNonNull(WmiQueryHandlerFFM.createInstance(),
36                  "WmiQueryHandlerFFM.createInstance() returned null").queryWMI(logicalDiskQuery);
37      }
38  }
