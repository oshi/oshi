5   package oshi.driver.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.driver.common.windows.wmi.Win32DiskDriveToDiskPartition;
9   import oshi.util.platform.windows.WbemcliUtilFFM.WmiQuery;
10  import oshi.util.platform.windows.WbemcliUtilFFM.WmiResult;
11  import oshi.util.platform.windows.WmiQueryHandlerFFM;
12  
13  /**
14   * Utility to query WMI class {@code Win32_DiskDriveToDiskPartition} using FFM.
15   */
16  @ThreadSafe
17  public final class Win32DiskDriveToDiskPartitionFFM extends Win32DiskDriveToDiskPartition {
18  
19      private Win32DiskDriveToDiskPartitionFFM() {
20      }
21  
22      /**
23       * Queries the association between disk drive and partition.
24       *
25       * @param h An instantiated {@link WmiQueryHandlerFFM}. User should have already initialized COM.
26       * @return Antecedent-dependent pairs of disk and partition.
27       */
28      public static WmiResult<DriveToPartitionProperty> queryDriveToPartition(WmiQueryHandlerFFM h) {
29          WmiQuery<DriveToPartitionProperty> driveToPartitionQuery = new WmiQuery<>(WIN32_DISK_DRIVE_TO_DISK_PARTITION,
30                  DriveToPartitionProperty.class);
31          return h.queryWMI(driveToPartitionQuery, false);
32      }
33  }
