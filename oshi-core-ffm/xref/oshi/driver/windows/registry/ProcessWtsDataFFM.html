18  /**
19   * Utility to read process data from WMI as a substitute for WTS native calls.
20   */
21  @ThreadSafe
22  public final class ProcessWtsDataFFM {
23  
24      private ProcessWtsDataFFM() {
25      }
26  
27      /**
28       * Query WMI for process performance counters.
29       *
30       * @param pids An optional collection of process IDs to filter the list to. May be null for no filtering.
31       * @return A map with Process ID as the key and a {@link WtsInfo} object populated with data.
32       */
33      public static Map<Integer, WtsInfo> queryProcessWtsMap(Collection<Integer> pids) {
34          Map<Integer, WtsInfo> wtsMap = new HashMap<>();
35          WmiResult<ProcessXPProperty> processWmiResult = Win32ProcessFFM.queryProcesses(pids);
36          for (int i = 0; i < processWmiResult.getResultCount(); i++) {
37              wtsMap.put(WmiUtilFFM.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i), new WtsInfo(
38                      WmiUtilFFM.getString(processWmiResult, ProcessXPProperty.NAME, i),
39                      WmiUtilFFM.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i),
40                      WmiUtilFFM.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i),
41                      // WMI Pagefile usage is in KB
42                      1024 * (WmiUtilFFM.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffff_ffffL),
43                      WmiUtilFFM.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10_000L,
44                      WmiUtilFFM.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10_000L,
45                      WmiUtilFFM.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i)));
46          }
47          return wtsMap;
48      }
49  }
