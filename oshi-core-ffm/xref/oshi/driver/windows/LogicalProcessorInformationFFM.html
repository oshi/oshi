35  /**
36   * FFM-based utility to query Logical Processor Information using GetLogicalProcessorInformationEx.
37   */
38  @ThreadSafe
39  public final class LogicalProcessorInformationFFM {
40  
41      private static final Logger LOG = LoggerFactory.getLogger(LogicalProcessorInformationFFM.class);
42  
43      // RelationAll = 0xffff
44      private static final int RELATION_ALL = 0xffff;
45      // Relationship types
46      private static final int RELATION_PROCESSOR_CORE = 0;
47      private static final int RELATION_NUMA_NODE = 1;
48      private static final int RELATION_CACHE = 2;
49      private static final int RELATION_PROCESSOR_PACKAGE = 3;
50  
51      private static final boolean IS_WIN10_OR_GREATER = VersionHelpersFFM.IsWindows10OrGreater();
52  
53      private LogicalProcessorInformationFFM() {
54      }
55  
56      /**
57       * Get a list of logical processors on this machine using GetLogicalProcessorInformationEx.
58       *
59       * @return A triplet of logical processors, physical processors, and processor caches
60       */
61      public static Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> getLogicalProcessorInformationEx() {
62          List<List<long[]>> packages = new ArrayList<>(); // each entry: list of [group, mask]
63          Set<ProcessorCache> caches = new HashSet<>();
64          List<long[]> cores = new ArrayList<>(); // each entry: [group, mask, efficiency]
65          List<long[]> numaNodes = new ArrayList<>(); // each entry: [nodeNumber, group, mask]
66  
67          try (Arena arena = Arena.ofConfined()) {
68              // First call to get required buffer size
69              MemorySegment returnedLength = arena.allocate(ValueLayout.JAVA_INT);
70              Kernel32FFM.GetLogicalProcessorInformationEx(RELATION_ALL, MemorySegment.NULL, returnedLength);
71              int bufferSize = returnedLength.get(ValueLayout.JAVA_INT, 0);
72              if (bufferSize == 0) {
73                  LOG.error("Failed to get buffer size for GetLogicalProcessorInformationEx");
74                  return new Triplet<>(new ArrayList<>(), new ArrayList<>(), new ArrayList<>());
75              }
76  
77              // Second call with properly sized buffer
78              MemorySegment buffer = arena.allocate(bufferSize);
79              returnedLength.set(ValueLayout.JAVA_INT, 0, bufferSize);
80              if (!Kernel32FFM.GetLogicalProcessorInformationEx(RELATION_ALL, buffer, returnedLength)) {
81                  LOG.error("Failed to get logical processor information");
82                  return new Triplet<>(new ArrayList<>(), new ArrayList<>(), new ArrayList<>());
83              }
84  
85              // Parse the buffer
86              long offset = 0;
87              while (offset < bufferSize) {
88                  int relationship = buffer.get(ValueLayout.JAVA_INT, offset);
89                  int size = buffer.get(ValueLayout.JAVA_INT, offset + 4L);
90  
91                  switch (relationship) {
92                      case RELATION_PROCESSOR_CORE:
93                          parseProcessorCore(buffer, offset, cores);
94                          break;
95                      case RELATION_NUMA_NODE:
96                          parseNumaNode(buffer, offset, numaNodes);
97                          break;
98                      case RELATION_CACHE:
99                          parseCache(buffer, offset, caches);
100                         break;
101                     case RELATION_PROCESSOR_PACKAGE:
102                         parsePackage(buffer, offset, packages);
103                         break;
104                     default:
105                         break;
106                 }
107                 offset += size;
108             }
109         }
110 
111         // Sort cores by group*64 + trailing zeros of mask
112         cores.sort(Comparator.comparing(c -> c[0] * 64L + Long.numberOfTrailingZeros(c[1])));
113         // Sort packages by first group affinity
114         packages.sort(Comparator.comparing(p -> p.get(0)[0] * 64L + Long.numberOfTrailingZeros(p.get(0)[1])));
115         // Sort numa nodes by node number
116         numaNodes.sort(Comparator.comparing(n -> n[0]));
117 
118         // Fetch processorIDs from WMI
119         Map<Integer, String> processorIdMap = new HashMap<>();
120         var processorId = Win32ProcessorFFM.queryProcessorId();
121         for (int pkg = 0; pkg < processorId.getResultCount(); pkg++) {
122             processorIdMap.put(pkg, WmiUtil.getString(processorId, ProcessorIdProperty.PROCESSORID, pkg));
123         }
124 
125         // Build logical processor list
126         List<LogicalProcessor> logProcs = new ArrayList<>();
127         Map<Integer, Integer> corePkgMap = new HashMap<>();
128         Map<Integer, String> pkgCpuidMap = new HashMap<>();
129         for (long[] node : numaNodes) {
130             int nodeNum = (int) node[0];
131             int group = (int) node[1];
132             long mask = node[2];
133             int lowBit = Long.numberOfTrailingZeros(mask);
134             int hiBit = 63 - Long.numberOfLeadingZeros(mask);
135             for (int lp = lowBit; lp <= hiBit; lp++) {
136                 if ((mask & (1L << lp)) != 0) {
137                     int coreId = getMatchingCore(cores, group, lp);
138                     int pkgId = getMatchingPackage(packages, group, lp);
139                     corePkgMap.put(coreId, pkgId);
140                     pkgCpuidMap.put(coreId, processorIdMap.getOrDefault(pkgId, ""));
141                     logProcs.add(new LogicalProcessor(lp, coreId, pkgId, nodeNum, group));
142                 }
143             }
144         }
145 
146         List<PhysicalProcessor> physProcs = new ArrayList<>();
147         for (int coreId = 0; coreId < cores.size(); coreId++) {
148             int efficiency = (int) cores.get(coreId)[2];
149             String cpuid = pkgCpuidMap.getOrDefault(coreId, "");
150             int pkgId = corePkgMap.getOrDefault(coreId, 0);
151             physProcs.add(new PhysicalProcessor(pkgId, coreId, efficiency, cpuid));
152         }
153 
154         return new Triplet<>(logProcs, physProcs, orderedProcCaches(caches));
155     }

157     private static void parseProcessorCore(MemorySegment buffer, long baseOffset, List<long[]> cores) {
158         long dataOffset = baseOffset + 8;
159         int efficiencyClass = IS_WIN10_OR_GREATER
160                 ? Byte.toUnsignedInt(buffer.get(ValueLayout.JAVA_BYTE, dataOffset + 1))
161                 : 0;
162         long mask = buffer.get(ValueLayout.JAVA_LONG, dataOffset + 24);
163         int group = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, dataOffset + 32));
164         cores.add(new long[] { group, mask, efficiencyClass });
165     }
166 
167     private static void parseNumaNode(MemorySegment buffer, long baseOffset, List<long[]> numaNodes) {
168         long dataOffset = baseOffset + 8;
169         int nodeNumber = buffer.get(ValueLayout.JAVA_INT, dataOffset);
170         int groupCount = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, dataOffset + 22));
171         if (groupCount == 0) {
172             groupCount = 1;
173         }
174         for (int i = 0; i < groupCount; i++) {
175             long affinityOffset = dataOffset + 24 + i * 16L;
176             long mask = buffer.get(ValueLayout.JAVA_LONG, affinityOffset);
177             int group = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, affinityOffset + 8));
178             numaNodes.add(new long[] { nodeNumber, group, mask });
179         }
180     }

182     private static void parseCache(MemorySegment buffer, long baseOffset, Set<ProcessorCache> caches) {
183         long dataOffset = baseOffset + 8;
184         int level = Byte.toUnsignedInt(buffer.get(ValueLayout.JAVA_BYTE, dataOffset));
185         int associativity = Byte.toUnsignedInt(buffer.get(ValueLayout.JAVA_BYTE, dataOffset + 1));
186         int lineSize = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, dataOffset + 2));
187         int cacheSize = buffer.get(ValueLayout.JAVA_INT, dataOffset + 4);
188         int type = buffer.get(ValueLayout.JAVA_INT, dataOffset + 8);
189         Type[] types = Type.values();
190         Type cacheType = (type >= 0 && type < types.length) ? types[type] : Type.UNIFIED;
191         caches.add(new ProcessorCache(level, associativity, lineSize, cacheSize, cacheType));
192     }

194     private static void parsePackage(MemorySegment buffer, long baseOffset, List<List<long[]>> packages) {
195         long dataOffset = baseOffset + 8;
196         int groupCount = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, dataOffset + 22));
197         List<long[]> groupMasks = new ArrayList<>();
198         for (int i = 0; i < groupCount; i++) {
199             long gmOffset = dataOffset + 24 + i * 16L;
200             long mask = buffer.get(ValueLayout.JAVA_LONG, gmOffset);
201             int group = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, gmOffset + 8));
202             groupMasks.add(new long[] { group, mask });
203         }
204         if (groupMasks.isEmpty()) {
205             // Fallback: read at least one
206             long mask = buffer.get(ValueLayout.JAVA_LONG, dataOffset + 24);
207             int group = Short.toUnsignedInt(buffer.get(ValueLayout.JAVA_SHORT, dataOffset + 24 + 8));
208             groupMasks.add(new long[] { group, mask });
209         }
210         packages.add(groupMasks);
211     }

213     private static int getMatchingPackage(List<List<long[]>> packages, int g, int lp) {
214         for (int i = 0; i < packages.size(); i++) {
215             for (long[] gm : packages.get(i)) {
216                 if ((gm[1] & (1L << lp)) != 0 && gm[0] == g) {
217                     return i;
218                 }
219             }
220         }
221         return 0;
222     }
223 
224     private static int getMatchingCore(List<long[]> cores, int g, int lp) {
225         for (int j = 0; j < cores.size(); j++) {
226             if ((cores.get(j)[1] & (1L << lp)) != 0 && cores.get(j)[0] == g) {
227                 return j;
228             }
229         }
230         return 0;
231     }
