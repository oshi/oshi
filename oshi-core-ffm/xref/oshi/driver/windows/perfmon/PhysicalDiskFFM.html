5   package oshi.driver.windows.perfmon;
6   
7   import static oshi.driver.common.windows.perfmon.PerfmonConstants.PHYSICAL_DISK;
8   import static oshi.driver.common.windows.perfmon.PerfmonConstants.WIN32_PERF_RAW_DATA_PERF_DISK_PHYSICAL_DISK_WHERE_NAME_NOT_TOTAL;
9   
10  import java.util.List;
11  import java.util.Map;
12  
13  import oshi.annotation.concurrent.ThreadSafe;
14  import oshi.driver.common.windows.perfmon.PhysicalDisk.PhysicalDiskProperty;
15  import oshi.util.platform.windows.PerfCounterWildcardQueryFFM;
16  import oshi.util.tuples.Pair;
17  
18  /**
19   * Utility to query PhysicalDisk performance counter using FFM.
20   */
21  @ThreadSafe
22  public final class PhysicalDiskFFM {
23  
24      private PhysicalDiskFFM() {
25      }
26  
27      /**
28       * Returns physical disk performance counters.
29       *
30       * @return Performance Counters for physical disks.
31       */
32      public static Pair<List<String>, Map<PhysicalDiskProperty, List<Long>>> queryDiskCounters() {
33          return PerfCounterWildcardQueryFFM.queryInstancesAndValues(PhysicalDiskProperty.class, PHYSICAL_DISK,
34                  WIN32_PERF_RAW_DATA_PERF_DISK_PHYSICAL_DISK_WHERE_NAME_NOT_TOTAL);
35      }
36  }
