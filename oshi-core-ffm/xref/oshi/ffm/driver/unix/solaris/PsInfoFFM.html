24  /**
25   * FFM equivalent of {@code oshi.driver.unix.solaris.PsInfo}. Parses {@code /proc/<pid>/psinfo},
26   * {@code /proc/<pid>/lwp/<tid>/lwpsinfo}, and {@code /proc/<pid>/usage} into pure-Java data classes (no JNA structs).
27   * <p>
28   * The arg/env scan that {@code oshi.driver.unix.solaris.PsInfo#queryArgsEnv} performs via {@code pread} on
29   * {@code /proc/<pid>/as} is replaced here with {@code pargs -e}, which is simpler and avoids needing a libc binding for
30   * {@code open}/{@code close}/{@code pread}. The functional result is the same (the same argv/envp the kernel exposed),
31   * at the cost of forking a child process per query.
32   */
33  @ThreadSafe
34  public final class PsInfoFFM {
35  
36      private static final Logger LOG = LoggerFactory.getLogger(PsInfoFFM.class);
37  
38      private PsInfoFFM() {
39      }

44          public int pr_nlwp;
45          public int pr_pid;
46          public int pr_ppid;
47          public int pr_pgid;

49          public int pr_uid;

51          public int pr_gid;

53          public long pr_addr; // user-space pointer as a raw address
54          public long pr_size; // Kbytes
55          public long pr_rssize;
56          public long pr_rssizepriv;
57          public long pr_ttydev;
58          public short pr_pctcpu;
59          public short pr_pctmem;
60          public Timestruc pr_start = new Timestruc();
61          public Timestruc pr_time = new Timestruc();
62          public Timestruc pr_ctime = new Timestruc();
63          public byte[] pr_fname = new byte[16];
64          public byte[] pr_psargs = new byte[80];
65          public int pr_wstat;
66          public int pr_argc;
67          public long pr_argv; // user-space pointer
68          public long pr_envp; // user-space pointer
69          public byte pr_dmodel; // 1=32-bit, 2=64-bit
70          public int pr_taskid;
71          public int pr_projid;

78          private PsInfo(ByteBuffer buff) {
79              this.pr_flag = FileUtil.readIntFromBuffer(buff);
80              this.pr_nlwp = FileUtil.readIntFromBuffer(buff);
81              this.pr_pid = FileUtil.readIntFromBuffer(buff);
82              this.pr_ppid = FileUtil.readIntFromBuffer(buff);
83              this.pr_pgid = FileUtil.readIntFromBuffer(buff);
84              this.pr_sid = FileUtil.readIntFromBuffer(buff);
85              this.pr_uid = FileUtil.readIntFromBuffer(buff);
86              this.pr_euid = FileUtil.readIntFromBuffer(buff);
87              this.pr_gid = FileUtil.readIntFromBuffer(buff);
88              this.pr_egid = FileUtil.readIntFromBuffer(buff);
89              this.pr_addr = FileUtil.readLongFromBuffer(buff);
90              this.pr_size = FileUtil.readLongFromBuffer(buff);
91              this.pr_rssize = FileUtil.readLongFromBuffer(buff);
92              this.pr_rssizepriv = FileUtil.readLongFromBuffer(buff);
93              this.pr_ttydev = FileUtil.readLongFromBuffer(buff);
94              this.pr_pctcpu = FileUtil.readShortFromBuffer(buff);
95              this.pr_pctmem = FileUtil.readShortFromBuffer(buff);
96              // Align to 8 bytes for following Timestrucs (pad 4 bytes after the two shorts)
97              FileUtil.readIntFromBuffer(buff);
98              this.pr_start.readFrom(buff);
99              this.pr_time.readFrom(buff);
100             this.pr_ctime.readFrom(buff);
101             FileUtil.readByteArrayFromBuffer(buff, this.pr_fname);
102             FileUtil.readByteArrayFromBuffer(buff, this.pr_psargs);
103             this.pr_wstat = FileUtil.readIntFromBuffer(buff);
104             this.pr_argc = FileUtil.readIntFromBuffer(buff);
105             this.pr_argv = FileUtil.readLongFromBuffer(buff);
106             this.pr_envp = FileUtil.readLongFromBuffer(buff);
107             this.pr_dmodel = FileUtil.readByteFromBuffer(buff);
108             // 3-byte pad (pr_pad2)
109             FileUtil.readByteFromBuffer(buff);
110             FileUtil.readByteFromBuffer(buff);
111             FileUtil.readByteFromBuffer(buff);
112             this.pr_taskid = FileUtil.readIntFromBuffer(buff);
113             this.pr_projid = FileUtil.readIntFromBuffer(buff);
114             this.pr_nzomb = FileUtil.readIntFromBuffer(buff);
115             this.pr_poolid = FileUtil.readIntFromBuffer(buff);
116             this.pr_zoneid = FileUtil.readIntFromBuffer(buff);
117             this.pr_contract = FileUtil.readIntFromBuffer(buff);
118             FileUtil.readIntFromBuffer(buff); // pr_filler
119             this.pr_lwp.readFrom(buff);
120         }
121     }

236             FileUtil.readLongFromBuffer(buff); // pr_sigs
237             this.pr_vctx = FileUtil.readLongFromBuffer(buff);
238             this.pr_ictx = FileUtil.readLongFromBuffer(buff);
239             // pr_sysc, pr_ioch, 10 fillers — ignored
240         }

243     /** Timestruc: tv_sec + tv_nsec, each 8 bytes on 64-bit Solaris. */
244     public static final class Timestruc {
245         public long tv_sec;
246         public long tv_nsec;
247 
248         void readFrom(ByteBuffer buff) {
249             this.tv_sec = FileUtil.readLongFromBuffer(buff);
250             this.tv_nsec = FileUtil.readLongFromBuffer(buff);
251         }
252 
253         /**
254          * Returns this timestamp converted to milliseconds since the epoch.
255          *
256          * @return the timestamp in milliseconds
257          */
258         public long toMillis() {
259             return tv_sec * 1000L + tv_nsec / 1_000_000L;
260         }

263     /**
264      * Reads {@code /proc/<pid>/psinfo}.
265      *
266      * @param pid the process ID
267      * @return populated {@link PsInfo}, or {@code null} if the file isn't readable
268      */
269     public static PsInfo queryPsInfo(int pid) {
270         String path = String.format(Locale.ROOT, "/proc/%d/psinfo", pid);
271         ByteBuffer buff = FileUtil.readAllBytesAsBuffer(path);
272         if (buff == null || buff.remaining() == 0) {
273             // Short-lived processes (e.g. test forks) commonly disappear from /proc
274             // between enumeration and read. Logging at debug avoids CI noise.
275             LOG.debug("psinfo file empty or unreadable for pid {} ({})", pid, path);
276             return null;
277         }
278         int sz = buff.remaining();
279         try {
280             return new PsInfo(buff);
281         } catch (Throwable t) {
282             LOG.warn("Failed to parse psinfo for pid {} (file size {} bytes)", pid, sz, t);
283             return null;
284         }
285     }

347     /**
348      * Reads the argument list and environment via {@code pargs -e}. The JNA driver uses {@code pread} on
349      * {@code /proc/<pid>/as} directly; the FFM driver substitutes a {@code pargs} invocation for simplicity.
350      *
351      * @param pid the process ID
352      * @return a pair of (arg list, env map). Either may be empty if {@code pargs} fails or has insufficient permission.
353      */
354     public static Pair<List<String>, Map<String, String>> queryArgsEnv(int pid) {
355         List<String> args = new ArrayList<>();
356         Map<String, String> env = new LinkedHashMap<>();
357         // `pargs -ae <pid>` emits one line per arg/env entry:
358         // argv[0]: /bin/foo
359         // envp[0]: FOO=bar
360         List<String> lines = ExecutingCommand.runNative("pargs -ae " + pid);
361         for (String line : lines) {
362             String s = line.trim();
363             if (s.startsWith("argv[")) {
364                 int colon = s.indexOf(": ");
365                 if (colon > 0 && colon + 2 < s.length()) {
366                     args.add(s.substring(colon + 2));
367                 }
368             } else if (s.startsWith("envp[")) {
369                 int colon = s.indexOf(": ");
370                 if (colon > 0 && colon + 2 < s.length()) {
371                     String entry = s.substring(colon + 2);
372                     int eq = entry.indexOf('=');
373                     if (eq > 0) {
374                         env.put(entry.substring(0, eq), entry.substring(eq + 1));
375                     }
376                 }
377             }
378         }
379         return new Pair<>(args, env);
380     }

382     /**
383      * Trims a NUL-terminated byte array to a UTF-8 String.
384      *
385      * @param bytes the buffer
386      * @return the trimmed String
387      */
388     public static String bytesToString(byte[] bytes) {
389         int len = 0;
390         while (len < bytes.length && bytes[len] != 0) {
391             len++;
392         }
393         return new String(Arrays.copyOf(bytes, len), java.nio.charset.StandardCharsets.UTF_8);
394     }
