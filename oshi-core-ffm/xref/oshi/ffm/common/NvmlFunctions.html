40      public static final int NVML_CLOCK_GRAPHICS = 0;

42      public static final int NVML_CLOCK_MEM = 2;

46      // nvmlUtilization_t: { unsigned int gpu; unsigned int memory; }
47      /** Layout for nvmlUtilization_t struct. */
48      public static final StructLayout UTILIZATION_LAYOUT = MemoryLayout.structLayout(JAVA_INT.withName("gpu"),
49              JAVA_INT.withName("memory"));
50  
51      // nvmlMemory_t: { unsigned long long total; unsigned long long free; unsigned long long used; }
52      /** Layout for nvmlMemory_t struct. */
53      public static final StructLayout MEMORY_LAYOUT = MemoryLayout.structLayout(JAVA_LONG.withName("total"),
54              JAVA_LONG.withName("free"), JAVA_LONG.withName("used"));
55  
56      // nvmlPciInfo_t: { char busIdLegacy[16]; unsigned int domain, bus, device, pciDeviceId, pciSubSystemId;
57      // char busId[32]; }
58      /** Layout for nvmlPciInfo_t struct. */
59      public static final StructLayout PCI_INFO_LAYOUT = MemoryLayout.structLayout(
60              MemoryLayout.sequenceLayout(16, JAVA_BYTE).withName("busIdLegacy"), JAVA_INT.withName("domain"),
61              JAVA_INT.withName("bus"), JAVA_INT.withName("device"), JAVA_INT.withName("pciDeviceId"),
62              JAVA_INT.withName("pciSubSystemId"), MemoryLayout.sequenceLayout(32, JAVA_BYTE).withName("busId"));

64      // Method handles
65      private static final MethodHandle nvmlInit_v2;
66      private static final MethodHandle nvmlShutdown;
67      private static final MethodHandle nvmlDeviceGetCount_v2;
68      private static final MethodHandle nvmlDeviceGetHandleByIndex_v2;
69      private static final MethodHandle nvmlDeviceGetName;
70      private static final MethodHandle nvmlDeviceGetPciInfo_v3;
71      private static final MethodHandle nvmlDeviceGetUtilizationRates;
72      private static final MethodHandle nvmlDeviceGetMemoryInfo;
73      private static final MethodHandle nvmlDeviceGetTemperature;
74      private static final MethodHandle nvmlDeviceGetPowerUsage;
75      private static final MethodHandle nvmlDeviceGetClockInfo;
76      private static final MethodHandle nvmlDeviceGetFanSpeed;
77  
78      private static final boolean AVAILABLE;
79  
80      static {
81          boolean available = false;
82          MethodHandle hInit = null;
83          MethodHandle hShutdown = null;
84          MethodHandle hGetCount = null;
85          MethodHandle hGetHandle = null;
86          MethodHandle hGetName = null;
87          MethodHandle hGetPci = null;
88          MethodHandle hGetUtil = null;
89          MethodHandle hGetMem = null;
90          MethodHandle hGetTemp = null;
91          MethodHandle hGetPower = null;
92          MethodHandle hGetClock = null;
93          MethodHandle hGetFan = null;
94          try {
95              String libName = System.getProperty("os.name", "").toLowerCase(java.util.Locale.ROOT).contains("win")
96                      ? "nvml"
97                      : "nvidia-ml";
98              SymbolLookup nvml = libraryLookup(libName);
99  
100             hInit = LINKER.downcallHandle(nvml.findOrThrow("nvmlInit_v2"), FunctionDescriptor.of(JAVA_INT));
101             hShutdown = LINKER.downcallHandle(nvml.findOrThrow("nvmlShutdown"), FunctionDescriptor.of(JAVA_INT));
102             hGetCount = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetCount_v2"),
103                     FunctionDescriptor.of(JAVA_INT, ADDRESS));
104             hGetHandle = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetHandleByIndex_v2"),
105                     FunctionDescriptor.of(JAVA_INT, JAVA_INT, ADDRESS));
106             hGetName = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetName"),
107                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS, JAVA_INT));
108             hGetPci = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetPciInfo_v3"),
109                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
110             hGetUtil = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetUtilizationRates"),
111                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
112             hGetMem = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetMemoryInfo"),
113                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
114             hGetTemp = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetTemperature"),
115                     FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS));
116             hGetPower = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetPowerUsage"),
117                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
118             hGetClock = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetClockInfo"),
119                     FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS));
120             hGetFan = LINKER.downcallHandle(nvml.findOrThrow("nvmlDeviceGetFanSpeed"),
121                     FunctionDescriptor.of(JAVA_INT, ADDRESS, ADDRESS));
122             available = true;
123             LOG.debug("NVML library loaded via FFM");
124         } catch (Throwable t) {
125             LOG.debug("NVML library not available via FFM: {}", t.getMessage());
126         }
127         nvmlInit_v2 = hInit;
128         nvmlShutdown = hShutdown;
129         nvmlDeviceGetCount_v2 = hGetCount;
130         nvmlDeviceGetHandleByIndex_v2 = hGetHandle;
131         nvmlDeviceGetName = hGetName;
132         nvmlDeviceGetPciInfo_v3 = hGetPci;
133         nvmlDeviceGetUtilizationRates = hGetUtil;
134         nvmlDeviceGetMemoryInfo = hGetMem;
135         nvmlDeviceGetTemperature = hGetTemp;
136         nvmlDeviceGetPowerUsage = hGetPower;
137         nvmlDeviceGetClockInfo = hGetClock;
138         nvmlDeviceGetFanSpeed = hGetFan;
139         AVAILABLE = available;
140     }

331     /**
332      * Reads a null-terminated string from a byte sequence within a struct segment.
333      *
334      * @param seg    the struct segment
335      * @param offset byte offset of the char array field
336      * @param maxLen maximum length of the char array
337      * @return the string, or empty if null/empty
338      */
339     public static String readString(MemorySegment seg, long offset, int maxLen) {
340         for (int i = 0; i < maxLen; i++) {
341             if (seg.get(JAVA_BYTE, offset + i) == 0) {
342                 if (i == 0) {
343                     return "";
344                 }
345                 return seg.getString(offset);
346             }
347         }
348         byte[] bytes = new byte[maxLen];
349         MemorySegment.copy(seg, JAVA_BYTE, offset, bytes, 0, maxLen);
350         return new String(bytes, java.nio.charset.StandardCharsets.UTF_8).trim();
351     }
