23  /**
24   * FFM bindings for OpenBSD libc functions used by OSHI.
25   * <p>
26   * OpenBSD uses {@code sysctl(int *name, u_int namelen, ...)} exclusively (no {@code sysctlbyname}). Additional bindings
27   * ({@code getthrid}, {@code getrlimit}) support process and resource limit queries.
28   */
29  public final class OpenBsdLibcFunctions extends ForeignFunctions {
30  
31      private OpenBsdLibcFunctions() {
32      }
33  
34      /** Layout of the C {@code size_t} type on OpenBSD (8 bytes on all supported 64-bit archs). */
35      public static final ValueLayout.OfLong SIZE_T = ValueLayout.JAVA_LONG;
36  
37      /** {@code getrlimit} resource: maximum number of open file descriptors. OpenBSD value (8). */
38      public static final int RLIMIT_NOFILE = 8;
39  
40      /**
41       * Layout of OpenBSD's {@code struct rlimit}: two {@code rlim_t} (LP64 long) fields.
42       */
43      public static final StructLayout RLIMIT_LAYOUT = MemoryLayout.structLayout(JAVA_LONG.withName("rlim_cur"),
44              JAVA_LONG.withName("rlim_max"));
45  
46      private static final VarHandle RLIMIT_CUR = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_cur"));
47      private static final VarHandle RLIMIT_MAX = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_max"));
48  
49      // libc is already loaded into the JVM process; defaultLookup() avoids versioning pitfalls.
50      private static final SymbolLookup LIBC = LINKER.defaultLookup();
51  
52      // int sysctl(int *name, u_int namelen, void *oldp, size_t *oldlenp, void *newp, size_t newlen);
53      private static final MethodHandle sysctl = LINKER.downcallHandle(LIBC.findOrThrow("sysctl"),
54              FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT, ADDRESS, ADDRESS, ADDRESS, SIZE_T), CAPTURE_CALL_STATE);
55  
56      /**
57       * Calls {@code sysctl(name, namelen, oldp, oldlenp, newp, newlen)} with errno capture.
58       *
59       * @param callState segment allocated with {@link ForeignFunctions#CAPTURED_STATE_LAYOUT} for errno capture
60       * @param name      MIB array segment (sequence of ints)
61       * @param namelen   number of ints in {@code name}
62       * @param oldp      output buffer for the current value, or {@link MemorySegment#NULL} to query size only
63       * @param oldlenp   pointer to a {@code size_t}: on input, the size of {@code oldp}; on output, the bytes written
64       * @param newp      new value buffer, or {@link MemorySegment#NULL} when reading
65       * @param newlen    size of {@code newp} in bytes, or 0 when reading
66       * @return 0 on success, -1 on error
67       * @throws Throwable on FFM invocation error
68       */
69      public static int sysctl(MemorySegment callState, MemorySegment name, int namelen, MemorySegment oldp,
70              MemorySegment oldlenp, MemorySegment newp, long newlen) throws Throwable {
71          return (int) sysctl.invokeExact(callState, name, namelen, oldp, oldlenp, newp, newlen);
72      }
73  
74      // int getthrid(void);
75      private static final MethodHandle getthrid = LINKER.downcallHandle(LIBC.findOrThrow("getthrid"),
76              FunctionDescriptor.of(JAVA_INT));
77  
78      /**
79       * Calls {@code getthrid()} — returns the thread ID of the calling thread.
80       *
81       * @return the thread ID of the calling thread
82       * @throws Throwable on FFM invocation error
83       */
84      public static int getthrid() throws Throwable {
85          return (int) getthrid.invokeExact();
86      }
87  
88      // pid_t getpid(void);
89      private static final MethodHandle getpid = LINKER.downcallHandle(LIBC.findOrThrow("getpid"),
90              FunctionDescriptor.of(JAVA_INT));
91  
92      /**
93       * Calls {@code getpid()}.
94       *
95       * @return the process ID of the calling process
96       * @throws Throwable on FFM invocation error
97       */
98      public static int getpid() throws Throwable {
99          return (int) getpid.invokeExact();
100     }
101 
102     // int getrlimit(int resource, struct rlimit *rlim);
103     private static final MethodHandle getrlimit = LINKER.downcallHandle(LIBC.findOrThrow("getrlimit"),
104             FunctionDescriptor.of(JAVA_INT, JAVA_INT, ADDRESS));
105 
106     /**
107      * Calls {@code getrlimit(resource, rlim)}.
108      *
109      * @param resource resource constant (e.g. {@link #RLIMIT_NOFILE})
110      * @param rlim     segment allocated with {@link #RLIMIT_LAYOUT}
111      * @return 0 on success, -1 on error
112      * @throws Throwable on FFM invocation error
113      */
114     public static int getrlimit(int resource, MemorySegment rlim) throws Throwable {
115         return (int) getrlimit.invokeExact(resource, rlim);
116     }
117 
118     /**
119      * Reads {@code rlim_cur} from an rlimit segment populated by {@link #getrlimit(int, MemorySegment)}.
120      *
121      * @param rlim segment allocated with {@link #RLIMIT_LAYOUT}
122      * @return the soft resource limit
123      */
124     public static long rlimitCur(MemorySegment rlim) {
125         return (long) RLIMIT_CUR.get(rlim, 0L);
126     }
127 
128     /**
129      * Reads {@code rlim_max} from an rlimit segment populated by {@link #getrlimit(int, MemorySegment)}.
130      *
131      * @param rlim segment allocated with {@link #RLIMIT_LAYOUT}
132      * @return the hard resource limit
133      */
134     public static long rlimitMax(MemorySegment rlim) {
135         return (long) RLIMIT_MAX.get(rlim, 0L);
136     }

172     // --- Bcachestats struct layout (18 int64_t fields) ---
173 
174     public static final StructLayout BCACHESTATS_LAYOUT = MemoryLayout.structLayout(JAVA_LONG.withName("numbufs"),
175             JAVA_LONG.withName("numbufpages"), JAVA_LONG.withName("numdirtypages"), JAVA_LONG.withName("numcleanpages"),
176             JAVA_LONG.withName("pendingwrites"), JAVA_LONG.withName("pendingreads"), JAVA_LONG.withName("numwrites"),
177             JAVA_LONG.withName("numreads"), JAVA_LONG.withName("cachehits"), JAVA_LONG.withName("busymapped"),
178             JAVA_LONG.withName("dmapages"), JAVA_LONG.withName("highpages"), JAVA_LONG.withName("delwribufs"),
179             JAVA_LONG.withName("kvaslots"), JAVA_LONG.withName("kvaslots_avail"), JAVA_LONG.withName("highflips"),
180             JAVA_LONG.withName("highflops"), JAVA_LONG.withName("dmaflips"));
181 
182     private static final VarHandle BCACHESTATS_NUMBUFPAGES = BCACHESTATS_LAYOUT
183             .varHandle(PathElement.groupElement("numbufpages"));
184 
185     /**
186      * Reads the {@code numbufpages} field from a Bcachestats segment.
187      *
188      * @param seg segment allocated with {@link #BCACHESTATS_LAYOUT}
189      * @return the number of pages in the buffer cache
190      */
191     public static long bcachestatsNumbufpages(MemorySegment seg) {
192         return (long) BCACHESTATS_NUMBUFPAGES.get(seg, 0L);
193     }

196     private static final MethodHandle getloadavg = LINKER.downcallHandle(LIBC.findOrThrow("getloadavg"),
197             FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT));

199     /**
200      * Calls {@code getloadavg(loadavg, nelem)}.
201      *
202      * @param loadavg pre-allocated segment of {@code nelem} doubles
203      * @param nelem   number of load average values to retrieve (1-3)
204      * @return number of samples set, or -1 on error
205      * @throws Throwable on FFM invocation error
206      */
207     public static int getloadavg(MemorySegment loadavg, int nelem) throws Throwable {
208         return (int) getloadavg.invokeExact(loadavg, nelem);
209     }
