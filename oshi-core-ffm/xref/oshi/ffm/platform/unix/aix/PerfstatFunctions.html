21  /**
22   * FFM bindings for the AIX {@code libperfstat} library, mapped to the surface OSHI actually consumes.
23   * <p>
24   * Struct sizes and field offsets are taken from JNA's {@code com.sun.jna.platform.unix.aix.Perfstat} (which has been
25   * reviewed against AIX's {@code <libperfstat.h>}); see the JNA file for full field documentation.
26   * <p>
27   * The library lives at {@code /usr/lib/libperfstat.a(shr_64.o)} (or {@code (shr.o)} for the 32-bit variant). Loading
28   * requires the AIX-specific {@code RTLD_MEMBER} flag, which FFM's {@link SymbolLookup#libraryLookup} can't pass; the
29   * dlopen call is wrapped behind {@link SharedObjectLoader}, mirroring JNA's contrib/platform
30   * {@code SharedObjectLoader}.
31   */
32  public final class PerfstatFunctions extends ForeignFunctions {

119     /**
120      * Copy {@code IDENTIFIER_LENGTH} bytes from {@code seg + offset} into a Java string, stopping at the first NUL.
121      *
122      * @param seg    a perfstat struct segment
123      * @param offset byte offset of the leading-name field within {@code seg}
124      * @return the decoded ASCII name
125      */
126     public static String readName(MemorySegment seg, long offset) {
127         byte[] bytes = new byte[IDENTIFIER_LENGTH];
128         for (int i = 0; i < IDENTIFIER_LENGTH; i++) {
129             byte b = seg.get(JAVA_BYTE, offset + i);
130             if (b == 0) {
131                 return new String(bytes, 0, i, java.nio.charset.StandardCharsets.US_ASCII);
132             }
133             bytes[i] = b;
134         }
135         return new String(bytes, java.nio.charset.StandardCharsets.US_ASCII);
136     }
