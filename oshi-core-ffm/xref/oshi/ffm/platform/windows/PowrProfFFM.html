5   package oshi.ffm.platform.windows;
6   
7   import static java.lang.foreign.MemoryLayout.sequenceLayout;
8   import static java.lang.foreign.MemoryLayout.structLayout;
9   import static java.lang.foreign.ValueLayout.JAVA_BYTE;
10  import static java.lang.foreign.ValueLayout.JAVA_INT;
11  import static java.lang.foreign.ValueLayout.JAVA_SHORT;
12  
13  import java.lang.foreign.MemoryLayout;
14  import java.lang.foreign.StructLayout;
15  
16  /**
17   * FFM mappings for Windows battery structures from PowrProf / batclass.h.
18   */
19  public final class PowrProfFFM extends WindowsForeignFunctions {
20  
21      // BATTERY_QUERY_INFORMATION: BatteryTag (DWORD) + InformationLevel (DWORD) + AtRate (LONG)
22      public static final StructLayout BATTERY_QUERY_INFORMATION = structLayout(JAVA_INT.withName("BatteryTag"),
23              JAVA_INT.withName("InformationLevel"), JAVA_INT.withName("AtRate"));
24  
25      // BATTERY_INFORMATION
26      public static final StructLayout BATTERY_INFORMATION = structLayout(JAVA_INT.withName("Capabilities"),
27              JAVA_BYTE.withName("Technology"), sequenceLayout(3, JAVA_BYTE).withName("Reserved"),
28              sequenceLayout(4, JAVA_BYTE).withName("Chemistry"), JAVA_INT.withName("DesignedCapacity"),
29              JAVA_INT.withName("FullChargedCapacity"), JAVA_INT.withName("DefaultAlert1"),
30              JAVA_INT.withName("DefaultAlert2"), JAVA_INT.withName("CriticalBias"), JAVA_INT.withName("CycleCount"));
31  
32      // BATTERY_WAIT_STATUS
33      public static final StructLayout BATTERY_WAIT_STATUS = structLayout(JAVA_INT.withName("BatteryTag"),
34              JAVA_INT.withName("Timeout"), JAVA_INT.withName("PowerState"), JAVA_INT.withName("LowCapacity"),
35              JAVA_INT.withName("HighCapacity"));
36  
37      // BATTERY_STATUS
38      public static final StructLayout BATTERY_STATUS = structLayout(JAVA_INT.withName("PowerState"),
39              JAVA_INT.withName("Capacity"), JAVA_INT.withName("Voltage"), JAVA_INT.withName("Rate"));
40  
41      // BATTERY_MANUFACTURE_DATE: Day (BYTE) + Month (BYTE) + Year (WORD)
42      public static final StructLayout BATTERY_MANUFACTURE_DATE = structLayout(JAVA_BYTE.withName("Day"),
43              JAVA_BYTE.withName("Month"), JAVA_SHORT.withName("Year"));
44  
45      public static final long OFF_BQI_TAG = offset(BATTERY_QUERY_INFORMATION, "BatteryTag");
46      public static final long OFF_BQI_LEVEL = offset(BATTERY_QUERY_INFORMATION, "InformationLevel");
47      public static final long OFF_BQI_ATRATE = offset(BATTERY_QUERY_INFORMATION, "AtRate");
48      public static final long OFF_BI_CAPABILITIES = offset(BATTERY_INFORMATION, "Capabilities");
49      public static final long OFF_BI_CHEMISTRY = offset(BATTERY_INFORMATION, "Chemistry");
50      public static final long OFF_BI_DESIGNED = offset(BATTERY_INFORMATION, "DesignedCapacity");
51      public static final long OFF_BI_FULL = offset(BATTERY_INFORMATION, "FullChargedCapacity");
52      public static final long OFF_BI_CYCLE = offset(BATTERY_INFORMATION, "CycleCount");
53      public static final long OFF_BWS_TAG = offset(BATTERY_WAIT_STATUS, "BatteryTag");
54      public static final long OFF_BS_POWERSTATE = offset(BATTERY_STATUS, "PowerState");
55      public static final long OFF_BS_CAPACITY = offset(BATTERY_STATUS, "Capacity");
56      public static final long OFF_BS_VOLTAGE = offset(BATTERY_STATUS, "Voltage");
57      public static final long OFF_BS_RATE = offset(BATTERY_STATUS, "Rate");
58      public static final long OFF_BMD_DAY = offset(BATTERY_MANUFACTURE_DATE, "Day");
59      public static final long OFF_BMD_MONTH = offset(BATTERY_MANUFACTURE_DATE, "Month");
60      public static final long OFF_BMD_YEAR = offset(BATTERY_MANUFACTURE_DATE, "Year");
61  
62      private static long offset(StructLayout layout, String name) {
63          return layout.byteOffset(MemoryLayout.PathElement.groupElement(name));
64      }
65  }
