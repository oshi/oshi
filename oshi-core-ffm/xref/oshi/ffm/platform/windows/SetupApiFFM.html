34       * SP_DEVICE_INTERFACE_DATA: cbSize (DWORD) + InterfaceClassGuid (GUID=16 bytes) + Flags (DWORD) + Reserved

38              MemoryLayout.sequenceLayout(16, JAVA_BYTE).withName("InterfaceClassGuid"), JAVA_INT.withName("Flags"),
39              ADDRESS.withName("Reserved"));

44      private static final MethodHandle SetupDiGetClassDevs = downcall(SETUPAPI, "SetupDiGetClassDevsW", ADDRESS, ADDRESS,
45              ADDRESS, ADDRESS, JAVA_INT);
46  
47      public static Optional<MemorySegment> SetupDiGetClassDevs(MemorySegment classGuid, int flags) {
48          try {
49              MemorySegment handle = (MemorySegment) SetupDiGetClassDevs.invokeExact(classGuid, MemorySegment.NULL,
50                      MemorySegment.NULL, flags);
51              if (Kernel32FFM.isInvalidHandle(handle)) {
52                  return Optional.empty();
53              }
54              return Optional.of(handle);
55          } catch (Throwable t) {
56              LOG.debug("SetupApiFFM.SetupDiGetClassDevs failed", t);
57              return Optional.empty();
58          }
59      }

61      private static final MethodHandle SetupDiEnumDeviceInterfaces = downcall(SETUPAPI, "SetupDiEnumDeviceInterfaces",
62              JAVA_INT, ADDRESS, ADDRESS, ADDRESS, JAVA_INT, ADDRESS);
63  
64      /**
65       * @param hDevInfo            the device info set handle
66       * @param classGuid           the interface class GUID
67       * @param memberIndex         zero-based index of the interface to retrieve
68       * @param deviceInterfaceData receives information about the device interface
69       * @return 0 if no more items, 1 if success, -1 on other error
70       */
71      public static int SetupDiEnumDeviceInterfaces(MemorySegment hDevInfo, MemorySegment classGuid, int memberIndex,
72              MemorySegment deviceInterfaceData) {
73          try {
74              int result = (int) SetupDiEnumDeviceInterfaces.invokeExact(hDevInfo, MemorySegment.NULL, classGuid,
75                      memberIndex, deviceInterfaceData);
76              if (isSuccess(result)) {
77                  return 1;
78              }
79              int err = Kernel32FFM.GetLastError().orElse(0);
80              return err == ERROR_NO_MORE_ITEMS ? 0 : -1;
81          } catch (Throwable t) {
82              LOG.debug("SetupApiFFM.SetupDiEnumDeviceInterfaces failed", t);
83              return -1;
84          }
85      }

87      private static final MethodHandle SetupDiGetDeviceInterfaceDetail = downcall(SETUPAPI,
88              "SetupDiGetDeviceInterfaceDetailW", JAVA_INT, ADDRESS, ADDRESS, ADDRESS, JAVA_INT, ADDRESS, ADDRESS);
89  
90      /**
91       * First call with null detail buffer to get required size.
92       *
93       * @param hDevInfo            the device info set handle
94       * @param deviceInterfaceData the device interface data
95       * @param arena               the arena to allocate the size buffer from
96       * @return required buffer size, or 0 on unexpected error
97       */
98      public static int SetupDiGetDeviceInterfaceDetailSize(MemorySegment hDevInfo, MemorySegment deviceInterfaceData,
99              Arena arena) {
100         try {
101             MemorySegment requiredSize = arena.allocate(JAVA_INT);
102             SetupDiGetDeviceInterfaceDetail.invokeExact(hDevInfo, deviceInterfaceData, MemorySegment.NULL, 0,
103                     requiredSize, MemorySegment.NULL);
104             int err = Kernel32FFM.GetLastError().orElse(0);
105             if (err == ERROR_INSUFFICIENT_BUFFER) {
106                 return requiredSize.get(JAVA_INT, 0);
107             }
108             return 0;
109         } catch (Throwable t) {
110             LOG.debug("SetupApiFFM.SetupDiGetDeviceInterfaceDetailSize failed", t);
111             return 0;
112         }
113     }
114 
115     /**
116      * Second call with allocated buffer to get device path. The detail buffer layout is: cbSize (DWORD=4) + DevicePath
117      * (WCHAR[]). On 64-bit, cbSize field value must be 8; on 32-bit it is 5 or 6.
118      *
119      * @param hDevInfo            the device info set handle
120      * @param deviceInterfaceData the device interface data
121      * @param requiredSize        the buffer size returned by {@link #SetupDiGetDeviceInterfaceDetailSize}
122      * @param arena               the arena to allocate buffers from
123      * @return the device path string, or empty if failed
124      */
125     public static Optional<String> SetupDiGetDeviceInterfaceDetail(MemorySegment hDevInfo,
126             MemorySegment deviceInterfaceData, int requiredSize, Arena arena) {
127         try {
128             MemorySegment detail = arena.allocate(requiredSize);
129             detail.set(JAVA_INT, 0, is64Bit() ? 8 : 6);
130             MemorySegment reqSize = arena.allocate(JAVA_INT);
131             int result = (int) SetupDiGetDeviceInterfaceDetail.invokeExact(hDevInfo, deviceInterfaceData, detail,
132                     requiredSize, reqSize, MemorySegment.NULL);
133             if (!isSuccess(result)) {
134                 return Optional.empty();
135             }
136             return Optional.of(readWideString(detail.asSlice(4)));
137         } catch (Throwable t) {
138             LOG.debug("SetupApiFFM.SetupDiGetDeviceInterfaceDetail failed", t);
139             return Optional.empty();
140         }
141     }

185             ADDRESS, ADDRESS, JAVA_INT, JAVA_INT, JAVA_INT, JAVA_INT);

187     /**
188      * Opens a registry key for device-specific configuration information.
189      *
190      * @param hDevInfo    the device info set handle
191      * @param devInfoData the device info data
192      * @param scope       DICS_FLAG_GLOBAL or DICS_FLAG_CONFIGSPECIFIC
193      * @param hwProfile   hardware profile (0 for current)
194      * @param keyType     DIREG_DEV or DIREG_DRV
195      * @param samDesired  registry access mask (e.g. KEY_QUERY_VALUE)
196      * @return registry key handle, or null if failed
197      */
198     public static MemorySegment SetupDiOpenDevRegKey(MemorySegment hDevInfo, MemorySegment devInfoData, int scope,
199             int hwProfile, int keyType, int samDesired) {
200         try {
201             MemorySegment key = (MemorySegment) SetupDiOpenDevRegKey.invokeExact(hDevInfo, devInfoData, scope,
202                     hwProfile, keyType, samDesired);
203             if (Kernel32FFM.isInvalidHandle(key)) {
204                 return null;
205             }
206             return key;
207         } catch (Throwable t) {
208             LOG.debug("SetupApiFFM.SetupDiOpenDevRegKey failed", t);
209             return null;
210         }
