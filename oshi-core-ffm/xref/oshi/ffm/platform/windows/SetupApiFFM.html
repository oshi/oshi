39       * SP_DEVICE_INTERFACE_DATA: cbSize (DWORD) + InterfaceClassGuid (GUID=16 bytes) + Flags (DWORD) + Reserved

49      private static final MethodHandle SetupDiGetClassDevs = downcall(SETUPAPI, "SetupDiGetClassDevsW", ADDRESS, ADDRESS,
50              ADDRESS, ADDRESS, JAVA_INT);
51  
52      public static Optional<MemorySegment> SetupDiGetClassDevs(MemorySegment classGuid, int flags) {
53          return getOptional(() -> {
54              MemorySegment handle = (MemorySegment) SetupDiGetClassDevs.invokeExact(classGuid, MemorySegment.NULL,
55                      MemorySegment.NULL, flags);
56              return Kernel32FFM.isInvalidHandle(handle) ? null : handle;
57          }, LOG, "SetupApiFFM.SetupDiGetClassDevs failed: {}");
58      }

60      private static final MethodHandle SetupDiEnumDeviceInterfaces = downcall(SETUPAPI, "SetupDiEnumDeviceInterfaces",
61              JAVA_INT, ADDRESS, ADDRESS, ADDRESS, JAVA_INT, ADDRESS);
62  
63      /**
64       * @param hDevInfo            the device info set handle
65       * @param classGuid           the interface class GUID
66       * @param memberIndex         zero-based index of the interface to retrieve
67       * @param deviceInterfaceData receives information about the device interface
68       * @return 0 if no more items, 1 if success, -1 on other error
69       */
70      public static int SetupDiEnumDeviceInterfaces(MemorySegment hDevInfo, MemorySegment classGuid, int memberIndex,
71              MemorySegment deviceInterfaceData) {
72          return getIntOrDefault(() -> {
73              int result = (int) SetupDiEnumDeviceInterfaces.invokeExact(hDevInfo, MemorySegment.NULL, classGuid,
74                      memberIndex, deviceInterfaceData);
75              if (isSuccess(result)) {
76                  return 1;
77              }
78              int err = Kernel32FFM.GetLastError().orElse(0);
79              return err == ERROR_NO_MORE_ITEMS ? 0 : -1;
80          }, -1, LOG, "SetupApiFFM.SetupDiEnumDeviceInterfaces failed: {}");
81      }

83      private static final MethodHandle SetupDiGetDeviceInterfaceDetail = downcall(SETUPAPI,
84              "SetupDiGetDeviceInterfaceDetailW", JAVA_INT, ADDRESS, ADDRESS, ADDRESS, JAVA_INT, ADDRESS, ADDRESS);
85  
86      /**
87       * First call with null detail buffer to get required size.
88       *
89       * @param hDevInfo            the device info set handle
90       * @param deviceInterfaceData the device interface data
91       * @param arena               the arena to allocate the size buffer from
92       * @return required buffer size, or 0 on unexpected error
93       */
94      public static int SetupDiGetDeviceInterfaceDetailSize(MemorySegment hDevInfo, MemorySegment deviceInterfaceData,
95              Arena arena) {
96          return getIntOrDefault(() -> {
97              MemorySegment requiredSize = arena.allocate(JAVA_INT);
98              SetupDiGetDeviceInterfaceDetail.invokeExact(hDevInfo, deviceInterfaceData, MemorySegment.NULL, 0,
99                      requiredSize, MemorySegment.NULL);
100             int err = Kernel32FFM.GetLastError().orElse(0);
101             return err == ERROR_INSUFFICIENT_BUFFER ? requiredSize.get(JAVA_INT, 0) : 0;
102         }, 0, LOG, "SetupApiFFM.SetupDiGetDeviceInterfaceDetailSize failed: {}");
103     }
104 
105     /**
106      * Second call with allocated buffer to get device path. The detail buffer layout is: cbSize (DWORD=4) + DevicePath
107      * (WCHAR[]). On 64-bit, cbSize field value must be 8; on 32-bit it is 5 or 6.
108      *
109      * @param hDevInfo            the device info set handle
110      * @param deviceInterfaceData the device interface data
111      * @param requiredSize        the buffer size returned by {@link #SetupDiGetDeviceInterfaceDetailSize}
112      * @param arena               the arena to allocate buffers from
113      * @return the device path string, or empty if failed
114      */
115     public static Optional<String> SetupDiGetDeviceInterfaceDetail(MemorySegment hDevInfo,
116             MemorySegment deviceInterfaceData, int requiredSize, Arena arena) {
117         return getOptional(() -> {
118             MemorySegment detail = arena.allocate(requiredSize);
119             detail.set(JAVA_INT, 0, is64Bit() ? 8 : 6);
120             MemorySegment reqSize = arena.allocate(JAVA_INT);
121             int result = (int) SetupDiGetDeviceInterfaceDetail.invokeExact(hDevInfo, deviceInterfaceData, detail,
122                     requiredSize, reqSize, MemorySegment.NULL);
123             return isSuccess(result) ? readWideString(detail.asSlice(4)) : null;
124         }, LOG, "SetupApiFFM.SetupDiGetDeviceInterfaceDetail failed: {}");
125     }

164     /**
165      * Opens a registry key for device-specific configuration information.
166      *
167      * @param hDevInfo    the device info set handle
168      * @param devInfoData the device info data
169      * @param scope       DICS_FLAG_GLOBAL or DICS_FLAG_CONFIGSPECIFIC
170      * @param hwProfile   hardware profile (0 for current)
171      * @param keyType     DIREG_DEV or DIREG_DRV
172      * @param samDesired  registry access mask (e.g. KEY_QUERY_VALUE)
173      * @return registry key handle, or null if failed
174      */
175     public static MemorySegment SetupDiOpenDevRegKey(MemorySegment hDevInfo, MemorySegment devInfoData, int scope,
176             int hwProfile, int keyType, int samDesired) {
177         return getOrDefault(() -> {
178             MemorySegment key = (MemorySegment) SetupDiOpenDevRegKey.invokeExact(hDevInfo, devInfoData, scope,
179                     hwProfile, keyType, samDesired);
180             return Kernel32FFM.isInvalidHandle(key) ? null : key;
181         }, null, LOG, "SetupApiFFM.SetupDiOpenDevRegKey failed: {}");
182     }
