114     public static Optional<String> GetComputerName() {
115         try (Arena arena = Arena.ofConfined()) {
116             int maxLen = 32;
117 
118             MemorySegment buffer = arena.allocate(JAVA_CHAR, maxLen);
119             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
120             sizeSegment.set(ValueLayout.JAVA_INT, 0, maxLen);
121 
122             boolean success = isSuccess((int) GetComputerName.invokeExact(buffer, sizeSegment));
123             if (!success) {
124                 int errorCode = (int) GetLastError.invokeExact();
125                 LOG.error("Failed to get computer name. Error code: {}", errorCode);
126                 return Optional.empty();
127             }
128 
129             return Optional.of(readWideString(buffer));
130         } catch (Throwable t) {
131             LOG.debug("Kernel32FFM.GetComputerName failed: {}", t.getMessage());
132             return Optional.empty();
133         }

346         return getBooleanOrDefault(

351     /**
352      * Retrieves the full name of the executable image for the specified process.
353      *
354      * @param hProcess Handle to the process
355      * @param dwFlags  Flags (0 for Win32 path format)
356      * @param arena    Arena for memory allocation
357      * @return The full process image name, or empty if failed
358      */
359     public static Optional<String> QueryFullProcessImageName(MemorySegment hProcess, int dwFlags, Arena arena) {
360         int maxPath = 260;
361         // Retry with growing buffer if needed (up to 32K which is max path on Windows)
362         while (maxPath <= 32768) {
363             MemorySegment buffer = arena.allocate(JAVA_CHAR, maxPath);
364             MemorySegment sizeSegment = arena.allocate(JAVA_INT);
365             sizeSegment.set(JAVA_INT, 0, maxPath);
366 
367             if (QueryFullProcessImageName(hProcess, dwFlags, buffer, sizeSegment)) {
368                 return Optional.of(readWideString(buffer));
369             }
370             // Check if buffer was too small (returned size >= maxPath)
371             int returnedSize = sizeSegment.get(JAVA_INT, 0);
372             if (returnedSize < maxPath) {
373                 break;
374             }
375             maxPath *= 2;
376         }
377         return Optional.empty();
378     }
