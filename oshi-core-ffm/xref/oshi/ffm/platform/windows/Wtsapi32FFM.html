5   package oshi.ffm.platform.windows;
6   
7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   
10  import java.lang.foreign.MemorySegment;
11  import java.lang.foreign.SymbolLookup;
12  import java.lang.invoke.MethodHandle;
13  
14  public final class Wtsapi32FFM extends WindowsForeignFunctions {
15  
16      private static final SymbolLookup WTS = lib("Wtsapi32");
17  
18      public static final MemorySegment WTS_CURRENT_SERVER_HANDLE = MemorySegment.NULL;
19  
20      /** Pass to WTSEnumerateProcessesEx to enumerate all sessions. */
21      public static final int WTS_ANY_SESSION = -2;
22  
23      /** Request WTS_PROCESS_INFO_EX structures from WTSEnumerateProcessesEx. */
24      public static final int WTS_PROCESS_INFO_LEVEL_1 = 1;
25  
26      /** WTS_TYPE_CLASS value for WTSTypeProcessInfoLevel1, used with WTSFreeMemoryEx. */
27      public static final int WTS_TYPE_PROCESS_INFO_LEVEL_1 = 1;
28  
29      // WTS_PROCESS_INFO_EXW struct layout (64-bit):
30      // SessionId(4) ProcessId(4) pProcessName(ptr) pUserSid(ptr)
31      // NumberOfThreads(4) HandleCount(4) PagefileUsage(4) PeakPagefileUsage(4)
32      // WorkingSetSize(4) PeakWorkingSetSize(4) UserTime(8) KernelTime(8) = 64 bytes
33      public static final long PROCESS_INFO_EX_SIZE = 64;
34      public static final long PROCESS_INFO_EX_PROCESS_ID = 4;
35      public static final long PROCESS_INFO_EX_PROCESS_NAME = 8;
36      public static final long PROCESS_INFO_EX_NUMBER_OF_THREADS = 24;
37      public static final long PROCESS_INFO_EX_HANDLE_COUNT = 28;
38      public static final long PROCESS_INFO_EX_PAGEFILE_USAGE = 32;
39      public static final long PROCESS_INFO_EX_USER_TIME = 48;
40      public static final long PROCESS_INFO_EX_KERNEL_TIME = 56;
41  
42      private static final MethodHandle WTSEnumerateSessionsW = downcall(WTS, "WTSEnumerateSessionsW", JAVA_INT, ADDRESS,
43              JAVA_INT, JAVA_INT, ADDRESS, ADDRESS);
44  
45      public static boolean WTSEnumerateSessions(MemorySegment hServer, int reserved, int version,
46              MemorySegment ppSessionInfo, MemorySegment pCount) throws Throwable {
47          return isSuccess((int) WTSEnumerateSessionsW.invokeExact(hServer, reserved, version, ppSessionInfo, pCount));
48      }
49  
50      private static final MethodHandle WTSEnumerateProcessesExW = downcall(WTS, "WTSEnumerateProcessesExW", JAVA_INT,
51              ADDRESS, ADDRESS, JAVA_INT, ADDRESS, ADDRESS);
52  
53      public static boolean WTSEnumerateProcessesEx(MemorySegment hServer, MemorySegment pLevel, int sessionId,
54              MemorySegment ppProcessInfo, MemorySegment pCount) throws Throwable {
55          return isSuccess((int) WTSEnumerateProcessesExW.invokeExact(hServer, pLevel, sessionId, ppProcessInfo, pCount));
56      }
57  
58      private static final MethodHandle WTSQuerySessionInformationW = downcall(WTS, "WTSQuerySessionInformationW",
59              JAVA_INT, ADDRESS, JAVA_INT, JAVA_INT, ADDRESS, ADDRESS);
60  
61      public static boolean WTSQuerySessionInformation(MemorySegment hServer, int sessionId, int wtsInfoClass,
62              MemorySegment ppBuffer, MemorySegment pBytesReturned) throws Throwable {
63          return isSuccess((int) WTSQuerySessionInformationW.invokeExact(hServer, sessionId, wtsInfoClass, ppBuffer,
64                  pBytesReturned));
65      }
66  
67      private static final MethodHandle WTSFreeMemory = downcall(WTS, "WTSFreeMemory", null, ADDRESS);
68  
69      public static void WTSFreeMemory(MemorySegment pMemory) throws Throwable {
70          WTSFreeMemory.invokeExact(pMemory);
71      }
72  
73      private static final MethodHandle WTSFreeMemoryExW = downcall(WTS, "WTSFreeMemoryExW", JAVA_INT, JAVA_INT, ADDRESS,
74              JAVA_INT);
75  
76      public static boolean WTSFreeMemoryEx(int wtsTypeClass, MemorySegment pMemory, int numberOfEntries)
77              throws Throwable {
78          return isSuccess((int) WTSFreeMemoryExW.invokeExact(wtsTypeClass, pMemory, numberOfEntries));
79      }
80  }
