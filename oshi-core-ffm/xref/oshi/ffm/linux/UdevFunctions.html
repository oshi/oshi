22  /**
23   * FFM bindings for the udev (userspace device management) library.
24   * <p>
25   * libudev provides APIs to introspect and enumerate devices on the local Linux system. All udev objects are opaque
26   * {@link MemorySegment} handles. Callers must unref context, enumerate, and device handles when done.
27   * <p>
28   * Loading can be suppressed via the {@code oshi.os.linux.allowudev} configuration property.
29   */
30  public final class UdevFunctions extends ForeignFunctions {

95                  hEnumScan = LINKER.downcallHandle(lib.findOrThrow("udev_enumerate_scan_devices"),
96                          FunctionDescriptor.of(JAVA_INT, ADDRESS));
97                  hEnumGetList = LINKER.downcallHandle(lib.findOrThrow("udev_enumerate_get_list_entry"),

102                         FunctionDescriptor.of(ADDRESS, ADDRESS));
103                 hDevNewFromSyspath = LINKER.downcallHandle(lib.findOrThrow("udev_device_new_from_syspath"),
104                         FunctionDescriptor.of(ADDRESS, ADDRESS, ADDRESS));
105                 hDevUnref = LINKER.downcallHandle(lib.findOrThrow("udev_device_unref"),
106                         FunctionDescriptor.ofVoid(ADDRESS));
107                 hDevGetParent = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_parent"),
108                         FunctionDescriptor.of(ADDRESS, ADDRESS));
109                 hDevGetParentSubDev = LINKER.downcallHandle(
110                         lib.findOrThrow("udev_device_get_parent_with_subsystem_devtype"),
111                         FunctionDescriptor.of(ADDRESS, ADDRESS, ADDRESS, ADDRESS));
112                 hDevGetSyspath = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_syspath"),
113                         FunctionDescriptor.of(ADDRESS, ADDRESS));
114                 hDevGetSysname = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_sysname"),
115                         FunctionDescriptor.of(ADDRESS, ADDRESS));
116                 hDevGetDevnode = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_devnode"),
117                         FunctionDescriptor.of(ADDRESS, ADDRESS));
118                 hDevGetDevtype = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_devtype"),
119                         FunctionDescriptor.of(ADDRESS, ADDRESS));
120                 hDevGetSubsystem = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_subsystem"),
121                         FunctionDescriptor.of(ADDRESS, ADDRESS));
122                 hDevGetSysattr = LINKER.downcallHandle(lib.findOrThrow("udev_device_get_sysattr_value"),
123                         FunctionDescriptor.of(ADDRESS, ADDRESS, ADDRESS));

143         udev_device_get_parent = hDevGetParent;

154     /**
155      * Attempts to load libudev by trying the unversioned name first, then known versioned sonames. The unversioned
156      * {@code libudev.so} is only present when the development package is installed; the versioned names are present on
157      * all runtime installations.
158      *
159      * @return the {@link SymbolLookup} for libudev
160      * @throws IllegalArgumentException if none of the candidate names can be loaded
161      */
162     private static SymbolLookup loadUdevLibrary() {
163         for (String name : new String[] { "udev", "libudev.so.1", "libudev.so.0" }) {
164             try {
165                 return name.startsWith("lib") ? SymbolLookup.libraryLookup(name, LIBRARY_ARENA) : libraryLookup(name);
166             } catch (IllegalArgumentException ignored) {
167                 // try next candidate name
168             }
169         }
170         throw new IllegalArgumentException("Cannot open library: libudev (tried udev, libudev.so.1, libudev.so.0)");
171     }
