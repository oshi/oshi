5   package oshi.ffm.unix.solaris;

7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   import static java.lang.foreign.ValueLayout.JAVA_LONG;
10  
11  import java.lang.foreign.FunctionDescriptor;
12  import java.lang.foreign.MemoryLayout;
13  import java.lang.foreign.MemoryLayout.PathElement;
14  import java.lang.foreign.MemorySegment;
15  import java.lang.foreign.StructLayout;
16  import java.lang.foreign.SymbolLookup;
17  import java.lang.foreign.ValueLayout;
18  import java.lang.invoke.MethodHandle;
19  import java.lang.invoke.VarHandle;
20  
21  import oshi.ffm.ForeignFunctions;
22  
23  /**
24   * FFM bindings for Solaris/illumos libc functions used by OSHI.
25   * <p>
26   * Solaris uses kstat as its primary kernel-statistics interface, so the libc surface here is small: thread/process IDs
27   * and {@code getrlimit}. The {@code RLIMIT_NOFILE} resource constant is {@code 5} on Solaris/illumos (not {@code 7} as
28   * on Linux — see <a href="https://illumos.org/man/2/getrlimit">illumos getrlimit(2)</a>).
29   */
30  public final class SolarisLibcFunctions extends ForeignFunctions {
31  
32      private SolarisLibcFunctions() {
33      }
34  
35      /** Layout of the C {@code size_t} type on 64-bit Solaris/illumos. */
36      public static final ValueLayout.OfLong SIZE_T = ValueLayout.JAVA_LONG;
37  
38      /**
39       * {@code getrlimit} resource: maximum number of open file descriptors. illumos/Solaris value (5). JNA's
40       * {@code Resource.RLIMIT_NOFILE} is the Linux value (7), which on Solaris corresponds to {@code RLIM_NLIMITS}
41       * (invalid); the JNA-side {@code oshi.jna.platform.unix.SolarisLibc#RLIMIT_NOFILE} shadows it with the correct
42       * value.
43       */
44      public static final int RLIMIT_NOFILE = 5;
45  
46      /** Layout of Solaris/illumos {@code struct rlimit}: two {@code rlim_t} (LP64 long) fields. */
47      public static final StructLayout RLIMIT_LAYOUT = MemoryLayout.structLayout(JAVA_LONG.withName("rlim_cur"),
48              JAVA_LONG.withName("rlim_max"));
49  
50      private static final VarHandle RLIMIT_CUR = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_cur"));
51      private static final VarHandle RLIMIT_MAX = RLIMIT_LAYOUT.varHandle(PathElement.groupElement("rlim_max"));
52  
53      // libc is already loaded into the JVM process; defaultLookup() avoids versioning pitfalls.
54      private static final SymbolLookup LIBC = LINKER.defaultLookup();
55  
56      // pid_t getpid(void);
57      private static final MethodHandle getpid = LINKER.downcallHandle(LIBC.findOrThrow("getpid"),
58              FunctionDescriptor.of(JAVA_INT));
59  
60      /**
61       * Calls {@code getpid()}.
62       *
63       * @return the process ID of the calling process
64       * @throws Throwable on FFM invocation error
65       */
66      public static int getpid() throws Throwable {
67          return (int) getpid.invokeExact();
68      }
69  
70      // thread_t thr_self(void); // Solaris-specific
71      private static final MethodHandle thr_self = LINKER.downcallHandle(LIBC.findOrThrow("thr_self"),
72              FunctionDescriptor.of(JAVA_INT));
73  
74      /**
75       * Calls {@code thr_self()} — returns the thread ID of the calling thread (Solaris/illumos).
76       *
77       * @return the thread ID of the calling thread
78       * @throws Throwable on FFM invocation error
79       */
80      public static int thr_self() throws Throwable {
81          return (int) thr_self.invokeExact();
82      }
83  
84      // int getrlimit(int resource, struct rlimit *rlim);
85      private static final MethodHandle getrlimit = LINKER.downcallHandle(LIBC.findOrThrow("getrlimit"),
86              FunctionDescriptor.of(JAVA_INT, JAVA_INT, ADDRESS));
87  
88      /**
89       * Calls {@code getrlimit(resource, rlim)}.
90       *
91       * @param resource resource constant (e.g. {@link #RLIMIT_NOFILE})
92       * @param rlim     segment allocated with {@link #RLIMIT_LAYOUT}
93       * @return 0 on success, -1 on error
94       * @throws Throwable on FFM invocation error
95       */
96      public static int getrlimit(int resource, MemorySegment rlim) throws Throwable {
97          return (int) getrlimit.invokeExact(resource, rlim);
98      }
99  
100     /**
101      * Reads {@code rlim_cur} from an rlimit segment populated by {@link #getrlimit(int, MemorySegment)}.
102      *
103      * @param rlim segment allocated with {@link #RLIMIT_LAYOUT}
104      * @return the soft resource limit
105      */
106     public static long rlimitCur(MemorySegment rlim) {
107         return (long) RLIMIT_CUR.get(rlim, 0L);
108     }
109 
110     /**
111      * Reads {@code rlim_max} from an rlimit segment populated by {@link #getrlimit(int, MemorySegment)}.
112      *
113      * @param rlim segment allocated with {@link #RLIMIT_LAYOUT}
114      * @return the hard resource limit
115      */
116     public static long rlimitMax(MemorySegment rlim) {
117         return (long) RLIMIT_MAX.get(rlim, 0L);
118     }
119 
120     // int gethostname(char *name, size_t namelen);
121     private static final MethodHandle gethostname = LINKER.downcallHandle(LIBC.findOrThrow("gethostname"),
122             FunctionDescriptor.of(JAVA_INT, ADDRESS, SIZE_T));
123 
124     /**
125      * Calls {@code gethostname(name, namelen)}.
126      *
127      * @param name    buffer for the hostname (allocated by caller)
128      * @param namelen size of {@code name} in bytes
129      * @return 0 on success, -1 on error
130      * @throws Throwable on FFM invocation error
131      */
132     public static int gethostname(MemorySegment name, long namelen) throws Throwable {
133         return (int) gethostname.invokeExact(name, namelen);
134     }
135 
136     // int getloadavg(double loadavg[], int nelem);
137     private static final MethodHandle getloadavg = LINKER.downcallHandle(LIBC.findOrThrow("getloadavg"),
138             FunctionDescriptor.of(JAVA_INT, ADDRESS, JAVA_INT));
139 
140     /**
141      * Calls {@code getloadavg(loadavg, nelem)}.
142      *
143      * @param loadavg pre-allocated segment of {@code nelem} doubles
144      * @param nelem   number of load average values to retrieve (1-3)
145      * @return number of samples set, or -1 on error
146      * @throws Throwable on FFM invocation error
147      */
148     public static int getloadavg(MemorySegment loadavg, int nelem) throws Throwable {
149         return (int) getloadavg.invokeExact(loadavg, nelem);
150     }
151 }
