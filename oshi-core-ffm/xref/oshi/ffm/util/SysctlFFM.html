5   package oshi.ffm.util;

7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   import static java.lang.foreign.ValueLayout.JAVA_LONG;
9   import static oshi.ffm.ForeignFunctions.callInArenaBooleanOrDefault;
10  import static oshi.ffm.ForeignFunctions.callInArenaIntOrDefault;
11  import static oshi.ffm.ForeignFunctions.callInArenaLongOrDefault;
12  import static oshi.ffm.ForeignFunctions.callInArenaOrDefault;
13  
14  import java.lang.foreign.Arena;
15  import java.lang.foreign.MemorySegment;
16  import java.lang.foreign.ValueLayout;
17  
18  import org.slf4j.Logger;
19  
20  import oshi.annotation.concurrent.ThreadSafe;
21  import oshi.util.LogLevel;
22  import oshi.util.ParseUtil;

24  /**
25   * Shared implementation of the FFM {@code sysctl} result handling used by the per-platform utility classes. The only
26   * per-platform difference is the native call itself (macOS and FreeBSD use {@code sysctlbyname(String)}; OpenBSD uses
27   * {@code sysctl(int[])}); callers supply that call as a {@link SysctlOp} so the arena management, buffer sizing, and
28   * result decoding live in one place. The {@code SysctlOp} is responsible for allocating its own name/MIB segment,
29   * invoking the native function, and logging any errno on failure.
30   */
31  @ThreadSafe
32  public final class SysctlFFM {
33  
34      /** All supported platforms define {@code size_t} as an unsigned 64-bit value. */
35      public static final ValueLayout.OfLong SIZE_T = JAVA_LONG;
36  
37      private SysctlFFM() {
38      }
39  
40      /**
41       * A single native {@code sysctl}/{@code sysctlbyname} invocation. Implementations allocate their name or MIB
42       * segment from the supplied arena, invoke the native function, log any errno on failure, and report success.
43       */
44      @FunctionalInterface
45      public interface SysctlOp {
46          /**
47           * Invokes the bound native sysctl call.
48           *
49           * @param arena   the confined arena for any scratch allocations
50           * @param oldp    buffer to receive the result, or {@link MemorySegment#NULL} to query the required size
51           * @param oldlenp in/out size of {@code oldp}
52           * @return {@code true} on success; {@code false} on failure
53           * @throws Throwable on FFM invocation error
54           */
55          boolean invoke(Arena arena, MemorySegment oldp, MemorySegment oldlenp) throws Throwable;
56      }
57  
58      /**
59       * Executes a sysctl call with an int result.
60       *
61       * @param op   the bound native sysctl call
62       * @param def  default int value
63       * @param log  logger for the calling class
64       * @param name name of the sysctl, for logging
65       * @return The int result of the call if successful; the default otherwise
66       */
67      public static int sysctl(SysctlOp op, int def, Logger log, Object name) {
68          return callInArenaIntOrDefault(arena -> {
69              MemorySegment valueSeg = arena.allocate(JAVA_INT);
70              MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, JAVA_INT.byteSize());
71              if (!op.invoke(arena, valueSeg, sizeSeg)) {
72                  return def;
73              }
74              return valueSeg.get(JAVA_INT, 0);
75          }, log, LogLevel.WARN, "Failed to get sysctl value for " + name, def);
76      }
77  
78      /**
79       * Executes a sysctl call with a long result.
80       *
81       * @param op   the bound native sysctl call
82       * @param def  default long value
83       * @param log  logger for the calling class
84       * @param name name of the sysctl, for logging
85       * @return The long result of the call if successful; the default otherwise
86       */
87      public static long sysctl(SysctlOp op, long def, Logger log, Object name) {
88          return callInArenaLongOrDefault(arena -> {
89              MemorySegment valueSeg = arena.allocate(JAVA_LONG);
90              MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, JAVA_LONG.byteSize());
91              if (!op.invoke(arena, valueSeg, sizeSeg)) {
92                  return def;
93              }
94              // Some OIDs exposed as a "long" are actually kernel ints (e.g. FreeBSD hw.clockrate is a 4-byte int).
95              // sysctl updates oldlenp in place with the number of bytes actually written, so read only that many,
96              // widening a 4-byte int as unsigned.
97              return sizeSeg.get(SIZE_T, 0) == JAVA_INT.byteSize()
98                      ? ParseUtil.unsignedIntToLong(valueSeg.get(JAVA_INT, 0))
99                      : valueSeg.get(JAVA_LONG, 0);
100         }, log, LogLevel.WARN, "Failed to get sysctl value for " + name, def);
101     }
102 
103     /**
104      * Executes a sysctl call with a String result.
105      *
106      * @param op   the bound native sysctl call
107      * @param def  default String value
108      * @param log  logger for the calling class
109      * @param name name of the sysctl, for logging
110      * @return The String result of the call if successful; the default otherwise
111      */
112     public static String sysctl(SysctlOp op, String def, Logger log, Object name) {
113         return callInArenaOrDefault(arena -> {
114             MemorySegment sizeSeg = arena.allocate(SIZE_T);
115             if (!op.invoke(arena, MemorySegment.NULL, sizeSeg)) {
116                 return def;
117             }
118             long size = sizeSeg.get(SIZE_T, 0);
119             MemorySegment valueSeg = arena.allocate(size + 1);
120             if (!op.invoke(arena, valueSeg, sizeSeg)) {
121                 return def;
122             }
123             return valueSeg.getString(0);
124         }, log, LogLevel.WARN, "Failed to get sysctl value for " + name, def);
125     }
126 
127     /**
128      * Executes a sysctl call that populates a caller-provided struct segment.
129      *
130      * @param op     the bound native sysctl call
131      * @param struct segment sized to the expected struct layout
132      * @param log    logger for the calling class
133      * @param name   name of the sysctl, for logging
134      * @return {@code true} if the struct was populated, {@code false} otherwise
135      */
136     public static boolean sysctl(SysctlOp op, MemorySegment struct, Logger log, Object name) {
137         return callInArenaBooleanOrDefault(arena -> {
138             MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, struct.byteSize());
139             return op.invoke(arena, struct, sizeSeg);
140         }, log, LogLevel.WARN, "Failed to get sysctl value for " + name, false);
141     }
142 
143     /**
144      * Executes a sysctl call returning a freshly allocated buffer of the natural size.
145      *
146      * @param op   the bound native sysctl call
147      * @param log  logger for the calling class
148      * @param name name of the sysctl, for logging
149      * @return an auto-arena {@link MemorySegment} containing the result on success; {@code null} otherwise
150      */
151     public static MemorySegment sysctl(SysctlOp op, Logger log, Object name) {
152         return callInArenaOrDefault(arena -> {
153             MemorySegment sizeSeg = arena.allocate(SIZE_T);
154             if (!op.invoke(arena, MemorySegment.NULL, sizeSeg)) {
155                 return null;
156             }
157             long size = sizeSeg.get(SIZE_T, 0);
158             MemorySegment valueSeg = arena.allocate(size);
159             if (!op.invoke(arena, valueSeg, sizeSeg)) {
160                 return null;
161             }
162             MemorySegment returnSeg = Arena.ofAuto().allocate(size);
163             returnSeg.copyFrom(valueSeg);
164             return returnSeg;
165         }, log, LogLevel.WARN, "Failed to get sysctl value for " + name, null);
166     }
