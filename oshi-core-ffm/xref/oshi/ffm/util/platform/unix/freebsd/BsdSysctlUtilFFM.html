7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   import static java.lang.foreign.ValueLayout.JAVA_LONG;
9   import static oshi.ffm.ForeignFunctions.CAPTURED_STATE_LAYOUT;
10  import static oshi.ffm.ForeignFunctions.getErrno;
11  import static oshi.ffm.unix.freebsd.FreeBsdLibcFunctions.SIZE_T;
12  
13  import java.lang.foreign.Arena;
14  import java.lang.foreign.MemorySegment;
15  
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;

19  import oshi.annotation.concurrent.ThreadSafe;
20  import oshi.ffm.unix.freebsd.FreeBsdLibcFunctions;

22  /**
23   * Provides access to sysctl calls on FreeBSD via the FFM API.
24   * <p>
25   * Mirrors the API of {@code oshi.util.platform.unix.freebsd.BsdSysctlUtil} (JNA) so per-class FFM concretes can be
26   * swapped in without changing call sites.
27   */
28  @ThreadSafe
29  public final class BsdSysctlUtilFFM {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(BsdSysctlUtilFFM.class);
32  
33      private static final String SYSCTL_FAIL = "Failed sysctl call: {}, Error code: {}";
34  
35      private BsdSysctlUtilFFM() {
36      }
37  
38      /**
39       * Executes a sysctl call with an int result.
40       *
41       * @param name name of the sysctl
42       * @param def  default int value
43       * @return The int result of the call if successful; the default otherwise
44       */
45      public static int sysctl(String name, int def) {
46          try (Arena arena = Arena.ofConfined()) {
47              MemorySegment nameSeg = arena.allocateFrom(name);
48              MemorySegment valueSeg = arena.allocate(JAVA_INT);
49              MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, JAVA_INT.byteSize());
50              if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg)) {
51                  return def;
52              }
53              return valueSeg.get(JAVA_INT, 0);
54          } catch (Throwable e) {
55              LOG.warn("Failed to get sysctl value for {}", name, e);
56              return def;
57          }
58      }
59  
60      /**
61       * Executes a sysctl call with a long result.
62       *
63       * @param name name of the sysctl
64       * @param def  default long value
65       * @return The long result of the call if successful; the default otherwise
66       */
67      public static long sysctl(String name, long def) {
68          try (Arena arena = Arena.ofConfined()) {
69              MemorySegment nameSeg = arena.allocateFrom(name);
70              MemorySegment valueSeg = arena.allocate(JAVA_LONG);
71              MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, JAVA_LONG.byteSize());
72              if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg)) {
73                  return def;
74              }
75              return valueSeg.get(JAVA_LONG, 0);
76          } catch (Throwable e) {
77              LOG.warn("Failed to get sysctl value for {}", name, e);
78              return def;
79          }
80      }
81  
82      /**
83       * Executes a sysctl call with a String result.
84       *
85       * @param name name of the sysctl
86       * @param def  default String value
87       * @return The String result of the call if successful; the default otherwise
88       */
89      public static String sysctl(String name, String def) {
90          try (Arena arena = Arena.ofConfined()) {
91              MemorySegment nameSeg = arena.allocateFrom(name);
92              MemorySegment sizeSeg = arena.allocate(SIZE_T);
93              if (!sysctlbyname(arena, nameSeg, MemorySegment.NULL, sizeSeg)) {
94                  return def;
95              }
96              long size = sizeSeg.get(SIZE_T, 0);
97              // +1 for null terminator
98              MemorySegment valueSeg = arena.allocate(size + 1);
99              if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg)) {
100                 return def;
101             }
102             return valueSeg.getString(0);
103         } catch (Throwable e) {
104             LOG.warn("Failed to get sysctl value for {}", name, e);
105             return def;
106         }
107     }
108 
109     /**
110      * Executes a sysctl call that populates a caller-provided struct segment.
111      *
112      * @param name   name of the sysctl
113      * @param struct segment sized to the expected struct layout
114      * @return {@code true} if the struct was populated, {@code false} otherwise
115      */
116     public static boolean sysctl(String name, MemorySegment struct) {
117         try (Arena arena = Arena.ofConfined()) {
118             MemorySegment nameSeg = arena.allocateFrom(name);
119             MemorySegment sizeSeg = arena.allocateFrom(SIZE_T, struct.byteSize());
120             return sysctlbyname(arena, nameSeg, struct, sizeSeg);
121         } catch (Throwable e) {
122             LOG.warn("Failed to get sysctl value for {}", name, e);
123             return false;
124         }
125     }
126 
127     /**
128      * Executes a sysctl call returning a freshly allocated buffer of the natural size.
129      *
130      * @param name name of the sysctl
131      * @return an auto-arena {@link MemorySegment} containing the result on success; {@code null} otherwise
132      */
133     public static MemorySegment sysctl(String name) {
134         try (Arena arena = Arena.ofConfined()) {
135             MemorySegment nameSeg = arena.allocateFrom(name);
136             MemorySegment sizeSeg = arena.allocate(SIZE_T);
137             if (!sysctlbyname(arena, nameSeg, MemorySegment.NULL, sizeSeg)) {
138                 return null;
139             }
140             long size = sizeSeg.get(SIZE_T, 0);
141             MemorySegment valueSeg = arena.allocate(size);
142             if (!sysctlbyname(arena, nameSeg, valueSeg, sizeSeg)) {
143                 return null;
144             }
145             // Copy to an auto-arena segment so it outlives this confined arena
146             MemorySegment returnSeg = Arena.ofAuto().allocate(size);
147             returnSeg.copyFrom(valueSeg);
148             return returnSeg;
149         } catch (Throwable e) {
150             LOG.warn("Failed to get sysctl value for {}", name, e);
151             return null;
152         }
153     }
154 
155     private static boolean sysctlbyname(Arena arena, MemorySegment name, MemorySegment oldp, MemorySegment oldlenp)
156             throws Throwable {
157         MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
158         int result = FreeBsdLibcFunctions.sysctlbyname(callState, name, oldp, oldlenp, MemorySegment.NULL, 0L);
159         if (result != 0) {
160             LOG.warn(SYSCTL_FAIL, name.getString(0), getErrno(callState));
161             return false;
162         }
163         return true;
164     }
165 }
