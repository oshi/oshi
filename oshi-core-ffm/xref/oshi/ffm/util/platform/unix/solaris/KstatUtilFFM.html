44  /**
45   * FFM equivalent of {@code oshi.util.platform.unix.solaris.KstatUtil}: thread-safe access to kstat data via
46   * {@link LibKstatFunctions}. Mirrors the JNA wrapper's API surface (chain lookup, named-data accessors). Kstat2 methods
47   * are intentionally not implemented here yet — see {@code Kstat2Functions} for the gated probe.
48   */
49  @ThreadSafe
50  public final class KstatUtilFFM {
51  
52      private static final Logger LOG = LoggerFactory.getLogger(KstatUtilFFM.class);
53  
54      private static final Lock CHAIN = new ReentrantLock();
55  
56      /** Cached {@code kstat_ctl_t} pointer for the JVM lifetime; opened lazily. */
57      @GuardedBy("CHAIN")
58      private static MemorySegment kstatCtl = MemorySegment.NULL;
59  
60      private KstatUtilFFM() {
61      }
62  
63      /**
64       * A copy of the Kstat chain. Only one thread may actively use this object at any time. Acquired via
65       * {@link #openChain()} and released via {@link #close()}.
66       */
67      public static final class KstatChain implements AutoCloseable {
68  
69          private final MemorySegment ctl;
70  
71          private KstatChain(MemorySegment ctl) {
72              this.ctl = ctl;
73              update();
74          }
75  
76          /**
77           * Calls {@code kstat_read(kc, ksp, NULL)} to refresh the named-data buffer pointed to by {@code ksp}.
78           *
79           * @param ksp segment addressing a {@code kstat_t}; should be reinterpreted to
80           *            {@link LibKstatFunctions#KSTAT_LAYOUT}
81           * @return {@code true} on success, {@code false} on failure
82           */
83          @GuardedBy("CHAIN")
84          public boolean read(MemorySegment ksp) {
85              try {
86                  int retry = 0;
87                  while (LibKstatFunctions.kstat_read(ctl, ksp, MemorySegment.NULL) == -1) {
88                      if (++retry >= 5) {
89                          return false;
90                      }
91                      // Exponential backoff mirroring the JNA KstatUtil (16/32/64/128/256 ms).
92                      // Avoids busy-spinning on transient EAGAINs while still bounded.
93                      try {
94                          Thread.sleep(8L << retry);
95                      } catch (InterruptedException _) {
96                          Thread.currentThread().interrupt();
97                          return false;
98                      }
99                  }
100                 return true;
101             } catch (Throwable t) {
102                 LOG.warn("kstat_read failed", t);
103                 return false;
104             }
105         }
106 
107         /**
108          * Convenience method for {@code kstat_lookup}. {@code module}/{@code name} may be {@code null} (wildcard);
109          * {@code instance} may be {@code -1} (wildcard).
110          *
111          * @param module   module name or {@code null}
112          * @param instance instance number or {@code -1}
113          * @param name     kstat name or {@code null}
114          * @return the first match reinterpreted to {@link LibKstatFunctions#KSTAT_LAYOUT}, or
115          *         {@link MemorySegment#NULL}
116          */
117         @GuardedBy("CHAIN")
118         public MemorySegment lookup(String module, int instance, String name) {
119             try (Arena arena = Arena.ofConfined()) {
120                 MemorySegment modSeg = module == null ? MemorySegment.NULL : arena.allocateFrom(module);
121                 MemorySegment nameSeg = name == null ? MemorySegment.NULL : arena.allocateFrom(name);
122                 MemorySegment hit = LibKstatFunctions.kstat_lookup(ctl, modSeg, instance, nameSeg);
123                 if (hit.address() == 0L) {
124                     return MemorySegment.NULL;
125                 }
126                 return hit.reinterpret(KSTAT_LAYOUT.byteSize());
127             } catch (Throwable t) {
128                 LOG.warn("kstat_lookup failed for {}:{}:{}", module, instance, name, t);
129                 return MemorySegment.NULL;
130             }
131         }
132 
133         /**
134          * Convenience method for iterating all kstats matching {@code (module, instance, name)}.
135          * {@code null}/{@code -1} mean wildcard.
136          *
137          * @param module   module name or {@code null}
138          * @param instance instance number or {@code -1}
139          * @param name     kstat name or {@code null}
140          * @return all matching kstats (each reinterpreted to {@link LibKstatFunctions#KSTAT_LAYOUT})
141          */
142         @GuardedBy("CHAIN")
143         public List<MemorySegment> lookupAll(String module, int instance, String name) {
144             List<MemorySegment> matches = new ArrayList<>();
145             MemorySegment ksp = lookup(module, instance, name);
146             while (ksp.address() != 0L) {
147                 if ((module == null || module.equals(kstatModule(ksp)))
148                         && (instance < 0 || instance == kstatInstance(ksp))
149                         && (name == null || name.equals(kstatName(ksp)))) {
150                     matches.add(ksp);
151                 }
152                 ksp = kstatNext(ksp);
153             }
154             return matches;
155         }
156 
157         /**
158          * Calls {@code kstat_chain_update(kc)}.
159          *
160          * @return the new KCID if changed, 0 if unchanged, -1 on failure
161          */
162         @GuardedBy("CHAIN")
163         public int update() {
164             try {
165                 return LibKstatFunctions.kstat_chain_update(ctl);
166             } catch (Throwable t) {
167                 LOG.warn("kstat_chain_update failed", t);
168                 return -1;
169             }
170         }
171 
172         @Override
173         public void close() {
174             CHAIN.unlock();
175         }
176     }
177 
178     /**
179      * Acquires the chain lock and returns a {@link KstatChain} bound to the (lazily opened) {@code kstat_ctl_t}. Caller
180      * must close the returned object to release the lock.
181      *
182      * @return a locked chain wrapper
183      */
184     public static synchronized KstatChain openChain() {
185         CHAIN.lock();
186         if (kstatCtl.address() == 0L) {
187             try {
188                 MemorySegment ctl = LibKstatFunctions.kstat_open();
189                 if (ctl.address() == 0L) {
190                     CHAIN.unlock();
191                     throw new IllegalStateException("kstat_open returned NULL");
192                 }
193                 kstatCtl = ctl;
194             } catch (Throwable t) {
195                 CHAIN.unlock();
196                 throw new IllegalStateException("kstat_open failed", t);
197             }
198         }
199         return new KstatChain(kstatCtl);
200     }

202     /**
203      * Searches the kstat's named-data section for the record with the given name. Caller must hold the chain lock
204      * (typically via an enclosing {@link KstatChain}) and must have called {@link KstatChain#read(MemorySegment)}
205      * first.
206      *
207      * @param ksp  kstat reinterpreted to {@link LibKstatFunctions#KSTAT_LAYOUT}
208      * @param name record key
209      * @return the value as a string (formatted per data type), or empty if not found / unsupported type
210      */
211     public static String dataLookupString(MemorySegment ksp, String name) {
212         byte type = kstatType(ksp);
213         if (type != KSTAT_TYPE_NAMED && type != KSTAT_TYPE_TIMER) {
214             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
215         }
216         try (Arena arena = Arena.ofConfined()) {
217             MemorySegment nameSeg = arena.allocateFrom(name);
218             MemorySegment data = LibKstatFunctions.kstat_data_lookup(ksp, nameSeg);
219             if (data.address() == 0L) {
220                 LOG.debug("Failed to lookup kstat value for key {}", name);
221                 return "";
222             }
223             MemorySegment named = data.reinterpret(KSTAT_NAMED_LAYOUT.byteSize());
224             byte dt = namedDataType(named);
225             switch (dt) {
226                 case KSTAT_DATA_CHAR:
227                     return namedValueChar(named);
228                 case KSTAT_DATA_INT32:
229                     return Integer.toString(namedValueInt32(named));
230                 case KSTAT_DATA_UINT32:
231                     return FormatUtil.toUnsignedString(namedValueInt32(named));
232                 case KSTAT_DATA_INT64:
233                     return Long.toString(namedValueInt64(named));
234                 case KSTAT_DATA_UINT64:
235                     return FormatUtil.toUnsignedString(namedValueInt64(named));
236                 case KSTAT_DATA_STRING:
237                     return namedValueString(named);
238                 default:
239                     LOG.error("Unimplemented kstat data type {}", dt);
240                     return "";
241             }
242         } catch (Throwable t) {
243             LOG.warn("kstat_data_lookup failed for key {}", name, t);
244             return "";
245         }
246     }
247 
248     /**
249      * Searches the kstat's named-data section for a numeric record. Non-numeric data types return {@code 0}.
250      *
251      * @param ksp  kstat reinterpreted to {@link LibKstatFunctions#KSTAT_LAYOUT}
252      * @param name record key
253      * @return the value as a long, or {@code 0} if not found / unsupported
254      */
255     public static long dataLookupLong(MemorySegment ksp, String name) {
256         byte type = kstatType(ksp);
257         if (type != KSTAT_TYPE_NAMED && type != KSTAT_TYPE_TIMER) {
258             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
259         }
260         try (Arena arena = Arena.ofConfined()) {
261             MemorySegment nameSeg = arena.allocateFrom(name);
262             MemorySegment data = LibKstatFunctions.kstat_data_lookup(ksp, nameSeg);
263             if (data.address() == 0L) {
264                 if (LOG.isDebugEnabled()) {
265                     LOG.debug("Failed to lookup kstat value on {}:{}:{} for key {}", kstatModule(ksp),
266                             kstatInstance(ksp), kstatName(ksp), name);
267                 }
268                 return 0L;
269             }
270             MemorySegment named = data.reinterpret(KSTAT_NAMED_LAYOUT.byteSize());
271             byte dt = namedDataType(named);
272             switch (dt) {
273                 case KSTAT_DATA_INT32:
274                     return namedValueInt32(named);
275                 case KSTAT_DATA_UINT32:
276                     return FormatUtil.getUnsignedInt(namedValueInt32(named));
277                 case KSTAT_DATA_INT64, KSTAT_DATA_UINT64:
278                     return namedValueInt64(named);
279                 default:
280                     LOG.error("Unimplemented or non-numeric kstat data type {}", dt);
281                     return 0L;
282             }
283         } catch (Throwable t) {
284             LOG.warn("kstat_data_lookup failed for key {}", name, t);
285             return 0L;
286         }
287     }

293      * @return the class string
294      */

296         return kstatClass(ksp);
