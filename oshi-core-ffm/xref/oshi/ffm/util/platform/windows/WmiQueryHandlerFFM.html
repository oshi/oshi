71      /**
72       * Factory method to create an instance of this class. To override this class, use {@link #setInstanceClass(Class)}
73       * to define a subclass which extends {@link WmiQueryHandlerFFM}.
74       *
75       * @return An instance of this class or a class defined by {@link #setInstanceClass(Class)}
76       */
77      public static synchronized WmiQueryHandlerFFM createInstance() {
78          if (customClass == null) {
79              return new WmiQueryHandlerFFM();
80          }
81          try {
82              return customClass.getConstructor(EMPTY_CLASS_ARRAY).newInstance(EMPTY_OBJECT_ARRAY);
83          } catch (NoSuchMethodException | SecurityException _) {
84              LOG.error("Failed to find or access a no-arg constructor for {}", customClass);
85          } catch (InstantiationException | IllegalAccessException | IllegalArgumentException
86                  | InvocationTargetException _) {
87              LOG.error("Failed to create a new instance of {}", customClass);
88          }
89          customClass = null;
90          return new WmiQueryHandlerFFM();
91      }

115     /**
116      * Query WMI for values.
117      *
118      * @param <T>     WMI queries use an Enum to identify the fields to query, and use the enum values as keys to
119      *                retrieve the results.
120      * @param query   A WmiQuery object encapsulating the namespace, class, and properties
121      * @param initCom Whether to initialize COM. If {@code true}, initializes COM before the query and uninitializes
122      *                after. If {@code false}, assumes the caller has already called {@link #initCOM()} and will call
123      *                {@link #unInitCOM()} when done. This can improve WMI query performance.
124      * @return a WmiResult object containing the query results, wrapping an EnumMap
125      */
126     public <T extends Enum<T>> WbemcliUtilFFM.WmiResult<T> queryWMI(WbemcliUtilFFM.WmiQuery<T> query, boolean initCom) {
127         WbemcliUtilFFM.WmiResult<T> result = new WbemcliUtilFFM.WmiResult<>(query.getPropertyEnum());
128         if (failedWmiClassNames.contains(query.getWmiClassName())) {
129             return result;
130         }
131         boolean comInit = false;
132         try (Arena arena = Arena.ofConfined()) {
133             if (initCom) {
134                 comInit = initCOM();
135                 if (!comInit) {
136                     return result;
137                 }
138             }
139 
140             Optional<MemorySegment> pLocatorOpt = IWbemLocatorFFM.create(arena);
141             if (pLocatorOpt.isEmpty()) {
142                 LOG.debug("Failed to create IWbemLocator. COM may not be initialized.");
143                 return result;
144             }
145             MemorySegment pLocator = pLocatorOpt.get();
146 
147             try {
148                 Optional<MemorySegment> pServicesOpt = IWbemLocatorFFM.connectServer(pLocator, query.getNameSpace(),
149                         arena);
150                 if (pServicesOpt.isEmpty()) {
151                     failedWmiClassNames.add(query.getWmiClassName());
152                     return result;
153                 }
154                 MemorySegment pServices = pServicesOpt.get();
155 
156                 try {
157                     Ole32FFM.CoSetProxyBlanket(pServices, -1, -1, Ole32FFM.RPC_C_AUTHN_LEVEL_CALL,
158                             Ole32FFM.RPC_C_IMP_LEVEL_IMPERSONATE, Ole32FFM.EOAC_NONE);
159 
160                     // Build WQL query: SELECT prop1,prop2,... FROM ClassName [WHERE ...]
161                     T[] props = query.getPropertyEnum().getEnumConstants();
162                     StringBuilder sb = new StringBuilder("SELECT ");
163                     sb.append(props[0].name());
164                     for (int i = 1; i < props.length; i++) {
165                         sb.append(',').append(props[i].name());
166                     }
167                     sb.append(" FROM ").append(query.getWmiClassName());
168 
169                     Optional<MemorySegment> pEnumOpt = IWbemServicesFFM.execQuery(pServices, sb.toString(), arena);
170                     if (pEnumOpt.isEmpty()) {
171                         failedWmiClassNames.add(query.getWmiClassName());
172                         return result;
173                     }
174                     MemorySegment pEnum = pEnumOpt.get();
175 
176                     try {
177                         while (true) {
178                             IEnumWbemClassObjectFFM.NextResult nextResult = IEnumWbemClassObjectFFM.next(pEnum,
179                                     wmiTimeout, arena);
180                             if (nextResult.isComplete() || !nextResult.hasObject()) {
181                                 break;
182                             }
183                             MemorySegment pObject = nextResult.pObject();
184                             try {
185                                 WbemcliUtilFFM.populateRow(pObject, arena, result);
186                             } finally {
187                                 IUnknownFFM.safeRelease(pObject, arena);
188                             }
189                         }
190                     } finally {
191                         IUnknownFFM.safeRelease(pEnum, arena);
192                     }
193                 } finally {
194                     IUnknownFFM.safeRelease(pServices, arena);
195                 }
196             } finally {
197                 IUnknownFFM.safeRelease(pLocator, arena);
198             }
199         } catch (FfmComException e) {
200             int hresult = e.getHresult();
201             switch (hresult) {
202                 case WbemcliFFM.WBEM_E_INVALID_NAMESPACE:
203                     LOG.warn("COM exception: Invalid Namespace {}", query.getNameSpace());
204                     break;
205                 case WbemcliFFM.WBEM_E_INVALID_CLASS:
206                     LOG.warn("COM exception: Invalid Class {}", query.getWmiClassName());
207                     break;
208                 case WbemcliFFM.WBEM_E_INVALID_QUERY:
209                     LOG.warn("COM exception: Invalid Query for {}", query.getWmiClassName());
210                     break;
211                 default:
212                     handleComException(query, e);
213                     break;
214             }
215             failedWmiClassNames.add(query.getWmiClassName());
216         } catch (Exception e) {
217             LOG.debug("WMI query failed for {}: {}", query.getWmiClassName(), e.getMessage());
218         } finally {
219             if (comInit) {
220                 unInitCOM();
221             }
222         }
223         return result;
224     }

276     /**
277      * Executes a WMI query, processing each result row with a custom processor.
278      *
279      * @param <T>           the type of result object
280      * @param namespace     the WMI namespace
281      * @param wmiClassName  the WMI class name
282      * @param whereClause   optional WHERE clause (null for none)
283      * @param resultFactory factory to create result objects per row
284      * @param rowProcessor  processor to populate result object from WMI row
285      * @return list of result objects, empty list if query fails
286      */
287     public <T> List<T> queryWMI(String namespace, String wmiClassName, String whereClause,
288             java.util.function.Supplier<T> resultFactory, TriConsumer<MemorySegment, Arena, T> rowProcessor) {
289         if (failedWmiClassNames.contains(wmiClassName)) {
290             return new ArrayList<>();
291         }
292         List<T> results = new ArrayList<>();
293         boolean comInit = false;
294         try (Arena arena = Arena.ofConfined()) {
295             comInit = initCOM();
296             if (!comInit) {
297                 return results;
298             }
299             Optional<MemorySegment> pLocatorOpt = IWbemLocatorFFM.create(arena);
300             if (pLocatorOpt.isEmpty()) {
301                 return results;
302             }
303             MemorySegment pLocator = pLocatorOpt.get();
304             try {
305                 Optional<MemorySegment> pServicesOpt = IWbemLocatorFFM.connectServer(pLocator, namespace, arena);
306                 if (pServicesOpt.isEmpty()) {
307                     failedWmiClassNames.add(wmiClassName);
308                     return results;
309                 }
310                 MemorySegment pServices = pServicesOpt.get();
311                 try {
312                     Ole32FFM.CoSetProxyBlanket(pServices, -1, -1, Ole32FFM.RPC_C_AUTHN_LEVEL_CALL,
313                             Ole32FFM.RPC_C_IMP_LEVEL_IMPERSONATE, Ole32FFM.EOAC_NONE);
314                     String query = "SELECT * FROM " + wmiClassName;
315                     if (whereClause != null && !whereClause.isEmpty()) {
316                         query += " " + whereClause;
317                     }
318                     Optional<MemorySegment> pEnumOpt = IWbemServicesFFM.execQuery(pServices, query, arena);
319                     if (pEnumOpt.isEmpty()) {
320                         failedWmiClassNames.add(wmiClassName);
321                         return results;
322                     }
323                     MemorySegment pEnum = pEnumOpt.get();
324                     try {
325                         while (true) {
326                             IEnumWbemClassObjectFFM.NextResult nextResult = IEnumWbemClassObjectFFM.next(pEnum,
327                                     wmiTimeout, arena);
328                             if (nextResult.isComplete() || !nextResult.hasObject()) {
329                                 break;
330                             }
331                             MemorySegment pObject = nextResult.pObject();
332                             try {
333                                 T result = resultFactory.get();
334                                 rowProcessor.accept(pObject, arena, result);
335                                 results.add(result);
336                             } finally {
337                                 IUnknownFFM.safeRelease(pObject, arena);
338                             }
339                         }
340                     } finally {
341                         IUnknownFFM.safeRelease(pEnum, arena);
342                     }
343                 } finally {
344                     IUnknownFFM.safeRelease(pServices, arena);
345                 }
346             } finally {
347                 IUnknownFFM.safeRelease(pLocator, arena);
348             }
349         } catch (FfmComException _) {
350             failedWmiClassNames.add(wmiClassName);
351         } catch (Exception e) {
352             LOG.debug("WMI query failed for {}: {}", wmiClassName, e.getMessage());
353         } finally {
354             if (comInit) {
355                 unInitCOM();
356             }
357         }
358         return results;
359     }

361     /**
362      * Functional interface for processing WMI rows with three parameters.
363      *
364      * @param <A> the first parameter type
365      * @param <B> the second parameter type
366      * @param <C> the third parameter type

368     @FunctionalInterface
369     public interface TriConsumer<A, B, C> {

371          * Performs an operation on the given arguments.
372          *
373          * @param a the first argument
374          * @param b the second argument
375          * @param c the third argument
376          */
377         void accept(A a, B b, C c);

380     /**
381      * COM Exception handler. Logs at debug level for known-optional classes, otherwise warns.
382      *
383      * @param query a WmiQuery object
384      * @param ex    a FfmComException object
385      */
386     protected void handleComException(WbemcliUtilFFM.WmiQuery<?> query, FfmComException ex) {
387         String msg = "COM exception querying {}, which might not be on your system."
388                 + " Will not attempt to query it again. Error was {}: {}";
389         Object[] args = { query.getWmiClassName(), ex.getHresult(), ex.getMessage() };
390         if ("MSAcpi_ThermalZoneTemperature".equals(query.getWmiClassName())) {
391             LOG.debug(msg, args);
392         } else {
393             LOG.warn(msg, args);
394         }
395     }

397     /**
398      * Initializes COM library and sets security to impersonate the local user.
399      *
400      * @return True if COM was initialized and needs to be uninitialized, false otherwise
401      */
402     public boolean initCOM() {
403         int threading = getComThreading();
404         boolean comInit = initCOM(threading);
405         if (!comInit) {
406             int switched = switchComThreadingFrom(threading);
407             if (switched != threading) {
408                 comInit = initCOM(switched);
409             }
410         }
411         if (comInit && !securityInitialized) {
412             var hrOpt = Ole32FFM.CoInitializeSecurity(Ole32FFM.RPC_C_AUTHN_LEVEL_DEFAULT,
413                     Ole32FFM.RPC_C_IMP_LEVEL_IMPERSONATE, Ole32FFM.EOAC_NONE);
414             if (hrOpt.isPresent()) {
415                 int hr = hrOpt.getAsInt();
416                 if (Ole32FFM.succeeded(hr) || hr == Ole32FFM.RPC_E_TOO_LATE) {
417                     securityInitialized = true;
418                 } else {
419                     Ole32FFM.CoUninitialize();
420                     throw new FfmComException("Failed to initialize security.", hr);
421                 }
422             }
423         }
424         return comInit;
425     }

427     /**
428      * Initializes COM with a specific threading model.
429      *
430      * @param coInitThreading The threading model
431      * @return True if COM was initialized and needs to be uninitialized, false otherwise
432      */
433     protected boolean initCOM(int coInitThreading) {
434         var hrOpt = Ole32FFM.CoInitializeEx(coInitThreading);
435         if (hrOpt.isEmpty()) {
436             return false;
437         }
438         int hr = hrOpt.getAsInt();
439         return switch (hr) {
440             case Ole32FFM.S_OK, Ole32FFM.S_FALSE -> true;
441             case Ole32FFM.RPC_E_CHANGED_MODE -> false;
442             default -> throw new FfmComException("Failed to initialize COM library.", hr);
443         };
444     }

462     /**
463      * Switches the current threading model for COM initialization.
464      *
465      * @return The new threading model after switching
466      */
467     public synchronized int switchComThreading() {
468         if (comThreading == Ole32FFM.COINIT_APARTMENTTHREADED) {
469             comThreading = Ole32FFM.COINIT_MULTITHREADED;
470         } else {
471             comThreading = Ole32FFM.COINIT_APARTMENTTHREADED;
472         }
473         return comThreading;
474     }
475 
476     /**
477      * Switches the current threading model only if it still matches the expected value.
478      *
479      * @param expected the threading model observed before the failed initCOM attempt
480      * @return the new threading model if switched, or the current value if already switched by another thread
481      */
482     public synchronized int switchComThreadingFrom(int expected) {
483         if (comThreading != expected) {
484             return comThreading;
485         }
486         return switchComThreading();
487     }
