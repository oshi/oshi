5   package oshi.ffm.util.platform.windows;
6   
7   import java.util.Objects;
8   
9   import oshi.driver.common.windows.wmi.WmiQuery;
10  import oshi.driver.common.windows.wmi.WmiQueryExecutor;
11  import oshi.driver.common.windows.wmi.WmiResult;
12  
13  /**
14   * Adapts the FFM {@link WmiQueryHandlerFFM} to the common {@link WmiQueryExecutor} interface.
15   */
16  public class WmiQueryExecutorFFM implements WmiQueryExecutor {
17  
18      private final WmiQueryHandlerFFM handler;
19  
20      /**
21       * Creates an executor wrapping the given handler.
22       *
23       * @param handler the FFM WMI query handler
24       */
25      public WmiQueryExecutorFFM(WmiQueryHandlerFFM handler) {
26          this.handler = Objects.requireNonNull(handler, "handler");
27      }
28  
29      /**
30       * Creates an executor using a new default handler instance.
31       *
32       * @return a new executor, or {@code null} if handler creation fails
33       */
34      public static WmiQueryExecutorFFM createInstance() {
35          WmiQueryHandlerFFM h = WmiQueryHandlerFFM.createInstance();
36          return h == null ? null : new WmiQueryExecutorFFM(h);
37      }
38  
39      /**
40       * Returns the underlying FFM handler, for COM init/uninit operations.
41       *
42       * @return the FFM handler
43       */
44      public WmiQueryHandlerFFM getHandler() {
45          return handler;
46      }
47  
48      @Override
49      public <T extends Enum<T>> WmiResult<T> queryWMI(WmiQuery<T> query) {
50          return queryWMI(query, true);
51      }
52  
53      @Override
54      public <T extends Enum<T>> WmiResult<T> queryWMI(WmiQuery<T> query, boolean initCom) {
55          WbemcliUtilFFM.WmiQuery<T> ffmQuery = new WbemcliUtilFFM.WmiQuery<>(query.getNameSpace(),
56                  query.getWmiClassName(), query.getPropertyEnum());
57          WbemcliUtilFFM.WmiResult<T> ffmResult = handler.queryWMI(ffmQuery, initCom);
58          return new FfmWmiResult<>(ffmResult);
59      }
60  
61      private static final class FfmWmiResult<T extends Enum<T>> implements WmiResult<T> {
62          private final WbemcliUtilFFM.WmiResult<T> delegate;
63  
64          FfmWmiResult(WbemcliUtilFFM.WmiResult<T> delegate) {
65              this.delegate = delegate;
66          }
67  
68          @Override
69          public int getResultCount() {
70              return delegate.getResultCount();
71          }
72  
73          @Override
74          public Object getValue(T property, int index) {
75              return delegate.getValue(property, index);
76          }
77  
78          @Override
79          public int getVtType(T property) {
80              return delegate.getVtType(property);
81          }
82  
83          @Override
84          public int getCIMType(T property) {
85              return delegate.getCIMType(property);
86          }
87      }
88  }
