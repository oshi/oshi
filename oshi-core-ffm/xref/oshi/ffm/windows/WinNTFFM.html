64      // OSVERSIONINFOEX structure: 284 bytes
65      // dwOSVersionInfoSize(4) + dwMajorVersion(4) + dwMinorVersion(4) + dwBuildNumber(4) + dwPlatformId(4)
66      // + szCSDVersion(128 chars = 256 bytes) + wServicePackMajor(2) + wServicePackMinor(2)
67      // + wSuiteMask(2) + wProductType(1) + wReserved(1)
68      StructLayout OSVERSIONINFOEX = structLayout(JAVA_INT.withName("dwOSVersionInfoSize"),
69              JAVA_INT.withName("dwMajorVersion"), JAVA_INT.withName("dwMinorVersion"),
70              JAVA_INT.withName("dwBuildNumber"), JAVA_INT.withName("dwPlatformId"),
71              MemoryLayout.sequenceLayout(128, JAVA_CHAR).withName("szCSDVersion"),
72              JAVA_SHORT.withName("wServicePackMajor"), JAVA_SHORT.withName("wServicePackMinor"),
73              JAVA_SHORT.withName("wSuiteMask"), JAVA_BYTE.withName("wProductType"), JAVA_BYTE.withName("wReserved"));
74  
75      long OSVERSIONINFOEX_MAJOR_VERSION_OFFSET = OSVERSIONINFOEX
76              .byteOffset(MemoryLayout.PathElement.groupElement("dwMajorVersion"));
77      long OSVERSIONINFOEX_MINOR_VERSION_OFFSET = OSVERSIONINFOEX
78              .byteOffset(MemoryLayout.PathElement.groupElement("dwMinorVersion"));
79      long OSVERSIONINFOEX_SP_MAJOR_OFFSET = OSVERSIONINFOEX
80              .byteOffset(MemoryLayout.PathElement.groupElement("wServicePackMajor"));
81      long OSVERSIONINFOEX_PRODUCT_TYPE_OFFSET = OSVERSIONINFOEX
82              .byteOffset(MemoryLayout.PathElement.groupElement("wProductType"));
83  
84      StructLayout EVENTLOGRECORD = structLayout(JAVA_INT.withName("Length"), JAVA_INT.withName("Reserved"),

86              JAVA_INT.withName("EventID"), JAVA_SHORT.withName("EventType"), JAVA_SHORT.withName("NumStrings"),
87              JAVA_SHORT.withName("EventCategory"), JAVA_SHORT.withName("ReservedFlags"), MemoryLayout.paddingLayout(4),
88              JAVA_INT.withName("ClosingRecordNumber"), JAVA_INT.withName("StringOffset"),
