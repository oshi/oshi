5   package oshi.ffm.windows;
6   
7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   import static java.lang.foreign.ValueLayout.JAVA_LONG;
10  
11  import java.lang.foreign.MemoryLayout;
12  import java.lang.foreign.MemorySegment;
13  import java.lang.foreign.StructLayout;
14  import java.lang.foreign.SymbolLookup;
15  import java.lang.invoke.MethodHandle;
16  
17  public final class PsapiFFM extends WindowsForeignFunctions {
18  
19      private static final SymbolLookup PSAPI = lib("Psapi");
20  
21      private static final MethodHandle GetPerformanceInfo = downcall(PSAPI, "GetPerformanceInfo", JAVA_INT, ADDRESS,
22              JAVA_INT);
23  
24      // PERFORMANCE_INFORMATION struct (64-bit): DWORD cb + 10 SIZE_T + 3 DWORD
25      public static final StructLayout PERFORMANCE_INFORMATION_LAYOUT = MemoryLayout.structLayout(JAVA_INT.withName("cb"),
26              MemoryLayout.paddingLayout(4), JAVA_LONG.withName("CommitTotal"), JAVA_LONG.withName("CommitLimit"),
27              JAVA_LONG.withName("CommitPeak"), JAVA_LONG.withName("PhysicalTotal"),
28              JAVA_LONG.withName("PhysicalAvailable"), JAVA_LONG.withName("SystemCache"),
29              JAVA_LONG.withName("KernelTotal"), JAVA_LONG.withName("KernelPaged"), JAVA_LONG.withName("KernelNonpaged"),
30              JAVA_LONG.withName("PageSize"), JAVA_INT.withName("HandleCount"), JAVA_INT.withName("ProcessCount"),
31              JAVA_INT.withName("ThreadCount"), MemoryLayout.paddingLayout(4));
32  
33      public static boolean GetPerformanceInfo(MemorySegment pPerfInfo, int size) throws Throwable {
34          return isSuccess((int) GetPerformanceInfo.invokeExact(pPerfInfo, size));
35      }
36  }
