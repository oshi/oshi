146     /**
147      * Executes an operation in a confined arena, returning a default value if the operation throws.
148      * <p>
149      * This helper centralizes the common FFM call pattern where temporary native memory is scoped to a confined arena
150      * and failures from method-handle invocation or native binding are logged at the caller's chosen level. The
151      * provided arena is closed before this method returns, so returned objects must not depend on memory allocated from
152      * it.
153      *
154      * @param <T>          the return type
155      * @param callable     the operation to execute
156      * @param logger       the logger for the calling class
157      * @param level        the level at which to log thrown failures
158      * @param message      the message to log if the operation throws
159      * @param defaultValue the value to return if the operation throws
160      * @return the operation result, or {@code defaultValue} if the operation throws
161      */
162     public static <T> T callInArenaOrDefault(ArenaCallable<T> callable, Logger logger, Level level, String message,
163             T defaultValue) {
164         Objects.requireNonNull(callable, "callable");
165         try (Arena arena = Arena.ofConfined()) {
166             return callable.call(arena);
167         } catch (Throwable t) {
168             logThrowable(logger, level, message, t);
169             return defaultValue;
170         }
171     }

304     // ---- dlopen/dlsym escape hatch ----
305     // SymbolLookup.libraryLookup() can't pass custom flags; some platforms need them (notably AIX
306     // archive-member loading via RTLD_MEMBER). We bind dlopen/dlsym from libc directly so callers
307     // can dlopen with whatever flags they need and wrap the handle into a SymbolLookup.
308     //
309     // Bill Pugh holder pattern: these symbols are POSIX-only (Windows has LoadLibrary +
310     // GetProcAddress instead), so binding them eagerly in <clinit> on ForeignFunctions would break
311     // every Windows FFM class that extends this one. The holder class is only loaded on first
312     // access — i.e., on the first call to dlopenWithFlags — which Windows code paths never make.
313 
314     private static final class DlSyms {
315         static final MethodHandle DLOPEN = LINKER.downcallHandle(LINKER.defaultLookup().findOrThrow("dlopen"),
316                 FunctionDescriptor.of(java.lang.foreign.ValueLayout.ADDRESS, java.lang.foreign.ValueLayout.ADDRESS,
317                         java.lang.foreign.ValueLayout.JAVA_INT));
318 
319         static final MethodHandle DLSYM = LINKER.downcallHandle(LINKER.defaultLookup().findOrThrow("dlsym"),
320                 FunctionDescriptor.of(java.lang.foreign.ValueLayout.ADDRESS, java.lang.foreign.ValueLayout.ADDRESS,
321                         java.lang.foreign.ValueLayout.ADDRESS));
322     }
323 
324     /**
325      * Calls {@code dlopen(path, flags)} directly through the C runtime, returning a {@link SymbolLookup} over the
326      * loaded library. Use this when {@link SymbolLookup#libraryLookup(java.nio.file.Path, Arena)} can't pass the flag
327      * set the platform requires — most notably AIX's {@code RTLD_MEMBER} for archive-member loading.
328      * <p>
329      * The library is loaded into {@link Arena#global()} so the lookup persists for the JVM lifetime, mirroring how
330      * statically-bound system libraries are loaded.
331      *
332      * @param path  absolute path passed verbatim to {@code dlopen} (must include any platform-specific syntax, e.g.
333      *              {@code "/usr/lib/libperfstat.a(shr_64.o)"} on AIX)
334      * @param flags bitwise-OR of the {@code RTLD_*} flags from the platform's {@code <dlfcn.h>}
335      * @return a {@link SymbolLookup} backed by {@code dlsym} on the loaded library
336      * @throws UnsatisfiedLinkError if {@code dlopen} returns {@code NULL}
337      */
338     public static SymbolLookup dlopenWithFlags(String path, int flags) {
339         try {
340             Arena global = Arena.global();
341             MemorySegment pathSeg = global.allocateFrom(path);
342             MemorySegment handle = (MemorySegment) DlSyms.DLOPEN.invokeExact(pathSeg, flags);
343             if (handle.address() == 0L) {
344                 throw new UnsatisfiedLinkError("dlopen returned NULL for " + path);
345             }
346             return dlsymLookup(handle);
347         } catch (UnsatisfiedLinkError e) {
348             throw e;
349         } catch (Throwable t) {
350             throw new UnsatisfiedLinkError("dlopen failed for " + path + ": " + t.getMessage());
351         }
352     }
353 
354     /**
355      * Calls {@link #dlopenWithFlags(String, int)} on each path in turn, returning the first lookup that succeeds.
356      * Useful for libraries that may live under multiple names — e.g. AIX's 64-bit/32-bit shared-object members of
357      * {@code libperfstat.a}.
358      *
359      * @param paths candidate absolute paths to try, in order
360      * @param flags bitwise-OR of {@code RTLD_*} flags
361      * @return the first successful {@link SymbolLookup}
362      * @throws UnsatisfiedLinkError if every path's {@code dlopen} returns {@code NULL}
363      */
364     public static SymbolLookup dlopenFirstAvailable(java.util.List<String> paths, int flags) {
365         UnsatisfiedLinkError last = null;
366         for (String path : paths) {
367             try {
368                 return dlopenWithFlags(path, flags);
369             } catch (UnsatisfiedLinkError e) {
370                 last = e;
371             }
372         }
373         throw new UnsatisfiedLinkError(
374                 "dlopen returned NULL for every candidate: " + paths + (last == null ? "" : " (" + last + ")"));
375     }

377     private static SymbolLookup dlsymLookup(MemorySegment libHandle) {
378         return name -> {
379             try (Arena local = Arena.ofConfined()) {
380                 MemorySegment nameSeg = local.allocateFrom(name);
381                 MemorySegment sym = (MemorySegment) DlSyms.DLSYM.invokeExact(libHandle, nameSeg);
382                 return sym.address() == 0L ? java.util.Optional.empty()
383                         : java.util.Optional.of(MemorySegment.ofAddress(sym.address()));
384             } catch (Throwable _) {
385                 return java.util.Optional.empty();
386             }
387         };
388     }

390     /**
391      * Reinterpret a raw native pointer as a struct of the given layout, scoped to the provided arena.
392      *
393      * @param pointer the native pointer
394      * @param layout  the struct layout
395      * @param arena   the arena to scope the resulting segment to
396      * @return a memory segment over the struct, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
397      */
398     public static MemorySegment getStructFromNativePointer(MemorySegment pointer, StructLayout layout, Arena arena) {
399         if (pointer == null || pointer.equals(MemorySegment.NULL)) {
400             return null;
401         }
402         return MemorySegment.ofAddress(pointer.address()).reinterpret(layout.byteSize(), arena, null);
403     }

405     /**
406      * Read a null-terminated UTF-8 string from a raw native pointer.
407      *
408      * @param pointer the native pointer
409      * @param arena   the arena to scope the reinterpreted segment to
410      * @return the Java string, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
411      */
412     public static String getStringFromNativePointer(MemorySegment pointer, Arena arena) {
413         if (pointer == null || pointer.equals(MemorySegment.NULL)) {
414             return null;
415         }
416         // More than enough space for 255 UTF characters
417         return MemorySegment.ofAddress(pointer.address()).reinterpret(1024, arena, null).getString(0);
418     }

420     /**
421      * Copy {@code length} bytes from a raw native pointer into a Java byte array.
422      *
423      * @param pointer the native pointer
424      * @param length  the number of bytes to copy
425      * @param arena   the arena to scope the reinterpreted segment to
426      * @return the byte array, or {@code null} if the pointer is null or {@link MemorySegment#NULL}
427      */
428     public static byte[] getByteArrayFromNativePointer(MemorySegment pointer, long length, Arena arena) {
429         if (pointer == null || pointer.equals(MemorySegment.NULL)) {
430             return null;
431         }
432         MemorySegment bytesSegment = MemorySegment.ofAddress(pointer.address()).reinterpret(length, arena, null);
433         byte[] result = new byte[(int) length];
434         MemorySegment.copy(bytesSegment, JAVA_BYTE, 0, result, 0, (int) length);
435         return result;
436     }

448     /**
449      * Create a downcall handle for a symbol in a library.
450      *
451      * @param lib        the symbol lookup
452      * @param symbol     the symbol name
453      * @param resLayout  the return layout
454      * @param argLayouts the argument layouts
455      * @return the method handle
456      */
457     public static MethodHandle downcall(SymbolLookup lib, String symbol, MemoryLayout resLayout,
458             MemoryLayout... argLayouts) {
459         MemorySegment sym = lib.findOrThrow(symbol);
460         FunctionDescriptor fd = (resLayout == null) ? FunctionDescriptor.ofVoid(argLayouts)
461                 : FunctionDescriptor.of(resLayout, argLayouts);
462         return LINKER.downcallHandle(sym, fd);
463     }
