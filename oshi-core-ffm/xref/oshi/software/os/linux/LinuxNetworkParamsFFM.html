5   package oshi.software.os.linux;
6   
7   import static org.slf4j.event.Level.WARN;
8   import static oshi.ffm.ForeignFunctions.callInArenaOrDefault;
9   
10  import java.lang.foreign.MemorySegment;
11  import java.lang.foreign.ValueLayout;
12  import java.net.InetAddress;
13  import java.net.UnknownHostException;
14  
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  import oshi.ffm.linux.LinuxLibcFunctions;
19  import oshi.software.common.os.linux.LinuxNetworkParams;
20  
21  /**
22   * FFM-based Linux network parameters. Overrides {@code gethostname} and {@code getaddrinfo} calls to use FFM.
23   */
24  final class LinuxNetworkParamsFFM extends LinuxNetworkParams {
25  
26      private static final Logger LOG = LoggerFactory.getLogger(LinuxNetworkParamsFFM.class);
27  
28      private static final int HOST_NAME_MAX = 255;
29  
30      @Override
31      public String getHostName() {
32          String hostname = callInArenaOrDefault(arena -> {
33              MemorySegment buf = arena.allocate(HOST_NAME_MAX + 1L);
34              if (0 == LinuxLibcFunctions.gethostname(buf, HOST_NAME_MAX + 1L)) {
35                  return buf.getString(0);
36              }
37              return null;
38          }, LOG, WARN, "FFM gethostname failed", null);
39          return hostname == null ? super.getHostName() : hostname;
40      }
41  
42      @Override
43      public String getDomainName() {
44          String hostname;
45          try {
46              hostname = InetAddress.getLocalHost().getHostName();
47          } catch (UnknownHostException e) {
48              LOG.warn("Unknown host exception when getting address of local host: {}", e.getMessage());
49              return "";
50          }
51          String domainName = callInArenaOrDefault(arena -> {
52              // Build hints: ai_flags = AI_CANONNAME, rest zero
53              MemorySegment hints = arena.allocate(LinuxLibcFunctions.ADDRINFO_LAYOUT);
54              hints.set(ValueLayout.JAVA_INT, 0, LinuxLibcFunctions.AI_CANONNAME);
55  
56              // Output pointer-to-pointer: one ADDRESS slot
57              MemorySegment resPP = arena.allocate(ValueLayout.ADDRESS);
58  
59              MemorySegment nodeSeg = arena.allocateFrom(hostname);
60              int rc = LinuxLibcFunctions.getaddrinfo(nodeSeg, MemorySegment.NULL, hints, resPP);
61              if (rc != 0) {
62                  LOG.error("Failed getaddrinfo(): {}", LinuxLibcFunctions.gaiStrerror(rc, arena));
63                  return "";
64              }
65              // resPP holds the pointer to the first addrinfo struct
66              MemorySegment resPtr = resPP.get(ValueLayout.ADDRESS, 0);
67              try {
68                  String canonname = LinuxLibcFunctions.addrinfoCanoname(resPtr, arena);
69                  return canonname == null ? hostname : canonname.trim();
70              } finally {
71                  LinuxLibcFunctions.freeaddrinfo(resPtr);
72              }
73          }, LOG, WARN, "FFM getaddrinfo failed", null);
74          return domainName == null ? super.getDomainName() : domainName;
75      }
76  }
