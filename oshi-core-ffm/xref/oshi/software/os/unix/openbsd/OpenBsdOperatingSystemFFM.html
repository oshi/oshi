5   package oshi.software.os.unix.openbsd;
6   
7   import java.util.Map;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.ffm.ForeignFunctions;
14  import oshi.ffm.platform.unix.openbsd.OpenBsdLibcFunctions;
15  import oshi.ffm.util.platform.unix.openbsd.OpenBsdSysctlUtilFFM;
16  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
17  import oshi.software.common.os.unix.openbsd.OpenBsdOperatingSystem;
18  import oshi.software.os.FileSystem;
19  import oshi.software.os.OSProcess;
20  import oshi.util.LogLevel;
21  
22  /**
23   * FFM-backed OpenBSD operating system.
24   */
25  @ThreadSafe
26  public class OpenBsdOperatingSystemFFM extends OpenBsdOperatingSystem {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(OpenBsdOperatingSystemFFM.class);
29  
30      @Override
31      protected String querySysctl(int[] mib, String def) {
32          return OpenBsdSysctlUtilFFM.sysctl(mib, def);
33      }
34  
35      @Override
36      public FileSystem getFileSystem() {
37          return new OpenBsdFileSystemFFM();
38      }
39  
40      @Override
41      public int getProcessId() {
42          return ForeignFunctions.callInArenaIntOrDefault(arena -> OpenBsdLibcFunctions.getpid(), LOG, LogLevel.WARN,
43                  "getpid failed", 0);
44      }
45  
46      @Override
47      public int getThreadId() {
48          return ForeignFunctions.callInArenaIntOrDefault(arena -> OpenBsdLibcFunctions.getthrid(), LOG, LogLevel.WARN,
49                  "getthrid failed", 0);
50      }
51  
52      @Override
53      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
54          return new OpenBsdOSProcessFFM(pid, psMap, this);
55      }
56  }
