5   package oshi.software.os.unix.openbsd;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.ffm.util.platform.unix.openbsd.OpenBsdSysctlUtilFFM;
9   import oshi.software.common.os.unix.openbsd.OpenBsdFileSystem;
10  
11  /**
12   * FFM-backed OpenBSD file system. All file-store enumeration (mount/df parsing) lives on {@link OpenBsdFileSystem};
13   * only the file-descriptor sysctl reads are FFM-specific.
14   */
15  @ThreadSafe
16  public final class OpenBsdFileSystemFFM extends OpenBsdFileSystem {
17  
18      @Override
19      protected long querySysctl(String name) {
20          return OpenBsdSysctlUtilFFM.sysctl(name, 0);
21      }
22  }
