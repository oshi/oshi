31  /**
32   * FFM-backed FreeBSD operating system.
33   */
34  @ThreadSafe
35  public class FreeBsdOperatingSystemFFM extends FreeBsdOperatingSystem {
36  
37      private static final Logger LOG = LoggerFactory.getLogger(FreeBsdOperatingSystemFFM.class);
38  
39      @Override
40      protected String querySysctl(String name, String def) {
41          return BsdSysctlUtilFFM.sysctl(name, def);
42      }
43  
44      @Override
45      public FileSystem getFileSystem() {
46          return new FreeBsdFileSystemFFM();
47      }
48  
49      @Override
50      public InternetProtocolStats getInternetProtocolStats() {
51          return new FreeBsdInternetProtocolStatsFFM();
52      }
53  
54      @Override
55      public NetworkParams getNetworkParams() {
56          return new FreeBsdNetworkParamsFFM();
57      }
58  
59      @Override
60      protected List<OSSession> queryWhoSessions() {
61          return WhoFFM.queryUtxent();
62      }
63  
64      @Override
65      public int getProcessId() {
66          return callInArenaIntOrDefault(arena -> FreeBsdLibcFunctions.getpid(), LOG, Level.WARN, "getpid failed", 0);
67      }
68  
69      @Override
70      public int getThreadId() {
71          return callInArenaIntOrDefault(arena -> {
72              MemorySegment id = arena.allocate(JAVA_LONG);
73              if (FreeBsdLibcFunctions.thr_self(id) < 0) {
74                  return 0;
75              }
76              return (int) id.get(JAVA_LONG, 0);
77          }, LOG, Level.WARN, "thr_self failed", 0);
78      }
79  
80      @Override
81      protected OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap) {
82          return new FreeBsdOSProcessFFM(pid, psMap, this);
83      }
84  
85      @Override
86      protected long queryKernBoottimeSeconds() {
87          try (Arena arena = Arena.ofConfined()) {
88              // struct timeval: two longs (tv_sec, tv_usec) = 16 bytes on LP64 FreeBSD.
89              MemorySegment tv = arena.allocate(JAVA_LONG.byteSize() * 2);
90              if (BsdSysctlUtilFFM.sysctl("kern.boottime", tv) && tv.get(JAVA_LONG, 0) != 0L) {
91                  return tv.get(JAVA_LONG, 0);
92              }
93          }
94          return 0L;
95      }
96  }
