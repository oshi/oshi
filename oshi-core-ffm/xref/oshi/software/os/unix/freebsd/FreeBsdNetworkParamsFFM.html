5   package oshi.software.os.unix.freebsd;
6   
7   import static java.lang.foreign.ValueLayout.ADDRESS;
8   import static java.lang.foreign.ValueLayout.JAVA_INT;
9   import static oshi.ffm.ForeignFunctions.callInArenaOrDefault;
10  import static oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions.ADDRINFO_LAYOUT;
11  import static oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions.AI_CANONNAME;
12  import static oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions.HOST_NAME_MAX;
13  
14  import java.lang.foreign.MemorySegment;
15  
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  import oshi.annotation.concurrent.ThreadSafe;
20  import oshi.ffm.platform.unix.freebsd.FreeBsdLibcFunctions;
21  import oshi.software.common.os.unix.freebsd.FreeBsdNetworkParams;
22  import oshi.util.LogLevel;
23  
24  /**
25   * FFM-backed FreeBSD network params. Command-line gateway lookup and the {@code getHostName} fallback live on
26   * {@link FreeBsdNetworkParams}; only the libc bindings ({@code getaddrinfo}, {@code gethostname}) are FFM-specific.
27   */
28  @ThreadSafe
29  public class FreeBsdNetworkParamsFFM extends FreeBsdNetworkParams {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(FreeBsdNetworkParamsFFM.class);
32  
33      @Override
34      protected String queryDomainName() {
35          String hostname = getHostName();
36          if (hostname == null || hostname.isEmpty()) {
37              return "";
38          }
39          return callInArenaOrDefault(arena -> {
40              MemorySegment hint = arena.allocate(ADDRINFO_LAYOUT);
41              hint.set(JAVA_INT, 0, AI_CANONNAME);
42              MemorySegment node = arena.allocateFrom(hostname);
43              MemorySegment resPtr = arena.allocate(ADDRESS);
44              int rc = FreeBsdLibcFunctions.getaddrinfo(node, MemorySegment.NULL, hint, resPtr);
45              if (rc != 0) {
46                  if (LOG.isWarnEnabled()) {
47                      LOG.warn("Failed getaddrinfo(): {}", FreeBsdLibcFunctions.gaiStrerror(rc, arena));
48                  }
49                  return "";
50              }
51              MemorySegment res = resPtr.get(ADDRESS, 0);
52              try {
53                  String canonname = FreeBsdLibcFunctions.addrinfoCanonname(res, arena);
54                  return canonname == null ? "" : canonname.trim();
55              } finally {
56                  FreeBsdLibcFunctions.freeaddrinfo(res);
57              }
58          }, LOG, LogLevel.WARN, "Failed to query domain name", "");
59      }
60  
61      @Override
62      protected String queryHostName() {
63          return callInArenaOrDefault(arena -> {
64              MemorySegment buf = arena.allocate(HOST_NAME_MAX + 1L);
65              if (0 != FreeBsdLibcFunctions.gethostname(buf, HOST_NAME_MAX + 1L)) {
66                  return null;
67              }
68              return buf.getString(0);
69          }, LOG, LogLevel.WARN, "Failed to get hostname", null);
70      }
71  }
