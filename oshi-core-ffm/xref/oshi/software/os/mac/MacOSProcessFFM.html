99      private static final int ARGMAX = sysctl("kern.argmax", 0);
100     private static final long TICKS_PER_MS;
101     static {
102         // default to 1 tick per nanosecond
103         long ticksPerSec = 1_000_000_000L;
104         IOIterator iter = IOKitUtilFFM.getMatchingServices("IOPlatformDevice");
105         if (iter != null) {
106             IORegistryEntry cpu = iter.next();
107             while (cpu != null) {
108                 try (IORegistryEntry current = cpu) {
109                     String s = current.getName().toLowerCase(Locale.ROOT);
110                     if (s.startsWith("cpu") && s.length() > 3) {
111                         // Frequency is typically only on lowest-numbered P-core
112                         byte[] data = current.getByteArrayProperty("timebase-frequency");
113                         if (data != null) {
114                             ticksPerSec = ParseUtil.byteArrayToLong(data, 4, false);
115                             break;
116                         }
117                     }
118                 }
119                 cpu = iter.next();
120             }
121             iter.release();
122         }
123         // Convert to ticks per millisecond
124         TICKS_PER_MS = ticksPerSec / 1000L;
125     }

132     @Override
133     protected Pair<List<String>, Map<String, String>> queryArgsAndEnvironment() {
134         int pid = getProcessID();
135         // Set up return objects
136         List<String> args = new ArrayList<>();
137         // API does not specify any particular order of entries, but it is reasonable to
138         // maintain whatever order the OS provided to the end user
139         Map<String, String> env = new LinkedHashMap<>();
140 
141         // Get command line via sysctl
142         int[] mib = { 1, 49, pid }; // CTL_KERN, KERN_PROCARGS2, pid
143         try (Arena arena = Arena.ofConfined()) {
144             MemorySegment procargs = arena.allocate(ARGMAX);
145             procargs.fill((byte) 0);
146 
147             long size = sysctl(mib, procargs);
148             if (size > 0) {
149                 // Procargs contains an int representing total # of args, followed by a
150                 // null-terminated execpath string and then the arguments, each
151                 // null-terminated (possible multiple consecutive nulls),
152                 // The execpath string is also the first arg.
153                 // Following this is an int representing total # of env, followed by
154                 // null-terminated envs in similar format
155                 int nargs = procargs.get(JAVA_INT, 0);
156                 // Sanity check
157                 if (nargs > 0 && nargs <= 1024) {
158                     // Skip first int (containing value of nargs)
159                     long offset = Integer.BYTES;
160                     // Skip exec_command
161                     offset += procargs.getString(offset).length();
162                     // Iterate character by character using offset
163                     // Build each arg and add to list
164                     while (offset < size) {
165                         // Skip null bytes
166                         while (offset < size && procargs.get(JAVA_BYTE, offset) == 0) {
167                             offset++;
168                         }
169                         if (offset >= size) {
170                             break;
171                         }
172                         // Read string until null terminator
173                         String arg = procargs.getString(offset);
174                         if (arg.isEmpty()) {
175                             break;
176                         }
177                         if (nargs-- > 0) {
178                             // Still processing arguments
179                             args.add(arg);
180                         } else {
181                             // Processing environment variables
182                             int idx = arg.indexOf('=');
183                             if (idx > 0) {
184                                 env.put(arg.substring(0, idx), arg.substring(idx + 1));
185                             }
186                         }
187                         offset += arg.length() + 1; // +1 for null terminator
188                     }
189                 }
190             } else {
191                 // Don't warn for pid 0
192                 if (pid > 0 && LOG_MAC_SYSCTL_WARNING) {
193                     LOG.warn("Failed sysctl call for process arguments (kern.procargs2), process {} may not exist.",
194                             pid);
195                 }
196             }
197         }
198         return new Pair<>(Collections.unmodifiableList(args), Collections.unmodifiableMap(env));
199     }

206     @Override
207     public long getSoftOpenFileLimit() {
208         if (getProcessID() == this.os.getProcessId()) {
209             return callInArenaLongOrDefault(arena -> {
210                 MemorySegment buffer = arena.allocate(RLIMIT);
211                 int result = getrlimit(MAC_RLIMIT_NOFILE, buffer);
212                 if (result == 0) {
213                     return buffer.get(JAVA_LONG, RLIMIT.byteOffset(RLIM_CUR));
214                 }
215                 return 0L;
216             }, LOG, DEBUG, "Failed to query soft open file limit", 0L);
217         }
218         return -1L; // not supported
219     }

236     @Override
237     public boolean updateAttributes() {
238         long now = System.currentTimeMillis();
239         int pid = getProcessID();
240         boolean updated = callInArenaBooleanOrDefault(arena -> {
241             MemorySegment taskAllInfo = arena.allocate(PROC_TASK_ALL_INFO);
242             int infoResult = proc_pidinfo(pid, PROC_PIDTASKALLINFO, 0L, taskAllInfo,
243                     (int) PROC_TASK_ALL_INFO.byteSize());
244             if (infoResult <= 0) {
245                 return false;
246             }
247             MemorySegment ptinfo = taskAllInfo.asSlice(PROC_TASK_ALL_INFO.byteOffset(PTINFO),
248                     PROC_TASK_INFO.byteSize());
249             if (ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_THREADNUM)) < 1) {
250                 return false;
251             }
252             MemorySegment pbsd = taskAllInfo.asSlice(PROC_TASK_ALL_INFO.byteOffset(PBSD), PROC_BSD_INFO.byteSize());
253 
254             // Get process path
255             MemorySegment pathBuf = arena.allocate(PROC_PIDPATHINFO_MAXSIZE);
256             if (proc_pidpath(pid, pathBuf, PROC_PIDPATHINFO_MAXSIZE) > 0) {
257                 this.path = pathBuf.getString(0).trim();
258                 // Overwrite name with last part of path
259                 String[] pathSplit = this.path.split("/");
260                 if (pathSplit.length > 0) {
261                     this.name = pathSplit[pathSplit.length - 1];
262                 }
263             }
264             if (this.name.isEmpty()) {
265                 // pbi_comm contains first 16 characters of name
266                 this.name = readFixedWidthString(pbsd, PROC_BSD_INFO.byteOffset(PBI_COMM), MAXCOMLEN);
267             }
268 
269             // Get Process state based on status
270             int status = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_STATUS));
271             this.state = stateFromStatus(status);
272 
273             // User and group info
274             this.parentProcessID = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_PPID));
275             int uid = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_UID));
276             this.userID = Integer.toString(uid);
277             MemorySegment pwuid = getpwuid(uid);
278             if (pwuid != null) {
279                 MemorySegment passwdStruct = ForeignFunctions.getStructFromNativePointer(pwuid, PASSWD, arena);
280                 MemorySegment nameAddress = passwdStruct.get(ADDRESS, PASSWD.byteOffset(groupElement("pw_name")));
281                 this.user = ForeignFunctions.getStringFromNativePointer(nameAddress, arena);
282             } else {
283                 this.user = this.userID;
284             }
285 
286             int gid = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_GID));
287             this.groupID = Integer.toString(gid);
288             MemorySegment grgid = getgrgid(gid);
289             if (grgid != null) {
290                 MemorySegment groupStruct = ForeignFunctions.getStructFromNativePointer(grgid, GROUP, arena);
291                 MemorySegment nameAddress = groupStruct.get(ADDRESS, GROUP.byteOffset(groupElement("gr_name")));
292                 this.group = ForeignFunctions.getStringFromNativePointer(nameAddress, arena);
293             } else {
294                 this.group = this.groupID;
295             }
296 
297             // Process metrics
298             this.threadCount = ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_THREADNUM));
299             this.priority = ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_PRIORITY));
300             this.virtualSize = ptinfo.get(JAVA_LONG, PROC_TASK_INFO.byteOffset(PTI_VIRTUAL_SIZE));
301             this.residentSetSize = ptinfo.get(JAVA_LONG, PROC_TASK_INFO.byteOffset(PTI_RESIDENT_SIZE));
302             // Default/fallback: RSS. Will be overwritten by phys_footprint when available.
303             this.memoryFootprint = this.residentSetSize;
304             this.kernelTime = ptinfo.get(JAVA_LONG, PROC_TASK_INFO.byteOffset(PTI_TOTAL_SYSTEM)) / TICKS_PER_MS;
305             this.userTime = ptinfo.get(JAVA_LONG, PROC_TASK_INFO.byteOffset(PTI_TOTAL_USER)) / TICKS_PER_MS;
306 
307             long startSec = pbsd.get(JAVA_LONG, PROC_BSD_INFO.byteOffset(PBI_START_TVSEC));
308             long startUsec = pbsd.get(JAVA_LONG, PROC_BSD_INFO.byteOffset(PBI_START_TVUSEC));
309             this.startTime = startSec * 1000L + startUsec / 1000L;
310             this.upTime = now - this.startTime;
311 
312             this.openFiles = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_NFILES));
313             int flags = pbsd.get(JAVA_INT, PROC_BSD_INFO.byteOffset(PBI_FLAGS));
314             this.bitness = (flags & P_LP64) == 0 ? 32 : 64;
315 
316             this.majorFaults = ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_PAGEINS));
317             int totalFaults = ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_FAULTS));
318             // testing using getrusage confirms pti_faults includes both major and minor
319             this.minorFaults = totalFaults - this.majorFaults;
320             this.contextSwitches = ptinfo.get(JAVA_INT, PROC_TASK_INFO.byteOffset(PTI_CSW));
321             // getrusage(RUSAGE_SELF) aggregates across all threads for current process
322             if (getProcessID() == this.os.getProcessId()) {
323                 try {
324                     MemorySegment rusageSegment = arena.allocate(MacSystemFunctions.RUSAGE_SIZE);
325                     if (0 == MacSystemFunctions.getrusage(MacSystemFunctions.RUSAGE_SELF, rusageSegment)) {
326                         this.voluntaryContextSwitches = rusageSegment.get(JAVA_LONG,
327                                 MacSystemFunctions.RUSAGE_NVCSW_OFFSET);
328                         this.involuntaryContextSwitches = rusageSegment.get(JAVA_LONG,
329                                 MacSystemFunctions.RUSAGE_NIVCSW_OFFSET);
330                         this.contextSwitches = this.voluntaryContextSwitches + this.involuntaryContextSwitches;
331                     } else {
332                         this.voluntaryContextSwitches = 0L;
333                         this.involuntaryContextSwitches = 0L;
334                     }
335                 } catch (Throwable t) {
336                     LOG.debug("FFM getrusage failed: {}", t.toString());
337                     this.voluntaryContextSwitches = 0L;
338                     this.involuntaryContextSwitches = 0L;
339                 }
340             }
341 
342             // Get rusage info for newer OS versions
343             if (this.majorVersion > 10 || (this.majorVersion == 10 && this.minorVersion >= 9)) {
344                 MemorySegment rusage = arena.allocate(RUSAGEINFOV2);
345                 if (0 == proc_pid_rusage(pid, RUSAGE_INFO_V2, rusage)) {
346                     this.bytesRead = rusage.get(JAVA_LONG, RUSAGEINFOV2.byteOffset(RI_DISKIO_BYTESREAD));
347                     this.bytesWritten = rusage.get(JAVA_LONG, RUSAGEINFOV2.byteOffset(RI_DISKIO_BYTESWRITTEN));
348                     this.memoryFootprint = rusage.get(JAVA_LONG, RUSAGEINFOV2.byteOffset(RI_PHYS_FOOTPRINT));
349                 }
350             }
351 
352             // Get working directory info
353             MemorySegment vnodeInfo = arena.allocate(VNODE_PATH_INFO);
354             if (proc_pidinfo(pid, PROC_PIDVNODEPATHINFO, 0L, vnodeInfo, (int) VNODE_PATH_INFO.byteSize()) > 0) {
355                 // Get the current directory path using nested path elements
356                 this.currentWorkingDirectory = vnodeInfo.asSlice(VNODE_PATH_INFO.byteOffset(PVI_CDIR))
357                         .asSlice(VNODE_INFO_PATH.byteOffset(VIP_PATH), MAXPATHLEN).getString(0);
358             }
359             return true;
360         }, LOG, TRACE, "Failed to update process attributes", false);
361         if (!updated) {
362             this.state = INVALID;
363         }
364         return updated;
365     }
