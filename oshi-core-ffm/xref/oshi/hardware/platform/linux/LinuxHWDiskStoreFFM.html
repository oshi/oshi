44      /**
45       * Gets the disks on this machine
46       *
47       * @return a list of {@link HWDiskStore} objects representing the disks
48       */
49      public static List<HWDiskStore> getDisks() {
50          return getDisks(null);
51      }

53      static List<HWDiskStore> getDisks(LinuxHWDiskStore storeToUpdate) {
54          if (!HAS_UDEV) {
55              if (storeToUpdate == null) {
56                  LOG.warn("Disk Store information requires libudev, which is not present.");
57              }
58              return Collections.emptyList();
59          }
60          LinuxHWDiskStoreFFM store = null;
61          List<HWDiskStore> result = new ArrayList<>();
62          Map<String, String> mountsMap = readMountsMap();
63  
64          try (Arena arena = Arena.ofConfined()) {
65              MemorySegment udev = UdevFunctions.udev_new();
66              if (MemorySegment.NULL.equals(udev)) {
67                  if (storeToUpdate == null) {
68                      LOG.warn("Failed to create udev context for disk store enumeration.");
69                  }
70                  return Collections.emptyList();
71              }
72              try {
73                  MemorySegment enumerate = UdevFunctions.udev_enumerate_new(udev);
74                  try {
75                      UdevFunctions.addMatchSubsystem(enumerate, BLOCK, arena);
76                      UdevFunctions.udev_enumerate_scan_devices(enumerate);
77                      for (MemorySegment entry = UdevFunctions
78                              .udev_enumerate_get_list_entry(enumerate); !MemorySegment.NULL
79                                      .equals(entry); entry = UdevFunctions.udev_list_entry_get_next(entry)) {
80                          String syspath = UdevFunctions.getString(UdevFunctions.udev_list_entry_get_name(entry), arena);
81                          if (syspath == null) {
82                              continue;
83                          }
84                          MemorySegment device = UdevFunctions.deviceNewFromSyspath(udev, syspath, arena);
85                          if (MemorySegment.NULL.equals(device)) {
86                              continue;
87                          }
88                          try {
89                              String devnode = UdevFunctions.getString(UdevFunctions.udev_device_get_devnode(device),
90                                      arena);
91                              if (devnode != null && !devnode.startsWith(DevPath.LOOP)
92                                      && !devnode.startsWith(DevPath.RAM)) {
93                                  String devtype = UdevFunctions.getString(UdevFunctions.udev_device_get_devtype(device),
94                                          arena);
95                                  if (DISK.equals(devtype)) {
96                                      String devModel = UdevFunctions.getPropertyValue(device, ID_MODEL, arena);
97                                      String devSerial = UdevFunctions.getPropertyValue(device, ID_SERIAL_SHORT, arena);
98                                      long devSize = ParseUtil.parseLongOrDefault(
99                                              UdevFunctions.getSysattrValue(device, SIZE, arena), 0L) * SECTORSIZE;
100                                     if (devnode.startsWith(DevPath.DM)) {
101                                         devModel = LOGICAL_VOLUME_GROUP;
102                                         devSerial = UdevFunctions.getPropertyValue(device, DM_UUID, arena);
103                                         store = new LinuxHWDiskStoreFFM(devnode, devModel,
104                                                 devSerial == null ? Constants.UNKNOWN : devSerial, devSize, "Virtual");
105                                         if (devSerial != null && devSerial.startsWith("LVM-")) {
106                                             String vgName = UdevFunctions.getPropertyValue(device, DM_VG_NAME, arena);
107                                             String lvName = UdevFunctions.getPropertyValue(device, DM_LV_NAME, arena);
108                                             String fsType = UdevFunctions.getPropertyValue(device, ID_FS_TYPE, arena);
109                                             String fsUuid = UdevFunctions.getPropertyValue(device, ID_FS_UUID, arena);
110                                             String fsLabel = UdevFunctions.getPropertyValue(device, ID_FS_LABEL, arena);
111                                             String sysname = UdevFunctions
112                                                     .getString(UdevFunctions.udev_device_get_sysname(device), arena);
113                                             store.getMutablePartitionList().add(new HWPartition(
114                                                     getPartitionNameForDmDevice(vgName, lvName), sysname,
115                                                     fsType == null ? PARTITION : fsType, fsUuid == null ? "" : fsUuid,
116                                                     fsLabel == null ? "" : fsLabel,
117                                                     ParseUtil.parseLongOrDefault(
118                                                             UdevFunctions.getSysattrValue(device, SIZE, arena), 0L)
119                                                             * SECTORSIZE,
120                                                     ParseUtil.parseIntOrDefault(
121                                                             UdevFunctions.getPropertyValue(device, MAJOR, arena), 0),
122                                                     ParseUtil.parseIntOrDefault(
123                                                             UdevFunctions.getPropertyValue(device, MINOR, arena), 0),
124                                                     getMountPointOfDmDevice(vgName, lvName)));
125                                         }
126                                     } else {
127                                         store = new LinuxHWDiskStoreFFM(devnode,
128                                                 devModel == null ? Constants.UNKNOWN : devModel,
129                                                 devSerial == null ? Constants.UNKNOWN : devSerial, devSize,
130                                                 detectDiskType(device, arena));
131                                     }
132                                     if (storeToUpdate == null) {
133                                         computeDiskStats(store, UdevFunctions.getSysattrValue(device, STAT, arena));
134                                         result.add(store);
135                                     } else if (store.getName().equals(storeToUpdate.getName())
136                                             && store.getModel().equals(storeToUpdate.getModel())
137                                             && store.getSerial().equals(storeToUpdate.getSerial())
138                                             && store.getSize() == storeToUpdate.getSize()) {
139                                         computeDiskStats(storeToUpdate,
140                                                 UdevFunctions.getSysattrValue(device, STAT, arena));
141                                         result.add(storeToUpdate);
142                                         break;
143                                     }
144                                 } else if (storeToUpdate == null && store != null && PARTITION.equals(devtype)) {
145                                     MemorySegment parent = UdevFunctions.getParentWithSubsystemDevtype(device, BLOCK,
146                                             DISK, arena);
147                                     if (!MemorySegment.NULL.equals(parent)) {
148                                         String parentDevnode = UdevFunctions
149                                                 .getString(UdevFunctions.udev_device_get_devnode(parent), arena);
150                                         if (store.getName().equals(parentDevnode)) {
151                                             String name = devnode;
152                                             String sysname = UdevFunctions
153                                                     .getString(UdevFunctions.udev_device_get_sysname(device), arena);
154                                             String sysPath = UdevFunctions
155                                                     .getString(UdevFunctions.udev_device_get_syspath(device), arena);
156                                             String fsType = UdevFunctions.getPropertyValue(device, ID_FS_TYPE, arena);
157                                             String fsUuid = UdevFunctions.getPropertyValue(device, ID_FS_UUID, arena);
158                                             String fsLabel = UdevFunctions.getPropertyValue(device, ID_FS_LABEL, arena);
159                                             store.getMutablePartitionList().add(new HWPartition(name, sysname,
160                                                     fsType == null ? PARTITION : fsType, fsUuid == null ? "" : fsUuid,
161                                                     fsLabel == null ? "" : fsLabel,
162                                                     ParseUtil.parseLongOrDefault(
163                                                             UdevFunctions.getSysattrValue(device, SIZE, arena), 0L)
164                                                             * SECTORSIZE,
165                                                     ParseUtil.parseIntOrDefault(
166                                                             UdevFunctions.getPropertyValue(device, MAJOR, arena), 0),
167                                                     ParseUtil.parseIntOrDefault(
168                                                             UdevFunctions.getPropertyValue(device, MINOR, arena), 0),
169                                                     mountsMap.getOrDefault(name,
170                                                             getDependentNamesFromHoldersDirectory(sysPath))));
171                                         }
172                                     }
173                                 }
174                             }
175                         } finally {
176                             UdevFunctions.udev_device_unref(device);
177                         }
178                     }
179                 } finally {
180                     UdevFunctions.udev_enumerate_unref(enumerate);
181                 }
182             } finally {
183                 UdevFunctions.udev_unref(udev);
184             }
185         } catch (Throwable e) {
186             if (storeToUpdate == null) {
187                 LOG.warn("Error enumerating disk stores: {}", e.toString());
188             }
189             return Collections.emptyList();
190         }
191         finalizePartitions(result);
192         return result;
193     }
194 
195     @Override
196     public boolean updateAttributes() {
197         return !getDisks(this).isEmpty();
198     }
199 
200     private static String detectDiskType(MemorySegment device, Arena arena) {
201         try {
202             String removable = UdevFunctions.getSysattrValue(device, "removable", arena);
203             if ("1".equals(removable)) {
204                 return "Removable";
205             }
206             String rotational = UdevFunctions.getSysattrValue(device, "queue/rotational", arena);
207             if ("0".equals(rotational)) {
208                 return "SSD";
209             } else if ("1".equals(rotational)) {
210                 return "HDD";
211             }
212         } catch (Throwable t) { // NOSONAR
213             LOG.debug("Unable to read disk type sysattr", t);
214         }
215         return "Unknown";
216     }
