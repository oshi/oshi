25  /**
26   * FFM-based Linux network interface implementation.
27   */
28  @ThreadSafe
29  public final class LinuxNetworkIFFFM extends LinuxNetworkIF {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(LinuxNetworkIFFFM.class);
32  
33      LinuxNetworkIFFFM(NetworkInterface netint) throws InstantiationException {
34          super(netint, queryIfModel(netint));
35      }
36  
37      private static String queryIfModel(NetworkInterface netint) {
38          String name = netint.getName();
39          if (!HAS_UDEV) {
40              return queryIfModelFromSysfs(name);
41          }
42          return callInArenaOrDefault(arena -> {
43              MemorySegment udev = UdevFunctions.udev_new();
44              if (MemorySegment.NULL.equals(udev)) {
45                  return queryIfModelFromSysfs(name);
46              }
47              // wrapped only to release the native handle on close
48              try (var _ = NativeHandle.of(udev, UdevFunctions::udev_unref)) {
49                  MemorySegment device = UdevFunctions.deviceNewFromSyspath(udev, SysPath.NET + name, arena);
50                  if (!MemorySegment.NULL.equals(device)) {
51                      // wrapped only to release the native handle on close
52                      try (var _ = NativeHandle.of(device, UdevFunctions::udev_device_unref)) {
53                          String devVendor = UdevFunctions.getPropertyValue(device, "ID_VENDOR_FROM_DATABASE", arena);
54                          String devModel = UdevFunctions.getPropertyValue(device, "ID_MODEL_FROM_DATABASE", arena);
55                          return formatModel(devVendor, devModel, name);
56                      }
57                  }
58              }
59              return name;
60          }, LOG, WARN, "Error querying network interface model for " + name, name);
61      }
62  
63      /**
64       * Gets network interfaces on this machine
65       *
66       * @param includeLocalInterfaces include local interfaces in the result
67       * @return A list of {@link NetworkIF} objects representing the interfaces
68       */
69      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
70          return getNetworks(includeLocalInterfaces, LinuxNetworkIFFFM::new);
71      }
72  }
