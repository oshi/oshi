37  final class WindowsLogicalVolumeGroupFFM extends AbstractLogicalVolumeGroup {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(WindowsLogicalVolumeGroupFFM.class);
40  
41      private static final Pattern SP_OBJECT_ID = Pattern.compile(".*ObjectId=.*SP:(\\{.*\\}).*");
42      private static final Pattern PD_OBJECT_ID = Pattern.compile(".*ObjectId=.*PD:(\\{.*\\}).*");
43      private static final Pattern VD_OBJECT_ID = Pattern.compile(".*ObjectId=.*VD:(\\{.*\\})(\\{.*\\}).*");
44  
45      private static final boolean IS_WINDOWS8_OR_GREATER = VersionHelpersFFM.IsWindows8OrGreater();
46  
47      WindowsLogicalVolumeGroupFFM(String name, Map<String, Set<String>> lvMap, Set<String> pvSet) {
48          super(name, lvMap, pvSet);
49      }
50  
51      static List<LogicalVolumeGroup> getLogicalVolumeGroups() {
52          if (!IS_WINDOWS8_OR_GREATER) {
53              return Collections.emptyList();
54          }
55          WmiQueryHandlerFFM h = Objects.requireNonNull(WmiQueryHandlerFFM.createInstance());
56          boolean comInit = false;
57          try {
58              comInit = h.initCOM();
59              WmiResult<StoragePoolProperty> sp = MSFTStorageFFM.queryStoragePools(h);
60              int count = sp.getResultCount();
61              if (count == 0) {
62                  return Collections.emptyList();
63              }
64  
65              Map<String, String> vdMap = new HashMap<>();
66              WmiResult<VirtualDiskProperty> vds = MSFTStorageFFM.queryVirtualDisks(h);
67              count = vds.getResultCount();
68              for (int i = 0; i < count; i++) {
69                  String vdObjectId = WmiUtilFFM.getString(vds, VirtualDiskProperty.OBJECTID, i);
70                  Matcher m = VD_OBJECT_ID.matcher(vdObjectId);
71                  if (m.matches()) {
72                      vdObjectId = m.group(2) + " " + m.group(1);
73                  }
74                  vdMap.put(vdObjectId, WmiUtilFFM.getString(vds, VirtualDiskProperty.FRIENDLYNAME, i));
75              }
76  
77              Map<String, Pair<String, String>> pdMap = new HashMap<>();
78              WmiResult<PhysicalDiskProperty> pds = MSFTStorageFFM.queryPhysicalDisks(h);
79              count = pds.getResultCount();
80              for (int i = 0; i < count; i++) {
81                  String pdObjectId = WmiUtilFFM.getString(pds, PhysicalDiskProperty.OBJECTID, i);
82                  Matcher m = PD_OBJECT_ID.matcher(pdObjectId);
83                  if (m.matches()) {
84                      pdObjectId = m.group(1);
85                  }
86                  pdMap.put(pdObjectId, new Pair<>(WmiUtilFFM.getString(pds, PhysicalDiskProperty.FRIENDLYNAME, i),
87                          WmiUtilFFM.getString(pds, PhysicalDiskProperty.PHYSICALLOCATION, i)));
88              }
89  
90              Map<String, String> sppdMap = new HashMap<>();
91              WmiResult<StoragePoolToPhysicalDiskProperty> sppd = MSFTStorageFFM.queryStoragePoolPhysicalDisks(h);
92              count = sppd.getResultCount();
93              for (int i = 0; i < count; i++) {
94                  String spObjectId = WmiUtilFFM.getRefString(sppd, StoragePoolToPhysicalDiskProperty.STORAGEPOOL, i);
95                  Matcher m = SP_OBJECT_ID.matcher(spObjectId);
96                  if (m.matches()) {
97                      spObjectId = m.group(1);
98                  }
99                  String pdObjectId = WmiUtilFFM.getRefString(sppd, StoragePoolToPhysicalDiskProperty.PHYSICALDISK, i);
100                 m = PD_OBJECT_ID.matcher(pdObjectId);
101                 if (m.matches()) {
102                     pdObjectId = m.group(1);
103                 }
104                 sppdMap.put(spObjectId + " " + pdObjectId, pdObjectId);
105             }
106 
107             List<LogicalVolumeGroup> lvgList = new ArrayList<>();
108             count = sp.getResultCount();
109             for (int i = 0; i < count; i++) {
110                 String name = WmiUtilFFM.getString(sp, StoragePoolProperty.FRIENDLYNAME, i);
111                 String spObjectId = WmiUtilFFM.getString(sp, StoragePoolProperty.OBJECTID, i);
112                 Matcher m = SP_OBJECT_ID.matcher(spObjectId);
113                 if (m.matches()) {
114                     spObjectId = m.group(1);
115                 }
116                 Set<String> physicalVolumeSet = new HashSet<>();
117                 for (Entry<String, String> entry : sppdMap.entrySet()) {
118                     if (entry.getKey().contains(spObjectId)) {
119                         String pdObjectId = entry.getValue();
120                         Pair<String, String> nameLoc = pdMap.get(pdObjectId);
121                         if (nameLoc != null) {
122                             physicalVolumeSet.add(nameLoc.getA() + " @ " + nameLoc.getB());
123                         }
124                     }
125                 }
126                 Map<String, Set<String>> logicalVolumeMap = new HashMap<>();
127                 for (Entry<String, String> entry : vdMap.entrySet()) {
128                     if (entry.getKey().contains(spObjectId)) {
129                         String vdObjectId = ParseUtil.whitespaces.split(entry.getKey())[0];
130                         logicalVolumeMap.put(entry.getValue() + " " + vdObjectId, physicalVolumeSet);
131                     }
132                 }
133                 lvgList.add(new WindowsLogicalVolumeGroupFFM(name, logicalVolumeMap, physicalVolumeSet));
134             }
135             return lvgList;
136         } catch (FfmComException e) {
137             LOG.warn("COM exception: {}", e.getMessage());
138             return Collections.emptyList();
139         } finally {
140             if (comInit) {
141                 h.unInitCOM();
142             }
143         }
144     }
