106     @Override
107     protected ProcessorIdentifier queryProcessorId() {
108         String cpuVendor = "";
109         String cpuName = "";
110         String cpuIdentifier = "";
111         String cpuFamily = "";
112         String cpuModel = "";
113         String cpuStepping = "";
114         long cpuVendorFreq = 0L;
115         String processorID;
116         boolean cpu64bit = false;
117 
118         final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
119         try (Arena arena = Arena.ofConfined()) {
120             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
121             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, cpuRegistryRoot);
122             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
123                 MemorySegment openedKey = hKey.get(ValueLayout.ADDRESS, 0);
124                 try {
125                     // Get first subkey name
126                     MemorySegment nameBuffer = arena.allocate(256 * 2L); // WCHAR[256]
127                     MemorySegment nameLen = arena.allocate(ValueLayout.JAVA_INT);
128                     nameLen.set(ValueLayout.JAVA_INT, 0, 256);
129                     if (Advapi32FFM.RegEnumKeyEx(openedKey, 0, nameBuffer, nameLen, MemorySegment.NULL,
130                             MemorySegment.NULL, MemorySegment.NULL, MemorySegment.NULL) == 0) {
131                         String firstKey = WindowsForeignFunctions.readWideString(nameBuffer);
132                         String cpuRegistryPath = cpuRegistryRoot + firstKey;
133                         cpuVendor = registryGetString(arena, cpuRegistryPath, "VendorIdentifier");
134                         cpuName = registryGetString(arena, cpuRegistryPath, "ProcessorNameString");
135                         cpuIdentifier = registryGetString(arena, cpuRegistryPath, "Identifier");
136                         cpuVendorFreq = registryGetDword(arena, cpuRegistryPath, "~MHz") * 1_000_000L;
137                     }
138                 } finally {
139                     Advapi32FFM.RegCloseKey(openedKey);
140                 }
141             }
142         } catch (Throwable t) {
143             LOG.debug("Failed to read processor registry info: {}", t.getMessage());
144         }
145 
146         if (!cpuIdentifier.isEmpty()) {
147             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
148             cpuModel = parseIdentifier(cpuIdentifier, "Model");
149             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
150         }
151 
152         // GetNativeSystemInfo to determine 64-bit
153         try (Arena arena = Arena.ofConfined()) {
154             // SYSTEM_INFO is 48 bytes on x64
155             MemorySegment sysInfo = arena.allocate(48);
156             if (Kernel32FFM.GetNativeSystemInfo(sysInfo)) {
157                 // wProcessorArchitecture is at offset 0 (WORD in union)
158                 int arch = Short.toUnsignedInt(sysInfo.get(ValueLayout.JAVA_SHORT, 0));
159                 if (arch == 9 || arch == 12 || arch == 6) { // AMD64, ARM64, IA64
160                     cpu64bit = true;
161                 }
162             }
163         }
164 
165         var processorIdResult = Win32ProcessorFFM.queryProcessorId();
166         if (processorIdResult.getResultCount() > 0) {
167             processorID = WmiUtilFFM.getString(processorIdResult, ProcessorIdProperty.PROCESSORID, 0);
168         } else {
169             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
170                     cpu64bit ? new String[] { "ia64" } : new String[0]);
171         }
172         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
173                 cpuVendorFreq);
174     }

176     @Override
177     protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
178         Triplet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>> lpi = LogicalProcessorInformationFFM
179                 .getLogicalProcessorInformationEx();
180         buildNumaNodeProcMap(lpi.getA());
181 
182         List<String> featureFlags = IntStream.range(0, PROCESSOR_FEATURES.length)
183                 .filter(i -> Kernel32FFM.IsProcessorFeaturePresent(PROCESSOR_FEATURES[i]))
184                 .mapToObj(i -> PROCESSOR_FEATURE_NAMES[i]).collect(Collectors.toList());
185         return new Quartet<>(lpi.getA(), lpi.getB(), lpi.getC(), featureFlags);
186     }

188     @Override
189     public long[] querySystemCpuLoadTicks() {
190         long[] ticks = new long[TickType.values().length];
191         if (USE_LEGACY_SYSTEM_COUNTERS) {
192             try (Arena arena = Arena.ofConfined()) {
193                 // FILETIME is 8 bytes (dwLowDateTime + dwHighDateTime)
194                 MemorySegment lpIdleTime = arena.allocate(8);
195                 MemorySegment lpKernelTime = arena.allocate(8);
196                 MemorySegment lpUserTime = arena.allocate(8);
197                 if (!Kernel32FFM.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
198                     LOG.error("Failed to update system idle/kernel/user times. Error code: {}",
199                             Kernel32FFM.GetLastError().orElse(-1));
200                     return ticks;
201                 }
202 
203                 Map<SystemTickCountProperty, Long> valueMap = ProcessorInformationFFM.querySystemCounters();
204                 ticks[TickType.IRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTINTERRUPTTIME, 0L)
205                         / 10_000L;
206                 ticks[TickType.SOFTIRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTDPCTIME, 0L)
207                         / 10_000L;
208 
209                 ticks[TickType.IDLE.getIndex()] = lpIdleTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L;
210                 ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L
211                         - ticks[TickType.IDLE.getIndex()];
212                 ticks[TickType.USER.getIndex()] = lpUserTime.get(ValueLayout.JAVA_LONG, 0) / 10_000L;
213                 ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()]
214                         + ticks[TickType.SOFTIRQ.getIndex()];
215             }
216             return ticks;
217         }
218         long[][] procTicks = getProcessorCpuLoadTicks();
219         for (int i = 0; i < ticks.length; i++) {
220             for (long[] procTick : procTicks) {
221                 ticks[i] += procTick[i];
222             }
223         }
224         return ticks;
225     }

227     @Override
228     public long[] queryCurrentFreq() {
229         if (VersionHelpersFFM.IsWindows7OrGreater()) {
230             long maxFreq = this.getMaxFreq();
231             if (maxFreq > 0) {
232                 // Prefer % Processor Performance from WMI formatted table (Win8+, reports >100% with turbo)
233                 Pair<List<String>, Map<ProcessorPerformanceProperty, List<Long>>> perfPair = ProcessorInformationFFM
234                         .queryProcessorPerformanceCounters();
235                 List<Long> perfList = perfPair.getB().get(ProcessorPerformanceProperty.PERCENTPROCESSORPERFORMANCE);
236                 long[] freqs = mapPercentToFreqs(perfPair.getA(), perfList, maxFreq);
237                 if (freqs != null) {
238                     return freqs;
239                 }
240                 // Fall back to % of Maximum Frequency (Win7, caps at 100%)
241                 Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> freqPair = ProcessorInformationFFM
242                         .queryFrequencyCounters();
243                 List<Long> percentMaxList = freqPair.getB().get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
244                 freqs = mapPercentToFreqs(freqPair.getA(), percentMaxList, maxFreq);
245                 if (freqs != null) {
246                     return freqs;
247                 }
248             }
249         }
250         return queryNTPower(2);
251     }

253     private long[] mapPercentToFreqs(List<String> instances, List<Long> percentList, long maxFreq) {
254         if (instances.isEmpty() || percentList == null) {
255             return null;
256         }
257         long[] freqs = new long[getLogicalProcessorCount()];
258         boolean populated = false;
259         for (int i = 0; i < instances.size(); i++) {
260             String instance = instances.get(i);
261             int cpu;
262             if (instance.contains(",")) {
263                 if (!getNumaNodeProcToLogicalProcMap().containsKey(instance)) {
264                     continue;
265                 }
266                 cpu = getNumaNodeProcToLogicalProcMap().get(instance);
267             } else {
268                 cpu = ParseUtil.parseIntOrDefault(instance, -1);
269             }
270             if (cpu < 0 || cpu >= getLogicalProcessorCount() || i >= percentList.size()) {
271                 continue;
272             }
273             freqs[cpu] = percentList.get(i) * maxFreq / 100L;
274             if (freqs[cpu] > 0) {
275                 populated = true;
276             }
277         }
278         return populated ? freqs : null;
279     }

287     private long[] queryNTPower(int fieldIndex) {
288         long[] freqs = new long[getLogicalProcessorCount()];
289         int totalSize = PPI_SIZE * freqs.length;
290         try (Arena arena = Arena.ofConfined()) {
291             MemorySegment buffer = arena.allocate(totalSize);
292             int status = Kernel32FFM.CallNtPowerInformation(PROCESSOR_INFORMATION_LEVEL, MemorySegment.NULL, 0, buffer,
293                     totalSize);
294             if (status != 0) {
295                 LOG.error("Unable to get Processor Information, status: {}", status);
296                 Arrays.fill(freqs, -1L);
297                 return freqs;
298             }
299             for (int i = 0; i < freqs.length; i++) {
300                 long offset = i * (long) PPI_SIZE;
301                 // ProcessorPowerInformation: number(4), maxMhz(4), currentMhz(4), mhzLimit(4), maxIdleState(4),
302                 // currentIdleState(4)
303                 if (fieldIndex == 1) {
304                     freqs[i] = Integer.toUnsignedLong(buffer.get(ValueLayout.JAVA_INT, offset + 4)) * 1_000_000L;
305                 } else if (fieldIndex == 2) {
306                     freqs[i] = Integer.toUnsignedLong(buffer.get(ValueLayout.JAVA_INT, offset + 8)) * 1_000_000L;
307                 } else {
308                     freqs[i] = -1L;
309                 }
310                 if (freqs[i] == 0) {
311                     freqs[i] = getProcessorIdentifier().getVendorFreq();
312                 }
313             }
314         }
315         return freqs;
316     }

360             initUserList = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
361             initBase = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
362             initSystemUtility = getInitialUtilityCounters()

381                 processorUtility, processorUtilityBase, initSystemList, initUserList, initBase, initSystemUtility,
382                 initProcessorUtility, initProcessorUtilityBase);

401     private static String registryGetString(Arena arena, String path, String valueName) {
402         try {
403             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
404             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, path);
405             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
406                 MemorySegment openedKey = hKey.get(ValueLayout.ADDRESS, 0);
407                 try {
408                     MemorySegment valueNameSeg = WindowsForeignFunctions.toWideString(arena, valueName);
409                     MemorySegment dataSize = arena.allocate(ValueLayout.JAVA_INT);
410                     dataSize.set(ValueLayout.JAVA_INT, 0, 512);
411                     MemorySegment data = arena.allocate(512);
412                     if (Advapi32FFM.RegQueryValueEx(openedKey, valueNameSeg, 0, MemorySegment.NULL, data,
413                             dataSize) == 0) {
414                         return WindowsForeignFunctions.readWideString(data);
415                     }
416                 } finally {
417                     Advapi32FFM.RegCloseKey(openedKey);
418                 }
419             }
420         } catch (Throwable t) {
421             LOG.debug("Failed to read registry string {}/{}: {}", path, valueName, t.getMessage());
422         }
423         return "";
424     }

426     private static long registryGetDword(Arena arena, String path, String valueName) {
427         try {
428             MemorySegment hKey = arena.allocate(ValueLayout.ADDRESS);
429             MemorySegment subKey = WindowsForeignFunctions.toWideString(arena, path);
430             if (Advapi32FFM.RegOpenKeyEx(HKLM, subKey, 0, WinNTFFM.KEY_READ, hKey) == 0) {
431                 MemorySegment openedKey = hKey.get(ValueLayout.ADDRESS, 0);
432                 try {
433                     MemorySegment valueNameSeg = WindowsForeignFunctions.toWideString(arena, valueName);
434                     MemorySegment dataSize = arena.allocate(ValueLayout.JAVA_INT);
435                     dataSize.set(ValueLayout.JAVA_INT, 0, 4);
436                     MemorySegment data = arena.allocate(4);
437                     if (Advapi32FFM.RegQueryValueEx(openedKey, valueNameSeg, 0, MemorySegment.NULL, data,
438                             dataSize) == 0) {
439                         return Integer.toUnsignedLong(data.get(ValueLayout.JAVA_INT, 0));
440                     }
441                 } finally {
442                     Advapi32FFM.RegCloseKey(openedKey);
443                 }
444             }
445         } catch (Throwable t) {
446             LOG.debug("Failed to read registry DWORD {}/{}: {}", path, valueName, t.getMessage());
447         }
448         return 0L;
449     }
