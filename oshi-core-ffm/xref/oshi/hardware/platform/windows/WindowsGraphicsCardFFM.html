48      private static final MemorySegment HKLM = MemorySegment.ofAddress(WinRegFFM.HKEY_LOCAL_MACHINE);

50      private static final String ADAPTER_STRING = "HardwareInformation.AdapterString";
51      private static final String DRIVER_DESC = "DriverDesc";
52      private static final String DRIVER_VERSION = "DriverVersion";
53      private static final String VENDOR = "ProviderName";
54      private static final String QW_MEMORY_SIZE = "HardwareInformation.qwMemorySize";
55      private static final String MATCHING_DEVICE_ID = "MatchingDeviceId";
56      private static final String LOCATION_INFORMATION = "LocationInformation";
57      private static final String DISPLAY_DEVICES_REGISTRY_PATH = "SYSTEM\\CurrentControlSet\\Control\\Class\\{4d36e968-e325-11ce-bfc1-08002be10318}\\";

67          this.luidPrefix = luidPrefix;
68          this.lhmParent = lhmParent;
69          this.pciBusNumber = pciBusNumber;
70          this.pciBusId = pciBusId;

78      public static List<GraphicsCard> getGraphicsCards() {
79          List<DxgiAdapterInfo> dxgiAdapters = DxgiFFM.queryAdapters();
80          boolean dxgiAvailable = !dxgiAdapters.isEmpty();
81          List<DxgiAdapterInfo> remainingDxgi = new ArrayList<>(dxgiAdapters);
82  
83          Map<String, String> lhmParentMap = buildLhmParentMap();
84  
85          TreeMap<Integer, GraphicsCard> dxgiOrdered = new TreeMap<>();
86          List<GraphicsCard> cardList = new ArrayList<>();
87  
88          try {
89              String[] keys = Advapi32UtilFFM.registryGetKeys(HKLM, DISPLAY_DEVICES_REGISTRY_PATH, 0);
90              int index = 1;
91              for (String key : keys) {
92                  if (!key.startsWith("0")) {
93                      continue;
94                  }
95                  try {
96                      String fullKey = DISPLAY_DEVICES_REGISTRY_PATH + key;
97                      if (!Advapi32UtilFFM.registryValueExists(HKLM, fullKey, ADAPTER_STRING)) {
98                          continue;
99                      }
100 
101                     String name = registryString(fullKey, DRIVER_DESC);
102                     String deviceId = "VideoController" + index++;
103                     String vendor = registryString(fullKey, VENDOR);
104                     String versionInfo = registryString(fullKey, DRIVER_VERSION);
105 
106                     String matchingDeviceId = registryString(fullKey, MATCHING_DEVICE_ID);
107                     Pair<Integer, Integer> pciIds = ParseUtil.parseDeviceIdToVendorProductIds(matchingDeviceId);
108                     int pciVendorId = pciIds == null ? 0 : pciIds.getA();
109                     int pciDeviceId = pciIds == null ? 0 : pciIds.getB();
110 
111                     long vram = -1L;
112                     int dxgiIndex = -1;
113                     String luidPrefix = "";
114                     int pciBusNumber = -1;
115                     String pciBusId = "";
116                     String locationInfo = registryString(fullKey, LOCATION_INFORMATION);
117                     DxgiAdapterInfo dxgiMatch = DxgiUtil.findMatch(remainingDxgi, pciVendorId, pciDeviceId, name);
118                     if (dxgiMatch != null) {
119                         vram = dxgiMatch.getDedicatedVideoMemory();
120                         dxgiIndex = dxgiAdapters.indexOf(dxgiMatch);
121                         luidPrefix = buildLuidPrefix(dxgiMatch);
122                         pciBusNumber = DxgiUtil.parsePciBusNumber(locationInfo);
123                         pciBusId = DxgiUtil.buildPciBusId(locationInfo);
124                     } else if (dxgiAvailable) {
125                         continue; // ghost device
126                     }
127 
128                     if (vram < 0 && Advapi32UtilFFM.registryValueExists(HKLM, fullKey, QW_MEMORY_SIZE)) {
129                         Object regValue = Advapi32UtilFFM.registryGetValue(HKLM, fullKey, QW_MEMORY_SIZE);
130                         vram = DxgiUtil.registryValueToVram(regValue);
131                     }
132                     if (vram < 0) {
133                         vram = 0L;
134                     }
135 
136                     String lhmParent = lhmParentMap.getOrDefault(DxgiUtil.normalizeName(Util.isBlank(name) ? "" : name),
137                             "");
138 
139                     GraphicsCard card = new WindowsGraphicsCardFFM(Util.isBlank(name) ? Constants.UNKNOWN : name,
140                             Util.isBlank(deviceId) ? Constants.UNKNOWN : deviceId,
141                             Util.isBlank(vendor) ? Constants.UNKNOWN : vendor,
142                             Util.isBlank(versionInfo) ? Constants.UNKNOWN : versionInfo, vram, luidPrefix, lhmParent,
143                             pciBusNumber, pciBusId);
144                     if (dxgiMatch != null) {
145                         remainingDxgi.remove(dxgiMatch);
146                     }
147                     if (dxgiIndex >= 0) {
148                         dxgiOrdered.put(dxgiIndex, card);
149                     } else {
150                         cardList.add(card);
151                     }
152                 } catch (Throwable t) {
153                     LOG.debug("Error reading graphics card registry key {}: {}", key, t.getMessage());
154                 }
155             }
156         } catch (Throwable t) {
157             LOG.debug("Failed to enumerate display device registry keys: {}", t.getMessage());
158         }
159 
160         List<GraphicsCard> result = new ArrayList<>(dxgiOrdered.values());
161         result.addAll(cardList);
162 
163         if (result.isEmpty()) {
164             return getGraphicsCardsFromWmi(dxgiAdapters, remainingDxgi, lhmParentMap);
165         }
166         return result;
167     }

169     private static List<GraphicsCard> getGraphicsCardsFromWmi(List<DxgiAdapterInfo> dxgiAdapters,
170             List<DxgiAdapterInfo> remainingDxgi, Map<String, String> lhmParentMap) {
171         boolean dxgiAvailable = !dxgiAdapters.isEmpty();
172         List<DxgiAdapterInfo> working = new ArrayList<>(remainingDxgi);
173         TreeMap<Integer, GraphicsCard> dxgiOrdered = new TreeMap<>();
174         List<GraphicsCard> cardList = new ArrayList<>();
175 
176         WmiResult<VideoControllerProperty> cards = Win32VideoControllerFFM.queryVideoController();
177         for (int index = 0; index < cards.getResultCount(); index++) {
178             if (dxgiAvailable
179                     && WmiUtilFFM.getUint32(cards, VideoControllerProperty.CONFIGMANAGERERRORCODE, index) != 0) {
180                 continue;
181             }
182             String name = WmiUtilFFM.getString(cards, VideoControllerProperty.NAME, index);
183             Triplet<String, String, String> idPair = ParseUtil.parseDeviceIdToVendorProductSerial(
184                     WmiUtilFFM.getString(cards, VideoControllerProperty.PNPDEVICEID, index));
185             String deviceId = idPair == null ? Constants.UNKNOWN : idPair.getB();
186             String vendor = WmiUtilFFM.getString(cards, VideoControllerProperty.ADAPTERCOMPATIBILITY, index);
187             if (idPair != null) {
188                 if (Util.isBlank(vendor)) {
189                     deviceId = idPair.getA();
190                 } else {
191                     vendor = vendor + " (" + idPair.getA() + ")";
192                 }
193             }
194             String versionInfo = WmiUtilFFM.getString(cards, VideoControllerProperty.DRIVERVERSION, index);
195             if (!Util.isBlank(versionInfo)) {
196                 versionInfo = "DriverVersion=" + versionInfo;
197             } else {
198                 versionInfo = Constants.UNKNOWN;
199             }
200 
201             Pair<Integer, Integer> pciIds = ParseUtil.parseDeviceIdToVendorProductIds(
202                     WmiUtilFFM.getString(cards, VideoControllerProperty.PNPDEVICEID, index));
203             int pciVendorId = pciIds == null ? 0 : pciIds.getA();
204             int pciDeviceId = pciIds == null ? 0 : pciIds.getB();
205             DxgiAdapterInfo dxgiMatch = DxgiUtil.findMatch(working, pciVendorId, pciDeviceId, name);
206             long vram;
207             int dxgiIndex = -1;
208             String luidPrefix = "";
209             int pciBusNumber = -1;
210             String pciBusId = "";
211             if (dxgiMatch != null) {
212                 vram = dxgiMatch.getDedicatedVideoMemory();
213                 dxgiIndex = dxgiAdapters.indexOf(dxgiMatch);
214                 luidPrefix = DxgiUtil.buildLuidPrefix(dxgiMatch);
215             } else {
216                 vram = WmiUtilFFM.getUint32asLong(cards, VideoControllerProperty.ADAPTERRAM, index);
217             }
218             String lhmParent = lhmParentMap.getOrDefault(DxgiUtil.normalizeName(Util.isBlank(name) ? "" : name), "");
219             GraphicsCard card = new WindowsGraphicsCardFFM(Util.isBlank(name) ? Constants.UNKNOWN : name, deviceId,
220                     Util.isBlank(vendor) ? Constants.UNKNOWN : vendor, versionInfo, vram, luidPrefix, lhmParent,
221                     pciBusNumber, pciBusId);
222             if (dxgiMatch != null) {
223                 working.remove(dxgiMatch);
224             }
225             if (dxgiIndex >= 0) {
226                 dxgiOrdered.put(dxgiIndex, card);
227             } else {
228                 cardList.add(card);
229             }
230         }
231         List<GraphicsCard> result = new ArrayList<>(dxgiOrdered.values());
232         result.addAll(cardList);
233         return result;
234     }

236     private static Map<String, String> buildLhmParentMap() {
237         Map<String, String> map = new HashMap<>();
238         try {
239             WmiResult<LhmHardwareProperty> hw = LhmSensorFFM.queryGpuHardware();
240             for (int i = 0; i < hw.getResultCount(); i++) {
241                 String identifier = WmiUtilFFM.getString(hw, LhmHardwareProperty.IDENTIFIER, i);
242                 String hwName = WmiUtilFFM.getString(hw, LhmHardwareProperty.NAME, i);
243                 if (!identifier.isEmpty() && !hwName.isEmpty()) {
244                     String norm = DxgiUtil.normalizeName(hwName);
245                     if (map.containsKey(norm)) {
246                         map.put(norm, "");
247                     } else {
248                         map.put(norm, identifier);
249                     }
250                 }
251             }
252         } catch (Exception e) {
253             LOG.debug("LHM GPU hardware query failed (LHM may not be running): {}", e.getMessage());
254         }
255         return map;
256     }

258     private static String buildLuidPrefix(DxgiAdapterInfo adapter) {
259         String prefix = DxgiUtil.buildLuidPrefix(adapter);
260         if (!prefix.isEmpty()) {
261             return prefix;
262         }
263         return buildLuidPrefixFromPdh();
264     }
265 
266     private static String buildLuidPrefixFromPdh() {
267         Pair<List<String>, Map<GpuAdapterMemoryProperty, List<Long>>> adapterData = GpuInformationFFM
268                 .queryGpuAdapterMemoryCounters();
269         List<String> instances = adapterData.getA();
270         if (instances.isEmpty()) {
271             return "";
272         }
273         if (instances.size() == 1) {
274             return instances.get(0);
275         }
276         LOG.debug("Multiple GPU Adapter Memory instances found ({}); LUID matching not yet implemented for multi-GPU",
277                 instances.size());
278         return "";
279     }

281     private static String registryString(String keyPath, String valueName) {
282         Object val = Advapi32UtilFFM.registryGetValue(HKLM, keyPath, valueName);
283         if (val instanceof String) {
284             return (String) val;
285         }
286         return "";
287     }
