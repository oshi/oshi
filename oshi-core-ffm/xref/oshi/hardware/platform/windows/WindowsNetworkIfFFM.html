24  /**
25   * WindowsNetworks class using FFM.
26   */
27  @ThreadSafe
28  public final class WindowsNetworkIfFFM extends WindowsNetworkIF {
29  
30      private static final Logger LOG = LoggerFactory.getLogger(WindowsNetworkIfFFM.class);
31  
32      public WindowsNetworkIfFFM(NetworkInterface netint) throws InstantiationException {
33          super(netint);
34      }
35  
36      /**
37       * Gets all network interfaces on this machine
38       *
39       * @param includeLocalInterfaces include local interfaces in the result
40       * @return A list of {@link NetworkIF} objects representing the interfaces
41       */
42      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
43          return getNetworks(includeLocalInterfaces, WindowsNetworkIfFFM::new);
44      }
45  
46      @Override
47      protected IfStats queryStats() {
48          // FFM targets JDK 25+ which requires Vista+; use GetIfEntry2 directly
49          return callInArenaOrDefault(arena -> {
50              MemorySegment ifRow = arena.allocate(IPHlpAPIFFM.MIB_IF_ROW2_SIZE);
51              ifRow.fill((byte) 0);
52              // Set InterfaceIndex at offset 8
53              ifRow.set(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_INTERFACE_INDEX, queryNetworkInterface().getIndex());
54              if (0 != IPHlpAPIFFM.GetIfEntry2(ifRow)) {
55                  LOG.error("Failed to retrieve data for interface {}, {}", queryNetworkInterface().getIndex(),
56                          getName());
57                  return null;
58              }
59              IfStats stats = new IfStats();
60              stats.ifType = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_TYPE);
61              stats.ndisPhysicalMediumType = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_PHYSICAL_MEDIUM_TYPE);
62              byte flags = ifRow.get(ValueLayout.JAVA_BYTE, IPHlpAPIFFM.OFFSET_FLAGS);
63              stats.connectorPresent = (flags & CONNECTOR_PRESENT_BIT) > 0;
64              stats.speed = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_RECEIVE_LINK_SPEED);
65              stats.bytesRecv = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_OCTETS);
66              stats.packetsRecv = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_UCAST_PKTS);
67              stats.inErrors = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_ERRORS);
68              stats.inDrops = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_IN_DISCARDS);
69              stats.bytesSent = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_OCTETS);
70              stats.packetsSent = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_UCAST_PKTS);
71              stats.outErrors = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_ERRORS);
72              stats.collisions = ifRow.get(ValueLayout.JAVA_LONG, IPHlpAPIFFM.OFFSET_OUT_DISCARDS);
73              // Alias is WCHAR[257] at OFFSET_ALIAS
74              MemorySegment aliasSlice = ifRow.asSlice(IPHlpAPIFFM.OFFSET_ALIAS, 257 * 2L);
75              stats.ifAlias = readWideString(aliasSlice);
76              int operStatus = ifRow.get(ValueLayout.JAVA_INT, IPHlpAPIFFM.OFFSET_OPER_STATUS);
77              stats.ifOperStatus = IfOperStatus.byValue(operStatus);
78              return stats;
79          }, LOG, ERROR, "Failed to retrieve data for interface " + queryNetworkInterface().getIndex() + ", " + getName(),
80                  null);
81      }
82  
83      private static String readWideString(MemorySegment seg) {
84          int len = 0;
85          while (len < seg.byteSize() / 2 && seg.get(ValueLayout.JAVA_SHORT, len * 2L) != 0) {
86              len++;
87          }
88          byte[] bytes = new byte[len * 2];
89          MemorySegment.copy(seg, ValueLayout.JAVA_BYTE, 0, bytes, 0, bytes.length);
90          return new String(bytes, StandardCharsets.UTF_16LE);
91      }
92  }
