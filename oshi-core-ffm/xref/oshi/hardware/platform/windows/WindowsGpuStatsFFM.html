71      @Override
72      public synchronized GpuTicks getGpuTicks() {
73          checkOpen();
74          if (luidPrefix.isEmpty()) {
75              return new GpuTicks(0L, 0L);
76          }
77          Pair<List<String>, Map<GpuEngineProperty, List<Long>>> engineData = GpuInformationFFM.queryGpuEngineCounters();
78          List<String> instances = engineData.getA();
79          Map<GpuEngineProperty, List<Long>> values = engineData.getB();
80          List<Long> runningTimes = values.get(GpuEngineProperty.RUNNING_TIME);
81          List<Long> runningTimeBases = values.get(GpuEngineProperty.RUNNING_TIME_BASE);
82          if (instances.isEmpty() || runningTimes == null || runningTimeBases == null) {
83              return new GpuTicks(0L, 0L);
84          }
85          Map<String, Long> activeByType = new HashMap<>();
86          Map<String, Long> baseByType = new HashMap<>();
87          String luidLower = luidPrefix.toLowerCase(Locale.ROOT);
88          int limit = Math.min(instances.size(), Math.min(runningTimes.size(), runningTimeBases.size()));
89          for (int i = 0; i < limit; i++) {
90              String inst = instances.get(i).toLowerCase(Locale.ROOT);
91              if (!inst.contains(luidLower)) {
92                  continue;
93              }
94              int engTypeIdx = inst.lastIndexOf("_engtype_");
95              String engType = engTypeIdx >= 0 ? inst.substring(engTypeIdx) : inst;
96              activeByType.merge(engType, runningTimes.get(i), Long::sum);
97              baseByType.merge(engType, runningTimeBases.get(i), Long::sum);
98          }
99          if (activeByType.isEmpty()) {
100             return new GpuTicks(0L, 0L);
101         }
102         long totalActive = 0L;
103         long totalBase = 0L;
104         for (Map.Entry<String, Long> entry : activeByType.entrySet()) {
105             totalActive += entry.getValue();
106             totalBase += baseByType.getOrDefault(entry.getKey(), 0L);
107         }
108         long idle = totalBase >= totalActive ? totalBase - totalActive : 0L;
109         return new GpuTicks(totalActive, idle);
110     }

112     @Override
113     public synchronized double getGpuUtilization() {
114         checkOpen();
115         if (!lhmParent.isEmpty()) {
116             try {
117                 WmiResult<LhmSensorProperty> sensors = LhmSensorFFM.querySensors(lhmParent, "Load");
118                 for (int i = 0; i < sensors.getResultCount(); i++) {
119                     if (GPU_CORE.equals(WmiUtilFFM.getString(sensors, LhmSensorProperty.NAME, i))) {
120                         return WmiUtilFFM.getFloat(sensors, LhmSensorProperty.VALUE, i);
121                     }
122                 }
123             } catch (Exception e) {
124                 LOG.debug("LHM GPU utilization query failed: {}", e.getMessage());
125             }
126         }
127         GpuTicks curr = getGpuTicks();
128         if (prevUtilTicks != null) {
129             long dActive = curr.getActiveTicks() - prevUtilTicks.getActiveTicks();
130             long dIdle = curr.getIdleTicks() - prevUtilTicks.getIdleTicks();
131             if (dActive < 0 || dIdle < 0) {
132                 // Counter reset or empty snapshot; discard and wait for next sample
133                 prevUtilTicks = curr;
134                 return -1d;
135             }
136             long dTotal = dActive + dIdle;
137             prevUtilTicks = curr;
138             return dTotal > 0 ? dActive * 100.0 / dTotal : -1d;
139         }
140         prevUtilTicks = curr;
141         return -1d;
142     }

144     @Override
145     public synchronized long getVramUsed() {
146         checkOpen();
147         long pdhResult = queryAdapterMemory(GpuAdapterMemoryProperty.DEDICATED_USAGE);
148         if (pdhResult >= 0) {
149             return pdhResult;
150         }
151         if (!lhmParent.isEmpty()) {
152             try {
153                 WmiResult<LhmSensorProperty> sensors = LhmSensorFFM.querySensors(lhmParent, "SmallData");
154                 for (int i = 0; i < sensors.getResultCount(); i++) {
155                     if ("GPU Memory Used".equals(WmiUtilFFM.getString(sensors, LhmSensorProperty.NAME, i))) {
156                         float mb = WmiUtilFFM.getFloat(sensors, LhmSensorProperty.VALUE, i);
157                         return (long) (mb * MB_TO_BYTES);
158                     }
159                 }
160             } catch (Exception e) {
161                 LOG.debug("LHM GPU memory used query failed: {}", e.getMessage());
162             }
163         }
164         return -1L;
165     }

176     @Override
177     public synchronized double getTemperature() {
178         checkOpen();
179         String nvmlDevice = findNvmlDevice();
180         if (nvmlDevice != null) {
181             double val = NvmlUtilFFM.getTemperature(nvmlDevice);
182             if (val >= 0) {
183                 return val;
184             }
185         }
186         int adlIndex = findAdlIndex();
187         if (adlIndex >= 0) {
188             double val = AdlUtilFFM.getTemperature(adlIndex);
189             if (val >= 0) {
190                 return val;
191             }
192         }
193         return lhmFloatSensor("Temperature", GPU_CORE);
194     }

286     private void checkOpen() {
287         if (closed) {
288             throw new IllegalStateException(
289                     "GpuStats session has been closed. Obtain a new session via GraphicsCard.createStatsSession().");
290         }
291     }

293     private long queryAdapterMemory(GpuAdapterMemoryProperty property) {
294         if (luidPrefix.isEmpty()) {
295             return -1L;
296         }
297         Pair<List<String>, Map<GpuAdapterMemoryProperty, List<Long>>> adapterData = GpuInformationFFM
298                 .queryGpuAdapterMemoryCounters();
299         List<String> instances = adapterData.getA();
300         List<Long> values = adapterData.getB().get(property);
301         if (values != null) {
302             String luidLower = luidPrefix.toLowerCase(Locale.ROOT);
303             int limit = Math.min(instances.size(), values.size());
304             for (int i = 0; i < limit; i++) {
305                 if (instances.get(i).toLowerCase(Locale.ROOT).contains(luidLower)) {
306                     return values.get(i);
307                 }
308             }
309         }
310         return -1L;
311     }

313     private String findNvmlDevice() {
314         if (cachedNvmlDevice != null) {
315             return cachedNvmlDevice.isEmpty() ? null : cachedNvmlDevice;
316         }
317         if (!NvmlUtilFFM.isAvailable()) {
318             cachedNvmlDevice = "";
319             return null;
320         }
321         String id = null;
322         if (!pciBusId.isEmpty()) {
323             id = NvmlUtilFFM.findDevice(pciBusId);
324         }
325         if (id == null) {
326             id = NvmlUtilFFM.findDeviceByName(cardName);
327         }
328         cachedNvmlDevice = id != null ? id : "";
329         return id;
330     }
331 
332     private int findAdlIndex() {
333         if (cachedAdlIndex != Integer.MIN_VALUE) {
334             return cachedAdlIndex;
335         }
336         if (!AdlUtilFFM.isAvailable() || pciBusNumber < 0) {
337             cachedAdlIndex = -1;
338             return -1;
339         }
340         cachedAdlIndex = AdlUtilFFM.findAdapterIndex(pciBusNumber);
341         return cachedAdlIndex;
342     }

344     private double lhmFloatSensor(String sensorType, String sensorName) {
345         if (lhmParent.isEmpty()) {
346             return -1d;
347         }
348         try {
349             WmiResult<LhmSensorProperty> sensors = LhmSensorFFM.querySensors(lhmParent, sensorType);
350             for (int i = 0; i < sensors.getResultCount(); i++) {
351                 if (sensorName.equals(WmiUtilFFM.getString(sensors, LhmSensorProperty.NAME, i))) {
352                     return WmiUtilFFM.getFloat(sensors, LhmSensorProperty.VALUE, i);
353                 }
354             }
355         } catch (Exception e) {
356             LOG.debug("LHM {} {} query failed: {}", sensorType, sensorName, e.getMessage());
357         }
358         return -1d;
359     }
