22  /**
23   * Windows {@link oshi.hardware.GpuStats} session using FFM.
24   */
25  @ThreadSafe
26  final class WindowsGpuStatsFFM extends WindowsGpuStats {
27  
28      WindowsGpuStatsFFM(String luidPrefix, String lhmParent, int pciBusNumber, String pciBusId, String cardName) {
29          super(luidPrefix, lhmParent, pciBusNumber, pciBusId, cardName);
30      }
31  
32      @Override
33      protected Pair<List<String>, Map<GpuEngineProperty, List<Long>>> queryGpuEngineCounters() {
34          return GpuInformationFFM.queryGpuEngineCounters();
35      }
36  
37      @Override
38      protected Pair<List<String>, Map<GpuAdapterMemoryProperty, List<Long>>> queryGpuAdapterMemoryCounters() {
39          return GpuInformationFFM.queryGpuAdapterMemoryCounters();
40      }
41  
42      @Override
43      protected WmiResult<LhmSensorProperty> queryLhmSensors(String parent, String sensorType) {
44          return LhmSensorFFM.querySensors(parent, sensorType);
45      }
46  
47      @Override
48      protected boolean isNvmlAvailable() {
49          return NvmlUtilFFM.isAvailable();
50      }
51  
52      @Override
53      protected String nvmlFindDevice(String pciBusId) {
54          return NvmlUtilFFM.findDevice(pciBusId);
55      }
56  
57      @Override
58      protected String nvmlFindDeviceByName(String gpuName) {
59          return NvmlUtilFFM.findDeviceByName(gpuName);
60      }
61  
62      @Override
63      protected double nvmlGetTemperature(String device) {
64          return NvmlUtilFFM.getTemperature(device);
65      }
66  
67      @Override
68      protected double nvmlGetPowerDraw(String device) {
69          return NvmlUtilFFM.getPowerDraw(device);
70      }
71  
72      @Override
73      protected long nvmlGetCoreClockMhz(String device) {
74          return NvmlUtilFFM.getCoreClockMhz(device);
75      }
76  
77      @Override
78      protected long nvmlGetMemoryClockMhz(String device) {
79          return NvmlUtilFFM.getMemoryClockMhz(device);
80      }
81  
82      @Override
83      protected double nvmlGetFanSpeedPercent(String device) {
84          return NvmlUtilFFM.getFanSpeedPercent(device);
85      }
86  
87      @Override
88      protected boolean isAdlAvailable() {
89          return AdlUtilFFM.isAvailable();
90      }
91  
92      @Override
93      protected int adlFindAdapterIndex(int pciBusNumber) {
94          return AdlUtilFFM.findAdapterIndex(pciBusNumber);
95      }
96  
97      @Override
98      protected double adlGetTemperature(int adapterIndex) {
99          return AdlUtilFFM.getTemperature(adapterIndex);
100     }
101 
102     @Override
103     protected double adlGetPowerDraw(int adapterIndex) {
104         return AdlUtilFFM.getPowerDraw(adapterIndex);
105     }
106 
107     @Override
108     protected long adlGetCoreClockMhz(int adapterIndex) {
109         return AdlUtilFFM.getCoreClockMhz(adapterIndex);
110     }
111 
112     @Override
113     protected long adlGetMemoryClockMhz(int adapterIndex) {
114         return AdlUtilFFM.getMemoryClockMhz(adapterIndex);
115     }
116 
117     @Override
118     protected double adlGetFanSpeedPercent(int adapterIndex) {
119         return AdlUtilFFM.getFanSpeedPercent(adapterIndex);
120     }
