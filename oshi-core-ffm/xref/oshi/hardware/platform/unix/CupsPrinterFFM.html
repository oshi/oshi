25  /**
26   * CUPS-based printer implementation using the Java FFM API. Falls back to lpstat when libcups is unavailable.
27   */
28  @Immutable
29  public final class CupsPrinterFFM extends CupsPrinter {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(CupsPrinterFFM.class);
32  
33      CupsPrinterFFM(String name, String driverName, String description, PrinterStatus status, String statusReason,
34              boolean isDefault, boolean isLocal, String portName) {
35          super(name, driverName, description, status, statusReason, isDefault, isLocal, portName);
36      }
37  
38      /**
39       * Gets printers using CUPS. Uses libcups via FFM if available, otherwise falls back to the {@code lpstat} command.
40       *
41       * @return a list of printers
42       */
43      public static List<Printer> getPrinters() {
44          if (CupsFunctions.isAvailable()) {
45              return getPrintersFromLibCups();
46          }
47          LOG.debug("libcups not available, falling back to lpstat.");
48          return getPrintersFromLpstat(CupsPrinterFFM::new);
49      }
50  
51      private static List<Printer> getPrintersFromLibCups() {
52          List<Printer> printers = new ArrayList<>();
53          try (Arena arena = Arena.ofConfined()) {
54              MemorySegment ptrSlot = arena.allocate(ADDRESS);
55              int numDests = CupsFunctions.cupsGetDests(ptrSlot);
56              if (numDests <= 0) {
57                  return printers;
58              }
59              MemorySegment destsPtr = ptrSlot.get(ADDRESS, 0);
60              if (destsPtr == null || destsPtr.equals(MemorySegment.NULL)) {
61                  return printers;
62              }
63              try {
64                  MemorySegment destsArray = destsPtr.reinterpret(CupsFunctions.CUPS_DEST_T.byteSize() * numDests, arena,
65                          null);
66                  for (int i = 0; i < numDests; i++) {
67                      long offset = i * CupsFunctions.CUPS_DEST_T.byteSize();
68                      MemorySegment dest = destsArray.asSlice(offset, CupsFunctions.CUPS_DEST_T.byteSize());
69  
70                      MemorySegment namePtr = (MemorySegment) CupsFunctions.CUPS_DEST_NAME.get(dest, 0L);
71                      if (namePtr == null || namePtr.equals(MemorySegment.NULL)) {
72                          continue;
73                      }
74                      String name = ForeignFunctions.getStringFromNativePointer(namePtr, arena);
75                      int isDefaultInt = (int) CupsFunctions.CUPS_DEST_IS_DEFAULT.get(dest, 0L);
76                      boolean isDefault = isDefaultInt != 0;
77                      int numOptions = (int) CupsFunctions.CUPS_DEST_NUM_OPTIONS.get(dest, 0L);
78                      MemorySegment options = (MemorySegment) CupsFunctions.CUPS_DEST_OPTIONS.get(dest, 0L);
79  
80                      String deviceUri = CupsFunctions.getOption("device-uri", numOptions, options, arena);
81                      String printerInfo = CupsFunctions.getOption("printer-info", numOptions, options, arena);
82                      String printerMakeModel = CupsFunctions.getOption("printer-make-and-model", numOptions, options,
83                              arena);
84                      String printerState = CupsFunctions.getOption("printer-state", numOptions, options, arena);
85                      String stateReasons = CupsFunctions.getOption("printer-state-reasons", numOptions, options, arena);
86                      String printerTypeStr = CupsFunctions.getOption("printer-type", numOptions, options, arena);
87  
88                      PrinterStatus status = parseStateFromCups(printerState, stateReasons);
89                      String statusReason = "none".equals(stateReasons) ? "" : stateReasons;
90                      int printerType = ParseUtil.parseIntOrDefault(printerTypeStr, 0);
91                      boolean isLocal = (printerType & CupsFunctions.CUPS_PRINTER_REMOTE) == 0;
92  
93                      printers.add(new CupsPrinterFFM(name, printerMakeModel, printerInfo, status, statusReason,
94                              isDefault, isLocal, deviceUri));
95                  }
96              } finally {
97                  CupsFunctions.cupsFreeDests(numDests, destsPtr);
98              }
99          } catch (Throwable e) {
100             LOG.warn("Failed to query printers from libcups: {}", e.getMessage(), e);
101         }
102         return printers;
103     }
104 
105     private static PrinterStatus parseStateFromCups(String state, String stateReasons) {
106         if (!stateReasons.isEmpty() && !"none".equals(stateReasons)) {
107             String lower = stateReasons.toLowerCase(Locale.ROOT);
108             if (lower.contains("error") || lower.contains("fault")) {
109                 return PrinterStatus.ERROR;
110             }
111         }
112         if (state.isEmpty()) {
113             return PrinterStatus.UNKNOWN;
114         }
115         switch (ParseUtil.parseIntOrDefault(state, -1)) {
116             case CupsFunctions.IPP_PRINTER_IDLE:
117                 return PrinterStatus.IDLE;
118             case CupsFunctions.IPP_PRINTER_PROCESSING:
119                 return PrinterStatus.PRINTING;
120             case CupsFunctions.IPP_PRINTER_STOPPED:
121                 return PrinterStatus.OFFLINE;
122             default:
123                 return PrinterStatus.UNKNOWN;
124         }
125     }
