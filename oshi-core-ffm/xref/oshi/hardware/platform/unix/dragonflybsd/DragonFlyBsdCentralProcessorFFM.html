5   package oshi.hardware.platform.unix.dragonflybsd;
6   
7   import static java.lang.foreign.ValueLayout.JAVA_LONG;
8   
9   import java.lang.foreign.MemorySegment;
10  
11  import oshi.ffm.util.platform.unix.freebsd.BsdSysctlUtilFFM;
12  import oshi.hardware.CentralProcessor;
13  import oshi.hardware.platform.unix.freebsd.FreeBsdCentralProcessorFFM;
14  
15  /**
16   * DragonFly BSD Central Processor (FFM). Inherits {@code kern.cp_time} system tick logic from
17   * {@link FreeBsdCentralProcessorFFM} and overrides per-CPU tick retrieval to use DragonFly's {@code kern.cputime}
18   * sysctl (FreeBSD uses {@code kern.cp_times}).
19   */
20  public class DragonFlyBsdCentralProcessorFFM extends FreeBsdCentralProcessorFFM {
21  
22      private static final int CP_USER = 0;
23      private static final int CP_NICE = 1;
24      private static final int CP_SYS = 2;
25      private static final int CP_INTR = 3;
26      private static final int CP_IDLE = 4;
27      private static final long UINT64_SIZE = Long.BYTES;
28  
29      @Override
30      public long[][] queryProcessorCpuLoadTicks() {
31          int cpus = getLogicalProcessorCount();
32          long[][] ticks = new long[cpus][CentralProcessor.TickType.values().length];
33          // DragonFly's kern.cputime returns a packed per-CPU array; the struct size per CPU is the total length
34          // divided by CPU count (the trailing fields can vary by DragonFly version).
35          MemorySegment buf = BsdSysctlUtilFFM.sysctl("kern.cputime");
36          if (buf == null) {
37              return ticks;
38          }
39          long arraySize = buf.byteSize();
40          if (arraySize <= 0) {
41              return ticks;
42          }
43          long structSize = arraySize / cpus;
44          // Guard: kern.cputime's per-CPU struct must hold at least the 5 tick fields we read (CP_IDLE is the highest
45          // index). A short return — e.g. a stub sysctl on a non-DragonFly kernel — would otherwise IOBE in buf.get().
46          if (structSize < (CP_IDLE + 1) * UINT64_SIZE) {
47              return ticks;
48          }
49          for (int cpu = 0; cpu < cpus; cpu++) {
50              long base = structSize * cpu;
51              ticks[cpu][TickType.USER.getIndex()] = buf.get(JAVA_LONG, base + CP_USER * UINT64_SIZE);
52              ticks[cpu][TickType.NICE.getIndex()] = buf.get(JAVA_LONG, base + CP_NICE * UINT64_SIZE);
53              ticks[cpu][TickType.SYSTEM.getIndex()] = buf.get(JAVA_LONG, base + CP_SYS * UINT64_SIZE);
54              ticks[cpu][TickType.IRQ.getIndex()] = buf.get(JAVA_LONG, base + CP_INTR * UINT64_SIZE);
55              ticks[cpu][TickType.IDLE.getIndex()] = buf.get(JAVA_LONG, base + CP_IDLE * UINT64_SIZE);
56          }
57          return ticks;
58      }
59  }
