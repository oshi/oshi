7   import static java.lang.foreign.ValueLayout.JAVA_INT;
8   import static oshi.ffm.ForeignFunctions.CAPTURED_STATE_LAYOUT;
9   import static oshi.ffm.ForeignFunctions.callInArenaDoubleOrDefault;
10  import static oshi.ffm.unix.freebsd.FreeBsdLibcFunctions.SIZE_T;
11  
12  import java.lang.foreign.MemorySegment;
13  import java.util.Locale;
14  
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  import org.slf4j.event.Level;
18  
19  import oshi.annotation.concurrent.ThreadSafe;
20  import oshi.ffm.unix.freebsd.FreeBsdLibcFunctions;
21  import oshi.hardware.common.platform.unix.freebsd.FreeBsdSensors;

23  /**
24   * FFM-backed FreeBSD sensors. The only native read is the per-CPU temperature from the {@code coretemp} kld module via
25   * {@code sysctlbyname("dev.cpu.%d.temperature")}. Fan speeds and CPU voltage have no FreeBSD-supported source and
26   * return the empty/zero defaults inherited from {@link FreeBsdSensors}.
27   */
28  @ThreadSafe
29  public class FreeBsdSensorsFFM extends FreeBsdSensors {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(FreeBsdSensorsFFM.class);
32  
33      @Override
34      public double queryCpuTemperature() {
35          return queryKldloadCoretemp();
36      }
37  
38      /*
39       * If user has loaded coretemp module via kldload coretemp, sysctl call will return temperature
40       *
41       * @return Temperature if successful, otherwise NaN
42       */
43      private static double queryKldloadCoretemp() {
44          return callInArenaDoubleOrDefault(arena -> {
45              MemorySegment p = arena.allocate(JAVA_INT);
46              MemorySegment size = arena.allocateFrom(SIZE_T, JAVA_INT.byteSize());
47              MemorySegment callState = arena.allocate(CAPTURED_STATE_LAYOUT);
48              int cpu = 0;
49              double sumTemp = 0d;
50              while (true) {
51                  MemorySegment name = arena.allocateFrom(String.format(Locale.ROOT, "dev.cpu.%d.temperature", cpu));
52                  if (0 != FreeBsdLibcFunctions.sysctlbyname(callState, name, p, size, MemorySegment.NULL, 0L)) {
53                      break;
54                  }
55                  sumTemp += p.get(JAVA_INT, 0) / 10d - 273.15;
56                  cpu++;
57              }
58              return cpu > 0 ? sumTemp / cpu : Double.NaN;
59          }, LOG, Level.WARN, "Failed reading CPU temperature", Double.NaN);
60      }
61  }
