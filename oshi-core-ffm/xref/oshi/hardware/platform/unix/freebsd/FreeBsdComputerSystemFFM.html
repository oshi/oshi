7   import oshi.annotation.concurrent.Immutable;
8   import oshi.ffm.util.platform.unix.freebsd.BsdSysctlUtilFFM;
9   import oshi.hardware.common.platform.unix.freebsd.FreeBsdComputerSystem;
10  import oshi.util.Constants;

12  /**
13   * FFM-backed FreeBSD computer system. {@code dmidecode} and {@code lshal} parsing live on
14   * {@link FreeBsdComputerSystem}; only the {@code kern.hostuuid} sysctl fallback is FFM-specific.
15   */
16  @Immutable
17  public class FreeBsdComputerSystemFFM extends FreeBsdComputerSystem {
18  
19      @Override
20      protected String queryHostUuid() {
21          return BsdSysctlUtilFFM.sysctl("kern.hostuuid", Constants.UNKNOWN);
22      }
23  }
