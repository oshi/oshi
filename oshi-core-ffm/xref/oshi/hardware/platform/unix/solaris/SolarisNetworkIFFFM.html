5   package oshi.hardware.platform.unix.solaris;
6   
7   import java.lang.foreign.MemorySegment;
8   import java.net.NetworkInterface;
9   import java.util.List;
10  
11  import oshi.annotation.concurrent.ThreadSafe;
12  import oshi.ffm.platform.unix.solaris.LibKstatFunctions;
13  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM;
14  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM.KstatChain;
15  import oshi.hardware.NetworkIF;
16  import oshi.hardware.common.platform.unix.solaris.SolarisNetworkIF;
17  
18  /**
19   * FFM-backed Solaris NetworkIF. Uses the legacy {@code kstat} chain only; Kstat2 exists only on the JDK 17-capped
20   * latest Solaris, so FFM (JDK 25) never needs it.
21   */
22  @ThreadSafe
23  public final class SolarisNetworkIFFFM extends SolarisNetworkIF {
24  
25      public SolarisNetworkIFFFM(NetworkInterface netint) throws InstantiationException {
26          super(netint);
27          updateAttributes();
28      }
29  
30      /**
31       * Gets all network interfaces on this machine.
32       *
33       * @param includeLocalInterfaces include local interfaces in the result
34       * @return a list of {@link NetworkIF} objects representing the interfaces
35       */
36      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
37          return getNetworks(includeLocalInterfaces, SolarisNetworkIFFFM::new);
38      }
39  
40      @Override
41      protected IfStats queryStats() {
42          try (KstatChain kc = KstatUtilFFM.openChain()) {
43              MemorySegment ksp = kc.lookup("link", -1, getName());
44              if (ksp.address() == 0L) {
45                  ksp = kc.lookup(null, -1, getName());
46              }
47              if (ksp.address() != 0L && kc.read(ksp)) {
48                  IfStats stats = new IfStats();
49                  stats.bytesSent = KstatUtilFFM.dataLookupLong(ksp, "obytes64");
50                  stats.bytesRecv = KstatUtilFFM.dataLookupLong(ksp, "rbytes64");
51                  stats.packetsSent = KstatUtilFFM.dataLookupLong(ksp, "opackets64");
52                  stats.packetsRecv = KstatUtilFFM.dataLookupLong(ksp, "ipackets64");
53                  stats.outErrors = KstatUtilFFM.dataLookupLong(ksp, "oerrors");
54                  stats.inErrors = KstatUtilFFM.dataLookupLong(ksp, "ierrors");
55                  stats.collisions = KstatUtilFFM.dataLookupLong(ksp, "collisions");
56                  stats.inDrops = KstatUtilFFM.dataLookupLong(ksp, "dl_idrops");
57                  stats.speed = KstatUtilFFM.dataLookupLong(ksp, "ifspeed");
58                  stats.timeStamp = LibKstatFunctions.kstatSnaptime(ksp) / 1_000_000L;
59                  return stats;
60              }
61          }
62          return null;
63      }
64  }
