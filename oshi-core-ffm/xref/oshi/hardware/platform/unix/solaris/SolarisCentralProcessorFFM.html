26  /**
27   * FFM-backed Solaris CPU. Uses the legacy {@code kstat} chain only; Kstat2 exists only on the JDK 17-capped latest
28   * Solaris, so FFM (JDK 25) never needs it.
29   */
30  @ThreadSafe
31  final class SolarisCentralProcessorFFM extends SolarisCentralProcessor {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(SolarisCentralProcessorFFM.class);
34  
35      private static final String CPU_INFO = "cpu_info";
36  
37      @Override
38      protected ProcessorIdentifier queryProcessorId() {
39          boolean cpu64bit = "64".equals(ExecutingCommand.getFirstAnswer("isainfo -b").trim());
40          String cpuVendor = "";
41          String cpuName = "";
42          String cpuFamily = "";
43          String cpuModel = "";
44          String cpuStepping = "";
45          long cpuFreq = 0L;
46  
47          try (KstatChain kc = KstatUtilFFM.openChain()) {
48              MemorySegment ksp = kc.lookup(CPU_INFO, -1, null);
49              if (ksp.address() != 0L && kc.read(ksp)) {
50                  cpuVendor = KstatUtilFFM.dataLookupString(ksp, "vendor_id");
51                  cpuName = KstatUtilFFM.dataLookupString(ksp, "brand");
52                  cpuFamily = KstatUtilFFM.dataLookupString(ksp, "family");
53                  cpuModel = KstatUtilFFM.dataLookupString(ksp, "model");
54                  cpuStepping = KstatUtilFFM.dataLookupString(ksp, "stepping");
55                  cpuFreq = KstatUtilFFM.dataLookupLong(ksp, "clock_MHz") * 1_000_000L;
56              }
57          }
58          String processorID = getProcessorID(cpuStepping, cpuModel, cpuFamily);
59          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
60                  cpuFreq);
61      }
62  
63      @Override
64      protected List<LogicalProcessor> queryLogicalProcessors() {
65          Map<Integer, Integer> numaNodeMap = mapNumaNodes();
66          List<LogicalProcessor> logProcs = new ArrayList<>();
67          try (KstatChain kc = KstatUtilFFM.openChain()) {
68              List<MemorySegment> kstats = kc.lookupAll(CPU_INFO, -1, null);
69              for (MemorySegment ksp : kstats) {
70                  if (ksp.address() != 0L && kc.read(ksp)) {
71                      int procId = logProcs.size();
72                      String chipId = KstatUtilFFM.dataLookupString(ksp, "chip_id");
73                      String coreId = KstatUtilFFM.dataLookupString(ksp, "core_id");
74                      logProcs.add(new LogicalProcessor(procId, ParseUtil.parseIntOrDefault(coreId, 0),
75                              ParseUtil.parseIntOrDefault(chipId, 0), numaNodeMap.getOrDefault(procId, 0)));
76                  }
77              }
78          }
79          return logProcs;
80      }

82      @Override
83      public long[] queryCurrentFreq() {
84          long[] freqs = new long[getLogicalProcessorCount()];
85          Arrays.fill(freqs, -1);
86          try (KstatChain kc = KstatUtilFFM.openChain()) {
87              for (int i = 0; i < freqs.length; i++) {
88                  for (MemorySegment ksp : kc.lookupAll(CPU_INFO, i, null)) {
89                      if (ksp.address() != 0L && kc.read(ksp)) {
90                          freqs[i] = KstatUtilFFM.dataLookupLong(ksp, "current_clock_Hz");
91                      }
92                  }
93              }
94          }
95          return freqs;
96      }
97  
98      @Override
99      public long queryMaxFreq() {
100         long max = -1L;
101         try (KstatChain kc = KstatUtilFFM.openChain()) {
102             for (MemorySegment ksp : kc.lookupAll(CPU_INFO, 0, null)) {
103                 if (kc.read(ksp)) {
104                     String suppFreq = KstatUtilFFM.dataLookupString(ksp, "supported_frequencies_Hz");
105                     if (!suppFreq.isEmpty()) {
106                         for (String s : suppFreq.split(":")) {
107                             long freq = ParseUtil.parseLongOrDefault(s, -1L);
108                             if (max < freq) {
109                                 max = freq;
110                             }
111                         }
112                     }
113                 }
114             }
115         }
116         return max;
117     }

119     @Override
120     public double[] getSystemLoadAverage(int nelem) {
121         if (nelem < 1 || nelem > 3) {
122             throw new IllegalArgumentException("Must include from one to three elements.");
123         }
124         double[] average = new double[nelem];
125         try (Arena arena = Arena.ofConfined()) {
126             MemorySegment buf = arena.allocate((long) nelem * Double.BYTES);
127             int retval = SolarisLibcFunctions.getloadavg(buf, nelem);
128             int validCount = Math.max(retval, 0);
129             for (int i = 0; i < validCount && i < nelem; i++) {
130                 average[i] = buf.get(ValueLayout.JAVA_DOUBLE, (long) i * Double.BYTES);
131             }
132             for (int i = validCount; i < nelem; i++) {
133                 average[i] = -1d;
134             }
135         } catch (Throwable t) {
136             LOG.warn("getloadavg failed", t);
137             Arrays.fill(average, -1d);
138         }
139         return average;
140     }

142     @Override
143     public long[][] queryProcessorCpuLoadTicks() {
144         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
145         int cpu = -1;
146         try (KstatChain kc = KstatUtilFFM.openChain()) {
147             List<MemorySegment> kstats = kc.lookupAll("cpu", -1, "sys");
148             for (MemorySegment ksp : kstats) {
149                 if (++cpu >= ticks.length) {
150                     break;
151                 }
152                 if (kc.read(ksp)) {
153                     ticks[cpu][TickType.IDLE.getIndex()] = KstatUtilFFM.dataLookupLong(ksp, "cpu_ticks_idle");
154                     ticks[cpu][TickType.SYSTEM.getIndex()] = KstatUtilFFM.dataLookupLong(ksp, "cpu_ticks_kernel");
155                     ticks[cpu][TickType.USER.getIndex()] = KstatUtilFFM.dataLookupLong(ksp, "cpu_ticks_user");
156                 }
157             }
158         }
159         return ticks;
160     }
