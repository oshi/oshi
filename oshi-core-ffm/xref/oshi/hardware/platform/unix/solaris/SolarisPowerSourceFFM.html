8   import java.time.LocalDate;
9   import java.util.Arrays;

12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM;
14  import oshi.ffm.util.platform.unix.solaris.KstatUtilFFM.KstatChain;
15  import oshi.hardware.PowerSource;

17  import oshi.util.Constants;
18  
19  @ThreadSafe
20  public final class SolarisPowerSourceFFM extends AbstractPowerSource {
21  
22      private static final String[] KSTAT_BATT_MOD = { null, "battery", "acpi_drv" };
23  
24      private static final int KSTAT_BATT_IDX;
25  
26      static {
27          int idx = 0;
28          try (KstatChain kc = KstatUtilFFM.openChain()) {
29              if (kc.lookup(KSTAT_BATT_MOD[1], 0, null).address() != 0L) {
30                  idx = 1;
31              } else if (kc.lookup(KSTAT_BATT_MOD[2], 0, null).address() != 0L) {
32                  idx = 2;
33              }
34          }
35          KSTAT_BATT_IDX = idx;
36      }
37  
38      public SolarisPowerSourceFFM(String psName, String psDeviceName, double psRemainingCapacityPercent,
39              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
40              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
41              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
42              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
43              String psSerialNumber, double psTemperature) {
44          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
45                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
46                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
47                  psManufacturer, psSerialNumber, psTemperature);
48      }
49  
50      @Override
51      protected List<PowerSource> queryPowerSources() {
52          return getPowerSources();
53      }
54  
55      public static List<PowerSource> getPowerSources() {
56          return Arrays.asList(getPowerSource("BAT0"));
57      }
58  
59      private static SolarisPowerSourceFFM getPowerSource(String name) {
60          String psName = name;
61          String psDeviceName = Constants.UNKNOWN;
62          double psRemainingCapacityPercent = 1d;
63          double psTimeRemainingEstimated = -1d;
64          double psTimeRemainingInstant = 0d;
65          double psPowerUsageRate = 0d;
66          double psVoltage = -1d;
67          double psAmperage = 0d;
68          boolean psPowerOnLine = false;
69          boolean psCharging = false;
70          boolean psDischarging = false;
71          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
72          int psCurrentCapacity = 0;
73          int psMaxCapacity = 1;
74          int psDesignCapacity = 1;
75          int psCycleCount = -1;
76          String psChemistry = Constants.UNKNOWN;
77          LocalDate psManufactureDate = null;
78          String psManufacturer = Constants.UNKNOWN;
79          String psSerialNumber = Constants.UNKNOWN;
80          double psTemperature = 0d;
81  
82          if (KSTAT_BATT_IDX > 0) {
83              try (KstatChain kc = KstatUtilFFM.openChain()) {
84                  MemorySegment ksp = kc.lookup(KSTAT_BATT_MOD[KSTAT_BATT_IDX], 0, "battery BIF0");
85                  if (ksp.address() != 0L && kc.read(ksp)) {
86                      long energyFull = KstatUtilFFM.dataLookupLong(ksp, "bif_last_cap");
87                      if (energyFull == 0xffffffffL || energyFull <= 0) {
88                          energyFull = KstatUtilFFM.dataLookupLong(ksp, "bif_design_cap");
89                      }
90                      if (energyFull != 0xffffffffL && energyFull > 0) {
91                          psMaxCapacity = (int) energyFull;
92                      }
93                      long unit = KstatUtilFFM.dataLookupLong(ksp, "bif_unit");
94                      if (unit == 0) {
95                          psCapacityUnits = CapacityUnits.MWH;
96                      } else if (unit == 1) {
97                          psCapacityUnits = CapacityUnits.MAH;
98                      }
99                      psDeviceName = KstatUtilFFM.dataLookupString(ksp, "bif_model");
100                     psSerialNumber = KstatUtilFFM.dataLookupString(ksp, "bif_serial");
101                     psChemistry = KstatUtilFFM.dataLookupString(ksp, "bif_type");
102                     psManufacturer = KstatUtilFFM.dataLookupString(ksp, "bif_oem_info");
103                 }
104 
105                 ksp = kc.lookup(KSTAT_BATT_MOD[KSTAT_BATT_IDX], 0, "battery BST0");
106                 if (ksp.address() != 0L && kc.read(ksp)) {
107                     long energyNow = KstatUtilFFM.dataLookupLong(ksp, "bst_rem_cap");
108                     if (energyNow >= 0) {
109                         psCurrentCapacity = (int) energyNow;
110                     }
111                     long powerNow = KstatUtilFFM.dataLookupLong(ksp, "bst_rate");
112                     if (powerNow == 0xFFFFFFFFL) {
113                         powerNow = 0L;
114                     }
115                     psPowerUsageRate = powerNow;
116                     // Battery State:
117                     // bit 0 = discharging
118                     // bit 1 = charging
119                     // bit 2 = critical energy state
120                     long bstState = KstatUtilFFM.dataLookupLong(ksp, "bst_state");
121                     psDischarging = (bstState & 0x1) > 0;
122                     psCharging = (bstState & 0x2) > 0;
123 
124                     if (psDischarging) {
125                         psTimeRemainingEstimated = powerNow > 0 ? 3600d * energyNow / powerNow : -1d;
126                     }
127 
128                     long voltageNow = KstatUtilFFM.dataLookupLong(ksp, "bst_voltage");
129                     if (voltageNow > 0) {
130                         psVoltage = voltageNow / 1000d;
131                         psAmperage = psPowerUsageRate * 1000d / voltageNow;
132                     }
133                 }
134             }
135         }
136 
137         return new SolarisPowerSourceFFM(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated,
138                 psTimeRemainingInstant, psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging,
139                 psDischarging, psCapacityUnits, psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount,
140                 psChemistry, psManufactureDate, psManufacturer, psSerialNumber, psTemperature);
141     }
