28   * MacNetworks FFM implementation.
29   */
30  @ThreadSafe
31  public final class MacNetworkIfFFM extends MacNetworkIF {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(MacNetworkIfFFM.class);
34  
35      public MacNetworkIfFFM(NetworkInterface netint, Map<Integer, IFdata> data) throws InstantiationException {
36          super(netint, queryIfDisplayName(netint));
37          updateNetworkStats(data);
38      }
39  
40      private static String queryIfDisplayName(NetworkInterface netint) {
41          String name = netint.getName();
42          try {
43              MemorySegment ifArray = SCNetworkInterfaceCopyAll();
44              if (ifArray.equals(MemorySegment.NULL)) {
45                  return name;
46              }
47              CFArrayRef cfArray = new CFArrayRef(ifArray);
48              try (cfArray) {
49                  int count = cfArray.getCount();
50                  for (int i = 0; i < count; i++) {
51                      MemorySegment pNetIf = cfArray.getValueAtIndex(i);
52                      MemorySegment cfNameSeg = SCNetworkInterfaceGetBSDName(pNetIf);
53                      if (!cfNameSeg.equals(MemorySegment.NULL) && name.equals(CFStringRef.stringValue(cfNameSeg))) {
54                          MemorySegment cfDisplayNameSeg = SCNetworkInterfaceGetLocalizedDisplayName(pNetIf);
55                          if (!cfDisplayNameSeg.equals(MemorySegment.NULL)) {
56                              return CFStringRef.stringValue(cfDisplayNameSeg);
57                          }
58                      }
59                  }
60              }
61          } catch (Throwable _) {
62              LOG.debug("Failed to query SC network interface display name for {}", name);
63          }
64          return name;
65      }
66  
67      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
68          Map<Integer, IFdata> data = NetStatFFM.queryIFdata(-1);
69          return getNetworks(includeLocalInterfaces, ni -> new MacNetworkIfFFM(ni, data));
70      }
71  
72      @Override
73      protected Map<Integer, IFdata> queryIFdata(int index) {
74          return NetStatFFM.queryIFdata(index);
75      }
76  }
