5   package oshi.ffm;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.notNullValue;
11  import static org.hamcrest.Matchers.oneOf;
12  
13  import java.util.List;
14  
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledForJreRange;
17  import org.junit.jupiter.api.condition.EnabledOnOs;
18  import org.junit.jupiter.api.condition.JRE;
19  import org.junit.jupiter.api.condition.OS;
20  
21  import oshi.software.os.OSProcess;
22  import oshi.software.os.windows.WindowsOperatingSystemFFM;
23  
24  @EnabledForJreRange(min = JRE.JAVA_25)
25  @EnabledOnOs(OS.WINDOWS)
26  class WindowsOperatingSystemFFMTest {
27  
28      private final TestWindowsOperatingSystemFFM os = new TestWindowsOperatingSystemFFM();
29  
30      @Test
31      void testQueryBitness() {
32          assertThat("64-bit JVM bitness should be returned directly", os.queryBitnessForTest(64), is(64));
33          assertThat("32-bit JVM bitness should be 32 or 64", os.queryBitnessForTest(32), is(oneOf(32, 64)));
34      }
35  
36      @Test
37      void testQueryChildProcesses() {
38          int pid = os.getProcessId();
39          assertThat("Current process id should be positive", pid, is(greaterThan(0)));
40          List<OSProcess> childProcesses = os.queryChildProcessesForTest(pid);
41          assertThat("Child process query should not be null", childProcesses, is(notNullValue()));
42          assertThat("Child process query should include the queried process",
43                  childProcesses.stream().anyMatch(p -> p.getProcessID() == pid), is(true));
44      }
45  
46      @Test
47      void testQueryDescendantProcesses() {
48          int pid = os.getProcessId();
49          assertThat("Current process id should be positive", pid, is(greaterThan(0)));
50          List<OSProcess> descendantProcesses = os.queryDescendantProcessesForTest(pid);
51          assertThat("Descendant process query should not be null", descendantProcesses, is(notNullValue()));
52          assertThat("Descendant process query should include the queried process",
53                  descendantProcesses.stream().anyMatch(p -> p.getProcessID() == pid), is(true));
54      }
55  
56      private static final class TestWindowsOperatingSystemFFM extends WindowsOperatingSystemFFM {
57          private int queryBitnessForTest(int jvmBitness) {
58              return queryBitness(jvmBitness);
59          }
60  
61          private List<OSProcess> queryChildProcessesForTest(int parentPid) {
62              return queryChildProcesses(parentPid);
63          }
64  
65          private List<OSProcess> queryDescendantProcessesForTest(int parentPid) {
66              return queryDescendantProcesses(parentPid);
67          }
68      }
69  }
