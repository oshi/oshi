25  /**
26   * Source-policy guard for a footgun that has already regressed twice (issues #3301 / #3422): a <em>void</em>
27   * {@link java.lang.invoke.MethodHandle#invokeExact} invoked from an expression-bodied lambda passed to a void-returning
28   * helper such as {@code ExceptionUtil.runOrLog}.
29   * <p>
30   * {@code invokeExact} is signature-polymorphic: its return type is inferred from the call site. In an expression lambda
31   * targeting a void functional interface, javac infers {@code Object} instead of {@code void}, so the call site
32   * ({@code (X)Object}) does not match the handle ({@code (X)void}) and throws {@code WrongMethodTypeException} at
33   * runtime. The failure only surfaces on the native platform and is swallowed by the logging helper, so a normal unit
34   * test can't see it. A block lambda ({@code () -> { handle.invokeExact(...); }}) keeps the call in a statement context
35   * where the return type is correctly {@code void}.
36   * <p>
37   * Scanning the sources catches the regression on every build (not only on the platform where the call runs), which
38   * matters because the fix keeps getting lost whenever the FFM code is moved between packages.
39   */
40  class VoidInvokeExactSourceTest {
41  
42      // run(OrLog|Silently)(() -> ... invokeExact with no '{' before invokeExact is an expression lambda -- the bug.
43      // Scanned against whole-file content, so a lambda wrapped across lines is still caught. [^{};]* stays within the
44      // single statement (no brace/semicolon crossing) to avoid matching an unrelated later invokeExact.
45      private static final Pattern EXPRESSION_LAMBDA_INVOKE_EXACT = Pattern
46              .compile("run(?:OrLog|Silently)\\(\\(\\)\\s*->[^{};]*invokeExact");
47  
48      @Test
49      void noVoidInvokeExactInExpressionLambda() throws IOException {
50          Path srcRoot = Path.of("src", "main", "java");
51          assumeTrue(Files.isDirectory(srcRoot), "FFM source tree not on the working directory; skipping source scan");
52  
53          List<String> offenders = new ArrayList<>();
54          try (Stream<Path> paths = Files.walk(srcRoot)) {
55              paths.filter(p -> p.toString().endsWith(".java")).forEach(p -> {
56                  try {
57                      String content = Files.readString(p, StandardCharsets.UTF_8);
58                      Matcher matcher = EXPRESSION_LAMBDA_INVOKE_EXACT.matcher(content);
59                      while (matcher.find()) {
60                          long line = content.substring(0, matcher.start()).chars().filter(c -> c == '\n').count() + 1;
61                          offenders.add(p + ":" + line + "  " + matcher.group().replaceAll("\\s+", " "));
62                      }
63                  } catch (IOException e) {
64                      throw new UncheckedIOException(e);
65                  }
66              });
67          }
68  
69          assertThat("void invokeExact in an expression lambda infers Object and throws WrongMethodTypeException at "
70                  + "runtime; use a block lambda instead: () -> { handle.invokeExact(...); }. Offenders: " + offenders,
71                  offenders, is(empty()));
72      }
73  }
