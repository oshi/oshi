9   import static org.hamcrest.Matchers.greaterThan;
10  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
11  import static org.hamcrest.Matchers.is;
12  import static org.hamcrest.Matchers.not;
13  import static org.hamcrest.Matchers.notNullValue;
14  import static org.junit.jupiter.api.Assertions.assertTrue;

16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.TestInstance;
18  import org.junit.jupiter.api.TestInstance.Lifecycle;
19  import org.junit.jupiter.api.condition.EnabledForJreRange;
20  import org.junit.jupiter.api.condition.EnabledOnOs;
21  import org.junit.jupiter.api.condition.JRE;
22  import org.junit.jupiter.api.condition.OS;
23  
24  import oshi.software.os.OSProcess;
25  import oshi.software.os.OperatingSystem;
26  import oshi.software.os.mac.MacOperatingSystemFFM;
27  
28  @EnabledForJreRange(min = JRE.JAVA_25)
29  @TestInstance(Lifecycle.PER_CLASS)
30  public class OperatingSystemFFMTest {
31  
32      @Test
33      void testOperatingSystem() {
34          OperatingSystem os = new SystemInfo().getOperatingSystem();
35          assertThat("OS shouldn't be null", os, is(notNullValue()));
36          assertThat("OS family shouldn't be null", os.getFamily(), is(notNullValue()));
37          assertThat("OS should have 1 or more currently running processes", os.getProcessCount(), is(greaterThan(0)));
38          assertThat("OS should have at least 1 currently running process", os.getProcesses(null, null, 0),
39                  is(not(empty())));
40      }
41  
42      @Test
43      void testCurrentProcessOpenFileLimits() {
44          OperatingSystem os = new SystemInfo().getOperatingSystem();
45          OSProcess process = os.getProcess(os.getProcessId());
46          assertThat("Current process shouldn't be null", process, is(notNullValue()));
47          long softLimit = process.getSoftOpenFileLimit();
48          long hardLimit = process.getHardOpenFileLimit();
49          assertThat("Soft open file limit for current process should be positive", softLimit, is(greaterThan(0L)));
50          assertThat("Hard open file limit for current process should be at least the soft limit", hardLimit,
51                  is(greaterThanOrEqualTo(softLimit)));
52      }
53  
54      @EnabledOnOs(OS.MAC)
55      @Test
56      void testGetMacOperatingSystem() {
57          assertTrue(new SystemInfo().getOperatingSystem() instanceof MacOperatingSystemFFM);
58      }
59  }
