69      @Test
70      void testCFTypeRefEqualsAndHashCode() {
71          try (var str1 = CFStringRef.createCFString("test");
72                  var str2 = CFStringRef.createCFString("test");
73                  var str3 = CFStringRef.createCFString("other")) {
74              assertThat(str1.equals(str2), is(true));
75              assertThat(str1.hashCode(), is(str2.hashCode()));
76              assertThat(str1.equals(str3), is(false));
77              assertThat(str1.equals(null), is(false));
78              assertThat(str1.equals(str1), is(true));
79              // Null refs are equal to each other
80              var nullRef1 = new CFTypeRef(NULL);
81              var nullRef2 = new CFTypeRef(NULL);
82              assertThat(nullRef1.equals(nullRef2), is(true));
83              assertThat(nullRef1.hashCode(), is(nullRef2.hashCode()));
84              assertThat(nullRef1.equals(str1), is(false));
85          }
86      }

88      @Test
89      void testCFTypeRefRetainRelease() throws Throwable {
90          var str = CFStringRef.createCFString("retain-test");
91          try {
92              long countBefore = CoreFoundationFunctions.CFGetRetainCount(str.segment());
93              str.retain();
94              long countAfter = CoreFoundationFunctions.CFGetRetainCount(str.segment());
95              assertThat(countAfter, is(countBefore + 1));
96              // Release the extra retain
97              str.release();
98              assertThat(CoreFoundationFunctions.CFGetRetainCount(str.segment()), is(countBefore));
99          } finally {
100             str.release();
101         }
102     }

315     @Test
316     void testCFArrayWithStrings() throws Throwable {
317         try (var arena = Arena.ofConfined();
318                 var str1 = CFStringRef.createCFString("a");
319                 var str2 = CFStringRef.createCFString("b")) {
320             var values = arena.allocate(ValueLayout.ADDRESS, 2);
321             values.setAtIndex(ValueLayout.ADDRESS, 0, str1.segment());
322             values.setAtIndex(ValueLayout.ADDRESS, 1, str2.segment());
323             var allocator = CoreFoundationFunctions.CFAllocatorGetDefault();
324             var callbacks = CF_LOOKUP.findOrThrow("kCFTypeArrayCallBacks");
325             var arraySeg = CoreFoundationFunctions.CFArrayCreate(allocator, values, 2, callbacks);
326             try (var cfArray = new CFArrayRef(arraySeg)) {
327                 assertThat(cfArray.isTypeID(CoreFoundationFunctions.ARRAY_TYPE_ID), is(true));
328                 assertThat(cfArray.getCount(), is(2));
329                 var elem0 = new CFStringRef(cfArray.getValueAtIndex(0));
330                 assertThat(elem0.stringValue(), is("a"));
331             }
332         }
333     }

346     @Test
347     void testCFMutableDictionarySetAndGet() throws Throwable {
348         try (var arena = Arena.ofConfined()) {
349             var allocator = CoreFoundationFunctions.CFAllocatorGetDefault();
350             var keyCallbacks = CF_LOOKUP.findOrThrow("kCFTypeDictionaryKeyCallBacks");
351             var valCallbacks = CF_LOOKUP.findOrThrow("kCFTypeDictionaryValueCallBacks");
352             var dictSeg = CoreFoundationFunctions.CFDictionaryCreateMutable(allocator, 0, keyCallbacks, valCallbacks);
353             try (var dict = new CFMutableDictionaryRef(dictSeg);
354                     var key = CFStringRef.createCFString("myKey");
355                     var val = CFStringRef.createCFString("myValue")) {
356                 dict.setValue(key, val);
357                 assertThat(dict.getCount(), is(1L));
358 
359                 var result = dict.getValue(key);
360                 assertThat(result, is(not(NULL)));
361                 assertThat(new CFStringRef(result).stringValue(), is("myValue"));
362 
363                 // getValueIfPresent
364                 var outPtr = arena.allocate(ValueLayout.ADDRESS);
365                 assertThat(dict.getValueIfPresent(key, outPtr), is(true));
366 
367                 // Missing key
368                 try (var missingKey = CFStringRef.createCFString("noSuchKey")) {
369                     assertThat(dict.getValue(missingKey), is(NULL));
370                     assertThat(dict.getValueIfPresent(missingKey, outPtr), is(false));
371                 }
372             }
373         }
374     }
