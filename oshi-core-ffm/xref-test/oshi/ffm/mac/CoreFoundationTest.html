69      @Test
70      void testCFTypeRefEqualsAndHashCode() {
71          var str1 = CFStringRef.createCFString("test");
72          var str2 = CFStringRef.createCFString("test");
73          var str3 = CFStringRef.createCFString("other");
74          try {
75              assertThat(str1.equals(str2), is(true));
76              assertThat(str1.hashCode(), is(str2.hashCode()));
77              assertThat(str1.equals(str3), is(false));
78              assertThat(str1.equals(null), is(false));
79              assertThat(str1.equals(str1), is(true));
80              // Null refs are equal to each other
81              var nullRef1 = new CFTypeRef(NULL);
82              var nullRef2 = new CFTypeRef(NULL);
83              assertThat(nullRef1.equals(nullRef2), is(true));
84              assertThat(nullRef1.hashCode(), is(nullRef2.hashCode()));
85              assertThat(nullRef1.equals(str1), is(false));
86          } finally {
87              str1.release();
88              str2.release();
89              str3.release();
90          }
91      }

93      @Test
94      void testCFTypeRefRetainRelease() throws Throwable {
95          var str = CFStringRef.createCFString("retain-test");
96          try {
97              long countBefore = CoreFoundationFunctions.CFGetRetainCount(str.segment());
98              str.retain();
99              long countAfter = CoreFoundationFunctions.CFGetRetainCount(str.segment());
100             assertThat(countAfter, is(countBefore + 1));
101             // Release the extra retain
102             str.release();
103             assertThat(CoreFoundationFunctions.CFGetRetainCount(str.segment()), is(countBefore));
104         } finally {
105             str.release();
106         }
107     }

153         try (var arena = Arena.ofConfined()) {

343     @Test
344     void testCFArrayWithStrings() throws Throwable {
345         var str1 = CFStringRef.createCFString("a");
346         var str2 = CFStringRef.createCFString("b");
347         try (var arena = Arena.ofConfined()) {
348             var values = arena.allocate(ValueLayout.ADDRESS, 2);
349             values.setAtIndex(ValueLayout.ADDRESS, 0, str1.segment());
350             values.setAtIndex(ValueLayout.ADDRESS, 1, str2.segment());
351             var allocator = CoreFoundationFunctions.CFAllocatorGetDefault();
352             var callbacks = CF_LOOKUP.findOrThrow("kCFTypeArrayCallBacks");
353             var arraySeg = CoreFoundationFunctions.CFArrayCreate(allocator, values, 2, callbacks);
354             var cfArray = new CFArrayRef(arraySeg);
355             try {
356                 assertThat(cfArray.isTypeID(CoreFoundationFunctions.ARRAY_TYPE_ID), is(true));
357                 assertThat(cfArray.getCount(), is(2));
358                 var elem0 = new CFStringRef(cfArray.getValueAtIndex(0));
359                 assertThat(elem0.stringValue(), is("a"));
360             } finally {
361                 cfArray.release();
362             }
363         } finally {
364             str1.release();
365             str2.release();
366         }
367     }

380     @Test
381     void testCFMutableDictionarySetAndGet() throws Throwable {
382         try (var arena = Arena.ofConfined()) {
383             var allocator = CoreFoundationFunctions.CFAllocatorGetDefault();
384             var keyCallbacks = CF_LOOKUP.findOrThrow("kCFTypeDictionaryKeyCallBacks");
385             var valCallbacks = CF_LOOKUP.findOrThrow("kCFTypeDictionaryValueCallBacks");
386             var dictSeg = CoreFoundationFunctions.CFDictionaryCreateMutable(allocator, 0, keyCallbacks, valCallbacks);
387             var dict = new CFMutableDictionaryRef(dictSeg);
388             try {
389                 var key = CFStringRef.createCFString("myKey");
390                 var val = CFStringRef.createCFString("myValue");
391                 try {
392                     dict.setValue(key, val);
393                     assertThat(dict.getCount(), is(1L));
394 
395                     var result = dict.getValue(key);
396                     assertThat(result, is(not(NULL)));
397                     assertThat(new CFStringRef(result).stringValue(), is("myValue"));
398 
399                     // getValueIfPresent
400                     var outPtr = arena.allocate(ValueLayout.ADDRESS);
401                     assertThat(dict.getValueIfPresent(key, outPtr), is(true));
402 
403                     // Missing key
404                     var missingKey = CFStringRef.createCFString("noSuchKey");
405                     try {
406                         assertThat(dict.getValue(missingKey), is(NULL));
407                         assertThat(dict.getValueIfPresent(missingKey, outPtr), is(false));
408                     } finally {
409                         missingKey.release();
410                     }
411                 } finally {
412                     key.release();
413                     val.release();
414                 }
415             } finally {
416                 dict.release();
417             }
418         }
419     }
