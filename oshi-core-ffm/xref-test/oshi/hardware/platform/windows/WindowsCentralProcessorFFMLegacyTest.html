5   package oshi.hardware.platform.windows;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.closeTo;
9   import static org.hamcrest.Matchers.greaterThan;
10  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
11  import static org.hamcrest.Matchers.is;
12  
13  import java.util.stream.LongStream;
14  
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledOnOs;
17  import org.junit.jupiter.api.condition.OS;
18  
19  import oshi.hardware.CentralProcessor.TickType;
20  
21  /**
22   * Exercises the FFM legacy {@code GetSystemTimes} system-CPU-tick path against the real system by forcing the
23   * {@code useLegacySystemCounters()} gate. Only the gate is overridden, so the genuine native call runs; a rotted legacy
24   * binding (e.g. a broken {@code Kernel32FFM.GetSystemTimes} or FILETIME read) surfaces as all-zero ticks or a total
25   * wildly inconsistent with the modern per-processor sum rather than as a mere coverage number.
26   */
27  @EnabledOnOs(OS.WINDOWS)
28  class WindowsCentralProcessorFFMLegacyTest {
29  
30      /** A real {@link WindowsCentralProcessorFFM} with only the legacy system-counter gate forced. */
31      private static final class ForcedLegacy extends WindowsCentralProcessorFFM {
32          private final boolean legacy;
33  
34          ForcedLegacy(boolean legacy) {
35              this.legacy = legacy;
36          }
37  
38          @Override
39          protected boolean useLegacySystemCounters() {
40              return legacy;
41          }
42      }
43  
44      /**
45       * The legacy {@code GetSystemTimes} path and the modern per-processor perfmon sum both measure cumulative CPU time
46       * since boot, so their totals must agree closely. A broken legacy binding shows up here as zero idle/user ticks or
47       * a total far from the modern one.
48       */
49      @Test
50      void legacySystemCpuLoadTicksAreSaneAndMatchModernTotal() {
51          long[] legacy = new ForcedLegacy(true).querySystemCpuLoadTicks();
52          long[] modern = new ForcedLegacy(false).querySystemCpuLoadTicks();
53  
54          assertThat("legacy ticks array length", legacy.length, is(TickType.values().length));
55          for (long tick : legacy) {
56              assertThat("legacy tick non-negative", tick, greaterThanOrEqualTo(0L));
57          }
58          // A successful GetSystemTimes yields non-zero idle and user time on any running system
59          assertThat("legacy idle ticks", legacy[TickType.IDLE.getIndex()], greaterThan(0L));
60          assertThat("legacy user ticks", legacy[TickType.USER.getIndex()], greaterThan(0L));
61  
62          long legacyTotal = LongStream.of(legacy).sum();
63          long modernTotal = LongStream.of(modern).sum();
64          assertThat("legacy total ticks", legacyTotal, greaterThan(0L));
65          // Both measure cumulative CPU time since boot; a generous tolerance catches a rotted binding without flaking
66          assertThat((double) legacyTotal, closeTo(modernTotal, modernTotal * 0.5));
67      }
68  }
