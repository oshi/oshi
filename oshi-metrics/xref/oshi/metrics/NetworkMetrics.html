8   import java.util.EnumMap;
9   import java.util.List;

11  import java.util.Map;
12  import java.util.concurrent.atomic.AtomicReference;
13  import java.util.function.Supplier;

15  import oshi.hardware.NetworkIF;
16  import oshi.software.os.InternetProtocolStats;
17  import oshi.software.os.InternetProtocolStats.IPConnection;
18  import oshi.software.os.InternetProtocolStats.TcpState;
19  
20  import io.micrometer.core.instrument.FunctionCounter;
21  import io.micrometer.core.instrument.Gauge;
22  import io.micrometer.core.instrument.MeterRegistry;
23  import io.micrometer.core.instrument.binder.MeterBinder;

25  /**
26   * {@link MeterBinder} for system network metrics following
27   * <a href="https://opentelemetry.io/docs/specs/semconv/system/system-metrics/#network-metrics">OpenTelemetry semantic
28   * conventions</a>.
29   *
30   * <p>
31   * Registers per network interface:
32   * <ul>
33   * <li>{@code system.network.io} — network bytes transferred by direction (transmit, receive)</li>
34   * <li>{@code system.network.packet.count} — network packets by direction (transmit, receive)</li>
35   * <li>{@code system.network.packet.dropped} — dropped packets (receive only; OSHI does not expose transmit drops)</li>
36   * <li>{@code system.network.errors} — network errors by direction (transmit, receive)</li>
37   * </ul>
38   *
39   * <p>
40   * Registers aggregate:
41   * <ul>
42   * <li>{@code system.network.connection.count} — connection count by protocol and state</li>
43   * </ul>
44   */
45  public class NetworkMetrics implements MeterBinder {
46  
47      private static final String NET_IO = "system.network.io";
48      private static final String NET_PACKETS = "system.network.packet.count";
49      private static final String NET_DROPPED = "system.network.packet.dropped";
50      private static final String NET_ERRORS = "system.network.errors";
51      private static final String NET_CONNECTIONS = "system.network.connection.count";
52      private static final String DEVICE_KEY = "system.device";
53      private static final String DIRECTION_KEY = "network.io.direction";
54      private static final String TRANSPORT_KEY = "network.transport";
55      private static final String STATE_KEY = "network.connection.state";
56      private static final long CACHE_TTL_MS = 1000L;
57  
58      private final Supplier<List<NetworkIF>> networkIFSupplier;
59      private final InternetProtocolStats ipStats;
60  
61      // Strong reference to prevent GC of NetworkIF objects used by FunctionCounter
62      private List<NetworkIF> networkIFs;
63  
64      // Connection count cache to avoid repeated getConnections() calls per scrape
65      private volatile long cacheTimestamp;
66      private final AtomicReference<Map<TcpState, Long>> tcpCounts = new AtomicReference<>(Collections.emptyMap());
67      private volatile long udpCount;
68  
69      /**
70       * Creates a new {@code NetworkMetrics} binder.
71       *
72       * @param networkIFSupplier supplier that returns the current list of {@link NetworkIF} instances
73       * @param ipStats           the {@link InternetProtocolStats} instance for connection counting
74       */
75      public NetworkMetrics(Supplier<List<NetworkIF>> networkIFSupplier, InternetProtocolStats ipStats) {
76          this.networkIFSupplier = networkIFSupplier;
77          this.ipStats = ipStats;
78      }
79  
80      @Override
81      public void bindTo(MeterRegistry registry) {
82          this.networkIFs = networkIFSupplier.get();
83  
84          for (NetworkIF net : networkIFs) {
85              String device = net.getName();
86  
87              // system.network.io — Counter, unit "By", attrs: network.io.direction, system.device
88              FunctionCounter.builder(NET_IO, net, n -> {
89                  n.updateAttributes();
90                  return n.getBytesRecv();
91              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "receive").description("Network bytes transferred")
92                      .baseUnit("By").register(registry);
93              FunctionCounter.builder(NET_IO, net, n -> {
94                  n.updateAttributes();
95                  return n.getBytesSent();
96              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "transmit").description("Network bytes transferred")
97                      .baseUnit("By").register(registry);
98  
99              // system.network.packet.count — Counter, unit "{packet}", attrs: network.io.direction, system.device
100             FunctionCounter.builder(NET_PACKETS, net, n -> {
101                 n.updateAttributes();
102                 return n.getPacketsRecv();
103             }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "receive").description("Network packets transferred")
104                     .baseUnit("{packet}").register(registry);
105             FunctionCounter.builder(NET_PACKETS, net, n -> {
106                 n.updateAttributes();
107                 return n.getPacketsSent();
108             }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "transmit").description("Network packets transferred")
109                     .baseUnit("{packet}").register(registry);
110 
111             // system.network.packet.dropped — Counter, unit "{packet}", attrs: network.io.direction, system.device
112             FunctionCounter.builder(NET_DROPPED, net, n -> {
113                 n.updateAttributes();
114                 return n.getInDrops();
115             }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "receive").description("Count of packets dropped")
116                     .baseUnit("{packet}").register(registry);
117 
118             // system.network.errors — Counter, unit "{error}", attrs: network.io.direction, system.device
119             FunctionCounter.builder(NET_ERRORS, net, n -> {
120                 n.updateAttributes();
121                 return n.getInErrors();
122             }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "receive").description("Network errors").baseUnit("{error}")
123                     .register(registry);
124             FunctionCounter.builder(NET_ERRORS, net, n -> {
125                 n.updateAttributes();
126                 return n.getOutErrors();
127             }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "transmit").description("Network errors").baseUnit("{error}")
128                     .register(registry);
129         }
130 
131         // system.network.connection.count — UpDownCounter (Gauge), unit "{connection}",
132         // attrs: network.transport, network.connection.state
133         registerConnectionCountGauges(registry);
134     }
135 
136     private void registerConnectionCountGauges(MeterRegistry registry) {
137         for (TcpState state : TcpState.values()) {
138             if (state == TcpState.NONE) {
139                 continue;
140             }
141             String stateValue = state.name().toLowerCase(Locale.ROOT);
142             Gauge.builder(NET_CONNECTIONS, this, self -> self.getCachedTcpCount(state)).tag(TRANSPORT_KEY, "tcp")
143                     .tag(STATE_KEY, stateValue).description("Total number of connections in each state")
144                     .baseUnit("{connection}").strongReference(true).register(registry);
145         }
146         Gauge.builder(NET_CONNECTIONS, this, self -> self.getCachedUdpCount()).tag(TRANSPORT_KEY, "udp")
147                 .description("Total number of UDP connections").baseUnit("{connection}").strongReference(true)
148                 .register(registry);
149     }
150 
151     private void refreshCache() {
152         if (System.currentTimeMillis() - cacheTimestamp > CACHE_TTL_MS) {
153             synchronized (this) {
154                 long now = System.currentTimeMillis();
155                 if (now - cacheTimestamp > CACHE_TTL_MS) {
156                     List<IPConnection> connections = ipStats.getConnections();
157                     Map<TcpState, Long> tcp = new EnumMap<>(TcpState.class);
158                     long udp = 0;
159                     for (IPConnection conn : connections) {
160                         if (conn.getType().startsWith("tcp")) {
161                             tcp.merge(conn.getState(), 1L, Long::sum);
162                         } else if (conn.getType().startsWith("udp")) {
163                             udp++;
164                         }
165                     }
166                     this.tcpCounts.set(Collections.unmodifiableMap(tcp));
167                     this.udpCount = udp;
168                     this.cacheTimestamp = now;
169                 }
170             }
171         }
172     }
173 
174     private double getCachedTcpCount(TcpState state) {
175         refreshCache();
176         return tcpCounts.get().getOrDefault(state, 0L);
177     }
178 
179     private double getCachedUdpCount() {
180         refreshCache();
181         return udpCount;
182     }
183 }
