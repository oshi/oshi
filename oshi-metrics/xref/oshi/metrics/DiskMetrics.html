5   package oshi.metrics;
6   
7   import java.util.List;
8   import java.util.function.Supplier;
9   
10  import oshi.hardware.HWDiskStore;
11  
12  import io.micrometer.core.instrument.FunctionCounter;
13  import io.micrometer.core.instrument.Gauge;
14  import io.micrometer.core.instrument.MeterRegistry;
15  import io.micrometer.core.instrument.binder.MeterBinder;
16  
17  /**
18   * {@link MeterBinder} for system disk controller metrics following
19   * <a href="https://opentelemetry.io/docs/specs/semconv/system/system-metrics/#disk-controller-metrics">OpenTelemetry
20   * semantic conventions</a>.
21   *
22   * <p>
23   * Registers per device:
24   * <ul>
25   * <li>{@code system.disk.io} — disk bytes transferred by direction (read, write)</li>
26   * <li>{@code system.disk.operations} — disk operations count by direction (read, write)</li>
27   * <li>{@code system.disk.io_time} — time disk spent activated, in seconds</li>
28   * <li>{@code system.disk.limit} — total storage capacity of the disk, in bytes</li>
29   * </ul>
30   *
31   * <p>
32   * Not implemented:
33   * <ul>
34   * <li>{@code system.disk.operation_time} — OSHI does not expose per-direction operation time</li>
35   * <li>{@code system.disk.merged} — OSHI does not expose merged operation counts</li>
36   * </ul>
37   */
38  public class DiskMetrics implements MeterBinder {
39  
40      private static final String DISK_IO = "system.disk.io";
41      private static final String DISK_OPERATIONS = "system.disk.operations";
42      private static final String DISK_IO_TIME = "system.disk.io_time";
43      private static final String DISK_LIMIT = "system.disk.limit";
44      private static final String DEVICE_KEY = "system.device";
45      private static final String DIRECTION_KEY = "disk.io.direction";
46      private static final double MS_PER_SECOND = 1000.0;
47  
48      private final Supplier<List<HWDiskStore>> diskStoreSupplier;
49      private List<HWDiskStore> diskStores;
50  
51      /**
52       * Creates a new {@code DiskMetrics} binder.
53       *
54       * @param diskStoreSupplier supplier that returns the current list of {@link HWDiskStore} instances
55       */
56      public DiskMetrics(Supplier<List<HWDiskStore>> diskStoreSupplier) {
57          this.diskStoreSupplier = diskStoreSupplier;
58      }
59  
60      @Override
61      public void bindTo(MeterRegistry registry) {
62          List<HWDiskStore> disks = diskStoreSupplier.get();
63          // Hold strong references to prevent GC (FunctionCounter uses WeakReference)
64          this.diskStores = disks;
65  
66          for (HWDiskStore disk : disks) {
67              String device = disk.getName();
68  
69              // system.disk.io — Counter, unit "By", attrs: disk.io.direction, system.device
70              FunctionCounter.builder(DISK_IO, disk, d -> {
71                  d.updateAttributes();
72                  return d.getReadBytes();
73              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "read").description("Disk bytes transferred").baseUnit("By")
74                      .register(registry);
75              FunctionCounter.builder(DISK_IO, disk, d -> {
76                  d.updateAttributes();
77                  return d.getWriteBytes();
78              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "write").description("Disk bytes transferred").baseUnit("By")
79                      .register(registry);
80  
81              // system.disk.operations — Counter, unit "{operation}", attrs: disk.io.direction, system.device
82              FunctionCounter.builder(DISK_OPERATIONS, disk, d -> {
83                  d.updateAttributes();
84                  return d.getReads();
85              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "read").description("Disk operations count")
86                      .baseUnit("{operation}").register(registry);
87              FunctionCounter.builder(DISK_OPERATIONS, disk, d -> {
88                  d.updateAttributes();
89                  return d.getWrites();
90              }).tag(DEVICE_KEY, device).tag(DIRECTION_KEY, "write").description("Disk operations count")
91                      .baseUnit("{operation}").register(registry);
92  
93              // system.disk.io_time — Counter, unit "s", attr: system.device
94              FunctionCounter.builder(DISK_IO_TIME, disk, d -> {
95                  d.updateAttributes();
96                  return d.getTransferTime() / MS_PER_SECOND;
97              }).tag(DEVICE_KEY, device).description("Time disk spent activated").baseUnit("s").register(registry);
98  
99              // system.disk.limit — UpDownCounter (Gauge), unit "By", attr: system.device
100             Gauge.builder(DISK_LIMIT, disk, HWDiskStore::getSize).tag(DEVICE_KEY, device)
101                     .description("The total storage capacity of the disk").baseUnit("By").strongReference(true)
102                     .register(registry);
103         }
104     }
105 }
