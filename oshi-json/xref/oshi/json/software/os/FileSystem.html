24  package oshi.json.software.os;
25  
26  import oshi.json.json.OshiJsonObject;
27  
28  /**
29   * The File System is a logical arrangement, usually in a hierarchial tree,
30   * where files are placed for storage and retrieval. It may consist of one or
31   * more file stores.
32   *
33   * @author widdis[at]gmail[dot]com
34   */
35  public interface FileSystem extends OshiJsonObject {
36  
37      /**
38       * Get file stores on this machine
39       *
40       * Instantiates an array of {@link OSFileStore} objects, representing a
41       * storage pool, device, partition, volume, concrete file system or other
42       * implementation specific means of file storage.
43       *
44       * @return An array of OSFileStore objects or an empty array if none are
45       *         present.
46       */
47      OSFileStore[] getFileStores();
48  
49      /**
50       * The current number of open file descriptors. A file descriptor is an
51       * abstract handle used to access I/O resources such as files and network
52       * connections. On UNIX-based systems there is a system-wide limit on the
53       * number of open file descriptors.
54       *
55       * On Windows systems, this method returns the total number of handles held
56       * by Processes. While Windows handles are conceptually similar to file
57       * descriptors, they may also refer to a number of non-I/O related objects.
58       *
59       * @return The number of open file descriptors if available, 0 otherwise.
60       */
61      long getOpenFileDescriptors();
62  
63      /**
64       * The maximum number of open file descriptors. A file descriptor is an
65       * abstract handle used to access I/O resources such as files and network
66       * connections. On UNIX-based systems there is a system-wide limit on the
67       * number of open file descriptors.
68       *
69       * On Windows systems, this method returns the theoretical max number of
70       * handles (2^24-2^15 on 32-bit, 2^24-2^16 on 64-bit). There may be a lower
71       * per-process limit. While Windows handles are conceptually similar to file
72       * descriptors, they may also refer to a number of non-I/O related objects.
73       *
74       * @return The maximum number of file descriptors if available, 0 otherwise.
75       */
76      long getMaxFileDescriptors();
77  }
