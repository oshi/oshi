37   * contains the program code and its current activity. Depending on the
38   * operating system (OS), a process may be made up of multiple threads of

175     /**
176      * @return Returns the priority of this process.
177      *
178      *         For Linux and Unix, priority is a value in the range -20 to 19
179      *         (20 on some systems). The default priority is 0; lower priorities
180      *         cause more favorable scheduling.
181      *
182      *         For Windows, priority values can range from 0 (lowest priority)
183      *         to 31 (highest priority).
184      *
185      *         Mac OS X has 128 priority levels, ranging from 0 (lowest
186      *         priority) to 127 (highest priority). They are divided into
187      *         several major bands: 0 through 51 are the normal levels; the
188      *         default priority is 31. 52 through 79 are the highest priority
189      *         regular threads; 80 through 95 are for kernel mode threads; and
190      *         96 through 127 correspond to real-time threads, which are treated
191      *         differently than other threads by the scheduler.
192      */
193     public int getPriority() {
194         return this.osProcess.getPriority();
195     }

536     /**
537      * {@inheritDoc}
538      */
539     @Override
540     public JsonObject toJSON(Properties properties) {
541         JsonObjectBuilder json = NullAwareJsonObjectBuilder.wrap(this.jsonFactory.createObjectBuilder());
542         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.name")) {
543             json.add("name", getName());
544         }
545         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.path")) {
546             json.add("path", getPath());
547         }
548         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.commandLine")) {
549             json.add("commandLine", getCommandLine());
550         }
551         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.currentWorkingDirectory")) {
552             json.add("currentWorkingDirectory", getCurrentWorkingDirectory());
553         }
554         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.user")) {
555             json.add("user", getUser());
556         }
557         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.userID")) {
558             json.add("userID", getUserID());
559         }
560         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.group")) {
561             json.add("group", getGroup());
562         }
563         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.groupID")) {
564             json.add("groupID", getGroupID());
565         }
566         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.state")) {
567             json.add("state", getState().name());
568         }
569         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.processID")) {
570             json.add("processID", getProcessID());
571         }
572         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.parentProcessID")) {
573             json.add("parentProcessID", getParentProcessID());
574         }
575         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.threadCount")) {
576             json.add("threadCount", getThreadCount());
577         }
578         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.priority")) {
579             json.add("priority", getPriority());
580         }
581         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.virtualSize")) {
582             json.add("virtualSize", getVirtualSize());
583         }
584         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.residentSetSize")) {
585             json.add("residentSetSize", getResidentSetSize());
586         }
587         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.kernelTime")) {
588             json.add("kernelTime", getKernelTime());
589         }
590         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.userTime")) {
591             json.add("userTime", getUserTime());
592         }
593         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.upTime")) {
594             json.add("upTime", getUpTime());
595         }
596         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.startTime")) {
597             json.add("startTime", getStartTime());
598         }
599         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.bytesRead")) {
600             json.add("bytesRead", getBytesRead());
601         }
602         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.bytesWritten")) {
603             json.add("bytesWritten", getBytesWritten());
604         }
605         if (PropertiesUtil.getBoolean(properties, "operatingSystem.processes.handles")) {
606             json.add("handles", getOpenFiles());
607         }
608         return json.build();
609     }
