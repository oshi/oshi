19  package oshi.json.hardware;
20  
21  import oshi.json.json.OshiJsonObject;
22  
23  /**
24   * A USB device is a device connected via a USB port, possibly
25   * internally/permanently. Hubs may contain ports to which other devices connect
26   * in a recursive fashion.
27   *
28   * @author widdis[at]gmail[dot]com
29   */
30  public interface UsbDevice extends OshiJsonObject {
31      /**
32       * Name of the USB device
33       *
34       * @return The device name
35       */
36      String getName();
37  
38      /**
39       * Vendor that manufactured the USB device
40       *
41       * @return The vendor name
42       */
43      String getVendor();
44  
45      /**
46       * ID of the vendor that manufactured the USB device
47       *
48       * @return The vendor ID, a 4-digit hex string
49       */
50      String getVendorId();
51  
52      /**
53       * Product ID of the USB device
54       *
55       * @return The product ID, a 4-digit hex string
56       */
57      String getProductId();
58  
59      /**
60       * Serial number of the USB device
61       *
62       * @return The serial number, if known
63       */
64      String getSerialNumber();
65  
66      /**
67       * Other devices connected to this hub
68       *
69       * @return An array of other devices connected to this hub, if any, or an
70       *         empty array if none
71       */
72      UsbDevice[] getConnectedDevices();
73  }
