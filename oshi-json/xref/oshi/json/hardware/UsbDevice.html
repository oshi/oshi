24  package oshi.json.hardware;
25  
26  import oshi.json.json.OshiJsonObject;
27  
28  /**
29   * A USB device is a device connected via a USB port, possibly
30   * internally/permanently. Hubs may contain ports to which other devices connect
31   * in a recursive fashion.
32   *
33   * @author widdis[at]gmail[dot]com
34   */
35  public interface UsbDevice extends OshiJsonObject {
36      /**
37       * Name of the USB device
38       *
39       * @return The device name
40       */
41      String getName();
42  
43      /**
44       * Vendor that manufactured the USB device
45       *
46       * @return The vendor name
47       */
48      String getVendor();
49  
50      /**
51       * ID of the vendor that manufactured the USB device
52       *
53       * @return The vendor ID, a 4-digit hex string
54       */
55      String getVendorId();
56  
57      /**
58       * Product ID of the USB device
59       *
60       * @return The product ID, a 4-digit hex string
61       */
62      String getProductId();
63  
64      /**
65       * Serial number of the USB device
66       *
67       * @return The serial number, if known
68       */
69      String getSerialNumber();
70  
71      /**
72       * Other devices connected to this hub
73       *
74       * @return An array of other devices connected to this hub, if any, or an
75       *         empty array if none
76       */
77      UsbDevice[] getConnectedDevices();
78  }
