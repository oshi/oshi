24  package oshi.json.hardware;
25  
26  import oshi.json.json.OshiJsonObject;
27  
28  /**
29   * Memory refers to the state information of a computing system, as it is kept
30   * active in some physical structure. The term "memory" is used for the
31   * information in physical systems which are fast (ie. RAM), as a distinction
32   * from physical systems which are slow to access (ie. data storage). By design,
33   * the term "memory" refers to temporary state devices, whereas the term
34   * "storage" is reserved for permanent data.
35   *
36   * @author dblock[at]dblock[dot]org
37   */
38  public interface GlobalMemory extends OshiJsonObject {
39      /**
40       * The amount of actual physical memory, in bytes.
41       *
42       * @return Total number of bytes.
43       */
44      long getTotal();
45  
46      /**
47       * The amount of physical memory currently available, in bytes.
48       *
49       * @return Available number of bytes.
50       */
51      long getAvailable();
52  
53      /**
54       * The current size of the paging/swap file(s), in bytes. If the paging/swap
55       * file can be extended, this is a soft limit.
56       *
57       * @return Total swap in bytes.
58       */
59      long getSwapTotal();
60  
61      /**
62       * The current memory committed to the paging/swap file(s), in bytes
63       *
64       * @return Swap used in bytes.
65       */
66      long getSwapUsed();
67  
68      /**
69       * Number of pages read from paging/swap file(s) to resolve hard page
70       * faults. (Hard page faults occur when a process requires code or data that
71       * is not in its working set or elsewhere in physical memory, and must be
72       * retrieved from disk.) This property was designed as a primary indicator
73       * of the kinds of faults that cause system-wide delays. It includes pages
74       * retrieved to satisfy faults in the file system cache (usually requested
75       * by applications) and in non-cached mapped memory files.
76       *
77       * @return Pages swapped in
78       */
79      long getSwapPagesIn();
80  
81      /**
82       * Number of pages written to paging/swap file(s) to free up space in
83       * physical memory. Pages are written back to disk only if they are changed
84       * in physical memory, so they are likely to hold data, not code. A high
85       * rate of pages output might indicate a memory shortage. The operating
86       * system writes more pages back to disk to free up space when physical
87       * memory is in short supply.
88       *
89       * @return Pages swapped out
90       */
91      long getSwapPagesOut();
92  
93      /**
94       * The number of bytes in a memory page
95       * 
96       * @return Page size in bytes.
97       */
98      long getPageSize();
99  }
