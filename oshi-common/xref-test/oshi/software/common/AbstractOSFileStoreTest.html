5   package oshi.software.common;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.containsString;
9   import static org.hamcrest.Matchers.is;
10  
11  import org.junit.jupiter.api.Test;
12  
13  class AbstractOSFileStoreTest {
14  
15      private static TestOSFileStore createFileStore() {
16          return new TestOSFileStore("TestFS", "/dev/sda1", "MyLabel", "/mnt/data", "rw,noatime", "1234-5678", true,
17                  "lv0", "Local Disk", "ext4", 1000L, 900L, 5000L, 100L, 500L);
18      }
19  
20      /** Test subclass exposing protected helpers. */
21      private static class TestOSFileStore extends AbstractOSFileStore {
22          TestOSFileStore(String name, String volume, String label, String mount, String options, String uuid,
23                  boolean local, String logicalVolume, String description, String fsType, long freeSpace,
24                  long usableSpace, long totalSpace, long freeInodes, long totalInodes) {
25              super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
26                      usableSpace, totalSpace, freeInodes, totalInodes);
27          }
28  
29          @Override
30          public boolean updateAttributes() {
31              return false;
32          }
33  
34          void callUpdateFrom(oshi.software.os.OSFileStore other) {
35              updateFrom(other);
36          }
37  
38          void callUpdateSpaceAndInodes(long free, long usable, long total, long freeIn, long totalIn) {
39              updateSpaceAndInodes(free, usable, total, freeIn, totalIn);
40          }
41  
42          void callUpdateSpace(long free, long usable, long total) {
43              updateSpace(free, usable, total);
44          }
45      }
46  
47      @Test
48      void testGetters() {
49          AbstractOSFileStore fs = createFileStore();
50          assertThat(fs.getName(), is("TestFS"));
51          assertThat(fs.getVolume(), is("/dev/sda1"));
52          assertThat(fs.getLabel(), is("MyLabel"));
53          assertThat(fs.getMount(), is("/mnt/data"));
54          assertThat(fs.getOptions(), is("rw,noatime"));
55          assertThat(fs.getUUID(), is("1234-5678"));
56          assertThat(fs.isLocal(), is(true));
57          assertThat(fs.getLogicalVolume(), is("lv0"));
58          assertThat(fs.getDescription(), is("Local Disk"));
59          assertThat(fs.getType(), is("ext4"));
60          assertThat(fs.getFreeSpace(), is(1000L));
61          assertThat(fs.getUsableSpace(), is(900L));
62          assertThat(fs.getTotalSpace(), is(5000L));
63          assertThat(fs.getFreeInodes(), is(100L));
64          assertThat(fs.getTotalInodes(), is(500L));
65      }
66  
67      @Test
68      void testToString() {
69          String s = createFileStore().toString();
70          assertThat(s, containsString("name=TestFS"));
71          assertThat(s, containsString("ext4"));
72          assertThat(s, containsString("lv0"));
73      }
74  
75      @Test
76      void testUpdateSpaceAndInodes() {
77          TestOSFileStore fs = createFileStore();
78          fs.callUpdateSpaceAndInodes(2000L, 1800L, 10000L, 200L, 1000L);
79          assertThat(fs.getFreeSpace(), is(2000L));
80          assertThat(fs.getUsableSpace(), is(1800L));
81          assertThat(fs.getTotalSpace(), is(10000L));
82          assertThat(fs.getFreeInodes(), is(200L));
83          assertThat(fs.getTotalInodes(), is(1000L));
84      }
85  
86      @Test
87      void testUpdateSpace() {
88          TestOSFileStore fs = createFileStore();
89          fs.callUpdateSpace(3000L, 2500L, 20000L);
90          assertThat(fs.getFreeSpace(), is(3000L));
91          assertThat(fs.getUsableSpace(), is(2500L));
92          assertThat(fs.getTotalSpace(), is(20000L));
93          // Inodes unchanged
94          assertThat(fs.getFreeInodes(), is(100L));
95          assertThat(fs.getTotalInodes(), is(500L));
96      }
97  
98      @Test
99      void testUpdateFrom() {
100         TestOSFileStore fs = createFileStore();
101         TestOSFileStore other = new TestOSFileStore("Other", "/dev/sdb1", "OtherLabel", "/mnt/other", "ro", "9999-0000",
102                 false, "lv1", "Network Disk", "nfs", 5000L, 4500L, 50000L, 400L, 2000L);
103         fs.callUpdateFrom(other);
104         assertThat(fs.getLogicalVolume(), is("lv1"));
105         assertThat(fs.getDescription(), is("Network Disk"));
106         assertThat(fs.getType(), is("nfs"));
107         assertThat(fs.getFreeSpace(), is(5000L));
108         assertThat(fs.getUsableSpace(), is(4500L));
109         assertThat(fs.getTotalSpace(), is(50000L));
110         assertThat(fs.getFreeInodes(), is(400L));
111         assertThat(fs.getTotalInodes(), is(2000L));
112     }
113 }
