5   package oshi.software.common.os.linux;
6   
7   import static org.junit.jupiter.api.Assertions.assertEquals;
8   import static org.junit.jupiter.api.Assertions.assertTrue;
9   
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.api.condition.EnabledIfEnvironmentVariable;
12  
13  import oshi.software.os.CgroupInfo;
14  
15  /**
16   * Integration tests that run inside a Docker container with known resource limits. Enabled only when the
17   * {@code OSHI_CGROUP_TEST} environment variable is set.
18   *
19   * <p>
20   * Expected container configuration:
21   * <ul>
22   * <li>{@code --cpus=1.5} (quota=150000, period=100000)</li>
23   * <li>{@code --memory=512m}</li>
24   * <li>{@code --pids-limit=256}</li>
25   * </ul>
26   */
27  @EnabledIfEnvironmentVariable(named = "OSHI_CGROUP_TEST", matches = "true")
28  class LinuxCgroupInfoContainerTest {
29  
30      private final LinuxCgroupInfo cgroup = new LinuxCgroupInfo();
31  
32      @Test
33      void isContainerized() {
34          assertTrue(cgroup.isContainerized(), "Should detect containerized environment");
35      }
36  
37      @Test
38      void versionIsV2OrV1() {
39          int version = cgroup.getVersion();
40          assertTrue(version == 1 || version == 2, "Version should be 1 or 2 in container, got " + version);
41      }
42  
43      @Test
44      void cpuQuotaMatchesLimit() {
45          // --cpus=1.5 sets quota=150000 period=100000
46          assertEquals(150000L, cgroup.getCpuQuota(), "CPU quota should be 150000 us");
47      }
48  
49      @Test
50      void cpuPeriodIsDefault() {
51          assertEquals(CgroupInfo.DEFAULT_CPU_PERIOD, cgroup.getCpuPeriod(), "CPU period should be 100000 us");
52      }
53  
54      @Test
55      void effectiveCpus() {
56          assertEquals(1.5d, cgroup.getEffectiveCpus(), "Effective CPUs should be 1.5");
57      }
58  
59      @Test
60      void cpuUsagePositive() {
61          assertTrue(cgroup.getCpuUsage() > 0, "CPU usage should be positive in a running container");
62      }
63  
64      @Test
65      void memoryLimitMatches512MB() {
66          // --memory=512m = 536870912 bytes
67          assertEquals(536870912L, cgroup.getMemoryLimit(), "Memory limit should be 512 MiB");
68      }
69  
70      @Test
71      void memoryUsagePositive() {
72          long usage = cgroup.getMemoryUsage();
73          assertTrue(usage > 0, "Memory usage should be positive");
74          assertTrue(usage <= 536870912L, "Memory usage should not exceed limit");
75      }
76  
77      @Test
78      void pidLimitMatches() {
79          assertEquals(256L, cgroup.getPidLimit(), "PID limit should be 256");
80      }
81  
82      @Test
83      void pidCurrentPositive() {
84          long current = cgroup.getPidCurrent();
85          assertTrue(current > 0, "PID current should be positive");
86          assertTrue(current <= 256, "PID current should not exceed limit");
87      }
88  }
