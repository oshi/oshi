305     @Test
306     void testParseServicesSystemctl() {
307         // Real `systemctl list-unit-files` columns: UNIT FILE, STATE, [VENDOR PRESET]. Only ".service"/"enabled"
308         // rows count; already-running services are dropped by full name or short (last dotted segment) name.
309         List<String> systemctl = Arrays.asList("UNIT FILE                              STATE      VENDOR PRESET",
310                 "cron.service                           enabled    enabled",
311                 "ssh.service                            enabled    enabled",
312                 "dbus-org.freedesktop.resolve1.service  enabled    enabled",
313                 "bluetooth.service                      disabled   enabled", "", "247 unit files listed.");
314         Set<String> running = new HashSet<>(Arrays.asList("ssh", "resolve1"));
315         List<OSService> services = LinuxOperatingSystem.parseServices(systemctl, running, "/nonexistent-init-dir");
316         // ssh (full-name match) and resolve1 (short-name match) deduped; bluetooth disabled; only cron remains
317         assertThat(services, hasSize(1));
318         assertThat(services.get(0).getName(), is("cron"));
319         assertThat(services.get(0).getState(), is(State.STOPPED));
320         // systemctl produced a service, so the init-dir fallback (nonexistent path) is not consulted
321     }

323     @Test
324     void testParseServicesSystemctlFoundSuppressesFallback(@TempDir Path initDir) throws IOException {
325         // The only enabled unit is already running, so no stopped service is emitted -- but systemctl DID produce
326         // output, so the /etc/init fallback must not run even though a .conf job is present in the init dir
327         Files.write(initDir.resolve("tty1.conf"), "start on runlevel\n".getBytes(StandardCharsets.UTF_8));
328         List<String> systemctl = Arrays.asList("UNIT FILE          STATE      VENDOR PRESET",
329                 "ssh.service        enabled    enabled");
330         Set<String> running = new HashSet<>(Collections.singletonList("ssh"));
331         List<OSService> services = LinuxOperatingSystem.parseServices(systemctl, running, initDir.toString());
332         assertThat(services, is(empty()));
333     }

335     @Test
336     void testParseServicesInitFallback(@TempDir Path initDir) throws IOException {
337         // No systemctl services -> scan the init directory for *.conf upstart jobs
338         Files.write(initDir.resolve("tty1.conf"), "start on runlevel\n".getBytes(StandardCharsets.UTF_8));
339         Files.write(initDir.resolve("ssh.conf"), "start on runlevel\n".getBytes(StandardCharsets.UTF_8));
340         Files.write(initDir.resolve("README.txt"), "not a job\n".getBytes(StandardCharsets.UTF_8));
341         Set<String> running = new HashSet<>(Collections.singletonList("ssh"));
342         List<OSService> services = LinuxOperatingSystem.parseServices(Collections.emptyList(), running,
343                 initDir.toString());
344         // ssh.conf deduped against running, README.txt is not a .conf; only tty1 remains
345         assertThat(services, hasSize(1));
346         assertThat(services.get(0).getName(), is("tty1"));
347         assertThat(services.get(0).getState(), is(State.STOPPED));
348     }

369     @Test
370     void testGetReleaseFilenameSkipsExcludedThenReleaseFallback(@TempDir Path etc) throws IOException {
371         // os-release/lsb-release/system-release are handled elsewhere and excluded from the scan; with only those
372         // present the scan falls back to the "release" file (Solaris-style) when it exists
373         Files.write(etc.resolve("os-release"), "NAME=Ubuntu\n".getBytes(StandardCharsets.UTF_8));
374         Files.write(etc.resolve("lsb-release"), "DISTRIB_ID=Ubuntu\n".getBytes(StandardCharsets.UTF_8));
375         Files.write(etc.resolve("release"), "Solaris\n".getBytes(StandardCharsets.UTF_8));
376         assertThat(LinuxOperatingSystem.getReleaseFilename(etc.toString()), is(etc.resolve("release").toString()));
377     }
