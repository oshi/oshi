5   package oshi.software.common.os.unix.bsd;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Collections;
11  import java.util.List;
12  import java.util.Map;
13  
14  import org.junit.jupiter.api.Test;
15  
16  import oshi.software.os.OSThread;
17  
18  /**
19   * Tests the shared open-file-limit skeleton and ELF-bitness parsing hoisted into {@link BsdOSProcess}.
20   */
21  class BsdOSProcessTest {
22  
23      // Builds a minimal BSD process whose open-file-limit hooks return sentinel values so the base's routing (current
24      // process via getrlimit vs. other process via the fallback) can be observed.
25      private static BsdOSProcess stubProcess(int pid, int ownPid) {
26          return new BsdOSProcess(pid) {
27              @Override
28              protected List<BsdPsKeyword> psKeywords() {
29                  return Collections.emptyList();
30              }
31  
32              @Override
33              protected String psCommandArgs() {
34                  return "";
35              }
36  
37              @Override
38              protected List<String> queryArguments() {
39                  return Collections.emptyList();
40              }
41  
42              @Override
43              protected Map<String, String> queryEnvironmentVariables() {
44                  return Collections.emptyMap();
45              }
46  
47              @Override
48              protected int queryBitness() {
49                  return 0;
50              }
51  
52              @Override
53              protected int queryOwnProcessId() {
54                  return ownPid;
55              }
56  
57              @Override
58              protected long queryRlimitNofile(boolean soft) {
59                  return soft ? 1024L : 4096L;
60              }
61  
62              @Override
63              protected long otherProcessOpenFileLimit(int index) {
64                  return index == 1 ? -11L : -22L;
65              }
66  
67              @Override
68              public String getCurrentWorkingDirectory() {
69                  return "";
70              }
71  
72              @Override
73              public long getOpenFiles() {
74                  return 0L;
75              }
76  
77              @Override
78              public List<OSThread> getThreadDetails() {
79                  return Collections.emptyList();
80              }
81          };
82      }
83  
84      @Test
85      void testOpenFileLimitsForCurrentProcess() {
86          // pid matches the current process id, so getrlimit values are returned
87          BsdOSProcess self = stubProcess(42, 42);
88          assertThat(self.getSoftOpenFileLimit(), is(1024L));
89          assertThat(self.getHardOpenFileLimit(), is(4096L));
90      }
91  
92      @Test
93      void testOpenFileLimitsForOtherProcess() {
94          // pid differs from the current process id, so the non-current fallback is used
95          BsdOSProcess other = stubProcess(42, 7);
96          assertThat(other.getSoftOpenFileLimit(), is(-11L));
97          assertThat(other.getHardOpenFileLimit(), is(-22L));
98      }
99  
100     @Test
101     void testElfBitness() {
102         assertThat(BsdOSProcess.elfBitness("FreeBSD ELF64"), is(64));
103         assertThat(BsdOSProcess.elfBitness("FreeBSD ELF32"), is(32));
104         assertThat(BsdOSProcess.elfBitness("not an abi string"), is(0));
105     }
106 }
