7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.aMapWithSize;
9   import static org.hamcrest.Matchers.anEmptyMap;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  import java.util.Map;
16  
17  import org.junit.jupiter.api.Test;
18  
19  import oshi.util.tuples.Pair;
20  
21  class SolarisFileSystemTest {
22  
23      @Test
24      void testParseDfgInodesTypical() {
25          List<String> lines = Arrays.asList(
26                  "/                  (/dev/md/dsk/d0    ):         8192 block size          1024 frag size",
27                  "41310292 total blocks   18193814 free blocks 17780712 available        2486848 total files",
28                  " 2293351 free files     22282240 filesys id",
29                  "     ufs fstype       0x00000004 flag             255 filename length",
30                  "/usr               (/dev/md/dsk/d1    ):         8192 block size          1024 frag size",
31                  "20000000 total blocks    5000000 free blocks  4800000 available        1000000 total files",
32                  "  800000 free files     33333333 filesys id",
33                  "     ufs fstype       0x00000004 flag             255 filename length");
34          Pair<Map<String, Long>, Map<String, Long>> result = SolarisFileSystem.parseDfgInodes(lines);
35          Map<String, Long> freeMap = result.getA();
36          Map<String, Long> totalMap = result.getB();
37          assertThat(freeMap, is(aMapWithSize(2)));
38          assertThat(freeMap.get("/"), is(2293351L));
39          assertThat(freeMap.get("/usr"), is(800000L));
40          assertThat(totalMap.get("/"), is(2486848L));
41          assertThat(totalMap.get("/usr"), is(1000000L));
42      }
43  
44      @Test
45      void testParseDfgInodesEmpty() {
46          Pair<Map<String, Long>, Map<String, Long>> result = SolarisFileSystem.parseDfgInodes(Collections.emptyList());
47          assertThat(result.getA(), is(anEmptyMap()));
48          assertThat(result.getB(), is(anEmptyMap()));
49      }
50  
51      @Test
52      void testParseDfgInodesSingleMount() {
53          List<String> lines = Arrays.asList(
54                  "/export/home       (/dev/dsk/c0t0d0s7):         8192 block size          1024 frag size",
55                  "10000000 total blocks    3000000 free blocks  2900000 available         500000 total files",
56                  "  450000 free files     44444444 filesys id",
57                  "     ufs fstype       0x00000004 flag             255 filename length");
58          Pair<Map<String, Long>, Map<String, Long>> result = SolarisFileSystem.parseDfgInodes(lines);
59          Map<String, Long> freeMap = result.getA();
60          Map<String, Long> totalMap = result.getB();
61          assertThat(freeMap, is(aMapWithSize(1)));
62          assertThat(freeMap.get("/export/home"), is(450000L));
63          assertThat(totalMap.get("/export/home"), is(500000L));
64      }
65  }
