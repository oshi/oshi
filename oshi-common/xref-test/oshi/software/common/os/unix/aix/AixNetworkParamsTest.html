5   package oshi.software.common.os.unix.aix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Arrays;
11  import java.util.Collections;
12  
13  import org.junit.jupiter.api.Test;
14  
15  import oshi.util.Constants;
16  
17  class AixNetworkParamsTest {
18  
19      @Test
20      void testParseDefaultGateway() {
21          // netstat -rnf inet: the "default" route's gateway is the second column
22          String gw = AixNetworkParams.parseDefaultGateway(Arrays.asList(//
23                  "Routing tables", //
24                  "Destination      Gateway         Flags  Refs   Use  If  Exp  Groups", //
25                  "Route Tree for Protocol Family 2 (Internet):", //
26                  "default          192.168.1.1     UG      1     123  en0   -    -", //
27                  "127/8            127.0.0.1       U       3       0  lo0   -    -"));
28          assertThat(gw, is("192.168.1.1"));
29      }
30  
31      @Test
32      void testParseDefaultGatewayNoDefaultRoute() {
33          String gw = AixNetworkParams.parseDefaultGateway(Arrays.asList(//
34                  "Destination      Gateway         Flags  Refs   Use  If  Exp  Groups", //
35                  "127/8            127.0.0.1       U       3       0  lo0   -    -"));
36          assertThat(gw, is(Constants.UNKNOWN));
37      }
38  
39      @Test
40      void testParseDefaultGatewayEmpty() {
41          assertThat(AixNetworkParams.parseDefaultGateway(Collections.emptyList()), is(Constants.UNKNOWN));
42      }
43  }
