171     @Test
172     void testGetProcessID() {
173         assertThat(createProcess(42, 0, 0, 0).getProcessID(), is(42));
174     }
175 
176     @Test
177     void testCpuLoadCumulativeZeroUptime() {
178         assertThat(createProcess(1, 100, 200, 0).getProcessCpuLoadCumulative(), is(0d));
179     }
180 
181     @Test
182     void testCpuLoadCumulative() {
183         // (300 + 700) / 2000 = 0.5
184         assertThat(createProcess(1, 300, 700, 2000).getProcessCpuLoadCumulative(), is(closeTo(0.5, 0.001)));
185     }
186 
187     @Test
188     void testCpuLoadBetweenTicksWithPrior() {
189         AbstractOSProcess prior = createProcess(1, 100, 100, 1000);
190         AbstractOSProcess current = createProcess(1, 200, 300, 2000);
191         // (200-100 + 300-100) / (2000-1000) = 0.3
192         assertThat(current.getProcessCpuLoadBetweenTicks(prior), is(closeTo(0.3, 0.001)));
193     }
194 
195     @Test
196     void testCpuLoadBetweenTicksNullPriorFallsToCumulative() {
197         AbstractOSProcess current = createProcess(1, 300, 700, 2000);
198         assertThat(current.getProcessCpuLoadBetweenTicks(null), is(closeTo(0.5, 0.001)));
199     }

201     @Test
202     void testCpuLoadBetweenTicksDifferentPidFallsToCumulative() {
203         AbstractOSProcess prior = createProcess(99, 100, 100, 1000);
204         AbstractOSProcess current = createProcess(1, 300, 700, 2000);
205         assertThat(current.getProcessCpuLoadBetweenTicks(prior), is(closeTo(0.5, 0.001)));
206     }

208     @Test

212         assertThat(current.getProcessCpuLoadBetweenTicks(prior), is(closeTo(0.5, 0.001)));
