5   package oshi.software.os;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.containsString;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.not;
11  
12  import java.util.LinkedHashMap;
13  import java.util.Map;
14  
15  import org.junit.jupiter.api.Test;
16  
17  class ApplicationInfoTest {
18  
19      @Test
20      void testGetters() {
21          Map<String, String> extra = new LinkedHashMap<>();
22          extra.put("location", "/Applications/App.app");
23          ApplicationInfo app = new ApplicationInfo("MyApp", "2.0", "Acme", 1_000_000L, extra);
24          assertThat(app.getName(), is("MyApp"));
25          assertThat(app.getVersion(), is("2.0"));
26          assertThat(app.getVendor(), is("Acme"));
27          assertThat(app.getTimestamp(), is(1_000_000L));
28          assertThat(app.getAdditionalInfo().get("location"), is("/Applications/App.app"));
29      }
30  
31      @Test
32      void testNullAdditionalInfoBecomesEmptyMap() {
33          ApplicationInfo app = new ApplicationInfo("App", "1.0", "Vendor", 0L, null);
34          assertThat(app.getAdditionalInfo().isEmpty(), is(true));
35      }
36  
37      @Test
38      void testEqualsAndHashCode() {
39          ApplicationInfo a = new ApplicationInfo("App", "1.0", "Vendor", 0L, null);
40          ApplicationInfo b = new ApplicationInfo("App", "1.0", "Vendor", 0L, null);
41          ApplicationInfo c = new ApplicationInfo("Other", "1.0", "Vendor", 0L, null);
42          assertThat(a, is(b));
43          assertThat(a.hashCode(), is(b.hashCode()));
44          assertThat(a, is(not(c)));
45          assertThat(a, is(a));
46          assertThat(a, is(not((Object) null)));
47      }
48  
49      @Test
50      void testToString() {
51          ApplicationInfo app = new ApplicationInfo("MyApp", "2.0", "Acme", 0L, null);
52          assertThat(app.toString(), containsString("MyApp"));
53          assertThat(app.toString(), containsString("Acme"));
54      }
55  }
