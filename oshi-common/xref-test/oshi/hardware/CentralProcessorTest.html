24      @Test
25      void testLogicalProcessorThreeArg() {
26          LogicalProcessor lp = new LogicalProcessor(2, 1, 0);
27          assertThat(lp.getProcessorNumber(), is(2));
28          assertThat(lp.getPhysicalProcessorNumber(), is(1));
29          assertThat(lp.getPhysicalPackageNumber(), is(0));
30          assertThat(lp.getNumaNode(), is(0));
31          assertThat(lp.getProcessorGroup(), is(0));
32          assertThat(lp.toString(), containsString("2"));
33      }
34

40      }

48      @Test

62          assertThat(pp.getIdString(), is("ARM Cortex-A78"));

65      @Test
66      void testProcessorCacheGetters() {
67          ProcessorCache cache = new ProcessorCache(2, 8, 64, 262144, Type.UNIFIED);
68          assertThat(cache.getLevel(), is((byte) 2));
69          assertThat(cache.getAssociativity(), is((byte) 8));
70          assertThat(cache.getLineSize(), is((short) 64));
71          assertThat(cache.getCacheSize(), is(262144));
72          assertThat(cache.getType(), is(Type.UNIFIED));
73          assertThat(cache.toString(), containsString("L2"));
74      }
75  
76      @Test
77      void testProcessorCacheEqualsAndHashCode() {
78          ProcessorCache a = new ProcessorCache(1, 4, 64, 32768, Type.DATA);
79          ProcessorCache b = new ProcessorCache(1, 4, 64, 32768, Type.DATA);
80          ProcessorCache c = new ProcessorCache(1, 4, 64, 32768, Type.INSTRUCTION);
81          assertThat(a, is(b));
82          assertThat(a.hashCode(), is(b.hashCode()));
83          assertThat(a, is(not(c)));
84          assertThat(a, is(not((Object) null)));
85          assertThat(a, is(a));
86      }
87  
88      @Test
89      void testProcessorCacheTypeToString() {
90          assertThat(Type.UNIFIED.toString(), is("Unified"));
91          assertThat(Type.INSTRUCTION.toString(), is("Instruction"));
92          assertThat(Type.DATA.toString(), is("Data"));
93          assertThat(Type.TRACE.toString(), is("Trace"));
94      }
95  
96      @Test
97      void testProcessorIdentifier() {
98          ProcessorIdentifier id = new ProcessorIdentifier("GenuineIntel", "Intel(R) Core(TM) i7 @ 3.60GHz", "6", "158",
99                  "10", "BFEBFBFF000906EA", true);
100         assertThat(id.getVendor(), is("GenuineIntel"));
101         assertThat(id.getName(), containsString("Intel"));
102         assertThat(id.getFamily(), is("6"));
103         assertThat(id.getModel(), is("158"));
104         assertThat(id.getStepping(), is("10"));
105         assertThat(id.getProcessorID(), is("BFEBFBFF000906EA"));
106         assertThat(id.isCpu64bit(), is(true));
107         assertThat(id.getVendorFreq(), is(3_600_000_000L));
108         assertThat(id.getIdentifier(), containsString("Intel64"));
109         assertThat(id.getMicroarchitecture(), is(notNullValue()));
110         assertThat(id.toString(), containsString("Intel64"));
111     }
