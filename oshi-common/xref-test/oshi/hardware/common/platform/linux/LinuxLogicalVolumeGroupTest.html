22  // parsePhysicalVolumes dereferences DevPath.DEV, whose static initializer validates that the configured /dev path
23  // exists. That holds on Linux, macOS, and the BSD/illumos CI hosts, but not on Windows, where class init throws.
24  @DisabledOnOs(OS.WINDOWS)
25  class LinuxLogicalVolumeGroupTest {
26  
27      @Test
28      void testParsePhysicalVolumes() {
29          // pvs -o vg_name,pv_name output: header row, then "vg pv" rows (leading whitespace)
30          List<String> pvs = Arrays.asList("  VG   PV", "  vg0  /dev/sda2", "  vg0  /dev/sdb1", "  vg1  /dev/sdc1");
31          Map<String, Set<String>> map = LinuxLogicalVolumeGroup.parsePhysicalVolumes(pvs);
32          assertThat(map.keySet(), containsInAnyOrder("vg0", "vg1"));
33          assertThat(map.get("vg0"), containsInAnyOrder("/dev/sda2", "/dev/sdb1"));
34          assertThat(map.get("vg1"), contains("/dev/sdc1"));
35      }
36  
37      @Test
38      void testParsePhysicalVolumesSkipsNonDeviceAndMalformed() {
39          // A single-token line (length != 2) and a second token that is not a /dev path are both ignored
40          List<String> pvs = Arrays.asList("  novg", "  vg2  notadevice", "  vg3  /dev/mapper/pv");
41          Map<String, Set<String>> map = LinuxLogicalVolumeGroup.parsePhysicalVolumes(pvs);
42          assertThat(map.keySet(), contains("vg3"));
43          assertThat(map.get("vg3"), contains("/dev/mapper/pv"));
44      }
45  
46      @Test
47      void testParsePhysicalVolumesEmpty() {
48          assertThat(LinuxLogicalVolumeGroup.parsePhysicalVolumes(Collections.emptyList()), anEmptyMap());
49      }
50  }
