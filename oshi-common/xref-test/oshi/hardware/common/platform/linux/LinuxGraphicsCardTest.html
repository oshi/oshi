32      @Test
33      void testConstructorAndGetters() {
34          StubGraphicsCard card = new StubGraphicsCard("RTX 4090", "0x2684", "NVIDIA", "Rev: 01", 24576L,
35                  "/sys/class/drm/card0/device", "nvidia", "0000:01:00.0");
36          assertThat(card.getName(), is("RTX 4090"));
37          assertThat(card.getDeviceId(), is("0x2684"));
38          assertThat(card.getVendor(), is("NVIDIA"));
39          assertThat(card.getVersionInfo(), is("Rev: 01"));
40          assertThat(card.getVRam(), is(24576L));
41          assertThat(card.getDrmDevicePath(), is("/sys/class/drm/card0/device"));
42          assertThat(card.getDriverName(), is("nvidia"));
43          assertThat(card.getPciBusId(), is("0000:01:00.0"));
44      }

126     @Test
127     void testReadUeventValue(@TempDir Path tempDir) throws IOException {
128         Path uevent = tempDir.resolve("uevent");
129         String content = "DRIVER=amdgpu\nPCI_SLOT_NAME=0000:01:00.0\nPCI_ID=1002:744C\n";
130         Files.write(uevent, content.getBytes(StandardCharsets.UTF_8));
131 
132         assertThat(LinuxGraphicsCard.readUeventValue(uevent.toString(), "PCI_SLOT_NAME"), is("0000:01:00.0"));
133         assertThat(LinuxGraphicsCard.readUeventValue(uevent.toString(), "DRIVER"), is("amdgpu"));
134         assertThat(LinuxGraphicsCard.readUeventValue(uevent.toString(), "MISSING_KEY"), is(""));
135     }

158     /**
159      * Creates a card directory with a driver symlink and uevent file.
160      *
161      * @param drmDir     the base DRM directory
162      * @param cardName   e.g. "card0"
163      * @param driverName e.g. "amdgpu"
164      * @param slotName   e.g. "0000:01:00.0"
165      * @throws IOException if file creation fails
166      */
167     private static void createCardWithDriver(Path drmDir, String cardName, String driverName, String slotName)
168             throws IOException {
169         Path deviceDir = drmDir.resolve(cardName + "/device");
170         Files.createDirectories(deviceDir);
171         // Create a fake driver target and symlink
172         Path driverTarget = drmDir.resolve("drivers/" + driverName);
173         Files.createDirectories(driverTarget);
174         Files.createSymbolicLink(deviceDir.resolve("driver"), driverTarget);
175         // Write uevent with PCI_SLOT_NAME
176         Files.write(deviceDir.resolve("uevent"), ("PCI_SLOT_NAME=" + slotName + "\n").getBytes(StandardCharsets.UTF_8));
177     }
