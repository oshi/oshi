24      @Test
25      void testParseProcessorId() {
26          // prtconf excerpt
27          Quartet<String, String, String, Boolean> id = AixCentralProcessor.parseProcessorId(Arrays.asList(//
28                  "System Model: IBM,9114-275", //
29                  "Processor Type: PowerPC_POWER7", //
30                  "Processor Version: PV_7_Compat", //
31                  "Number Of Processors: 8", //
32                  "CPU Type: 64-bit"));
33          assertThat(id.getA(), is("IBM"));
34          assertThat(id.getB(), is("PowerPC_POWER7"));
35          assertThat(id.getC(), is("PV_7_Compat"));
36          assertThat(id.getD(), is(true));
37      }

39      @Test
40      void testParseProcessorIdEmpty() {
41          Quartet<String, String, String, Boolean> id = AixCentralProcessor.parseProcessorId(Collections.emptyList());
42          assertThat(id.getA(), is(Constants.UNKNOWN));
43          assertThat(id.getB(), is(""));
44          assertThat(id.getC(), is(""));
45          assertThat(id.getD(), is(false));
46      }

48      @Test
49      void testParseProcessorIdEmptyMarkerValues() {
50          // Bare marker lines (no value after the colon) must not throw; they yield empty fields
51          Quartet<String, String, String, Boolean> id = AixCentralProcessor
52                  .parseProcessorId(Arrays.asList("Processor Type:", "Processor Version:", "CPU Type:"));
53          assertThat(id.getA(), is(Constants.UNKNOWN));
54          assertThat(id.getB(), is(""));
55          assertThat(id.getC(), is(""));
56          assertThat(id.getD(), is(false));
57      }

59      @Test
60      void testParsePowerVersion() {
61          // Real prtconf excerpt from a POWER8 box (IBM,8284-22A): the Processor Version line carries the generation
62          List<String> prtconf = Arrays.asList(//
63                  "System Model: IBM,8284-22A", //
64                  "Processor Type: PowerPC_POWER8", //
65                  "Processor Implementation Mode: POWER 8", //
66                  "Processor Version: PV_8_Compat", //
67                  "Number Of Processors: 10", //
68                  "CPU Type: 64-bit");
69          assertThat(AixCentralProcessor.parsePowerVersion(prtconf), is(8));
70          // POWER9 form
71          assertThat(AixCentralProcessor.parsePowerVersion(Collections.singletonList("Processor Version: PV_9_Compat")),
72                  is(9));
73          // No Processor Version line -> 0 (selects no caches, same as an unknown generation)
74          assertThat(AixCentralProcessor.parsePowerVersion(Collections.emptyList()), is(0));
75      }
76  
77      @Test
78      void testParseCurrentFreq() {
79          List<String> pmcycles = Arrays.asList("Cpu 0 runs at 3000 MHz", "Cpu 1 runs at 3000 MHz");
80          // exactly two CPUs
81          assertThat(AixCentralProcessor.parseCurrentFreq(pmcycles, 2),
82                  is(new long[] { 3_000_000_000L, 3_000_000_000L }));
83          // more logical processors than output lines: trailing entries stay -1
84          assertThat(AixCentralProcessor.parseCurrentFreq(pmcycles, 4),
85                  is(new long[] { 3_000_000_000L, 3_000_000_000L, -1L, -1L }));
86          // no output: all -1
87          assertThat(AixCentralProcessor.parseCurrentFreq(Collections.emptyList(), 2), is(new long[] { -1L, -1L }));
88      }
89  
90      @Test
91      void testParseSbits() {
92          assertThat(AixCentralProcessor.parseSbits(Collections.singletonList("#define SBITS 16")), is(16));
93          assertThat(AixCentralProcessor.parseSbits(Collections.singletonList("#define  SBITS  10")), is(10));
94          // default when not present
95          assertThat(AixCentralProcessor.parseSbits(Collections.singletonList("#define OTHER 5")), is(16));
96          assertThat(AixCentralProcessor.parseSbits(Collections.emptyList()), is(16));
97      }
98  
99      @Test
100     void testCachesForPowerVersion() {
101         assertThat(AixCentralProcessor.cachesForPowerVersion(7, 4), hasSize(4));
102         assertThat(AixCentralProcessor.cachesForPowerVersion(8, 4), hasSize(5));
103         // POWER9 sizes its shared L3 by core count: (cores * 10) MiB
104         List<ProcessorCache> p9 = AixCentralProcessor.cachesForPowerVersion(9, 4);
105         assertThat(p9, hasSize(4));
106         ProcessorCache l3 = p9.stream().filter(c -> c.getLevel() == 3).findFirst().orElseThrow(AssertionError::new);
107         assertThat(l3.getCacheSize(), is((4 * 10) << 20));
108         // unknown version yields no caches
109         assertThat(AixCentralProcessor.cachesForPowerVersion(99, 4), is(empty()));
110     }
