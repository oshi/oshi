27      // Representative OpenBSD/NetBSD dmesg topology lines
28      private static final List<String> DMESG_TOPOLOGY = Arrays.asList(//
29              "cpu0: smt 0, core 0, package 0", //
30              "cpu1: smt 0, core 1, package 0", //
31              "cpu2: smt 0, core 0, package 1", //
32              "cpu3: smt 0, core 1, package 1");
33  
34      @Test
35      void testParseTopology() {
36          Pair<Map<Integer, Integer>, Map<Integer, Integer>> result = BsdCentralProcessor.parseTopology(DMESG_TOPOLOGY);
37          Map<Integer, Integer> coreMap = result.getA();
38          Map<Integer, Integer> packageMap = result.getB();
39          assertThat(coreMap.size(), is(4));
40          assertThat(coreMap.get(0), is(0));
41          assertThat(coreMap.get(1), is(1));
42          assertThat(coreMap.get(2), is(0));
43          assertThat(coreMap.get(3), is(1));
44          assertThat(packageMap.get(0), is(0));
45          assertThat(packageMap.get(2), is(1));
46      }

56      @Test
57      void testParseDmesgModelsAndCachesIntel() {
58          // Representative x86 dmesg — cache entries are split across separate lines but each line ends with "cache"
59          List<String> dmesg = Arrays.asList(//
60                  "cpu0: Intel(R) Celeron(R) N4000 CPU @ 1.10GHz, 2491.67 MHz, 06-7a-01", //
61                  "cpu0: 32KB 64b/line 8-way D-cache, 32KB 64b/line 8-way I-cache", //
62                  "cpu0: 4MB 64b/line 16-way L2 cache", //
63                  "cpu0: FPU,VME,DE,PSE,TSC,MSR,PAE,MCE");
64          DmesgStrings result = BsdCentralProcessor.parseDmesgModelsAndCaches(dmesg);
65          assertThat(result.getCpuMap().get(0), is("Intel(R) Celeron(R) N4000 CPU @ 1.10GHz, 2491.67 MHz, 06-7a-01"));
66          assertThat(result.getCaches(), is(not(empty())));
67          // D-cache, I-cache (from first cache line), L2 cache (from second)
68          assertThat(result.getCaches().size(), is(3));
69      }

71      @Test
72      void testParseDmesgModelsAndCachesArm() {
73          // Representative ARM big.LITTLE dmesg
74          List<String> dmesg = Arrays.asList(//
75                  "cpu0 at mainbus0 mpidr 0: ARM Cortex-A53 r0p4", //
76                  "cpu0: 32KB 64b/line 2-way L1 VIPT I-cache, 32KB 64b/line 4-way L1 D-cache", //
77                  "cpu0: 512KB 64b/line 16-way L2 cache", //
78                  "cpu4 at mainbus0 mpidr 100: ARM Cortex-A72 r0p2", //
79                  "cpu4: 48KB 64b/line 3-way L1 PIPT I-cache, 32KB 64b/line 2-way L1 D-cache", //
80                  "cpu4: 1024KB 64b/line 16-way L2 cache");
81          DmesgStrings result = BsdCentralProcessor.parseDmesgModelsAndCaches(dmesg);
82          assertThat(result.getCpuMap().get(0), is("ARM Cortex-A53 r0p4"));
83          assertThat(result.getCpuMap().get(4), is("ARM Cortex-A72 r0p2"));
84          // Should have L1 I-cache, L1 D-cache, L2 from each CPU type (unique by cache params)
85          assertThat(result.getCaches().size(), is(6));
86      }

88      @Test
89      void testParseDmesgModelsAndCachesFeatureFlags() {
90          // Feature flags are lines starting with "cpu" having ": " with 4+ comma-delimited items
91          List<String> dmesg = Arrays.asList(//
92                  "cpu0: FPU,VME,DE,PSE,TSC,MSR,PAE,MCE", //
93                  "cpu0: SSE3,PCLMULQDQ,DTES64,MONITOR");
94          DmesgStrings result = BsdCentralProcessor.parseDmesgModelsAndCaches(dmesg);
95          assertThat(result.getFeatureFlags().size(), is(2));
96      }

106     @Test
107     void testParseCacheStrDCache() {
108         ProcessorCache cache = BsdCentralProcessor.parseCacheStr("32KB 64b/line 8-way D-cache");
109         assertThat(cache, is(not(nullValue())));
110         assertThat(cache.getType(), is(Type.DATA));
111         assertThat(cache.getLevel(), is((byte) 1));
112         assertThat(cache.getAssociativity(), is((byte) 8));
113         assertThat(cache.getLineSize(), is((short) 64));
114         assertThat(cache.getCacheSize(), is(32768));
115     }
