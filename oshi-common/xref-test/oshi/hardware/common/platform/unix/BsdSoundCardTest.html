5   package oshi.hardware.common.platform.unix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.SoundCard;
19  
20  class BsdSoundCardTest {
21  
22      // Representative NetBSD/OpenBSD dmesg output with audio devices
23      private static final List<String> DMESG = Arrays.asList(//
24              "azalia0 at pci0 dev 27 function 0 \"Intel 82801I HD Audio\" rev 0x03: msi", //
25              "azalia0: codec[0]: Realtek ALC888", //
26              "audio0 at azalia0", //
27              "hdaudio0 at pci0 dev 3 function 0 \"Intel HD Graphics Audio\" rev 0x05: msi", //
28              "hdaudio0: codec[0]: Intel Haswell HDMI", //
29              "audio1 at hdaudio0");
30  
31      @Test
32      void testParseDmesg() {
33          List<SoundCard> cards = BsdSoundCard.parseDmesg(DMESG);
34          assertThat(cards, hasSize(2));
35          // Find the azalia card
36          SoundCard azalia = cards.stream().filter(c -> c.getName().contains("82801I")).findFirst().orElse(null);
37          assertThat(azalia != null, is(true));
38          assertThat(azalia.getName(), is("Intel 82801I HD Audio"));
39          // Parser uses indexOf(':') to split — captures everything after the first colon in the codec line
40          assertThat(azalia.getCodec(), is("codec[0]: Realtek ALC888"));
41          assertThat(azalia.getDriverVersion(), is("rev 0x03"));
42      }
43  
44      @Test
45      void testParseDmesgNoAudio() {
46          List<String> dmesg = Arrays.asList(//
47                  "azalia0 at pci0 dev 27 function 0 \"Intel Audio\" rev 0x03: msi", //
48                  "azalia0: codec[0]: Realtek ALC888");
49          // No "audio0 at azalia0" line means azalia0 is not in the names set
50          List<SoundCard> cards = BsdSoundCard.parseDmesg(dmesg);
51          assertThat(cards, is(empty()));
52      }
53  
54      @Test
55      void testParseDmesgEmpty() {
56          List<SoundCard> cards = BsdSoundCard.parseDmesg(Collections.emptyList());
57          assertThat(cards, is(empty()));
58      }
59  
60      @Test
61      void testParseDmesgNoCodec() {
62          // Audio device is present but no codec line follows the PCI match
63          List<String> dmesg = Arrays.asList(//
64                  "audio0 at azalia0", //
65                  "azalia0 at pci0 dev 27 function 0 \"Intel Audio\" rev 0x01: msi", //
66                  "unrelated line here");
67          List<SoundCard> cards = BsdSoundCard.parseDmesg(dmesg);
68          assertThat(cards, hasSize(1));
69          assertThat(cards.get(0).getName(), is("Intel Audio"));
70          // No codec line found, map.get returns null
71          assertThat(cards.get(0).getCodec(), is((String) null));
72      }
73  }
