7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.emptyString;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.nullValue;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.util.tuples.Triplet;
19  
20  class FreeBsdFirmwareTest {
21  
22      @Test
23      void testParseDmiDecode() {
24          // Representative `dmidecode -t bios` (DMI type 0) block; MM/DD/YYYY date is normalized to ISO.
25          List<String> dmidecode = Arrays.asList(//
26                  "Handle 0x0000, DMI type 0, 24 bytes", //
27                  "BIOS Information", //
28                  "\tVendor: Parallels Software International Inc.", //
29                  "\tVersion: 11.2.1 (32626)", //
30                  "\tRelease Date: 07/15/2016", //
31                  "\tBIOS Revision: 11.2", //
32                  "\tFirmware Revision: 11.2");
33          Triplet<String, String, String> fw = FreeBsdFirmware.parseDmiDecode(dmidecode);
34          assertThat(fw.getA(), is("Parallels Software International Inc."));
35          assertThat(fw.getB(), is("11.2.1 (32626)"));
36          assertThat(fw.getC(), is("2016-07-15"));
37      }
38  
39      @Test
40      void testParseDmiDecodeEmpty() {
41          // No output (e.g. dmidecode not available / not root): the parser returns raw absent values (null
42          // manufacturer/version, empty date); readDmiDecode applies the Constants.UNKNOWN fallbacks.
43          Triplet<String, String, String> fw = FreeBsdFirmware.parseDmiDecode(Collections.emptyList());
44          assertThat(fw.getA(), is(nullValue()));
45          assertThat(fw.getB(), is(nullValue()));
46          assertThat(fw.getC(), is(emptyString()));
47      }
48  }
