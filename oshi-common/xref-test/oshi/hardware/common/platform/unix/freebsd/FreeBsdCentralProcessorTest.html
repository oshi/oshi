24  class FreeBsdCentralProcessorTest {
25  
26      // Representative dmesg.boot header (indented Origin/Features lines, as FreeBSD writes them)
27      private static final List<String> DMESG_ID = Arrays.asList(
28              "CPU: Intel(R) Xeon(R) CPU E5-2683 v3 @ 2.00GHz (2000.05-MHz K8-class CPU)",
29              "  Origin=\"GenuineIntel\"  Id=0x306f2  Family=0x6  Model=0x3f  Stepping=2",
30              "  Features=0xbfebfbff<FPU,VME,DE,PSE,TSC,MSR,PAE,MCE,CX8,APIC>",
31              "  Features2=0x7ffefbff<SSE3,PCLMULQDQ,DTES64>");
32  
33      @Test
34      void testParseProcessorIdFromDmesg() {
35          DmesgProcessorId id = FreeBsdCentralProcessor.parseProcessorIdFromDmesg(DMESG_ID, "");
36          assertThat(id.vendor(), is("GenuineIntel"));
37          assertThat(id.name(), is("Intel(R) Xeon(R) CPU E5-2683 v3 @ 2.00GHz (2000.05-MHz K8-class CPU)"));
38          assertThat(id.family(), is("6"));
39          assertThat(id.model(), is("63"));
40          assertThat(id.stepping(), is("2"));
41          // low 32 bits from Id, high 32 bits from the first Features field
42          assertThat(id.processorIdBits(), is(0xbfebfbff000306f2L));
43      }
44  
45      @Test
46      void testParseProcessorIdFromDmesgPrefersSysctlName() {
47          // A non-empty initial name (from sysctl hw.model) is kept over the CPU: line
48          DmesgProcessorId id = FreeBsdCentralProcessor.parseProcessorIdFromDmesg(DMESG_ID, "Preset Model");
49          assertThat(id.name(), is("Preset Model"));
50          assertThat(id.vendor(), is("GenuineIntel"));
51      }
52  
53      @Test
54      void testParseProcessorIdFromDmesgEmpty() {
55          DmesgProcessorId id = FreeBsdCentralProcessor.parseProcessorIdFromDmesg(Collections.emptyList(), "");
56          assertThat(id.vendor(), is(""));
57          assertThat(id.name(), is(""));
58          assertThat(id.processorIdBits(), is(0L));
59      }
60  
61      @Test
62      void testParseProcModelsAndFlags() {
63          List<String> dmesg = Arrays.asList("cpu0: <ACPI CPU> on acpi0", "cpu1: <ACPI CPU> on acpi0",
64                  "  Origin=\"GenuineIntel\"  Id=0x306f2", "  Features=0xbfebfbff<FPU,VME>",
65                  "  Features2=0x7ffefbff<SSE3>", "real memory  = 8589934592");
66          Pair<Map<Integer, String>, List<String>> r = FreeBsdCentralProcessor.parseProcModelsAndFlags(dmesg);
67          assertThat(r.getA().get(0), is("<ACPI CPU>"));
68          assertThat(r.getA().get(1), is("<ACPI CPU>"));
69          assertThat(r.getB(), contains("Features=0xbfebfbff<FPU,VME>", "Features2=0x7ffefbff<SSE3>"));
70      }
71  
72      @Test
73      void testParseProcModelsAndFlagsArmHybrid() {
74          List<String> dmesg = Arrays.asList("CPU 0: ARM Cortex-A53 r0p4 affinity: 0 0",
75                  "CPU 1: ARM Cortex-A72 r0p3 affinity: 0 1");
76          Pair<Map<Integer, String>, List<String>> r = FreeBsdCentralProcessor.parseProcModelsAndFlags(dmesg);
77          assertThat(r.getA().get(0), is("ARM Cortex-A53 r0p4"));
78          assertThat(r.getA().get(1), is("ARM Cortex-A72 r0p3"));
79          assertThat(r.getB(), is(empty()));
80      }
81  
82      @Test
83      void testParseCachesFromLscpu() {
84          List<String> lscpu = Arrays.asList("L1d cache:                       32K",
85                  "L1i cache:                       32K", "L2 cache:                        256K",
86                  "L3 cache:                        8M");
87          List<ProcessorCache> caches = FreeBsdCentralProcessor.parseCachesFromLscpu(lscpu);
88          assertThat(caches, hasSize(4));
89          assertThat(caches.stream().anyMatch(c -> c.getLevel() == 3), is(true));
90          assertThat(caches.stream().anyMatch(c -> c.getLevel() == 1 && c.getType() == ProcessorCache.Type.DATA),
91                  is(true));
92      }
93  
94      @Test
95      void testParseCachesFromLscpuEmpty() {
96          assertThat(FreeBsdCentralProcessor.parseCachesFromLscpu(Collections.emptyList()), is(empty()));
97      }
98  }
