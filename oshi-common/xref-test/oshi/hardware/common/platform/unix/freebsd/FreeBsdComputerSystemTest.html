5   package oshi.hardware.common.platform.unix.freebsd;

7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.hamcrest.Matchers.nullValue;
10  
11  import java.util.Arrays;
12  import java.util.Collections;
13  import java.util.List;
14  
15  import org.junit.jupiter.api.Test;
16  
17  import oshi.util.tuples.Quintet;
18  
19  class FreeBsdComputerSystemTest {
20  
21      @Test
22      void testParseDmiDecode() {
23          // Representative `dmidecode -t system` (DMI type 1) block.
24          List<String> dmidecode = Arrays.asList(//
25                  "Handle 0x0001, DMI type 1, 27 bytes", //
26                  "System Information", //
27                  "\tManufacturer: Dell Inc.", //
28                  "\tProduct Name: PowerEdge R640", //
29                  "\tVersion: Not Specified", //
30                  "\tSerial Number: 7XY1234", //
31                  "\tUUID: 4c4c4544-0058-5910-8034-c4c04f313233", //
32                  "\tWake-up Type: Power Switch", //
33                  "\tSKU Number: SKU=NotProvided", //
34                  "\tFamily: PowerEdge");
35          Quintet<String, String, String, String, String> dmi = FreeBsdComputerSystem.parseDmiDecode(dmidecode);
36          assertThat(dmi.getA(), is("Dell Inc."));
37          assertThat(dmi.getB(), is("PowerEdge R640"));
38          assertThat(dmi.getC(), is("7XY1234"));
39          assertThat(dmi.getD(), is("4c4c4544-0058-5910-8034-c4c04f313233"));
40          assertThat(dmi.getE(), is("Not Specified"));
41      }
42  
43      @Test
44      void testParseDmiDecodeEmpty() {
45          // No output (e.g. dmidecode not available / not root): every field is null so the caller can fall back.
46          Quintet<String, String, String, String, String> dmi = FreeBsdComputerSystem
47                  .parseDmiDecode(Collections.emptyList());
48          assertThat(dmi.getA(), is(nullValue()));
49          assertThat(dmi.getC(), is(nullValue()));
50          assertThat(dmi.getD(), is(nullValue()));
51      }
52  }
