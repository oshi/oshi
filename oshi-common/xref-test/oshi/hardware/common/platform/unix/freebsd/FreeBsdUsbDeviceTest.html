7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;

12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;

16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.UsbDevice;
19  
20  class FreeBsdUsbDeviceTest {
21  
22      // A minimal but real-shaped lshal tree: a host controller, its usbus, and one device beneath the usbus.
23      // getUsbDevices() promotes the usbus's children onto the controller and drops the usbus node itself.
24      private static final List<String> LSHAL = Arrays.asList(//
25              "udi = '/org/freedesktop/Hal/devices/pci_1022_43d5'", //
26              "  info.product = 'xHCI Host Controller'  (string)", //
27              "udi = '/org/freedesktop/Hal/devices/usb_device_0_0_usbus0'", //
28              "  freebsd.driver = 'usbus'  (string)", //
29              "  info.parent = '/org/freedesktop/Hal/devices/pci_1022_43d5'  (string)", //
30              "udi = '/org/freedesktop/Hal/devices/usb_device_781_5567'", //
31              "  info.parent = '/org/freedesktop/Hal/devices/usb_device_0_0_usbus0'  (string)", //
32              "  usb_device.product = 'Ultra USB 3.0'  (string)", //
33              "  usb_device.vendor = 'SanDisk'  (string)", //
34              "  usb_device.vendor_id = 1921  (0x781)  (int)", //
35              "  usb_device.product_id = 21863  (0x5567)  (int)", //
36              "  usb_device.serial = '4C530001120510117433'  (string)");
37  
38      @Test
39      void testParseUsbDevices() {
40          List<UsbDevice> controllers = FreeBsdUsbDevice.parseUsbDevices(LSHAL);
41          assertThat(controllers, hasSize(1));
42  
43          UsbDevice controller = controllers.get(0);
44          assertThat(controller.getName(), is("xHCI Host Controller"));
45          assertThat(controller.getUniqueDeviceId(), is("/org/freedesktop/Hal/devices/pci_1022_43d5"));
46          // The usbus node is skipped; its child is promoted directly onto the controller.
47          assertThat(controller.getConnectedDevices(), hasSize(1));
48  
49          UsbDevice device = controller.getConnectedDevices().get(0);
50          assertThat(device.getName(), is("Ultra USB 3.0"));
51          assertThat(device.getVendor(), is("SanDisk"));
52          assertThat(device.getVendorId(), is("0781"));
53          assertThat(device.getProductId(), is("5567"));
54          assertThat(device.getSerialNumber(), is("4C530001120510117433"));
55      }
56  
57      @Test
58      void testParseUsbDevicesHexSerial() {
59          // A 0x-prefixed serial is decoded from its hex byte string.
60          List<String> lshal = Arrays.asList(//
61                  "udi = '/org/freedesktop/Hal/devices/pci_0'", //
62                  "  info.product = 'Root Hub'  (string)", //
63                  "udi = '/org/freedesktop/Hal/devices/usbus0'", //
64                  "  freebsd.driver = 'usbus'  (string)", //
65                  "  info.parent = '/org/freedesktop/Hal/devices/pci_0'  (string)", //
66                  "udi = '/org/freedesktop/Hal/devices/dev0'", //
67                  "  info.parent = '/org/freedesktop/Hal/devices/usbus0'  (string)", //
68                  "  usb_device.serial = '0x41424344'  (string)");
69          List<UsbDevice> controllers = FreeBsdUsbDevice.parseUsbDevices(lshal);
70          assertThat(controllers, hasSize(1));
71          assertThat(controllers.get(0).getConnectedDevices(), hasSize(1));
72          // 0x41424344 -> "ABCD"
73          assertThat(controllers.get(0).getConnectedDevices().get(0).getSerialNumber(), is("ABCD"));
74      }
75  
76      @Test
77      void testParseUsbDevicesSkipsInterfaceNodes() {
78          // An interface node's udi is "<device-udi>_ifN"; its .parent points at the device, so key.replace(parent)
79          // begins with "_if" and the node must be skipped rather than treated as a child device.
80          List<String> lshal = Arrays.asList(//
81                  "udi = '/dev/pci0'", //
82                  "  info.product = 'Root Hub'  (string)", //
83                  "udi = '/dev/usbus0'", //
84                  "  freebsd.driver = 'usbus'  (string)", //
85                  "  info.parent = '/dev/pci0'  (string)", //
86                  "udi = '/dev/dev0'", //
87                  "  info.parent = '/dev/usbus0'  (string)", //
88                  "  usb_device.product = 'Widget'  (string)", //
89                  "udi = '/dev/dev0_if0'", //
90                  "  info.parent = '/dev/dev0'  (string)", //
91                  "  usb.interface.number = 0  (int)");
92          List<UsbDevice> controllers = FreeBsdUsbDevice.parseUsbDevices(lshal);
93          assertThat(controllers, hasSize(1));
94          // Only the real device hangs off the controller; the _if interface node is not a child.
95          assertThat(controllers.get(0).getConnectedDevices(), hasSize(1));
96          assertThat(controllers.get(0).getConnectedDevices().get(0).getName(), is("Widget"));
97      }
98  
99      @Test
100     void testParseUsbDevicesEmpty() {
101         assertThat(FreeBsdUsbDevice.parseUsbDevices(Collections.emptyList()), is(empty()));
102     }
