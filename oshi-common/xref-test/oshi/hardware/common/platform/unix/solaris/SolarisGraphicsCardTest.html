5   package oshi.hardware.common.platform.unix.solaris;

7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.GraphicsCard;
19  
20  class SolarisGraphicsCardTest {
21  
22      @Test
23      void testParsePrtconf() {
24          // prtconf -pv output with a display-class device (class-code 0003xxxx)
25          List<String> prtconf = Arrays.asList(//
26                  "    Node 0x1f8e610", //
27                  "      name:  'pci8086,2a42'", //
28                  "      model:  'Intel GM45 Integrated Graphics'", //
29                  "      vendor-id:  00008086", //
30                  "      device-id:  00002a42", //
31                  "      revision-id:  00000007", //
32                  "      class-code:  00030000");
33          List<GraphicsCard> cards = SolarisGraphicsCard.parsePrtconf(prtconf);
34          assertThat(cards, hasSize(1));
35          GraphicsCard card = cards.get(0);
36          assertThat(card.getName(), is("Intel GM45 Integrated Graphics"));
37          assertThat(card.getVendor(), is("0x8086"));
38          assertThat(card.getDeviceId(), is("0x2a42"));
39          assertThat(card.getVersionInfo(), is("revision-id:  00000007"));
40          assertThat(card.getVRam(), is(0L));
41      }
42  
43      @Test
44      void testParsePrtconfNoDisplayDevice() {
45          // prtconf -pv output with NO display-class device
46          List<String> prtconf = Arrays.asList(//
47                  "    Node 0x1f8e610", //
48                  "      name:  'pci8086,29c0'", //
49                  "      vendor-id:  00008086", //
50                  "      device-id:  000029c0", //
51                  "      class-code:  00060000"); // class 0006 = Bridge, not Display
52          List<GraphicsCard> cards = SolarisGraphicsCard.parsePrtconf(prtconf);
53          assertThat(cards, is(empty()));
54      }
55  
56      @Test
57      void testParsePrtconfEmpty() {
58          List<GraphicsCard> cards = SolarisGraphicsCard.parsePrtconf(Collections.emptyList());
59          assertThat(cards, is(empty()));
60      }
61  
62      @Test
63      void testParsePrtconfDisplayAtEnd() {
64          // Display device at end of output (no following Node header to flush)
65          List<String> prtconf = Arrays.asList(//
66                  "    Node 0xaabbcc", //
67                  "      name:  'pci1002,67df'", //
68                  "      model:  'AMD Radeon RX 580'", //
69                  "      vendor-id:  00001002", //
70                  "      device-id:  000067df", //
71                  "      revision-id:  000000e7", //
72                  "      class-code:  00030000");
73          List<GraphicsCard> cards = SolarisGraphicsCard.parsePrtconf(prtconf);
74          assertThat(cards, hasSize(1));
75          assertThat(cards.get(0).getName(), is("AMD Radeon RX 580"));
76          assertThat(cards.get(0).getVendor(), is("0x1002"));
77          assertThat(cards.get(0).getDeviceId(), is("0x67df"));
78      }
79  }
