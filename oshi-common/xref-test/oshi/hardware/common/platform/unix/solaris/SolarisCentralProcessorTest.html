21      @Test
22      void testParseDmesgCpuInfo() {
23          List<String> dmesg = Arrays.asList(//
24                  "Jan  9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: Intel(r) Celeron(r) CPU J3455 @ 1.50GHz", //
25                  "Jan  9 14:04:28 solaris unix: [ID 950921 kern.info] cpu0: x86 (chipid 0x0 GenuineIntel 506C9 family 6 model 92 step 9 clock 1500 MHz)", //
26                  "Jan  9 14:04:28 solaris unix: [ID 950921 kern.info] cpu1: Intel(r) Celeron(r) CPU J3455 @ 1.50GHz");
27          Map<Integer, String> result = SolarisCentralProcessor.parseDmesgCpuInfo(dmesg);
28          // Only lines matching ".* cpu(\d+): ((ARM|AMD|Intel).+)" are captured
29          assertThat(result.size(), is(2));
30          assertThat(result, hasEntry(0, "Intel(r) Celeron(r) CPU J3455 @ 1.50GHz"));
31          assertThat(result, hasEntry(1, "Intel(r) Celeron(r) CPU J3455 @ 1.50GHz"));
32      }

37          assertThat(result.isEmpty(), is(true));
38      }

40      @Test
41      void testParseNumaNodes() {
42          List<String> lgrpinfo = Arrays.asList(//
43                  "lgroup 0 (root):", //
44                  "CPUs: 0-3", //
45                  "lgroup 1 (leaf):", //
46                  "CPUs: 4 5 6 7");
47          Map<Integer, Integer> result = SolarisCentralProcessor.parseNumaNodes(lgrpinfo);
48          assertThat(result.size(), is(8));
49          assertThat(result, hasEntry(0, 0));
50          assertThat(result, hasEntry(1, 0));
51          assertThat(result, hasEntry(2, 0));
52          assertThat(result, hasEntry(3, 0));
53          assertThat(result, hasEntry(4, 1));
54          assertThat(result, hasEntry(5, 1));
55          assertThat(result, hasEntry(6, 1));
56          assertThat(result, hasEntry(7, 1));
57      }

62          assertThat(result.isEmpty(), is(true));

65      @Test
66      void testParseIsainfoFlags() {
67          // isainfo -v output: first a 64-bit header, then indented flags, then 32-bit header
68          List<String> isainfo = Arrays.asList(//
69                  "64-bit amd64 applications", //
70                  "        avx512f avx512cd sha sse4.2 popcnt", //
71                  "32-bit i386 applications", //
72                  "        sse2 sse mmx");
73          String[] flags = SolarisCentralProcessor.parseIsainfoFlags(isainfo);
74          // Should only contain flags from the 64-bit section, lowercased, split on whitespace.
75          // The first element is empty because the flags StringBuilder starts with a space.
76          assertThat(flags, arrayContaining("", "avx512f", "avx512cd", "sha", "sse4.2", "popcnt"));
77      }
78  
79      @Test
80      void testParseIsainfoFlagsEmpty() {
81          String[] flags = SolarisCentralProcessor.parseIsainfoFlags(Collections.emptyList());
82          // Splitting an empty string produces a single-element array with ""
83          assertThat(flags.length, is(1));
84          assertThat(flags[0], is(""));
85      }
