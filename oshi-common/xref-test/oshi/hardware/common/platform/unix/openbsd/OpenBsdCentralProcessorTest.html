21  /**
22   * Tests the native-free tick-mapping and load-average logic hoisted into {@link OpenBsdCentralProcessor}, using a stub
23   * that supplies raw CPU-state and load-average values without native calls.

27      @Test
28      void testFillTicksFiveElement() {
29          // Pre-6.4 layout: user, nice, sys, intr, idle
30          StubOpenBsdCentralProcessor cpu = new StubOpenBsdCentralProcessor(new long[] { 10L, 20L, 30L, 40L, 50L },
31                  new double[0], 0);
32          long[] ticks = cpu.querySystemCpuLoadTicks();
33          assertThat(ticks[TickType.USER.getIndex()], is(10L));
34          assertThat(ticks[TickType.NICE.getIndex()], is(20L));
35          assertThat(ticks[TickType.SYSTEM.getIndex()], is(30L));
36          assertThat(ticks[TickType.IRQ.getIndex()], is(40L));
37          assertThat(ticks[TickType.IDLE.getIndex()], is(50L));
38      }
39  
40      @Test
41      void testFillTicksSixElement() {
42          // OpenBSD 6.4+ inserts a spin element (99) between sys and intr, shifting IRQ and IDLE by one
43          StubOpenBsdCentralProcessor cpu = new StubOpenBsdCentralProcessor(new long[] { 10L, 20L, 30L, 99L, 40L, 50L },
44                  new double[0], 0);
45          long[] ticks = cpu.querySystemCpuLoadTicks();
46          assertThat(ticks[TickType.USER.getIndex()], is(10L));
47          assertThat(ticks[TickType.NICE.getIndex()], is(20L));
48          assertThat(ticks[TickType.SYSTEM.getIndex()], is(30L));
49          assertThat(ticks[TickType.IRQ.getIndex()], is(40L));
50          assertThat(ticks[TickType.IDLE.getIndex()], is(50L));
51      }

54      void testFillTicksShortArrayLeavesZeros() {
55          // Fewer than CPUSTATES elements: mapping is skipped, ticks stay zero
56          StubOpenBsdCentralProcessor cpu = new StubOpenBsdCentralProcessor(new long[] { 1L, 2L, 3L }, new double[0], 0);
57          long[] ticks = cpu.querySystemCpuLoadTicks();

96      @Test
97      void testParseVmStats() {
98          // Real OpenBSD vmstat -s: "software interrupts" appears before bare "interrupts"
99          List<String> vmstat = Arrays.asList(//
100                 "      142983 cpu context switches", //
101                 "           0 software interrupts", //
102                 "       28301 interrupts", //
103                 "       50000 some other stat");
104         Pair<Long, Long> result = OpenBsdCentralProcessor.parseVmStats(vmstat);
105         assertThat(result.getA(), is(142983L));
106         assertThat(result.getB(), is(28301L));

116     @Test
117     void testCpuidToFamilyModelStepping() {
118         // Intel Core i7-8700K: CPUID = 0x000906EA → family=6, model=158, stepping=10
119         Triplet<Integer, Integer, Integer> fms = OpenBsdCentralProcessor.cpuidToFamilyModelStepping(0x000906EA);
120         assertThat(fms.getA(), is(6));
121         assertThat(fms.getB(), is(158));
122         assertThat(fms.getC(), is(10));
123     }
124 
125     @Test
126     void testCpuidToFamilyModelSteppingAmd() {
127         // AMD Zen 2 (Ryzen 3600): CPUID = 0x00800F12 → family=143 (0x8F), model=1, stepping=2
128         // formula: family = (cpuid >> 16 & 0xff0) | (cpuid >> 8 & 0xf)
129         // model = (cpuid >> 12 & 0xf0) | (cpuid >> 4 & 0xf)
130         // 0x00800F12 >> 16 = 0x0080, & 0xff0 = 0x80; >> 8 = 0x00800F, & 0xf = 0xF; family = 0x80|0xF = 0x8F = 143
131         // 0x00800F12 >> 12 = 0x00800, & 0xf0 = 0x00; >> 4 = 0x0080_0F1, & 0xf = 0x1; model = 0|1 = 1
132         Triplet<Integer, Integer, Integer> fms = OpenBsdCentralProcessor.cpuidToFamilyModelStepping(0x00800F12);
133         assertThat(fms.getA(), is(143));
134         assertThat(fms.getB(), is(1));
135         assertThat(fms.getC(), is(2));
136     }

138     @Test
139     void testCpuidToFamilyModelSteppingZero() {

141         assertThat(fms.getA(), is(0));
142         assertThat(fms.getB(), is(0));
143         assertThat(fms.getC(), is(0));

146     /**
147      * Minimal stub supplying raw CPU-state and load-average values without native calls.
148      */
149     static class StubOpenBsdCentralProcessor extends OpenBsdCentralProcessor {
150 
151         private final long[] systemCpTime;
152         private final double[] loadavgValues;
153         private final int loadavgRetval;
154 
155         StubOpenBsdCentralProcessor(long[] systemCpTime, double[] loadavgValues, int loadavgRetval) {
156             this.systemCpTime = systemCpTime;
157             this.loadavgValues = loadavgValues;
158             this.loadavgRetval = loadavgRetval;
159         }
160 
161         @Override
162         protected int sysctl(String name, int def) {
163             return def;
164         }
165 
166         @Override
167         protected String sysctl(String name, String def) {
168             return def;
169         }
170 
171         @Override
172         protected String sysctl(int[] mib, String def) {
173             return def;
174         }
175 
176         @Override
177         protected int sysctl(int[] mib, int def) {
178             return def;
179         }
180 
181         @Override
182         protected Pair<Long, Long> queryVmStats() {
183             return new Pair<>(0L, 0L);
184         }
185 
186         @Override
187         protected long[] querySystemCpTime() {
188             return systemCpTime;
189         }
190 
191         @Override
192         protected long[] queryProcessorCpTime(int cpu) {
193             return systemCpTime;
194         }
195 
196         @Override
197         protected int getloadavgNative(double[] loadavg, int nelem) {
198             for (int i = 0; i < nelem && i < loadavgValues.length; i++) {
199                 loadavg[i] = loadavgValues[i];
200             }
201             return loadavgRetval;
202         }
203     }
