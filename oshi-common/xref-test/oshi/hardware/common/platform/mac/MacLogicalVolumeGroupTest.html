5   package oshi.hardware.common.platform.mac;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasItem;
10  import static org.hamcrest.Matchers.hasKey;
11  import static org.hamcrest.Matchers.hasSize;
12  import static org.hamcrest.Matchers.is;
13  
14  import java.util.Arrays;
15  import java.util.Collections;
16  import java.util.List;
17  
18  import org.junit.jupiter.api.Test;
19  
20  import oshi.hardware.LogicalVolumeGroup;
21  
22  class MacLogicalVolumeGroupTest {
23  
24      // Fixture: typical diskutil cs list output with one CoreStorage volume group
25      private static final List<String> DISKUTIL_CS_OUTPUT = Arrays.asList("CoreStorage logical volume groups (1 found)",
26              "|", "+-- Logical Volume Group B6D1A2C3-4E5F-6789-ABCD-EF0123456789",
27              "    =========================================================", "    Name:         Macintosh HD",
28              "    Status:       Online", "    Size:         499248103424 B (499.2 GB)",
29              "    Free Space:   12345678 B (12.3 MB)", "    |",
30              "    +-< Physical Volume 0A1B2C3D-4E5F-6789-ABCD-EF0123456789",
31              "    |   ----------------------------------------------------", "    |   Index:    0",
32              "    |   Disk:     disk0s2", "    |   Status:   Online", "    |   Size:     499248103424 B (499.2 GB)",
33              "    |", "    +-> Logical Volume Family 1A2B3C4D-5E6F-7890-ABCD-EF0123456789",
34              "        ----------------------------------------------------------",
35              "        +-> Logical Volume 2A3B4C5D-6E7F-8901-ABCD-EF0123456789",
36              "            ---------------------------------------------------",
37              "            Disk:                  disk1", "            Status:                Online",
38              "            Size (Total):          498877931520 B (498.9 GB)");
39  
40      @Test
41      void testParseDiskutilCsListSingleGroup() {
42          List<LogicalVolumeGroup> groups = MacLogicalVolumeGroup.parseDiskutilCsList(DISKUTIL_CS_OUTPUT);
43          assertThat(groups, hasSize(1));
44  
45          LogicalVolumeGroup group = groups.get(0);
46          assertThat(group.getName(), is("Macintosh HD"));
47          assertThat(group.getPhysicalVolumes(), hasItem("disk0s2"));
48          assertThat(group.getLogicalVolumes(), hasKey("disk1"));
49      }
50  
51      @Test
52      void testParseDiskutilCsListEmpty() {
53          List<LogicalVolumeGroup> groups = MacLogicalVolumeGroup.parseDiskutilCsList(Collections.emptyList());
54          assertThat(groups, is(empty()));
55      }
56  
57      @Test
58      void testParseDiskutilCsListNoGroups() {
59          List<String> noGroups = Arrays.asList("No CoreStorage logical volume groups found");
60          List<LogicalVolumeGroup> groups = MacLogicalVolumeGroup.parseDiskutilCsList(noGroups);
61          assertThat(groups, is(empty()));
62      }
63  
64      @Test
65      void testParseDiskutilCsListMultipleGroups() {
66          List<String> twoGroups = Arrays.asList("+-- Logical Volume Group AAA", "    Name:         Group One",
67                  "    +-< Physical Volume PV1", "    |   Disk:     disk0s2", "    +-> Logical Volume LV1",
68                  "            Disk:                  disk1", "+-- Logical Volume Group BBB",
69                  "    Name:         Group Two", "    +-< Physical Volume PV2", "    |   Disk:     disk2s1",
70                  "    +-> Logical Volume LV2", "            Disk:                  disk3");
71          List<LogicalVolumeGroup> groups = MacLogicalVolumeGroup.parseDiskutilCsList(twoGroups);
72          assertThat(groups, hasSize(2));
73          assertThat(groups.get(0).getName(), is("Group One"));
74          assertThat(groups.get(1).getName(), is("Group Two"));
75          assertThat(groups.get(0).getPhysicalVolumes(), hasItem("disk0s2"));
76          assertThat(groups.get(1).getPhysicalVolumes(), hasItem("disk2s1"));
77      }
78  
79      @Test
80      void testParseDiskutilCsListMultiplePVsAndLVs() {
81          List<String> multiPvLv = Arrays.asList("+-- Logical Volume Group CCC", "    Name:         Fusion Drive",
82                  "    +-< Physical Volume PV1", "    |   Disk:     disk0s2", "    +-< Physical Volume PV2",
83                  "    |   Disk:     disk1s1", "    +-> Logical Volume LV1", "            Disk:                  disk2",
84                  "    +-> Logical Volume LV2", "            Disk:                  disk3");
85          List<LogicalVolumeGroup> groups = MacLogicalVolumeGroup.parseDiskutilCsList(multiPvLv);
86          assertThat(groups, hasSize(1));
87          assertThat(groups.get(0).getPhysicalVolumes(), hasItem("disk0s2"));
88          assertThat(groups.get(0).getPhysicalVolumes(), hasItem("disk1s1"));
89          assertThat(groups.get(0).getLogicalVolumes(), hasKey("disk2"));
90          assertThat(groups.get(0).getLogicalVolumes(), hasKey("disk3"));
91      }
92  }
