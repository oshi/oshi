25      /** Builds a plain AbstractUsbDevice so buildDeviceTree can be exercised without a platform backend. */
26      private static final AbstractUsbDevice.UsbDeviceFactory FACTORY = (name, vendor, vendorId, productId, serialNumber,
27              uniqueDeviceId, connectedDevices) -> new AbstractUsbDevice(name, vendor, vendorId, productId, serialNumber,
28                      uniqueDeviceId, connectedDevices) {
29              };

31      @Test
32      void testGetters() {
33          AbstractUsbDevice device = new AbstractUsbDevice("Mouse", "Logitech", "046d", "c077", "SN1", "USB1",
34                  Collections.emptyList()) {
35          };
36          assertThat(device.getName(), is("Mouse"));
37          assertThat(device.getVendor(), is("Logitech"));
38          assertThat(device.getVendorId(), is("046d"));
39          assertThat(device.getProductId(), is("c077"));
40          assertThat(device.getSerialNumber(), is("SN1"));
41          assertThat(device.getUniqueDeviceId(), is("USB1"));
42          assertThat(device.getConnectedDevices(), is(Collections.emptyList()));
43      }

45      @Test
46      void testConnectedDevices() {
47          AbstractUsbDevice child = new AbstractUsbDevice("Mouse", "Logitech", "", "", "", "", Collections.emptyList()) {
48          };
49          AbstractUsbDevice parent = new AbstractUsbDevice("Hub", "Generic", "", "", "", "",
50                  Collections.singletonList(child)) {
51          };
52          assertThat(parent.getConnectedDevices(), hasSize(1));
53          assertThat(parent.getConnectedDevices().get(0).getName(), is("Mouse"));
54      }

82          nameMap.put("usb1/1-1", "USB Mouse");

84          vendorMap.put("usb1/1-1", "Logitech");

88          productIdMap.put("usb1/1-1", "c077");
89          Map<String, String> serialMap = new HashMap<>();
90          serialMap.put("usb1/1-1", "SN123");
91          Map<String, List<String>> hubMap = Collections.emptyMap();
92  
93          UsbDevice device = AbstractUsbDevice.buildDeviceTree("usb1/1-1", "0000", "0000", nameMap, vendorMap,
94                  vendorIdMap, productIdMap, serialMap, hubMap, FACTORY);

97          assertThat(device.getVendor(), is("Logitech"));
98          assertThat(device.getVendorId(), is("046d"));
99          assertThat(device.getProductId(), is("c077"));
100         assertThat(device.getSerialNumber(), is("SN123"));
101         assertThat(device.getUniqueDeviceId(), is("usb1/1-1"));
102         assertThat(device.getConnectedDevices(), is(empty()));
103     }

127     @Test
128     void testBuildDeviceTreeHubWithSortedChildren() {
129         Map<String, String> nameMap = new HashMap<>();
130         nameMap.put("usb1", "Root Hub");
131         nameMap.put("usb1/1-1", "Zebra Device");
132         nameMap.put("usb1/1-2", "Alpha Device");
133         Map<String, String> vendorMap = Collections.emptyMap();
134         Map<String, String> vendorIdMap = new HashMap<>();
135         vendorIdMap.put("usb1", "1d6b");
136         vendorIdMap.put("usb1/1-1", "aaaa");
137         vendorIdMap.put("usb1/1-2", "bbbb");
138         Map<String, String> productIdMap = new HashMap<>();
139         productIdMap.put("usb1", "0002");
140         productIdMap.put("usb1/1-1", "0001");
141         productIdMap.put("usb1/1-2", "0002");
142         Map<String, String> serialMap = Collections.emptyMap();
143         Map<String, List<String>> hubMap = new HashMap<>();
144         hubMap.put("usb1", Arrays.asList("usb1/1-1", "usb1/1-2"));
145 
146         UsbDevice hub = AbstractUsbDevice.buildDeviceTree("usb1", "0000", "0000", nameMap, vendorMap, vendorIdMap,
147                 productIdMap, serialMap, hubMap, FACTORY);
148 
149         assertThat(hub.getName(), is("Root Hub"));
150         List<UsbDevice> children = hub.getConnectedDevices();
151         assertThat(children, hasSize(2));
152         // Children sorted by name: "Alpha Device" before "Zebra Device"
153         assertThat(children.get(0).getName(), is("Alpha Device"));
154         assertThat(children.get(1).getName(), is("Zebra Device"));
155     }

161     @Test
162     void testBuildDeviceTreeNestedHierarchy() {
163         Map<String, String> nameMap = new HashMap<>();
164         nameMap.put("usb1", "Root Hub");
165         nameMap.put("usb1/1-1", "Hub");
166         nameMap.put("usb1/1-1/1-1.1", "Keyboard");
167         Map<String, String> vendorIdMap = new HashMap<>();
168         vendorIdMap.put("usb1", "1d6b");
169         vendorIdMap.put("usb1/1-1", "0424");
170         vendorIdMap.put("usb1/1-1/1-1.1", "04f2");
171         Map<String, String> productIdMap = new HashMap<>();
172         productIdMap.put("usb1", "0002");
173         productIdMap.put("usb1/1-1", "2514");
174         productIdMap.put("usb1/1-1/1-1.1", "0112");
175         Map<String, List<String>> hubMap = new HashMap<>();
176         hubMap.put("usb1", Collections.singletonList("usb1/1-1"));
177         hubMap.put("usb1/1-1", Collections.singletonList("usb1/1-1/1-1.1"));
178 
179         UsbDevice root = AbstractUsbDevice.buildDeviceTree("usb1", "0000", "0000", nameMap, Collections.emptyMap(),
180                 vendorIdMap, productIdMap, Collections.emptyMap(), hubMap, FACTORY);
181 
182         assertThat(root.getConnectedDevices(), hasSize(1));
183         UsbDevice hub = root.getConnectedDevices().get(0);
184         assertThat(hub.getName(), is("Hub"));
185         assertThat(hub.getConnectedDevices(), hasSize(1));
186         assertThat(hub.getConnectedDevices().get(0).getName(), is("Keyboard"));
187     }
