150     @Test
151     void testOrderedProcCaches() {
152         // Model real Apple Silicon-style caches where P-core and E-core variants
153         // of the same level+type have different sizes but the same highestOneBit,
154         // producing sort key ties (e.g., 131072 and 196608 both -> highestOneBit 131072)
155         Set<ProcessorCache> caches = new HashSet<>();
156         ProcessorCache l2p = new ProcessorCache(2, 0, 128, 16777216, Type.UNIFIED);
157         ProcessorCache l2e = new ProcessorCache(2, 0, 128, 25165824, Type.UNIFIED);
158         ProcessorCache l1ip = new ProcessorCache(1, 0, 128, 131072, Type.INSTRUCTION);
159         ProcessorCache l1ie = new ProcessorCache(1, 0, 128, 196608, Type.INSTRUCTION);
160         ProcessorCache l1dp = new ProcessorCache(1, 0, 128, 131072, Type.DATA);
161         ProcessorCache l1de = new ProcessorCache(1, 0, 128, 196608, Type.DATA);
162         caches.add(l2p);
163         caches.add(l2e);
164         caches.add(l1ip);
165         caches.add(l1ie);
166         caches.add(l1dp);
167         caches.add(l1de);
168 
169         List<ProcessorCache> ordered = AbstractCentralProcessor.orderedProcCaches(caches);
170         assertThat(ordered, hasSize(6));
171         // L2 sorts before L1; within same level+type+highestOneBit, order is not guaranteed
172         assertThat(ordered.subList(0, 2), containsInAnyOrder(l2p, l2e));
173         assertThat(ordered.subList(2, 4), containsInAnyOrder(l1ip, l1ie));
174         assertThat(ordered.subList(4, 6), containsInAnyOrder(l1dp, l1de));
175     }

191     private static AbstractCentralProcessor createHybridProcessor() {
192         return new AbstractCentralProcessor() {
193             @Override
194             protected Quartet<List<LogicalProcessor>, List<PhysicalProcessor>, List<ProcessorCache>, List<String>> initProcessorCounts() {
195                 List<LogicalProcessor> logProcs = Arrays.asList(new LogicalProcessor(0, 0, 0),
196                         new LogicalProcessor(1, 1, 0), new LogicalProcessor(2, 2, 0), new LogicalProcessor(3, 3, 0));
197                 List<PhysicalProcessor> physProcs = Arrays.asList(new PhysicalProcessor(0, 0, 1, "P-core"),
198                         new PhysicalProcessor(0, 1, 1, "P-core"), new PhysicalProcessor(0, 2, 0, "E-core"),
199                         new PhysicalProcessor(0, 3, 0, "E-core"));
200                 List<ProcessorCache> caches = Collections
201                         .singletonList(new ProcessorCache(2, 0, 64, 4 * 1024 * 1024, Type.UNIFIED));
202                 return new Quartet<>(logProcs, physProcs, caches, Arrays.asList("sse", "avx"));
203             }

270     @Test
271     void testCreateProcListFromDmesg() {
272         AbstractCentralProcessor cpu = createProcessor();
273         List<LogicalProcessor> logProcs = Arrays.asList(new LogicalProcessor(0, 0, 0), new LogicalProcessor(1, 1, 0));
274         Map<Integer, String> dmesg = new HashMap<>();
275         dmesg.put(0, "ARM Cortex-A73");
276         dmesg.put(1, "ARM Cortex-A53");
277 
278         List<PhysicalProcessor> result = cpu.createProcListFromDmesg(logProcs, dmesg);
279         assertThat(result.size(), is(2));
280         // A73 (>=70) should be efficiency 1 (big), A53 (<70) should be 0 (LITTLE)
281         assertThat(result.get(0).getEfficiency(), is(1));
282         assertThat(result.get(1).getEfficiency(), is(0));
283     }
