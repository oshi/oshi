25      @Test
26      void testToString() {
27          AbstractGlobalMemory mem = new AbstractGlobalMemory() {
28              @Override
29              public long getTotal() {
30                  return 8L * 1024 * 1024 * 1024;
31              }
32  
33              @Override
34              public long getAvailable() {
35                  return 4L * 1024 * 1024 * 1024;
36              }
37  
38              @Override
39              public long getPageSize() {
40                  return 4096L;
41              }
42  
43              @Override
44              public VirtualMemory getVirtualMemory() {
45                  return null;
46              }
47          };
48          String s = mem.toString();
49          assertThat(s, containsString("Available:"));
50          assertThat(s, containsString("4 GiB"));
51          assertThat(s, containsString("8 GiB"));
52      }

73      @Test
74      void testParsePhysicalMemoryTwoDimms() {
75          List<PhysicalMemory> result = AbstractGlobalMemory.getPhysicalMemory(DMI_TYPE_17);
76          assertThat(result, hasSize(2));
77  
78          PhysicalMemory first = result.get(0);
79          assertThat(first.getBankLabel(), is("NODE 0/DIMM_A1"));
80          assertThat(first.getCapacity(), is(16384L * 1024 * 1024));
81          assertThat(first.getClockSpeed(), is(2_666_000_000L));
82          assertThat(first.getManufacturer(), is("Samsung"));
83          assertThat(first.getMemoryType(), is("DDR4"));
84          assertThat(first.getPartNumber(), is("M393A2K43CB2-CTD"));
85          assertThat(first.getSerialNumber(), is("12345ABC"));
86  
87          PhysicalMemory second = result.get(1);
88          assertThat(second.getBankLabel(), is("NODE 1/DIMM_B1"));
89          assertThat(second.getCapacity(), is(8192L * 1024 * 1024));
90          assertThat(second.getClockSpeed(), is(2_400_000_000L));
91          assertThat(second.getManufacturer(), is("Micron"));
92          assertThat(second.getMemoryType(), is("DDR4"));
93          assertThat(second.getPartNumber(), is("MTA18ASF1G72PZ"));
94          assertThat(second.getSerialNumber(), is("67890DEF"));

102     @Test
103     void testParsePhysicalMemorySingleDimm() {
104         List<String> single = Arrays.asList("Handle 0x0038, DMI type 17, 40 bytes", "Memory Device", "\tSize: 4096 MB",
105                 "\tLocator: DIMM0", "\tBank Locator: Bank0", "\tType: DDR3", "\tSpeed: 1600 MT/s",
106                 "\tManufacturer: Kingston", "\tPart Number: KVR16N11/4", "\tSerial Number: AABB1122");
107         List<PhysicalMemory> result = AbstractGlobalMemory.getPhysicalMemory(single);
108         assertThat(result, hasSize(1));
109         assertThat(result.get(0).getBankLabel(), is("Bank0/DIMM0"));
110         assertThat(result.get(0).getCapacity(), is(4096L * 1024 * 1024));
111     }

113     @Test
114     void testParsePhysicalMemoryNoStateLeak() {
115         // Second DIMM omits Manufacturer and Part Number — should NOT inherit from first
116         List<String> twoEntries = Arrays.asList("Handle 0x0038, DMI type 17, 40 bytes", "Memory Device",
117                 "\tSize: 8192 MB", "\tLocator: DIMM_A1", "\tBank Locator: Bank0", "\tType: DDR4", "\tSpeed: 2666 MT/s",
118                 "\tManufacturer: Samsung", "\tPart Number: M393A2K43CB2-CTD", "\tSerial Number: SN111",
119                 "Handle 0x0039, DMI type 17, 40 bytes", "Memory Device", "\tSize: 4096 MB", "\tLocator: DIMM_B1",
120                 "\tBank Locator: Bank1", "\tType: DDR4", "\tSpeed: 2400 MT/s", "\tSerial Number: SN222");
121         List<PhysicalMemory> result = AbstractGlobalMemory.getPhysicalMemory(twoEntries);
122         assertThat(result, hasSize(2));
123 
124         assertThat(result.get(0).getManufacturer(), is("Samsung"));
125         assertThat(result.get(0).getPartNumber(), is("M393A2K43CB2-CTD"));
126 
127         // Second DIMM should have defaults, not Samsung/M393...
128         assertThat(result.get(1).getBankLabel(), is("Bank1/DIMM_B1"));
129         assertThat(result.get(1).getCapacity(), is(4096L * 1024 * 1024));
130         assertThat(result.get(1).getManufacturer(), is(Constants.UNKNOWN));
131         assertThat(result.get(1).getPartNumber(), is(Constants.UNKNOWN));
132     }
