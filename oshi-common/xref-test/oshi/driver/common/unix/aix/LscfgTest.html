5   package oshi.driver.common.unix.aix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.hamcrest.Matchers.nullValue;
10  
11  import java.util.Arrays;
12  import java.util.Collections;
13  
14  import org.junit.jupiter.api.Test;
15  
16  import oshi.util.tuples.Pair;
17  
18  class LscfgTest {
19  
20      @Test
21      void testParseModelSerial() {
22          // lscfg -vl hdisk0: the "Machine Type and Model" line overrides the first-line description
23          Pair<String, String> modSer = Lscfg.parseModelSerial(Arrays.asList(//
24                  "  hdisk0           U78CB.001.WZS00MP-P1-C2-T1-L0  MPIO IBM 2076 FC Disk", //
25                  "        Manufacturer................IBM", //
26                  "        Machine Type and Model......2076", //
27                  "        Serial Number...............0123456789AB", //
28                  "        EC Level....................D77161"), "hdisk0");
29          assertThat(modSer.getA(), is("2076"));
30          assertThat(modSer.getB(), is("0123456789AB"));
31      }
32  
33      @Test
34      void testParseModelSerialFallsBackToDescription() {
35          // With no "Machine Type and Model" line, the model is the description after the location code
36          Pair<String, String> modSer = Lscfg.parseModelSerial(
37                  Collections.singletonList("  cd0              U78CB.001.WZS00MP-P2-D1  SATA DVD-RAM Drive"), "cd0");
38          assertThat(modSer.getA(), is("SATA DVD-RAM Drive"));
39          assertThat(modSer.getB(), is(nullValue()));
40      }
41  
42      @Test
43      void testParseModelSerialDeviceAtEndOfLine() {
44          // A line ending in the device name splits to a single element; the fallback must be skipped, not crash
45          Pair<String, String> modSer = Lscfg.parseModelSerial(Collections.singletonList("  Adapter containing hdisk0"),
46                  "hdisk0");
47          assertThat(modSer.getA(), is(nullValue()));
48          assertThat(modSer.getB(), is(nullValue()));
49      }
50  
51      @Test
52      void testParseModelSerialEmpty() {
53          Pair<String, String> modSer = Lscfg.parseModelSerial(Collections.emptyList(), "hdisk0");
54          assertThat(modSer.getA(), is(nullValue()));
55          assertThat(modSer.getB(), is(nullValue()));
56      }
57  }
