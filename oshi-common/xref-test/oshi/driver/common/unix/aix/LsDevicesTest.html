31      @Test
32      void testParseDeviceMajorMinor() {
33          // ls -l /dev: block devices begin with 'b'; major,minor are the 2nd and 3rd integers, name is the last token
34          Map<String, Pair<Integer, Integer>> map = Ls.parseDeviceMajorMinor(Arrays.asList(//
35                  "total 0", //
36                  "crw-rw-rw-  1 root system  2,  2 Jun 28  1970 null", // not a block device, skipped
37                  "brw-rw----  1 root system 10,  5 Sep 12  2017 hd2", //
38                  "brw-------  1 root system 20,  0 Jun 28  1970 hdisk0"));
39          assertThat(map, not(hasKey("null")));
40          assertThat(map.get("hd2").getA(), is(10));
41          assertThat(map.get("hd2").getB(), is(5));
42          assertThat(map.get("hdisk0").getA(), is(20));
43          assertThat(map.get("hdisk0").getB(), is(0));
44      }
45  
46      @Test
47      void testParseDeviceMajorMinorEmpty() {
48          assertThat(Ls.parseDeviceMajorMinor(Collections.emptyList()), is(anEmptyMap()));
49      }
50  
51      @Test
52      @EnabledOnOs(OS.AIX)
53      void testQueryLsDevices() {
54          Map<String, Pair<Integer, Integer>> majMinMap = Ls.queryDeviceMajorMinor();
55          assertThat("Device Major Minor Map shouldn't be empty", majMinMap, not(anEmptyMap()));
56          // Key set is device name
57          boolean foundNonNull = false;
58          boolean foundPartitions = false;
59          for (String device : majMinMap.keySet()) {
60              Pair<String, String> modSer = Lscfg.queryModelSerial(device);
61              if (modSer.getA() != null || modSer.getB() != null) {
62                  foundNonNull = true;
63                  List<HWPartition> lvs = Lspv.queryLogicalVolumes(device, majMinMap);
64                  if (!lvs.isEmpty()) {
65                      foundPartitions = true;
66                  }
67              }
68              if (foundNonNull && foundPartitions) {
69                  break;
70              }
71          }
72          assertTrue(foundNonNull, "At least one device model/serial should be non null");
73          assertTrue(foundPartitions, "At least one device should have partitions");
74  
75          List<String> lscfg = Lscfg.queryAllDevices();
76          assertThat("Output of lscfg should be nonempty", lscfg.size(), greaterThan(0));
77  
78          Triplet<String, String, String> modSerVer = Lscfg.queryBackplaneModelSerialVersion(lscfg);
79          // Either all should be null or none should be null
80          if (!(modSerVer.getA() == null && modSerVer.getB() == null && modSerVer.getC() == null)) {
81              assertNotNull(modSerVer.getA(), "Backplane Model should not be null");
82              assertNotNull(modSerVer.getB(), "Backplane Serial should not be null");
83              assertNotNull(modSerVer.getC(), "Backplane Version should not be null");
84          }
85      }
