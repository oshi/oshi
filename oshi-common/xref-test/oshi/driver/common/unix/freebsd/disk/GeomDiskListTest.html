7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.anEmptyMap;
9   import static org.hamcrest.Matchers.is;
10  
11  import java.util.Arrays;
12  import java.util.Collections;
13  import java.util.List;
14  import java.util.Map;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.util.Constants;
19  import oshi.util.tuples.Triplet;
20  
21  class GeomDiskListTest {
22  
23      @Test
24      void testParseGeomDiskListTypicalOutput() {
25          List<String> geom = Arrays.asList("Geom name: ada0", "Providers:", "1. Name: ada0",
26                  "   Mediasize: 500107862016 (465G)", "   Sectorsize: 512", "   Mode: r2w2e3",
27                  "   descr: Samsung SSD 860 EVO 500GB", "   ident: S3Z2NB0K123456X", "   rotationrate: 0", "",
28                  "Geom name: da0", "Providers:", "1. Name: da0", "   Mediasize: 8053063680 (7.5G)", "   Sectorsize: 512",
29                  "   Mode: r0w0e0", "   descr: USB Flash Drive", "   ident: (null)", "   rotationrate: unknown");
30  
31          Map<String, Triplet<String, String, Long>> result = GeomDiskList.parseGeomDiskList(geom);
32  
33          assertThat(result.size(), is(2));
34  
35          Triplet<String, String, Long> ada0 = result.get("ada0");
36          assertThat(ada0.getA(), is("Samsung SSD 860 EVO 500GB"));
37          assertThat(ada0.getB(), is("S3Z2NB0K123456X"));
38          assertThat(ada0.getC(), is(500107862016L));
39  
40          Triplet<String, String, Long> da0 = result.get("da0");
41          assertThat(da0.getA(), is("USB Flash Drive"));
42          assertThat(da0.getB(), is(""));
43          assertThat(da0.getC(), is(8053063680L));
44      }
45  
46      @Test
47      void testParseGeomDiskListEmptyInput() {
48          Map<String, Triplet<String, String, Long>> result = GeomDiskList.parseGeomDiskList(Collections.emptyList());
49          assertThat(result, is(anEmptyMap()));
50      }
51  
52      @Test
53      void testParseGeomDiskListNoMediasizeDefaultsToZero() {
54          List<String> geom = Arrays.asList("Geom name: vtbd0", "   descr: VirtIO Block Device", "   ident: serial123");
55  
56          Map<String, Triplet<String, String, Long>> result = GeomDiskList.parseGeomDiskList(geom);
57  
58          assertThat(result.size(), is(1));
59          Triplet<String, String, Long> vtbd0 = result.get("vtbd0");
60          assertThat(vtbd0.getA(), is("VirtIO Block Device"));
61          assertThat(vtbd0.getB(), is("serial123"));
62          assertThat(vtbd0.getC(), is(0L));
63      }
64  
65      @Test
66      void testParseGeomDiskListMissingFieldsDefaultToUnknown() {
67          List<String> geom = Arrays.asList("Geom name: nvd0", "   Mediasize: 1024000000000 (953G)");
68  
69          Map<String, Triplet<String, String, Long>> result = GeomDiskList.parseGeomDiskList(geom);
70  
71          assertThat(result.size(), is(1));
72          Triplet<String, String, Long> nvd0 = result.get("nvd0");
73          assertThat(nvd0.getA(), is(Constants.UNKNOWN));
74          assertThat(nvd0.getB(), is(Constants.UNKNOWN));
75          assertThat(nvd0.getC(), is(1024000000000L));
76      }
77  }
