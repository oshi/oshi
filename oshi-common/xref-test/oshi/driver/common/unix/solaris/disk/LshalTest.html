5   package oshi.driver.common.unix.solaris.disk;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.anEmptyMap;
9   import static org.hamcrest.Matchers.is;
10  
11  import java.util.Arrays;
12  import java.util.Collections;
13  import java.util.List;
14  import java.util.Map;
15  
16  import org.junit.jupiter.api.Test;
17  
18  class LshalTest {
19  
20      @Test
21      void testParseLshalTypicalOutput() {
22          List<String> lshal = Arrays.asList("udi = '/org/freedesktop/Hal/devices/storage_serial_VBOX_HARDDISK_VB12345'",
23                  "  block.major = 102  (0x66)  (int)", "  block.minor = 0  (0x0)  (int)",
24                  "  info.category = 'storage'  (string)", "",
25                  "udi = '/org/freedesktop/Hal/devices/storage_serial_disk1'", "  block.major = 91  (0x5b)  (int)",
26                  "  block.minor = 1  (0x1)  (int)");
27  
28          Map<String, Integer> result = Lshal.parseLshal(lshal);
29  
30          assertThat(result.size(), is(2));
31          assertThat(result.get("storage_serial_VBOX_HARDDISK_VB12345"), is(102));
32          assertThat(result.get("storage_serial_disk1"), is(91));
33      }
34  
35      @Test
36      void testParseLshalEmptyInput() {
37          Map<String, Integer> result = Lshal.parseLshal(Collections.emptyList());
38          assertThat(result, is(anEmptyMap()));
39      }
40  
41      @Test
42      void testParseLshalNoBlockMajorLine() {
43          List<String> lshal = Arrays.asList("udi = '/org/freedesktop/Hal/devices/computer'",
44                  "  info.category = 'computer'  (string)", "  info.product = 'Computer'  (string)");
45  
46          Map<String, Integer> result = Lshal.parseLshal(lshal);
47          assertThat(result, is(anEmptyMap()));
48      }
49  }
