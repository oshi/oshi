22      @Test
23      void testParsePrtvtocTypicalOutput() {
24          // Typical prtvtoc output for a Solaris disk
25          List<String> prtvtoc = Arrays.asList("* /dev/dsk/c1t0d0s0 partition map", "*", "* Dimensions:",
26                  "*     512 bytes/sector", "*      63 sectors/track", "*      16 tracks/cylinder",
27                  "*    1008 sectors/cylinder", "*   16384 cylinders", "*   16382 accessible cylinders", "*", "* Flags:",
28                  "*   1: unmountable", "*  10: read-only", "*", "* Volume Name: MyVolume", "*",
29                  "*          First     Sector    Last",
30                  "* Partition  Tag  Flags  Sector     Count    Sector  Mount Directory",
31                  "       0      2    00       1008   14329440   14330447   /",
32                  "       1      3    01    14330448    2097648   16428095",
33                  "       2      5    00           0   16515072   16515071",
34                  "       3      7    00    16428096      86976   16515071   /var");
35  
36          List<HWPartition> result = Prtvtoc.parsePrtvtoc(prtvtoc, "c1t0d0", 102);
37  
38          // Partition 2 (backup/whole disk) is always skipped
39          assertThat(result.size(), is(3));
40  
41          // Partition 0: tag=2 (root), flags=00 (wm)
42          HWPartition p0 = result.get(0);
43          assertThat(p0.getIdentification(), is("c1t0d0s0"));
44          assertThat(p0.getName(), is("root"));
45          assertThat(p0.getType(), is("wm"));
46          assertThat(p0.getSize(), is(512L * 14329440L));
47          assertThat(p0.getMajor(), is(102));
48          assertThat(p0.getMinor(), is(0));
49          assertThat(p0.getMountPoint(), is("/"));
50          assertThat(p0.getLabel(), is("MyVolume"));
51  
52          // Partition 1: tag=3 (swap), flags=01 (wu)
53          HWPartition p1 = result.get(1);
54          assertThat(p1.getIdentification(), is("c1t0d0s1"));
55          assertThat(p1.getName(), is("swap"));
56          assertThat(p1.getType(), is("wu"));
57          assertThat(p1.getSize(), is(512L * 2097648L));
58          assertThat(p1.getMajor(), is(102));
59          assertThat(p1.getMinor(), is(1));
60          assertThat(p1.getMountPoint(), is(""));
61  
62          // Partition 3: tag=7 (var), flags=00 (wm)
63          HWPartition p3 = result.get(2);
64          assertThat(p3.getIdentification(), is("c1t0d0s3"));
65          assertThat(p3.getName(), is("var"));
66          assertThat(p3.getType(), is("wm"));
67          assertThat(p3.getSize(), is(512L * 86976L));
68          assertThat(p3.getMajor(), is(102));
69          assertThat(p3.getMinor(), is(3));
70          assertThat(p3.getMountPoint(), is("/var"));
71      }

73      @Test
74      void testParsePrtvtocTagMappings() {
75          // Test various tag values to verify name mappings
76          List<String> prtvtoc = Arrays.asList("* header line", "*     512 bytes/sector",
77                  "       0      1    00       0   1000    999", "       1     24    00       0   1000    999",
78                  "       3      4    10       0   1000    999", "       4      6    00       0   1000    999",
79                  "       5      8    00       0   1000    999", "       6      9    00       0   1000    999",
80                  "       7     10    00       0   1000    999", "       8     11    00       0   1000    999",
81                  "       9     12    00       0   1000    999", "      10     14    00       0   1000    999",
82                  "      11     15    00       0   1000    999", "      12     99    00       0   1000    999");
83  
84          List<HWPartition> result = Prtvtoc.parsePrtvtoc(prtvtoc, "c0d0", 10);
85  
86          assertThat(result.size(), is(12));
87          // tag 0x01 -> boot
88          assertThat(result.get(0).getName(), is("boot"));
89          // tag 0x18 (24) -> boot
90          assertThat(result.get(1).getName(), is("boot"));
91          // tag 0x04 -> usr
92          assertThat(result.get(2).getName(), is("usr"));
93          // tag 0x06 -> stand
94          assertThat(result.get(3).getName(), is("stand"));
95          // tag 0x08 -> home
96          assertThat(result.get(4).getName(), is("home"));
97          // tag 0x09 -> altsctr
98          assertThat(result.get(5).getName(), is("altsctr"));
99          // tag 0x0a (10) -> cache
100         assertThat(result.get(6).getName(), is("cache"));
101         // tag 0x0b (11) -> reserved
102         assertThat(result.get(7).getName(), is("reserved"));
103         // tag 0x0c (12) -> system
104         assertThat(result.get(8).getName(), is("system"));
105         // tag 0x0e (14) -> public region
106         assertThat(result.get(9).getName(), is("public region"));
107         // tag 0x0f (15) -> private region
108         assertThat(result.get(10).getName(), is("private region"));
109         // tag 99 -> unknown
110         assertThat(result.get(11).getName(), is(Constants.UNKNOWN));
111     }

113     @Test
114     void testParsePrtvtocFlagMappings() {
115         // Test flag values: 00=wm, 10=rm, 01=wu, other=ru
116         List<String> prtvtoc = Arrays.asList("* header", "*     512 bytes/sector",
117                 "       0      2    00       0   1000    999", "       1      2    10       0   1000    999",
118                 "       3      2    01       0   1000    999", "       4      2    11       0   1000    999");
119 
120         List<HWPartition> result = Prtvtoc.parsePrtvtoc(prtvtoc, "c0d0", 5);
121 
122         assertThat(result.size(), is(4));
123         assertThat(result.get(0).getType(), is("wm"));
124         assertThat(result.get(1).getType(), is("rm"));
125         assertThat(result.get(2).getType(), is("wu"));
126         assertThat(result.get(3).getType(), is("ru"));
127     }

144         // If bytes/sector is never set, partition lines are ignored

148         List<HWPartition> result = Prtvtoc.parsePrtvtoc(prtvtoc, "c0d0", 10);
149         assertThat(result, is(empty()));
