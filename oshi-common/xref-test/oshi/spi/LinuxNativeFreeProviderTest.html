5   package oshi.spi;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.greaterThanOrEqualTo;
10  import static org.hamcrest.Matchers.is;
11  import static org.hamcrest.Matchers.not;
12  import static org.hamcrest.Matchers.notNullValue;
13  
14  import org.junit.jupiter.api.BeforeAll;
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledOnOs;
17  import org.junit.jupiter.api.condition.OS;
18  
19  import oshi.hardware.CentralProcessor;
20  import oshi.hardware.GlobalMemory;
21  import oshi.hardware.HardwareAbstractionLayer;
22  import oshi.nativefree.SystemInfo;
23  import oshi.software.os.OperatingSystem;
24  
25  @EnabledOnOs(OS.LINUX)
26  class LinuxNativeFreeProviderTest {
27  
28      private static SystemInfo provider;
29  
30      @BeforeAll
31      static void setUp() {
32          provider = new SystemInfo();
33      }
34  
35      @Test
36      void testIsAvailable() {
37          assertThat(provider.isAvailable(), is(true));
38      }
39  
40      @Test
41      void testPriority() {
42          assertThat(provider.getPriority(), is(0));
43      }
44  
45      @Test
46      void testOperatingSystem() {
47          OperatingSystem os = provider.getOperatingSystem();
48          assertThat(os, is(notNullValue()));
49          assertThat(os.getProcessId(), greaterThan(0));
50          assertThat(os.getThreadCount(), greaterThan(0));
51          assertThat(os.getFamily(), is(notNullValue()));
52          assertThat(os.getFamily(), is(not("")));
53      }
54  
55      @Test
56      void testProcessor() {
57          HardwareAbstractionLayer hal = provider.getHardware();
58          CentralProcessor cpu = hal.getProcessor();
59          assertThat(cpu, is(notNullValue()));
60          assertThat(cpu.getLogicalProcessorCount(), greaterThan(0));
61          assertThat(cpu.getPhysicalProcessorCount(), greaterThan(0));
62          assertThat(cpu.getProcessorIdentifier().getName(), is(not("")));
63      }
64  
65      @Test
66      void testMemory() {
67          HardwareAbstractionLayer hal = provider.getHardware();
68          GlobalMemory mem = hal.getMemory();
69          assertThat(mem, is(notNullValue()));
70          assertThat(mem.getTotal(), greaterThan(0L));
71          assertThat(mem.getAvailable(), greaterThanOrEqualTo(0L));
72      }
73  
74      @Test
75      void testFileSystem() {
76          OperatingSystem os = provider.getOperatingSystem();
77          assertThat(os.getFileSystem(), is(notNullValue()));
78          assertThat(os.getFileSystem().getFileStores().size(), greaterThan(0));
79      }
80  }
