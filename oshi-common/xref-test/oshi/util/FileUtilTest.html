46      @Test
47      void testReadFile() {
48          // Write to a temp file
49          Path multilineFile = null;
50          try {
51              multilineFile = Files.createTempFile("oshitest.multiline", null);
52              String s = "Line 1\nLine 2\nThe third line\nLine 4\nLine 5\n";
53              Files.write(multilineFile, s.getBytes(StandardCharsets.UTF_8));
54          } catch (IOException e) {
55              fail("IO Exception creating or writing to temporary multiline file.");
56          }
57  
58          // Try the new temp file
59          List<String> tempFileStrings = FileUtil.readFile(multilineFile.toString());
60          assertThat("Temp file line one mismatch", tempFileStrings.get(0), is("Line 1"));
61          List<String> matchingLines = tempFileStrings.stream().filter(s -> s.startsWith("Line "))
62                  .collect(Collectors.toList());
63          assertThat("Matching lines mismatch", matchingLines.size(), is(4));
64  
65          // Delete the temp file
66          try {
67              Files.deleteIfExists(multilineFile);
68          } catch (IOException e) {
69              fail("IO Exception deleting temporary multiline file.");
70          }
71  
72          // Try file not found on deleted file
73          assertThat("Deleted file should return empty", FileUtil.readFile(multilineFile.toString()), is(empty()));
74      }

79      @Test
80      void testGetFromFile() {
81          // Write to temp file
82          Path integerFile = null;
83          try {
84              integerFile = Files.createTempFile("oshitest.int", null);
85              Files.write(integerFile, "123\n".getBytes(StandardCharsets.UTF_8));
86          } catch (IOException e) {
87              fail("IO Exception creating or writing to temporary integer file.");
88          }
89          assertThat("unsigned long from int", FileUtil.getUnsignedLongFromFile(integerFile.toString()), is(123L));
90          assertThat("long from int", FileUtil.getLongFromFile(integerFile.toString()), is(123L));
91          assertThat("int from int", FileUtil.getIntFromFile(integerFile.toString()), is(123));
92          assertThat("string from int", FileUtil.getStringFromFile(integerFile.toString()), is("123"));
93  
94          // Delete the temp file
95          try {
96              Files.deleteIfExists(integerFile);
97          } catch (IOException e) {
98              fail("IO Exception deleting temporary integer file.");
99          }
100 
101         // Write to temp file
102         Path stringFile = null;
103         try {
104             stringFile = Files.createTempFile("oshitest.str", null);
105             Files.write(stringFile, "foo bar\n".getBytes(StandardCharsets.UTF_8));
106         } catch (IOException e) {
107             fail("IO Exception creating or writing to temporary string file.");
108         }
109 
110         assertThat("unsigned long from string", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
111         assertThat("long from string", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
112         assertThat("int from string", FileUtil.getIntFromFile(stringFile.toString()), is(0));
113         assertThat("string from string", FileUtil.getStringFromFile(stringFile.toString()), is("foo bar"));
114         // Delete the temp file
115         try {
116             Files.deleteIfExists(stringFile);
117         } catch (IOException e) {
118             fail("IO Exception deleting temporary string file.");
119         }
120 
121         // Try file not found on deleted file
122         assertThat("unsigned long from invalid", FileUtil.getUnsignedLongFromFile(stringFile.toString()), is(0L));
123         assertThat("long from invalid", FileUtil.getLongFromFile(stringFile.toString()), is(0L));
124         assertThat("int from invalid", FileUtil.getIntFromFile(stringFile.toString()), is(0));
125         assertThat("string from invalid ", FileUtil.getStringFromFile(stringFile.toString()), is(emptyString()));
126     }

128     @Test
129     void testReadProcIo() {
130         Map<String, String> expected = new LinkedHashMap<>();
131         expected.put("rchar", "124788352");
132         expected.put("wchar", "124802481");
133         expected.put("syscr", "135");
134         expected.put("syscw", "1547");
135         expected.put("read_bytes", "40304640");
136         expected.put("write_bytes", "124780544");
137         expected.put("cancelled_write_bytes", "42");
138         // Write this to a temp file
139         Path procIoFile = null;
140         try {
141             procIoFile = Files.createTempFile("oshitest.procio", null);
142             for (Entry<String, String> e : expected.entrySet()) {
143                 String s = e.getKey() + ": " + e.getValue() + "\n";
144                 Files.write(procIoFile, s.getBytes(StandardCharsets.UTF_8), StandardOpenOption.APPEND);
145             }
146         } catch (IOException e) {
147             fail("IO Exception creating or writing to temporary procIo file.");
148         }
149         // Read into map
150         Map<String, String> actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
151         assertThat("procio size", actual, is(aMapWithSize(expected.size())));
152         for (Entry<String, String> entry : expected.entrySet()) {
153             assertThat("procio entry", actual, hasEntry(entry.getKey(), entry.getValue()));
154         }
155 
156         // Cleanup
157         try {
158             Files.deleteIfExists(procIoFile);
159         } catch (IOException e) {
160             fail("IO Exception deleting temporary procIo file.");
161         }
162 
163         // Test deleted file
164         actual = FileUtil.getKeyValueMapFromFile(procIoFile.toString(), ":");
165         assertThat("procio size", actual, anEmptyMap());
166     }

176     @Test
177     void testReadBytesFromURL() throws IOException {
178         // Create temporary files
179         Path file1 = Files.createTempFile("oshitest.file1", null);
180         Path file2 = Files.createTempFile("oshitest.file2", null);
181         Path file3 = Files.createTempFile("oshitest.file3", null);
182 
183         Files.write(file1, "Same".getBytes(StandardCharsets.UTF_8), StandardOpenOption.WRITE);
184         Files.write(file2, "Same".getBytes(StandardCharsets.UTF_8), StandardOpenOption.WRITE);
185         Files.write(file3, "Different".getBytes(StandardCharsets.UTF_8), StandardOpenOption.WRITE);
186 
187         byte[] bytes1 = FileUtil.readFileAsBytes(file1.toUri().toURL());
188         byte[] bytes2 = FileUtil.readFileAsBytes(file2.toUri().toURL());
189         byte[] bytes3 = FileUtil.readFileAsBytes(file3.toUri().toURL());
190 
191         assertArrayEquals(bytes1, bytes2, "Byte arrays should match");
192         assertFalse(Arrays.equals(bytes1, bytes3), "Byte arrays should not match");
193     }

195     @Test
196     void testReadBinaryFile() {
197         ByteBuffer buff = ByteBuffer.allocate(18);
198         buff.order(ByteOrder.nativeOrder());
199         buff.putLong(123L);
200         buff.putInt(45);
201         buff.putShort((short) 67);
202         buff.put((byte) 89);
203         byte[] arr = new byte[] { 1, 2, 3 };
204         buff.put(arr);
205 
206         // Write to temp file
207         Path binaryFile = null;
208         try {
209             binaryFile = Files.createTempFile("oshitest.binary", null);
210             Files.write(binaryFile, buff.array());
211         } catch (IOException e) {
212             fail("IO Exception creating or writing to temporary binary file.");
213         }
214 
215         // Read from file
216         buff = FileUtil.readAllBytesAsBuffer(binaryFile.toString());
217         assertThat("Buffer size should match bytes written", buff.limit(), is(18));
218         assertThat("Long from buffer should match", FileUtil.readLongFromBuffer(buff), is(123L));
219         assertThat("Int from buffer should match", FileUtil.readIntFromBuffer(buff), is(45));
220         assertThat("Short from buffer should match", FileUtil.readShortFromBuffer(buff), is((short) 67));
221         assertThat("Byte from buffer should match", FileUtil.readByteFromBuffer(buff), is((byte) 89));
222         byte[] array = new byte[3];
223         FileUtil.readByteArrayFromBuffer(buff, array);
224         assertArrayEquals(arr, array, "Byte array from buffer should match");
225         // Test reads past end of file
226         assertThat("Long from buffer at limit should be 0", FileUtil.readLongFromBuffer(buff), is(0L));
227         assertThat("Int from buffer at limit should be 0", FileUtil.readIntFromBuffer(buff), is(0));
228         assertThat("Short from buffer at limit should be 0", FileUtil.readShortFromBuffer(buff), is((short) 0));
229         assertThat("Byte from buffer at limit should be 0", FileUtil.readByteFromBuffer(buff), is((byte) 0));
230         byte[] arr0 = new byte[] { 0, 0, 0 };
231         array = new byte[3];
232         FileUtil.readByteArrayFromBuffer(buff, array);
233         assertArrayEquals(arr0, array, "Byte array from buffer at limit should be all 0s");
234 
235         // Cleanup
236         try {
237             Files.deleteIfExists(binaryFile);
238         } catch (IOException e) {
239             fail("IO Exception deleting temporary procIo file.");
240         }
241     }
