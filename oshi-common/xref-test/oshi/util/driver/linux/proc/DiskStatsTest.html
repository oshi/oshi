26  class DiskStatsTest {
27  
28      @Test
29      void testParseDiskStats() {
30          // /proc/diskstats: major minor name reads reads_merged sectors_read ms_read writes writes_merged ...
31          List<String> diskstats = Arrays.asList("   8       0 sda 100 5 2000 50 200 10 4000 80 0 130 130",
32                  "   8       1 sda1 40 2 800 20 90 3 1200 30 0 45 45");
33          Map<String, Map<IoStat, Long>> map = DiskStats.parseDiskStats(diskstats);
34          assertThat(map.keySet(), containsInAnyOrder("sda", "sda1"));
35          Map<IoStat, Long> sda = map.get("sda");
36          assertThat(sda.get(IoStat.MAJOR), is(8L));
37          assertThat(sda.get(IoStat.READS), is(100L));
38          assertThat(sda.get(IoStat.READS_SECTOR), is(2000L));
39          assertThat(sda.get(IoStat.WRITES), is(200L));
40          assertThat(sda.get(IoStat.WRITES_SECTOR), is(4000L));
41          // NAME is used as the map key, not stored in the per-device map
42          assertThat(sda.containsKey(IoStat.NAME), is(false));
43          assertThat(map.get("sda1").get(IoStat.READS), is(40L));
44      }
45  
46      @Test
47      void testParseDiskStatsEmpty() {
48          assertThat(DiskStats.parseDiskStats(Collections.emptyList()), is(anEmptyMap()));
49      }
50  
51      @Test
52      @EnabledOnOs(OS.LINUX)
53      void testGetDiskStats() {
54          Map<String, Map<IoStat, Long>> map = DiskStats.getDiskStats();
55          assertNotNull(map, "DiskStats should not be null");
56  
57          map.forEach((key, value) -> {
58              assertNotNull(value, "Entry should not have a null map!");
59              assertInstanceOf(EnumMap.class, value, "Value should be enum map!");
60          });
61      }
