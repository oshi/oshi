26  class DmidecodeTest {
27  
28      // Fixture: typical dmidecode -t system output
29      private static final List<String> SYSTEM_OUTPUT = Arrays.asList("# dmidecode 3.2",
30              "Getting SMBIOS data from sysfs.", "SMBIOS 2.7 present.", "", "Handle 0x0001, DMI type 1, 27 bytes",
31              "System Information", "\tManufacturer: Dell Inc.", "\tProduct Name: PowerEdge R720",
32              "\tVersion: Not Specified", "\tSerial Number: ABC1234", "\tUUID: 4C4C4544-0044-4810-8031-B4C04F333132",
33              "\tWake-up Type: Power Switch", "\tSKU Number: Not Specified", "\tFamily: Not Specified");
34  
35      // Fixture: typical dmidecode -t bios output (uppercase BIOS Revision, as seen in real dmidecode)
36      private static final List<String> BIOS_OUTPUT = Arrays.asList("# dmidecode 3.2", "SMBIOS 2.7 present.", "",
37              "Handle 0x0000, DMI type 0, 24 bytes", "BIOS Information", "\tVendor: Dell Inc.", "\tVersion: 2.5.2",
38              "\tRelease Date: 01/20/2014", "\tAddress: 0xF0000", "\tRuntime Size: 64 kB", "\tROM Size: 16 MB",
39              "\tBIOS Revision: 2.5");
40  
41      @Test
42      void testQuerySerialNumber() {
43          assertThat(Dmidecode.querySerialNumber(SYSTEM_OUTPUT), is("ABC1234"));
44      }
45  
46      @Test
47      void testQuerySerialNumberNotFound() {
48          assertThat(Dmidecode.querySerialNumber(Collections.emptyList()), is(nullValue()));
49      }
50  
51      @Test
52      void testQueryUUID() {
53          assertThat(Dmidecode.queryUUID(SYSTEM_OUTPUT), is("4C4C4544-0044-4810-8031-B4C04F333132"));
54      }
55  
56      @Test
57      void testQueryUUIDNotFound() {
58          assertThat(Dmidecode.queryUUID(Collections.emptyList()), is(nullValue()));
59      }
60  
61      @Test
62      void testQueryBiosNameRev() {
63          Pair<String, String> result = Dmidecode.queryBiosNameRev(BIOS_OUTPUT);
64          assertThat(result.getA(), is("SMBIOS 2.7"));
65          assertThat(result.getB(), is("2.5"));
66      }
67  
68      @Test
69      void testQueryBiosNameRevEmpty() {
70          Pair<String, String> result = Dmidecode.queryBiosNameRev(Collections.emptyList());
71          assertThat(result.getA(), is(nullValue()));
72          assertThat(result.getB(), is(nullValue()));
73      }
74  
75      @Test
76      void testQueryBiosNameRevNoRevision() {
77          List<String> noRev = Arrays.asList("SMBIOS 3.0 present.", "Handle 0x0000, DMI type 0, 24 bytes");
78          Pair<String, String> result = Dmidecode.queryBiosNameRev(noRev);
79          assertThat(result.getA(), is("SMBIOS 3.0"));
80          assertThat(result.getB(), is(nullValue()));
81      }
82  
83      @Test
84      void testQueryBiosNameRevLowercaseVariant() {
85          List<String> lowerCase = Arrays.asList("SMBIOS 3.1 present.", "\tbios revision: 1.2");
86          Pair<String, String> result = Dmidecode.queryBiosNameRev(lowerCase);
87          assertThat(result.getA(), is("SMBIOS 3.1"));
88          assertThat(result.getB(), is("1.2"));
89      }
90  
91      @Nested
92      @EnabledOnOs(OS.LINUX)
93      class LiveTests {
94          @Test
95          void testQueries() {
96              assertDoesNotThrow(() -> Dmidecode.querySerialNumber());
97              assertDoesNotThrow(() -> Dmidecode.queryBiosNameRev());
98  
99              final String uuid = Dmidecode.queryUUID();
100             if (uuid != null) {
101                 assertThat("Test Dmidecode queryUUID format", uuid, matchesRegex(TestConstants.UUID_REGEX));
102             }
103         }
104 
105         @Test
106         void testQueryBiosNameRevReturnsPair() {
107             Pair<String, String> result = Dmidecode.queryBiosNameRev();
108             assertThat(result, notNullValue());
109         }
110     }
