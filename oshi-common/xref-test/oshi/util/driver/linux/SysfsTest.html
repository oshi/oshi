5   package oshi.util.driver.linux;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.containsString;
9   import static org.hamcrest.Matchers.is;
10  import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
11  
12  import org.junit.jupiter.api.Test;
13  import org.junit.jupiter.api.condition.EnabledOnOs;
14  import org.junit.jupiter.api.condition.OS;
15  
16  @EnabledOnOs(OS.LINUX)
17  class SysfsTest {
18  
19      @Test
20      void testQueries() {
21          // On most Linux systems (including CI), these sysfs files exist and return non-null
22          // We verify actual values rather than just assertDoesNotThrow
23          assertDoesNotThrow(Sysfs::querySystemVendor);
24          assertDoesNotThrow(Sysfs::queryProductModel);
25          assertDoesNotThrow(Sysfs::queryProductSerial);
26          assertDoesNotThrow(Sysfs::queryUUID);
27          assertDoesNotThrow(Sysfs::queryBoardVendor);
28          assertDoesNotThrow(Sysfs::queryBoardModel);
29          assertDoesNotThrow(Sysfs::queryBoardVersion);
30          assertDoesNotThrow(Sysfs::queryBoardSerial);
31          assertDoesNotThrow(Sysfs::queryBiosVendor);
32          assertDoesNotThrow(Sysfs::queryBiosDescription);
33          assertDoesNotThrow(Sysfs::queryBiosReleaseDate);
34      }
35  
36      @Test
37      void testAtLeastOneSysfsQueryReturnsValue() {
38          // On any Linux system with DMI support (including CI), at least one of these
39          // sysfs queries should return a non-null value. This catches blanket regressions.
40          String vendor = Sysfs.querySystemVendor();
41          String model = Sysfs.queryProductModel();
42          String desc = Sysfs.queryBiosDescription();
43          String date = Sysfs.queryBiosReleaseDate();
44          assertThat("At least one sysfs query should return non-null on Linux",
45                  vendor != null || model != null || desc != null || date != null, is(true));
46      }
47  
48      @Test
49      void testQueryBiosVersion() {
50          Sysfs.queryBiosVersion("");
51          final String biosRevisionSuffix = "biosRevision";
52          final String biosRevision = Sysfs.queryBiosVersion(biosRevisionSuffix);
53          if (biosRevision != null) {
54              assertThat("Test Sysfs queryBiosVersion with biosRevision", biosRevision,
55                      containsString(biosRevisionSuffix));
56          }
57      }
58  }
