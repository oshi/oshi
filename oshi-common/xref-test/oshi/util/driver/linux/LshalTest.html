7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.hamcrest.Matchers.matchesRegex;
10  import static org.hamcrest.Matchers.nullValue;
11  import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
12  
13  import java.util.Arrays;
14  import java.util.Collections;
15  import java.util.List;
16  
17  import org.junit.jupiter.api.Test;
18  
19  import oshi.TestConstants;
20  
21  class LshalTest {
22  
23      // Fixture: lshal output with system hardware info
24      private static final List<String> LSHAL_OUTPUT = Arrays.asList("udi = '/org/freedesktop/Hal/devices/computer'",
25              "  system.hardware.vendor = 'Dell Inc.'  (string)",
26              "  system.hardware.product = 'PowerEdge R720'  (string)", "  system.hardware.serial = 'ABC1234'  (string)",
27              "  system.hardware.uuid = '4C4C4544-0044-4810-8031-B4C04F333132'  (string)",
28              "  system.firmware.vendor = 'Dell Inc.'  (string)");
29  
30      @Test
31      void testQuerySerialNumber() {
32          assertThat(Lshal.querySerialNumber(LSHAL_OUTPUT), is("ABC1234"));
33      }
34  
35      @Test
36      void testQuerySerialNumberNotFound() {
37          assertThat(Lshal.querySerialNumber(Collections.emptyList()), is(nullValue()));
38      }
39  
40      @Test
41      void testQueryUUID() {
42          assertThat(Lshal.queryUUID(LSHAL_OUTPUT), is("4C4C4544-0044-4810-8031-B4C04F333132"));
43      }
44  
45      @Test
46      void testQueryUUIDNotFound() {
47          assertThat(Lshal.queryUUID(Collections.emptyList()), is(nullValue()));
48      }
49  
50      @Test
51      void testLiveQueries() {
52          assertDoesNotThrow(() -> Lshal.querySerialNumber());
53  
54          final String uuid = Lshal.queryUUID();
55          if (uuid != null) {
56              assertThat("Test Lshal queryUUID format", uuid, matchesRegex(TestConstants.UUID_REGEX));
57          }
58      }
59  }
