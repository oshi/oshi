101     @Test
102     void testQueryNetstat() {
103         List<IPConnection> connections = NetStat.queryNetstat(NETSTAT_N);
104         assertThat(connections, hasSize(2));
105         assertThat(connections.get(0).getType(), is("tcp4"));
106         assertThat(connections.get(0).getState(), is(TcpState.ESTABLISHED));
107         assertThat(connections.get(0).getLocalPort(), is(55362));
108         assertThat(connections.get(0).getForeignPort(), is(443));
109         assertThat(connections.get(1).getType(), is("udp4"));
110         assertThat(connections.get(1).getState(), is(TcpState.NONE));
111         assertThat(connections.get(1).getLocalPort(), is(5353));
112     }

161     @Test
162     void testQueryTcpStatsBsd() {
163         TcpStats stats = NetStat.queryTcpStats(TCP_STATS_BSD);
164         assertThat(stats.getConnectionsEstablished(), is(15L));
165         assertThat(stats.getConnectionFailures(), is(5L));
166         assertThat(stats.getConnectionsReset(), is(7L));
167         assertThat(stats.getSegmentsSent(), is(100L));
168         assertThat(stats.getSegmentsReceived(), is(200L));
169         // "42 retransmitted 3 data packet" matches the special-case pattern
170         assertThat(stats.getSegmentsRetransmitted(), is(42L));
171         // 2 bad checksum + 3 bad header offset field
172         assertThat(stats.getInErrors(), is(5L));
173         assertThat(stats.getOutResets(), is(10L));
174     }

176     @Test
177     void testQueryTcpStatsRetransmitSpecialCase() {
178         // Test the "N retransmitted M data packet" pattern in isolation (BSD format)
179         List<String> retransmitLines = Arrays.asList("tcp:", "    99 retransmitted 5 data packet");
180         TcpStats stats = NetStat.queryTcpStats(retransmitLines);
181         // The special-case pattern uses += on segmentsRetransmitted
182         assertThat(stats.getSegmentsRetransmitted(), is(99L));
183     }

214     @Test
215     void testQueryNetstatIpv6Addresses() {
216         List<IPConnection> connections = NetStat.queryNetstat(NETSTAT_IPV6);
217         assertThat(connections, hasSize(3));
218         // First: 2001:db8::1 port 443
219         assertThat(connections.get(0).getType(), is("tcp6"));
220         assertThat(connections.get(0).getLocalPort(), is(443));
221         assertThat(connections.get(0).getForeignPort(), is(54321));
222         assertThat(connections.get(0).getState(), is(TcpState.ESTABLISHED));
223         // Second: fe80::1: port 8080 (trailing colon in IP, tests parseIP fallback)
224         assertThat(connections.get(1).getType(), is("tcp6"));
225         assertThat(connections.get(1).getLocalPort(), is(8080));
226         assertThat(connections.get(1).getForeignPort(), is(9090));
227         assertThat(connections.get(1).getState(), is(TcpState.TIME_WAIT));
228         // Third: ::1 port 55364
229         assertThat(connections.get(2).getType(), is("tcp6"));
230         assertThat(connections.get(2).getLocalPort(), is(55364));
231         assertThat(connections.get(2).getForeignPort(), is(8080));
232         assertThat(connections.get(2).getState(), is(TcpState.ESTABLISHED));
233     }

256         @Test
257         void testQueryStatsLinux() {
258             TcpStats tcpStats = NetStat.queryTcpStats("netstat -st4");
259             assertThat("tcp connections must be nonnegative", tcpStats.getConnectionsEstablished(),
260                     greaterThanOrEqualTo(0L));
261             UdpStats udp4Stats = NetStat.queryUdpStats("netstat -su4");
262             assertThat("udp4 datagrams sent must be nonnegative", udp4Stats.getDatagramsSent(),
263                     greaterThanOrEqualTo(0L));
264         }
