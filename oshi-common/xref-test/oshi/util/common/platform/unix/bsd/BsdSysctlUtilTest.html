5   package oshi.util.common.platform.unix.bsd;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.not;
11  
12  import java.util.EnumSet;
13  import java.util.Set;
14  
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledIf;
17  
18  import oshi.util.PlatformEnum;
19  
20  /**
21   * Tests the command-line {@code sysctl -n} implementation shared across all BSD-family operating systems. Enabled on
22   * every platform that ships a {@code sysctl} command so both the success and failure paths of all three overloads are
23   * exercised.
24   */
25  class BsdSysctlUtilTest {
26  
27      /** Platforms that provide a {@code sysctl} command whose {@code -n} flag prints a bare value. */
28      private static final Set<PlatformEnum> SYSCTL_PLATFORMS = EnumSet.of(PlatformEnum.MACOS, PlatformEnum.FREEBSD,
29              PlatformEnum.OPENBSD, PlatformEnum.NETBSD, PlatformEnum.DRAGONFLYBSD);
30  
31      /**
32       * Condition method for {@link EnabledIf}: true when the current platform ships {@code sysctl}.
33       *
34       * @return whether the sysctl command-line tests should run
35       */
36      static boolean hasSysctlCommand() {
37          return SYSCTL_PLATFORMS.contains(PlatformEnum.getCurrentPlatform());
38      }
39  
40      @Test
41      @EnabledIf("hasSysctlCommand")
42      void testSysctlStringSuccess() {
43          // kern.ostype is present on macOS and every BSD; its value is the kernel name.
44          assertThat("kern.ostype should not be empty", BsdSysctlUtil.sysctl("kern.ostype", ""), is(not("")));
45      }
46  
47      @Test
48      @EnabledIf("hasSysctlCommand")
49      void testSysctlStringFailureReturnsDefault() {
50          assertThat(BsdSysctlUtil.sysctl("invalid.bogus.name", "mydefault"), is("mydefault"));
51      }
52  
53      @Test
54      @EnabledIf("hasSysctlCommand")
55      void testSysctlIntSuccess() {
56          // hw.pagesize is a positive int on macOS and every BSD.
57          assertThat("hw.pagesize should be positive", BsdSysctlUtil.sysctl("hw.pagesize", 0), greaterThan(0));
58      }
59  
60      @Test
61      @EnabledIf("hasSysctlCommand")
62      void testSysctlIntFailureReturnsDefault() {
63          assertThat(BsdSysctlUtil.sysctl("invalid.bogus.name", -99), is(-99));
64      }
65  
66      @Test
67      @EnabledIf("hasSysctlCommand")
68      void testSysctlLongSuccess() {
69          // hw.pagesize also parses as a positive long.
70          assertThat("hw.pagesize should be positive", BsdSysctlUtil.sysctl("hw.pagesize", 0L), greaterThan(0L));
71      }
72  
73      @Test
74      @EnabledIf("hasSysctlCommand")
75      void testSysctlLongFailureReturnsDefault() {
76          assertThat(BsdSysctlUtil.sysctl("invalid.bogus.name", -123L), is(-123L));
77      }
78  }
