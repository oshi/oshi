5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants and property enums for WMI class {@code Win32_PhysicalMemory}.
11   */
12  @ThreadSafe
13  public class Win32PhysicalMemory {
14  
15      /**
16       * The WMI class name.
17       */
18      public static final String WIN32_PHYSICAL_MEMORY = "Win32_PhysicalMemory";
19  
20      /**
21       * Physical Memory properties for Win10 and later.
22       */
23      public enum PhysicalMemoryProperty {
24          /** BANKLABEL property. */
25          BANKLABEL,
26          /** CAPACITY property. */
27          CAPACITY,
28          /** SPEED property. */
29          SPEED,
30          /** MANUFACTURER property. */
31          MANUFACTURER,
32          /** PARTNUMBER property. */
33          PARTNUMBER,
34          /** SMBIOSMEMORYTYPE property. */
35          SMBIOSMEMORYTYPE,
36          /** SERIALNUMBER property. */
37          SERIALNUMBER
38      }
39  
40      /**
41       * Physical Memory properties for Win8 and earlier.
42       */
43      public enum PhysicalMemoryPropertyWin8 {
44          /** BANKLABEL property. */
45          BANKLABEL,
46          /** CAPACITY property. */
47          CAPACITY,
48          /** SPEED property. */
49          SPEED,
50          /** MANUFACTURER property. */
51          MANUFACTURER,
52          /** MEMORYTYPE property. */
53          MEMORYTYPE,
54          /** PARTNUMBER property. */
55          PARTNUMBER,
56          /** SERIALNUMBER property. */
57          SERIALNUMBER
58      }
59  
60      /**
61       * Constructor.
62       */
63      protected Win32PhysicalMemory() {
64      }
65  
66      /**
67       * Queries physical memory information.
68       *
69       * @param h An instantiated {@link WmiQueryExecutor}.
70       * @return Physical memory information.
71       */
72      public static WmiResult<PhysicalMemoryProperty> queryPhysicalMemory(WmiQueryExecutor h) {
73          WmiQuery<PhysicalMemoryProperty> query = new WmiQuery<>(WIN32_PHYSICAL_MEMORY, PhysicalMemoryProperty.class);
74          return h.queryWMI(query);
75      }
76  
77      /**
78       * Queries physical memory information (Win8+ properties).
79       *
80       * @param h An instantiated {@link WmiQueryExecutor}.
81       * @return Physical memory information with Win8+ properties.
82       */
83      public static WmiResult<PhysicalMemoryPropertyWin8> queryPhysicalMemoryWin8(WmiQueryExecutor h) {
84          WmiQuery<PhysicalMemoryPropertyWin8> query = new WmiQuery<>(WIN32_PHYSICAL_MEMORY,
85                  PhysicalMemoryPropertyWin8.class);
86          return h.queryWMI(query);
87      }
88  }
