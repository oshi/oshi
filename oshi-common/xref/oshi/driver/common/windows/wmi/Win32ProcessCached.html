5   package oshi.driver.common.windows.wmi;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   import java.util.concurrent.locks.ReentrantLock;
10  
11  import oshi.annotation.concurrent.GuardedBy;
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.driver.common.windows.wmi.Win32Process.CommandLineProperty;
14  import oshi.util.tuples.Pair;
15  
16  /**
17   * Base for the cached {@code Win32_Process} command-line query. Holds the command-line cache and the lookup logic; the
18   * per-backend (JNA/FFM) WMI query is supplied by {@link #queryCommandLines()}.
19   */
20  @ThreadSafe
21  public abstract class Win32ProcessCached {
22  
23      @GuardedBy("commandLineCacheLock")
24      private final Map<Integer, Pair<Long, String>> commandLineCache = new HashMap<>();
25      private final ReentrantLock commandLineCacheLock = new ReentrantLock();
26  
27      /**
28       * Default constructor.
29       */
30      protected Win32ProcessCached() {
31      }
32  
33      /**
34       * Queries the command lines of all running processes via WMI.
35       *
36       * @return the {@code Win32_Process} command-line WMI result
37       */
38      protected abstract WmiResult<CommandLineProperty> queryCommandLines();
39  
40      /**
41       * Gets the process command line, while also querying and caching command lines for all running processes if the
42       * specified process is not in the cache.
43       *
44       * @param processId The process ID for which to return the command line.
45       * @param startTime The start time of the process, in milliseconds since the 1970 epoch.
46       * @return The command line of the specified process.
47       */
48      public String getCommandLine(int processId, long startTime) {
49          commandLineCacheLock.lock();
50          try {
51              Pair<Long, String> pair = commandLineCache.get(processId);
52              if (pair != null && startTime < pair.getA()) {
53                  return pair.getB();
54              } else {
55                  long now = System.currentTimeMillis();
56                  WmiResult<CommandLineProperty> commandLineAllProcs = queryCommandLines();
57                  if (commandLineCache.size() > commandLineAllProcs.getResultCount() * 2) {
58                      commandLineCache.clear();
59                  }
60                  String result = "";
61                  for (int i = 0; i < commandLineAllProcs.getResultCount(); i++) {
62                      int pid = WmiUtil.getUint32(commandLineAllProcs, CommandLineProperty.PROCESSID, i);
63                      String cl = WmiUtil.getString(commandLineAllProcs, CommandLineProperty.COMMANDLINE, i);
64                      commandLineCache.put(pid, new Pair<>(now, cl));
65                      if (pid == processId) {
66                          result = cl;
67                      }
68                  }
69                  return result;
70              }
71          } finally {
72              commandLineCacheLock.unlock();
73          }
74      }
75  }
