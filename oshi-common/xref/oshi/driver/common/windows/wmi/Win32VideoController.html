5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants and property enum for WMI class {@code Win32_VideoController}.
11   */
12  @ThreadSafe
13  public class Win32VideoController {
14  
15      /**
16       * The WMI class name.
17       */
18      public static final String WIN32_VIDEO_CONTROLLER = "Win32_VideoController";
19  
20      /**
21       * Video Controller properties.
22       */
23      public enum VideoControllerProperty {
24          /** ADAPTERCOMPATIBILITY property. */
25          ADAPTERCOMPATIBILITY,
26          /** ADAPTERRAM property. */
27          ADAPTERRAM,
28          /** CONFIGMANAGERERRORCODE property. */
29          CONFIGMANAGERERRORCODE,
30          /** DRIVERVERSION property. */
31          DRIVERVERSION,
32          /** NAME property. */
33          NAME,
34          /** PNPDEVICEID property. */
35          PNPDEVICEID
36      }
37  
38      /**
39       * Constructor.
40       */
41      protected Win32VideoController() {
42      }
43  
44      /**
45       * Queries video controller information.
46       *
47       * @param h An instantiated {@link WmiQueryExecutor}.
48       * @return Video controller information.
49       */
50      public static WmiResult<VideoControllerProperty> queryVideoController(WmiQueryExecutor h) {
51          WmiQuery<VideoControllerProperty> query = new WmiQuery<>(WIN32_VIDEO_CONTROLLER, VideoControllerProperty.class);
52          return h.queryWMI(query);
53      }
54  }
