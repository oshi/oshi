5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants, property enums, and WHERE clause builders for LibreHardwareMonitor WMI sensor data.
11   *
12   * <p>
13   * LHM publishes sensor data to {@code ROOT\LibreHardwareMonitor} when it is running. This class provides the shared
14   * constants and query builders for the {@code Sensor} and {@code Hardware} tables.
15   */
16  @ThreadSafe
17  public class LhmSensor {
18  
19      /**
20       * The WMI namespace for LibreHardwareMonitor.
21       */
22      public static final String LHM_NAMESPACE = "ROOT\\LibreHardwareMonitor";
23  
24      /**
25       * The WMI class name for sensors.
26       */
27      public static final String SENSOR = "Sensor";
28  
29      /**
30       * The WMI class name for hardware.
31       */
32      public static final String HARDWARE = "Hardware";
33  
34      /**
35       * Sensor properties returned by LHM WMI queries.
36       */
37      public enum LhmSensorProperty {
38          NAME, VALUE, PARENT;
39      }
40  
41      /**
42       * LHM Hardware properties.
43       */
44      public enum LhmHardwareProperty {
45          IDENTIFIER, NAME;
46      }
47  
48      protected LhmSensor() {
49      }
50  
51      /**
52       * Builds the WMI class name with WHERE clause for sensor queries.
53       *
54       * @param parent     the LHM hardware identifier (e.g. {@code /gpu-nvidia/0})
55       * @param sensorType the sensor type string (e.g. {@code "Load"}, {@code "SmallData"})
56       * @return the WMI class name with WHERE clause
57       */
58      public static String buildSensorWmiClassNameWithWhere(String parent, String sensorType) {
59          StringBuilder sb = new StringBuilder(SENSOR);
60          sb.append(" WHERE Parent=\"").append(parent);
61          sb.append("\" AND SensorType=\"").append(sensorType).append('"');
62          return sb.toString();
63      }
64  
65      /**
66       * Builds the WMI class name with WHERE clause for GPU hardware queries.
67       *
68       * @return the WMI class name with WHERE clause filtering GPU hardware types
69       */
70      public static String buildGpuHardwareWmiClassName() {
71          return HARDWARE + " WHERE HardwareType=\"GpuNvidia\" OR HardwareType=\"GpuAmd\" OR HardwareType=\"GpuIntel\"";
72      }
73  }
