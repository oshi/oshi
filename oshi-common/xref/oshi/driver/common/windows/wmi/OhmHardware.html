5   package oshi.driver.common.windows.wmi;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants, property enum, and WHERE clause builder for Open Hardware Monitor WMI Hardware data.
11   */
12  @ThreadSafe
13  public class OhmHardware {
14  
15      /**
16       * The WMI namespace for Open Hardware Monitor.
17       */
18      public static final String OHM_NAMESPACE = "ROOT\\OpenHardwareMonitor";
19  
20      /**
21       * The WMI class name for hardware.
22       */
23      public static final String HARDWARE = "Hardware";
24  
25      /**
26       * HW Identifier Property.
27       */
28      public enum IdentifierProperty {
29          IDENTIFIER;
30      }
31  
32      protected OhmHardware() {
33      }
34  
35      /**
36       * Builds the WMI class name with WHERE clause for hardware identifier queries.
37       *
38       * @param typeToQuery which type to filter based on
39       * @param typeName    the name of the type
40       * @return the WMI class name with WHERE clause
41       */
42      public static String buildHardwareWmiClassNameWithWhere(String typeToQuery, String typeName) {
43          StringBuilder sb = new StringBuilder(HARDWARE);
44          sb.append(" WHERE ").append(typeToQuery).append("Type=\"").append(typeName).append('"');
45          return sb.toString();
46      }
47  }
