5   package oshi.driver.common.windows.wmi;
6   
7   import java.util.Collection;
8   import java.util.Set;
9   import java.util.stream.Collectors;
10  
11  import oshi.annotation.concurrent.ThreadSafe;
12  
13  /**
14   * Constants, property enums, and WHERE clause builder for WMI class {@code Win32_Process}.
15   */
16  @ThreadSafe
17  public class Win32Process {
18  
19      /**
20       * The WMI class name.
21       */
22      public static final String WIN32_PROCESS = "Win32_Process";
23  
24      /**
25       * Process command lines.
26       */
27      public enum CommandLineProperty {
28          /** PROCESSID property. */
29          PROCESSID,
30          /** COMMANDLINE property. */
31          COMMANDLINE
32      }
33  
34      /**
35       * Process properties accessible from WTSEnumerateProcesses in Vista+.
36       */
37      public enum ProcessXPProperty {
38          /** PROCESSID property. */
39          PROCESSID,
40          /** NAME property. */
41          NAME,
42          /** KERNELMODETIME property. */
43          KERNELMODETIME,
44          /** USERMODETIME property. */
45          USERMODETIME,
46          /** THREADCOUNT property. */
47          THREADCOUNT,
48          /** PAGEFILEUSAGE property. */
49          PAGEFILEUSAGE,
50          /** HANDLECOUNT property. */
51          HANDLECOUNT,
52          /** EXECUTABLEPATH property. */
53          EXECUTABLEPATH
54      }
55  
56      /**
57       * Constructor.
58       */
59      protected Win32Process() {
60      }
61  
62      /**
63       * Builds the WMI class name with optional WHERE clause filtering by process IDs.
64       *
65       * @param pids Process IDs to filter, or {@code null} to query all processes
66       * @return the WMI class name with WHERE clause appended if pids is non-null
67       */
68      public static String buildWmiClassNameWithPidFilter(Collection<Integer> pids) {
69          if (pids == null || pids.isEmpty()) {
70              return WIN32_PROCESS;
71          }
72          return WIN32_PROCESS + " WHERE ProcessID="
73                  + pids.stream().map(String::valueOf).collect(Collectors.joining(" OR PROCESSID="));
74      }
75  
76      /**
77       * Returns process command lines.
78       *
79       * @param h           An instantiated {@link WmiQueryExecutor}.
80       * @param pidsToQuery Process IDs to query for command lines. Pass {@code null} to query all processes.
81       * @return A {@link WmiResult} containing process IDs and command lines.
82       */
83      public static WmiResult<CommandLineProperty> queryCommandLines(WmiQueryExecutor h, Set<Integer> pidsToQuery) {
84          WmiQuery<CommandLineProperty> commandLineQuery = new WmiQuery<>(buildWmiClassNameWithPidFilter(pidsToQuery),
85                  CommandLineProperty.class);
86          return h.queryWMI(commandLineQuery);
87      }
88  
89      /**
90       * Returns process info.
91       *
92       * @param h    An instantiated {@link WmiQueryExecutor}.
93       * @param pids Process IDs to query.
94       * @return Information on the provided processes.
95       */
96      public static WmiResult<ProcessXPProperty> queryProcesses(WmiQueryExecutor h, Collection<Integer> pids) {
97          WmiQuery<ProcessXPProperty> processQueryXP = new WmiQuery<>(buildWmiClassNameWithPidFilter(pids),
98                  ProcessXPProperty.class);
99          return h.queryWMI(processQueryXP);
100     }
101 }
