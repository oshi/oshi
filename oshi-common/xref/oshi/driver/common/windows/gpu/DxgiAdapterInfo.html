5   package oshi.driver.common.windows.gpu;
6   
7   import java.util.Locale;
8   
9   /**
10   * Immutable snapshot of a DXGI adapter's identity and dedicated video memory.
11   *
12   * <p>
13   * {@code DedicatedVideoMemory} from {@code DXGI_ADAPTER_DESC} is the authoritative Windows API source for dedicated GPU
14   * memory; it is not subject to the 2 GiB cap that affects the 32-bit registry value
15   * {@code HardwareInformation.MemorySize}.
16   */
17  public final class DxgiAdapterInfo {
18  
19      private final String description;
20      private final int vendorId;
21      private final int deviceId;
22      private final long dedicatedVideoMemory;
23      private final int luidLowPart;
24      private final int luidHighPart;
25  
26      /**
27       * @param description          adapter description string from {@code DXGI_ADAPTER_DESC.Description}
28       * @param vendorId             PCI vendor ID
29       * @param deviceId             PCI device ID
30       * @param dedicatedVideoMemory dedicated video memory in bytes
31       * @param luidLowPart          low 32 bits of the adapter LUID
32       * @param luidHighPart         high 32 bits of the adapter LUID
33       */
34      public DxgiAdapterInfo(String description, int vendorId, int deviceId, long dedicatedVideoMemory, int luidLowPart,
35              int luidHighPart) {
36          this.description = description;
37          this.vendorId = vendorId;
38          this.deviceId = deviceId;
39          this.dedicatedVideoMemory = dedicatedVideoMemory;
40          this.luidLowPart = luidLowPart;
41          this.luidHighPart = luidHighPart;
42      }
43  
44      public String getDescription() {
45          return description;
46      }
47  
48      public int getVendorId() {
49          return vendorId;
50      }
51  
52      public int getDeviceId() {
53          return deviceId;
54      }
55  
56      public long getDedicatedVideoMemory() {
57          return dedicatedVideoMemory;
58      }
59  
60      public int getLuidLowPart() {
61          return luidLowPart;
62      }
63  
64      public int getLuidHighPart() {
65          return luidHighPart;
66      }
67  
68      @Override
69      public String toString() {
70          return String.format(Locale.ROOT,
71                  "DxgiAdapterInfo{desc='%s', vendor=0x%04X, device=0x%04X, vram=%d, luid=0x%08X_0x%08X}", description,
72                  vendorId, deviceId, dedicatedVideoMemory, luidHighPart, luidLowPart);
73      }
74  }
