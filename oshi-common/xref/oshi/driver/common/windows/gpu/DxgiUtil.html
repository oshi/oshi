22      /**
23       * Finds the best-matching DXGI adapter for a given vendor ID, device ID, and adapter name.
24       *
25       * <p>
26       * Matching priority:
27       * <ol>
28       * <li>Vendor ID + Device ID (exact, both non-zero)</li>
29       * <li>Normalized name match (case-insensitive, ignoring {@code (R)}, {@code (TM)}, extra spaces)</li>
30       * </ol>
31       *
32       * @param adapters    list of adapters
33       * @param vendorId    PCI vendor ID parsed from the registry key (0 if unknown)
34       * @param deviceId    PCI device ID parsed from the registry key (0 if unknown)
35       * @param adapterName adapter name from the registry {@code DriverDesc} value
36       * @return best-matching adapter, or {@code null}
37       */
38      public static DxgiAdapterInfo findMatch(List<DxgiAdapterInfo> adapters, int vendorId, int deviceId,
39              String adapterName) {
40          if (vendorId != 0 && deviceId != 0) {
41              for (DxgiAdapterInfo a : adapters) {
42                  if (a.getVendorId() == vendorId && a.getDeviceId() == deviceId) {
43                      return a;
44                  }
45              }
46          }
47          if (adapterName != null && !adapterName.isEmpty()) {
48              String norm = normalizeName(adapterName);
49              for (DxgiAdapterInfo a : adapters) {
50                  if (normalizeName(a.getDescription()).equals(norm)) {
51                      return a;
52                  }
53              }
54          }
55          return null;
56      }

58      /**
59       * Converts a registry value (REG_QWORD as Long, REG_DWORD as Integer, or REG_BINARY as byte[]) to a VRAM size in
60       * bytes. REG_BINARY is interpreted as little-endian.
61       *
62       * @param value the registry value object
63       * @return the VRAM size in bytes, or 0 if the value type is unrecognised
64       */
65      public static long registryValueToVram(Object value) {
66          if (value instanceof Long) {
67              return (long) value;
68          } else if (value instanceof Integer) {
69              return Integer.toUnsignedLong((int) value);
70          } else if (value instanceof byte[]) {
71              byte[] bytes = (byte[]) value;
72              long total = 0L;
73              int size = Math.min(bytes.length, 8);
74              for (int i = 0; i < size; i++) {
75                  total = total << 8 | bytes[size - i - 1] & 0xff;
76              }
77              return total;
78          }
79          return 0L;
80      }

82      /**
83       * Normalizes an adapter name for fuzzy matching: lower-case, strips {@code (R)}/{@code (TM)}, collapses whitespace.
84       *
85       * @param name the raw adapter name, may be {@code null}
86       * @return normalized name, never {@code null}
87       */
88      public static String normalizeName(String name) {
89          if (name == null) {
90              return "";
91          }
92          return name.toLowerCase(Locale.ROOT).replace("(r)", "").replace("(tm)", "").replaceAll("\\s+", " ").trim();
93      }

95      /**
96       * Builds the PDH LUID instance prefix for the given DXGI adapter. The prefix has the form
97       * {@code luid_0xHHHHHHHH_0xLLLLLLLL_phys_0}.
98       *
99       * @param adapter the DXGI adapter info containing the LUID
100      * @return PDH LUID instance prefix string, or empty string if the LUID is zero
101      */
102     public static String buildLuidPrefix(DxgiAdapterInfo adapter) {
103         int low = adapter.getLuidLowPart();
104         int high = adapter.getLuidHighPart();
105         if (low == 0 && high == 0) {
106             return "";
107         }
108         return String.format(Locale.ROOT, "luid_0x%08x_0x%08x_phys_0", high, low);
109     }

111     /**
112      * Parses the PCI bus number from a Windows registry {@code LocationInformation} string of the form
113      * {@code "PCI bus N, device N, function N"}.
114      *
115      * @param locationInfo the LocationInformation registry value
116      * @return PCI bus number, or -1 if not parseable
117      */
118     public static int parsePciBusNumber(String locationInfo) {
119         if (locationInfo == null || locationInfo.isEmpty()) {
120             return -1;
121         }
122         String lower = locationInfo.toLowerCase(Locale.ROOT);
123         int busIdx = lower.indexOf("pci bus ");
124         if (busIdx < 0) {
125             return -1;
126         }
127         int start = busIdx + 8;
128         int end = lower.indexOf(',', start);
129         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
130         return ParseUtil.parseIntOrDefault(numStr, -1);
131     }

133     /**
134      * Parses the PCI device number from a Windows registry {@code LocationInformation} string.
135      *
136      * @param locationInfo the LocationInformation registry value
137      * @return PCI device number, or -1 if not parseable
138      */

140         if (locationInfo == null || locationInfo.isEmpty()) {
141             return -1;
142         }
143         String lower = locationInfo.toLowerCase(Locale.ROOT);
144         int devIdx = lower.indexOf("device ");
145         if (devIdx < 0) {
146             return -1;
147         }
148         int start = devIdx + 7;
149         int end = lower.indexOf(',', start);
150         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
151         return ParseUtil.parseIntOrDefault(numStr, -1);
152     }

154     /**
155      * Parses the PCI function number from a Windows registry {@code LocationInformation} string.
156      *
157      * @param locationInfo the LocationInformation registry value
158      * @return PCI function number, or -1 if not parseable
159      */

161         if (locationInfo == null || locationInfo.isEmpty()) {
162             return -1;
163         }
164         String lower = locationInfo.toLowerCase(Locale.ROOT);
165         int fnIdx = lower.indexOf("function ");
166         if (fnIdx < 0) {
167             return -1;
168         }
169         int start = fnIdx + 9;
170         int end = lower.indexOf(',', start);
171         String numStr = end > start ? locationInfo.substring(start, end).trim() : locationInfo.substring(start).trim();
172         return ParseUtil.parseIntOrDefault(numStr, -1);
173     }

175     /**
176      * Builds a PCI bus ID string in {@code "0000:BB:DD.F"} format from a Windows registry {@code LocationInformation}
177      * string.
178      *
179      * @param locationInfo the LocationInformation registry value
180      * @return PCI bus ID string, or empty string if not parseable
181      */
182     public static String buildPciBusId(String locationInfo) {
183         int bus = parsePciBusNumber(locationInfo);
184         int device = parsePciDevice(locationInfo);
185         int function = parsePciFunction(locationInfo);
186         if (bus < 0 || device < 0 || function < 0) {
187             return "";
188         }
189         return String.format(Locale.ROOT, "0000:%02x:%02x.%x", bus, device, function);
190     }
