18  /**
19   * Common logic for building process performance data maps from registry or performance counter results. Callers
20   * (JNA/FFM variants) supply the platform-specific pre-fetched registry or performance-counter data.
21   */
22  @ThreadSafe
23  public final class ProcessPerformanceData {
24  
25      /**
26       * The performance object name for process counters.
27       */
28      public static final String PROCESS = "Process";
29  
30      private ProcessPerformanceData() {
31      }
32  
33      /**
34       * Builds a process map from registry performance data that has already been read.
35       *
36       * @param pids        An optional collection of process IDs to filter the list to. May be null for no filtering.
37       * @param processData The raw registry data triplet (instance maps, perfTime100nSec, now in ms)
38       * @return A map with Process ID as the key and a {@link ProcessPerfCounterBlock} object populated with performance
39       *         counter information, or null if processData is null.
40       */
41      public static Map<Integer, ProcessPerfCounterBlock> buildProcessMapFromRegistry(Collection<Integer> pids,
42              Triplet<List<Map<ProcessPerformanceProperty, Object>>, Long, Long> processData) {
43          if (processData == null) {
44              return null;
45          }
46          List<Map<ProcessPerformanceProperty, Object>> processInstanceMaps = processData.getA();
47          long now = processData.getC(); // 1970 epoch
48  
49          Map<Integer, ProcessPerfCounterBlock> processMap = new HashMap<>();
50          for (Map<ProcessPerformanceProperty, Object> processInstanceMap : processInstanceMaps) {
51              int pid = ((Integer) processInstanceMap.get(ProcessPerformanceProperty.IDPROCESS)).intValue();
52              String name = (String) processInstanceMap.get(ProcessPerformanceProperty.NAME);
53              if ((pids == null || pids.contains(pid)) && !"_Total".equals(name)) {
54                  long ctime = (Long) processInstanceMap.get(ProcessPerformanceProperty.ELAPSEDTIME);
55                  if (ctime > now) {
56                      ctime = ParseUtil.filetimeToUtcMs(ctime, false);
57                  }
58                  long upTime = now - ctime;
59                  if (upTime < 1L) {
60                      upTime = 1L;
61                  }
62                  processMap.put(pid, new ProcessPerfCounterBlock(name,
63                          (Integer) processInstanceMap.get(ProcessPerformanceProperty.CREATINGPROCESSID),
64                          (Integer) processInstanceMap.get(ProcessPerformanceProperty.PRIORITYBASE),
65                          (Long) processInstanceMap.get(ProcessPerformanceProperty.WORKINGSETPRIVATE),
66                          (Long) processInstanceMap.get(ProcessPerformanceProperty.WORKINGSET), ctime, upTime,
67                          (Long) processInstanceMap.get(ProcessPerformanceProperty.IOREADBYTESPERSEC),
68                          (Long) processInstanceMap.get(ProcessPerformanceProperty.IOWRITEBYTESPERSEC),
69                          Integer.toUnsignedLong(
70                                  (Integer) processInstanceMap.get(ProcessPerformanceProperty.PAGEFAULTSPERSEC))));
71              }
72          }
73          return processMap;
74      }
75  
76      /**
77       * Builds a process map from performance counter query results.
78       *
79       * @param pids           An optional collection of process IDs to filter the list to. May be null for no filtering.
80       * @param instanceValues The query results as a pair of (instances, valueMap)
81       * @return A map with Process ID as the key and a {@link ProcessPerfCounterBlock} object populated with performance
82       *         counter information, or null if instanceValues is null.
83       */
84      public static Map<Integer, ProcessPerfCounterBlock> buildProcessMapFromPerfCounters(Collection<Integer> pids,
85              Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> instanceValues) {
86          if (instanceValues == null) {
87              return null;
88          }
89          Map<Integer, ProcessPerfCounterBlock> processMap = new HashMap<>();
90          long now = System.currentTimeMillis(); // 1970 epoch
91          List<String> instances = instanceValues.getA();
92          Map<ProcessPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
93          List<Long> pidList = valueMap.get(ProcessPerformanceProperty.IDPROCESS);
94          List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.CREATINGPROCESSID);
95          List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.PRIORITYBASE);
96          List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.IOREADBYTESPERSEC);
97          List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.IOWRITEBYTESPERSEC);
98          List<Long> privateWorkingSetList = valueMap.get(ProcessPerformanceProperty.WORKINGSETPRIVATE);
99          List<Long> workingSetList = valueMap.get(ProcessPerformanceProperty.WORKINGSET);
100         List<Long> elapsedTimeList = valueMap.get(ProcessPerformanceProperty.ELAPSEDTIME);
101         List<Long> pageFaultsList = valueMap.get(ProcessPerformanceProperty.PAGEFAULTSPERSEC);
102 
103         for (int inst = 0; inst < instances.size(); inst++) {
104             int pid = pidList.get(inst).intValue();
105             if (pids == null || pids.contains(pid)) {
106                 long ctime = elapsedTimeList.get(inst);
107                 if (ctime > now) {
108                     ctime = ParseUtil.filetimeToUtcMs(ctime, false);
109                 }
110                 long upTime = now - ctime;
111                 if (upTime < 1L) {
112                     upTime = 1L;
113                 }
114                 processMap.put(pid,
115                         new ProcessPerfCounterBlock(instances.get(inst), ppidList.get(inst).intValue(),
116                                 priorityList.get(inst).intValue(), privateWorkingSetList.get(inst),
117                                 workingSetList.get(inst), ctime, upTime, ioReadList.get(inst), ioWriteList.get(inst),
118                                 pageFaultsList.get(inst).longValue()));
119             }
120         }
121         return processMap;
122     }
