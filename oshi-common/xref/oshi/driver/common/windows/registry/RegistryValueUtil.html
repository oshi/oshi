5   package oshi.driver.common.windows.registry;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.util.ParseUtil;
9   
10  /**
11   * Utility for interpreting raw Windows Registry values. Shared by the JNA and FFM backends so the two cannot diverge in
12   * how they coerce registry types to Java values.
13   */
14  @ThreadSafe
15  public final class RegistryValueUtil {
16  
17      private static final long THIRTY_YEARS_IN_SECS = 30L * 365 * 24 * 60 * 60;
18  
19      private RegistryValueUtil() {
20      }
21  
22      /**
23       * Interprets a registry value as a String, handling the three types encountered in practice: {@code REG_SZ} /
24       * {@code REG_EXPAND_SZ} (a {@link String}), {@code REG_DWORD} (an {@link Integer}), and {@code REG_BINARY} (a
25       * {@code byte[]}).
26       *
27       * @param val the raw registry value, or {@code null}
28       * @return the value as a String, or {@code null} if the value is {@code null} or an unsupported type
29       */
30      public static String registryValueToString(Object val) {
31          // REG_SZ / REG_EXPAND_SZ
32          if (val instanceof String) {
33              return ((String) val).trim();
34          }
35          // REG_DWORD (unsigned 32-bit)
36          if (val instanceof Integer) {
37              return Integer.toUnsignedString((Integer) val);
38          }
39          // REG_BINARY
40          if (val instanceof byte[]) {
41              return ParseUtil.decodeBinaryToString((byte[]) val);
42          }
43          return null;
44      }
45  
46      /**
47       * Interprets a registry value as a long. Supports {@link Integer} values (treating those in a plausible Unix
48       * timestamp range as seconds and converting to milliseconds) and {@link String} dates ({@code yyyyMMdd} or
49       * {@code MM/dd/yyyy}).
50       *
51       * @param val the raw registry value, or {@code null}
52       * @return the value as a long, or {@code 0} if the value is {@code null} or an unsupported type
53       */
54      public static long registryValueToLong(Object val) {
55          if (val == null) {
56              return 0L;
57          }
58          // Calculate reasonable timestamp bounds (current time to 30 years ago)
59          long currentTimeSecs = System.currentTimeMillis() / 1000L;
60          long minSaneTimestamp = currentTimeSecs - THIRTY_YEARS_IN_SECS;
61          if (val instanceof Integer) {
62              // REG_DWORD is unsigned; interpret it as such so timestamps past 2038 (high bit set) are not treated as
63              // negative.
64              long value = Integer.toUnsignedLong((Integer) val);
65              if (value > minSaneTimestamp && value < currentTimeSecs) {
66                  return value * 1000L;
67              }
68              return value;
69          } else if (val instanceof String) {
70              String dateStr = ((String) val).trim();
71              // Try yyyyMMdd first
72              long epoch = ParseUtil.parseDateToEpoch(dateStr, "yyyyMMdd");
73              if (epoch == 0) {
74                  // If that fails, try MM/dd/yyyy
75                  epoch = ParseUtil.parseDateToEpoch(dateStr, "MM/dd/yyyy");
76              }
77              return epoch;
78          }
79          return 0L;
80      }
81  }
