5   package oshi.driver.common.windows.perfmon;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   
9   /**
10   * Constants used in Perfmon driver classes
11   */
12  @ThreadSafe
13  public final class PerfmonConstants {
14  
15      /** Instance filter: _Total instance only. */
16      public static final String TOTAL_INSTANCE = "_Total";
17      /** Instance filter: _Total or Idle instances. */
18      public static final String TOTAL_OR_IDLE_INSTANCES = "_Total|Idle";
19      /** Instance filter: all _Total instances (wildcard). */
20      public static final String TOTAL_INSTANCES = "*_Total";
21      /** Instance filter: exclude _Total instance. */
22      public static final String NOT_TOTAL_INSTANCE = "^" + TOTAL_INSTANCE;
23      /** Instance filter: exclude all _Total instances. */
24      public static final String NOT_TOTAL_INSTANCES = "^" + TOTAL_INSTANCES;
25  
26      /** PDH object name for Memory counters. */
27      public static final String MEMORY = "Memory";
28      /** WMI class for raw Memory performance data. */
29      public static final String WIN32_PERF_RAW_DATA_PERF_OS_MEMORY = "Win32_PerfRawData_PerfOS_Memory";
30  
31      /** PDH object name for Paging File counters. */
32      public static final String PAGING_FILE = "Paging File";
33      /** WMI class for raw Paging File performance data. */
34      public static final String WIN32_PERF_RAW_DATA_PERF_OS_PAGING_FILE = "Win32_PerfRawData_PerfOS_PagingFile";
35  
36      /** PDH object name for Physical Disk counters. */
37      public static final String PHYSICAL_DISK = "PhysicalDisk";
38      /** WMI class for raw Physical Disk data excluding _Total. */
39      public static final String WIN32_PERF_RAW_DATA_PERF_DISK_PHYSICAL_DISK_WHERE_NAME_NOT_TOTAL = "Win32_PerfRawData_PerfDisk_PhysicalDisk WHERE Name!=\"_Total\"";
40  
41      /** PDH object name for Process counters. */
42      public static final String PROCESS = "Process";
43      /** WMI class for raw Process performance data. */
44      public static final String WIN32_PERFPROC_PROCESS = "Win32_PerfRawData_PerfProc_Process";
45      /** WMI query for Process data excluding _Total. */
46      public static final String WIN32_PERFPROC_PROCESS_WHERE_NOT_NAME_LIKE_TOTAL = WIN32_PERFPROC_PROCESS
47              + " WHERE NOT Name LIKE \"%_Total\"";
48      /** WMI query for Process _Total data. */
49      public static final String WIN32_PERFPROC_PROCESS_WHERE_NAME_TOTAL = WIN32_PERFPROC_PROCESS
50              + " WHERE Name=\"_Total\"";
51      /** WMI query for the System Idle Process (PID 0). */
52      public static final String WIN32_PERFPROC_PROCESS_WHERE_IDPROCESS_0 = "Win32_PerfRawData_PerfProc_Process WHERE IDProcess=0";
53  
54      /** PDH object name for Thread counters. */
55      public static final String THREAD = "Thread";
56      /** WMI class for raw Thread performance data. */
57      public static final String WIN32_PERF_RAW_DATA_PERF_PROC_THREAD = "Win32_PerfRawData_PerfProc_Thread";
58      /** WMI query for Thread data excluding _Total. */
59      public static final String WIN32_PERF_RAW_DATA_PERF_PROC_THREAD_WHERE_NOT_NAME_LIKE_TOTAL = "Win32_PerfRawData_PerfProc_Thread WHERE NOT Name LIKE \"%_Total\"";
60  
61      /** PDH object name for Processor counters (Vista and earlier). */
62      public static final String PROCESSOR = "Processor";
63      /** WMI query for Processor data excluding _Total. */
64      public static final String WIN32_PERF_RAW_DATA_PERF_OS_PROCESSOR_WHERE_NAME_NOT_TOTAL = "Win32_PerfRawData_PerfOS_Processor WHERE Name!=\"_Total\"";
65      /** WMI query for Processor _Total data. */
66      public static final String WIN32_PERF_RAW_DATA_PERF_OS_PROCESSOR_WHERE_NAME_TOTAL = "Win32_PerfRawData_PerfOS_Processor WHERE Name=\"_Total\"";
67  
68      /** PDH object name for Processor Information counters (Win7+). */
69      public static final String PROCESSOR_INFORMATION = "Processor Information";
70      /** WMI query for raw Processor Information data excluding _Total. */
71      public static final String WIN32_PERF_RAW_DATA_COUNTERS_PROCESSOR_INFORMATION_WHERE_NOT_NAME_LIKE_TOTAL = "Win32_PerfRawData_Counters_ProcessorInformation WHERE NOT Name LIKE \"%_Total\"";
72      /** WMI query for formatted (cooked) Processor Information data excluding _Total. */
73      public static final String WIN32_PERF_FORMATTED_DATA_COUNTERS_PROCESSOR_INFORMATION_WHERE_NOT_NAME_LIKE_TOTAL = "Win32_PerfFormattedData_Counters_ProcessorInformation WHERE NOT Name LIKE \"%_Total\"";
74  
75      /** PDH object name for System counters. */
76      public static final String SYSTEM = "System";
77      /** WMI class for raw System performance data. */
78      public static final String WIN32_PERF_RAW_DATA_PERF_OS_SYSTEM = "Win32_PerfRawData_PerfOS_System";
79  
80      /** PDH object name for GPU Engine counters. */
81      public static final String GPU_ENGINE = "GPU Engine";
82      /** PDH object name for GPU Adapter Memory counters. */
83      public static final String GPU_ADAPTER_MEMORY = "GPU Adapter Memory";
84  
85      /**
86       * Everything in this class is static, never instantiate it
87       */
88      private PerfmonConstants() {
89          throw new AssertionError();
90      }
91  }
