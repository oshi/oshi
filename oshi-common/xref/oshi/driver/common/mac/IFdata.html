5   package oshi.driver.common.mac;
6   
7   import oshi.annotation.concurrent.Immutable;
8   
9   /**
10   * Encapsulates network interface statistics read from the macOS {@code NET_RT_IFLIST2} sysctl, shared between the JNA
11   * and FFM NetStat drivers.
12   */
13  @Immutable
14  public class IFdata {
15      private final int ifType;
16      private final long oPackets;
17      private final long iPackets;
18      private final long oBytes;
19      private final long iBytes;
20      private final long oErrors;
21      private final long iErrors;
22      private final long collisions;
23      private final long iDrops;
24      private final long speed;
25      private final long timeStamp;
26  
27      /**
28       * Creates an immutable interface-data record. The {@code long} counters are masked to 32 bits because the
29       * underlying {@code if_data64} fields wrap at {@code 2^32}.
30       *
31       * @param ifType     the interface type
32       * @param oPackets   packets sent
33       * @param iPackets   packets received
34       * @param oBytes     bytes sent
35       * @param iBytes     bytes received
36       * @param oErrors    output errors
37       * @param iErrors    input errors
38       * @param collisions collisions
39       * @param iDrops     input drops
40       * @param speed      interface speed (baud rate)
41       * @param timeStamp  the time the data was sampled, in milliseconds since the epoch
42       */
43      public IFdata(int ifType, // NOSONAR squid:S00107
44              long oPackets, long iPackets, long oBytes, long iBytes, long oErrors, long iErrors, long collisions,
45              long iDrops, long speed, long timeStamp) {
46          this.ifType = ifType;
47          this.oPackets = oPackets & 0xffffffffL;
48          this.iPackets = iPackets & 0xffffffffL;
49          this.oBytes = oBytes & 0xffffffffL;
50          this.iBytes = iBytes & 0xffffffffL;
51          this.oErrors = oErrors & 0xffffffffL;
52          this.iErrors = iErrors & 0xffffffffL;
53          this.collisions = collisions & 0xffffffffL;
54          this.iDrops = iDrops & 0xffffffffL;
55          this.speed = speed & 0xffffffffL;
56          this.timeStamp = timeStamp;
57      }
58  
59      /**
60       * @return the ifType
61       */
62      public int getIfType() {
63          return ifType;
64      }
65  
66      /**
67       * @return the oPackets
68       */
69      public long getOPackets() {
70          return oPackets;
71      }
72  
73      /**
74       * @return the iPackets
75       */
76      public long getIPackets() {
77          return iPackets;
78      }
79  
80      /**
81       * @return the oBytes
82       */
83      public long getOBytes() {
84          return oBytes;
85      }
86  
87      /**
88       * @return the iBytes
89       */
90      public long getIBytes() {
91          return iBytes;
92      }
93  
94      /**
95       * @return the oErrors
96       */
97      public long getOErrors() {
98          return oErrors;
99      }
100 
101     /**
102      * @return the iErrors
103      */
104     public long getIErrors() {
105         return iErrors;
106     }
107 
108     /**
109      * @return the collisions
110      */
111     public long getCollisions() {
112         return collisions;
113     }
114 
115     /**
116      * @return the iDrops
117      */
118     public long getIDrops() {
119         return iDrops;
120     }
121 
122     /**
123      * @return the speed
124      */
125     public long getSpeed() {
126         return speed;
127     }
128 
129     /**
130      * @return the timeStamp
131      */
132     public long getTimeStamp() {
133         return timeStamp;
134     }
135 }
