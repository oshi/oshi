5   package oshi.driver.common.unix.solaris;
6   
7   import java.nio.ByteBuffer;
8   
9   import oshi.util.FileUtil;
10  
11  /**
12   * 64-bit Solaris {@code timestruc_t}: seconds and nanoseconds, each 8 bytes.
13   */
14  public class SolarisTimestruc {
15  
16      public long tv_sec; // seconds
17      public long tv_nsec; // nanoseconds
18  
19      public SolarisTimestruc(ByteBuffer buff) {
20          this.tv_sec = FileUtil.readLongFromBuffer(buff);
21          this.tv_nsec = FileUtil.readLongFromBuffer(buff);
22      }
23  
24      /**
25       * Returns this timestamp converted to milliseconds since the epoch.
26       *
27       * @return the timestamp in milliseconds
28       */
29      public long toMillis() {
30          return tv_sec * 1000L + tv_nsec / 1_000_000L;
31      }
32  }
