5   package oshi.driver.common.unix.freebsd.disk;
6   
7   import java.util.HashMap;
8   import java.util.List;
9   import java.util.Map;
10  import java.util.regex.Matcher;
11  import java.util.regex.Pattern;
12  
13  import oshi.annotation.concurrent.ThreadSafe;
14  import oshi.util.ExecutingCommand;
15  
16  /**
17   * Utility to query mount
18   */
19  @ThreadSafe
20  public final class Mount {
21  
22      private static final String MOUNT_CMD = "mount";
23      private static final Pattern MOUNT_PATTERN = Pattern.compile("/dev/(\\S+p\\d+) on (\\S+) .*");
24  
25      private Mount() {
26      }
27  
28      /**
29       * Query mount to map partitions to mount points
30       *
31       * @return A map with partitions as the key and mount points as the value
32       */
33      public static Map<String, String> queryPartitionToMountMap() {
34          return parseMountOutput(ExecutingCommand.runNative(MOUNT_CMD));
35      }
36  
37      /**
38       * Parses the output of the {@code mount} command into a partition-to-mount-point map.
39       *
40       * @param mountOutput the lines of output from {@code mount}
41       * @return A map with partitions as the key and mount points as the value
42       */
43      static Map<String, String> parseMountOutput(List<String> mountOutput) {
44          Map<String, String> mountMap = new HashMap<>();
45          for (String mnt : mountOutput) {
46              Matcher m = MOUNT_PATTERN.matcher(mnt);
47              if (m.matches()) {
48                  mountMap.put(m.group(1), m.group(2));
49              }
50          }
51          return mountMap;
52      }
53  }
