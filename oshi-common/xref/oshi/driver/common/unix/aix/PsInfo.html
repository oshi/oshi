5   package oshi.driver.common.unix.aix;
6   
7   import java.util.Locale;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.util.FileUtil;
14  import oshi.util.tuples.Triplet;
15  
16  /**
17   * Shared utility to query the pure-Java parts of AIX {@code /proc/<pid>/psinfo} and
18   * {@code /proc/<pid>/lwp/<tid>/lwpsinfo}.
19   * <p>
20   * The address-space reads {@code queryArgsEnv} performs need libc {@code open}/{@code pread} and live in the JNA
21   * ({@code PsInfoJNA}) and FFM ({@code PsInfoFFM}) drivers respectively.
22   */
23  @ThreadSafe
24  public final class PsInfo {
25  
26      private static final Logger LOG = LoggerFactory.getLogger(PsInfo.class);
27  
28      private PsInfo() {
29      }
30  
31      /**
32       * Reads {@code /proc/<pid>/psinfo} and returns the parsed structure.
33       *
34       * @param pid the process ID
35       * @return a structure containing information for the requested process
36       */
37      public static AixPsInfo queryPsInfo(int pid) {
38          return new AixPsInfo(FileUtil.readAllBytesAsBuffer(String.format(Locale.ROOT, "/proc/%d/psinfo", pid)));
39      }
40  
41      /**
42       * Reads {@code /proc/<pid>/lwp/<tid>/lwpsinfo} and returns the parsed structure.
43       *
44       * @param pid the process ID
45       * @param tid the thread ID (lwpid)
46       * @return a structure containing information for the requested thread
47       */
48      public static AixLwpsInfo queryLwpsInfo(int pid, int tid) {
49          return new AixLwpsInfo(
50                  FileUtil.readAllBytesAsBuffer(String.format(Locale.ROOT, "/proc/%d/lwp/%d/lwpsinfo", pid, tid)));
51      }
52  
53      /**
54       * Reads the {@code pr_argc}, {@code pr_argv}, and {@code pr_envp} fields from a {@code psinfo_t}.
55       *
56       * @param pid    the process ID (used for diagnostic logging only)
57       * @param psinfo a populated {@link AixPsInfo} containing the offset pointers
58       * @return a triplet of {@code (argc, argv, envp)}, or {@code null} if the psinfo is unusable
59       */
60      public static Triplet<Integer, Long, Long> queryArgsEnvAddrs(int pid, AixPsInfo psinfo) {
61          if (psinfo != null) {
62              int argc = psinfo.pr_argc;
63              // Must have at least one argc (the command itself) so failure here means exit
64              if (argc > 0) {
65                  return new Triplet<>(argc, psinfo.pr_argv, psinfo.pr_envp);
66              }
67              LOG.trace("Failed argc sanity check: argc={}", argc);
68              return null;
69          }
70          LOG.trace("Failed to read psinfo file for pid: {} ", pid);
71          return null;
72      }
73  }
