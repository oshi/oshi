5   package oshi.hardware;
6   
7   import oshi.annotation.concurrent.Immutable;
8   
9   /**
10   * An immutable snapshot of cumulative GPU active and idle tick counters in opaque, platform-native units.
11   *
12   * <p>
13   * Because {@code activeTicks + idleTicks = total elapsed ticks}, callers can compute utilization from two snapshots
14   * without needing a separate timestamp:
15   *
16   * <pre>{@code
17   * long dActive = curr.getActiveTicks() - prev.getActiveTicks();
18   * long dIdle = curr.getIdleTicks() - prev.getIdleTicks();
19   * long dTotal = dActive + dIdle;
20   * double util = dTotal > 0 ? dActive * 100.0 / dTotal : -1d;
21   * }</pre>
22   *
23   * <p>
24   * Both counters are {@code 0} when tick-level metrics are not available on this platform. Because {@code 0} is also the
25   * natural starting value of a real counter, callers cannot distinguish "not available" from "counter just started" —
26   * but the {@code dTotal > 0} guard in the utilization formula handles both cases correctly.
27   */
28  @Immutable
29  public final class GpuTicks {
30  
31      private final long activeTicks;
32      private final long idleTicks;
33  
34      /**
35       * Creates a new snapshot.
36       *
37       * @param activeTicks cumulative GPU active ticks in platform-native units
38       * @param idleTicks   cumulative GPU idle ticks in platform-native units
39       */
40      public GpuTicks(long activeTicks, long idleTicks) {
41          this.activeTicks = activeTicks;
42          this.idleTicks = idleTicks;
43      }
44  
45      /**
46       * @return cumulative GPU active ticks in platform-native units
47       */
48      public long getActiveTicks() {
49          return activeTicks;
50      }
51  
52      /**
53       * @return cumulative GPU idle ticks in platform-native units
54       */
55      public long getIdleTicks() {
56          return idleTicks;
57      }
58  
59      @Override
60      public String toString() {
61          return "GpuTicks{activeTicks=" + activeTicks + ", idleTicks=" + idleTicks + '}';
62      }
63  }
