5   package oshi.hardware;
6   
7   import oshi.annotation.PublicApi;
8   import oshi.annotation.concurrent.Immutable;
9   
10  /**
11   * Display refers to the information regarding a video source and monitor identified by the EDID standard.
12   * <p>
13   * {@link #getDisplayInfo()} returns a {@link DisplayInfo} exposing the display's decoded attributes — manufacturer ID,
14   * product ID, serial number, physical dimensions, preferred resolution, model name, and the raw (or synthesized) EDID
15   * byte array.
16   * <p>
17   * Example: extracting monitor manufacturer and dimensions:
18   *
19   * <pre>{@code
20   * for (Display display : hal.getDisplays()) {
21   *     DisplayInfo info = display.getDisplayInfo();
22   *     System.out.println("Manufacturer: " + info.getManufacturerID());
23   *     System.out.println("Product ID:   " + info.getProductID());
24   *     System.out.printf("Size: %d cm x %d cm%n", info.getHcm(), info.getVcm());
25   * }
26   * }</pre>
27   *
28   * For displays that report their attributes without providing an EDID (such as a built-in macOS Retina panel),
29   * {@link DisplayInfo#isEdidSynthetic()} returns {@code true} and {@link DisplayInfo#getEdid()} returns an EDID
30   * synthesized from those attributes.
31   *
32   * @see DisplayInfo
33   * @see oshi.util.EdidUtil
34   */
35  @PublicApi
36  @Immutable
37  public interface Display {
38      /**
39       * The EDID byte array.
40       *
41       * @return The EDID byte array, either reported by the display or, when {@link DisplayInfo#isEdidSynthetic()} is
42       *         {@code true}, synthesized from the display's reported attributes.
43       * @deprecated As of 7.4.0, use {@link #getDisplayInfo()}.{@link DisplayInfo#getEdid() getEdid()} instead; the
44       *             decoded attributes are also available directly from {@link DisplayInfo}. Scheduled for removal in the
45       *             next major release.
46       */
47      @Deprecated
48      byte[] getEdid();
49  
50      /**
51       * The decoded display information.
52       *
53       * @return A {@link DisplayInfo} holding the display's decoded attributes.
54       */
55      DisplayInfo getDisplayInfo();
56  }
