41      /**
42       * Initialization-on-demand holder: the map is only constructed if the non-udev path is reached, avoiding
43       * unnecessary work on systems where udev is available.
44       */
45      private static final class PropByName {
46          static final Map<String, Prop> MAP = new HashMap<>();
47          static {
48              for (Prop p : Prop.values()) {
49                  MAP.put(p.name(), p);
50              }
51          }
52      }

89      /**
90       * Gets Battery Information from the specified power supply path.
91       *
92       * @param powerSupplyPath The path to the power supply directory.
93       * @return An array of PowerSource objects representing batteries, etc.
94       */
95      static List<PowerSource> getPowerSources(String powerSupplyPath) {
96          List<PowerSource> psList = new ArrayList<>();
97          File psDir = new File(powerSupplyPath);
98          File[] psArr = psDir.listFiles();
99          if (psArr == null) {
100             return Collections.emptyList();
101         }
102         for (File ps : psArr) {
103             String name = ps.getName();
104             if (!name.startsWith("ADP") && !name.startsWith("AC") && !name.contains("USBC")) {
105                 List<String> psInfo = FileUtil.readFile(powerSupplyPath + "/" + name + "/uevent", false);
106                 Map<Prop, String> props = new EnumMap<>(Prop.class);
107                 for (String line : psInfo) {
108                     String[] split = line.split("=", 2);
109                     if (split.length > 1 && !split[1].isEmpty()) {
110                         Prop p = PropByName.MAP.get(split[0]);
111                         if (p != null) {
112                             props.put(p, split[1]);
113                         }
114                     }
115                 }
116                 if (ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_PRESENT), 1) > 0) {
117                     psList.add(buildPowerSource(name, props));
118                 }
119             }
120         }
121         return psList;
122     }

124     protected static LinuxPowerSource buildPowerSource(String name, Map<Prop, String> props) {
125         String psName = props.getOrDefault(Prop.POWER_SUPPLY_NAME, name);
126         String status = props.get(Prop.POWER_SUPPLY_STATUS);
127         boolean psCharging = "Charging".equals(status);
128         boolean psDischarging = "Discharging".equals(status);
129         double psRemainingCapacityPercent = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CAPACITY), -100)
130                 / 100d;
131 
132         // Debian/Ubuntu provides Energy (µWh). Fedora/RHEL provides Charge (µAh).
133         // Parse raw values in micro-units first, fall back across Energy/Charge, then
134         // divide by 1000 at the end to avoid integer division truncating the -1 sentinel.
135         int rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_NOW), -1);
136         CapacityUnits psCapacityUnits;
137         if (rawNow >= 0) {
138             psCapacityUnits = CapacityUnits.MWH;
139         } else {
140             rawNow = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_NOW), -1);
141             psCapacityUnits = CapacityUnits.MAH;
142         }
143         int psCurrentCapacity = rawNow >= 0 ? rawNow / 1000 : -1;
144 
145         int rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL), -1);
146         if (rawFull < 0) {
147             rawFull = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL), -1);
148         }
149         int psMaxCapacity = rawFull >= 0 ? rawFull / 1000 : -1;
150 
151         int rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_ENERGY_FULL_DESIGN), -1);
152         if (rawDesign < 0) {
153             rawDesign = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CHARGE_FULL_DESIGN), -1);
154         }
155         int psDesignCapacity = rawDesign >= 0 ? rawDesign / 1000 : -1;
156 
157         // Debian/Ubuntu provides Voltage and Power. Fedora/RHEL provides Voltage and Current.
158         double psVoltage = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_VOLTAGE_NOW), -1d) / 1_000_000d;
159         double psPowerUsageRate = 0d;
160         double psAmperage = 0d;
161         // From Physics we know P = IV so I = P/V
162         // Linux always reports POWER_NOW and CURRENT_NOW as non-negative; sign is
163         // conveyed by STATUS. Negate when discharging to match the PowerSource contract
164         // (positive = charging, negative = discharging).
165         double sign = psDischarging ? -1d : 1d;
166         if (psVoltage > 0) {
167             double rawPower = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_POWER_NOW), -1d);
168             double rawCurrent = ParseUtil.parseDoubleOrDefault(props.get(Prop.POWER_SUPPLY_CURRENT_NOW), -1d);
169             if (rawPower >= 0) {
170                 psPowerUsageRate = sign * rawPower / 1000d;
171             }
172             if (rawCurrent >= 0) {
173                 psAmperage = sign * rawCurrent / 1000d;
174             }
175             if (rawPower < 0 && rawCurrent >= 0) {
176                 psPowerUsageRate = psAmperage * psVoltage;
177             } else if (rawCurrent < 0 && rawPower >= 0) {
178                 psAmperage = psPowerUsageRate / psVoltage;
179             }
180         }
181 
182         int psCycleCount = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_CYCLE_COUNT), -1);
183         String psChemistry = props.getOrDefault(Prop.POWER_SUPPLY_TECHNOLOGY, UNKNOWN);
184         String psDeviceName = props.getOrDefault(Prop.POWER_SUPPLY_MODEL_NAME, UNKNOWN);
185         String psManufacturer = props.getOrDefault(Prop.POWER_SUPPLY_MANUFACTURER, UNKNOWN);
186         String psSerialNumber = props.getOrDefault(Prop.POWER_SUPPLY_SERIAL_NUMBER, UNKNOWN);
187 
188         // TEMP is in tenths of degrees Celsius
189         int rawTemp = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_TEMP), -1);
190         double psTemperature = rawTemp >= 0 ? rawTemp / 10d : 0d;
191 
192         // TIME_TO_EMPTY_NOW when discharging, TIME_TO_FULL_NOW when charging; both in seconds
193         Prop timeKey = psCharging ? Prop.POWER_SUPPLY_TIME_TO_FULL_NOW : Prop.POWER_SUPPLY_TIME_TO_EMPTY_NOW;
194         double psTimeRemainingInstant = ParseUtil.parseIntOrDefault(props.get(timeKey), -1);
195 
196         int year = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_YEAR), -1);
197         int month = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_MONTH), -1);
198         int day = ParseUtil.parseIntOrDefault(props.get(Prop.POWER_SUPPLY_MANUFACTURE_DAY), -1);
199         LocalDate psManufactureDate = (year > 0 && month > 0 && day > 0) ? LocalDate.of(year, month, day) : null;
200 
201         return new LinuxPowerSource(psName, psDeviceName, psRemainingCapacityPercent, -1d, psTimeRemainingInstant,
202                 psPowerUsageRate, psVoltage, psAmperage, false, psCharging, psDischarging, psCapacityUnits,
203                 psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
204                 psManufacturer, psSerialNumber, psTemperature);
205     }
