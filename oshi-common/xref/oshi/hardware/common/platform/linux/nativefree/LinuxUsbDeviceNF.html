5   package oshi.hardware.common.platform.linux.nativefree;
6   
7   import java.io.File;
8   import java.util.ArrayList;
9   import java.util.HashMap;
10  import java.util.List;
11  import java.util.Map;
12  
13  import oshi.hardware.UsbDevice;
14  import oshi.hardware.common.platform.linux.LinuxUsbDevice;
15  import oshi.util.FileUtil;
16  import oshi.util.linux.SysPath;
17  
18  /**
19   * Native-free Linux USB device enumeration from {@code /sys/bus/usb/devices/}.
20   */
21  public final class LinuxUsbDeviceNF extends LinuxUsbDevice {
22  
23      private static final String SYS_USB = SysPath.SYS + "bus/usb/devices/";
24  
25      private LinuxUsbDeviceNF() {
26      }
27  
28      /**
29       * Gets USB devices on this machine.
30       *
31       * @param tree if true, returns controllers with device tree; if false, flat list excluding controllers
32       * @return a list of {@link UsbDevice} objects
33       */
34      public static List<UsbDevice> getUsbDevices(boolean tree) {
35          List<UsbDevice> devices = queryUsbDevices(SYS_USB);
36          if (tree) {
37              return devices;
38          }
39          List<UsbDevice> deviceList = new ArrayList<>();
40          for (UsbDevice device : devices) {
41              addDevicesToList(deviceList, device.getConnectedDevices());
42          }
43          return deviceList;
44      }
45  
46      /**
47       * Queries USB devices from the specified sysfs path.
48       *
49       * @param sysUsbPath the path to the USB devices directory (e.g., /sys/bus/usb/devices/)
50       * @return a list of USB device trees rooted at controllers
51       */
52      static List<UsbDevice> queryUsbDevices(String sysUsbPath) {
53          if (!sysUsbPath.endsWith(File.separator)) {
54              sysUsbPath += File.separator;
55          }
56          File usbDir = new File(sysUsbPath);
57          File[] deviceDirs = usbDir.listFiles();
58          if (deviceDirs == null) {
59              return new ArrayList<>();
60          }
61  
62          List<String> usbControllers = new ArrayList<>();
63          Map<String, String> nameMap = new HashMap<>();
64          Map<String, String> vendorMap = new HashMap<>();
65          Map<String, String> vendorIdMap = new HashMap<>();
66          Map<String, String> productIdMap = new HashMap<>();
67          Map<String, String> serialMap = new HashMap<>();
68          Map<String, List<String>> hubMap = new HashMap<>();
69  
70          for (File devDir : deviceDirs) {
71              String syspath = devDir.getAbsolutePath();
72              String devtype = FileUtil.getStringFromFile(syspath + "/devnum").trim();
73              if (devtype.isEmpty()) {
74                  continue;
75              }
76  
77              String product = FileUtil.getStringFromFile(syspath + "/product").trim();
78              if (!product.isEmpty()) {
79                  nameMap.put(syspath, product);
80              }
81              String manufacturer = FileUtil.getStringFromFile(syspath + "/manufacturer").trim();
82              if (!manufacturer.isEmpty()) {
83                  vendorMap.put(syspath, manufacturer);
84              }
85              String idVendor = FileUtil.getStringFromFile(syspath + "/idVendor").trim();
86              if (!idVendor.isEmpty()) {
87                  vendorIdMap.put(syspath, idVendor);
88              }
89              String idProduct = FileUtil.getStringFromFile(syspath + "/idProduct").trim();
90              if (!idProduct.isEmpty()) {
91                  productIdMap.put(syspath, idProduct);
92              }
93              String serial = FileUtil.getStringFromFile(syspath + "/serial").trim();
94              if (!serial.isEmpty()) {
95                  serialMap.put(syspath, serial);
96              }
97  
98              // Determine parent: USB root hubs have names like "usbN", children have "N-N.N..."
99              String name = devDir.getName();
100             if (name.startsWith("usb")) {
101                 usbControllers.add(syspath);
102             } else {
103                 String parentName;
104                 if (name.contains(".")) {
105                     parentName = name.substring(0, name.lastIndexOf('.'));
106                 } else if (name.contains("-")) {
107                     parentName = "usb" + name.substring(0, name.indexOf('-'));
108                 } else {
109                     parentName = name;
110                 }
111                 String parentPath = sysUsbPath + parentName;
112                 hubMap.computeIfAbsent(parentPath, x -> new ArrayList<>()).add(syspath);
113             }
114         }
115 
116         List<UsbDevice> controllerDevices = new ArrayList<>();
117         for (String controller : usbControllers) {
118             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
119                     productIdMap, serialMap, hubMap));
120         }
121         return controllerDevices;
122     }
