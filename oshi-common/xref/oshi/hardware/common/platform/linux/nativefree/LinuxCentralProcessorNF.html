5   package oshi.hardware.common.platform.linux.nativefree;
6   
7   import java.util.List;
8   import java.util.Map;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.hardware.common.platform.linux.LinuxCentralProcessor;
12  import oshi.util.ExecutingCommand;
13  import oshi.util.ParseUtil;
14  import oshi.util.tuples.Quartet;
15  
16  /**
17   * Native-free Linux central processor implementation. Extends {@link LinuxCentralProcessor}, overriding udev-dependent
18   * methods with sysfs-based alternatives.
19   */
20  @ThreadSafe
21  public final class LinuxCentralProcessorNF extends LinuxCentralProcessor {
22  
23      private static final long USER_HZ = queryUserHz();
24  
25      /**
26       * Creates a new native-free Linux central processor.
27       */
28      public LinuxCentralProcessorNF() {
29          super(USER_HZ);
30      }
31  
32      private static long queryUserHz() {
33          long hz = ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf CLK_TCK"), 0L);
34          return hz > 0 ? hz : 100L;
35      }
36  
37      @Override
38      protected Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> readTopologyWithUdev() {
39          return readTopologyFromSysfs();
40      }
41  
42      @Override
43      protected boolean queryCurrentFreqFromUdev(long[] freqs) {
44          return queryCurrentFreqFromSysfs(freqs);
45      }
46  
47      @Override
48      protected long queryMaxFreqFromUdev() {
49          return queryMaxFreqFromSysfs();
50      }
51  }
