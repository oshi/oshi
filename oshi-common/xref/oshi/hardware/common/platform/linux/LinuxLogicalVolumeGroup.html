5   package oshi.hardware.common.platform.linux;
6   
7   import java.util.HashMap;
8   import java.util.HashSet;
9   import java.util.Map;
10  import java.util.Set;
11  
12  import oshi.hardware.common.AbstractLogicalVolumeGroup;
13  import oshi.util.ExecutingCommand;
14  import oshi.util.ParseUtil;
15  import oshi.util.linux.DevPath;
16  
17  /**
18   * Linux implementation of LogicalVolumeGroup.
19   */
20  public class LinuxLogicalVolumeGroup extends AbstractLogicalVolumeGroup {
21  
22      /** Sysfs block device type. */
23      protected static final String BLOCK = "block";
24      /** Device-mapper UUID property. */
25      protected static final String DM_UUID = "DM_UUID";
26      /** Device-mapper volume group name property. */
27      protected static final String DM_VG_NAME = "DM_VG_NAME";
28      /** Device-mapper logical volume name property. */
29      protected static final String DM_LV_NAME = "DM_LV_NAME";
30  
31      /**
32       * Creates a LinuxLogicalVolumeGroup.
33       *
34       * @param name  the volume group name
35       * @param lvMap the logical volume map
36       * @param pvSet the physical volume set
37       */
38      protected LinuxLogicalVolumeGroup(String name, Map<String, Set<String>> lvMap, Set<String> pvSet) {
39          super(name, lvMap, pvSet);
40      }
41  
42      /**
43       * Populates the physical volumes map from the {@code pvs} command output. Requires elevated permissions; may return
44       * an empty map if the command fails.
45       *
46       * @return map of VG name to set of PV device paths
47       */
48      protected static Map<String, Set<String>> queryPhysicalVolumes() {
49          Map<String, Set<String>> physicalVolumesMap = new HashMap<>();
50          for (String s : ExecutingCommand.runNative("pvs -o vg_name,pv_name")) {
51              String[] split = ParseUtil.whitespaces.split(s.trim());
52              if (split.length == 2 && split[1].startsWith(DevPath.DEV)) {
53                  physicalVolumesMap.computeIfAbsent(split[0], k -> new HashSet<>()).add(split[1]);
54              }
55          }
56          return physicalVolumesMap;
57      }
58  }
