53      // Get a list of orders to pass to ParseUtil
54      protected static final int[] UDEV_STAT_ORDERS = new int[UdevStat.values().length];
55      static {
56          for (UdevStat stat : UdevStat.values()) {
57              UDEV_STAT_ORDERS[stat.ordinal()] = stat.getOrder();
58          }
59      }
60  
61      // There are at least 11 elements in udev stat output or sometimes 15. We want
62      // the rightmost 11 or 15 if there is leading text.
63      protected static final int UDEV_STAT_LENGTH;
64      static {
65          String stat = FileUtil.getStringFromFile(ProcPath.DISKSTATS);
66          int statLength = 11;
67          if (!stat.isEmpty()) {
68              statLength = ParseUtil.countStringToLongArray(stat, ' ');
69          }
70          UDEV_STAT_LENGTH = statLength;
71      }

149     protected static void computeDiskStats(LinuxHWDiskStore store, String devstat) {
150         long[] devstatArray = ParseUtil.parseStringToLongArray(devstat, UDEV_STAT_ORDERS, UDEV_STAT_LENGTH, ' ');
151         store.setDiskStats(devstatArray[UdevStat.READS.ordinal()],
152                 devstatArray[UdevStat.READ_BYTES.ordinal()] * SECTORSIZE, devstatArray[UdevStat.WRITES.ordinal()],
153                 devstatArray[UdevStat.WRITE_BYTES.ordinal()] * SECTORSIZE,
154                 devstatArray[UdevStat.QUEUE_LENGTH.ordinal()], devstatArray[UdevStat.ACTIVE_MS.ordinal()],
155                 System.currentTimeMillis());
156     }
157 
158     protected static Map<String, String> readMountsMap() {
159         Map<String, String> mountsMap = new HashMap<>();
160         List<String> mounts = FileUtil.readFile(ProcPath.MOUNTS);
161         for (String mount : mounts) {
162             String[] split = ParseUtil.whitespaces.split(mount);
163             if (split.length < 2 || !split[0].startsWith(DevPath.DEV)) {
164                 continue;
165             }
166             mountsMap.put(split[0], split[1]);
167         }
168         return mountsMap;
169     }
170 
171     protected static String getPartitionNameForDmDevice(String vgName, String lvName) {
172         return DevPath.DEV + vgName + '/' + lvName;
173     }
174 
175     protected static String getMountPointOfDmDevice(String vgName, String lvName) {
176         return DevPath.MAPPER + vgName + '-' + lvName;
177     }
178 
179     protected static String getDependentNamesFromHoldersDirectory(String sysPath) {
180         File holdersDir = new File(sysPath + "/holders");
181         File[] holders = holdersDir.listFiles();
182         if (holders != null) {
183             return Arrays.stream(holders).map(File::getName).collect(Collectors.joining(" "));
184         }
185         return "";
186     }
187 
188     protected static void finalizePartitions(List<HWDiskStore> result) {
189         for (HWDiskStore hwds : result) {
190             LinuxHWDiskStore store = (LinuxHWDiskStore) hwds;
191             store.setPartitionList(Collections.unmodifiableList(store.getPartitions().stream()
192                     .sorted(java.util.Comparator.comparing(HWPartition::getName)).collect(Collectors.toList())));
193         }
194     }

196     // Order the field is in udev stats
197     protected enum UdevStat {
198         // The parsing implementation in ParseUtil requires these to be declared
199         // in increasing order. Use 0-ordered index here
200         READS(0), READ_BYTES(2), WRITES(4), WRITE_BYTES(6), QUEUE_LENGTH(8), ACTIVE_MS(9);
201 
202         private int order;
203 
204         public int getOrder() {
205             return this.order;
206         }
207 
208         UdevStat(int order) {
209             this.order = order;
210         }
211     }
