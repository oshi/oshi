5   package oshi.hardware.common.platform.mac;
6   
7   import java.time.LocalDate;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.AbstractPowerSource;
11  
12  /**
13   * A Power Source
14   */
15  @ThreadSafe
16  public abstract class MacPowerSource extends AbstractPowerSource {
17  
18      /**
19       * Creates a MacPowerSource.
20       *
21       * @param psName                     power source name
22       * @param psDeviceName               device name
23       * @param psRemainingCapacityPercent remaining capacity percentage
24       * @param psTimeRemainingEstimated   estimated time remaining
25       * @param psTimeRemainingInstant     instant time remaining
26       * @param psPowerUsageRate           power usage rate
27       * @param psVoltage                  voltage
28       * @param psAmperage                 amperage
29       * @param psPowerOnLine              whether on AC power
30       * @param psCharging                 whether charging
31       * @param psDischarging              whether discharging
32       * @param psCapacityUnits            capacity units
33       * @param psCurrentCapacity          current capacity
34       * @param psMaxCapacity              max capacity
35       * @param psDesignCapacity           design capacity
36       * @param psCycleCount               cycle count
37       * @param psChemistry                chemistry
38       * @param psManufactureDate          manufacture date
39       * @param psManufacturer             manufacturer
40       * @param psSerialNumber             serial number
41       * @param psTemperature              temperature
42       */
43      public MacPowerSource(String psName, String psDeviceName, double psRemainingCapacityPercent,
44              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
45              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
46              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
47              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
48              String psSerialNumber, double psTemperature) {
49          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
50                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
51                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
52                  psManufacturer, psSerialNumber, psTemperature);
53      }
54  }
