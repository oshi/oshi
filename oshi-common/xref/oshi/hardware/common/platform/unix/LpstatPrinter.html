5   package oshi.hardware.common.platform.unix;
6   
7   import java.util.List;
8   
9   import oshi.annotation.concurrent.Immutable;
10  import oshi.hardware.Printer;
11  
12  /**
13   * Native-free CUPS printer implementation. Queries printers via the {@code lpstat} command rather than libcups, so it
14   * works without JNA or FFM native access. Used by platforms whose hardware abstraction layer lives entirely in
15   * {@code oshi-common} (currently NetBSD).
16   */
17  @Immutable
18  public final class LpstatPrinter extends CupsPrinter {
19  
20      LpstatPrinter(String name, String driverName, String description, PrinterStatus status, String statusReason,
21              boolean isDefault, boolean isLocal, String portName) {
22          super(name, driverName, description, status, statusReason, isDefault, isLocal, portName);
23      }
24  
25      /**
26       * Gets the list of printers known to CUPS by parsing {@code lpstat -p} output.
27       *
28       * @return list of printers, or an empty list if {@code lpstat} is unavailable.
29       */
30      public static List<Printer> getPrinters() {
31          return getPrintersFromLpstat(LpstatPrinter::new);
32      }
33  }
