17  /**
18   * Shared OpenBSD CentralProcessor logic. The processor identification, frequency, and {@code vmstat} parsing are common
19   * to the JNA and FFM backends; subclasses supply only the sysctl reads (mib- and name-based) and the native CPU-tick
20   * and load-average calls.
21   */
22  public abstract class OpenBsdCentralProcessor extends BsdCentralProcessor {
23  
24      // OpenBSD sysctl(3) ABI identifiers, stable across releases (mirrored from the JNA/FFM libc backends).
25      private static final int CTL_HW = 6;
26      private static final int HW_MACHINE = 1;
27      private static final int HW_MODEL = 2;
28      private static final int HW_CPUSPEED = 12;
29  
30      // OpenBSD CPU state indices into the kern.cptime / kern.cptime2 arrays
31      private static final int CP_USER = 0;
32      private static final int CP_NICE = 1;
33      private static final int CP_SYS = 2;
34      private static final int CP_INTR = 3;
35      private static final int CP_IDLE = 4;
36      private static final int CPUSTATES = 5;
37  
38      /**
39       * Reads a string sysctl value by name.
40       *
41       * @param name the sysctl name
42       * @param def  the default value
43       * @return the sysctl string value or the default
44       */
45      protected abstract String sysctl(String name, String def);
46  
47      /**
48       * Reads a string sysctl value addressed by a Management Information Base (MIB) array.
49       *
50       * @param mib the sysctl MIB
51       * @param def the default value
52       * @return the sysctl string value or the default
53       */
54      protected abstract String sysctl(int[] mib, String def);
55  
56      /**
57       * Reads an integer sysctl value addressed by a Management Information Base (MIB) array.
58       *
59       * @param mib the sysctl MIB
60       * @param def the default value
61       * @return the sysctl integer value or the default
62       */
63      protected abstract int sysctl(int[] mib, int def);
64  
65      @Override
66      protected ProcessorIdentifier queryProcessorId() {
67          String cpuVendor = sysctl("machdep.cpuvendor", "");
68          int[] mib = { CTL_HW, HW_MODEL };
69          String cpuName = sysctl(mib, "");
70          int cpuid = ParseUtil.hexStringToInt(sysctl("machdep.cpuid", ""), 0);
71          int cpufeature = ParseUtil.hexStringToInt(sysctl("machdep.cpufeature", ""), 0);
72          Triplet<Integer, Integer, Integer> cpu = cpuidToFamilyModelStepping(cpuid);
73          String cpuFamily = cpu.getA().toString();
74          String cpuModel = cpu.getB().toString();
75          String cpuStepping = cpu.getC().toString();
76          long cpuFreq = ParseUtil.parseHertz(cpuName);
77          if (cpuFreq < 0) {
78              cpuFreq = queryMaxFreq();
79          }
80          mib = new int[] { CTL_HW, HW_MACHINE };
81          String machine = sysctl(mib, "");
82          boolean cpu64bit = machine != null && machine.contains("64")
83                  || ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64");
84          String processorID = String.format(Locale.ROOT, "%08x%08x", cpufeature, cpuid);
85  
86          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
87                  cpuFreq);
88      }
89  
90      private static Triplet<Integer, Integer, Integer> cpuidToFamilyModelStepping(int cpuid) {
91          // family is bits 27:20 | 11:8
92          int family = cpuid >> 16 & 0xff0 | cpuid >> 8 & 0xf;
93          // model is bits 19:16 | 7:4
94          int model = cpuid >> 12 & 0xf0 | cpuid >> 4 & 0xf;
95          // stepping is bits 3:0
96          int stepping = cpuid & 0xf;
97          return new Triplet<>(family, model, stepping);
98      }

100     @Override
101     protected long[] queryCurrentFreq() {
102         long[] freq = new long[1];
103         int[] mib = { CTL_HW, HW_CPUSPEED };
104         freq[0] = sysctl(mib, 0) * 1_000_000L;
105         return freq;
106     }
107 
108     @Override
109     protected Pair<Long, Long> queryVmStats() {
110         long contextSwitches = 0L;
111         long interrupts = 0L;
112         List<String> vmstat = ExecutingCommand.runNative("vmstat -s");
113         for (String line : vmstat) {
114             if (line.endsWith("cpu context switches")) {
115                 contextSwitches = ParseUtil.parseLongOrDefault(line.trim().split("\\s+")[0], 0L);
116             } else if (line.endsWith("interrupts")) {
117                 interrupts = ParseUtil.parseLongOrDefault(line.trim().split("\\s+")[0], 0L);
118             }
119         }
120         return new Pair<>(contextSwitches, interrupts);
121     }

123     /**
124      * Reads the system-wide CPU time counters ({@code kern.cptime}) as an array of 5 or 6 {@code long} values.
125      *
126      * @return the CPU state ticks, or a shorter/empty array if unavailable
127      */
128     protected abstract long[] querySystemCpTime();
129 
130     /**
131      * Reads a single processor's CPU time counters ({@code kern.cptime2}) as an array of 5 or 6 {@code long} values.
132      *
133      * @param cpu the logical processor index
134      * @return the CPU state ticks, or a shorter/empty array if unavailable
135      */
136     protected abstract long[] queryProcessorCpTime(int cpu);
137 
138     /**
139      * Native {@code getloadavg(3)} call, filling {@code loadavg} with up to {@code nelem} samples.
140      *
141      * @param loadavg the array to populate
142      * @param nelem   the number of elements requested
143      * @return the number of samples retrieved
144      */
145     protected abstract int getloadavgNative(double[] loadavg, int nelem);
146 
147     /**
148      * Maps a 5- or 6-element OpenBSD CPU-state array to the corresponding {@link TickType} slots of {@code ticks}.
149      * OpenBSD 6.4+ adds a spin-time element, giving a 6-element array; the extra element shifts IRQ and IDLE by one.
150      *
151      * @param ticks   the destination tick array
152      * @param cpTicks the source CPU-state values (5 or 6 longs)
153      */
154     private static void fillTicks(long[] ticks, long[] cpTicks) {
155         if (cpTicks.length < CPUSTATES) {
156             return;
157         }
158         ticks[TickType.USER.getIndex()] = cpTicks[CP_USER];
159         ticks[TickType.NICE.getIndex()] = cpTicks[CP_NICE];
160         ticks[TickType.SYSTEM.getIndex()] = cpTicks[CP_SYS];
161         int offset = cpTicks.length > CPUSTATES ? 1 : 0;
162         ticks[TickType.IRQ.getIndex()] = cpTicks[CP_INTR + offset];
163         ticks[TickType.IDLE.getIndex()] = cpTicks[CP_IDLE + offset];
164     }
165 
166     @Override
167     protected long[] querySystemCpuLoadTicks() {
168         long[] ticks = new long[TickType.values().length];
169         fillTicks(ticks, querySystemCpTime());
170         return ticks;
171     }
172 
173     @Override
174     protected long[][] queryProcessorCpuLoadTicks() {
175         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
176         for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
177             fillTicks(ticks[cpu], queryProcessorCpTime(cpu));
178         }
179         return ticks;
180     }
181 
182     @Override
183     public double[] getSystemLoadAverage(int nelem) {
184         if (nelem < 1 || nelem > 3) {
185             throw new IllegalArgumentException("Must include from one to three elements.");
186         }
187         double[] average = new double[nelem];
188         int retval = getloadavgNative(average, nelem);
189         if (retval < nelem) {
190             Arrays.fill(average, -1d);
191         }
192         return average;
193     }
