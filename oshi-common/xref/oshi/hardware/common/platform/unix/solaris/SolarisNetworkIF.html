5   package oshi.hardware.common.platform.unix.solaris;
6   
7   import java.net.NetworkInterface;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.AbstractNetworkIF;
11  
12  /**
13   * Abstract base for Solaris NetworkIF. The interface enumeration ({@link #getNetworks}) is shared; the {@code kstat}
14   * per-interface stats are native and fetched by the JNA and FFM subclasses via {@link #queryStats()}.
15   */
16  @ThreadSafe
17  public abstract class SolarisNetworkIF extends AbstractNetworkIF {
18  
19      /** Per-interface stats the Solaris HAL reads. Both JNA and FFM concrete subclasses populate this. */
20      public static final class IfStats {
21          public long bytesSent;
22          public long bytesRecv;
23          public long packetsSent;
24          public long packetsRecv;
25          public long outErrors;
26          public long inErrors;
27          public long collisions;
28          public long inDrops;
29          public long speed;
30          public long timeStamp;
31      }
32  
33      protected SolarisNetworkIF(NetworkInterface netint) throws InstantiationException {
34          super(netint);
35      }
36  
37      @Override
38      public boolean updateAttributes() {
39          // Initialize to a sane default value
40          this.timeStamp = System.currentTimeMillis();
41          IfStats stats = queryStats();
42          if (stats == null) {
43              return false;
44          }
45          this.bytesSent = stats.bytesSent;
46          this.bytesRecv = stats.bytesRecv;
47          this.packetsSent = stats.packetsSent;
48          this.packetsRecv = stats.packetsRecv;
49          this.outErrors = stats.outErrors;
50          this.inErrors = stats.inErrors;
51          this.collisions = stats.collisions;
52          this.inDrops = stats.inDrops;
53          this.speed = stats.speed;
54          this.timeStamp = stats.timeStamp;
55          return true;
56      }
57  
58      /**
59       * Looks up this interface's stats from the subclass's kstat data source.
60       *
61       * @return stats POJO, or {@code null} if no entry was found for this interface name
62       */
63      protected abstract IfStats queryStats();
64  }
