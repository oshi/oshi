5   package oshi.hardware.common.platform.windows;
6   
7   import java.net.NetworkInterface;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.AbstractNetworkIF;
11  
12  /**
13   * Common Windows NetworkIF logic shared between the JNA and FFM implementations. The per-interface statistics are
14   * gathered from the IP Helper API by the subclasses (JNA {@code MIB_IF_ROW2}/{@code MIB_IFROW} structures or FFM
15   * {@code GetIfEntry2} offsets) into an {@link IfStats} carrier; mapping those onto the interface fields lives here.
16   */
17  @ThreadSafe
18  public abstract class WindowsNetworkIF extends AbstractNetworkIF {
19  
20      /** Bit in the interface flags indicating a connector is present. */
21      protected static final byte CONNECTOR_PRESENT_BIT = 0b00000100;
22  
23      private int ifType;
24      private int ndisPhysicalMediumType;
25      private boolean connectorPresent;
26      private String ifAlias = "";
27      private IfOperStatus ifOperStatus = IfOperStatus.UNKNOWN;
28  
29      /**
30       * Creates a WindowsNetworkIF.
31       *
32       * @param netint the network interface
33       * @throws InstantiationException if the interface cannot be instantiated
34       */
35      protected WindowsNetworkIF(NetworkInterface netint) throws InstantiationException {
36          super(netint);
37          updateAttributes();
38      }
39  
40      /** Interface statistics gathered by a backend from the IP Helper API. */
41      public static final class IfStats {
42          public int ifType;
43          public int ndisPhysicalMediumType;
44          public boolean connectorPresent;
45          public long bytesSent;
46          public long bytesRecv;
47          public long packetsSent;
48          public long packetsRecv;
49          public long outErrors;
50          public long inErrors;
51          public long collisions;
52          public long inDrops;
53          public long speed;
54          public String ifAlias = "";
55          public IfOperStatus ifOperStatus = IfOperStatus.UNKNOWN;
56      }
57  
58      @Override
59      public int getIfType() {
60          return this.ifType;
61      }
62  
63      @Override
64      public int getNdisPhysicalMediumType() {
65          return this.ndisPhysicalMediumType;
66      }
67  
68      @Override
69      public boolean isConnectorPresent() {
70          return this.connectorPresent;
71      }
72  
73      @Override
74      public String getIfAlias() {
75          return ifAlias;
76      }
77  
78      @Override
79      public IfOperStatus getIfOperStatus() {
80          return ifOperStatus;
81      }
82  
83      @Override
84      public boolean updateAttributes() {
85          IfStats stats = queryStats();
86          if (stats == null) {
87              return false;
88          }
89          this.ifType = stats.ifType;
90          this.ndisPhysicalMediumType = stats.ndisPhysicalMediumType;
91          this.connectorPresent = stats.connectorPresent;
92          this.bytesSent = stats.bytesSent;
93          this.bytesRecv = stats.bytesRecv;
94          this.packetsSent = stats.packetsSent;
95          this.packetsRecv = stats.packetsRecv;
96          this.outErrors = stats.outErrors;
97          this.inErrors = stats.inErrors;
98          this.collisions = stats.collisions;
99          this.inDrops = stats.inDrops;
100         this.speed = stats.speed;
101         this.ifAlias = stats.ifAlias;
102         this.ifOperStatus = stats.ifOperStatus;
103         this.timeStamp = System.currentTimeMillis();
104         return true;
105     }
106 
107     /**
108      * Queries this interface's statistics from the native (JNA or FFM) IP Helper API.
109      *
110      * @return the stats, or {@code null} if the query failed
111      */
112     protected abstract IfStats queryStats();
113 }
