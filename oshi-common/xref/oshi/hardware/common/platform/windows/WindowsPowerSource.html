5   package oshi.hardware.common.platform.windows;
6   
7   import java.time.LocalDate;
8   
9   import oshi.annotation.concurrent.ThreadSafe;
10  import oshi.hardware.common.AbstractPowerSource;
11  
12  /**
13   * A Power Source
14   */
15  @ThreadSafe
16  public abstract class WindowsPowerSource extends AbstractPowerSource {
17  
18      // IOCTL codes
19      protected static final int IOCTL_BATTERY_QUERY_TAG = 0x294040;
20      protected static final int IOCTL_BATTERY_QUERY_INFORMATION = 0x294044;
21      protected static final int IOCTL_BATTERY_QUERY_STATUS = 0x29404c;
22  
23      // Capability flags (BATTERY_INFORMATION.Capabilities / BATTERY_STATUS.PowerState)
24      protected static final int BATTERY_SYSTEM_BATTERY = 0x80000000;
25      protected static final int BATTERY_IS_SHORT_TERM = 0x20000000;
26      protected static final int BATTERY_CAPACITY_RELATIVE = 0x40000000;
27      protected static final int BATTERY_POWER_ON_LINE = 0x00000001;
28      protected static final int BATTERY_DISCHARGING = 0x00000002;
29      protected static final int BATTERY_CHARGING = 0x00000004;
30  
31      // BATTERY_QUERY_INFORMATION_LEVEL ordinals
32      protected static final int BATTERY_INFORMATION_LEVEL = 0;
33      protected static final int BATTERY_TEMPERATURE_LEVEL = 2;
34      protected static final int BATTERY_ESTIMATED_TIME_LEVEL = 3;
35      protected static final int BATTERY_DEVICE_NAME_LEVEL = 4;
36      protected static final int BATTERY_MANUFACTURE_DATE_LEVEL = 5;
37      protected static final int BATTERY_MANUFACTURE_NAME_LEVEL = 6;
38      protected static final int BATTERY_SERIAL_NUMBER_LEVEL = 8;
39  
40      protected WindowsPowerSource(String psName, String psDeviceName, double psRemainingCapacityPercent,
41              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
42              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
43              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
44              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
45              String psSerialNumber, double psTemperature) {
46          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
47                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
48                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
49                  psManufacturer, psSerialNumber, psTemperature);
50      }
51  
52  }
