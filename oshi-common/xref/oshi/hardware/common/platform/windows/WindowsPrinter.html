5   package oshi.hardware.common.platform.windows;
6   
7   import oshi.annotation.concurrent.Immutable;
8   import oshi.hardware.Printer;
9   import oshi.hardware.common.AbstractPrinter;
10  
11  /**
12   * Common Windows printer logic shared between JNA and FFM implementations.
13   */
14  @Immutable
15  public abstract class WindowsPrinter extends AbstractPrinter {
16  
17      // DetectedErrorState: 0=Unknown, 1=Other, 2=No Error, 3=Low Paper, 4=No Paper,
18      // 5=Low Toner, 6=No Toner, 7=Door Open, 8=Jammed, 9=Offline, 10=Service Requested, 11=Output Bin Full
19      private static final String[] ERROR_STATE_NAMES = { "Unknown", "Other", "", "Low Paper", "No Paper", "Low Toner",
20              "No Toner", "Door Open", "Jammed", "Offline", "Service Requested", "Output Bin Full" };
21  
22      protected WindowsPrinter(String name, String driverName, String description, PrinterStatus status,
23              String statusReason, boolean isDefault, boolean isLocal, String portName) {
24          super(name, driverName, description, status, statusReason, isDefault, isLocal, portName);
25      }
26  
27      /**
28       * Parses Win32_Printer PrinterStatus and DetectedErrorState codes.
29       *
30       * @param statusCode The PrinterStatus code from WMI.
31       * @param errorState The DetectedErrorState code from WMI.
32       * @return The corresponding PrinterStatus enum value.
33       */
34      protected static Printer.PrinterStatus parseStatus(int statusCode, int errorState) {
35          if (errorState == 4 || errorState == 6 || errorState >= 7) {
36              return Printer.PrinterStatus.ERROR;
37          }
38          switch (statusCode) {
39              case 1: // Other
40              case 2: // Unknown
41                  return Printer.PrinterStatus.UNKNOWN;
42              case 3: // Idle
43                  return Printer.PrinterStatus.IDLE;
44              case 4: // Printing
45                  return Printer.PrinterStatus.PRINTING;
46              case 5: // Warmup
47                  return Printer.PrinterStatus.IDLE;
48              case 6: // Stopped Printing
49              case 7: // Offline
50                  return Printer.PrinterStatus.OFFLINE;
51              default:
52                  return Printer.PrinterStatus.UNKNOWN;
53          }
54      }
55  
56      /**
57       * Converts a DetectedErrorState code to a human-readable string.
58       *
59       * @param errorState The DetectedErrorState code from WMI.
60       * @return A string describing the error state.
61       */
62      protected static String parseErrorState(int errorState) {
63          if (errorState >= 0 && errorState < ERROR_STATE_NAMES.length) {
64              return ERROR_STATE_NAMES[errorState];
65          }
66          return "";
67      }
68  }
