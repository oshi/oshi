24  /**
25   * Windows USB device. The controller device tree (controller ids + parent/name/deviceId/manufacturer maps) is queried
26   * by the backend — JNA {@code DeviceTree} or FFM {@code DeviceTreeFFM} — and supplied to {@link #getUsbDevices}; the
27   * recursive tree assembly is shared between those backends here.
28   * <p>
29   * This intentionally does not use the shared {@code AbstractUsbDevice.buildDeviceTree}: the Windows model keys each
30   * child to its parent (rather than parent-to-children), parses vendor/product/serial from PnP device-id strings, and
31   * inherits serials from the parent.
32   */
33  @Immutable
34  public final class WindowsUsbDevice extends AbstractUsbDevice {
35  
36      private WindowsUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
37              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
38          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
39      }
40  
41      /**
42       * Builds the USB controller device tree from a backend-supplied device-tree query.
43       *
44       * @param deviceTreeSupplier supplies the controller device tree: (controller ids, child→parent map, name map,
45       *                           deviceId map, manufacturer map)
46       * @return a list of USB controllers, each with its connected-device tree
47       */
48      public static List<UsbDevice> getUsbDevices(
49              Supplier<Quintet<Set<Integer>, Map<Integer, Integer>, Map<Integer, String>, Map<Integer, String>, Map<Integer, String>>> deviceTreeSupplier) {
50          Quintet<Set<Integer>, Map<Integer, Integer>, Map<Integer, String>, Map<Integer, String>, Map<Integer, String>> controllerDevices = deviceTreeSupplier
51                  .get();
52          Map<Integer, Integer> parentMap = controllerDevices.getB();
53          Map<Integer, String> nameMap = controllerDevices.getC();
54          Map<Integer, String> deviceIdMap = controllerDevices.getD();
55          Map<Integer, String> mfgMap = controllerDevices.getE();
56  
57          List<UsbDevice> usbDevices = new ArrayList<>();
58          // recursively build results
59          for (Integer controllerDevice : controllerDevices.getA()) {
60              WindowsUsbDevice deviceAndChildren = queryDeviceAndChildren(controllerDevice, parentMap, nameMap,
61                      deviceIdMap, mfgMap, "0000", "0000", "");
62              if (deviceAndChildren != null) {
63                  usbDevices.add(deviceAndChildren);
64              }
65          }
66          return usbDevices;
67      }
68  
69      private static WindowsUsbDevice queryDeviceAndChildren(Integer device, Map<Integer, Integer> parentMap,
70              Map<Integer, String> nameMap, Map<Integer, String> deviceIdMap, Map<Integer, String> mfgMap, String vid,
71              String pid, String parentSerial) {
72          // Parse vendor and product IDs from the device ID; fall back to the parent's IDs if that fails
73          String vendorId = vid;
74          String productId = pid;
75          String serial = parentSerial;
76          Triplet<String, String, String> idsAndSerial = ParseUtil
77                  .parseDeviceIdToVendorProductSerial(deviceIdMap.get(device));
78          if (idsAndSerial != null) {
79              vendorId = idsAndSerial.getA();
80              productId = idsAndSerial.getB();
81              serial = idsAndSerial.getC();
82              if (serial.isEmpty() && vendorId.equals(vid) && productId.equals(pid)) {
83                  serial = parentSerial;
84              }
85          }
86          // Recursively find children in the parent map
87          Set<Integer> childDeviceSet = parentMap.entrySet().stream().filter(e -> e.getValue().equals(device))
88                  .map(Entry::getKey).collect(Collectors.toSet());
89          List<UsbDevice> childDevices = new ArrayList<>();
90          for (Integer child : childDeviceSet) {
91              WindowsUsbDevice deviceAndChildren = queryDeviceAndChildren(child, parentMap, nameMap, deviceIdMap, mfgMap,
92                      vendorId, productId, serial);
93              if (deviceAndChildren != null) {
94                  childDevices.add(deviceAndChildren);
95              }
96          }
97          sort(childDevices);
98          if (nameMap.containsKey(device)) {
99              String name = nameMap.get(device);
100             if (name.isEmpty()) {
101                 name = vendorId + ":" + productId;
102             }
103             String deviceId = deviceIdMap.get(device);
104             String mfg = mfgMap.get(device);
105             return new WindowsUsbDevice(name, mfg, vendorId, productId, serial, deviceId, childDevices);
106         }
107         return null;
108     }
