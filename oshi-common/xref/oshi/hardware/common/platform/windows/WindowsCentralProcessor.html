63      /**
64       * Checks whether the OS version is Windows 8 or greater using system property 'os.version'.
65       *
66       * @return true if Windows 8 or greater
67       */
68      private static boolean isWindows8OrGreater() {
69          // Use system property check that works without native access
70          String osVersion = System.getProperty("os.version", "");
71          String[] parts = osVersion.split("\\.");
72          if (parts.length >= 2) {
73              int major = ParseUtil.parseIntOrDefault(parts[0], 0);
74              int minor = ParseUtil.parseIntOrDefault(parts[1], 0);
75              return major > 6 || (major == 6 && minor >= 2);
76          }
77          return false;
78      }

107     /**
108      * Builds the numaNodeProcToLogicalProcMap from the logical processor list.
109      *
110      * @param logProcs the list of logical processors
111      */
112     protected void buildNumaNodeProcMap(List<LogicalProcessor> logProcs) {
113         Map<Integer, Integer> nextProcIndexByNode = new HashMap<>();
114         int lp = 0;
115         this.numaNodeProcToLogicalProcMap = new HashMap<>();
116         for (LogicalProcessor logProc : logProcs) {
117             int node = logProc.getNumaNode();
118             int procNum = nextProcIndexByNode.getOrDefault(node, 0);
119             numaNodeProcToLogicalProcMap.put(String.format(Locale.ROOT, "%d,%d", node, procNum), lp++);
120             nextProcIndexByNode.put(node, procNum + 1);
121         }
122     }

124     /**
125      * Parses identifier string
126      *
127      * @param identifier the full identifier string
128      * @param key        the key to retrieve
129      * @return the string following id
130      */
131     protected static String parseIdentifier(String identifier, String key) {
132         String[] idSplit = ParseUtil.whitespaces.split(identifier);
133         boolean found = false;
134         for (String s : idSplit) {
135             if (found) {
136                 return s;
137             }
138             found = s.equals(key);
139         }
140         return "";
141     }

143     /**
144      * Lazily calculate the capacity tick multiplier once.
145      *
146      * @param deltaBase The difference in base ticks.
147      * @param deltaT    The difference in elapsed 100NS time
148      * @return The ratio of elapsed time to base ticks
149      */
150     protected synchronized long lazilyCalculateMultiplier(long deltaBase, long deltaT) {
151         if (utilityBaseMultiplier == null) {
152             // If too much time has elapsed from class instantiation, re-initialize the
153             // ticks and return without calculating. Approx 7 minutes for 100NS counter to
154             // exceed max unsigned int.
155             if (deltaT >> 32 > 0) {
156                 setInitialUtilityCounters(queryProcessorUtilityCounters());
157                 return 0L;
158             }
159             // Base counter wraps approximately every 115 minutes
160             // If deltaBase is nonpositive assume it has wrapped
161             if (deltaBase <= 0) {
162                 deltaBase += 1L << 32;
163             }
164             long multiplier = Math.round((double) deltaT / deltaBase);
165             // If not enough time has elapsed, return the value this one time but don't
166             // persist. 5000 ms = 50 million 100NS ticks
167             if (deltaT < 50_000_000L) {
168                 return multiplier;
169             }
170             utilityBaseMultiplier = multiplier;
171         }
172         return utilityBaseMultiplier;
173     }

198     @Override
199     public long[][] queryProcessorCpuLoadTicks() {
200         // These are used in all cases
201         List<String> instances;
202         List<Long> systemList;
203         List<Long> userList;
204         List<Long> irqList;
205         List<Long> softIrqList;
206         List<Long> idleList;
207         // These are only used with USE_CPU_UTILITY
208         List<Long> baseList = null;
209         List<Long> systemUtility = null;
210         List<Long> processorUtility = null;
211         List<Long> processorUtilityBase = null;
212         List<Long> initSystemList = null;
213         List<Long> initUserList = null;
214         List<Long> initBase = null;
215         List<Long> initSystemUtility = null;
216         List<Long> initProcessorUtility = null;
217         List<Long> initProcessorUtilityBase = null;
218         if (USE_CPU_UTILITY) {
219             Pair<List<String>, Map<ProcessorUtilityTickCountProperty, List<Long>>> instanceValuePair = processorUtilityCounters
220                     .get();
221             instances = instanceValuePair.getA();
222             Map<ProcessorUtilityTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
223             systemList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
224             userList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
225             irqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTINTERRUPTTIME);
226             softIrqList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTDPCTIME);
227             // % Processor Time is actually Idle time
228             idleList = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORTIME);
229             baseList = valueMap.get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
230             // Utility ticks, if configured
231             systemUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
232             processorUtility = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
233             processorUtilityBase = valueMap.get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
234 
235             initSystemList = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDTIME);
236             initUserList = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.PERCENTUSERTIME);
237             initBase = getInitialUtilityCounters().get(ProcessorUtilityTickCountProperty.TIMESTAMP_SYS100NS);
238             initSystemUtility = getInitialUtilityCounters()
239                     .get(ProcessorUtilityTickCountProperty.PERCENTPRIVILEGEDUTILITY);
240             initProcessorUtility = getInitialUtilityCounters()
241                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY);
242             initProcessorUtilityBase = getInitialUtilityCounters()
243                     .get(ProcessorUtilityTickCountProperty.PERCENTPROCESSORUTILITY_BASE);
244         } else {
245             Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = queryProcessorCounters();
246             instances = instanceValuePair.getA();
247             Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
248             systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
249             userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
250             irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
251             softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
252             // % Processor Time is actually Idle time
253             idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
254         }
255 
256         return processTickData(instances, systemList, userList, irqList, softIrqList, idleList, baseList, systemUtility,
257                 processorUtility, processorUtilityBase, initSystemList, initUserList, initBase, initSystemUtility,
258                 initProcessorUtility, initProcessorUtilityBase);
259     }

261     /**
262      * Processes raw perfmon tick data into the standard tick array format. Handles both legacy and utility-based
263      * approaches.
264      *
265      * @param instances                the perfmon instance names
266      * @param systemList               system (privileged) time ticks
267      * @param userList                 user time ticks
268      * @param irqList                  IRQ time ticks
269      * @param softIrqList              soft IRQ (DPC) time ticks
270      * @param idleList                 idle time ticks
271      * @param baseList                 timestamp base (for utility mode), may be null
272      * @param systemUtility            system utility ticks, may be null
273      * @param processorUtility         processor utility ticks, may be null
274      * @param processorUtilityBase     processor utility base, may be null
275      * @param initSystemList           initial system ticks (for utility mode), may be null
276      * @param initUserList             initial user ticks (for utility mode), may be null
277      * @param initBase                 initial base (for utility mode), may be null
278      * @param initSystemUtility        initial system utility, may be null
279      * @param initProcessorUtility     initial processor utility, may be null
280      * @param initProcessorUtilityBase initial processor utility base, may be null
281      * @return the processed tick array
282      */
283     protected long[][] processTickData(List<String> instances, List<Long> systemList, List<Long> userList,
284             List<Long> irqList, List<Long> softIrqList, List<Long> idleList, List<Long> baseList,
285             List<Long> systemUtility, List<Long> processorUtility, List<Long> processorUtilityBase,
286             List<Long> initSystemList, List<Long> initUserList, List<Long> initBase, List<Long> initSystemUtility,
287             List<Long> initProcessorUtility, List<Long> initProcessorUtilityBase) {
288 
289         int ncpu = getLogicalProcessorCount();
290         long[][] ticks = new long[ncpu][TickType.values().length];
291         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
292                 || idleList == null
293                 || (USE_CPU_UTILITY && (baseList == null || systemUtility == null || processorUtility == null
294                         || processorUtilityBase == null || initSystemList == null || initUserList == null
295                         || initBase == null || initSystemUtility == null || initProcessorUtility == null
296                         || initProcessorUtilityBase == null))) {
297             return ticks;
298         }
299         int size = instances.size();
300         if (systemList.size() < size || userList.size() < size || irqList.size() < size || softIrqList.size() < size
301                 || idleList.size() < size) {
302             return ticks;
303         }
304         if (USE_CPU_UTILITY && (baseList.size() < size || systemUtility.size() < size || processorUtility.size() < size
305                 || processorUtilityBase.size() < size || initSystemList.size() < size || initUserList.size() < size
306                 || initBase.size() < size || initSystemUtility.size() < size || initProcessorUtility.size() < size
307                 || initProcessorUtilityBase.size() < size)) {
308             return ticks;
309         }
310         for (int i = 0; i < instances.size(); i++) {
311             String instance = instances.get(i);
312             int cpu;
313             if (instance.contains(",")) {
314                 if (!getNumaNodeProcToLogicalProcMap().containsKey(instance)) {
315                     continue;
316                 }
317                 cpu = getNumaNodeProcToLogicalProcMap().get(instance);
318             } else {
319                 cpu = ParseUtil.parseIntOrDefault(instance, -1);
320             }
321             if (cpu < 0 || cpu >= ncpu) {
322                 continue;
323             }
324             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(i);
325             ticks[cpu][TickType.USER.getIndex()] = userList.get(i);
326             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(i);
327             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(i);
328             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(i);
329 
330             if (USE_CPU_UTILITY) {
331                 long deltaT = baseList.get(i) - initBase.get(i);
332                 if (deltaT > 0) {
333                     long deltaBase = processorUtilityBase.get(i) - initProcessorUtilityBase.get(i);
334                     long multiplier = lazilyCalculateMultiplier(deltaBase, deltaT);
335                     if (multiplier > 0) {
336                         long deltaProc = processorUtility.get(i) - initProcessorUtility.get(i);
337                         long deltaSys = systemUtility.get(i) - initSystemUtility.get(i);
338                         long newUser = initUserList.get(i) + multiplier * (deltaProc - deltaSys) / 100;
339                         long newSystem = initSystemList.get(i) + multiplier * deltaSys / 100;
340                         long delta = newUser - ticks[cpu][TickType.USER.getIndex()];
341                         ticks[cpu][TickType.USER.getIndex()] = newUser;
342                         delta += newSystem - ticks[cpu][TickType.SYSTEM.getIndex()];
343                         ticks[cpu][TickType.SYSTEM.getIndex()] = newSystem;
344                         ticks[cpu][TickType.IDLE.getIndex()] -= delta;
345                     }
346                 }
347             }
348 
349             // Decrement IRQ from system to avoid double counting in the total array
350             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
351                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
352 
353             // Raw value is cumulative 100NS-ticks
354             // Divide by 10_000 to get milliseconds
355             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
356             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
357             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
358             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
359             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
360         }
361         return ticks;
362     }
