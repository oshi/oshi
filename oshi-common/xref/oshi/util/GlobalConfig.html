12  /**
13   * The global configuration utility. See {@code src/main/resources/oshi.properties} for default values.
14   * <p>
15   * Configuration values set as Java System Properties using {@link System#setProperty(String, String)} will override
16   * values from the {@code oshi.properties} file, but may then be later altered using {@link #set(String, Object)} or
17   * {@link #remove(String)}.
18   * <p>
19   * This class is not thread safe if methods manipulating the configuration are used. These methods are intended for use
20   * by a single thread at startup, before instantiation of any other OSHI classes. OSHI does not guarantee re- reading of
21   * any configuration changes.
22   */
23  @NotThreadSafe
24  public final class GlobalConfig {
25  
26      private static final String OSHI_PROPERTIES = "oshi.properties";
27  
28      private static final Properties CONFIG = FileUtil.readPropertiesFromFilename(OSHI_PROPERTIES);
29      static {
30          System.getProperties().forEach((k, v) -> {
31              String key = k.toString();
32              if (key.startsWith("oshi.")) {
33                  set(key, v);
34              }
35          });
36      }

112      * Whether to perform WMI queries for command lines in a batch for all running processes. Individual WMI queries

116     public static final String OSHI_OS_WINDOWS_COMMANDLINE_BATCH = "oshi.os.windows.commandline.batch";
117     /**
118      * Whether to attempt to fetch Windows performance counter data for processes and threads from

131      * Whether to start a daemon thread to provide load averages on Windows. While Windows does not natively provide

133      * similar metric. Default is {@code false}.
134      *
135      * @see CentralProcessor#getSystemLoadAverage(int)

137     public static final String OSHI_OS_WINDOWS_LOADAVERAGE = "oshi.os.windows.loadaverage";
138     /**
139      * Whether to use "Processor Utility" for System and per-processor CPU Load ticks (on Windows 8 and higher) to match
140      * CPU usage with the Windows Task Manager.
141      * <p>
142      * By default, OSHI uses "Processor Time" which measures active and idle times, matching Unix-based systems and
143      * Windows 7 Task Manager. Setting this to {@code true} switches to "Processor Utility" which measures work
144      * completed relative to nominal frequency. Because of features like Intel Turbo Boost and AMD Precision Boost, this
145      * value can exceed 100% (though Task Manager caps it). OSHI does not cap the value.
146      * <p>
147      * For this calculation to produce meaningful data, the ticks must come from the same CentralProcessor instance and
148      * the first polling interval must be less than 7 minutes. Default is {@code false}.
149      *
150      * @see CentralProcessor#getSystemCpuLoadTicks()
151      */
152     public static final String OSHI_OS_WINDOWS_CPU_UTILITY = "oshi.os.windows.cpu.utility";

233     /**
234      * Get the property associated with the given key.
235      *
236      * @param key The property key
237      * @return The property value if it exists, or null otherwise
238      */
239     public static String get(String key) {
240         return CONFIG.getProperty(key);
241     }
242 
243     /**
244      * Get the {@code String} property associated with the given key.
245      *
246      * @param key The property key
247      * @param def The default value
248      * @return The property value or the given default if not found
249      */
250     public static String get(String key, String def) {
251         return CONFIG.getProperty(key, def);
252     }
253 
254     /**
255      * Get the {@code int} property associated with the given key.
256      *
257      * @param key The property key
258      * @param def The default value
259      * @return The property value or the given default if not found
260      */
261     public static int get(String key, int def) {
262         String value = CONFIG.getProperty(key);
263         return value == null ? def : ParseUtil.parseIntOrDefault(value, def);
264     }
265 
266     /**
267      * Get the {@code double} property associated with the given key.
268      *
269      * @param key The property key
270      * @param def The default value
271      * @return The property value or the given default if not found
272      */
273     public static double get(String key, double def) {
274         String value = CONFIG.getProperty(key);
275         return value == null ? def : ParseUtil.parseDoubleOrDefault(value, def);
276     }
277 
278     /**
279      * Get the {@code boolean} property associated with the given key.
280      *
281      * @param key The property key
282      * @param def The default value
283      * @return The property value or the given default if not found
284      */
285     public static boolean get(String key, boolean def) {
286         String value = CONFIG.getProperty(key);
287         return value == null ? def : Boolean.parseBoolean(value);
288     }
289 
290     /**
291      * Set the given property, overwriting any existing value. If the given value is {@code null}, the property is
292      * removed.
293      *
294      * @param key The property key
295      * @param val The new value
296      */
297     public static void set(String key, Object val) {
298         if (val == null) {
299             CONFIG.remove(key);
300         } else {
301             CONFIG.setProperty(key, val.toString());
302         }
303     }
304 
305     /**
306      * Reset the given property to its default value.
307      *
308      * @param key The property key
309      */
310     public static void remove(String key) {
311         CONFIG.remove(key);
312     }
313 
314     /**
315      * Clear the configuration.
316      */
317     public static void clear() {
318         CONFIG.clear();
319     }
320 
321     /**
322      * Load the given {@link java.util.Properties} into the global configuration.
323      *
324      * @param properties The new properties
325      */
326     public static void load(Properties properties) {
327         CONFIG.putAll(properties);
328     }

330     /**
331      * Indicates that a configuration value is invalid.
332      */
333     public static class PropertyException extends RuntimeException {
334 
335         private static final long serialVersionUID = -7482581936621748005L;
336 
337         /**
338          * @param property The property name
339          */
340         public PropertyException(String property) {
341             super("Invalid property: \"" + property + "\" = " + GlobalConfig.get(property, null));
342         }
343 
344         /**
345          * @param property The property name
346          * @param message  An exception message
347          */
348         public PropertyException(String property, String message) {
349             super("Invalid property \"" + property + "\": " + message);
350         }
351     }
