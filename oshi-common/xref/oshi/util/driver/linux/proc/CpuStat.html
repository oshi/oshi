15  /**
16   * Utility to read CPU statistics from {@code /proc/stat}
17   */
18  @ThreadSafe
19  public final class CpuStat {
20  
21      private CpuStat() {
22      }
23  
24      /**
25       * Gets the System CPU ticks array from {@code /proc/stat}
26       *
27       * @return Array of CPU ticks
28       */
29      public static long[] getSystemCpuLoadTicks() {
30          return parseSystemCpuLoadTicks(FileUtil.readLines(ProcPath.STAT, 1));
31      }
32  
33      /**
34       * Parses the overall CPU ticks from the first line of {@code /proc/stat}. Package-private for testing.
35       *
36       * @param procStat the first line(s) of {@code /proc/stat}
37       * @return array of CPU ticks
38       */
39      static long[] parseSystemCpuLoadTicks(List<String> procStat) {
40          long[] ticks = new long[TickType.values().length];
41          // /proc/stat expected format
42          // first line is overall user,nice,system,idle,iowait,irq, etc.
43          // cpu 3357 0 4313 1362393 ...
44          if (procStat.isEmpty()) {
45              return ticks;
46          }
47          String tickStr = procStat.get(0);
48  
49          // Split the line. Note the first (0) element is "cpu" so remaining
50          // elements are offset by 1 from the enum index
51          String[] tickArr = ParseUtil.whitespaces.split(tickStr);
52          if (tickArr.length <= TickType.IDLE.getIndex()) {
53              // If ticks don't at least go user/nice/system/idle, abort
54              return ticks;
55          }
56          // Note tickArr is offset by 1 because first element is "cpu". Stop if a truncated line runs out of
57          // fields, leaving the zero defaults for any missing values.
58          for (int i = 0; i < TickType.values().length && i + 1 < tickArr.length; i++) {
59              ticks[i] = ParseUtil.parseLongOrDefault(tickArr[i + 1], 0L);
60          }
61          // Ignore guest or guest_nice, they are included in user/nice
62          return ticks;
63      }
64  
65      /**
66       * Gets an array of Processor CPU ticks array from /proc/stat
67       *
68       * @param logicalProcessorCount The number of logical processors, which corresponds to the number of lines to read
69       *                              from the file.
70       * @return Array of CPU ticks for each processor
71       */
72      public static long[][] getProcessorCpuLoadTicks(int logicalProcessorCount) {
73          return parseProcessorCpuLoadTicks(FileUtil.readFile(ProcPath.STAT), logicalProcessorCount);
74      }
75  
76      /**
77       * Parses the per-processor CPU ticks from {@code /proc/stat} lines. Package-private for testing.
78       *
79       * @param procStat              the lines of {@code /proc/stat}
80       * @param logicalProcessorCount the number of logical processors
81       * @return array of CPU ticks for each processor
82       */
83      static long[][] parseProcessorCpuLoadTicks(List<String> procStat, int logicalProcessorCount) {
84          long[][] ticks = new long[logicalProcessorCount][TickType.values().length];
85          // /proc/stat expected format
86          // first line is overall user,nice,system,idle, etc.
87          // cpu 3357 0 4313 1362393 ...
88          // per-processor subsequent lines for cpu0, cpu1, etc.
89          int cpu = 0;
90          for (String stat : procStat) {
91              if (stat.startsWith("cpu") && !stat.startsWith("cpu ")) {
92                  // Split the line. Note the first (0) element is "cpu" so
93                  // remaining
94                  // elements are offset by 1 from the enum index
95                  String[] tickArr = ParseUtil.whitespaces.split(stat);
96                  if (tickArr.length <= TickType.IDLE.getIndex()) {
97                      // If ticks don't at least go user/nice/system/idle, abort
98                      return ticks;
99                  }
100                 // Note tickArr is offset by 1. Stop if a truncated line runs out of fields, leaving the zero
101                 // defaults for any missing values.
102                 for (int i = 0; i < TickType.values().length && i + 1 < tickArr.length; i++) {
103                     ticks[cpu][i] = ParseUtil.parseLongOrDefault(tickArr[i + 1], 0L);
104                 }
105                 // Ignore guest or guest_nice, they are included in
106                 if (++cpu >= logicalProcessorCount) {
107                     break;
108                 }
109             }
110         }
111         return ticks;
112     }

114     /**
115      * Gets the number of context switches from /proc/stat
116      *
117      * @return The number of context switches if available, -1 otherwise
118      */
119     public static long getContextSwitches() {
120         return parseContextSwitches(FileUtil.readFile(ProcPath.STAT));
121     }
122 
123     /**
124      * Parses the number of context switches from the {@code ctxt} line of {@code /proc/stat}. Package-private for
125      * testing.
126      *
127      * @param procStat the lines of {@code /proc/stat}
128      * @return the number of context switches, or 0 if unavailable
129      */
130     static long parseContextSwitches(List<String> procStat) {
131         for (String stat : procStat) {
132             if (stat.startsWith("ctxt ")) {
133                 String[] ctxtArr = ParseUtil.whitespaces.split(stat);
134                 if (ctxtArr.length == 2) {
135                     return ParseUtil.parseLongOrDefault(ctxtArr[1], 0);
136                 }
137             }
138         }
139         return 0L;
140     }
141 
142     /**
143      * Gets the number of interrupts from /proc/stat
144      *
145      * @return The number of interrupts if available, -1 otherwise
146      */
147     public static long getInterrupts() {
148         return parseInterrupts(FileUtil.readFile(ProcPath.STAT));
149     }
150 
151     /**
152      * Parses the number of interrupts from the {@code intr} line of {@code /proc/stat}. Package-private for testing.
153      *
154      * @param procStat the lines of {@code /proc/stat}
155      * @return the number of interrupts, or 0 if unavailable
156      */
157     static long parseInterrupts(List<String> procStat) {
158         for (String stat : procStat) {
159             if (stat.startsWith("intr ")) {
160                 String[] intrArr = ParseUtil.whitespaces.split(stat);
161                 if (intrArr.length > 2) {
162                     return ParseUtil.parseLongOrDefault(intrArr[1], 0);
163                 }
164             }
165         }
166         return 0L;
167     }
168 
169     /**
170      * Gets the boot time from /proc/stat
171      *
172      * @return The boot time if available, 0 otherwise
173      */
174     public static long getBootTime() {
175         return parseBootTime(FileUtil.readFile(ProcPath.STAT));
176     }
177 
178     /**
179      * Parses the boot time from the {@code btime} line of {@code /proc/stat}. Package-private for testing.
180      *
181      * @param procStat the lines of {@code /proc/stat}
182      * @return the boot time in seconds since the epoch, or 0 if unavailable
183      */
184     static long parseBootTime(List<String> procStat) {
185         // Boot time given by btime variable in /proc/stat.
186         for (String stat : procStat) {
187             if (stat.startsWith("btime")) {
188                 String[] bTime = ParseUtil.whitespaces.split(stat);
189                 if (bTime.length >= 2) {
190                     return ParseUtil.parseLongOrDefault(bTime[1], 0L);
191                 }
192             }
193         }
194         return 0;
195     }
