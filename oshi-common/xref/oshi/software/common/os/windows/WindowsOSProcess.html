184     @Override
185     public List<OSThread> getThreadDetails() {
186         Map<Integer, ThreadPerfCounterBlock> threads = tcb == null
187                 ? queryMatchingThreads(Collections.singleton(this.getProcessID()))
188                 : tcb;
189         if (threads == null) {
190             threads = Collections.emptyMap();
191         }
192         return threads.entrySet().stream().parallel()
193                 .filter(entry -> entry.getValue().getOwningProcessID() == this.getProcessID())
194                 .map(entry -> createOSThread(getProcessID(), entry.getKey(), this.name, entry.getValue()))
195                 .collect(Collectors.toList());
196     }

209     /**
210      * Updates process attributes from performance counter and WTS data, then performs native-specific updates.
211      * Subclasses should call {@code super.updateAttributes(pcb, wts)} and then perform native handle-based updates.
212      *
213      * @param pcb Performance counter block for this process, or null if unavailable
214      * @param wts WTS info for this process, or null if unavailable
215      * @return true if the process is valid after the update
216      */
217     protected boolean updateAttributes(ProcessPerfCounterBlock pcb, WtsInfo wts) {
218         if (pcb == null) {
219             this.state = INVALID;
220             return false;
221         }
222         this.name = pcb.getName();
223         this.parentProcessID = pcb.getParentProcessID();
224         this.priority = pcb.getPriority();
225         this.workingSetSize = pcb.getWorkingSetSize();
226         this.privateWorkingSetSize = pcb.getPrivateWorkingSetSize();
227         this.startTime = pcb.getStartTime();
228         this.upTime = pcb.getUpTime();
229         this.bytesRead = pcb.getBytesRead();
230         this.bytesWritten = pcb.getBytesWritten();
231         this.pageFaults = pcb.getPageFaults();
232         if (wts != null) {
233             this.path = wts.getPath();
234             this.threadCount = wts.getThreadCount();
235             this.virtualSize = wts.getVirtualSize();
236             this.kernelTime = wts.getKernelTime();
237             this.userTime = wts.getUserTime();
238             this.openFiles = wts.getOpenFiles();
239         }
240 
241         this.state = RUNNING;
242         if (this.tcb != null) {
243             int pid = this.getProcessID();
244             for (ThreadPerfCounterBlock tpd : this.tcb.values()) {
245                 if (tpd.getOwningProcessID() == pid) {
246                     if (tpd.getThreadWaitReason() == 5) {
247                         this.state = SUSPENDED;
248                     } else {
249                         this.state = RUNNING;
250                         break;
251                     }
252                 }
253             }
254         }
255 
256         return !this.state.equals(INVALID);
257     }

326     private String queryCwd() {
327         if (!cwdCmdEnv.get().getA().isEmpty()) {
328             return cwdCmdEnv.get().getA();
329         }
330         if (getProcessID() == this.os.getProcessId()) {
331             String cwd = new File(".").getAbsolutePath();
332             if (!cwd.isEmpty()) {
333                 return cwd.substring(0, cwd.length() - 1);
334             }
335         }
336         return "";
337     }
