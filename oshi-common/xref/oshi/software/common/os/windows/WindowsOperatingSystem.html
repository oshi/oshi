5   package oshi.software.common.os.windows;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  import oshi.annotation.concurrent.ThreadSafe;
11  import oshi.software.common.AbstractOperatingSystem;
12  import oshi.util.GlobalConfig;
13  
14  /**
15   * Common base class for Windows operating system implementations.
16   */
17  @ThreadSafe
18  public abstract class WindowsOperatingSystem extends AbstractOperatingSystem {
19  
20      /**
21       * Default constructor.
22       */
23      protected WindowsOperatingSystem() {
24      }
25  
26      protected static final boolean USE_PROCSTATE_SUSPENDED = GlobalConfig
27              .get(GlobalConfig.OSHI_OS_WINDOWS_PROCSTATE_SUSPENDED, false);
28  
29      @Override
30      protected String queryManufacturer() {
31          return "Microsoft";
32      }
33  
34      /**
35       * Gets suites available on the system and return as a codename.
36       *
37       * @param suiteMask The suite mask bitmask
38       * @return Suites
39       */
40      protected static String parseCodeName(int suiteMask) {
41          List<String> suites = new ArrayList<>();
42          if ((suiteMask & 0x00000002) != 0) {
43              suites.add("Enterprise");
44          }
45          if ((suiteMask & 0x00000004) != 0) {
46              suites.add("BackOffice");
47          }
48          if ((suiteMask & 0x00000008) != 0) {
49              suites.add("Communications Server");
50          }
51          if ((suiteMask & 0x00000080) != 0) {
52              suites.add("Datacenter");
53          }
54          if ((suiteMask & 0x00000200) != 0) {
55              suites.add("Home");
56          }
57          if ((suiteMask & 0x00000400) != 0) {
58              suites.add("Web Server");
59          }
60          if ((suiteMask & 0x00002000) != 0) {
61              suites.add("Storage Server");
62          }
63          if ((suiteMask & 0x00004000) != 0) {
64              suites.add("Compute Cluster");
65          }
66          if ((suiteMask & 0x00008000) != 0) {
67              suites.add("Home Server");
68          }
69          return String.join(",", suites);
70      }
71  }
