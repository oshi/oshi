82      // Matches addr= or mountaddr= in NFS mount options; captures up to the next comma
83      // so IPv6 addresses (e.g. addr=2001:db8::1) are not truncated at the first colon.
84      private static final Pattern NFS_ADDR_PATTERN = Pattern.compile("(?:^|,)(?:mount)?addr=([^,]+)");

86      // Maximum number of concurrent NFS reachability-probe threads. Hardcoded: probes are
87      // short-lived blocking I/O waits, not CPU work, so this only bounds the pathological
88      // many-stale-mounts case rather than acting as a throughput throttle.
89      private static final int NFS_PROBE_MAX_THREADS = 64;

107     /**
108      * Builds a map of filesystem UUIDs to device paths.
109      *
110      * @return the UUID map
111      */
112     protected static Map<String, String> buildUuidMap() {
113         // Map of volume with device path as key
114         Map<String, String> volumeDeviceMap = new HashMap<>();
115         File devMapper = new File(DevPath.MAPPER);
116         File[] volumes = devMapper.listFiles();
117         if (volumes != null) {
118             for (File volume : volumes) {
119                 try {
120                     volumeDeviceMap.put(volume.getCanonicalPath(), volume.getAbsolutePath());
121                 } catch (IOException e) {
122                     LOG.debug("Couldn't get canonical path for {}. {}", volume.getName(), e.getMessage());
123                 }
124             }
125         }
126         // Map uuids with device path as key
127         Map<String, String> uuidMap = new HashMap<>();
128         File uuidDir = new File(DevPath.DISK_BY_UUID);
129         File[] uuids = uuidDir.listFiles();
130         if (uuids != null) {
131             for (File uuid : uuids) {
132                 try {
133                     // Store UUID as value with path (e.g., /dev/sda1) and volumes as key
134                     String canonicalPath = uuid.getCanonicalPath();
135                     uuidMap.put(canonicalPath, uuid.getName().toLowerCase(Locale.ROOT));
136                     if (volumeDeviceMap.containsKey(canonicalPath)) {
137                         uuidMap.put(volumeDeviceMap.get(canonicalPath), uuid.getName().toLowerCase(Locale.ROOT));
138                     }
139                 } catch (IOException e) {
140                     LOG.debug("Couldn't get canonical path for {}. {}", uuid.getName(), e.getMessage());
141                 }
142             }
143         }
144         return uuidMap;
145     }

147     List<OSFileStore> getFileStoreMatching(String nameToMatch, Map<String, String> uuidMap, boolean localOnly) {
148         List<OSFileStore> fsList = new ArrayList<>();
149 
150         Map<String, String> labelMap = queryLabelMap();
151 
152         // Parse /proc/mounts to get fs types
153         List<String> mounts = FileUtil.readFile(ProcPath.MOUNTS);
154 
155         // Pre-probe all unique NFS hosts in parallel before the loop. Skipped for localOnly
156         // scans, which filter out network filesystems anyway, so a local-only query never
157         // waits on a stale NFS mount.
158         Map<String, Boolean> nfsHostReachable = CHECK_NFS && !localOnly ? probeNfsHosts(mounts)
159                 : Collections.emptyMap();
160 
161         for (String mount : mounts) {
162             String[] split = mount.split(" ");
163             // As reported in fstab(5) manpage, struct is:
164             // 1st field is volume name
165             // 2nd field is path with spaces escaped as \040
166             // 3rd field is fs type
167             // 4th field is mount options
168             // 5th field is used by dump(8) (ignored)
169             // 6th field is fsck order (ignored)
170             if (split.length < 6) {
171                 continue;
172             }
173 
174             // Exclude pseudo file systems
175             String volume = split[0].replace(UNICODE_SPACE, " ");
176             String name = volume;
177             String path = split[1].replace(UNICODE_SPACE, " ");
178             if (path.equals("/")) {
179                 name = "/";
180             }
181             String type = split[2];
182 
183             // Skip non-local drives if requested, and exclude pseudo file systems
184             boolean isLocal = !NETWORK_FS_TYPES.contains(type);
185             if ((localOnly && !isLocal)
186                     || !path.equals("/") && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path,
187                             volume, FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
188                 continue;
189             }
190 
191             String options = split[3];
192 
193             // If only updating for one name, skip others
194             if (nameToMatch != null && !nameToMatch.equals(name)) {
195                 continue;
196             }
197 
198             String uuid = uuidMap != null ? uuidMap.getOrDefault(split[0], "") : "";
199 
200             String description;
201             if (volume.startsWith(DevPath.DEV)) {
202                 description = "Local Disk";
203             } else if (volume.equals("tmpfs")) {
204                 description = "Ram Disk";
205             } else if (NETWORK_FS_TYPES.contains(type)) {
206                 description = "Network Disk";
207             } else {
208                 description = "Mount Point";
209             }
210 
211             // Add in logical volume found at /dev/mapper, useful when linking
212             // file system with drive.
213             String logicalVolume = "";
214             Path link = Paths.get(volume);
215             if (link.toFile().exists() && Files.isSymbolicLink(link)) {
216                 try {
217                     Path slink = Files.readSymbolicLink(link);
218                     Path full = Paths.get(DevPath.MAPPER + slink.toString());
219                     if (full.toFile().exists()) {
220                         logicalVolume = full.normalize().toString();
221                     }
222                 } catch (IOException e) {
223                     LOG.debug("Couldn't access symbolic path  {}. {}", link, e.getMessage());
224                 }
225             }
226 
227             long totalInodes = 0L;
228             long freeInodes = 0L;
229             long totalSpace = 0L;
230             long usableSpace = 0L;
231             long freeSpace = 0L;
232 
233             // For NFS mounts, skip statvfs if the server was found unreachable during
234             // the parallel pre-probe above. Scoped to nfs/nfs4 only so non-NFS network
235             // filesystems (cifs, fuse, ...) are not probed on port 2049.
236             if (CHECK_NFS && isNfsType(type)) {
237                 String host = parseNfsAddr(options);
238                 if (host != null && Boolean.FALSE.equals(nfsHostReachable.get(host))) {
239                     description = "Network Disk [unreachable]";
240                     fsList.add(new LinuxOSFileStore(name, volume, labelMap.getOrDefault(path, name), path, options,
241                             uuid, isLocal, logicalVolume, description, type, 0L, 0L, 0L, 0L, 0L, this, true));
242                     continue;
243                 }
244             }
245 
246             long[] vfs = queryStatvfs(path);
247             if (vfs != null) {
248                 totalInodes = vfs[0];
249                 freeInodes = vfs[1];
250                 totalSpace = vfs[2];
251                 usableSpace = vfs[3];
252                 freeSpace = vfs[4];
253             }
254             // If native methods failed use JVM methods
255             if (totalSpace == 0L) {
256                 File tmpFile = new File(path);
257                 totalSpace = tmpFile.getTotalSpace();
258                 usableSpace = tmpFile.getUsableSpace();
259                 freeSpace = tmpFile.getFreeSpace();
260             }
261 
262             fsList.add(new LinuxOSFileStore(name, volume, labelMap.getOrDefault(path, name), path, options, uuid,
263                     isLocal, logicalVolume, description, type, freeSpace, usableSpace, totalSpace, freeInodes,
264                     totalInodes, this));
265         }
266         return fsList;
267     }

269     /**
270      * Returns true for mount types that use the NFS protocol and port 2049.
271      *
272      * @param type the filesystem type from {@code /proc/mounts}
273      * @return {@code true} if the type is {@code nfs} or {@code nfs4}, otherwise {@code false}
274      */
275     static boolean isNfsType(String type) {
276         return "nfs".equals(type) || "nfs4".equals(type);
277     }

279     /**
280      * Extracts the NFS server address from mount options.
281      *
282      * @param options the mount options field from {@code /proc/mounts}
283      * @return the server address from the {@code addr=} or {@code mountaddr=} field, or {@code null} if neither is
284      *         present
285      */
286     static String parseNfsAddr(String options) {
287         Matcher m = NFS_ADDR_PATTERN.matcher(options);
288         return m.find() ? m.group(1) : null;
289     }

291     /**
292      * Attempts a TCP connection to {@code host:port} within {@code timeoutMs}.
293      * <p>
294      * Only a completed connection counts as reachable. Any {@link IOException} — connection timeout, connection
295      * refused, host unreachable, or any other network error — is treated as unreachable and returns {@code false}.
296      *
297      * @param host      the server address to probe
298      * @param port      the TCP port to connect to
299      * @param timeoutMs the connection timeout in milliseconds
300      * @return {@code true} only if the connection succeeds; {@code false} on any {@link IOException}
301      */
302     private static boolean tcpReachable(String host, int port, int timeoutMs) {
303         // TCP connect to the standard NFS port (2049).
304         // More reliable than InetAddress.isReachable() which uses ICMP (may need root)
305         // or port 7 (echo, commonly firewalled).
306         try (Socket socket = new Socket()) {
307             socket.connect(new InetSocketAddress(host, port), timeoutMs);
308             return true;
309         } catch (IOException e) {
310             LOG.debug("NFS host {} not reachable on port {}: {}", host, port, e.getMessage());
311             return false;
312         }
313     }

315     /**
316      * Probes the unique NFS server hosts found in {@code mounts} for reachability, in parallel. Only {@code nfs}/
317      * {@code nfs4} mounts that expose an {@code addr=} or {@code mountaddr=} server address are probed; mounts without
318      * a parseable address are omitted and left for the normal {@code statvfs} path.
319      * <p>
320      * Probes run on a fixed pool capped at {@code NFS_PROBE_MAX_THREADS} (64) threads, each with a 2-second connect
321      * timeout. The worst-case wait is therefore roughly {@code 2s * ceil(unreachableHosts / 64)} rather than a strict 2
322      * seconds: up to 64 hosts resolve within one timeout window, and any excess queues into additional windows.
323      *
324      * @param mounts the lines read from {@code /proc/mounts}
325      * @return a map of NFS server address to reachability ({@code true} if reachable, {@code false} otherwise)
326      */
327     private static Map<String, Boolean> probeNfsHosts(List<String> mounts) {
328         Set<String> hosts = new HashSet<>();
329         for (String mount : mounts) {
330             String[] split = mount.split(" ");
331             if (split.length >= 6 && isNfsType(split[2])) {
332                 String host = parseNfsAddr(split[3]);
333                 if (host != null) {
334                     hosts.add(host);
335                 }
336             }
337         }
338         Map<String, Boolean> reachable = new ConcurrentHashMap<>();
339         if (hosts.isEmpty()) {
340             return reachable;
341         }
342         ExecutorService pool = Executors.newFixedThreadPool(Math.min(hosts.size(), NFS_PROBE_MAX_THREADS));
343         try {
344             CompletableFuture<?>[] futures = hosts.stream()
345                     .map(h -> CompletableFuture.runAsync(() -> reachable.put(h, tcpReachable(h, 2049, 2_000)), pool))
346                     .toArray(CompletableFuture[]::new);
347             CompletableFuture.allOf(futures).join();
348         } finally {
349             pool.shutdownNow();
350         }
351         return reachable;
352     }

354     private static Map<String, String> queryLabelMap() {
355         Map<String, String> labelMap = new HashMap<>();
356         for (String line : ExecutingCommand.runNative("lsblk -o mountpoint,label")) {
357             String[] split = ParseUtil.whitespaces.split(line, 2);
358             if (split.length == 2) {
359                 labelMap.put(split[0], split[1]);
360             }
361         }
362         return labelMap;
363     }

380     /**
381      * Returns a value from the Linux system file /proc/sys/fs/file-nr.
382      *
383      * @param index The index of the value to retrieve. 0 returns the total allocated file descriptors. 1 returns the
384      *              number of used file descriptors for kernel 2.4, or the number of unused file descriptors for kernel
385      *              2.6. 2 returns the maximum number of file descriptors that can be allocated.
386      * @return Corresponding file descriptor value from the Linux system file.
387      */
388     private static long getFileDescriptors(int index) {
389         String filename = ProcPath.SYS_FS_FILE_NR;
390         if (index < 0 || index > 2) {
391             throw new IllegalArgumentException("Index must be between 0 and 2.");
392         }
393         List<String> osDescriptors = FileUtil.readFile(filename);
394         if (!osDescriptors.isEmpty()) {
395             String[] splittedLine = osDescriptors.get(0).split("\\D+");
396             return ParseUtil.parseLongOrDefault(splittedLine[index], 0L);
397         }
398         return 0L;
399     }

401     private static long getFileDescriptorsPerProcess() {
402         return FileUtil.getLongFromFile(ProcPath.SYS_FS_FILE_MAX);
