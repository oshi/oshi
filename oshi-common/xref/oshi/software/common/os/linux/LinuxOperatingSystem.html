71          long tempBT = CpuStat.getBootTime();

96      @Override
97      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
98          Triplet<String, String, String> familyVersionCodename = queryFamilyVersionCodenameFromReleaseFiles();
99          String buildNumber = null;
100         List<String> procVersion = FileUtil.readFile(ProcPath.VERSION);
101         if (!procVersion.isEmpty()) {
102             String[] split = ParseUtil.whitespaces.split(procVersion.get(0));
103             for (String s : split) {
104                 if (!"Linux".equals(s) && !"version".equals(s)) {
105                     buildNumber = s;
106                     break;
107                 }
108             }
109         }
110         OSVersionInfo versionInfo = new OSVersionInfo(familyVersionCodename.getB(), familyVersionCodename.getC(),
111                 buildNumber);
112         return new Pair<>(familyVersionCodename.getA(), versionInfo);
113     }

133     @Override
134     public List<OSProcess> queryChildProcesses(int parentPid) {
135         File[] pidFiles = ProcessStat.getPidFiles();
136         if (parentPid >= 0) {
137             return queryProcessList(getChildrenOrDescendants(getParentPidsFromProcFiles(pidFiles), parentPid, false));
138         }
139         Set<Integer> descendantPids = new HashSet<>();
140         for (File procFile : pidFiles) {
141             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), -2);
142             if (pid != -2) {
143                 descendantPids.add(pid);
144             }
145         }
146         return queryProcessList(descendantPids);
147     }
148 
149     @Override
150     public List<OSProcess> queryDescendantProcesses(int parentPid) {
151         File[] pidFiles = ProcessStat.getPidFiles();
152         return queryProcessList(getChildrenOrDescendants(getParentPidsFromProcFiles(pidFiles), parentPid, true));
153     }

155     private List<OSProcess> queryProcessList(Set<Integer> descendantPids) {
156         List<OSProcess> procs = new ArrayList<>();
157         for (int pid : descendantPids) {
158             OSProcess proc = createOSProcess(pid);
159             if (!proc.getState().equals(State.INVALID)) {
160                 procs.add(proc);
161             }
162         }
163         return procs;
164     }

174     protected static Map<Integer, Integer> getParentPidsFromProcFiles(File[] pidFiles) {
175         Map<Integer, Integer> parentPidMap = new HashMap<>();
176         for (File procFile : pidFiles) {
177             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), 0);
178             parentPidMap.put(pid, getParentPidFromProcFile(pid));
179         }
180         return parentPidMap;
181     }

183     private static int getParentPidFromProcFile(int pid) {
184         String stat = FileUtil.getStringFromFile(String.format(Locale.ROOT, "/proc/%d/stat", pid));
185         // A race condition may leave us with an empty string
186         if (stat.isEmpty()) {
187             return 0;
188         }
189         // Grab PPID
190         long[] statArray = ParseUtil.parseStringToLongArray(stat, PPID_INDEX, ProcessStat.PROC_PID_STAT_LENGTH, ' ');
191         return (int) statArray[0];
192     }

219     private static Triplet<String, String, String> queryFamilyVersionCodenameFromReleaseFiles() {
220         Triplet<String, String, String> familyVersionCodename;
221         // There are two competing options for family/version information.
222         // Newer systems are adopting a standard /etc/os-release file:
223         // https://www.freedesktop.org/software/systemd/man/os-release.html
224         //
225         // Some systems are still using the lsb standard which parses a
226         // variety of /etc/*-release files and is most easily accessed via
227         // the commandline lsb_release -a, see here:
228         // https://linux.die.net/man/1/lsb_release
229         // In this case, the /etc/lsb-release file (if it exists) has
230         // optional overrides to the information in the /etc/distrib-release
231         // files, which show: "Distributor release x.x (Codename)"
232 
233         // Attempt to read /etc/system-release which has more details than
234         // os-release on (CentOS and Fedora)
235         if ((familyVersionCodename = readDistribRelease("/etc/system-release")) != null) {
236             return familyVersionCodename;
237         }
238 
239         // Attempt to read /etc/os-release file.
240         if ((familyVersionCodename = readOsRelease()) != null) {
241             return familyVersionCodename;
242         }
243 
244         // Attempt to execute the `lsb_release` command
245         if ((familyVersionCodename = execLsbRelease()) != null) {
246             return familyVersionCodename;
247         }
248 
249         // Attempt to read /etc/lsb-release file
250         if ((familyVersionCodename = readLsbRelease()) != null) {
251             return familyVersionCodename;
252         }
253 
254         // Search for any /etc/*-release (or similar) filename
255         String etcDistribRelease = getReleaseFilename();
256         if ((familyVersionCodename = readDistribRelease(etcDistribRelease)) != null) {
257             return familyVersionCodename;
258         }
259         String family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
260                 .replace("version", "").replace("-", "").replace("_", ""));
261         return new Triplet<>(family, Constants.UNKNOWN, Constants.UNKNOWN);
262     }

264     /**
265      * Attempts to read /etc/os-release
266      *
267      * @return a triplet with the parsed family, versionID and codeName if file successfully read and NAME= found, null
268      *         otherwise
269      */
270     private static Triplet<String, String, String> readOsRelease() {
271         String family = null;
272         String versionId = Constants.UNKNOWN;
273         String codeName = Constants.UNKNOWN;
274         List<String> osRelease = FileUtil.readFile("/etc/os-release");
275         // Search for NAME=
276         for (String line : osRelease) {
277             if (line.startsWith("VERSION=")) {
278                 LOG.debug(OS_RELEASE_LOG, line);
279                 line = line.replace("VERSION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
280                 String[] split = line.split("[()]");
281                 if (split.length <= 1) {
282                     split = line.split(", ");
283                 }
284                 if (split.length > 0) {
285                     versionId = split[0].trim();
286                 }
287                 if (split.length > 1) {
288                     codeName = split[1].trim();
289                 }
290             } else if (line.startsWith("NAME=") && family == null) {
291                 LOG.debug(OS_RELEASE_LOG, line);
292                 family = line.replace("NAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
293             } else if (line.startsWith("VERSION_ID=") && versionId.equals(Constants.UNKNOWN)) {
294                 LOG.debug(OS_RELEASE_LOG, line);
295                 versionId = line.replace("VERSION_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
296             }
297         }
298         return family == null ? null : new Triplet<>(family, versionId, codeName);
299     }

301     /**
302      * Attempts to execute `lsb_release -a`
303      *
304      * @return a triplet with the parsed family, versionID and codeName if the command successfully executed and
305      *         Distributor ID: or Description: found, null otherwise
306      */
307     private static Triplet<String, String, String> execLsbRelease() {
308         String family = null;
309         String versionId = Constants.UNKNOWN;
310         String codeName = Constants.UNKNOWN;
311         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
312             if (line.startsWith("Description:")) {
313                 LOG.debug(LSB_RELEASE_A_LOG, line);
314                 line = line.replace("Description:", "").trim();
315                 if (line.contains(RELEASE_DELIM)) {
316                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
317                     family = triplet.getA();
318                     if (versionId.equals(Constants.UNKNOWN)) {
319                         versionId = triplet.getB();
320                     }
321                     if (codeName.equals(Constants.UNKNOWN)) {
322                         codeName = triplet.getC();
323                     }
324                 }
325             } else if (line.startsWith("Distributor ID:") && family == null) {
326                 LOG.debug(LSB_RELEASE_A_LOG, line);
327                 family = line.replace("Distributor ID:", "").trim();
328             } else if (line.startsWith("Release:") && versionId.equals(Constants.UNKNOWN)) {
329                 LOG.debug(LSB_RELEASE_A_LOG, line);
330                 versionId = line.replace("Release:", "").trim();
331             } else if (line.startsWith("Codename:") && codeName.equals(Constants.UNKNOWN)) {
332                 LOG.debug(LSB_RELEASE_A_LOG, line);
333                 codeName = line.replace("Codename:", "").trim();
334             }
335         }
336         return family == null ? null : new Triplet<>(family, versionId, codeName);
337     }

339     /**
340      * Attempts to read /etc/lsb-release
341      *
342      * @return a triplet with the parsed family, versionID and codeName if file successfully read and and DISTRIB_ID or
343      *         DISTRIB_DESCRIPTION, null otherwise
344      */
345     private static Triplet<String, String, String> readLsbRelease() {
346         String family = null;
347         String versionId = Constants.UNKNOWN;
348         String codeName = Constants.UNKNOWN;
349         List<String> osRelease = FileUtil.readFile("/etc/lsb-release");
350         for (String line : osRelease) {
351             if (line.startsWith("DISTRIB_DESCRIPTION=")) {
352                 LOG.debug(LSB_RELEASE_LOG, line);
353                 line = line.replace("DISTRIB_DESCRIPTION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
354                 if (line.contains(RELEASE_DELIM)) {
355                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
356                     family = triplet.getA();
357                     if (versionId.equals(Constants.UNKNOWN)) {
358                         versionId = triplet.getB();
359                     }
360                     if (codeName.equals(Constants.UNKNOWN)) {
361                         codeName = triplet.getC();
362                     }
363                 }
364             } else if (line.startsWith("DISTRIB_ID=") && family == null) {
365                 LOG.debug(LSB_RELEASE_LOG, line);
366                 family = line.replace("DISTRIB_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
367             } else if (line.startsWith("DISTRIB_RELEASE=") && versionId.equals(Constants.UNKNOWN)) {
368                 LOG.debug(LSB_RELEASE_LOG, line);
369                 versionId = line.replace("DISTRIB_RELEASE=", "").replaceAll(DOUBLE_QUOTES, "").trim();
370             } else if (line.startsWith("DISTRIB_CODENAME=") && codeName.equals(Constants.UNKNOWN)) {
371                 LOG.debug(LSB_RELEASE_LOG, line);
372                 codeName = line.replace("DISTRIB_CODENAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
373             }
374         }
375         return family == null ? null : new Triplet<>(family, versionId, codeName);
376     }

378     /**
379      * Attempts to read /etc/distrib-release (for some value of distrib)
380      *
381      * @param filename The /etc/distrib-release file
382      * @return a triplet with the parsed family, versionID and codeName if file successfully read and " release " or "
383      *         VERSION " found, null otherwise
384      */
385     private static Triplet<String, String, String> readDistribRelease(String filename) {
386         if (new File(filename).exists()) {
387             List<String> osRelease = FileUtil.readFile(filename);
388             for (String line : osRelease) {
389                 LOG.debug("{}: {}", filename, line);
390                 if (line.contains(RELEASE_DELIM)) {
391                     return parseRelease(line, RELEASE_DELIM);
392                 } else if (line.contains(" VERSION ")) {
393                     return parseRelease(line, " VERSION ");
394                 }
395             }
396         }
397         return null;
398     }

400     /**
401      * Helper method to parse version description line style
402      *
403      * @param line      a String of the form "Distributor release x.x (Codename)"
404      * @param splitLine A regex to split on, e.g. " release "
405      * @return a triplet with the parsed family, versionID and codeName
406      */
407     private static Triplet<String, String, String> parseRelease(String line, String splitLine) {
408         String[] split = line.split(splitLine);
409         String family = split[0].trim();
410         String versionId = Constants.UNKNOWN;
411         String codeName = Constants.UNKNOWN;
412         if (split.length > 1) {
413             split = split[1].split("[()]");
414             if (split.length > 0) {
415                 versionId = split[0].trim();
416             }
417             if (split.length > 1) {
418                 codeName = split[1].trim();
419             }
420         }
421         return new Triplet<>(family, versionId, codeName);
422     }

424     /**
425      * Looks for a collection of possible distrib-release filenames
426      *
427      * @return The first valid matching filename
428      */
429     protected static String getReleaseFilename() {
430         // Look for any /etc/*-release, *-version, and variants
431         File etc = new File("/etc");
432         File[] matchingFiles = etc.listFiles(//
433                 f -> (f.getName().endsWith("-release") || //
434                         f.getName().endsWith("-version") || //
435                         f.getName().endsWith("_release") || //
436                         f.getName().endsWith("_version")) //
437                         && !(f.getName().endsWith("os-release") || //
438                                 f.getName().endsWith("lsb-release") || //
439                                 f.getName().endsWith("system-release")));
440         if (matchingFiles != null && matchingFiles.length > 0) {
441             return matchingFiles[0].getPath();
442         }
443         if (new File("/etc/release").exists()) {
444             return "/etc/release";
445         }
446         return "/etc/issue";
447     }

449     /**
450      * Converts a portion of a filename (e.g. the 'redhat' in /etc/redhat-release) to a mixed case string representing
451      * the family (e.g., Red Hat)
452      *
453      * @param name Stripped version of filename after removing /etc and -release
454      * @return Mixed case family
455      */
456     private static String filenameToFamily(String name) {
457         if (name.isEmpty()) {
458             return "Solaris";
459         } else if ("issue".equalsIgnoreCase(name)) {
460             return "Unknown";
461         } else {
462             Properties filenameProps = FileUtil.readPropertiesFromFilename(FILENAME_PROPERTIES);
463             String family = filenameProps.getProperty(name.toLowerCase(Locale.ROOT));
464             return family != null ? family : name.substring(0, 1).toUpperCase(Locale.ROOT) + name.substring(1);
465         }
466     }

468     @Override
469     public List<OSService> getServices() {
470         List<OSService> services = new ArrayList<>();
471         Set<String> running = new HashSet<>();
472         for (OSProcess p : getChildProcesses(1, ProcessFiltering.ALL_PROCESSES, ProcessSorting.PID_ASC, 0)) {
473             OSService s = new OSService(p.getName(), p.getProcessID(), RUNNING);
474             services.add(s);
475             running.add(p.getName());
476         }
477         boolean systemctlFound = false;
478         List<String> systemctl = ExecutingCommand.runNative("systemctl list-unit-files");
479         for (String str : systemctl) {
480             String[] split = ParseUtil.whitespaces.split(str);
481             if (split.length >= 2 && split[0].endsWith(".service") && "enabled".equals(split[1])) {
482                 String name = split[0].substring(0, split[0].length() - 8);
483                 int index = name.lastIndexOf('.');
484                 String shortName = (index < 0 || index > name.length() - 2) ? name : name.substring(index + 1);
485                 if (!running.contains(name) && !running.contains(shortName)) {
486                     OSService s = new OSService(name, 0, STOPPED);
487                     services.add(s);
488                     systemctlFound = true;
489                 }
490             }
491         }
492         if (!systemctlFound) {
493             File dir = new File("/etc/init");
494             if (dir.exists() && dir.isDirectory()) {
495                 for (File f : dir.listFiles((f, name) -> name.toLowerCase(Locale.ROOT).endsWith(".conf"))) {
496                     String name = f.getName().substring(0, f.getName().length() - 5);
497                     int index = name.lastIndexOf('.');
498                     String shortName = (index < 0 || index > name.length() - 2) ? name : name.substring(index + 1);
499                     if (!running.contains(name) && !running.contains(shortName)) {
500                         OSService s = new OSService(name, 0, STOPPED);
501                         services.add(s);
502                     }
503                 }
504             } else {
505                 LOG.error("Directory: /etc/init does not exist");
506             }
507         }
508         return services;
509     }
