5   package oshi.software.common.os.unix.netbsd;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.software.common.AbstractOSFileStore;
9   import oshi.software.os.OSFileStore;
10  
11  /**
12   * OSFileStore implementation
13   */
14  @ThreadSafe
15  public class NetBsdOSFileStore extends AbstractOSFileStore {
16  
17      public NetBsdOSFileStore(String name, String volume, String label, String mount, String options, String uuid,
18              boolean local, String logicalVolume, String description, String fsType, long freeSpace, long usableSpace,
19              long totalSpace, long freeInodes, long totalInodes) {
20          super(name, volume, label, mount, options, uuid, local, logicalVolume, description, fsType, freeSpace,
21                  usableSpace, totalSpace, freeInodes, totalInodes);
22      }
23  
24      @Override
25      public boolean updateAttributes() {
26          for (OSFileStore fileStore : NetBsdFileSystem.getFileStoreMatching(getName(), isLocal())) {
27              if (getName().equals(fileStore.getName()) && getVolume().equals(fileStore.getVolume())
28                      && getMount().equals(fileStore.getMount())) {
29                  updateFrom(fileStore);
30                  return true;
31              }
32          }
33          return false;
34      }
35  }
