28  /**
29   * Solaris is a non-free Unix operating system originally developed by Sun Microsystems. It superseded the company's
30   * earlier SunOS in 1993. In 2010, after the Sun acquisition by Oracle, it was renamed Oracle Solaris.
31   * <p>
32   * Process enumeration, services, and version queries are shared; the native bits ({@code kstat} uptime/boot time,
33   * {@code getpid}/{@code thr_self}, and the {@link OSProcess}/{@code FileSystem}/{@code NetworkParams}/{@code OSThread}
34   * factories) are provided by the JNA and FFM subclasses.
35   */
36  @ThreadSafe
37  public abstract class SolarisOperatingSystem extends AbstractOperatingSystem {

39      private static final String VERSION;
40      private static final String BUILD_NUMBER;
41      static {
42          String[] split = ParseUtil.whitespaces.split(ExecutingCommand.getFirstAnswer("uname -rv"));
43          VERSION = split[0];
44          BUILD_NUMBER = split.length > 1 ? split[1] : "";
45      }

47      @Override
48      public String queryManufacturer() {
49          return "Oracle";
50      }

52      @Override
53      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
54          return new Pair<>("SunOS", new OSVersionInfo(VERSION, "Solaris", BUILD_NUMBER));
55      }

57      @Override
58      protected int queryBitness(int jvmBitness) {
59          if (jvmBitness == 64) {
60              return 64;
61          }
62          return ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("isainfo -b"), 32);
63      }

65      @Override
66      public InternetProtocolStats getInternetProtocolStats() {
67          return new SolarisInternetProtocolStats();
68      }

70      @Override
71      public OSProcess getProcess(int pid) {
72          List<OSProcess> procs = getProcessListFromProcfs(pid);
73          if (procs.isEmpty()) {
74              return null;
75          }
76          return procs.get(0);
77      }
78  
79      @Override
80      public List<OSProcess> queryAllProcesses() {
81          return getProcessListFromProcfs(-1);
82      }
83  
84      @Override
85      public List<OSProcess> queryChildProcesses(int parentPid) {
86          List<OSProcess> allProcs = queryAllProcesses();
87          Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, false);
88          return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
89      }
90  
91      @Override
92      public List<OSProcess> queryDescendantProcesses(int parentPid) {
93          List<OSProcess> allProcs = queryAllProcesses();
94          Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, true);
95          return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
96      }

98      private List<OSProcess> getProcessListFromProcfs(int pid) {
99          List<OSProcess> procs = new ArrayList<>();
100 
101         File[] numericFiles = null;
102         if (pid < 0) {
103             // If no pid, get process files in proc
104             File directory = new File("/proc");
105             numericFiles = directory.listFiles(file -> Constants.DIGITS.matcher(file.getName()).matches());
106         } else {
107             // If pid specified just find that file
108             File pidFile = new File("/proc/" + pid);
109             if (pidFile.exists()) {
110                 numericFiles = new File[1];
111                 numericFiles[0] = pidFile;
112             }
113         }
114         if (numericFiles == null) {
115             return procs;
116         }
117 
118         // Iterate files
119         for (File pidFile : numericFiles) {
120             int pidNum = ParseUtil.parseIntOrDefault(pidFile.getName(), 0);
121             OSProcess proc = createProcess(pidNum);
122             if (proc.getState() != INVALID) {
123                 procs.add(proc);
124             }
125         }
126         return procs;
127     }

129     @Override

131         return ProcessStat.getPidFiles().length;
132     }

134     @Override
135     public int getThreadCount() {
136         List<String> threadList = ExecutingCommand.runNative("ps -eLo pid");
137         if (!threadList.isEmpty()) {
138             // Subtract 1 for header
139             return threadList.size() - 1;
140         }
141         return getProcessCount();
142     }

144     @Override
145     public List<OSService> getServices() {
146         // Get legacy RC service name possibilities
147         List<String> legacySvcs = new ArrayList<>();
148         File dir = new File("/etc/init.d");
149         File[] listFiles;
150         if (dir.exists() && dir.isDirectory() && (listFiles = dir.listFiles()) != null) {
151             for (File f : listFiles) {
152                 legacySvcs.add(f.getName());
153             }
154         }
155         // Iterate service list
156         List<String> svcs = ExecutingCommand.runNative("svcs -p");
157         return parseSvcs(svcs, legacySvcs);
158     }

160     /**
161      * Parses the output of {@code svcs -p} to build a list of OS services.
162      *
163      * @param svcs       the lines emitted by {@code svcs -p}
164      * @param legacySvcs the list of legacy service names from {@code /etc/init.d}
165      * @return a list of {@link OSService} objects
166      */
167     static List<OSService> parseSvcs(List<String> svcs, List<String> legacySvcs) {
168         List<OSService> services = new ArrayList<>();
169         /*-
170          Output:
171          STATE          STIME    FRMI
172          legacy_run     23:56:49 lrc:/etc/rc2_d/S47pppd
173          legacy_run     23:56:49 lrc:/etc/rc2_d/S81dodatadm_udaplt
174          legacy_run     23:56:49 lrc:/etc/rc2_d/S89PRESERVE
175          online         23:56:25 svc:/system/early-manifest-import:default
176          online         23:56:25 svc:/system/svc/restarter:default
177                         23:56:24       13 svc.startd
178                         ...
179          */
180         for (String line : svcs) {
181             if (line.startsWith("online")) {
182                 int delim = line.lastIndexOf(":/");
183                 if (delim > 0) {
184                     String name = line.substring(delim + 1);
185                     if (name.endsWith(":default")) {
186                         name = name.substring(0, name.length() - 8);
187                     }
188                     services.add(new OSService(name, 0, STOPPED));
189                 }
190             } else if (line.startsWith(" ")) {
191                 String[] split = ParseUtil.whitespaces.split(line.trim());
192                 if (split.length == 3) {
193                     services.add(new OSService(split[2], ParseUtil.parseIntOrDefault(split[1], 0), RUNNING));
194                 }
195             } else if (line.startsWith("legacy_run")) {
196                 for (String svc : legacySvcs) {
197                     if (line.endsWith(svc)) {
198                         services.add(new OSService(svc, 0, STOPPED));
199                         break;
200                     }
201                 }
202             }
203         }
204         return services;
205     }
