5   package oshi.software.common.os.unix.aix;
6   
7   import oshi.annotation.concurrent.ThreadSafe;
8   import oshi.driver.common.unix.aix.AixLwpsInfo;
9   import oshi.driver.common.unix.aix.PsInfo;
10  import oshi.software.common.AbstractOSThread;
11  import oshi.software.os.OSProcess;
12  
13  /**
14   * AIX OSThread implementation. Reads {@code /proc/<pid>/lwp/<tid>/lwpsinfo} via {@link PsInfo} so it has no native
15   * dependency and is shared by the JNA and FFM backends.
16   */
17  @ThreadSafe
18  public class AixOSThread extends AbstractOSThread {
19  
20      public AixOSThread(int pid, int tid) {
21          super(pid);
22          this.threadId = tid;
23          updateAttributes();
24      }
25  
26      @Override
27      public boolean updateAttributes() {
28          AixLwpsInfo lwpsinfo = PsInfo.queryLwpsInfo(getOwningProcessId(), getThreadId());
29          if (lwpsinfo == null) {
30              this.state = OSProcess.State.INVALID;
31              return false;
32          }
33          this.threadId = (int) lwpsinfo.pr_lwpid; // 64 bit storage but always 32 bit
34          this.startMemoryAddress = lwpsinfo.pr_addr;
35          this.state = AixProcessState.getStateFromOutput((char) lwpsinfo.pr_sname);
36          this.priority = lwpsinfo.pr_pri;
37          return true;
38      }
39  }
