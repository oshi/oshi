7   import static oshi.software.os.OperatingSystem.ProcessFiltering.VALID_PROCESS;

9   import java.util.List;
10  import java.util.Map;
11  import java.util.function.Predicate;
12  import java.util.stream.Collectors;
13  
14  import oshi.annotation.concurrent.ThreadSafe;
15  import oshi.software.common.os.unix.bsd.BsdOperatingSystem;
16  import oshi.software.common.os.unix.bsd.BsdPsKeyword;
17  import oshi.software.os.OSProcess;
18  import oshi.software.os.OSThread;
19  import oshi.util.ExecutingCommand;
20  import oshi.util.ParseUtil;
21  
22  /**
23   * Abstract base shared by the FreeBSD OperatingSystem implementations (JNA and FFM). The cross-BSD-common pieces live
24   * in {@link BsdOperatingSystem}; this layer adds the FreeBSD {@code ps} process enumeration and current-thread factory.
25   * The native bits (the {@link OSProcess}/{@code FileSystem}/{@code NetworkParams}/{@code InternetProtocolStats}
26   * factories, {@code getpid}/{@code thr_self}, sysctl version and boot-time queries, and {@code who} sessions) are
27   * provided by the JNA and FFM subclasses.
28   */
29  @ThreadSafe
30  public abstract class FreeBsdOperatingSystem extends BsdOperatingSystem {
31  
32      @Override
33      protected List<OSProcess> getProcessListFromPS(int pid) {
34          String psCommand = "ps -awwxo " + FreeBsdOSProcess.PS_COMMAND_ARGS;
35          if (pid >= 0) {
36              psCommand += " -p " + pid;
37          }
38  
39          Predicate<Map<BsdPsKeyword, String>> hasKeywordArgs = psMap -> psMap.containsKey(BsdPsKeyword.ARGS);
40          return ExecutingCommand.runNative(psCommand).stream().skip(1).parallel()
41                  .map(proc -> ParseUtil.stringToEnumMap(BsdPsKeyword.class, FreeBsdOSProcess.PS_KEYWORDS, proc.trim(),
42                          ' '))
43                  .filter(hasKeywordArgs)
44                  .map(psMap -> createProcess(pid < 0 ? ParseUtil.parseIntOrDefault(psMap.get(BsdPsKeyword.PID), 0) : pid,
45                          psMap))
46                  .filter(VALID_PROCESS).collect(Collectors.toList());
47      }
48  
49      @Override
50      public OSThread getCurrentThread() {
51          OSProcess proc = getCurrentProcess();
52          final int tid = getThreadId();
53          return proc.getThreadDetails().stream().filter(t -> t.getThreadId() == tid).findFirst()
54                  .orElse(new FreeBsdOSThread(proc.getProcessID(), tid));
55      }
56  
57      /**
58       * Creates a backend-specific {@link OSProcess} from a parsed {@code ps} row.
59       *
60       * @param pid   the process ID
61       * @param psMap the parsed {@code ps} columns for this process
62       * @return a new OSProcess
63       */
64      protected abstract OSProcess createProcess(int pid, Map<BsdPsKeyword, String> psMap);
65  }
