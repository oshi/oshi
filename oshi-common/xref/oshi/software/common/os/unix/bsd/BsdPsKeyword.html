7   /**
8    * The union of {@code ps} output columns used by the BSD-family {@code OSProcess} implementations (FreeBSD, OpenBSD,
9    * DragonFly, NetBSD). Each platform declares its own ordered subset (see {@code PS_KEYWORDS} on the per-platform
10   * {@code OSProcess} base) and parses {@code ps} output positionally against that subset. The constant order here is not
11   * significant; only the per-platform subset order is.
12   */
13  public enum BsdPsKeyword {
14      STATE, PID, PPID, USER, UID, GROUP, GID, RGID, NLWP, PRI, VSZ, RSS, ETIME, ETIMES, CPUTIME, SYSTIME, TIME, COMM,
15      UCOMM, MAJFLT, MINFLT, NVCSW, NIVCSW, ARGS, COMMAND;
16  }
