41  public abstract class OpenBsdOSProcess extends BsdOSProcess {
42  
43      /**
44       * Ordered {@code ps} columns queried for each process. Shared by the OpenBsdOSProcess subclasses (JNA/FFM) and the
45       * OpenBSD OperatingSystem classes so the column list and parsing stay in lockstep. {@code ARGS} must remain last.
46       */
47      public static final List<BsdPsKeyword> PS_KEYWORDS = Collections.unmodifiableList(Arrays.asList(STATE, PID, PPID,
48              USER, UID, GROUP, GID, PRI, VSZ, RSS, ETIME, CPUTIME, COMM, MAJFLT, MINFLT, NVCSW, NIVCSW, ARGS));
49  
50      public static final String PS_COMMAND_ARGS = PS_KEYWORDS.stream().map(Enum::name)
51              .map(name -> name.toLowerCase(Locale.ROOT)).collect(Collectors.joining(","));
52  
53      /**
54       * Columns requested from {@code ps -aHwwxo} when enumerating threads. Shared by OpenBsdOSProcess subclasses and
55       * OpenBsdOSThread so the column list and parsing enum stay in lockstep.
56       */
57      public enum PsThreadColumns {
58          TID, STATE, ETIME, CPUTIME, NIVCSW, NVCSW, MAJFLT, MINFLT, PRI, ARGS;
59      }
60  
61      public static final String PS_THREAD_COLUMNS = Arrays.stream(PsThreadColumns.values()).map(Enum::name)
62              .map(name -> name.toLowerCase(Locale.ROOT)).collect(Collectors.joining(","));
63  
64      protected OpenBsdOSProcess(int pid) {
65          super(pid);
66      }
67  
68      @Override
69      protected List<BsdPsKeyword> psKeywords() {
70          return PS_KEYWORDS;
71      }
72  
73      @Override
74      protected String psCommandArgs() {
75          return PS_COMMAND_ARGS;
76      }
77  
78      @Override
79      protected void updateThreadCount() {
80          List<String> threadList = ExecutingCommand.runNative("ps -axHo tid -p " + getProcessID());
81          if (!threadList.isEmpty()) {
82              // Subtract 1 for header
83              this.threadCount = threadList.size() - 1;
84          }
85          // A live process always has at least one thread
86          this.threadCount = Math.max(this.threadCount, 1);
87      }
88  
89      @Override
90      public List<OSThread> getThreadDetails() {
91          String psCommand = "ps -aHwwxo " + PS_THREAD_COLUMNS;
92          if (getProcessID() >= 0) {
93              psCommand += " -p " + getProcessID();
94          }
95          Predicate<Map<PsThreadColumns, String>> hasColumnsArgs = threadMap -> threadMap
96                  .containsKey(PsThreadColumns.ARGS);
97          return ExecutingCommand.runNative(psCommand).stream().skip(1).parallel()
98                  .map(thread -> ParseUtil.stringToEnumMap(PsThreadColumns.class, thread.trim(), ' '))
99                  .filter(hasColumnsArgs).map(threadMap -> new OpenBsdOSThread(getProcessID(), threadMap))
100                 .filter(VALID_THREAD).collect(Collectors.toList());
101     }
102 }
