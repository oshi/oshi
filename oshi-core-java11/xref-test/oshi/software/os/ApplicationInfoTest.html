7   import org.junit.jupiter.api.BeforeEach;
8   import org.junit.jupiter.api.Test;
9   
10  import java.util.ArrayList;
11  import java.util.LinkedHashSet;
12  import java.util.LinkedHashMap;
13  import java.util.List;
14  import java.util.Map;
15  import java.util.Set;
16  
17  import static org.junit.jupiter.api.Assertions.assertEquals;
18  import static org.junit.jupiter.api.Assertions.assertNotNull;
19  import static org.junit.jupiter.api.Assertions.assertNotEquals;
20  import static org.junit.jupiter.api.Assertions.assertFalse;
21  import static org.junit.jupiter.api.Assertions.assertTrue;
22  
23  public class ApplicationInfoTest {
24  
25      private ApplicationInfo appInfo;
26      private final String name = "zstd";
27      private final String version = "1.5.5+dfsg2-2build1";
28      private final String vendor = "Ubuntu Developers <ubuntu-devel-discuss@lists.ubuntu.com>";
29      private final long timestamp = 1713886075L;
30      private Map<String, String> additionalInfo;
31  
32      @BeforeEach
33      void setUp() {
34          additionalInfo = new LinkedHashMap<>();
35          additionalInfo.put("source", "libzstd");
36          additionalInfo.put("installedSize", "1798");
37          additionalInfo.put("architecture", "amd64");
38          additionalInfo.put("homepage", "https://github.com/facebook/zstd");
39  
40          appInfo = new ApplicationInfo(name, version, vendor, timestamp, additionalInfo);
41      }
42  
43      @Test
44      void testGetName() {
45          assertEquals(name, appInfo.getName(), "Application name should match");
46      }
47  
48      @Test
49      void testGetVersion() {
50          assertEquals(version, appInfo.getVersion(), "Version should match");
51      }
52  
53      @Test
54      void testGetVendor() {
55          assertEquals(vendor, appInfo.getVendor(), "Vendor should match");
56      }
57  
58      @Test
59      void testGetTimestamp() {
60          assertEquals(timestamp, appInfo.getTimestamp(), "Timestamp should match");
61      }
62  
63      @Test
64      void testGetAdditionalInfo() {
65          assertNotNull(appInfo.getAdditionalInfo(), "Additional info map should not be null");
66          assertEquals("libzstd", appInfo.getAdditionalInfo().get("source"));
67          assertEquals("1798", appInfo.getAdditionalInfo().get("installedSize"));
68          assertEquals("amd64", appInfo.getAdditionalInfo().get("architecture"));
69          assertEquals("https://github.com/facebook/zstd", appInfo.getAdditionalInfo().get("homepage"));
70      }
71  
72      @Test
73      void testImmutableAdditionalInfo() {
74          // Ensure modifying the original map doesn't affect the stored map
75          additionalInfo.put("NewKey", "NewValue");
76          assertFalse(appInfo.getAdditionalInfo().containsKey("NewKey"), "Stored map should remain unchanged");
77      }
78  
79      @Test
80      public void equalsWithSameReferenceShouldReturnTrue() {
81          ApplicationInfo app = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
82                  1746576000000L, new LinkedHashMap<>());
83          assertTrue(app.equals(app)); // this == o
84      }
85  
86      @Test
87      public void equalsWithDifferentObjectTypeShouldReturnFalse() {
88          ApplicationInfo app = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
89                  1746576000000L, new LinkedHashMap<>());
90          assertFalse(app.equals("NotAnApplicationInfo")); // !(o instanceof ApplicationInfo)
91      }
92  
93      @Test
94      public void testEqualsAndHashCodeSameValues() {
95          Map<String, String> info1 = new LinkedHashMap<>();
96          info1.put("installLocation", null);
97          info1.put("installSource",
98                  "C:\\ProgramData\\Package Cache\\{FE8C7838-D3E6-4CEA-87BE-216E42391827}v20.2.37.0\\");
99  
100         ApplicationInfo app1 = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
101                 1746576000000L, info1);
102         ApplicationInfo app2 = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
103                 1746576000000L, new LinkedHashMap<>(info1));
104 
105         assertEquals(app1, app2);
106         assertEquals(app1.hashCode(), app2.hashCode());
107     }
108 
109     @Test
110     public void testEqualsAndHashCodeDifferentVersion() {
111         ApplicationInfo app1 = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
112                 1746576000000L, new LinkedHashMap<>());
113         ApplicationInfo app2 = new ApplicationInfo("SQL Server Management Studio", "20.3.37.0", "Microsoft Corp.",
114                 1746576000000L, new LinkedHashMap<>());
115 
116         assertNotEquals(app1, app2);
117     }
118 
119     @Test
120     public void testDeduplicationWithListResult() {
121         ApplicationInfo app1 = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
122                 1746576000000L, new LinkedHashMap<>());
123 
124         ApplicationInfo app2 = new ApplicationInfo("SQL Server Management Studio", "20.2.37.0", "Microsoft Corp.",
125                 1746576000000L, new LinkedHashMap<>());
126 
127         Set<ApplicationInfo> dedupedSet = new LinkedHashSet<>();
128         dedupedSet.add(app1);
129         dedupedSet.add(app2); // Duplicate
130 
131         List<ApplicationInfo> resultList = new ArrayList<>(dedupedSet);
132 
133         assertEquals(1, resultList.size());
134         assertEquals(app1, resultList.get(0));
135     }
