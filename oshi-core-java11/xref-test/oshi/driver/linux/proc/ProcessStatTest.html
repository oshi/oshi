24  package oshi.driver.linux.proc;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.greaterThan;
28  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
29  import static org.junit.jupiter.api.Assertions.assertNotNull;
30  import static org.junit.jupiter.api.Assertions.assertTrue;
31  
32  import java.io.File;
33  import java.util.Map;
34  
35  import org.junit.jupiter.api.Test;
36  import org.junit.jupiter.api.condition.EnabledOnOs;
37  import org.junit.jupiter.api.condition.OS;
38  
39  import oshi.driver.linux.proc.ProcessStat.PidStat;
40  import oshi.driver.linux.proc.ProcessStat.PidStatM;
41  import oshi.software.os.OSProcess;
42  import oshi.software.os.OSProcess.State;
43  import oshi.util.ParseUtil;
44  import oshi.util.tuples.Triplet;
45  
46  @EnabledOnOs(OS.LINUX)
47  class ProcessStatTest {
48  
49      @Test
50      void testGetPidFiles() {
51          File[] files = ProcessStat.getPidFiles();
52          assertThat("Files should be non-empty array", files.length, greaterThan(0));
53          // No need to test every file, but no guarantee any one file exists due to race
54          // condition. Track that each test succeeds at least once.
55          boolean fileDescriptorFilesTested = false;
56          boolean pidStatMTested = false;
57          boolean pidStatsTested = false;
58          boolean threadIdsTested = false;
59          for (File f : files) {
60              int pid = ParseUtil.parseIntOrDefault(f.getName(), -1);
61              assertThat("Pid " + f.getName() + " must be numeric", pid, greaterThanOrEqualTo(0));
62              if (!fileDescriptorFilesTested) {
63                  fileDescriptorFilesTested = testGetFileDescriptorFiles(pid);
64              }
65              if (!pidStatMTested) {
66                  pidStatMTested = testGetPidStatM(pid);
67              }
68              if (!pidStatsTested) {
69                  pidStatsTested = testGetPidStats(pid);
70              }
71              if (!threadIdsTested) {
72                  threadIdsTested = testGetThreadIds(pid);
73              }
74              // If every test has passed at least once, we're done
75              if (fileDescriptorFilesTested && pidStatMTested && pidStatsTested && threadIdsTested) {
76                  break;
77              }
78          }
79          assertTrue(fileDescriptorFilesTested, "File Descriptor tests failed");
80          assertTrue(pidStatMTested, "PidStatM tests failed");
81          assertTrue(pidStatsTested, "PidStats tests failed");
82          assertTrue(threadIdsTested, "ThreadIds tests failed");
83      }
84  
85      private boolean testGetFileDescriptorFiles(int pid) {
86          // Pids should have STDIN, STDOUT, and STDERR fds
87          return ProcessStat.getFileDescriptorFiles(pid).length >= 3;
88      }
89  
90      private boolean testGetPidStatM(int pid) {
91          Map<PidStatM, Long> statM = ProcessStat.getPidStatM(pid);
92          assertNotNull(statM);
93          return !statM.isEmpty();
94      }
95  
96      private boolean testGetPidStats(int pid) {
97          Triplet<String, Character, Map<PidStat, Long>> stats = ProcessStat.getPidStats(pid);
98          assertNotNull(stats);
99          State procState = ProcessStat.getState(stats.getB());
100         return stats.getA().length() > 0 // at least one process should have nonempty name
101                 // At least one process should be running or sleeping
102                 && (procState.equals(OSProcess.State.RUNNING) || procState.equals(OSProcess.State.SLEEPING))
103                 // Have keys at least up to RSS
104                 && stats.getC().containsKey(PidStat.RSS);
105     }
106 
107     private boolean testGetThreadIds(int pid) {
108         // Every process should have at least one thread (itself)!
109         return !ProcessStat.getThreadIds(pid).isEmpty();
110     }
111 
112     @Test
113     void testQuerySocketToPidMap() {
114         assertThat("Socket to pid map shouldn't be empty.", ProcessStat.querySocketToPidMap().size(), greaterThan(0));
115     }
116 }
