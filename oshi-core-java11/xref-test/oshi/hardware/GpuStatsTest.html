116     /**
117      * Full lifecycle: sentinel/priming phase, poll phase, shutdown, then verify all methods throw post-close.
118      */
119     @Test
120     void testRealCardSessionFullLifecycle() throws InterruptedException {
121         SystemInfo si = new SystemInfo();
122         for (GraphicsCard card : si.getHardware().getGraphicsCards()) {
123             GpuStats stats = card.createStatsSession();
124             try {
125                 // Phase 1: sentinel / priming — delta-based metrics return -1 on first call
126                 assertThat("Session open after creation", stats.isClosed(), is(false));
127                 assertThat("getGpuTicks non-null on first call", stats.getGpuTicks(), is(notNullValue()));
128                 assertThat("getGpuUtilization >= -1 on first call", stats.getGpuUtilization() >= -1d, is(true));
129                 assertThat("getPowerDraw >= -1 on first call", stats.getPowerDraw() >= -1d, is(true));
130                 assertThat("getVramUsed >= -1", stats.getVramUsed() >= -1L, is(true));
131                 assertThat("getSharedMemoryUsed >= -1", stats.getSharedMemoryUsed() >= -1L, is(true));
132                 assertThat("getTemperature >= -1", stats.getTemperature() >= -1d, is(true));
133                 assertThat("getCoreClockMhz >= -1", stats.getCoreClockMhz() >= -1L, is(true));
134                 assertThat("getMemoryClockMhz >= -1", stats.getMemoryClockMhz() >= -1L, is(true));
135                 assertThat("getFanSpeedPercent >= -1", stats.getFanSpeedPercent() >= -1d, is(true));
136 
137                 // Phase 2: poll — after an interval, delta-based metrics return valid value or sentinel
138                 Thread.sleep(500L);
139                 assertThat("getGpuTicks non-null after interval", stats.getGpuTicks(), is(notNullValue()));
140                 assertThat("getGpuUtilization >= -1 after interval", stats.getGpuUtilization() >= -1d, is(true));
141                 assertThat("getPowerDraw >= -1 after interval", stats.getPowerDraw() >= -1d, is(true));
142                 assertThat("getVramUsed >= -1 after interval", stats.getVramUsed() >= -1L, is(true));
143                 assertThat("getTemperature >= -1 after interval", stats.getTemperature() >= -1d, is(true));
144             } finally {
145                 // Phase 3: shutdown
146                 stats.close();
147             }
148 
149             // Phase 4: post-close — all metric methods must throw IllegalStateException
150             assertThat("Session closed after close()", stats.isClosed(), is(true));
151             assertThrows(IllegalStateException.class, stats::getGpuTicks);
152             assertThrows(IllegalStateException.class, stats::getGpuUtilization);
153             assertThrows(IllegalStateException.class, stats::getVramUsed);
154             assertThrows(IllegalStateException.class, stats::getSharedMemoryUsed);
155             assertThrows(IllegalStateException.class, stats::getTemperature);
156             assertThrows(IllegalStateException.class, stats::getPowerDraw);
157             assertThrows(IllegalStateException.class, stats::getCoreClockMhz);
158             assertThrows(IllegalStateException.class, stats::getMemoryClockMhz);
159             assertThrows(IllegalStateException.class, stats::getFanSpeedPercent);
160         }
161     }

163     @Test
164     void testConcurrentAccessDoesNotThrow() throws Exception {
165         GpuStats stats = new NoOpGpuStats();
166         int threads = 8;
167         ExecutorService pool = Executors.newFixedThreadPool(threads);
168         List<Callable<Void>> tasks = new ArrayList<>();
169         for (int i = 0; i < threads; i++) {
170             tasks.add(() -> {
171                 stats.getGpuTicks();
172                 stats.getTemperature();
173                 stats.getPowerDraw();
174                 return null;
175             });
176         }
177         List<Future<Void>> futures;
178         try {
179             futures = pool.invokeAll(tasks);
180         } finally {
181             pool.shutdown();
182             if (!pool.awaitTermination(5L, TimeUnit.SECONDS)) {
183                 pool.shutdownNow();
184             }
185         }
186         stats.close();
187         for (Future<Void> f : futures) {
188             f.get();
189         }
190     }
