5   package oshi.software.os;
6   
7   import java.time.Instant;
8   import java.time.LocalDateTime;
9   import java.time.ZoneId;
10  import java.time.format.DateTimeFormatter;
11  
12  import oshi.annotation.concurrent.Immutable;
13  
14  /**
15   * This class encapsulates information about users who are currently logged in to an operating system.
16   */
17  @Immutable
18  public class OSSession {
19      private static final DateTimeFormatter LOGIN_FORMAT = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm");
20  
21      private final String userName;
22      private final String terminalDevice;
23      private final long loginTime;
24      private final String host;
25  
26      public OSSession(String userName, String terminalDevice, long loginTime, String host) {
27          this.userName = userName;
28          this.terminalDevice = terminalDevice;
29          this.loginTime = loginTime;
30          this.host = host;
31      }
32  
33      /**
34       * Gets the login name of the user
35       *
36       * @return the userName
37       */
38      public String getUserName() {
39          return userName;
40      }
41  
42      /**
43       * Gets the terminal device (such as tty, pts, etc.) the user used to log in
44       *
45       * @return the terminalDevice
46       */
47      public String getTerminalDevice() {
48          return terminalDevice;
49      }
50  
51      /**
52       * Gets the time the user logged in
53       *
54       * @return the loginTime, in milliseconds since the 1970 epoch
55       */
56      public long getLoginTime() {
57          return loginTime;
58      }
59  
60      /**
61       * Gets the remote host from which the user logged in
62       *
63       * @return the host as either an IPv4 or IPv6 representation. If the host is unspecified, may also be an empty
64       *         string, depending on the platform.
65       */
66      public String getHost() {
67          return host;
68      }
69  
70      @Override
71      public String toString() {
72          String loginStr = loginTime == 0 ? "No login"
73                  : LocalDateTime.ofInstant(Instant.ofEpochMilli(loginTime), ZoneId.systemDefault()).format(LOGIN_FORMAT);
74          String hostStr = "";
75          if (!host.isEmpty() && !host.equals("::") && !host.equals("0.0.0.0")) {
76              hostStr = ", (" + host + ")";
77          }
78          return String.format("%s, %s, %s%s", userName, terminalDevice, loginStr, hostStr);
79      }
80  }
