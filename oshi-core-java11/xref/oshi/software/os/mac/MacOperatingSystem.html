38  /**
39   * macOS, previously Mac OS X and later OS X) is a series of proprietary graphical operating systems developed and
40   * marketed by Apple Inc. since 2001. It is the primary operating system for Apple's Mac computers.
41   */
42  @ThreadSafe
43  public abstract class MacOperatingSystem extends AbstractOperatingSystem {
44  
45      private static final Logger LOG = LoggerFactory.getLogger(MacOperatingSystemJNA.class);
46  
47      public static final String MACOS_VERSIONS_PROPERTIES = "oshi.macos.versions.properties";
48  
49      private static final String SYSTEM_LIBRARY_LAUNCH_AGENTS = "/System/Library/LaunchAgents";
50      private static final String SYSTEM_LIBRARY_LAUNCH_DAEMONS = "/System/Library/LaunchDaemons";
51  
52      protected final int maxProc;
53  
54      protected final String osXVersion;
55      protected final int major;
56      protected final int minor;
57  
58      private final Supplier<List<ApplicationInfo>> installedAppsSupplier = Memoizer
59              .memoize(MacInstalledApps::queryInstalledApps, installedAppsExpiration());
60  
61      protected MacOperatingSystem(int maxproc) {
62          String version = System.getProperty("os.version");
63          int verMajor = ParseUtil.getFirstIntValue(version);
64          int verMinor = ParseUtil.getNthIntValue(version, 2);
65          // Big Sur (11.x) may return 10.16
66          if (verMajor == 10 && verMinor > 15) {
67              String swVers = ExecutingCommand.getFirstAnswer("sw_vers -productVersion");
68              if (!swVers.isEmpty()) {
69                  version = swVers;
70              }
71              verMajor = ParseUtil.getFirstIntValue(version);
72              verMinor = ParseUtil.getNthIntValue(version, 2);
73          }
74          this.osXVersion = version;
75          this.major = verMajor;
76          this.minor = verMinor;
77          // Set max processes
78          this.maxProc = maxproc;
79      }
80  
81      @Override
82      public String queryManufacturer() {
83          return "Apple";
84      }
85  
86      protected String parseCodeName() {
87          Properties verProps = FileUtil.readPropertiesFromFilename(MACOS_VERSIONS_PROPERTIES);
88          String codeName = null;
89          if (this.major > 10) {
90              codeName = verProps.getProperty(Integer.toString(this.major));
91          } else if (this.major == 10) {
92              codeName = verProps.getProperty(this.major + "." + this.minor);
93          }
94          if (Util.isBlank(codeName)) {
95              LOG.warn("Unable to parse version {}.{} to a codename.", this.major, this.minor);
96          }
97          return codeName;
98      }
99  
100     @Override
101     protected int queryBitness(int jvmBitness) {
102         if (jvmBitness == 64 || (this.major == 10 && this.minor > 6)) {
103             return 64;
104         }
105         return ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("getconf LONG_BIT"), 32);
106     }
107 
108     @Override
109     public List<OSProcess> queryChildProcesses(int parentPid) {
110         List<OSProcess> allProcs = queryAllProcesses();
111         Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, false);
112         return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
113     }
114 
115     @Override
116     public List<OSProcess> queryDescendantProcesses(int parentPid) {
117         List<OSProcess> allProcs = queryAllProcesses();
118         Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, true);
119         return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
120     }

122     @Override
123     public int getThreadId() {
124         OSThread thread = getCurrentThread();
125         if (thread == null) {
126             return 0;
127         }
128         return thread.getThreadId();
129     }
130 
131     @Override
132     public OSThread getCurrentThread() {
133         // Get oldest thread
134         return getCurrentProcess().getThreadDetails().stream().sorted(Comparator.comparingLong(OSThread::getStartTime))
135                 .findFirst().orElse(new MacOSThread(getProcessId()));
136     }

138     @Override
139     public long getSystemUptime() {
140         return System.currentTimeMillis() / 1000 - getSystemBootTime();
141     }

143     @Override
144     public List<OSService> getServices() {
145         // Get running services
146         List<OSService> services = new ArrayList<>();
147         Set<String> running = new HashSet<>();
148         for (OSProcess p : getChildProcesses(1, ProcessFiltering.ALL_PROCESSES, ProcessSorting.PID_ASC, 0)) {
149             OSService s = new OSService(p.getName(), p.getProcessID(), RUNNING);
150             services.add(s);
151             running.add(p.getName());
152         }
153         // Get Directories for stopped services
154         ArrayList<File> files = new ArrayList<>();
155         File dir = new File(SYSTEM_LIBRARY_LAUNCH_AGENTS);
156         if (dir.exists() && dir.isDirectory()) {
157             files.addAll(Arrays.asList(dir.listFiles((f, name) -> name.toLowerCase(Locale.ROOT).endsWith(".plist"))));
158         } else {
159             LOG.error("Directory: /System/Library/LaunchAgents does not exist");
160         }
161         dir = new File(SYSTEM_LIBRARY_LAUNCH_DAEMONS);
162         if (dir.exists() && dir.isDirectory()) {
163             files.addAll(Arrays.asList(dir.listFiles((f, name) -> name.toLowerCase(Locale.ROOT).endsWith(".plist"))));
164         } else {
165             LOG.error("Directory: /System/Library/LaunchDaemons does not exist");
166         }
167         for (File f : files) {
168             // remove .plist extension
169             String name = f.getName().substring(0, f.getName().length() - 6);
170             int index = name.lastIndexOf('.');
171             String shortName = (index < 0 || index > name.length() - 2) ? name : name.substring(index + 1);
172             if (!running.contains(name) && !running.contains(shortName)) {
173                 OSService s = new OSService(name, 0, STOPPED);
174                 services.add(s);
175             }
176         }
177         return services;
178     }
179 
180     @Override
181     public List<ApplicationInfo> getInstalledApplications() {
182         return installedAppsSupplier.get();
183     }
