35      private static final long BOOTTIME;
36      static {
37          try (CloseableTimeval tv = new CloseableTimeval()) {
38              if (!SysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec.longValue() == 0L) {
39                  // Usually this works. If it doesn't, fall back to text parsing.
40                  // Boot time will be the first consecutive string of digits.
41                  BOOTTIME = ParseUtil.parseLongOrDefault(
42                          ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
43                          System.currentTimeMillis() / 1000);
44              } else {
45                  // tv now points to a 64-bit timeval structure for boot time.
46                  // First 4 bytes are seconds, second 4 bytes are microseconds
47                  // (we ignore)
48                  BOOTTIME = tv.tv_sec.longValue();
49              }
50          }
51      }

76      @Override
77      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
78          String family = this.major > 10 || (this.major == 10 && this.minor >= 12) ? "macOS"
79                  : System.getProperty("os.name");
80          String codeName = parseCodeName();
81          String buildNumber = SysctlUtil.sysctl("kern.osversion", "");
82          return new Pair<>(family, new OSVersionInfo(this.osXVersion, codeName, buildNumber));
83      }

90      @Override
91      public List<OSProcess> queryAllProcesses() {
92          List<OSProcess> procs = new ArrayList<>();
93          int[] pids = new int[this.maxProc];
94          Arrays.fill(pids, -1);
95          int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids,
96                  pids.length * SystemB.INT_SIZE) / SystemB.INT_SIZE;
97          for (int i = 0; i < numberOfProcesses; i++) {
98              if (pids[i] >= 0) {
99                  OSProcess proc = getProcess(pids[i]);
100                 if (proc != null) {
101                     procs.add(proc);
102                 }
103             }
104         }
105         return procs;
106     }

108     @Override
109     public OSProcess getProcess(int pid) {
110         OSProcess proc = new MacOSProcess(pid, this.major, this.minor, this);
111         return proc.getState().equals(OSProcess.State.INVALID) ? null : proc;
112     }

122     }

124     @Override
125     public int getThreadCount() {
126         // Get current pids, then slightly pad in case new process starts while
127         // allocating array space
128         int[] pids = new int[getProcessCount() + 10];
129         int numberOfProcesses = SystemB.INSTANCE.proc_listpids(SystemB.PROC_ALL_PIDS, 0, pids, pids.length)
130                 / SystemB.INT_SIZE;
131         int numberOfThreads = 0;
132         try (Struct.CloseableProcTaskInfo taskInfo = new Struct.CloseableProcTaskInfo()) {
133             for (int i = 0; i < numberOfProcesses; i++) {
134                 int exit = SystemB.INSTANCE.proc_pidinfo(pids[i], SystemB.PROC_PIDTASKINFO, 0, taskInfo,
135                         taskInfo.size());
136                 if (exit != -1) {
137                     numberOfThreads += taskInfo.pti_threadnum;
138                 }
139             }
140         }
141         return numberOfThreads;
142     }
