5   package oshi.driver.windows;
6   
7   import java.util.Locale;
8   
9   /**
10   * Immutable snapshot of a DXGI adapter's identity and dedicated video memory.
11   *
12   * <p>
13   * Populated by {@link oshi.jna.platform.windows.WindowsDxgi#queryAdapters()} and used by
14   * {@code oshi.hardware.platform.windows.WindowsGraphicsCard} to supply accurate VRAM values.
15   * {@code DedicatedVideoMemory} from {@code DXGI_ADAPTER_DESC} is the authoritative Windows API source for dedicated GPU
16   * memory; it is not subject to the 2 GiB cap that affects the 32-bit registry value
17   * {@code HardwareInformation.MemorySize}.
18   */
19  public final class DxgiAdapterInfo {
20  
21      private final String description;
22      private final int vendorId;
23      private final int deviceId;
24      private final long dedicatedVideoMemory;
25      private final int luidLowPart;
26      private final int luidHighPart;
27  
28      /**
29       * @param description          adapter description string from {@code DXGI_ADAPTER_DESC.Description}
30       * @param vendorId             PCI vendor ID
31       * @param deviceId             PCI device ID
32       * @param dedicatedVideoMemory dedicated video memory in bytes
33       * @param luidLowPart          low 32 bits of the adapter LUID
34       * @param luidHighPart         high 32 bits of the adapter LUID
35       */
36      public DxgiAdapterInfo(String description, int vendorId, int deviceId, long dedicatedVideoMemory, int luidLowPart,
37              int luidHighPart) {
38          this.description = description;
39          this.vendorId = vendorId;
40          this.deviceId = deviceId;
41          this.dedicatedVideoMemory = dedicatedVideoMemory;
42          this.luidLowPart = luidLowPart;
43          this.luidHighPart = luidHighPart;
44      }
45  
46      /**
47       * Returns the adapter description string.
48       *
49       * @return adapter description string
50       */
51      public String getDescription() {
52          return description;
53      }
54  
55      /**
56       * Returns the PCI vendor ID.
57       *
58       * @return PCI vendor ID
59       */
60      public int getVendorId() {
61          return vendorId;
62      }
63  
64      /**
65       * Returns the PCI device ID.
66       *
67       * @return PCI device ID
68       */
69      public int getDeviceId() {
70          return deviceId;
71      }
72  
73      /**
74       * Returns the dedicated video memory in bytes.
75       *
76       * @return dedicated video memory in bytes
77       */
78      public long getDedicatedVideoMemory() {
79          return dedicatedVideoMemory;
80      }
81  
82      /**
83       * Returns the low 32 bits of the adapter LUID.
84       *
85       * @return LUID low part
86       */
87      public int getLuidLowPart() {
88          return luidLowPart;
89      }
90  
91      /**
92       * Returns the high 32 bits of the adapter LUID.
93       *
94       * @return LUID high part
95       */
96      public int getLuidHighPart() {
97          return luidHighPart;
98      }
99  
100     @Override
101     public String toString() {
102         return String.format(Locale.ROOT,
103                 "DxgiAdapterInfo{desc='%s', vendor=0x%04X, device=0x%04X, vram=%d, luid=0x%08X_0x%08X}", description,
104                 vendorId, deviceId, dedicatedVideoMemory, luidHighPart, luidLowPart);
105     }
106 }
