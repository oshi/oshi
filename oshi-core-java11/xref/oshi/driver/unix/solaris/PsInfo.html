89      /**
90       * Reads the pr_argc, pr_argv, pr_envp, and pr_dmodel fields from /proc/pid/psinfo
91       *
92       * @param pid    The process ID
93       * @param psinfo A populated {@link SolarisPsInfo} structure containing the offset pointers for these fields
94       * @return A quartet containing the argc, argv, envp and dmodel values, or null if unable to read
95       */
96      public static Quartet<Integer, Long, Long, Byte> queryArgsEnvAddrs(int pid, SolarisPsInfo psinfo) {
97          if (psinfo != null) {
98              int argc = psinfo.pr_argc;
99              // Must have at least one argc (the command itself) so failure here means exit
100             if (argc > 0) {
101                 long argv = Pointer.nativeValue(psinfo.pr_argv);
102                 long envp = Pointer.nativeValue(psinfo.pr_envp);
103                 // Process data model 1 = 32 bit, 2 = 64 bit
104                 byte dmodel = psinfo.pr_dmodel;
105                 // Sanity check
106                 if (dmodel * 4 == (envp - argv) / (argc + 1)) {
107                     return new Quartet<>(argc, argv, envp, dmodel);
108                 }
109                 LOG.trace("Failed data model and offset increment sanity check: dm={} diff={}", dmodel, envp - argv);
110                 return null;
111             }
112             LOG.trace("Failed argc sanity check: argc={}", argc);
113             return null;
114         }
115         LOG.trace("Failed to read psinfo file for pid: {} ", pid);
116         return null;
117     }

119     /**
120      * Read the argument and environment strings from process address space
121      *
122      * @param pid    the process id
123      * @param psinfo A populated {@link SolarisPsInfo} structure containing the offset pointers for these fields
124      * @return A pair containing a list of the arguments and a map of environment variables
125      */
126     public static Pair<List<String>, Map<String, String>> queryArgsEnv(int pid, SolarisPsInfo psinfo) {
127         List<String> args = new ArrayList<>();
128         Map<String, String> env = new LinkedHashMap<>();
129 
130         // Get the arg count and list of env vars
131         Quartet<Integer, Long, Long, Byte> addrs = queryArgsEnvAddrs(pid, psinfo);
132         if (addrs != null) {
133             // Open a file descriptor to the address space
134             String procas = "/proc/" + pid + "/as";
135             int fd = LIBC.open(procas, 0);
136             if (fd < 0) {
137                 LOG.trace("No permission to read file: {} ", procas);
138                 return new Pair<>(args, env);
139             }
140             try {
141                 // Non-null addrs means argc > 0
142                 int argc = addrs.getA();
143                 long argv = addrs.getB();
144                 long envp = addrs.getC();
145                 long increment = addrs.getD() * 4L;
146 
147                 // Reusable buffer
148                 long bufStart = 0;
149                 try (Memory buffer = new Memory(PAGE_SIZE * 2)) {
150                     size_t bufSize = new size_t(buffer.size());
151 
152                     // Read the pointers to the arg strings
153                     // We know argc so we can count them
154                     long[] argp = new long[argc];
155                     long offset = argv;
156                     for (int i = 0; i < argc; i++) {
157                         bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, offset);
158                         argp[i] = bufStart == 0 ? 0 : getOffsetFromBuffer(buffer, offset - bufStart, increment);
159                         offset += increment;
160                     }
161 
162                     // Also read the pointers to the env strings
163                     // We don't know how many, so stop when we get to null pointer
164                     List<Long> envPtrList = new ArrayList<>();
165                     offset = envp;
166                     long addr = 0;
167                     int limit = 500; // sane max env strings to stop at
168                     do {
169                         bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, offset);
170                         addr = bufStart == 0 ? 0 : getOffsetFromBuffer(buffer, offset - bufStart, increment);
171                         if (addr != 0) {
172                             envPtrList.add(addr);
173                         }
174                         offset += increment;
175                     } while (addr != 0 && --limit > 0);
176 
177                     // Now read the arg strings from the buffer
178                     for (int i = 0; i < argp.length && argp[i] != 0; i++) {
179                         bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, argp[i]);
180                         if (bufStart != 0) {
181                             String argStr = buffer.getString(argp[i] - bufStart);
182                             if (!argStr.isEmpty()) {
183                                 args.add(argStr);
184                             }
185                         }
186                     }
187 
188                     // And now read the env strings from the buffer
189                     for (Long envPtr : envPtrList) {
190                         bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, envPtr);
191                         if (bufStart != 0) {
192                             String envStr = buffer.getString(envPtr - bufStart);
193                             int idx = envStr.indexOf('=');
194                             if (idx > 0) {
195                                 env.put(envStr.substring(0, idx), envStr.substring(idx + 1));
196                             }
197                         }
198                     }
199                 }
200             } finally {
201                 LIBC.close(fd);
202             }
203         }
204         return new Pair<>(args, env);
205     }

207     /**
208      * Reads the page containing addr into buffer, unless the buffer already contains that page (as indicated by the
209      * bufStart address), in which case nothing is changed.
210      *
211      * @param fd       The file descriptor for the address space
212      * @param buffer   An allocated buffer, possibly with data reread from bufStart
213      * @param bufSize  The size of the buffer
214      * @param bufStart The start of data currently in bufStart, or 0 if uninitialized
215      * @param addr     THe address whose page to read into the buffer
216      * @return The new starting pointer for the buffer
217      */
218     private static long conditionallyReadBufferFromStartOfPage(int fd, Memory buffer, size_t bufSize, long bufStart,
219             long addr) {
220         // If we don't have the right buffer, update it
221         if (addr < bufStart || addr - bufStart > PAGE_SIZE) {
222             long newStart = Math.floorDiv(addr, PAGE_SIZE) * PAGE_SIZE;
223             ssize_t result = LIBC.pread(fd, buffer, bufSize, new NativeLong(newStart));
224             // May return less than asked but should be at least a full page
225             if (result.longValue() < PAGE_SIZE) {
226                 LOG.debug("Failed to read page from address space: {} bytes read", result.longValue());
227                 return 0;
228             }
229             return newStart;
230         }
231         return bufStart;
232     }
233 
234     private static long getOffsetFromBuffer(Memory buffer, long offset, long increment) {
235         return increment == 8 ? buffer.getLong(offset) : buffer.getInt(offset);
236     }
