13  import java.time.temporal.ChronoField;

17  import java.util.Locale;
18  import java.util.regex.Matcher;
19  import java.util.regex.Pattern;
20  
21  import com.sun.jna.Platform;
22  
23  import oshi.annotation.concurrent.ThreadSafe;
24  import oshi.software.os.OSSession;
25  import oshi.util.Constants;

28  /**
29   * Utility to query logged in users.
30   */
31  @ThreadSafe
32  public final class Who {
33  
34      // sample format:
35      // oshi pts/0 2020-05-14 21:23 (192.168.1.23)
36      private static final Pattern WHO_FORMAT_LINUX = Pattern
37              .compile("(\\S+)\\s+(\\S+)\\s+(\\d{4}-\\d{2}-\\d{2})\\s+(\\d{2}:\\d{2})\\s*(?:\\((.+)\\))?");
38      private static final DateTimeFormatter WHO_DATE_FORMAT_LINUX = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm");
39      // oshi ttys000 May 4 23:50 (192.168.1.23)
40      // middle 12 characters from Thu Nov 24 18:22:48 1986
41      private static final Pattern WHO_FORMAT_UNIX = Pattern
42              .compile("(\\S+)\\s+(\\S+)\\s+(\\S+)\\s+(\\d+)\\s+(\\d{2}:\\d{2})\\s*(?:\\((.+)\\))?");
43      private static final DateTimeFormatter WHO_DATE_FORMAT_UNIX = new DateTimeFormatterBuilder()
44              .appendPattern("MMM d HH:mm").parseDefaulting(ChronoField.YEAR, Year.now().getValue())
45              .toFormatter(Locale.US);
46  
47      private Who() {
48      }
49  
50      /**
51       * Query {@code who} to get logged in users
52       *
53       * @return A list of logged in user sessions
54       */
55      public static synchronized List<OSSession> queryWho() {
56          List<OSSession> whoList = new ArrayList<>();
57          List<String> who = ExecutingCommand.runNative("who");
58          for (String s : who) {
59              boolean matched = false;
60              if (Platform.isLinux()) {
61                  matched = matchLinux(whoList, s);
62              }
63              if (!matched) {
64                  matchUnix(whoList, s);
65              }
66          }
67          return whoList;
68      }
69  
70      /**
71       * Attempt to match Linux WHO format and add to the list
72       *
73       * @param whoList the list to add to
74       * @param s       the string to match
75       * @return true if successful, false otherwise
76       */
77      private static boolean matchLinux(List<OSSession> whoList, String s) {
78          Matcher m = WHO_FORMAT_LINUX.matcher(s);
79          if (m.matches()) {
80              try {
81                  whoList.add(new OSSession(m.group(1), m.group(2),
82                          LocalDateTime.parse(m.group(3) + " " + m.group(4), WHO_DATE_FORMAT_LINUX)
83                                  .atZone(ZoneId.systemDefault()).toInstant().toEpochMilli(),
84                          m.group(5) == null ? Constants.UNKNOWN : m.group(5)));
85                  return true;
86              } catch (DateTimeParseException | NullPointerException e) {
87                  // shouldn't happen if regex matches and OS is producing sensible dates
88              }
89          }
90          return false;
91      }
92  
93      /**
94       * Attempt to match Unix WHO format and add to the list
95       *
96       * @param whoList the list to add to
97       * @param s       the string to match
98       * @return true if successful, false otherwise
99       */
100     private static boolean matchUnix(List<OSSession> whoList, String s) {
101         Matcher m = WHO_FORMAT_UNIX.matcher(s);
102         if (m.matches()) {
103             try {
104                 // Missing year, parse date time with current year
105                 LocalDateTime login = LocalDateTime.parse(m.group(3) + " " + m.group(4) + " " + m.group(5),
106                         WHO_DATE_FORMAT_UNIX);
107                 // If this date is in the future, subtract a year
108                 if (login.isAfter(LocalDateTime.now())) {
109                     login = login.minus(1, ChronoUnit.YEARS);
110                 }
111                 long millis = login.atZone(ZoneId.systemDefault()).toInstant().toEpochMilli();
112                 whoList.add(new OSSession(m.group(1), m.group(2), millis, m.group(6) == null ? "" : m.group(6)));
113                 return true;
114             } catch (DateTimeParseException | NullPointerException e) {
115                 // shouldn't happen if regex matches and OS is producing sensible dates
116             }
117         }
118         return false;
119     }
120 }
