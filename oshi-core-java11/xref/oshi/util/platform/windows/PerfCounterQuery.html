27  /**
28   * Enables queries of Performance Counters using wild cards to filter instances
29   */
30  @ThreadSafe
31  public final class PerfCounterQuery {
32  
33      private static final Logger LOG = LoggerFactory.getLogger(PerfCounterQuery.class);
34  
35      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
36  
37      // Use a thread safe set to cache failed pdh queries
38      private static final Set<String> FAILED_QUERY_CACHE = ConcurrentHashMap.newKeySet();
39  
40      // A map to cache localization strings
41      private static final ConcurrentHashMap<String, String> LOCALIZE_CACHE = new ConcurrentHashMap<>();
42  
43      /*
44       * Multiple classes use these constants
45       */
46      public static final String TOTAL_INSTANCE = "_Total";
47      public static final String TOTAL_OR_IDLE_INSTANCES = "_Total|Idle";
48      public static final String TOTAL_INSTANCES = "*_Total";
49      public static final String NOT_TOTAL_INSTANCE = "^" + TOTAL_INSTANCE;
50      public static final String NOT_TOTAL_INSTANCES = "^" + TOTAL_INSTANCES;
51  
52      private PerfCounterQuery() {
53      }
54  
55      /**
56       * Query the a Performance Counter using PDH, with WMI backup on failure, for values corresponding to the property
57       * enum.
58       *
59       * @param <T>          The enum type of {@code propertyEnum}
60       * @param propertyEnum An enum which implements
61       *                     {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty} and contains the WMI
62       *                     field (Enum value) and PDH Counter string (instance and counter)
63       * @param perfObject   The PDH object for this counter; all counters on this object will be refreshed at the same
64       *                     time
65       * @param perfWmiClass The WMI PerfData_RawData_* class corresponding to the PDH object
66       * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} on success, or an empty map if both PDH
67       *         and WMI queries failed.
68       */
69      public static <T extends Enum<T>> Map<T, Long> queryValues(Class<T> propertyEnum, String perfObject,
70              String perfWmiClass) {
71          if (!FAILED_QUERY_CACHE.contains(perfObject)) {
72              Map<T, Long> valueMap = queryValuesFromPDH(propertyEnum, perfObject);
73              if (!valueMap.isEmpty()) {
74                  return valueMap;
75              }
76              // If we are here, query failed
77              LOG.warn("Disabling further attempts to query {}.", perfObject);
78              FAILED_QUERY_CACHE.add(perfObject);
79          }
80          return queryValuesFromWMI(propertyEnum, perfWmiClass);
81      }
82  
83      /**
84       * Query the a Performance Counter using PDH for values corresponding to the property enum.
85       *
86       * @param <T>          The enum type of {@code propertyEnum}
87       * @param propertyEnum An enum which implements
88       *                     {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty} and contains the WMI
89       *                     field (Enum value) and PDH Counter string (instance and counter)
90       * @param perfObject   The PDH object for this counter; all counters on this object will be refreshed at the same
91       *                     time
92       * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} on success, or an empty map if the PDH
93       *         query failed.
94       */
95      public static <T extends Enum<T>> Map<T, Long> queryValuesFromPDH(Class<T> propertyEnum, String perfObject) {
96          T[] props = propertyEnum.getEnumConstants();
97          // If pre-Vista, localize the perfObject
98          String perfObjectLocalized = PerfCounterQuery.localizeIfNeeded(perfObject, false);
99          EnumMap<T, PerfCounter> counterMap = new EnumMap<>(propertyEnum);
100         EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
101         try (PerfCounterQueryHandler pdhQueryHandler = new PerfCounterQueryHandler()) {
102             // Set up the query and counter handles
103             for (T prop : props) {
104                 PerfCounter counter = PerfDataUtil.createCounter(perfObjectLocalized,
105                         ((PdhCounterProperty) prop).getInstance(), ((PdhCounterProperty) prop).getCounter());
106                 counterMap.put(prop, counter);
107                 if (!pdhQueryHandler.addCounterToQuery(counter)) {
108                     return valueMap;
109                 }
110             }
111             // And then query. Zero timestamp means update failed
112             if (0 < pdhQueryHandler.updateQuery()) {
113                 for (T prop : props) {
114                     valueMap.put(prop, pdhQueryHandler.queryCounter(counterMap.get(prop)));
115                 }
116             }
117         }
118         return valueMap;
119     }

121     /**
122      * Query the a Performance Counter using WMI for values corresponding to the property enum.
123      *
124      * @param <T>          The enum type of {@code propertyEnum}
125      * @param propertyEnum An enum which implements
126      *                     {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty} and contains the WMI
127      *                     field (Enum value) and PDH Counter string (instance and counter)
128      * @param wmiClass     The WMI PerfData_RawData_* class corresponding to the PDH object
129      * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} if successful, an empty map if the WMI
130      *         query failed.
131      */
132     public static <T extends Enum<T>> Map<T, Long> queryValuesFromWMI(Class<T> propertyEnum, String wmiClass) {
133         WmiQuery<T> query = new WmiQuery<>(wmiClass, propertyEnum);
134         WmiResult<T> result = Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(query);
135         EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
136         if (result.getResultCount() > 0) {
137             for (T prop : propertyEnum.getEnumConstants()) {
138                 switch (result.getCIMType(prop)) {
139                 case Wbemcli.CIM_UINT16:
140                     valueMap.put(prop, (long) WmiUtil.getUint16(result, prop, 0));
141                     break;
142                 case Wbemcli.CIM_UINT32:
143                     valueMap.put(prop, WmiUtil.getUint32asLong(result, prop, 0));
144                     break;
145                 case Wbemcli.CIM_UINT64:
146                     valueMap.put(prop, WmiUtil.getUint64(result, prop, 0));
147                     break;
148                 case Wbemcli.CIM_DATETIME:
149                     valueMap.put(prop, WmiUtil.getDateTime(result, prop, 0).toInstant().toEpochMilli());
150                     break;
151                 default:
152                     throw new ClassCastException("Unimplemented CIM Type Mapping.");
153                 }
154             }
155         }
156         return valueMap;
157     }

159     /**
160      * Localize a PerfCounter string. English counter names should normally be in
161      * {@code HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows
162      * NT\CurrentVersion\Perflib\009\Counter}, but language manipulations may delete the {@code 009} index. In this case
163      * we can assume English must be the language and continue. We may still fail to match the name if the assumption is
164      * wrong but it's better than nothing.
165      *
166      * @param perfObject A String to localize
167      * @param force      If true, always localize
168      * @return The localized string if localization successful, or the original string otherwise.
169      */
170     public static String localizeIfNeeded(String perfObject, boolean force) {
171         return !force && IS_VISTA_OR_GREATER ? perfObject
172                 : LOCALIZE_CACHE.computeIfAbsent(perfObject, PerfCounterQuery::localizeUsingPerfIndex);
173     }
174 
175     private static String localizeUsingPerfIndex(String perfObject) {
176         String localized = perfObject;
177         try {
178             localized = PdhUtil.PdhLookupPerfNameByIndex(null, PdhUtil.PdhLookupPerfIndexByEnglishName(perfObject));
179         } catch (Win32Exception e) {
180             LOG.warn(
181                     "Unable to locate English counter names in registry Perflib 009. Assuming English counters. Error {}. {}",
182                     String.format("0x%x", e.getHR().intValue()),
183                     "See https://support.microsoft.com/en-us/help/300956/how-to-manually-rebuild-performance-counter-library-values");
184         } catch (PdhException e) {
185             LOG.warn("Unable to localize {} performance counter.  Error {}.", perfObject,
186                     String.format("0x%x", e.getErrorCode()));
187         }
188         if (localized.isEmpty()) {
189             return perfObject;
190         }
191         LOG.debug("Localized {} to {}", perfObject, localized);
192         return localized;
193     }
