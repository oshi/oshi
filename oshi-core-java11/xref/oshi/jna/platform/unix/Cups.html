5   package oshi.jna.platform.unix;
6   
7   import com.sun.jna.Library;
8   import com.sun.jna.Native;
9   import com.sun.jna.Pointer;
10  import com.sun.jna.Structure;
11  import com.sun.jna.Structure.FieldOrder;
12  import com.sun.jna.ptr.PointerByReference;
13  
14  /**
15   * CUPS (Common Unix Printing System) library. This class should be considered non-API as it may be removed if/when its
16   * code is incorporated into the JNA project.
17   */
18  public interface Cups extends Library {
19  
20      Cups INSTANCE = Native.load("cups", Cups.class);
21  
22      // Printer state constants from cups/cups.h
23      int IPP_PRINTER_IDLE = 3;
24      int IPP_PRINTER_PROCESSING = 4;
25      int IPP_PRINTER_STOPPED = 5;
26  
27      // Printer type bit flags from cups/cups.h
28      int CUPS_PRINTER_REMOTE = 0x0002;
29  
30      /**
31       * CUPS destination (printer) structure.
32       */
33      @FieldOrder({ "name", "instance", "is_default", "num_options", "options" })
34      class CupsDest extends Structure {
35          public String name;
36          public String instance;
37          public int is_default;
38          public int num_options;
39          public Pointer options; // cups_option_t*
40  
41          public CupsDest() {
42              super();
43          }
44  
45          public CupsDest(Pointer p) {
46              super(p);
47              read();
48          }
49      }
50  
51      /**
52       * Gets all available destinations (printers and classes).
53       *
54       * @param dests Pointer to receive destination array
55       * @return Number of destinations
56       */
57      int cupsGetDests(PointerByReference dests);
58  
59      /**
60       * Frees the memory used by a destination array.
61       *
62       * @param num_dests Number of destinations
63       * @param dests     Pointer to destination array
64       */
65      void cupsFreeDests(int num_dests, Pointer dests);
66  
67      /**
68       * Gets the default printer name.
69       *
70       * @return Default printer name or null if none
71       */
72      String cupsGetDefault();
73  
74      /**
75       * Gets an option value from a destination.
76       *
77       * @param name        Option name
78       * @param num_options Number of options
79       * @param options     Pointer to options array
80       * @return Option value or null if not found
81       */
82      String cupsGetOption(String name, int num_options, Pointer options);
83  }
