5   package oshi.hardware.platform.mac;
6   
7   import com.sun.jna.Native;
8   import com.sun.jna.platform.mac.SystemB;
9   
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.hardware.VirtualMemory;
14  import oshi.jna.ByRef.CloseableLongByReference;
15  import oshi.jna.Struct.CloseableVMStatistics;
16  import oshi.jna.ByRef.CloseableIntByReference;
17  import oshi.util.platform.mac.SysctlUtil;
18  
19  @ThreadSafe
20  final class MacGlobalMemoryJNA extends MacGlobalMemory {
21  
22      private static final Logger LOG = LoggerFactory.getLogger(MacGlobalMemoryJNA.class);
23  
24      @Override
25      protected long queryVmStats() {
26          try (CloseableVMStatistics vmStats = new CloseableVMStatistics();
27                  CloseableIntByReference size = new CloseableIntByReference(vmStats.size() / SystemB.INT_SIZE)) {
28              if (0 != SystemB.INSTANCE.host_statistics(SystemB.INSTANCE.mach_host_self(), SystemB.HOST_VM_INFO, vmStats,
29                      size)) {
30                  LOG.error("Failed to get host VM info. Error code: {}", Native.getLastError());
31                  return 0L;
32              }
33              return (vmStats.free_count + vmStats.inactive_count) * getPageSize();
34          }
35      }
36  
37      @Override
38      protected long sysctl(String name, long defaultValue) {
39          return SysctlUtil.sysctl(name, defaultValue);
40      }
41  
42      @Override
43      protected long host_page_size() {
44          try (CloseableLongByReference pPageSize = new CloseableLongByReference()) {
45              if (0 == SystemB.INSTANCE.host_page_size(SystemB.INSTANCE.mach_host_self(), pPageSize)) {
46                  return pPageSize.getValue();
47              }
48          }
49          return -1;
50      }
51  
52      protected VirtualMemory createVirtualMemory() {
53          return new MacVirtualMemoryJNA(this);
54      }
55  
56  }
