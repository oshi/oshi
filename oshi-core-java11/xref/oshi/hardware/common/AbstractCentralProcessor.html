79      /**
80       * Create a Processor
81       */
82      protected AbstractCentralProcessor() {
83          Pair<List<LogicalProcessor>, List<PhysicalProcessor>> processorLists = initProcessorCounts();
84          // Populate logical processor lists.
85          this.logicalProcessors = Collections.unmodifiableList(processorLists.getA());
86          if (processorLists.getB() == null) {
87              Set<Integer> pkgCoreKeys = this.logicalProcessors.stream()
88                      .map(p -> (p.getPhysicalPackageNumber() << 16) + p.getPhysicalProcessorNumber())
89                      .collect(Collectors.toSet());
90              List<PhysicalProcessor> physProcs = pkgCoreKeys.stream().sorted()
91                      .map(k -> new PhysicalProcessor(k >> 16, k & 0xffff)).collect(Collectors.toList());
92              this.physicalProcessors = Collections.unmodifiableList(physProcs);
93          } else {
94              this.physicalProcessors = Collections.unmodifiableList(processorLists.getB());
95          }
96          // Init processor counts
97          Set<Integer> physPkgs = new HashSet<>();
98          for (LogicalProcessor logProc : this.logicalProcessors) {
99              int pkg = logProc.getPhysicalPackageNumber();
100             physPkgs.add(pkg);
101         }
102         this.logicalProcessorCount = this.logicalProcessors.size();
103         this.physicalProcessorCount = this.physicalProcessors.size();
104         this.physicalPackageCount = physPkgs.size();
105     }

214     @Override
215     public double getSystemCpuLoadBetweenTicks(long[] oldTicks) {
216         if (oldTicks.length != TickType.values().length) {
217             throw new IllegalArgumentException(
218                     "Tick array " + oldTicks.length + " should have " + TickType.values().length + " elements");
219         }
220         long[] ticks = getSystemCpuLoadTicks();
221         // Calculate total
222         long total = 0;
223         for (int i = 0; i < ticks.length; i++) {
224             total += ticks[i] - oldTicks[i];
225         }
226         // Calculate idle from difference in idle and IOwait
227         long idle = ticks[TickType.IDLE.getIndex()] + ticks[TickType.IOWAIT.getIndex()]
228                 - oldTicks[TickType.IDLE.getIndex()] - oldTicks[TickType.IOWAIT.getIndex()];
229         LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
230 
231         return total > 0 ? (double) (total - idle) / total : 0d;
232     }
233 
234     @Override
235     public double[] getProcessorCpuLoadBetweenTicks(long[][] oldTicks) {
236         if (oldTicks.length != this.logicalProcessorCount || oldTicks[0].length != TickType.values().length) {
237             throw new IllegalArgumentException(
238                     "Tick array " + oldTicks.length + " should have " + this.logicalProcessorCount
239                             + " arrays, each of which has " + TickType.values().length + " elements");
240         }
241         long[][] ticks = getProcessorCpuLoadTicks();
242         double[] load = new double[this.logicalProcessorCount];
243         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
244             long total = 0;
245             for (int i = 0; i < ticks[cpu].length; i++) {
246                 total += ticks[cpu][i] - oldTicks[cpu][i];
247             }
248             // Calculate idle from difference in idle and IOwait
249             long idle = ticks[cpu][TickType.IDLE.getIndex()] + ticks[cpu][TickType.IOWAIT.getIndex()]
250                     - oldTicks[cpu][TickType.IDLE.getIndex()] - oldTicks[cpu][TickType.IOWAIT.getIndex()];
251             LOG.trace("CPU: {}  Total ticks: {}  Idle ticks: {}", cpu, total, idle);
252             // update
253             load[cpu] = total > 0 && idle >= 0 ? (double) (total - idle) / total : 0d;
254         }
255         return load;
256     }

273     /**
274      * Creates a Processor ID by encoding the stepping, model, family, and feature
275      * flags.
276      *
277      * @param stepping
278      *            The CPU stepping
279      * @param model
280      *            The CPU model
281      * @param family
282      *            The CPU family
283      * @param flags
284      *            A space-delimited list of CPU feature flags
285      * @return The Processor ID string
286      */
287     protected static String createProcessorID(String stepping, String model, String family, String[] flags) {
288         long processorIdBytes = 0L;
289         long steppingL = ParseUtil.parseLongOrDefault(stepping, 0L);
290         long modelL = ParseUtil.parseLongOrDefault(model, 0L);
291         long familyL = ParseUtil.parseLongOrDefault(family, 0L);
292         // 3:0 – Stepping
293         processorIdBytes |= steppingL & 0xf;
294         // 19:16,7:4 – Model
295         processorIdBytes |= (modelL & 0x0f) << 4;
296         processorIdBytes |= (modelL & 0xf0) << 16;
297         // 27:20,11:8 – Family
298         processorIdBytes |= (familyL & 0x0f) << 8;
299         processorIdBytes |= (familyL & 0xf0) << 20;
300         // 13:12 – Processor Type, assume 0
301         long hwcap = 0L;
302         if (Platform.isLinux()) {
303             hwcap = Auxv.queryAuxv().getOrDefault(Auxv.AT_HWCAP, 0L);
304         }
305         if (hwcap > 0) {
306             processorIdBytes |= hwcap << 32;
307         } else {
308             for (String flag : flags) {
309                 switch (flag) { // NOSONAR squid:S1479
310                 case "fpu":
311                     processorIdBytes |= 1L << 32;
312                     break;
313                 case "vme":
314                     processorIdBytes |= 1L << 33;
315                     break;
316                 case "de":
317                     processorIdBytes |= 1L << 34;
318                     break;
319                 case "pse":
320                     processorIdBytes |= 1L << 35;
321                     break;
322                 case "tsc":
323                     processorIdBytes |= 1L << 36;
324                     break;
325                 case "msr":
326                     processorIdBytes |= 1L << 37;
327                     break;
328                 case "pae":
329                     processorIdBytes |= 1L << 38;
330                     break;
331                 case "mce":
332                     processorIdBytes |= 1L << 39;
333                     break;
334                 case "cx8":
335                     processorIdBytes |= 1L << 40;
336                     break;
337                 case "apic":
338                     processorIdBytes |= 1L << 41;
339                     break;
340                 case "sep":
341                     processorIdBytes |= 1L << 43;
342                     break;
343                 case "mtrr":
344                     processorIdBytes |= 1L << 44;
345                     break;
346                 case "pge":
347                     processorIdBytes |= 1L << 45;
348                     break;
349                 case "mca":
350                     processorIdBytes |= 1L << 46;
351                     break;
352                 case "cmov":
353                     processorIdBytes |= 1L << 47;
354                     break;
355                 case "pat":
356                     processorIdBytes |= 1L << 48;
357                     break;
358                 case "pse-36":
359                     processorIdBytes |= 1L << 49;
360                     break;
361                 case "psn":
362                     processorIdBytes |= 1L << 50;
363                     break;
364                 case "clfsh":
365                     processorIdBytes |= 1L << 51;
366                     break;
367                 case "ds":
368                     processorIdBytes |= 1L << 53;
369                     break;
370                 case "acpi":
371                     processorIdBytes |= 1L << 54;
372                     break;
373                 case "mmx":
374                     processorIdBytes |= 1L << 55;
375                     break;
376                 case "fxsr":
377                     processorIdBytes |= 1L << 56;
378                     break;
379                 case "sse":
380                     processorIdBytes |= 1L << 57;
381                     break;
382                 case "sse2":
383                     processorIdBytes |= 1L << 58;
384                     break;
385                 case "ss":
386                     processorIdBytes |= 1L << 59;
387                     break;
388                 case "htt":
389                     processorIdBytes |= 1L << 60;
390                     break;
391                 case "tm":
392                     processorIdBytes |= 1L << 61;
393                     break;
394                 case "ia64":
395                     processorIdBytes |= 1L << 62;
396                     break;
397                 case "pbe":
398                     processorIdBytes |= 1L << 63;
399                     break;
400                 default:
401                     break;
402                 }
403             }
404         }
405         return String.format("%016X", processorIdBytes);
406     }

408     protected List<PhysicalProcessor> createProcListFromDmesg(List<LogicalProcessor> logProcs,
409             Map<Integer, String> dmesg) {
410         // Check if multiple CPU types
411         boolean isHybrid = dmesg.values().stream().distinct().count() > 1;
412         List<PhysicalProcessor> physProcs = new ArrayList<>();
413         Set<Integer> pkgCoreKeys = new HashSet<>();
414         for (LogicalProcessor logProc : logProcs) {
415             int pkgId = logProc.getPhysicalPackageNumber();
416             int coreId = logProc.getPhysicalProcessorNumber();
417             int pkgCoreKey = (pkgId << 16) + coreId;
418             if (!pkgCoreKeys.contains(pkgCoreKey)) {
419                 pkgCoreKeys.add(pkgCoreKey);
420                 String idStr = dmesg.getOrDefault(logProc.getProcessorNumber(), "");
421                 int efficiency = 0;
422                 // ARM v8 big.LITTLE chips just use the # for efficiency class
423                 // High-performance CPU (big): Cortex-A73, Cortex-A75, Cortex-A76
424                 // High-efficiency CPU (LITTLE): Cortex-A53, Cortex-A55
425                 if (isHybrid && idStr.startsWith("ARM Cortex")) {
426                     efficiency = ParseUtil.getFirstIntValue(idStr) >= 70 ? 1 : 0;
427                 }
428                 physProcs.add(new PhysicalProcessor(pkgId, coreId, efficiency, idStr));
429             }
430         }
431         physProcs.sort(Comparator.comparingInt(PhysicalProcessor::getPhysicalPackageNumber)
432                 .thenComparingInt(PhysicalProcessor::getPhysicalProcessorNumber));
433         return physProcs;
434     }

436     @Override
437     public String toString() {
438         StringBuilder sb = new StringBuilder(getProcessorIdentifier().getName());
439         sb.append("\n ").append(getPhysicalPackageCount()).append(" physical CPU package(s)");
440         sb.append("\n ").append(getPhysicalProcessorCount()).append(" physical CPU core(s)");
441         Map<Integer, Integer> efficiencyCount = new HashMap<>();
442         int maxEfficiency = 0;
443         for (PhysicalProcessor cpu : getPhysicalProcessors()) {
444             int eff = cpu.getEfficiency();
445             efficiencyCount.merge(eff, 1, Integer::sum);
446             if (eff > maxEfficiency) {
447                 maxEfficiency = eff;
448             }
449         }
450         int pCores = efficiencyCount.getOrDefault(maxEfficiency, 0);
451         int eCores = getPhysicalProcessorCount() - pCores;
452         if (eCores > 0) {
453             sb.append(" (").append(pCores).append(" performance + ").append(eCores).append(" efficiency)");
454         }
455         sb.append("\n ").append(getLogicalProcessorCount()).append(" logical CPU(s)");
456         sb.append('\n').append("Identifier: ").append(getProcessorIdentifier().getIdentifier());
457         sb.append('\n').append("ProcessorID: ").append(getProcessorIdentifier().getProcessorID());
458         sb.append('\n').append("Microarchitecture: ").append(getProcessorIdentifier().getMicroarchitecture());
459         return sb.toString();
460     }
