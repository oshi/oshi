5   package oshi.benchmark;
6   
7   import java.util.concurrent.TimeUnit;
8   
9   import org.openjdk.jmh.annotations.Benchmark;
10  import org.openjdk.jmh.annotations.BenchmarkMode;
11  import org.openjdk.jmh.annotations.Fork;
12  import org.openjdk.jmh.annotations.Measurement;
13  import org.openjdk.jmh.annotations.Mode;
14  import org.openjdk.jmh.annotations.OutputTimeUnit;
15  import org.openjdk.jmh.annotations.Scope;
16  import org.openjdk.jmh.annotations.Setup;
17  import org.openjdk.jmh.annotations.State;
18  import org.openjdk.jmh.annotations.Warmup;
19  import org.openjdk.jmh.infra.Blackhole;
20  import org.openjdk.jmh.runner.Runner;
21  import org.openjdk.jmh.runner.RunnerException;
22  import org.openjdk.jmh.runner.options.Options;
23  import org.openjdk.jmh.runner.options.OptionsBuilder;
24  
25  import oshi.hardware.GlobalMemory;
26  import oshi.hardware.VirtualMemory;
27  import oshi.util.GlobalConfig;
28  
29  /**
30   * Side-by-side benchmarks of JNA vs FFM implementations of {@link GlobalMemory} and {@link VirtualMemory} queries.
31   * <p>
32   * Only benchmarks methods that make native calls on each invocation: {@link GlobalMemory#getAvailable()},
33   * {@link VirtualMemory#getSwapUsed()}, and {@link VirtualMemory#getSwapTotal()}. Static values like
34   * {@link GlobalMemory#getTotal()} and {@link GlobalMemory#getPageSize()} are permanently cached and excluded.
35   */
36  @BenchmarkMode(Mode.AverageTime)
37  @OutputTimeUnit(TimeUnit.MICROSECONDS)
38  @State(Scope.Benchmark)
39  @Warmup(iterations = 3, time = 2)
40  @Measurement(iterations = 5, time = 2)
41  @Fork(value = 1, jvmArgsPrepend = "--enable-native-access=ALL-UNNAMED")
42  public class MemoryBenchmark {
43  
44      private GlobalMemory jnaMem;
45      private GlobalMemory ffmMem;
46      private VirtualMemory jnaVm;
47      private VirtualMemory ffmVm;
48  
49      /** Creates a new benchmark instance. Required by JMH for {@code @State} classes. */
50      public MemoryBenchmark() {
51      }
52  
53      /**
54       * Initializes JNA and FFM {@link GlobalMemory} and {@link VirtualMemory} instances with memoization disabled.
55       */
56      @Setup
57      public void setup() {
58          GlobalConfig.set(GlobalConfig.OSHI_UTIL_MEMOIZER_EXPIRATION, 0);
59          jnaMem = new oshi.SystemInfo().getHardware().getMemory();
60          ffmMem = new oshi.ffm.SystemInfo().getHardware().getMemory();
61          jnaVm = jnaMem.getVirtualMemory();
62          ffmVm = ffmMem.getVirtualMemory();
63      }
64  
65      /**
66       * Benchmarks the JNA implementation of memory and virtual-memory queries.
67       *
68       * @param bh JMH black hole to prevent dead-code elimination
69       */
70      @Benchmark
71      public void jna(Blackhole bh) {
72          bh.consume(jnaMem.getAvailable());
73          bh.consume(jnaVm.getSwapTotal());
74          bh.consume(jnaVm.getSwapUsed());
75      }
76  
77      /**
78       * Benchmarks the FFM implementation of memory and virtual-memory queries.
79       *
80       * @param bh JMH black hole to prevent dead-code elimination
81       */
82      @Benchmark
83      public void ffm(Blackhole bh) {
84          bh.consume(ffmMem.getAvailable());
85          bh.consume(ffmVm.getSwapTotal());
86          bh.consume(ffmVm.getSwapUsed());
87      }
88  
89      /**
90       * Standalone entry point for running this benchmark outside the fat jar.
91       *
92       * @param args command-line arguments (unused)
93       * @throws RunnerException if the benchmark fails
94       */
95      public static void main(String[] args) throws RunnerException {
96          Options opt = new OptionsBuilder().include(MemoryBenchmark.class.getSimpleName()).build();
97          new Runner(opt).run();
98      }
99  }
