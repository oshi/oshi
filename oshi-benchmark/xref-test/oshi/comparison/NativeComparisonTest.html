335     @Test
336     void currentProcess() {
337         int pid = jnaOs.getProcessId();
338         assertThat(ffmOs.getProcessId()).isEqualTo(pid);
339         OSProcess jna = jnaOs.getProcess(pid);
340         OSProcess ffm = ffmOs.getProcess(pid);
341         assertThat(ffm).isNotNull();
342         assertThat(ffm.getProcessID()).isEqualTo(jna.getProcessID());
343         assertThat(ffm.getName()).isEqualTo(jna.getName());
344         assertThat(ffm.getPath()).isEqualTo(jna.getPath());
345         assertThat(ffm.getUser()).isEqualTo(jna.getUser());
346         assertThat(ffm.getUserID()).isEqualTo(jna.getUserID());
347         assertThat(ffm.getGroup()).isEqualTo(jna.getGroup());
348         assertThat(ffm.getGroupID()).isEqualTo(jna.getGroupID());
349         assertThat(ffm.getState()).isEqualTo(jna.getState());
350         assertThat(ffm.getParentProcessID()).isEqualTo(jna.getParentProcessID());
351         assertThat(ffm.getPriority()).isEqualTo(jna.getPriority());
352         // Memory values should be in the same ballpark
353         // Virtual size can differ significantly between JNA and FFM due to timing of memory-mapped regions
354         // and DLL loading between the two snapshots; 0.95 tolerance accommodates this variance
355         assertWithinRatio(ffm.getVirtualSize(), jna.getVirtualSize(), 0.95, "process.virtualSize");
356         assertWithinRatio(ffm.getResidentMemory(), jna.getResidentMemory(), 0.75, "process.residentMemory");
357         // Time counters: snapshots taken close together, allow small difference
358         assertThat(ffm.getKernelTime()).as("process.kernelTime").isGreaterThanOrEqualTo(jna.getKernelTime());
359         assertThat(ffm.getUserTime()).as("process.userTime").isGreaterThanOrEqualTo(jna.getUserTime());
360         // Uptime may differ due to memoization (300ms TTL), use max of 10% or 300ms
361         assertThat(Math.abs(ffm.getUpTime() - jna.getUpTime())).as("process.upTime")
362                 .isLessThanOrEqualTo(Math.max(jna.getUpTime() / 10, 300L));
363         assertThat(ffm.getStartTime()).isEqualTo(jna.getStartTime());
364         assertThat(ffm.getCommandLine()).isEqualTo(jna.getCommandLine());
365     }

487     @Test
488     void processListStructure() {
489         // Both should return a non-empty process list
490         List<OSProcess> jna = jnaOs.getProcesses(null, null, 0);
491         List<OSProcess> ffm = ffmOs.getProcesses(null, null, 0);
492         assertThat(ffm).isNotEmpty();
493         // Process counts can differ slightly but should be in the same ballpark
494         assertWithinRatio(ffm.size(), jna.size(), 0.25, "processCount");
495 
496         // The longest-lived processes should be present in both lists
497         List<OSProcess> jnaSorted = jna.stream().sorted(Comparator.comparingLong(OSProcess::getStartTime)).limit(10)
498                 .collect(Collectors.toList());
499         Map<Integer, OSProcess> ffmByPid = ffm.stream()
500                 .collect(Collectors.toMap(OSProcess::getProcessID, Function.identity(), (a, b) -> a));
501         for (OSProcess j : jnaSorted) {
502             OSProcess f = ffmByPid.get(j.getProcessID());
503             if (f != null) {
504                 // Kernel worker threads (kworker) are renamed dynamically based on current work; skip name check
505                 if (!j.getName().startsWith("kworker/")) {
506                     assertThat(f.getName()).as("process[%d].name", j.getProcessID()).isEqualTo(j.getName());
507                 }
508                 assertThat(f.getStartTime()).as("process[%d].startTime", j.getProcessID()).isEqualTo(j.getStartTime());
509             }
510         }
511     }
