86      // This test intentionally validates every WMI value type (string, uint16, uint32, uint64)
87      // so the remaining driver tests can focus on result count and a few representative fields.

89      @Test
90      void testLogicalDiskUnfiltered() {
91          WmiResult<LogicalDiskProperty> jna = Win32LogicalDiskJNA.queryLogicalDisk(null, false);
92          WbemcliUtilFFM.WmiResult<LogicalDiskProperty> ffm = Win32LogicalDiskFFM.queryLogicalDisk(null, false);
93  
94          assertThat(ffm.getResultCount()).as("LogicalDisk result count").isEqualTo(jna.getResultCount());
95  
96          var jnaByName = buildJnaStringMap(jna, LogicalDiskProperty.NAME);
97          var ffmByName = buildFfmStringMap(ffm, LogicalDiskProperty.NAME);
98  
99          assertThat(ffmByName.keySet()).as("LogicalDisk names").containsExactlyInAnyOrderElementsOf(jnaByName.keySet());
100 
101         for (String name : jnaByName.keySet()) {
102             List<Integer> jnaIndices = jnaByName.get(name);
103             List<Integer> ffmIndices = ffmByName.get(name);
104             assertThat(ffmIndices).as("LogicalDisk index count [" + name + "]").hasSameSizeAs(jnaIndices);
105             for (int idx = 0; idx < jnaIndices.size(); idx++) {
106                 int ji = jnaIndices.get(idx);
107                 int fi = ffmIndices.get(idx);
108                 for (LogicalDiskProperty key : LogicalDiskProperty.values()) {
109                     String desc = "LogicalDisk " + key + " [" + name + "]";
110                     switch (key) {
111                         case NAME, DESCRIPTION, FILESYSTEM, VOLUMENAME, PROVIDERNAME -> assertThat(
112                                 WmiUtilFFM.getString(ffm, key, fi)).as(desc).isEqualTo(WmiUtil.getString(jna, key, ji));
113                         case DRIVETYPE -> assertThat(WmiUtilFFM.getUint32(ffm, key, fi)).as(desc)
114                                 .isEqualTo(WmiUtil.getUint32(jna, key, ji));
115                         case ACCESS -> assertThat(WmiUtilFFM.getUint16(ffm, key, fi)).as(desc)
116                                 .isEqualTo(WmiUtil.getUint16(jna, key, ji));
117                         case SIZE -> assertThat(WmiUtilFFM.getUint64(ffm, key, fi)).as(desc)
118                                 .isEqualTo(WmiUtil.getUint64(jna, key, ji));
119                         default -> assertWithinRatio(WmiUtilFFM.getUint64(ffm, key, fi),
120                                 WmiUtil.getUint64(jna, key, ji), 0.05, desc);
121                     }
122                 }
123             }
124         }
125     }

386     @Test
387     void testDiskDrivers() {
388         WmiQueryHandler jnaHandler = WmiQueryHandler.createInstance();
389         Assumptions.assumeTrue(jnaHandler != null, "JNA WmiQueryHandler unavailable");
390         WmiQueryHandlerFFM ffmHandler = WmiQueryHandlerFFM.createInstance();
391         Assumptions.assumeTrue(ffmHandler != null, "FFM WmiQueryHandlerFFM unavailable");
392 
393         boolean jnaComInit = jnaHandler.initCOM();
394         boolean ffmComInit = ffmHandler.initCOM();
395         try {
396             // DiskDrive
397             WmiResult<DiskDriveProperty> jnaDd = Win32DiskDriveJNA.queryDiskDrive(jnaHandler);
398             WbemcliUtilFFM.WmiResult<DiskDriveProperty> ffmDd = Win32DiskDriveFFM.queryDiskDrive(ffmHandler);
399             assertThat(ffmDd.getResultCount()).as("DiskDrive count").isEqualTo(jnaDd.getResultCount());
400 
401             // DiskDriveToDiskPartition
402             WmiResult<DriveToPartitionProperty> jnaDtp = Win32DiskDriveToDiskPartitionJNA
403                     .queryDriveToPartition(jnaHandler);
404             WbemcliUtilFFM.WmiResult<DriveToPartitionProperty> ffmDtp = Win32DiskDriveToDiskPartitionFFM
405                     .queryDriveToPartition(ffmHandler);
406             assertThat(ffmDtp.getResultCount()).as("DriveToPartition count").isEqualTo(jnaDtp.getResultCount());
407 
408             // DiskPartition
409             WmiResult<DiskPartitionProperty> jnaDp = Win32DiskPartitionJNA.queryPartition(jnaHandler);
410             WbemcliUtilFFM.WmiResult<DiskPartitionProperty> ffmDp = Win32DiskPartitionFFM.queryPartition(ffmHandler);
411             assertThat(ffmDp.getResultCount()).as("DiskPartition count").isEqualTo(jnaDp.getResultCount());
412 
413             // LogicalDiskToPartition
414             WmiResult<DiskToPartitionProperty> jnaLtp = Win32LogicalDiskToPartitionJNA.queryDiskToPartition(jnaHandler);
415             WbemcliUtilFFM.WmiResult<DiskToPartitionProperty> ffmLtp = Win32LogicalDiskToPartitionFFM
416                     .queryDiskToPartition(ffmHandler);
417             assertThat(ffmLtp.getResultCount()).as("LogicalDiskToPartition count").isEqualTo(jnaLtp.getResultCount());
418         } finally {
419             if (ffmComInit) {
420                 ffmHandler.unInitCOM();
421             }
422             if (jnaComInit) {
