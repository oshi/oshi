54   * values in the FFM port. JNA runs first (stable baseline), FFM runs second.
55   *
56   * <p>
57   * Activate with: {@code mvn test -pl oshi-benchmark -Pnative-comparison}
58   */

68      /**
69       * Initializes JNA (baseline) and FFM hardware and OS instances with memoization disabled and Windows-specific
70       * configuration options enabled for maximum coverage.
71       */
72      @BeforeAll
73      static void setup() {
74          // Disable memoization so each call gets fresh data for accurate comparison
75          GlobalConfig.set(GlobalConfig.OSHI_UTIL_MEMOIZER_EXPIRATION, 0);
76          // Enable CPU utility counters (Windows 8+) to exercise that branch;
77          // normal CI tests use the default (false) so both paths get coverage
78          GlobalConfig.set(GlobalConfig.OSHI_OS_WINDOWS_CPU_UTILITY, true);
79          // Enable suspended process state detection to exercise thread-based state logic
80          GlobalConfig.set(GlobalConfig.OSHI_OS_WINDOWS_PROCSTATE_SUSPENDED, true);
81          // Enable batch command line queries to exercise Win32ProcessCached
82          GlobalConfig.set(GlobalConfig.OSHI_OS_WINDOWS_COMMANDLINE_BATCH, true);
83  
84          oshi.SystemInfo jnaSi = new oshi.SystemInfo();
85          jnaHal = jnaSi.getHardware();
86          jnaOs = jnaSi.getOperatingSystem();
87  
88          oshi.ffm.SystemInfo ffmSi = new oshi.ffm.SystemInfo();
89          ffmHal = ffmSi.getHardware();
90          ffmOs = ffmSi.getOperatingSystem();
91      }

428         // Kernel and user time are monotonically nondecreasing
429         assertThat(jna.getKernelTime()).as("JNA kernelTime").isGreaterThanOrEqualTo(jnaKernelBefore);
430         assertThat(jna.getUserTime()).as("JNA userTime").isGreaterThanOrEqualTo(jnaUserBefore);

462     @Test
463     void diskStoreUpdateAttributes() {
464         List<HWDiskStore> jna = jnaHal.getDiskStores();
465         List<HWDiskStore> ffm = ffmHal.getDiskStores();
466         assertThat(ffm).isNotEmpty();
467         Map<String, HWDiskStore> ffmByName = ffm.stream()
468                 .collect(Collectors.toMap(HWDiskStore::getName, Function.identity()));
469         int matched = 0;
470         for (HWDiskStore j : jna) {
471             HWDiskStore f = ffmByName.get(j.getName());
472             if (f == null) {
473                 continue;
474             }
475             matched++;
476             // Snapshot before values
477             long jnaReadsBefore = j.getReads();
478             long jnaWritesBefore = j.getWrites();
479             long ffmReadsBefore = f.getReads();
480             long ffmWritesBefore = f.getWrites();
481             // Delay to allow I/O activity
482             Util.sleep(100);
483             // Refresh. Some kstat-backed disks (e.g. removable / installer / iSCSI volumes whose stats
484             // come from a flaky module) intermittently return false on Solaris; both implementations are
485             // entitled to fail individually, but a false from either side means we can't validate this
486             // disk this pass — skip it rather than treating it as a parity bug.
487             boolean jnaUpdated = j.updateAttributes();
488             boolean ffmUpdated = f.updateAttributes();
489             if (!jnaUpdated || !ffmUpdated) {
490                 continue;
491             }
492             // Reads and writes are monotonically nondecreasing
493             assertThat(j.getReads()).as("JNA reads(%s)", j.getName()).isGreaterThanOrEqualTo(jnaReadsBefore);

837     static boolean isLinux() {
838         return PlatformEnum.getCurrentPlatform() == PlatformEnum.LINUX;

841     static boolean isBsd() {
842         PlatformEnum p = PlatformEnum.getCurrentPlatform();
843         return p == PlatformEnum.FREEBSD || p == PlatformEnum.OPENBSD;
844     }

846     /**
847      * On FreeBSD/OpenBSD, OSProcess#getStartTime() is derived as {@code now - elapsedTime} where elapsedTime is the
848      * seconds-resolution {@code ps -o etimes/etime} value. JNA and FFM each capture {@code now} at slightly different
849      * timestamps, so the derived startTimes can differ by tens of milliseconds. Linux/Mac/Windows expose the kernel's
850      * exact start time, so they keep strict equality.
851      */
852     private static void assertStartTimeMatches(long actual, long expected, String description) {
853         if (isBsd()) {
854             assertThat(Math.abs(actual - expected)).as(description + " (BSD tolerance)").isLessThanOrEqualTo(2000L);
855         } else {
856             assertThat(actual).as(description).isEqualTo(expected);
857         }
858     }
