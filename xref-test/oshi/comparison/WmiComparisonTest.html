111     @Test
112     void testLogicalDiskUnfiltered() {
113         WmiResult<LogicalDiskProperty> jna = Win32LogicalDiskJNA.queryLogicalDisk(null, false);
114         WbemcliUtilFFM.WmiResult<LogicalDiskProperty> ffm = Win32LogicalDiskFFM.queryLogicalDisk(null, false);
115 
116         assertThat(ffm.getResultCount()).as("LogicalDisk result count").isEqualTo(jna.getResultCount());
117 
118         var jnaByName = buildJnaStringMap(jna, LogicalDiskProperty.NAME);
119         var ffmByName = buildFfmStringMap(ffm, LogicalDiskProperty.NAME);
120 
121         assertThat(ffmByName.keySet()).as("LogicalDisk names").containsExactlyInAnyOrderElementsOf(jnaByName.keySet());
122 
123         for (String name : jnaByName.keySet()) {
124             List<Integer> jnaIndices = jnaByName.get(name);
125             List<Integer> ffmIndices = ffmByName.get(name);
126             assertThat(ffmIndices).as("LogicalDisk index count [" + name + "]").hasSameSizeAs(jnaIndices);
127             for (int idx = 0; idx < jnaIndices.size(); idx++) {
128                 int ji = jnaIndices.get(idx);
129                 int fi = ffmIndices.get(idx);
130                 for (LogicalDiskProperty key : LogicalDiskProperty.values()) {
131                     String desc = "LogicalDisk " + key + " [" + name + "]";
132                     switch (key) {
133                         case NAME, DESCRIPTION, FILESYSTEM, VOLUMENAME, PROVIDERNAME -> assertThat(
134                                 WmiUtilFFM.getString(ffm, key, fi)).as(desc).isEqualTo(WmiUtil.getString(jna, key, ji));
135                         case DRIVETYPE -> assertThat(WmiUtilFFM.getUint32(ffm, key, fi)).as(desc)
136                                 .isEqualTo(WmiUtil.getUint32(jna, key, ji));
137                         case ACCESS -> assertThat(WmiUtilFFM.getUint16(ffm, key, fi)).as(desc)
138                                 .isEqualTo(WmiUtil.getUint16(jna, key, ji));
139                         case SIZE -> assertThat(WmiUtilFFM.getUint64(ffm, key, fi)).as(desc)
140                                 .isEqualTo(WmiUtil.getUint64(jna, key, ji));
141                         default -> assertWithinRatio(WmiUtilFFM.getUint64(ffm, key, fi),
142                                 WmiUtil.getUint64(jna, key, ji), 0.05, desc);
143                     }
144                 }
145             }
146         }
147     }

173         String firstName = WmiUtil.getString(all, LogicalDiskProperty.NAME, 0);

180 
181         assertThat(WmiUtilFFM.getString(ffm, LogicalDiskProperty.NAME, 0)).as("LogicalDisk filtered NAME")
182                 .isEqualTo(firstName);

541     private static <T extends Enum<T>> Map<String, List<Integer>> buildJnaStringMap(WmiResult<T> result, T keyProp) {
542         Map<String, List<Integer>> map = new HashMap<>();
543         for (int i = 0; i < result.getResultCount(); i++) {
544             map.computeIfAbsent(WmiUtil.getString(result, keyProp, i), k -> new ArrayList<>()).add(i);
545         }
546         return map;
547     }
