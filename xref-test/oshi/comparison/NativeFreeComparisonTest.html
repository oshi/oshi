33  /**
34   * Compares the native-free (NF) provider against the JNA provider on Linux to verify that procfs/sysfs/command-line
35   * implementations produce the same results as native calls.
36   * <p>
37   * JNA is the baseline (stable, well-tested). NF values should match for deterministic fields and be within tolerance
38   * for dynamic values.
39   */
40  @EnabledOnOs(OS.LINUX)
41  class NativeFreeComparisonTest {
42  
43      private static HardwareAbstractionLayer jnaHal;
44      private static HardwareAbstractionLayer nfHal;
45      private static OperatingSystem jnaOs;
46      private static OperatingSystem nfOs;
47  
48      @BeforeAll
49      static void setup() {
50          oshi.SystemInfo jnaSi = new oshi.SystemInfo();
51          jnaHal = jnaSi.getHardware();
52          jnaOs = jnaSi.getOperatingSystem();
53  
54          oshi.nativefree.SystemInfo nfSi = new oshi.nativefree.SystemInfo();
55          nfHal = nfSi.getHardware();
56          nfOs = nfSi.getOperatingSystem();
57      }

71      }

178     @Test
179     void diskStores() {
180         List<HWDiskStore> jna = jnaHal.getDiskStores();
181         List<HWDiskStore> nf = nfHal.getDiskStores();
182         assertThat(nf).hasSameSizeAs(jna);
183         Map<String, HWDiskStore> nfByName = nf.stream()
184                 .collect(Collectors.toMap(HWDiskStore::getName, Function.identity()));
185         for (HWDiskStore j : jna) {
186             HWDiskStore n = nfByName.get(j.getName());
187             assertThat(n).as("disk %s", j.getName()).isNotNull();
188             // Model may differ in space vs underscore: sysfs uses raw kernel string,
189             // udev may normalize spaces to underscores (or vice versa depending on device)
190             assertThat(n.getModel().replace('_', ' ')).isEqualTo(j.getModel().replace('_', ' '));
191             // Serial: udev ID_SERIAL_SHORT may expose serials not available in sysfs device/serial
192             if (!n.getSerial().isEmpty()) {
193                 assertThat(n.getSerial()).isEqualTo(j.getSerial());
194             }
195             assertThat(n.getSize()).isEqualTo(j.getSize());
196             assertThat(n.getReads()).as("reads(%s)", j.getName()).isGreaterThanOrEqualTo(j.getReads());
197             assertThat(n.getPartitions().size()).as("partitions(%s)", j.getName()).isEqualTo(j.getPartitions().size());
198         }
199     }

215             assertThat(n.getIPv6addr()).isEqualTo(j.getIPv6addr());

223     @Test
224     void fileSystem() {
225         FileSystem jnaFs = jnaOs.getFileSystem();
226         FileSystem nfFs = nfOs.getFileSystem();
227 
228         List<OSFileStore> jnaStores = jnaFs.getFileStores();
229         List<OSFileStore> nfStores = nfFs.getFileStores();
230         assertThat(nfStores).hasSameSizeAs(jnaStores);
231         Map<String, OSFileStore> nfByMount = nfStores.stream()
232                 .collect(Collectors.toMap(OSFileStore::getMount, Function.identity(), (a, b) -> a));
233         for (OSFileStore j : jnaStores) {
234             OSFileStore nf = nfByMount.get(j.getMount());
235             assertThat(nf).as("fileStore at %s", j.getMount()).isNotNull();
236             assertThat(nf.getName()).isEqualTo(j.getName());
237             assertThat(nf.getType()).isEqualTo(j.getType());
238             assertThat(nf.getVolume()).isEqualTo(j.getVolume());
239             assertThat(nf.getTotalSpace()).isEqualTo(j.getTotalSpace());
240             assertWithinRatio(nf.getUsableSpace(), j.getUsableSpace(), 0.25, "usableSpace(" + j.getMount() + ")");
241         }
242     }
