22  /**
23   * Compares FFM PerfCounters (PDH) and FFM Registry (HKEY_PERFORMANCE_DATA) implementations. Both ultimately read the
24   * same kernel performance data, so results should be nearly identical — differences only arise from the time between
25   * the two calls and from dynamic fields the scheduler can change.
26   */
27  @DisabledIf("isNotWindows")
28  class RegistryComparisonTest {
29  
30      // Absolute tolerance for the zero-value branch in assertWithinRatio: a process may transiently show 0 bytes
31      // for a memory gauge while the other read captures a small allocation.
32      private static final long ZERO_TOLERANCE_BYTES = 1024L * 1024L;
33  
34      @Test
35      void testProcessData() {
36          // Registry first, then PerfCounters
37          Map<Integer, ProcessPerfCounterBlock> reg = ProcessPerformanceDataFFM.buildProcessMapFromRegistry(null);
38          Map<Integer, ProcessPerfCounterBlock> pdh = ProcessPerformanceDataFFM.buildProcessMapFromPerfCounters(null);
39          assertThat(reg).as("Registry process map").isNotNull().isNotEmpty();
40          assertThat(pdh).as("PerfCounter process map").isNotNull().isNotEmpty();
41  
42          // PID sets should have high overlap — processes rarely start/stop in a few ms
43          Set<Integer> commonPids = new HashSet<>(reg.keySet());
44          commonPids.retainAll(pdh.keySet());
45          int minSize = Math.min(reg.size(), pdh.size());
46          assertThat(commonPids.size()).as("common PID overlap (reg=%d, pdh=%d)", reg.size(), pdh.size())
47                  .isGreaterThanOrEqualTo((minSize * 90 + 99) / 100);
48  
49          for (int pid : commonPids) {
50              ProcessPerfCounterBlock r = reg.get(pid);
51              ProcessPerfCounterBlock p = pdh.get(pid);
52  
53              // Static fields — must be exact
54              assertThat(p.getName()).as("process[%d].name", pid).isEqualTo(r.getName());
55              assertThat(p.getParentProcessID()).as("process[%d].ppid", pid).isEqualTo(r.getParentProcessID());
56              assertThat(p.getPriority()).as("process[%d].priority", pid).isEqualTo(r.getPriority());
57              assertThat(p.getStartTime()).as("process[%d].startTime", pid).isEqualTo(r.getStartTime());
58  
59              // Cumulative counters — PDH called second, should be >=
60              assertThat(p.getBytesRead()).as("process[%d].bytesRead", pid).isGreaterThanOrEqualTo(r.getBytesRead());
61              assertThat(p.getBytesWritten()).as("process[%d].bytesWritten", pid)
62                      .isGreaterThanOrEqualTo(r.getBytesWritten());
63              assertThat(p.getPageFaults()).as("process[%d].pageFaults", pid).isGreaterThanOrEqualTo(r.getPageFaults());
64  
65              // Memory gauges — should be close, allow 50% fluctuation on CI
66              assertWithinRatio(p.getWorkingSetSize(), r.getWorkingSetSize(), 0.50,
67                      "process[" + pid + "].workingSetSize");
68              assertWithinRatio(p.getPrivateWorkingSetSize(), r.getPrivateWorkingSetSize(), 0.50,
69                      "process[" + pid + "].privateWorkingSetSize");
70  
71              // Uptime differs by the delta between the two reads
72              assertThat(Math.abs(p.getUpTime() - r.getUpTime())).as("process[%d].upTime delta", pid)
73                      .isLessThanOrEqualTo(Math.max(r.getUpTime() / 10, 1000L));
74          }
75      }
76  
77      @Test
78      void testThreadData() {
79          // Registry first, then PerfCounters
80          Map<Integer, ThreadPerfCounterBlock> reg = ThreadPerformanceDataFFM.buildThreadMapFromRegistry(null);
81          Map<Integer, ThreadPerfCounterBlock> pdh = ThreadPerformanceDataFFM.buildThreadMapFromPerfCounters(null);
82          assertThat(reg).as("Registry thread map").isNotNull().isNotEmpty();
83          assertThat(pdh).as("PerfCounter thread map").isNotNull().isNotEmpty();
84  
85          // TID sets should have high overlap
86          Set<Integer> commonTids = new HashSet<>(reg.keySet());
87          commonTids.retainAll(pdh.keySet());
88          int minSize = Math.min(reg.size(), pdh.size());
89          assertThat(commonTids.size()).as("common TID overlap (reg=%d, pdh=%d)", reg.size(), pdh.size())
90                  .isGreaterThanOrEqualTo((minSize * 90 + 99) / 100);
91  
92          for (int tid : commonTids) {
93              ThreadPerfCounterBlock r = reg.get(tid);
94              ThreadPerfCounterBlock p = pdh.get(tid);
95  
96              // Static fields — must be exact
97              assertThat(p.getOwningProcessID()).as("thread[%d].pid", tid).isEqualTo(r.getOwningProcessID());
98  
99              // Priority and thread state are dynamic — scheduler can change them between reads
100             assertThat(p.getPriority()).as("thread[%d].priority", tid).isGreaterThanOrEqualTo(0);
101 
102             // Cumulative counters — PDH called second, should be >=
103             assertThat(p.getUserTime()).as("thread[%d].userTime", tid).isGreaterThanOrEqualTo(r.getUserTime());
104             assertThat(p.getKernelTime()).as("thread[%d].kernelTime", tid).isGreaterThanOrEqualTo(r.getKernelTime());
105             assertThat(p.getContextSwitches()).as("thread[%d].contextSwitches", tid)
106                     .isGreaterThanOrEqualTo(r.getContextSwitches());
107 
108             // Start time — allow tolerance for Windows clock tick resolution (1/64s ≈ 15.625ms)
109             // and integer division rounding in the now - upTime computation
110             assertThat(Math.abs(p.getStartTime() - r.getStartTime())).as("thread[%d].startTime", tid)
111                     .isLessThanOrEqualTo(20L);
112         }
113     }

115     // ---- Helpers ----

117     private static void assertWithinRatio(double actual, double expected, double ratio, String description) {
118         if (expected == 0 && actual == 0) {
119             return;
120         }
121         if (expected == 0 || actual == 0) {
122             double nonZero = Math.max(Math.abs(expected), Math.abs(actual));
123             assertThat(nonZero).as("%s: one value is 0, other is %.0f", description, nonZero)
124                     .isLessThanOrEqualTo(ZERO_TOLERANCE_BYTES);
125             return;
126         }
127         double min = Math.min(Math.abs(actual), Math.abs(expected));
128         double max = Math.max(Math.abs(actual), Math.abs(expected));
129         assertThat(min / max).as("%s: expected=%f, actual=%f", description, expected, actual)
130                 .isGreaterThanOrEqualTo(1.0 - ratio);
131     }
132 
133     static boolean isNotWindows() {
134         return PlatformEnum.getCurrentPlatform() != PlatformEnum.WINDOWS;
135     }
