178     /**
179      * Test power source.
180      */
181     @Test
182     public void testPowerSource() {
183         SystemInfo si = new SystemInfo();
184         HardwareAbstractionLayer hal = si.getHardware();
185         if (hal.getPowerSources().length > 1) {
186             assertTrue(hal.getPowerSources()[0].getRemainingCapacity() >= 0
187                     && hal.getPowerSources()[0].getRemainingCapacity() <= 1);
188             double epsilon = 1E-6;
189             assertTrue(hal.getPowerSources()[0].getTimeRemaining() > 0
190                     || Math.abs(hal.getPowerSources()[0].getTimeRemaining() - -1) < epsilon
191                     || Math.abs(hal.getPowerSources()[0].getTimeRemaining() - -2) < epsilon);
192         }
193     }

195     /**
196      * Test file system.
197      *
198      * @throws IOException
199      *             Signals that an I/O exception has occurred.
200      */
201     @Test
202     public void testFileSystem() throws IOException {
203         SystemInfo si = new SystemInfo();
204         HardwareAbstractionLayer hal = si.getHardware();
205         if (hal.getFileStores().length > 1) {
206             assertTrue(hal.getFileStores()[0].getTotalSpace() >= 0);
207             assertTrue(hal.getFileStores()[0].getUsableSpace() <= hal.getFileStores()[0].getTotalSpace());
208         }
209         // Hack to extract path from FileStore.toString() is undocumented,
210         // this test will fail if toString format changes
211         if (Platform.isLinux()) {
212             FileStore store = Files.getFileStore((new File("/")).toPath());
213             assertEquals("/", store.toString().replace(" (" + store.name() + ")", ""));
214         }
215     }

252     /**
253      * The main method.
254      *
255      * @param args
256      *            the arguments
257      */
258     public static void main(String[] args) {
259         // Options: ERROR > WARN > INFO > DEBUG > TRACE
260         System.setProperty(org.slf4j.impl.SimpleLogger.DEFAULT_LOG_LEVEL_KEY, "INFO");
261         Logger LOG = LoggerFactory.getLogger(SystemInfoTest.class);
262 
263         LOG.info("Initializing System...");
264         SystemInfo si = new SystemInfo();
265         // software
266         // software: operating system
267         OperatingSystem os = si.getOperatingSystem();
268         System.out.println(os);
269 
270         LOG.info("Initializing Hardware...");
271         // hardware
272         HardwareAbstractionLayer hal = si.getHardware();
273 
274         // hardware: processors
275         System.out.println(hal.getProcessor());
276         System.out.println(" " + hal.getProcessor().getPhysicalProcessorCount() + " physical CPU(s)");
277         System.out.println(" " + hal.getProcessor().getLogicalProcessorCount() + " logical CPU(s)");
278 
279         System.out.println("Identifier: " + hal.getProcessor().getIdentifier());
280         System.out.println("Serial Num: " + hal.getProcessor().getSystemSerialNumber());
281 
282         // hardware: memory
283         LOG.info("Checking Memory...");
284         System.out.println("Memory: " + FormatUtil.formatBytes(hal.getMemory().getAvailable()) + "/"
285                 + FormatUtil.formatBytes(hal.getMemory().getTotal()));
286         // uptime
287         LOG.info("Checking Uptime...");
288         System.out.println("Uptime: " + FormatUtil.formatElapsedSecs(hal.getProcessor().getSystemUptime()));
289 
290         // CPU
291         LOG.info("Checking CPU...");
292         long[] prevTicks = hal.getProcessor().getSystemCpuLoadTicks();
293         System.out.println("CPU ticks @ 0 sec:" + Arrays.toString(prevTicks));
294         // Wait a second...
295         Util.sleep(1000);
296         long[] ticks = hal.getProcessor().getSystemCpuLoadTicks();
297         System.out.println("CPU ticks @ 1 sec:" + Arrays.toString(ticks));
298         long user = ticks[0] - prevTicks[0];
299         long nice = ticks[1] - prevTicks[1];
300         long sys = ticks[2] - prevTicks[2];
301         long idle = ticks[3] - prevTicks[3];
302         long totalCpu = user + nice + sys + idle;
303 
304         System.out.format("User: %.1f%% Nice: %.1f%% System: %.1f%% Idle: %.1f%%%n", 100d * user / totalCpu,
305                 100d * nice / totalCpu, 100d * sys / totalCpu, 100d * idle / totalCpu);
306         System.out.format("CPU load: %.1f%% (counting ticks)%n",
307                 hal.getProcessor().getSystemCpuLoadBetweenTicks() * 100);
308         System.out.format("CPU load: %.1f%% (OS MXBean)%n", hal.getProcessor().getSystemCpuLoad() * 100);
309         double loadAverage = hal.getProcessor().getSystemLoadAverage();
310         System.out.println("CPU load average: " + (loadAverage < 0 ? "N/A" : String.format("%.2f", loadAverage)));
311         // per core CPU
312         StringBuilder procCpu = new StringBuilder("CPU load per processor:");
313         double[] load = hal.getProcessor().getProcessorCpuLoadBetweenTicks();
314         for (int cpu = 0; cpu < load.length; cpu++) {
315             procCpu.append(String.format(" %.1f%%", load[cpu] * 100));
316         }
317         System.out.println(procCpu.toString());
318 
319         // hardware: power
320         LOG.info("Checking Power sources...");
321         StringBuilder sb = new StringBuilder("Power: ");
322         if (hal.getPowerSources().length == 0) {
323             sb.append("Unknown");
324         } else {
325             double timeRemaining = hal.getPowerSources()[0].getTimeRemaining();
326             if (timeRemaining < -1d)
327                 sb.append("Charging");
328             else if (timeRemaining < 0d)
329                 sb.append("Calculating time remaining");
330             else
331                 sb.append(String.format("%d:%02d remaining", (int) (timeRemaining / 3600),
332                         (int) (timeRemaining / 60) % 60));
333         }
334         for (PowerSource pSource : hal.getPowerSources()) {
335             sb.append(String.format("%n %s @ %.1f%%", pSource.getName(), pSource.getRemainingCapacity() * 100d));
336         }
337         System.out.println(sb.toString());
338 
339         // hardware: file system
340         LOG.info("Checking File System...");
341         System.out.println("File System:");
342 
343         OSFileStore[] fsArray = hal.getFileStores();
344         for (OSFileStore fs : fsArray) {
345             long usable = fs.getUsableSpace();
346             long total = fs.getTotalSpace();
347             System.out.format(" %s (%s) %s of %s free (%.1f%%)%n", fs.getName(),
348                     fs.getDescription().isEmpty() ? "file system" : fs.getDescription(), FormatUtil.formatBytes(usable),
349                     FormatUtil.formatBytes(fs.getTotalSpace()), 100d * usable / total);
350         }
351 
352         // hardware: displays
353         LOG.info("Checking Displays...");
354         System.out.println("Displays:");
355         int i = 0;
356         for (Display display : hal.getDisplays()) {
357             System.out.println(" Display " + i + ":");
358             byte[] edid = display.getEdid();
359             System.out.println("  Manuf. ID=" + EdidUtil.getManufacturerID(edid) + ", Product ID="
360                     + EdidUtil.getProductID(edid) + ", " + (EdidUtil.isDigital(edid) ? "Digital" : "Analog")
361                     + ", Serial=" + EdidUtil.getSerialNo(edid) + ", ManufDate=" + (EdidUtil.getWeek(edid) * 12 / 52 + 1)
362                     + "/" + EdidUtil.getYear(edid) + ", EDID v" + EdidUtil.getVersion(edid));
363             int hSize = EdidUtil.getHcm(edid);
364             int vSize = EdidUtil.getVcm(edid);
365             System.out.format("  %d x %d cm (%.1f x %.1f in)%n", hSize, vSize, hSize / 2.54, vSize / 2.54);
366             byte[][] desc = EdidUtil.getDescriptors(edid);
367             for (int d = 0; d < desc.length; d++) {
368                 switch (EdidUtil.getDescriptorType(desc[d])) {
369                 case 0xff:
370                     System.out.println("  Serial Number: " + EdidUtil.getDescriptorText(desc[d]));
371                     break;
372                 case 0xfe:
373                     System.out.println("  Unspecified Text: " + EdidUtil.getDescriptorText(desc[d]));
374                     break;
375                 case 0xfd:
376                     System.out.println("  Range Limits: " + EdidUtil.getDescriptorRangeLimits(desc[d]));
377                     break;
378                 case 0xfc:
379                     System.out.println("  Monitor Name: " + EdidUtil.getDescriptorText(desc[d]));
380                     break;
381                 case 0xfb:
382                     System.out.println("  White Point Data: " + EdidUtil.getDescriptorHex(desc[d]));
383                     break;
384                 case 0xfa:
385                     System.out.println("  Standard Timing ID: " + EdidUtil.getDescriptorHex(desc[d]));
386                     break;
387                 default:
388                     if (EdidUtil.getDescriptorType(desc[d]) <= 0x0f && EdidUtil.getDescriptorType(desc[d]) >= 0x00) {
389                         System.out.println("  Manufacturer Data: " + EdidUtil.getDescriptorHex(desc[d]));
390                     } else {
391                         System.out.println("  Preferred Timing: " + EdidUtil.getTimingDescriptor(desc[d]));
392                     }
393                 }
394             }
395             i++;
396         }
397         LOG.info("Printing JSON:");
398         System.out.println(si.toString());
399 
400     }
