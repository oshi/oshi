58      /**
59       * Test central processor.
60       */
61      @Test
62      public void testCentralProcessor() {
63          SystemInfo si = new SystemInfo();
64          CentralProcessor p = si.getHardware().getProcessor();
65  
66          assertNotNull(p.getVendor());
67          assertTrue(p.getVendorFreq() == -1 || p.getVendorFreq() > 0);
68          p.setVendor("v");
69          assertEquals(p.getVendor(), "v");
70  
71          assertNotNull(p.getName());
72          p.setName("n");
73          assertEquals(p.getName(), "n");
74  
75          assertNotNull(p.getIdentifier());
76          p.setIdentifier("i");
77          assertEquals(p.getIdentifier(), "i");
78  
79          p.setCpu64(true);
80          assertTrue(p.isCpu64bit());
81  
82          assertNotNull(p.getStepping());
83          p.setStepping("s");
84          assertEquals(p.getStepping(), "s");
85  
86          assertNotNull(p.getModel());
87          p.setModel("m");
88          assertEquals(p.getModel(), "m");
89  
90          assertNotNull(p.getFamily());
91          p.setFamily("f");
92          assertEquals(p.getFamily(), "f");
93  
94          assertTrue(p.getSystemCpuLoadBetweenTicks() >= 0 && p.getSystemCpuLoadBetweenTicks() <= 1);
95          assertEquals(p.getSystemCpuLoadTicks().length, 4);
96          assertTrue(p.getSystemIOWaitTicks() >= 0);
97          assertEquals(p.getSystemIrqTicks().length, 2);
98  
99          Util.sleep(500);
100         assertTrue(p.getSystemCpuLoad() >= 0.0 && p.getSystemCpuLoad() <= 1.0);
101         assertEquals(p.getSystemLoadAverage(3).length, 3);
102         if (Platform.isMac() || Platform.isLinux()) {
103             assertTrue(p.getSystemLoadAverage() >= 0.0);
104         }
105 
106         assertEquals(p.getProcessorCpuLoadBetweenTicks().length, p.getLogicalProcessorCount());
107         for (int cpu = 0; cpu < p.getLogicalProcessorCount(); cpu++) {
108             assertTrue(p.getProcessorCpuLoadBetweenTicks()[cpu] >= 0 && p.getProcessorCpuLoadBetweenTicks()[cpu] <= 1);
109             assertEquals(p.getProcessorCpuLoadTicks()[cpu].length, 4);
110         }
111 
112         assertTrue(p.getSystemUptime() > 0);
113         assertNotNull(p.getSystemSerialNumber());
114         assertTrue(p.getLogicalProcessorCount() >= p.getPhysicalProcessorCount());
115         assertTrue(p.getPhysicalProcessorCount() > 0);
116         assertTrue(p.getProcessCount() >= 1);
117         assertTrue(p.getThreadCount() >= p.getProcessCount());
118     }

201      */
202     @Test
203     public void testPowerSource() {
204         SystemInfo si = new SystemInfo();
205         PowerSource[] psArr = si.getHardware().getPowerSources();
206         for (PowerSource ps : psArr) {
207             assertTrue(ps.getRemainingCapacity() >= 0 && ps.getRemainingCapacity() <= 1);
208             double epsilon = 1E-6;
209             assertTrue(ps.getTimeRemaining() > 0 || Math.abs(ps.getTimeRemaining() - -1) < epsilon

245     /**
246      * Test file system.
247      *
248      * @throws IOException
249      *             Signals that an I/O exception has occurred.
250      */
251     @Test
252     public void testFileSystem() throws IOException {
253         SystemInfo si = new SystemInfo();
254         OSFileStore[] fs = si.getHardware().getFileStores();
255         for (int f = 0; f < fs.length; f++) {
256             assertNotNull(fs[f].getName());
257             assertNotNull(fs[f].getDescription());
258             assertNotNull(fs[f].getType());
259             assertTrue(fs[f].getTotalSpace() >= 0);
260             assertTrue(fs[f].getUsableSpace() <= fs[f].getTotalSpace());
261         }
262         // Hack to extract path from FileStore.toString() is undocumented,
263         // this test will fail if toString format changes
264         if (Platform.isLinux()) {
265             FileStore store = Files.getFileStore((new File("/")).toPath());
266             assertEquals("/", store.toString().replace(" (" + store.name() + ")", ""));
267         }

270     /**
271      * The main method.
272      *
273      * @param args
274      *            the arguments
275      */
276     public static void main(String[] args) {
277         // Options: ERROR > WARN > INFO > DEBUG > TRACE
278         System.setProperty(org.slf4j.impl.SimpleLogger.DEFAULT_LOG_LEVEL_KEY, "INFO");
279         Logger LOG = LoggerFactory.getLogger(SystemInfoTest.class);
280 
281         LOG.info("Initializing System...");
282         SystemInfo si = new SystemInfo();
283         // software
284         // software: operating system
285         OperatingSystem os = si.getOperatingSystem();
286         System.out.println(os);
287 
288         LOG.info("Initializing Hardware...");
289         // hardware
290         HardwareAbstractionLayer hal = si.getHardware();
291 
292         // hardware: processors
293         System.out.println(hal.getProcessor());
294         System.out.println(" " + hal.getProcessor().getPhysicalProcessorCount() + " physical CPU(s)");
295         System.out.println(" " + hal.getProcessor().getLogicalProcessorCount() + " logical CPU(s)");
296 
297         System.out.println("Identifier: " + hal.getProcessor().getIdentifier());
298         System.out.println("Serial Num: " + hal.getProcessor().getSystemSerialNumber());
299 
300         // hardware: memory
301         LOG.info("Checking Memory...");
302         System.out.println("Memory: " + FormatUtil.formatBytes(hal.getMemory().getAvailable()) + "/"
303                 + FormatUtil.formatBytes(hal.getMemory().getTotal()));
304         System.out.println("Swap used: " + FormatUtil.formatBytes(hal.getMemory().getSwapUsed()) + "/"
305                 + FormatUtil.formatBytes(hal.getMemory().getSwapTotal()));
306         // uptime
307         LOG.info("Checking Uptime...");
308         System.out.println("Uptime: " + FormatUtil.formatElapsedSecs(hal.getProcessor().getSystemUptime()));
309 
310         // CPU
311         LOG.info("Checking CPU...");
312         long[] prevTicks = hal.getProcessor().getSystemCpuLoadTicks();
313         System.out.println("CPU, IOWait, and IRQ ticks @ 0 sec:" + Arrays.toString(prevTicks) + ", "
314                 + hal.getProcessor().getSystemIOWaitTicks() + ", "
315                 + Arrays.toString(hal.getProcessor().getSystemIrqTicks()));
316         // Wait a second...
317         Util.sleep(1000);
318         long[] ticks = hal.getProcessor().getSystemCpuLoadTicks();
319         System.out.println("CPU, IOWait, and IRQ ticks @ 1 sec:" + Arrays.toString(ticks) + ", "
320                 + hal.getProcessor().getSystemIOWaitTicks() + ", "
321                 + Arrays.toString(hal.getProcessor().getSystemIrqTicks()));
322         long user = ticks[0] - prevTicks[0];
323         long nice = ticks[1] - prevTicks[1];
324         long sys = ticks[2] - prevTicks[2];
325         long idle = ticks[3] - prevTicks[3];
326         long totalCpu = user + nice + sys + idle;
327 
328         System.out.format("User: %.1f%% Nice: %.1f%% System: %.1f%% Idle: %.1f%%%n", 100d * user / totalCpu,
329                 100d * nice / totalCpu, 100d * sys / totalCpu, 100d * idle / totalCpu);
330         System.out.format("CPU load: %.1f%% (counting ticks)%n",
331                 hal.getProcessor().getSystemCpuLoadBetweenTicks() * 100);
332         System.out.format("CPU load: %.1f%% (OS MXBean)%n", hal.getProcessor().getSystemCpuLoad() * 100);
333         double[] loadAverage = hal.getProcessor().getSystemLoadAverage(3);
334         System.out.println("CPU load averages:" + (loadAverage[0] < 0 ? " N/A" : String.format(" %.2f", loadAverage[0]))
335                 + (loadAverage[1] < 0 ? " N/A" : String.format(" %.2f", loadAverage[1]))
336                 + (loadAverage[2] < 0 ? " N/A" : String.format(" %.2f", loadAverage[2])));
337         // per core CPU
338         StringBuilder procCpu = new StringBuilder("CPU load per processor:");
339         double[] load = hal.getProcessor().getProcessorCpuLoadBetweenTicks();
340         for (int cpu = 0; cpu < load.length; cpu++) {
341             procCpu.append(String.format(" %.1f%%", load[cpu] * 100));
342         }
343         System.out.println(procCpu.toString());
344         System.out.println("Processes: " + hal.getProcessor().getProcessCount() + ", Threads: "
345                 + hal.getProcessor().getThreadCount());
346         // hardware: sensors
347         LOG.info("Checking Sensors...");
348         System.out.println("Sensors:");
349         System.out.format(" CPU Temperature: %.1fÂ°C%n", hal.getSensors().getCpuTemperature());
350         System.out.println(" Fan Speeds:" + Arrays.toString(hal.getSensors().getFanSpeeds()));
351         System.out.format(" CPU Voltage: %.1fV%n", hal.getSensors().getCpuVoltage());
352 
353         // hardware: power
354         LOG.info("Checking Power sources...");
355         StringBuilder sb = new StringBuilder("Power: ");
356         if (hal.getPowerSources().length == 0) {
357             sb.append("Unknown");
358         } else {
359             double timeRemaining = hal.getPowerSources()[0].getTimeRemaining();
360             if (timeRemaining < -1d) {
361                 sb.append("Charging");
362             } else if (timeRemaining < 0d) {
363                 sb.append("Calculating time remaining");
364             } else {
365                 sb.append(String.format("%d:%02d remaining", (int) (timeRemaining / 3600),
366                         (int) (timeRemaining / 60) % 60));
367             }
368         }
369         for (PowerSource pSource : hal.getPowerSources()) {
370             sb.append(String.format("%n %s @ %.1f%%", pSource.getName(), pSource.getRemainingCapacity() * 100d));
371         }
372         System.out.println(sb.toString());
373 
374         // hardware: file system
375         LOG.info("Checking File System...");
376         System.out.println("File System:");
377 
378         OSFileStore[] fsArray = hal.getFileStores();
379         for (OSFileStore fs : fsArray) {
380             long usable = fs.getUsableSpace();
381             long total = fs.getTotalSpace();
382             System.out.format(" %s (%s) [%s] %s of %s free (%.1f%%)%n", fs.getName(),
383                     fs.getDescription().isEmpty() ? "file system" : fs.getDescription(), fs.getType(),
384                     FormatUtil.formatBytes(usable), FormatUtil.formatBytes(fs.getTotalSpace()), 100d * usable / total);
385         }
386 
387         // hardware: disks
388         LOG.info("Checking Disks...");
389         System.out.println("Disks:");
390 
391         HWDiskStore[] dskArray = hal.getDiskStores();
392         for (HWDiskStore dsk : dskArray) {
393             System.out.format(" %s: (model: %s - S/N: %s) size: %s, reads: %s, writes: %s %n", dsk.getName(),
394                     dsk.getModel(), dsk.getSerial(),
395                     dsk.getSize() > 0 ? FormatUtil.formatBytesDecimal(dsk.getSize()) : "?",
396                     dsk.getReads() > 0 ? FormatUtil.formatBytes(dsk.getReads()) : "?",
397                     dsk.getWrites() > 0 ? FormatUtil.formatBytes(dsk.getWrites()) : "?");
398         }
399 
400         // hardware: network interfaces
401         LOG.info("Checking Network interfaces...");
402         System.out.println("Network interfaces:");
403 
404         NetworkIF[] netArray = hal.getNetworkIFs();
405         for (NetworkIF net : netArray) {
406             System.out.format(" Name: %s (%s)%n", net.getName(), net.getDisplayName());
407             System.out.format("   MAC Address: %s %n", net.getMacaddr());
408             System.out.format("   MTU: %s, Speed: %s %n", net.getMTU(), FormatUtil.formatValue(net.getSpeed(), "bps"));
409             System.out.format("   IPv4: %s %n", Arrays.toString(net.getIPv4addr()));
410             System.out.format("   IPv6: %s %n", Arrays.toString(net.getIPv6addr()));
411             boolean hasData = net.getBytesRecv() > 0 || net.getBytesSent() > 0 || net.getPacketsRecv() > 0
412                     || net.getPacketsSent() > 0;
413             System.out.format("   Traffic: received %s/%s; transmitted %s/%s %n",
414                     hasData ? net.getPacketsRecv() + " packets" : "?",
415                     hasData ? FormatUtil.formatBytes(net.getBytesRecv()) : "?",
416                     hasData ? net.getPacketsSent() + " packets" : "?",
417                     hasData ? FormatUtil.formatBytes(net.getBytesSent()) : "?");
418         }
419 
420         // hardware: displays
421         LOG.info("Checking Displays...");
422         System.out.println("Displays:");
423         int i = 0;
424         for (Display display : hal.getDisplays()) {
425             System.out.println(" Display " + i + ":");
426             System.out.println(display.toString());
427             i++;
428         }
429         LOG.info("Printing JSON:");
430         // Compact JSON
431         // System.out.println(si.toJSON().toString());
432 
433         // Pretty JSON
434         System.out.println(ParseUtil.jsonPrettyPrint(si.toJSON()));
435     }
