24  package oshi.hardware;
25  
26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertNotNull;
28  import static org.junit.Assert.assertTrue;
29  
30  import org.junit.Test;
31  
32  import oshi.SystemInfo;
33  import oshi.hardware.CentralProcessor.ProcessorIdentifier;
34  import oshi.hardware.CentralProcessor.TickType;
35  import oshi.util.Util;
36  
37  /**
38   * Test CPU
39   */
40  public class CentralProcessorTest {
41  
42      /**
43       * Test central processor.
44       */
45      @Test
46      public void testCentralProcessor() {
47          SystemInfo si = new SystemInfo();
48          CentralProcessor p = si.getHardware().getProcessor();
49  
50          ProcessorIdentifier pi = p.getProcessorIdentifier();
51          assertNotNull(pi.getVendor());
52          assertTrue(pi.getVendorFreq() == -1 || pi.getVendorFreq() > 0);
53          assertNotNull(pi.getName());
54          assertNotNull(pi.getIdentifier());
55          assertNotNull(pi.getProcessorID());
56          assertNotNull(pi.getStepping());
57          assertNotNull(pi.getModel());
58          assertNotNull(pi.getFamily());
59  
60          long[] ticks = p.getSystemCpuLoadTicks();
61          long[][] procTicks = p.getProcessorCpuLoadTicks();
62          assertEquals(ticks.length, TickType.values().length);
63  
64          Util.sleep(500);
65  
66          assertTrue(p.getSystemCpuLoadBetweenTicks(ticks) >= 0 && p.getSystemCpuLoadBetweenTicks(ticks) <= 1);
67          assertEquals(3, p.getSystemLoadAverage(3).length);
68  
69          assertEquals(p.getProcessorCpuLoadBetweenTicks(procTicks).length, p.getLogicalProcessorCount());
70          for (int cpu = 0; cpu < p.getLogicalProcessorCount(); cpu++) {
71              assertTrue(p.getProcessorCpuLoadBetweenTicks(procTicks)[cpu] >= 0
72                      && p.getProcessorCpuLoadBetweenTicks(procTicks)[cpu] <= 1);
73              assertEquals(p.getProcessorCpuLoadTicks()[cpu].length, TickType.values().length);
74          }
75  
76          assertTrue(p.getLogicalProcessorCount() >= p.getPhysicalProcessorCount());
77          assertTrue(p.getPhysicalProcessorCount() > 0);
78          assertTrue(p.getPhysicalProcessorCount() >= p.getPhysicalPackageCount());
79          assertTrue(p.getPhysicalPackageCount() > 0);
80          assertTrue(p.getContextSwitches() >= 0);
81          assertTrue(p.getInterrupts() >= 0);
82  
83          long max = p.getMaxFreq();
84          long[] curr = p.getCurrentFreq();
85          assertEquals(curr.length, p.getLogicalProcessorCount());
86          if (max >= 0) {
87              for (int i = 0; i < curr.length; i++) {
88                  assertTrue(curr[i] <= max);
89              }
90          }
91      }
92  }
