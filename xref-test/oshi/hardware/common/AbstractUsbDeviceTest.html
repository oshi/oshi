5   package oshi.hardware.common;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.hasSize;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.lessThan;
11  
12  import java.util.Collections;
13  
14  import org.junit.jupiter.api.Test;
15  
16  class AbstractUsbDeviceTest {
17  
18      @Test
19      void testGetters() {
20          AbstractUsbDevice device = new AbstractUsbDevice("Mouse", "Logitech", "046d", "c077", "SN1", "USB1",
21                  Collections.emptyList()) {
22          };
23          assertThat(device.getName(), is("Mouse"));
24          assertThat(device.getVendor(), is("Logitech"));
25          assertThat(device.getVendorId(), is("046d"));
26          assertThat(device.getProductId(), is("c077"));
27          assertThat(device.getSerialNumber(), is("SN1"));
28          assertThat(device.getUniqueDeviceId(), is("USB1"));
29          assertThat(device.getConnectedDevices(), is(Collections.emptyList()));
30      }
31  
32      @Test
33      void testConnectedDevices() {
34          AbstractUsbDevice child = new AbstractUsbDevice("Mouse", "Logitech", "", "", "", "", Collections.emptyList()) {
35          };
36          AbstractUsbDevice parent = new AbstractUsbDevice("Hub", "Generic", "", "", "", "",
37                  Collections.singletonList(child)) {
38          };
39          assertThat(parent.getConnectedDevices(), hasSize(1));
40          assertThat(parent.getConnectedDevices().get(0).getName(), is("Mouse"));
41      }
42  
43      @Test
44      void testCompareTo() {
45          AbstractUsbDevice a = new AbstractUsbDevice("Alpha", "", "", "", "", "", Collections.emptyList()) {
46          };
47          AbstractUsbDevice b = new AbstractUsbDevice("Beta", "", "", "", "", "", Collections.emptyList()) {
48          };
49          assertThat(a.compareTo(b), is(lessThan(0)));
50      }
51  
52      @Test
53      void testToStringWithChildren() {
54          AbstractUsbDevice child = new AbstractUsbDevice("Mouse", "Logitech", "", "", "SN1", "",
55                  Collections.emptyList()) {
56          };
57          AbstractUsbDevice parent = new AbstractUsbDevice("Hub", "", "", "", "", "", Collections.singletonList(child)) {
58          };
59          assertThat(parent.toString(), is(" Hub\n |-- Mouse (Logitech) [s/n: SN1]"));
60      }
61  }
