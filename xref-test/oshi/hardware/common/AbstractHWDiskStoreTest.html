5   package oshi.hardware.common;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.containsString;
9   import static org.hamcrest.Matchers.is;
10  
11  import java.util.Collections;
12  import java.util.List;
13  
14  import org.junit.jupiter.api.Test;
15  
16  import oshi.hardware.HWPartition;
17  
18  class AbstractHWDiskStoreTest {
19  
20      private static TestHWDiskStore createDisk() {
21          return new TestHWDiskStore("sda", "Model X", "SN123", 500_000_000_000L);
22      }
23  
24      /** Test subclass exposing protected setters. */
25      private static class TestHWDiskStore extends AbstractHWDiskStore {
26          TestHWDiskStore(String name, String model, String serial, long size) {
27              super(name, model, serial, size);
28          }
29  
30          @Override
31          public boolean updateAttributes() {
32              return false;
33          }
34  
35          void applyStats(long reads, long readBytes, long writes, long writeBytes, long queueLen, long xferTime,
36                  long ts) {
37              setDiskStats(reads, readBytes, writes, writeBytes, queueLen, xferTime, ts);
38          }
39  
40          void applyCurrentQueueLength(long queueLen) {
41              setCurrentQueueLength(queueLen);
42          }
43  
44          void applyPartitions(List<HWPartition> parts) {
45              setPartitionList(parts);
46          }
47      }
48  
49      @Test
50      void testGettersAndSetDiskStats() {
51          TestHWDiskStore disk = createDisk();
52          assertThat(disk.getName(), is("sda"));
53          assertThat(disk.getModel(), is("Model X"));
54          assertThat(disk.getSerial(), is("SN123"));
55          assertThat(disk.getSize(), is(500_000_000_000L));
56          assertThat(disk.getDiskType(), is("Unknown"));
57  
58          disk.applyStats(100L, 2048L, 50L, 1024L, 3L, 500L, 99999L);
59          assertThat(disk.getReads(), is(100L));
60          assertThat(disk.getReadBytes(), is(2048L));
61          assertThat(disk.getWrites(), is(50L));
62          assertThat(disk.getWriteBytes(), is(1024L));
63          assertThat(disk.getCurrentQueueLength(), is(3L));
64          assertThat(disk.getTransferTime(), is(500L));
65          assertThat(disk.getTimeStamp(), is(99999L));
66      }
67  
68      @Test
69      void testSetCurrentQueueLength() {
70          TestHWDiskStore disk = createDisk();
71          disk.applyCurrentQueueLength(7L);
72          assertThat(disk.getCurrentQueueLength(), is(7L));
73      }
74  
75      @Test
76      void testPartitions() {
77          TestHWDiskStore disk = createDisk();
78          assertThat(disk.getPartitions(), is(Collections.emptyList()));
79          List<HWPartition> parts = Collections
80                  .singletonList(new HWPartition("sda1", "sda1", "ext4", "uuid1", 100_000L, 0, 8, "/"));
81          disk.applyPartitions(parts);
82          assertThat(disk.getPartitions().size(), is(1));
83      }
84  
85      @Test
86      void testToString() {
87          TestHWDiskStore disk = createDisk();
88          disk.applyStats(10L, 512L, 5L, 256L, 0L, 100L, 1000L);
89          String s = disk.toString();
90          assertThat(s, containsString("sda"));
91          assertThat(s, containsString("Model X"));
92      }
93  }
