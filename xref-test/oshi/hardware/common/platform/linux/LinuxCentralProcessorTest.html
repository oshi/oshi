73      @Test
74      void testCreateMIDRBasic() {
75          // vendor=0x41 (ARM), stepping=r1p3, model=0xd07, family=8
76          String midr = LinuxCentralProcessor.createMIDR("0x41", "r1p3", "0xd07", "8");
77          // Implementer 0x41 << 24 = 0x41000000
78          // Variant 1 << 20 = 0x00100000
79          // Architecture 8 << 16 = 0x00080000
80          // PartNum 0xd07 << 4 = 0x0000D070
81          // Revision 3 = 0x00000003
82          // Total = 0x41180073... let me compute:
83          // 0x41 << 24 = 0x41000000
84          // 1 << 20 = 0x00100000
85          // 8 << 16 = 0x00080000
86          // 0xd07 << 4 = 0x0000D070
87          // 3 = 0x00000003
88          // Sum = 0x4118D073
89          assertThat(midr, is("4118D073"));
90      }
91  
92      @Test
93      void testCreateMIDRNoStepping() {
94          // No rnpn stepping format
95          String midr = LinuxCentralProcessor.createMIDR("0x41", "", "0xd03", "8");
96          // 0x41 << 24 + 8 << 16 + 0xd03 << 4 = 0x4108D030
97          assertThat(midr, is("4108D030"));
98      }

418             "         \"ways\": 8,", "         \"type\": \"Instruction\",", "         \"level\": 1,",
419             "         \"sets\": null,", "         \"coherency-size\": 64", "      },", "      {",
420             "         \"name\": \"L2\",", "         \"one-size\": \"262144\",", "         \"all-size\": null,",
421             "         \"ways\": 4,", "         \"type\": \"Unified\",", "         \"level\": 2,",
422             "         \"sets\": null,", "         \"coherency-size\": 64", "      },", "      {",
423             "         \"name\": \"L3\",", "         \"one-size\": \"16777216\",", "         \"all-size\": null,",
424             "         \"ways\": null,", "         \"type\": \"Unified\",", "         \"level\": 3,",
425             "         \"sets\": null,", "         \"coherency-size\": 64", "      }", "   ]", "}");

427     @Test
428     void testMapCachesFromLscpu() {
429         Set<ProcessorCache> result = LinuxCentralProcessor.mapCachesFromLscpu(LSCPU_CACHE_JSON);
430         assertThat(result, hasSize(4));
431 
432         ProcessorCache l1d = new ProcessorCache(1, 8, 64, 32768L, ProcessorCache.Type.DATA);
433         assertThat(result.contains(l1d), is(true));
434 
435         ProcessorCache l1i = new ProcessorCache(1, 8, 64, 32768L, ProcessorCache.Type.INSTRUCTION);
436         assertThat(result.contains(l1i), is(true));

438         ProcessorCache l2 = new ProcessorCache(2, 4, 64, 262144L, ProcessorCache.Type.UNIFIED);
439         assertThat(result.contains(l2), is(true));
440 
441         // L3 ways is null, so associativity should be 0
442         ProcessorCache l3 = new ProcessorCache(3, 0, 64, 16777216L, ProcessorCache.Type.UNIFIED);
