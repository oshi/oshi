76      @Test
77      void testCreateMIDRBasic() {
78          // vendor=0x41 (ARM), stepping=r1p3, model=0xd07, family=8
79          String midr = LinuxCentralProcessor.createMIDR("0x41", "r1p3", "0xd07", "8");
80          // Implementer 0x41 << 24 = 0x41000000
81          // Variant 1 << 20 = 0x00100000
82          // Architecture 8 << 16 = 0x00080000
83          // PartNum 0xd07 << 4 = 0x0000D070
84          // Revision 3 = 0x00000003
85          // Total = 0x41180073... let me compute:
86          // 0x41 << 24 = 0x41000000
87          // 1 << 20 = 0x00100000
88          // 8 << 16 = 0x00080000
89          // 0xd07 << 4 = 0x0000D070
90          // 3 = 0x00000003
91          // Sum = 0x4118D073
92          assertThat(midr, is("4118D073"));
93      }
