38      @Test
39      void testCreateMIDRBasic() {
40          // vendor=0x41 (ARM), stepping=r1p3, model=0xd07, family=8
41          String midr = LinuxCentralProcessor.createMIDR("0x41", "r1p3", "0xd07", "8");
42          // Implementer 0x41 << 24 = 0x41000000
43          // Variant 1 << 20 = 0x00100000
44          // Architecture 8 << 16 = 0x00080000
45          // PartNum 0xd07 << 4 = 0x0000D070
46          // Revision 3 = 0x00000003
47          // Total = 0x41180073... let me compute:
48          // 0x41 << 24 = 0x41000000
49          // 1 << 20 = 0x00100000
50          // 8 << 16 = 0x00080000
51          // 0xd07 << 4 = 0x0000D070
52          // 3 = 0x00000003
53          // Sum = 0x4118D073
54          assertThat(midr, is("4118D073"));
55      }
56  
57      @Test
58      void testCreateMIDRNoStepping() {
59          // No rnpn stepping format
60          String midr = LinuxCentralProcessor.createMIDR("0x41", "", "0xd03", "8");
61          // 0x41 << 24 + 8 << 16 + 0xd03 << 4 = 0x4108D030
62          assertThat(midr, is("4108D030"));
63      }

110     @Test
111     void testQueryMaxFreqFromCpuFreqPathWithPolicies(@TempDir Path tempDir) throws IOException {
112         Path cpuFreq = tempDir.resolve("cpufreq");
113         Path policy0 = cpuFreq.resolve("policy0");
114         Path policy1 = cpuFreq.resolve("policy1");
115         Files.createDirectories(policy0);
116         Files.createDirectories(policy1);
117         writeFile(policy0.resolve("scaling_max_freq"), "3600000");
118         writeFile(policy1.resolve("scaling_max_freq"), "4200000");
119 
120         long maxFreq = LinuxCentralProcessor.queryMaxFreqFromCpuFreqPath(cpuFreq.toString());
121         // 4200000 kHz * 1000 = 4200000000 Hz
122         assertThat(maxFreq, is(4200000000L));
123     }

168     @Test
169     void testReadTopologyFromSysfsSingleCpu(@TempDir Path tempDir) throws IOException {
170         Path cpu0 = tempDir.resolve("cpu0");
171         Files.createDirectories(cpu0.resolve("topology"));
172         writeFile(cpu0.resolve("topology/core_id"), "0");
173         writeFile(cpu0.resolve("topology/physical_package_id"), "0");
174         writeFile(cpu0.resolve("cpu_capacity"), "1024");
175 
176         Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> result = LinuxCentralProcessor
177                 .readTopologyFromSysfs(tempDir.toString());
178         assertThat(result.getA(), hasSize(1));
179         LogicalProcessor lp = result.getA().get(0);
180         assertThat(lp.getPhysicalProcessorNumber(), is(0));
181         assertThat(lp.getPhysicalPackageNumber(), is(0));
182         // cpu_capacity value stored in coreEfficiencyMap keyed by (pkg<<16)+core = 0
183         assertThat(result.getC().containsKey(0), is(true));
184         assertThat(result.getC().get(0), is(1024));
185     }

201     @Test
202     void testReadTopologyFromSysfsWithCache(@TempDir Path tempDir) throws IOException {
203         Path cpu0 = tempDir.resolve("cpu0");
204         Files.createDirectories(cpu0.resolve("topology"));
205         writeFile(cpu0.resolve("topology/core_id"), "0");
206         writeFile(cpu0.resolve("topology/physical_package_id"), "0");
207 
208         Path index0 = cpu0.resolve("cache/index0");
209         Files.createDirectories(index0);
210         writeFile(index0.resolve("level"), "1");
211         writeFile(index0.resolve("type"), "Data");
212         writeFile(index0.resolve("ways_of_associativity"), "8");
213         writeFile(index0.resolve("coherency_line_size"), "64");
214         writeFile(index0.resolve("size"), "32K");
215 
216         Quartet<List<LogicalProcessor>, List<ProcessorCache>, Map<Integer, Integer>, Map<Integer, String>> result = LinuxCentralProcessor
217                 .readTopologyFromSysfs(tempDir.toString());
218         assertThat(result.getB(), hasSize(1));
219         ProcessorCache cache = result.getB().get(0);
220         assertThat(cache.getLevel(), is((byte) 1));
221         assertThat(cache.getType(), is(ProcessorCache.Type.DATA));
222         assertThat(cache.getAssociativity(), is((byte) 8));
223         assertThat(cache.getLineSize(), is((short) 64));
224         assertThat(cache.getCacheSize(), is(32768));
225     }
