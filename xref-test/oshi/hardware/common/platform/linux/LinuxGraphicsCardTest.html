166     /**
167      * Creates a card directory with a driver symlink and uevent file.
168      *
169      * @param drmDir     the base DRM directory
170      * @param cardName   e.g. "card0"
171      * @param driverName e.g. "amdgpu"
172      * @param slotName   e.g. "0000:01:00.0"
173      * @throws IOException if file creation fails
174      */
175     private static void createCardWithDriver(Path drmDir, String cardName, String driverName, String slotName)
176             throws IOException {
177         Path deviceDir = drmDir.resolve(cardName + "/device");
178         Files.createDirectories(deviceDir);
179         // Create a fake driver target and symlink
180         Path driverTarget = drmDir.resolve("drivers/" + driverName);
181         Files.createDirectories(driverTarget);
182         Files.createSymbolicLink(deviceDir.resolve("driver"), driverTarget);
183         // Write uevent with PCI_SLOT_NAME
184         Files.write(deviceDir.resolve("uevent"), ("PCI_SLOT_NAME=" + slotName + "\n").getBytes(StandardCharsets.UTF_8));
185     }

191     // Fixture: lspci -vnnm output with one VGA card
192     private static final List<String> LSPCI_VNNM = Arrays.asList("Slot:\t01:00.0",
193             "Class:\tVGA compatible controller [0300]", "Vendor:\tNVIDIA Corporation [10de]",
194             "Device:\tGA102 [GeForce RTX 3090] [2204]", "SVendor:\tASUS [1043]",
195             "SDevice:\tGA102 [GeForce RTX 3090] [8687]", "Rev:\ta1", "");
196 
197     private static final Function<LinuxGraphicsCard.Attrs, GraphicsCard> STUB_FACTORY = attrs -> new StubGraphicsCard(
198             attrs.getName(), attrs.getDeviceId(), attrs.getVendor(), attrs.getVersionInfo(), attrs.getVram(),
199             attrs.getDrmDevicePath(), attrs.getDriverName(), attrs.getPciBusId());
200 
201     // No-op lookups for pure parsing tests
202     private static final ToLongFunction<String> NO_VRAM = slot -> 0L;
203     private static final Function<String, Triplet<String, String, String>> NO_DRM = slot -> new Triplet<>("", "", "");
204 
205     @Test
206     void testGetGraphicsCardsFromLspciSingleCard() {
207         List<GraphicsCard> cards = LinuxGraphicsCard.getGraphicsCardsFromLspci(LSPCI_VNNM, STUB_FACTORY, NO_VRAM,
208                 NO_DRM);
209         assertThat(cards.size(), is(1));
210         GraphicsCard card = cards.get(0);
211         assertThat(card.getName(), is("GA102 [GeForce RTX 3090]"));
212         assertThat(card.getDeviceId(), is("0x2204"));
213         assertThat(card.getVendor(), is("NVIDIA Corporation (0x10de)"));
214         assertThat(card.getVersionInfo(), is("Rev:\ta1"));
215     }

250     @Test
251     void testQueryLspciMemorySize() {
252         List<String> lspciV = Arrays.asList("01:00.0 VGA compatible controller: NVIDIA Corporation",
253                 "\tMemory at f6000000 (32-bit, non-prefetchable) [size=16M]",
254                 "\tMemory at e0000000 (64-bit, prefetchable) [size=256M]",
255                 "\tMemory at f0000000 (64-bit, prefetchable) [size=32M]", "\tI/O ports at e000 [size=128]");
256         long vram = LinuxGraphicsCard.queryLspciMemorySize(lspciV);
257         // 256M + 32M = 288M
258         assertThat(vram, is(256L * 1024 * 1024 + 32L * 1024 * 1024));
259     }

299     // Fixture: `lshw -C display` output (real structure) with two display nodes. The second card has no bus info so
300     // its DRM lookup is with a null slot; resources memory ranges drive the VRAM total.
301     private static final List<String> LSHW = Arrays.asList("  *-display",
302             "       description: VGA compatible controller", "       product: GA102 [GeForce RTX 3090]",
303             "       vendor: NVIDIA Corporation", "       physical id: 0", "       bus info: pci@0000:01:00.0",
304             "       version: a1", "       width: 64 bits", "       clock: 33MHz",
305             "       capabilities: pm msi pciexpress vga_controller bus_master cap_list rom",
306             "       configuration: driver=nvidia latency=0",
307             "       resources: irq:178 memory:f6000000-f6ffffff memory:e0000000-efffffff", "  *-display",
308             "       description: VGA compatible controller", "       product: UHD Graphics 630",
309             "       vendor: Intel Corporation", "       physical id: 2",
310             "       resources: irq:24 memory:db000000-dbffffff");
311 
312     @Test
313     void testGetGraphicsCardsFromLshwTwoCards() {
314         List<GraphicsCard> cards = LinuxGraphicsCard.getGraphicsCardsFromLshw(LSHW, STUB_FACTORY, NO_DRM);
315         assertThat(cards.size(), is(2));
316 
317         GraphicsCard nvidia = cards.get(0);
318         assertThat(nvidia.getName(), is("GA102 [GeForce RTX 3090]"));
319         assertThat(nvidia.getVendor(), is("NVIDIA Corporation"));
320         assertThat(nvidia.getVersionInfo(), is("version: a1"));
321         // 16 MiB (f6000000-f6ffffff) + 256 MiB (e0000000-efffffff)
322         assertThat(nvidia.getVRam(), is(16L * 1024 * 1024 + 256L * 1024 * 1024));
323 
324         GraphicsCard intel = cards.get(1);
325         assertThat(intel.getName(), is("UHD Graphics 630"));
326         assertThat(intel.getVendor(), is("Intel Corporation"));
327         // No version line -> UNKNOWN version info; single 16 MiB memory range
328         assertThat(intel.getVersionInfo(), is(Constants.UNKNOWN));
329         assertThat(intel.getVRam(), is(16L * 1024 * 1024));
330     }
