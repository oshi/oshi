46      @Test
47      void testGetCardFoldersFiltersCorrectly(@TempDir Path tempDir) throws IOException {
48          Path asound = tempDir.resolve("asound");
49          Files.createDirectories(asound.resolve("card0"));
50          Files.createDirectories(asound.resolve("card1"));
51          // Not a card directory — should be excluded
52          Files.createDirectories(asound.resolve("timers"));
53          // File starting with "card" but not a directory
54          writeFile(asound.resolve("cards"), "content");
55  
56          List<File> folders = LinuxSoundCard.getCardFolders(asound.toString());
57          assertThat(folders, hasSize(2));
58          List<String> names = new ArrayList<>();
59          for (File f : folders) {
60              names.add(f.getName());
61          }
62          assertThat(names, containsInAnyOrder("card0", "card1"));
63      }

138         Files.createDirectories(card0);

163     @Test
164     void testGetSoundCardsIntegration(@TempDir Path tempDir) throws IOException {
165         Path asound = tempDir.resolve("asound/");
166         Files.createDirectories(asound);
167         writeFile(asound.resolve("version"), "ALSA v1.2.6");
168 
169         Path card0 = asound.resolve("card0");
170         Files.createDirectories(card0);
171         writeFile(card0.resolve("id"), "PCH");
172         writeFile(card0.resolve("codec#0"), "Codec: ALC892\nAddress: 0");
173         writeFile(asound.resolve("cards"), " 0 [PCH            ]: HDA-Intel - HDA Intel PCH");
174 
175         List<SoundCard> cards = LinuxSoundCard.getSoundCards(asound.toString());
176         assertThat(cards, hasSize(1));
177         SoundCard sc = cards.get(0);
178         assertThat(sc.getDriverVersion(), is("ALSA v1.2.6"));
179         assertThat(sc.getName(), is("HDA-Intel - HDA Intel PCH"));
180         assertThat(sc.getCodec(), is("ALC892"));
181     }
