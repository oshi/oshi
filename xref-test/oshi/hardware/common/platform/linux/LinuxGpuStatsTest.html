325     /**
326      * Creates a temp directory layout matching real sysfs: {@code card0/device} and {@code card0/gt/gt0}. The
327      * constructor computes gt0Path as {@code drmDevicePath + "/../gt/gt0"}, so the "device" dir must be a real
328      * subdirectory so that ".." resolves to the card directory.
329      *
330      * @param tmp the JUnit {@code @TempDir} root
331      * @return the {@code device} subdirectory path to pass as {@code drmDevicePath}
332      * @throws IOException if directory creation fails
333      */
334     private static Path createCardWithGt(Path tmp) throws IOException {
335         Path card = tmp.resolve("card0");
336         Path device = card.resolve("device");
337         Path gt0 = card.resolve("gt/gt0");
338         Files.createDirectories(device);
339         Files.createDirectories(gt0);
340         return device;
341     }
342

344     void testI915Utilization(@TempDir Path tmp) throws IOException {
345         Path device = createCardWithGt(tmp);

415     @Test
416     void testNvmlUnavailableCachesResult() {
417         StubLinuxGpuStats stats = new StubLinuxGpuStats("", "", "0000:01:00.0", "GPU");
418         // First call resolves and caches
419         assertThat(stats.findNvmlDevice() == null, is(true));
420         assertThat(stats.getFindDeviceCallCount(), is(0)); // nvmlIsAvailable() is false, never calls findDevice
421         // Second call uses cache — still no findDevice calls
422         assertThat(stats.findNvmlDevice() == null, is(true));
423         assertThat(stats.getFindDeviceCallCount(), is(0));
424         stats.close();
425     }
