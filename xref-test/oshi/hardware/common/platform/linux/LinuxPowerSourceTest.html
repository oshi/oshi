34          Map<LinuxPowerSource.Prop, String> props = new EnumMap<>(LinuxPowerSource.Prop.class);
35          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_NAME, "BAT0");
36          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_STATUS, "Discharging");
37          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_PRESENT, "1");
38          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_TECHNOLOGY, "Li-ion");
39          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CYCLE_COUNT, "481");
40          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_VOLTAGE_NOW, "7400000");
41          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_POWER_NOW, "9361000");
42          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_ENERGY_FULL_DESIGN, "48248000");
43          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_ENERGY_FULL, "40877000");
44          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_ENERGY_NOW, "20712000");
45          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CAPACITY, "50");
46          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_MODEL_NAME, "UX32-65");
47          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_MANUFACTURER, "ASUSTeK");
48          return props;

51      @Test
52      void testSampleUeventDischarging() {
53          LinuxPowerSource ps = LinuxPowerSource.buildPowerSource("BAT0", sampleProps());
54  
55          assertThat(ps.getName(), is("BAT0"));
56          assertThat(ps.getDeviceName(), is("UX32-65"));
57          assertThat(ps.getChemistry(), is("Li-ion"));
58          assertThat(ps.getManufacturer(), is("ASUSTeK"));
59          assertThat(ps.getCycleCount(), is(481));
60  
61          assertThat(ps.getRemainingCapacityPercent(), closeTo(0.50, EPS));
62          assertThat(ps.isCharging(), is(false));
63          assertThat(ps.isDischarging(), is(true));
64  
65          // Voltage: 7400000 µV → 7.4 V
66          assertThat(ps.getVoltage(), closeTo(7.4, EPS));
67  
68          // ENERGY_* present → MWh units
69          assertThat(ps.getCapacityUnits(), is(CapacityUnits.MWH));
70  
71          // ENERGY_NOW: 20712000 µWh → 20712 mWh
72          assertThat(ps.getCurrentCapacity(), is(20712));
73          // ENERGY_FULL: 40877000 µWh → 40877 mWh
74          assertThat(ps.getMaxCapacity(), is(40877));
75          // ENERGY_FULL_DESIGN: 48248000 µWh → 48248 mWh
76          assertThat(ps.getDesignCapacity(), is(48248));
77  
78          // POWER_NOW: 9361000 µW → 9361 mW, negated because discharging
79          assertThat(ps.getPowerUsageRate(), closeTo(-9361.0, EPS));
80          assertThat(ps.getPowerUsageRate(), lessThan(0d));
81  
82          // No CURRENT_NOW → amperage derived: P/V = -9361 mW / 7.4 V ≈ -1265 mA
83          assertThat(ps.getAmperage(), closeTo(-9361.0 / 7.4, 1e-3));
84      }

86      @Test
87      void testChargingSignConvention() {
88          Map<LinuxPowerSource.Prop, String> props = sampleProps();
89          props.put(LinuxPowerSource.Prop.POWER_SUPPLY_STATUS, "Charging");
90  
91          LinuxPowerSource ps = LinuxPowerSource.buildPowerSource("BAT0", props);
92  
93          assertThat(ps.isCharging(), is(true));
94          assertThat(ps.isDischarging(), is(false));
95          // Power and amperage should be positive when charging
96          assertThat(ps.getPowerUsageRate(), closeTo(9361.0, EPS));
97          assertThat(ps.getAmperage(), closeTo(9361.0 / 7.4, 1e-3));
98      }

100     @Test
101     void testChargeNowFallback() {
102         // No ENERGY_* keys → should fall back to CHARGE_* and use MAH units
103         Map<LinuxPowerSource.Prop, String> props = new EnumMap<>(LinuxPowerSource.Prop.class);
104         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_STATUS, "Discharging");
105         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CHARGE_NOW, "3000000");
106         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CHARGE_FULL, "5000000");
107         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CHARGE_FULL_DESIGN, "6000000");
108         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_VOLTAGE_NOW, "7400000");
109         props.put(LinuxPowerSource.Prop.POWER_SUPPLY_CURRENT_NOW, "1500000");
110 
111         LinuxPowerSource ps = LinuxPowerSource.buildPowerSource("BAT0", props);
112 
113         assertThat(ps.getCapacityUnits(), is(CapacityUnits.MAH));
114         assertThat(ps.getCurrentCapacity(), is(3000));
115         assertThat(ps.getMaxCapacity(), is(5000));
116         assertThat(ps.getDesignCapacity(), is(6000));
117 
118         // CURRENT_NOW: 1500000 µA → 1500 mA, negated because discharging
119         assertThat(ps.getAmperage(), closeTo(-1500.0, EPS));
120         // No POWER_NOW → derived: I * V = -1500 mA * 7.4 V = -11100 mW
121         assertThat(ps.getPowerUsageRate(), closeTo(-1500.0 * 7.4, 1e-3));
122     }

263         Path bat = tempDir.resolve("BAT0");
264         Files.createDirectories(bat);
265         String uevent = "POWER_SUPPLY_NAME=BAT0\n" + "POWER_SUPPLY_STATUS=Discharging\n" + "POWER_SUPPLY_PRESENT=1\n"
266                 + "POWER_SUPPLY_TECHNOLOGY=Li-ion\n" + "POWER_SUPPLY_VOLTAGE_NOW=7400000\n"
267                 + "POWER_SUPPLY_ENERGY_NOW=20712000\n" + "POWER_SUPPLY_ENERGY_FULL=40877000\n"
268                 + "POWER_SUPPLY_ENERGY_FULL_DESIGN=48248000\n" + "POWER_SUPPLY_POWER_NOW=9361000\n"
269                 + "POWER_SUPPLY_CAPACITY=50\n" + "IGNORED_KEY=should_be_skipped\n" + "MALFORMED_LINE\n";
270         Files.write(bat.resolve("uevent"), uevent.getBytes(StandardCharsets.UTF_8));
271 
272         List<PowerSource> result = LinuxPowerSource.getPowerSources(tempDir.toString());
273

275         PowerSource ps = result.get(0);
276         assertThat(ps.getName(), is("BAT0"));
277         assertThat(ps.isDischarging(), is(true));
278         assertThat(ps.getChemistry(), is("Li-ion"));
279         assertThat(ps.getCurrentCapacity(), is(20712));
280         assertThat(ps.getMaxCapacity(), is(40877));

294     @Test
295     void testGetPowerSourcesCopyRoundTrip(@TempDir Path tempDir) throws IOException {
296         Path bat = tempDir.resolve("BAT0");
297         Files.createDirectories(bat);
298         String uevent = "POWER_SUPPLY_NAME=BAT0\nPOWER_SUPPLY_STATUS=Charging\n"
299                 + "POWER_SUPPLY_ENERGY_NOW=20000000\nPOWER_SUPPLY_ENERGY_FULL=40000000\n";
300         Files.write(bat.resolve("uevent"), uevent.getBytes(StandardCharsets.UTF_8));
301 
302         List<PowerSource> sources = LinuxPowerSource.getPowerSources(tempDir.toString());
303         assertThat(sources, hasSize(1));
304         LinuxPowerSource ps = (LinuxPowerSource) sources.get(0);
305 
306         // queryPowerSources() delegates to getPowerSources(SysPath.POWER_SUPPLY) on real
307         // systems; here we verify the copy constructor round-trip used by that path
308         LinuxPowerSource copy = new LinuxPowerSource(ps);
309         assertThat(copy.getName(), is(ps.getName()));
310         assertThat(copy.isCharging(), is(true));
311     }
