28      @Test
29      void testLeafDevice() {
30          Map<String, String> nameMap = new HashMap<>();
31          nameMap.put("usb1/1-1", "USB Mouse");
32          Map<String, String> vendorMap = new HashMap<>();
33          vendorMap.put("usb1/1-1", "Logitech");
34          Map<String, String> vendorIdMap = new HashMap<>();
35          vendorIdMap.put("usb1/1-1", "046d");
36          Map<String, String> productIdMap = new HashMap<>();
37          productIdMap.put("usb1/1-1", "c077");
38          Map<String, String> serialMap = new HashMap<>();
39          serialMap.put("usb1/1-1", "SN123");
40          Map<String, List<String>> hubMap = Collections.emptyMap();
41  
42          LinuxUsbDevice device = LinuxUsbDevice.getDeviceAndChildren("usb1/1-1", "0000", "0000", nameMap, vendorMap,
43                  vendorIdMap, productIdMap, serialMap, hubMap);
44  
45          assertThat(device.getName(), is("USB Mouse"));
46          assertThat(device.getVendor(), is("Logitech"));
47          assertThat(device.getVendorId(), is("046d"));
48          assertThat(device.getProductId(), is("c077"));
49          assertThat(device.getSerialNumber(), is("SN123"));
50          assertThat(device.getUniqueDeviceId(), is("usb1/1-1"));
51          assertThat(device.getConnectedDevices(), is(empty()));
52      }

55      // getDeviceAndChildren — fallback to parent vid/pid

58      @Test
59      void testFallbackToParentVidPid() {
60          // Device not in vendorIdMap/productIdMap — should inherit parent vid/pid
61          LinuxUsbDevice device = LinuxUsbDevice.getDeviceAndChildren("usb1/1-2", "abcd", "1234", Collections.emptyMap(),
62                  Collections.emptyMap(), Collections.emptyMap(), Collections.emptyMap(), Collections.emptyMap(),
63                  Collections.emptyMap());
64  
65          assertThat(device.getVendorId(), is("abcd"));
66          assertThat(device.getProductId(), is("1234"));
67          // Name falls back to "vid:pid"
68          assertThat(device.getName(), is("abcd:1234"));
69          assertThat(device.getVendor(), is(""));
70          assertThat(device.getSerialNumber(), is(""));
71      }

73      // -------------------------------------------------------------------------
74      // getDeviceAndChildren — hub with children, sorted
75      // -------------------------------------------------------------------------
76  
77      @Test
78      void testHubWithSortedChildren() {
79          Map<String, String> nameMap = new HashMap<>();
80          nameMap.put("usb1", "Root Hub");
81          nameMap.put("usb1/1-1", "Zebra Device");
82          nameMap.put("usb1/1-2", "Alpha Device");
83          Map<String, String> vendorMap = Collections.emptyMap();
84          Map<String, String> vendorIdMap = new HashMap<>();
85          vendorIdMap.put("usb1", "1d6b");
86          vendorIdMap.put("usb1/1-1", "aaaa");
87          vendorIdMap.put("usb1/1-2", "bbbb");
88          Map<String, String> productIdMap = new HashMap<>();
89          productIdMap.put("usb1", "0002");
90          productIdMap.put("usb1/1-1", "0001");
91          productIdMap.put("usb1/1-2", "0002");
92          Map<String, String> serialMap = Collections.emptyMap();
93          Map<String, List<String>> hubMap = new HashMap<>();
94          hubMap.put("usb1", Arrays.asList("usb1/1-1", "usb1/1-2"));
95  
96          LinuxUsbDevice hub = LinuxUsbDevice.getDeviceAndChildren("usb1", "0000", "0000", nameMap, vendorMap,
97                  vendorIdMap, productIdMap, serialMap, hubMap);
98  
99          assertThat(hub.getName(), is("Root Hub"));
100         List<UsbDevice> children = hub.getConnectedDevices();
101         assertThat(children, hasSize(2));
102         // Children sorted by name: "Alpha Device" before "Zebra Device"
103         assertThat(children.get(0).getName(), is("Alpha Device"));
104         assertThat(children.get(1).getName(), is("Zebra Device"));
105     }

111     @Test
112     void testNestedHierarchy() {
113         Map<String, String> nameMap = new HashMap<>();
114         nameMap.put("usb1", "Root Hub");
115         nameMap.put("usb1/1-1", "Hub");
116         nameMap.put("usb1/1-1/1-1.1", "Keyboard");
117         Map<String, String> vendorIdMap = new HashMap<>();
118         vendorIdMap.put("usb1", "1d6b");
119         vendorIdMap.put("usb1/1-1", "0424");
120         vendorIdMap.put("usb1/1-1/1-1.1", "04f2");
121         Map<String, String> productIdMap = new HashMap<>();
122         productIdMap.put("usb1", "0002");
123         productIdMap.put("usb1/1-1", "2514");
124         productIdMap.put("usb1/1-1/1-1.1", "0112");
125         Map<String, List<String>> hubMap = new HashMap<>();
126         hubMap.put("usb1", Collections.singletonList("usb1/1-1"));
127         hubMap.put("usb1/1-1", Collections.singletonList("usb1/1-1/1-1.1"));
128 
129         LinuxUsbDevice root = LinuxUsbDevice.getDeviceAndChildren("usb1", "0000", "0000", nameMap,
130                 Collections.emptyMap(), vendorIdMap, productIdMap, Collections.emptyMap(), hubMap);
131 
132         assertThat(root.getConnectedDevices(), hasSize(1));
133         UsbDevice hub = root.getConnectedDevices().get(0);
134         assertThat(hub.getName(), is("Hub"));
135         assertThat(hub.getConnectedDevices(), hasSize(1));
136         assertThat(hub.getConnectedDevices().get(0).getName(), is("Keyboard"));
137     }
