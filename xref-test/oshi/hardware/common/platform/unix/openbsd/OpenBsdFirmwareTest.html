5   package oshi.hardware.common.platform.unix.openbsd;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.emptyString;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.nullValue;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.util.tuples.Triplet;
19  
20  class OpenBsdFirmwareTest {
21  
22      @Test
23      void testParseDmesg() {
24          // Representative OpenBSD dmesg bios0 output with multi-word vendor
25          List<String> dmesg = Arrays.asList(//
26                  "bios0 at mainbus0: SMBIOS rev. 2.8 @ 0xe9cb0 (53 entries)", //
27                  "bios0: vendor American Megatrends Inc. version \"F5\" date 03/18/2016");
28          Triplet<String, String, String> fw = OpenBsdFirmware.parseDmesg(dmesg);
29          assertThat(fw.getA(), is("American Megatrends Inc."));
30          assertThat(fw.getB(), is("F5"));
31          assertThat(fw.getC(), is("2016-03-18"));
32      }
33  
34      @Test
35      void testParseDmesgEmpty() {
36          Triplet<String, String, String> fw = OpenBsdFirmware.parseDmesg(Collections.emptyList());
37          assertThat(fw.getA(), is(nullValue()));
38          assertThat(fw.getB(), is(nullValue()));
39          assertThat(fw.getC(), is(emptyString()));
40      }
41  }
