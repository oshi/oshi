5   package oshi.hardware.common.platform.unix.aix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Arrays;
11  import java.util.Collections;
12  import java.util.List;
13  
14  import org.junit.jupiter.api.Test;
15  
16  import oshi.util.Constants;
17  
18  class AixComputerSystemTest {
19  
20      // lsattr -El sys0: attr, value, description, user-settable columns (whitespace-separated)
21      private static final List<String> LSATTR = Arrays.asList(//
22              "fwversion       IBM,RG080425_d79e22_r                Firmware version and revision levels   False", //
23              "modelname       IBM,9114-275                         Machine name                           False", //
24              "os_uuid         789f930f-b15c-4639-b842-b42603862704 N/A                                     True", //
25              "systemid        IBM,0110ACFDE                        Hardware system identifier             False");
26  
27      // lsmcode -c
28      private static final List<String> LSMCODE = Arrays.asList(//
29              "Platform Firmware level is 3F080425", //
30              "System Firmware level is RG080425_d79e22_regatta");
31  
32      @Test
33      void testParseLsattr() {
34          AixComputerSystem.LsattrStrings s = AixComputerSystem.parseLsattr(LSATTR, LSMCODE);
35          // The "IBM," prefix is split off manufacturer/vendor; the remaining first token is model/version
36          assertThat(s.manufacturer(), is("IBM"));
37          assertThat(s.model(), is("9114-275"));
38          assertThat(s.serialNumber(), is("IBM,0110ACFDE"));
39          assertThat(s.uuid(), is("789f930f-b15c-4639-b842-b42603862704"));
40          assertThat(s.biosVendor(), is("IBM"));
41          assertThat(s.biosVersion(), is("RG080425_d79e22_r"));
42          assertThat(s.biosPlatformVersion(), is("3F080425"));
43      }
44  
45      @Test
46      void testParseLsattrEmpty() {
47          // No lsattr/lsmcode output: vendor/manufacturer keep their IBM defaults, everything else is UNKNOWN
48          AixComputerSystem.LsattrStrings s = AixComputerSystem.parseLsattr(Collections.emptyList(),
49                  Collections.emptyList());
50          assertThat(s.manufacturer(), is("IBM"));
51          assertThat(s.biosVendor(), is("IBM"));
52          assertThat(s.model(), is(Constants.UNKNOWN));
53          assertThat(s.serialNumber(), is(Constants.UNKNOWN));
54          assertThat(s.uuid(), is(Constants.UNKNOWN));
55          assertThat(s.biosVersion(), is(Constants.UNKNOWN));
56          assertThat(s.biosPlatformVersion(), is(Constants.UNKNOWN));
57      }
58  }
