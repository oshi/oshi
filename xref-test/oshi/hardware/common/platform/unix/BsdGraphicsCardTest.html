5   package oshi.hardware.common.platform.unix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.GraphicsCard;
19  
20  class BsdGraphicsCardTest {
21  
22      // Representative pcidump -v output from OpenBSD with one display and one network device
23      private static final List<String> PCIDUMP = Arrays.asList(//
24              " 0:0:0: Intel 82G33/G31/P35/P31 DRAM Controller", //
25              "\tVendor ID: 8086", //
26              "\tProduct ID: 29c0", //
27              "\tClass: 06 Bridge, Host", //
28              " 0:2:0: Intel 82G33/G31 Integrated Graphics Controller", //
29              "\tVendor ID: 8086", //
30              "\tProduct ID: 29c2", //
31              "\tClass: 03 Display, VGA", //
32              "\tRevision: 02", //
33              " 0:25:0: Intel 82801I (ICH9) LAN Controller", //
34              "\tVendor ID: 8086", //
35              "\tProduct ID: 10f0", //
36              "\tClass: 02 Network, Ethernet");
37  
38      @Test
39      void testParsePciDump() {
40          List<GraphicsCard> cards = BsdGraphicsCard.parsePciDump(PCIDUMP);
41          assertThat(cards, hasSize(1));
42          GraphicsCard card = cards.get(0);
43          assertThat(card.getName(), is("Intel 82G33/G31 Integrated Graphics Controller"));
44          assertThat(card.getDeviceId(), is("0x29c2"));
45          assertThat(card.getVendor(), is("0x8086"));
46          assertThat(card.getVersionInfo(), is("Revision: 02"));
47          assertThat(card.getVRam(), is(0L));
48      }
49  
50      @Test
51      void testParsePciDumpMultipleDisplayDevices() {
52          List<String> pcidump = Arrays.asList(//
53                  " 0:2:0: NVIDIA GeForce GTX 1080", //
54                  "\tVendor ID: 10de", //
55                  "\tProduct ID: 1b80", //
56                  "\tClass: 03 Display, VGA", //
57                  "\tRevision: a1", //
58                  " 0:3:0: NVIDIA GeForce GTX 1080 Audio", //
59                  "\tVendor ID: 10de", //
60                  "\tProduct ID: 10f0", //
61                  "\tClass: 04 Multimedia, Audio", //
62                  " 0:5:0: AMD Radeon RX 580", //
63                  "\tVendor ID: 1002", //
64                  "\tProduct ID: 67df", //
65                  "\tClass: 03 Display, VGA", //
66                  "\tRevision: e7");
67          List<GraphicsCard> cards = BsdGraphicsCard.parsePciDump(pcidump);
68          assertThat(cards, hasSize(2));
69          assertThat(cards.get(0).getName(), is("NVIDIA GeForce GTX 1080"));
70          assertThat(cards.get(0).getDeviceId(), is("0x1b80"));
71          assertThat(cards.get(1).getName(), is("AMD Radeon RX 580"));
72          assertThat(cards.get(1).getDeviceId(), is("0x67df"));
73          assertThat(cards.get(1).getVersionInfo(), is("Revision: e7"));
74      }
75  
76      @Test
77      void testParsePciDumpNoDisplayDevices() {
78          List<String> pcidump = Arrays.asList(//
79                  " 0:0:0: Intel Host Bridge", //
80                  "\tVendor ID: 8086", //
81                  "\tProduct ID: 0001", //
82                  "\tClass: 06 Bridge, Host");
83          List<GraphicsCard> cards = BsdGraphicsCard.parsePciDump(pcidump);
84          assertThat(cards, is(empty()));
85      }
86  
87      @Test
88      void testParsePciDumpEmpty() {
89          List<GraphicsCard> cards = BsdGraphicsCard.parsePciDump(Collections.emptyList());
90          assertThat(cards, is(empty()));
91      }
92  
93      @Test
94      void testParsePciDumpDisplayAtEndOfOutput() {
95          // Display device is the last entry (no following header to flush it)
96          List<String> pcidump = Arrays.asList(//
97                  " 0:2:0: VGA Compatible Controller", //
98                  "\tVendor ID: abcd", //
99                  "\tProduct ID: 1234", //
100                 "\tClass: 03 Display, VGA", //
101                 "\tRevision: ff");
102         List<GraphicsCard> cards = BsdGraphicsCard.parsePciDump(pcidump);
103         assertThat(cards, hasSize(1));
104         assertThat(cards.get(0).getVendor(), is("0xabcd"));
105         assertThat(cards.get(0).getDeviceId(), is("0x1234"));
106     }
107 }
