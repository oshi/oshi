7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.SoundCard;
19  
20  class FreeBsdSoundCardTest {
21  
22      @Test
23      void testParseSoundCards() {
24          // A pcm-driven node carries the product/vendor; other nodes (e.g. the parent hdac) are ignored.
25          List<String> lshal = Arrays.asList(//
26                  "udi = '/org/freedesktop/Hal/devices/pci_8086_a348'", //
27                  "  info.product = 'Cannon Lake PCH cAVS'  (string)", //
28                  "udi = '/org/freedesktop/Hal/devices/pcm0'", //
29                  "  info.product = 'Realtek ALC892 (Analog)'  (string)", //
30                  "  info.vendor = 'Realtek'  (string)", //
31                  "  freebsd.driver = 'pcm'  (string)");
32          List<SoundCard> cards = FreeBsdSoundCard.parseSoundCards(lshal);
33          assertThat(cards, hasSize(1));
34          SoundCard card = cards.get(0);
35          assertThat(card.getName(), is("Realtek Realtek ALC892 (Analog)"));
36          assertThat(card.getCodec(), is("Realtek ALC892 (Analog)"));
37      }
38  
39      @Test
40      void testParseSoundCardsMissingVendor() {
41          // With no info.vendor, the name is just the product (leading space from the "vendor product" join).
42          List<String> lshal = Arrays.asList(//
43                  "udi = '/org/freedesktop/Hal/devices/pcm0'", //
44                  "  info.product = 'HDA Generic'  (string)", //
45                  "  freebsd.driver = 'pcm'  (string)");
46          List<SoundCard> cards = FreeBsdSoundCard.parseSoundCards(lshal);
47          assertThat(cards, hasSize(1));
48          assertThat(cards.get(0).getCodec(), is("HDA Generic"));
49      }
50  
51      @Test
52      void testParseSoundCardsEmpty() {
53          assertThat(FreeBsdSoundCard.parseSoundCards(Collections.emptyList()), is(empty()));
54      }
55  }
