7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  
16  import org.junit.jupiter.api.Test;
17  
18  import oshi.hardware.SoundCard;
19  
20  class SolarisSoundCardTest {
21  
22      @Test
23      void testParseLshal() {
24          // Representative lshal output with a sound device
25          List<String> lshal = Arrays.asList(//
26                  "udi = '/org/freedesktop/Hal/devices/pci_8086_293e'", //
27                  "  info.solaris.driver = 'audio810'  (string)", //
28                  "  info.product = 'ICH9 HD Audio Controller'  (string)", //
29                  "  info.vendor = 'Intel Corporation'  (string)", //
30                  "", //
31                  "udi = '/org/freedesktop/Hal/devices/pci_8086_0001'", //
32                  "  info.solaris.driver = 'e1000g'  (string)", //
33                  "  info.product = 'Network Adapter'  (string)", //
34                  "  info.vendor = 'Intel Corporation'  (string)");
35          List<SoundCard> cards = SolarisSoundCard.parseLshal(lshal);
36          assertThat(cards, hasSize(1));
37          SoundCard card = cards.get(0);
38          assertThat(card.getName(), is("Intel Corporation ICH9 HD Audio Controller"));
39          assertThat(card.getCodec(), is("ICH9 HD Audio Controller"));
40          assertThat(card.getDriverVersion(), is("ICH9 HD Audio Controller audio810"));
41      }
42  
43      @Test
44      void testParseLshalMultipleAudioDevices() {
45          List<String> lshal = Arrays.asList(//
46                  "udi = '/org/freedesktop/Hal/devices/pci_8086_293e'", //
47                  "  info.solaris.driver = 'audio810'  (string)", //
48                  "  info.product = 'HD Audio'  (string)", //
49                  "  info.vendor = 'Intel'  (string)", //
50                  "", //
51                  "udi = '/org/freedesktop/Hal/devices/pci_1002_aa38'", //
52                  "  info.solaris.driver = 'audio810'  (string)", //
53                  "  info.product = 'Radeon Audio'  (string)", //
54                  "  info.vendor = 'AMD'  (string)");
55          List<SoundCard> cards = SolarisSoundCard.parseLshal(lshal);
56          assertThat(cards, hasSize(2));
57          assertThat(cards.get(0).getName(), is("Intel HD Audio"));
58          assertThat(cards.get(1).getName(), is("AMD Radeon Audio"));
59      }
60  
61      @Test
62      void testParseLshalEmpty() {
63          List<SoundCard> cards = SolarisSoundCard.parseLshal(Collections.emptyList());
64          assertThat(cards, is(empty()));
65      }
66  
67      @Test
68      void testParseLshalNoAudioDriver() {
69          // Devices exist but none have audio810 driver
70          List<String> lshal = Arrays.asList(//
71                  "udi = '/org/freedesktop/Hal/devices/pci_8086_0001'", //
72                  "  info.solaris.driver = 'e1000g'  (string)", //
73                  "  info.product = 'Network Adapter'  (string)", //
74                  "  info.vendor = 'Intel Corporation'  (string)");
75          List<SoundCard> cards = SolarisSoundCard.parseLshal(lshal);
76          assertThat(cards, is(empty()));
77      }
