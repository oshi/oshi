22      @Test
23      void testParseSmbios() {
24          // Representative smbios output with BIOS, System, and Baseboard sections
25          List<String> smbios = Arrays.asList(//
26                  "ID    SIZE TYPE", //
27                  "0     87   SMB_TYPE_BIOS (BIOS Information)", //
28                  "", //
29                  "  Vendor: Parallels Software International Inc.", //
30                  "  Version String: 11.2.1 (32686)", //
31                  "  Release Date: 07/15/2016", //
32                  "  Address Segment: 0xf000", //
33                  "", //
34                  "ID    SIZE TYPE", //
35                  "1     177  SMB_TYPE_SYSTEM (system information)", //
36                  "", //
37                  "  Manufacturer: Parallels Software International Inc.", //
38                  "  Product: Parallels Virtual Platform", //
39                  "  Version: None", //
40                  "  Serial Number: Parallels-45 2E 7E 2D 57 5C 4B 59 B1 30 28 81 B7 81 89 34", //
41                  "  UUID: 452e7e2d-575c04b59-b130-2881b7818934", //
42                  "", //
43                  "ID    SIZE TYPE", //
44                  "2     90   SMB_TYPE_BASEBOARD (base board)", //
45                  "", //
46                  "  Manufacturer: Parallels Software International Inc.", //
47                  "  Product: Parallels Virtual Platform", //
48                  "  Version: None", //
49                  "  Serial Number: None");
50  
51          EnumMap<SmbType, Map<String, String>> result = SolarisComputerSystem.parseSmbios(smbios);
52  
53          Map<String, String> bios = result.get(SmbType.SMB_TYPE_BIOS);
54          assertThat(bios.get("Vendor"), is("Parallels Software International Inc."));
55          assertThat(bios.get("Version String"), is("11.2.1 (32686)"));
56          assertThat(bios.get("Release Date"), is("07/15/2016"));
57  
58          Map<String, String> system = result.get(SmbType.SMB_TYPE_SYSTEM);
59          assertThat(system.get("Manufacturer"), is("Parallels Software International Inc."));
60          assertThat(system.get("Product"), is("Parallels Virtual Platform"));
61          assertThat(system.get("Serial Number"), is("Parallels-45 2E 7E 2D 57 5C 4B 59 B1 30 28 81 B7 81 89 34"));
62          assertThat(system.get("UUID"), is("452e7e2d-575c04b59-b130-2881b7818934"));
63  
64          Map<String, String> baseboard = result.get(SmbType.SMB_TYPE_BASEBOARD);
65          assertThat(baseboard.get("Manufacturer"), is("Parallels Software International Inc."));
66          assertThat(baseboard.get("Product"), is("Parallels Virtual Platform"));
67          assertThat(baseboard.get("Serial Number"), is("None"));
68      }
69  
70      @Test
71      void testParseSmbiosEmpty() {
72          EnumMap<SmbType, Map<String, String>> result = SolarisComputerSystem.parseSmbios(Collections.emptyList());
73          assertThat(result.get(SmbType.SMB_TYPE_BIOS).isEmpty(), is(true));
74          assertThat(result.get(SmbType.SMB_TYPE_SYSTEM).isEmpty(), is(true));
75          assertThat(result.get(SmbType.SMB_TYPE_BASEBOARD).isEmpty(), is(true));
76      }

81                  "System Configuration:  Sun Microsystems  sun4u", //

83                  "      chassis-sn:  'ABC123'", //
84                  "    banner-name:  'Sun Ultra 5/10 UPA/PCI'");
85          assertThat(SolarisComputerSystem.parseSerialFromPrtconf(prtconf), is("ABC123"));
86      }
