5   package oshi.hardware.common.platform.unix.solaris;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.closeTo;
9   import static org.hamcrest.Matchers.is;
10  
11  import java.util.Arrays;
12  import java.util.Collections;
13  
14  import org.junit.jupiter.api.Test;
15  
16  class SolarisSensorsTest {
17  
18      @Test
19      void testParseCpuTemperature() {
20          assertThat(
21                  SolarisSensors.parseCpuTemperature(
22                          Arrays.asList("  :Temperature  42", "    Temperature:  36", "    Temperature:  42")),
23                  is(closeTo(42.0, 0.001)));
24      }
25  
26      @Test
27      void testParseCpuTemperatureMillidegrees() {
28          assertThat(
29                  SolarisSensors.parseCpuTemperature(Arrays.asList("    Temperature:  45000", "    Temperature:  42000")),
30                  is(closeTo(45.0, 0.001)));
31      }
32  
33      @Test
34      void testParseCpuTemperatureEmpty() {
35          assertThat(SolarisSensors.parseCpuTemperature(Collections.emptyList()), is(closeTo(0.0, 0.001)));
36      }
37  
38      @Test
39      void testParseFanSpeeds() {
40          int[] speeds = SolarisSensors
41                  .parseFanSpeeds(Arrays.asList("    Speed:  1200", "    Speed:  1500", "    Speed:  900"));
42          assertThat(speeds.length, is(3));
43          assertThat(speeds[0], is(1200));
44          assertThat(speeds[1], is(1500));
45          assertThat(speeds[2], is(900));
46      }
47  
48      @Test
49      void testParseFanSpeedsEmpty() {
50          int[] speeds = SolarisSensors.parseFanSpeeds(Collections.emptyList());
51          assertThat(speeds.length, is(0));
52      }
53  
54      @Test
55      void testParseVoltage() {
56          assertThat(SolarisSensors.parseVoltage(Arrays.asList("    Voltage:  1.200", "    Voltage:  3.300")),
57                  is(closeTo(1.2, 0.001)));
58      }
59  
60      @Test
61      void testParseVoltageEmpty() {
62          assertThat(SolarisSensors.parseVoltage(Collections.emptyList()), is(closeTo(0.0, 0.001)));
63      }
64  
65      @Test
66      void testParseVoltageNoMatch() {
67          assertThat(SolarisSensors.parseVoltage(Arrays.asList("  Other:  1.200", "  Something:  3.300")),
68                  is(closeTo(0.0, 0.001)));
69      }
70  }
