7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.hasSize;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.HashMap;
15  import java.util.List;
16  import java.util.Map;
17  
18  import org.junit.jupiter.api.Test;
19  
20  import oshi.hardware.Printer;
21  import oshi.hardware.Printer.PrinterStatus;
22  
23  class CupsPrinterTest {
24  
25      /** Minimal concrete subclass for testing. */
26      private static final class StubPrinter extends CupsPrinter {
27          StubPrinter(String name, String driverName, String description, PrinterStatus status, String statusReason,
28                  boolean isDefault, boolean isLocal, String portName) {
29              super(name, driverName, description, status, statusReason, isDefault, isLocal, portName);
30          }
31      }
32  
33      private static final CupsPrinter.PrinterFactory STUB_FACTORY = StubPrinter::new;
34  
35      // Fixture: lpstat -p output with two printers
36      private static final List<String> LPSTAT_P = Arrays.asList(
37              "printer HP_LaserJet is idle.  enabled since Mon 01 Jan 2024 12:00:00 AM",
38              "printer PDF_Printer disabled since Tue 02 Jan 2024 - paused by admin");
39  
40      private static final Map<String, String> PORT_MAP = new HashMap<>();
41      private static final Map<String, String> DESC_MAP = new HashMap<>();
42      static {
43          PORT_MAP.put("HP_LaserJet", "usb://HP/LaserJet");
44          PORT_MAP.put("PDF_Printer", "cups-pdf:/");
45          DESC_MAP.put("HP_LaserJet", "HP LaserJet Pro MFP");
46          DESC_MAP.put("PDF_Printer", "CUPS-PDF Virtual Printer");
47      }
48  
49      @Test
50      void testGetPrintersFromLpstatTwoPrinters() {
51          List<Printer> printers = CupsPrinter.getPrintersFromLpstat(LPSTAT_P, "HP_LaserJet", PORT_MAP, DESC_MAP,
52                  name -> "TestDriver", STUB_FACTORY);
53          assertThat(printers, hasSize(2));
54  
55          Printer hp = printers.get(0);
56          assertThat(hp.getName(), is("HP_LaserJet"));
57          assertThat(hp.getStatus(), is(PrinterStatus.IDLE));
58          assertThat(hp.isDefault(), is(true));
59          assertThat(hp.isLocal(), is(true));
60          assertThat(hp.getPortName(), is("usb://HP/LaserJet"));
61          assertThat(hp.getDescription(), is("HP LaserJet Pro MFP"));
62          assertThat(hp.getDriverName(), is("TestDriver"));
63  
64          Printer pdf = printers.get(1);
65          assertThat(pdf.getName(), is("PDF_Printer"));
66          assertThat(pdf.getStatus(), is(PrinterStatus.OFFLINE));
67          assertThat(pdf.isDefault(), is(false));
68          assertThat(pdf.getStatusReason(), is("paused by admin"));
69      }
70  
71      @Test
72      void testGetPrintersFromLpstatEmpty() {
73          List<Printer> printers = CupsPrinter.getPrintersFromLpstat(Collections.emptyList(), "", Collections.emptyMap(),
74                  Collections.emptyMap(), name -> "", STUB_FACTORY);
75          assertThat(printers, is(empty()));
76      }
77  
78      @Test
79      void testGetPrintersFromLpstatSkipsNonPrinterLines() {
80          List<String> mixed = Arrays.asList("scheduler is running",
81                  "printer MyPrinter now printing MyPrinter-42. enabled since Mon 01 Jan",
82                  "no system default destination");
83          List<Printer> printers = CupsPrinter.getPrintersFromLpstat(mixed, "", Collections.emptyMap(),
84                  Collections.emptyMap(), name -> "", STUB_FACTORY);
85          assertThat(printers, hasSize(1));
86          assertThat(printers.get(0).getName(), is("MyPrinter"));
87          assertThat(printers.get(0).getStatus(), is(PrinterStatus.PRINTING));
88      }
89  }
