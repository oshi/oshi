46      @Test
47      void testSyntheticRoundTrip() {
48          DisplayInfo info = new DisplayInfoImpl("AUO", "9227", "162C0C25", (byte) 44, 2012, "1.4", true, 60, 34,
49                  "2560x1440", "Thunderbolt", "C02JM2PFF2GC");
50          assertThat("synthetic", info.isEdidSynthetic(), is(true));
51          // Exercise the synthetic-branch field getters (return the stored values directly)
52          assertThat("synth manufacturerId", info.getManufacturerID(), is("AUO"));
53          assertThat("synth productId", info.getProductID(), is("9227"));
54          assertThat("synth serialNo", info.getSerialNo(), is("162C0C25"));
55          assertThat("synth week", info.getWeek(), is((byte) 44));
56          assertThat("synth year", info.getYear(), is(2012));
57          assertThat("synth version", info.getVersion(), is("1.4"));
58          assertThat("synth digital", info.isDigital(), is(true));
59          assertThat("synth hcm", info.getHcm(), is(60));
60          assertThat("synth vcm", info.getVcm(), is(34));
61          assertThat("synth preferredResolution", info.getPreferredResolution(), is("2560x1440"));
62          assertThat("synth model", info.getModel(), is("Thunderbolt"));
63          assertThat("synth productSerialNumber", info.getProductSerialNumber(), is("C02JM2PFF2GC"));
64          // getEdid() is idempotent and stable across calls
65          assertThat("stable edid", info.getEdid(), is(info.getEdid()));
66          // Re-parsing the synthesized EDID yields the same field values
67          DisplayInfo reparsed = new DisplayInfoImpl(info.getEdid());
68          assertThat("reparsed not synthetic", reparsed.isEdidSynthetic(), is(false));
69          assertThat("manufacturerId", reparsed.getManufacturerID(), is("AUO"));
70          assertThat("productId", reparsed.getProductID(), is("9227"));
71          assertThat("serialNo", reparsed.getSerialNo(), is("162C0C25"));
72          assertThat("week", reparsed.getWeek(), is((byte) 44));
73          assertThat("year", reparsed.getYear(), is(2012));
74          assertThat("version", reparsed.getVersion(), is("1.4"));
75          assertThat("digital", reparsed.isDigital(), is(true));
76          assertThat("hcm", reparsed.getHcm(), is(60));
77          assertThat("vcm", reparsed.getVcm(), is(34));
78          assertThat("preferredResolution", reparsed.getPreferredResolution(), is("2560x1440"));
79          assertThat("model", reparsed.getModel(), is("Thunderbolt"));
80          assertThat("productSerialNumber", reparsed.getProductSerialNumber(), is("C02JM2PFF2GC"));
81      }

83      @Test
84      void testSyntheticToStringDoesNotSynthesize() {
85          // A serial number that EdidUtil.setSerialNo string form rejects: toString must still work (it formats from the
86          // decoded fields) without triggering EDID synthesis. getEdid gracefully falls back to zeroed bytes 12-15.
87          DisplayInfo info = new DisplayInfoImpl("AUO", "9227", "BADSERIAL", (byte) 1, 2020, "1.4", true, 50, 30,
88                  "1920x1080", "Acme", "SN12345");
89          assertThat("toString model", info.toString(), containsString("Acme"));
90          assertThat("toString serial", info.toString(), containsString("SN12345"));
91          // getEdid no longer throws; it falls back to zero serial bytes
92          byte[] edid = info.getEdid();
93          assertThat("edid length", edid.length, is(128));
94      }

96      @Test
97      void testSyntheticNumericSerialFallback() {
98          // A serial like "FD626D62" fails EdidUtil.setSerialNo(String) because byte 0xFD is not alphanumeric while
99          // 0x62/'b' and 0x6D/'m' are, causing getSerialNo to return a mixed-format string that doesn't match.
100         // setSerialNoSafe falls back to the numeric overload, writing the bytes directly.
101         DisplayInfo info = new DisplayInfoImpl("APP", "a050", "FD626D62", (byte) 0, 1990, "1.4", true, 34, 22,
102                 "3456x2234", "Built-in Retina Display", "");
103         assertThat("serialNo field", info.getSerialNo(), is("FD626D62"));
104         // getEdid succeeds (no exception) and produces a valid 128-byte EDID
105         byte[] edid = info.getEdid();
106         assertThat("edid length", edid.length, is(128));
107         // Bytes 12-15 contain the numeric serial in little-endian
108         assertThat("edid[12]", edid[12] & 0xFF, is(0x62));
109         assertThat("edid[13]", edid[13] & 0xFF, is(0x6D));
110         assertThat("edid[14]", edid[14] & 0xFF, is(0x62));
111         assertThat("edid[15]", edid[15] & 0xFF, is(0xFD));
112     }

162     @Test
163     void testSyntheticNullProductSerialNumber() {
164         // A null product serial number is omitted from both the synthesized EDID and the toString output; an analog
165         // display also exercises the non-digital toString branch
166         DisplayInfo info = new DisplayInfoImpl("AUO", "9227", "162C0C25", (byte) 44, 2012, "1.4", false, 60, 34,
167                 "2560x1440", "Thunderbolt", null);
168         assertThat("analog", info.isDigital(), is(false));
169         assertThat("toString analog", info.toString(), containsString("Analog"));
170         // An absent product serial number is reported as an empty string, never null
171         assertThat("productSerialNumber", info.getProductSerialNumber(), is(""));
172         assertThat("toString omits serial", info.toString(), not(containsString("Serial Number")));
173         // The synthesized EDID has no serial-number descriptor, so a reparse also finds none
174         DisplayInfo reparsed = new DisplayInfoImpl(info.getEdid());
175         assertThat("reparsed productSerialNumber", reparsed.getProductSerialNumber(), is(""));
176     }
