5   package oshi.software.os.shared;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThanOrEqualTo;
9   import static org.hamcrest.Matchers.is;
10  import static org.hamcrest.Matchers.notNullValue;
11  
12  import java.util.List;
13  
14  import org.junit.jupiter.api.Test;
15  
16  import oshi.SystemInfo;
17  import oshi.software.os.FileSystem;
18  import oshi.software.os.OSFileStore;
19  import oshi.util.PlatformEnum;
20  
21  /**
22   * Test File System
23   */
24  class FileSystemTest {
25  
26      /**
27       * Test file system.
28       *
29       */
30      @Test
31      void testFileSystem() {
32          SystemInfo si = new SystemInfo();
33          FileSystem filesystem = si.getOperatingSystem().getFileSystem();
34          assertThat("File system open file descriptors should be 0 or higher", filesystem.getOpenFileDescriptors(),
35                  greaterThanOrEqualTo(0L));
36          assertThat("File system max open file descriptors should be 0 or higher", filesystem.getMaxFileDescriptors(),
37                  greaterThanOrEqualTo(0L));
38          assertThat("File system max open file descriptors per process should be 0 or higher",
39                  filesystem.getMaxFileDescriptorsPerProcess(), greaterThanOrEqualTo(0L));
40          filesystem.getMaxFileDescriptorsPerProcess();
41          List<OSFileStore> stores = filesystem.getFileStores();
42          int updateCount = 0;
43          for (OSFileStore store : stores) {
44              assertThat("File store name shouldn't be null", store.getName(), is(notNullValue()));
45              assertThat("File store volume shouldn't be null", store.getVolume(), is(notNullValue()));
46              assertThat("File store label shouldn't be null", store.getLabel(), is(notNullValue()));
47              assertThat("File store logical volume shouldn't be null", store.getLogicalVolume(), is(notNullValue()));
48              assertThat("File store description shouldn't be null", store.getDescription(), is(notNullValue()));
49              assertThat("File store type shouldn't be null", store.getType(), is(notNullValue()));
50              assertThat("File store options shouldn't be empty", store.getOptions().isEmpty(), is(false));
51              assertThat("File store mount shouldn't be null", store.getMount(), is(notNullValue()));
52              assertThat("File store UUID shouldn't be null", store.getUUID(), is(notNullValue()));
53              assertThat("File store total space should be 0 or higher", store.getTotalSpace() >= 0, is(true));
54              assertThat("File store usable space should be 0 or higher", store.getUsableSpace() >= 0, is(true));
55              assertThat("File store free space should be 0 or higher", store.getFreeSpace() >= 0, is(true));
56              if (PlatformEnum.getCurrentPlatform() != PlatformEnum.WINDOWS) {
57                  assertThat("Number of free inodes should be 0 or higher on non-Windows systems",
58                          store.getFreeInodes() >= 0, is(true));
59                  if (PlatformEnum.getCurrentPlatform() != PlatformEnum.SOLARIS) {
60                      assertThat(
61                              "Total number of inodes should be greater than or equal to number of free inodes on non-Windows/Solaris systems",
62                              store.getTotalInodes() >= store.getFreeInodes(), is(true));
63                  }
64              }
65              if (!store.getDescription().equals("Network drive")) {
66                  assertThat(
67                          "File store's usable space should be less than or equal to the total space on non-network drives",
68                          store.getUsableSpace() <= store.getTotalSpace(), is(true));
69              }
70              // updateAttributes should succeed and total space should not change
71              if (++updateCount <= 10) {
72                  long totalBefore = store.getTotalSpace();
73                  assertThat("File store updateAttributes should succeed for " + store.getMount(),
74                          store.updateAttributes(), is(true));
75                  if (PlatformEnum.getCurrentPlatform() != PlatformEnum.SOLARIS) {
76                      assertThat("File store total space should not change after update", store.getTotalSpace(),
77                              is(totalBefore));
78                  }
79                  assertThat("File store usable space should remain valid after update", store.getUsableSpace(),
80                          greaterThanOrEqualTo(0L));
81              }
82          }
83      }
84  }
