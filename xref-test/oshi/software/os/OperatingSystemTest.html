76          assertThat("OS family shouldn't be null", os.getFamily(), is(notNullValue()));
77          assertThat("OS manufacturer shouldn't be null", os.getManufacturer(), is(notNullValue()));
78          OSVersionInfo versionInfo = os.getVersionInfo();
79          assertThat("OS version info shouldn't be null", versionInfo, is(notNullValue()));

81          assertThat("OS uptime in seconds should be greater than 0", os.getSystemUptime(), is(greaterThan(0L)));
82          assertThat("OS boot time in seconds since Unix epoch should be greater than 0", os.getSystemBootTime(),
83                  is(greaterThan(0L)));
84          assertThat("OS boot time in seconds since Unix epoch should be before the current time", os.getSystemBootTime(),
85                  is(lessThan(System.currentTimeMillis() / 1000L)));
86  
87          assertThat("OS should have 1 or more currently running processes", os.getProcessCount(), is(greaterThan(0)));
88          assertThat("OS should have 1 or more currently running threads", os.getThreadCount(), is(greaterThan(0)));
89          assertThat("OS bitness should either be 32 or 64 ", os.getBitness(), is(oneOf(32, 64)));
90          assertThat("The current process id should be greater than 0", os.getProcessId(), is(greaterThan(0)));
91          // Just exercise this code without error
92          assertThat(
93                  "The current process' permissions (if has sudo or Administrator privileges) should be determined correctly",
94                  os.isElevated(), is(anything()));
95  
96          assertThat("OS should have at least 1 currently running process", os.getProcesses(0, null), is(not(empty())));

168      * Tests process query by pid list

170     @Test
171     void testProcessQueryByList() {
172         SystemInfo si = new SystemInfo();
173         OperatingSystem os = si.getOperatingSystem();
174         assertThat("OS family shouldn't be null", os.getFamily(), is(notNullValue()));
175         assertThat("OS manufacturer shouldn't be null", os.getManufacturer(), is(notNullValue()));
176         OSVersionInfo versionInfo = os.getVersionInfo();
177         assertThat("OS version info shouldn't be null", versionInfo, is(notNullValue()));
178 
179         assertThat("OS currently running processes should be 1 or higher", os.getProcessCount(), is(greaterThan(0)));
180         assertThat("OS thread count should be 1 or higher", os.getThreadCount(), is(greaterThan(0)));
181         assertThat("OS current running process id should be 0 or higher", os.getProcessId(),
182                 is(greaterThanOrEqualTo(0)));
183 
184         List<OSProcess> processes = os.getProcesses(5, null);
185         assertThat("Currently running processes shouldn't be null", processes, is(notNullValue()));
186         assertThat("every OS should have at least one process running on it", processes, is(not(empty())));
187         // the list of pids we want info on
188         List<Integer> pids = new ArrayList<>();
189         for (OSProcess p : processes) {
190             pids.add(p.getProcessID());
191         }
192         // query for just those processes
193         Collection<OSProcess> processes1 = os.getProcesses(pids);
194         // there's a potential for a race condition here, if a process we
195         // queried for initially wasn't running during the second query. In this case,
196         // try again with the shorter list
197         while (processes1.size() < pids.size()) {
198             pids.clear();
199             for (OSProcess p : processes1) {
200                 pids.add(p.getProcessID());
201             }
202             // query for just those processes
203             processes1 = os.getProcesses(pids);
204         }
205         assertThat("OS processes should match processes with pids we want info on", pids, hasSize(processes1.size()));
206 
207     }

209     /**
210      * Tests child process getter
211      */
212     @Test
213     void testGetChildProcesses() {
214         // Testing child processes is tricky because we don't really know a priori what
215         // processes might have children, and if we do test the full list vs. individual
216         // processes, we run into a race condition where child processes can start or
217         // stop before we measure a second time. So we can't really test for one-to-one
218         // correspondence of child process lists.
219         //
220         // We can expect code logic failures to occur all/most of the time for
221         // categories of processes, however, and allow occasional differences due to
222         // race conditions. So we will test three categories of processes: Those with 0
223         // children, those with exactly 1 child process, and those with multiple child
224         // processes. On the second poll, we expect at least half of processes in those
225         // categories to still be in the same category.
226         //
227         SystemInfo si = new SystemInfo();
228         OperatingSystem os = si.getOperatingSystem();
229         List<OSProcess> processes = os.getProcesses(0, null);
230         Set<Integer> zeroChildSet = new HashSet<>();
231         Set<Integer> oneChildSet = new HashSet<>();
232         Set<Integer> manyChildSet = new HashSet<>();
233         // Initialize all processes with no children
234         for (OSProcess p : processes) {
235             zeroChildSet.add(p.getProcessID());
236         }
237         // Move parents with 1 or more children to other set
238         for (OSProcess p : processes) {
239             if (zeroChildSet.contains(p.getParentProcessID())) {
240                 // Zero to One
241                 zeroChildSet.remove(p.getParentProcessID());
242                 oneChildSet.add(p.getParentProcessID());
243             } else if (oneChildSet.contains(p.getParentProcessID())) {
244                 // One to many
245                 oneChildSet.remove(p.getParentProcessID());
246                 manyChildSet.add(p.getParentProcessID());
247             }
248         }
249         // Now test that majority of each set is in same category
250         int matched = 0;
251         int total = 0;
252         for (Integer i : zeroChildSet) {
253             if (os.getChildProcesses(i, 0, null).isEmpty()) {
254                 matched++;
255             }
256             // Quit if enough to test
257             if (++total > 9) {
258                 break;
259             }
260         }
261         if (total > 4) {
262             assertThat("Most processes with no children should not suddenly have them.", matched,
263                     is(greaterThan(total / 2)));
264         }
265         matched = 0;
266         total = 0;
267         for (Integer i : oneChildSet) {
268             if (os.getChildProcesses(i, 0, null).size() == 1) {
269                 matched++;
270             }
271             // Quit if enough to test
272             if (++total > 9) {
273                 break;
274             }
275         }
276         if (total > 4) {
277             assertThat("Most processes with one child should not suddenly have zero or more than one.", matched,
278                     is(greaterThan(total / 2)));
279         }
280         matched = 0;
281         total = 0;
282         for (Integer i : manyChildSet) {
283             // Use a non-null sorting for test purposes
284             if (os.getChildProcesses(i, Integer.MAX_VALUE, ProcessSort.PID).size() > 1) {
285                 matched++;
286             }
287             // Quit if enough to test
288             if (++total > 9) {
289                 break;
290             }
291         }
292         if (total > 4) {
293             assertThat("Most processes with more than one child should not suddenly have one or less.", matched,
294                     is(greaterThan(total / 2)));
295         }
296     }

314     /**
315      * Tests services getter
316      */
317     @Test
318     void testGetServices() {
319         SystemInfo si = new SystemInfo();
320         OperatingSystem os = si.getOperatingSystem();
321         int stopped = 0;
322         int running = 0;
323         for (OSService svc : os.getServices()) {
324             assertThat(svc.getName(), is(not(emptyString())));
325             switch (svc.getState()) {
326             case STOPPED:
327                 stopped++;
328                 break;
329             case RUNNING:
330                 running++;
331                 break;
332             default:
333                 break;
334             }
335         }
336         // Should be at least one of each
337         assertThat("There should be at least 1 stopped service", stopped, is(greaterThan(0)));
338         assertThat("There should be at least 1 running service", running, is(greaterThan(0)));
339     }
