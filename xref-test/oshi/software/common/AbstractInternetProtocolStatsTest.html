5   package oshi.software.common;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import org.junit.jupiter.api.Test;
11  
12  import oshi.software.os.InternetProtocolStats.TcpStats;
13  import oshi.software.os.InternetProtocolStats.UdpStats;
14  
15  class AbstractInternetProtocolStatsTest {
16  
17      private static final AbstractInternetProtocolStats STATS = new AbstractInternetProtocolStats() {
18          @Override
19          public TcpStats getTCPv4Stats() {
20              return new TcpStats(1L, 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L, 0L);
21          }
22  
23          @Override
24          public UdpStats getUDPv4Stats() {
25              return new UdpStats(1L, 0L, 0L, 0L);
26          }
27      };
28  
29      @Test
30      void testDefaultTCPv6StatsAllZeros() {
31          TcpStats tcp6 = STATS.getTCPv6Stats();
32          assertThat(tcp6.getConnectionsEstablished(), is(0L));
33          assertThat(tcp6.getConnectionsActive(), is(0L));
34          assertThat(tcp6.getConnectionsPassive(), is(0L));
35          assertThat(tcp6.getConnectionFailures(), is(0L));
36          assertThat(tcp6.getConnectionsReset(), is(0L));
37          assertThat(tcp6.getSegmentsSent(), is(0L));
38          assertThat(tcp6.getSegmentsReceived(), is(0L));
39          assertThat(tcp6.getSegmentsRetransmitted(), is(0L));
40          assertThat(tcp6.getInErrors(), is(0L));
41          assertThat(tcp6.getOutResets(), is(0L));
42      }
43  
44      @Test
45      void testDefaultUDPv6StatsAllZeros() {
46          UdpStats udp6 = STATS.getUDPv6Stats();
47          assertThat(udp6.getDatagramsSent(), is(0L));
48          assertThat(udp6.getDatagramsReceived(), is(0L));
49          assertThat(udp6.getDatagramsNoPort(), is(0L));
50          assertThat(udp6.getDatagramsReceivedErrors(), is(0L));
51      }
52  }
