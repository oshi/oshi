21  class FreeBsdFileSystemTest {
22  
23      @Test
24      void testParseGeomPartListTypical() {
25          List<String> lines = Arrays.asList("Geom name: ada0", "Providers:", "1. Name: ada0p1",
26                  "   rawuuid: 3e1cbe42-e2b9-11e6-85c6-0025909460ac", "2. Name: ada0p2",
27                  "   rawuuid: 3e2e5580-e2b9-11e6-85c6-0025909460ac", "3. Name: ada0p3",
28                  "   rawuuid: 3e3ed3be-e2b9-11e6-85c6-0025909460ac");
29          Map<String, String> result = FreeBsdFileSystem.parseGeomPartList(lines);
30          assertThat(result, is(aMapWithSize(3)));
31          assertThat(result.get("ada0p1"), is("3e1cbe42-e2b9-11e6-85c6-0025909460ac"));
32          assertThat(result.get("ada0p2"), is("3e2e5580-e2b9-11e6-85c6-0025909460ac"));
33          assertThat(result.get("ada0p3"), is("3e3ed3be-e2b9-11e6-85c6-0025909460ac"));
34      }

36      @Test
37      void testParseGeomPartListEmpty() {
38          Map<String, String> result = FreeBsdFileSystem.parseGeomPartList(Collections.emptyList());
39          assertThat(result, is(anEmptyMap()));
40      }

42      @Test
43      void testParseDfInodesTypical() {
44          List<String> lines = Arrays.asList(
45                  "Filesystem    1K-blocks   Used   Avail Capacity iused  ifree %iused  Mounted on",
46                  "/dev/twed0s1a   2026030 584112 1279836    31%    2751 279871    1%   /",
47                  "/dev/twed0s1e  10240000 2048000 8192000    20%    5000 100000    5%   /usr");
48          Pair<Map<String, Long>, Map<String, Long>> result = FreeBsdFileSystem.parseDfInodes(lines);
49          Map<String, Long> freeMap = result.getA();
50          Map<String, Long> totalMap = result.getB();
51          assertThat(freeMap, is(aMapWithSize(2)));
52          assertThat(freeMap.get("/"), is(279871L));
53          assertThat(freeMap.get("/usr"), is(100000L));
54          assertThat(totalMap.get("/"), is(2751L + 279871L));
55          assertThat(totalMap.get("/usr"), is(5000L + 100000L));
56      }
57  
58      @Test
59      void testParseDfInodesEmpty() {
60          Pair<Map<String, Long>, Map<String, Long>> result = FreeBsdFileSystem.parseDfInodes(Collections.emptyList());
61          assertThat(result.getA(), is(anEmptyMap()));
62          assertThat(result.getB(), is(anEmptyMap()));
63      }
64  
65      @Test
66      void testParseDfInodesSkipsHeaderIncludesNonDevFs() {
67          List<String> lines = Arrays.asList(
68                  "Filesystem    1K-blocks   Used   Avail Capacity iused  ifree %iused  Mounted on",
69                  "devfs               1      1       0   100%       0      0  100%   /dev",
70                  "zroot/ROOT    95678456 8234567  87443889     9%  234567 9999999    2%   /");
71          Pair<Map<String, Long>, Map<String, Long>> result = FreeBsdFileSystem.parseDfInodes(lines);
72          Map<String, Long> freeMap = result.getA();
73          Map<String, Long> totalMap = result.getB();
74          // Header is skipped; devfs and zroot are both included
75          assertThat(freeMap.get("/dev"), is(0L));
76          assertThat(freeMap.get("/"), is(9999999L));
77          assertThat(totalMap.get("/"), is(234567L + 9999999L));
78      }
