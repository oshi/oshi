7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.aMapWithSize;
9   import static org.hamcrest.Matchers.anEmptyMap;
10  import static org.hamcrest.Matchers.is;
11  
12  import java.util.Arrays;
13  import java.util.Collections;
14  import java.util.List;
15  import java.util.Map;
16  
17  import org.junit.jupiter.api.Test;
18  
19  import oshi.util.tuples.Pair;
20  
21  class NetBsdFileSystemTest {
22  
23      @Test
24      void testParseDfInodesTypical() {
25          List<String> lines = Arrays.asList(
26                  "Filesystem  512-blocks      Used     Avail Capacity iused   ifree  %iused  Mounted on",
27                  "/dev/wd0a      2149212    908676   1133076    45%    8355  147163     5%   /",
28                  "/dev/wd0e      4050876        36   3848300     0%      10  285108     0%   /home",
29                  "/dev/wd0d      6082908   3343172   2435592    58%   27813  386905     7%   /usr");
30          Pair<Map<String, Long>, Map<String, Long>> result = NetBsdFileSystem.parseDfInodes(lines);
31          Map<String, Long> freeMap = result.getA();
32          Map<String, Long> usedMap = result.getB();
33          assertThat(freeMap, is(aMapWithSize(3)));
34          assertThat(freeMap.get("/dev/wd0a"), is(147163L));
35          assertThat(freeMap.get("/dev/wd0e"), is(285108L));
36          assertThat(freeMap.get("/dev/wd0d"), is(386905L));
37          assertThat(usedMap.get("/dev/wd0a"), is(8355L));
38          assertThat(usedMap.get("/dev/wd0e"), is(10L));
39          assertThat(usedMap.get("/dev/wd0d"), is(27813L));
40      }
41  
42      @Test
43      void testParseDfInodesEmpty() {
44          Pair<Map<String, Long>, Map<String, Long>> result = NetBsdFileSystem.parseDfInodes(Collections.emptyList());
45          assertThat(result.getA(), is(anEmptyMap()));
46          assertThat(result.getB(), is(anEmptyMap()));
47      }
48  
49      @Test
50      void testParseDfInodesSkipsHeader() {
51          List<String> lines = Arrays
52                  .asList("Filesystem  512-blocks      Used     Avail Capacity iused   ifree  %iused  Mounted on");
53          Pair<Map<String, Long>, Map<String, Long>> result = NetBsdFileSystem.parseDfInodes(lines);
54          assertThat(result.getA(), is(anEmptyMap()));
55          assertThat(result.getB(), is(anEmptyMap()));
56      }
57  }
