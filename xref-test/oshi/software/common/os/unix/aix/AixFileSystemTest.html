23      @Test
24      void testParseDfInodesTypical() {
25          List<String> lines = Arrays.asList("Filesystem    512-blocks     Ifree    Iused",
26                  "/dev/hd4         4194304   164951    15969", "/dev/hd2        52690944  2894117   196692",
27                  "/dev/hd9var      6291456   605443     2317");
28          Pair<Map<String, Long>, Map<String, Long>> result = AixFileSystem.parseDfInodes(lines);
29          Map<String, Long> freeMap = result.getA();
30          Map<String, Long> totalMap = result.getB();
31          assertThat(freeMap, is(aMapWithSize(3)));
32          assertThat(freeMap.get("/dev/hd4"), is(164951L));
33          assertThat(freeMap.get("/dev/hd2"), is(2894117L));
34          assertThat(freeMap.get("/dev/hd9var"), is(605443L));
35          assertThat(totalMap.get("/dev/hd4"), is(164951L + 15969L));
36          assertThat(totalMap.get("/dev/hd2"), is(2894117L + 196692L));
37          assertThat(totalMap.get("/dev/hd9var"), is(605443L + 2317L));
38      }
39  
40      @Test
41      void testParseDfInodesNfsMounts() {
42          List<String> lines = Arrays.asList("Filesystem    512-blocks     Ifree    Iused",
43                  "nfshost:/export  8388608   500000    10000");
44          Pair<Map<String, Long>, Map<String, Long>> result = AixFileSystem.parseDfInodes(lines);
45          Map<String, Long> freeMap = result.getA();
46          Map<String, Long> totalMap = result.getB();
47          // NFS mounts matching "hostname:/path" should be included via FS_PATTERN
48          assertThat(freeMap, is(aMapWithSize(1)));
49          assertThat(freeMap.get("nfshost:/export"), is(500000L));
50          assertThat(totalMap.get("nfshost:/export"), is(510000L));
51      }
52  
53      @Test
54      void testParseDfInodesSkipsProcAndHeader() {
55          List<String> lines = Arrays.asList("Filesystem    512-blocks     Ifree    Iused",
56                  "/proc                  -        -        -");
57          Pair<Map<String, Long>, Map<String, Long>> result = AixFileSystem.parseDfInodes(lines);
58          // Header doesn't match FS_PATTERN, /proc has dashes so parseLong returns 0
59          // Actually /proc starts with / so it does match FS_PATTERN
60          Map<String, Long> freeMap = result.getA();
61          assertThat(freeMap.get("/proc"), is(0L));
62      }
63  
64      @Test
65      void testParseDfInodesEmpty() {
66          Pair<Map<String, Long>, Map<String, Long>> result = AixFileSystem.parseDfInodes(Collections.emptyList());
67          assertThat(result.getA(), is(anEmptyMap()));
68          assertThat(result.getB(), is(anEmptyMap()));
69      }

71      @Test
72      void testParseOpenFileDescriptors() {
73          // lsof -nl: every row after the COMMAND header counts
74          List<String> lsof = Arrays.asList(//
75                  "COMMAND     PID     USER   FD   TYPE             DEVICE  SIZE/OFF  NODE NAME", //
76                  "init          1     root  cwd  VDIR              10,  5      256     2 /", //
77                  "init          1     root  txt  VREG              10,  6    12345    17 /usr/sbin/init");
78          assertThat(AixFileSystem.parseOpenFileDescriptors(lsof), is(2L));
79      }
80  
81      @Test
82      void testParseOpenFileDescriptorsNoHeaderOrEmpty() {
83          // No COMMAND header line: nothing is counted
84          assertThat(AixFileSystem.parseOpenFileDescriptors(Arrays.asList("garbage", "more garbage")), is(0L));
85          assertThat(AixFileSystem.parseOpenFileDescriptors(Collections.emptyList()), is(0L));
86      }

88      @Test
89      void testParseMaxFileDescriptorsPerProcess() {
90          List<String> limits = Arrays.asList("default:", "        fsize = -1", "        nofiles = 2000",
91                  "        core = 2097151");
92          assertThat(AixFileSystem.parseMaxFileDescriptorsPerProcess(limits), is(2000L));
93      }
94  
95      @Test
96      void testParseMaxFileDescriptorsPerProcessMissing() {
97          // No nofiles line: unlimited (Long.MAX_VALUE)
98          assertThat(AixFileSystem.parseMaxFileDescriptorsPerProcess(Arrays.asList("default:", "        fsize = -1")),
99                  is(Long.MAX_VALUE));
