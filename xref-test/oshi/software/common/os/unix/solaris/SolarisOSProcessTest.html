5   package oshi.software.common.os.unix.solaris;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Arrays;
11  import java.util.Collections;
12  import java.util.List;
13  
14  import org.junit.jupiter.api.Test;
15  
16  class SolarisOSProcessTest {
17  
18      @Test
19      void testParseOpenFileLimitSoft() {
20          List<String> plimit = Arrays.asList("1234:", "   coredumpsize      0           unlimited",
21                  "   nofiles(descriptors)  256         65536", "   vmemorysize       unlimited   unlimited");
22          assertThat(SolarisOSProcess.parseOpenFileLimit(plimit, 1), is(256L));
23      }
24  
25      @Test
26      void testParseOpenFileLimitHard() {
27          List<String> plimit = Arrays.asList("1234:", "   nofiles(descriptors)  256         65536");
28          assertThat(SolarisOSProcess.parseOpenFileLimit(plimit, 2), is(65536L));
29      }
30  
31      @Test
32      void testParseOpenFileLimitEmpty() {
33          assertThat(SolarisOSProcess.parseOpenFileLimit(Collections.emptyList(), 1), is(-1L));
34      }
35  
36      @Test
37      void testParseOpenFileLimitNoNofilesLine() {
38          List<String> plimit = Arrays.asList("1234:", "   coredumpsize      0           unlimited");
39          assertThat(SolarisOSProcess.parseOpenFileLimit(plimit, 1), is(-1L));
40      }
41  
42      @Test
43      void testParseOpenFileLimitIndexOutOfBounds() {
44          List<String> plimit = Arrays.asList("   nofiles(descriptors)  256         unlimited");
45          // "unlimited" is non-numeric, so split yields only ["", "256"]
46          assertThat(SolarisOSProcess.parseOpenFileLimit(plimit, 2), is(-1L));
47      }
48  
49      @Test
50      void testParsePbindOutputBound() {
51          String cpuset = "pid 101048 strongly bound to processor(s) 0 1 2 3.";
52          long mask = SolarisOSProcess.parsePbindOutput(cpuset);
53          // bits 0,1,2,3 set = 0b1111 = 15
54          assertThat(mask, is(15L));
55      }
56  
57      @Test
58      void testParsePbindOutputEmpty() {
59          assertThat(SolarisOSProcess.parsePbindOutput(""), is(0L));
60      }
61  
62      @Test
63      void testParsePbindOutputNotBound() {
64          // Some unexpected format that doesn't match the pattern
65          assertThat(SolarisOSProcess.parsePbindOutput("pid 1234 not bound"), is(0L));
66      }
67  
68      @Test
69      void testParsePsrinfo() {
70          List<String> psrinfo = Arrays.asList("0       on-line   since 01/01/2020 00:00:00",
71                  "1       on-line   since 01/01/2020 00:00:00", "4       on-line   since 01/01/2020 00:00:00");
72          long mask = SolarisOSProcess.parsePsrinfo(psrinfo);
73          // bits 0, 1, 4 set = 0b10011 = 19
74          assertThat(mask, is(19L));
75      }
76  
77      @Test
78      void testParsePsrinfoEmpty() {
79          assertThat(SolarisOSProcess.parsePsrinfo(Collections.emptyList()), is(0L));
80      }
81  }
