144     // --- V1 controller path resolution from /proc/self/cgroup ---
145 
146     // Real hybrid (v1 + unified) /proc/self/cgroup: numbered v1 hierarchies with comma-separated controllers, plus
147     // a trailing "0::" unified (v2) line whose controllers field is empty.
148     private static final List<String> SELF_CGROUP = Arrays.asList("12:hugetlb:/",
149             "11:cpu,cpuacct:/user.slice/user-1000.slice", "10:memory:/user.slice/user-1000.slice/session-3.scope",
150             "3:pids:/user.slice/user-1000.slice", "1:name=systemd:/user.slice/user-1000.slice/session-3.scope",
151             "0::/user.slice/user-1000.slice/session-3.scope");
152 
153     // SysPath's static initializer resolves the live sysfs mount, so these Linux-only paths are asserted on Linux CI
154     // (which is also where coverage is collected).
155 
156     @Test
157     @EnabledOnOs(OS.LINUX)
158     void resolveV1ControllerPathExactMatchInCommaList() {
159         // "cpu" is one of the comma-separated controllers; the full "cpu,cpuacct" mount name is used in the path
160         assertEquals(SysPath.CGROUP + "cpu,cpuacct/user.slice/user-1000.slice/",
161                 LinuxCgroupInfo.resolveV1ControllerPath(SELF_CGROUP, "cpu"));
162     }
163 
164     @Test
165     @EnabledOnOs(OS.LINUX)
166     void resolveV1ControllerPathSingleController() {
167         assertEquals(SysPath.CGROUP + "memory/user.slice/user-1000.slice/session-3.scope/",
168                 LinuxCgroupInfo.resolveV1ControllerPath(SELF_CGROUP, "memory"));
169     }
170 
171     @Test
172     @EnabledOnOs(OS.LINUX)
173     void resolveV1ControllerPathNotFoundUsesDefault() {
174         // No hierarchy lists "blkio"; fall back to the bare controller mount
175         assertEquals(SysPath.CGROUP + "blkio/", LinuxCgroupInfo.resolveV1ControllerPath(SELF_CGROUP, "blkio"));
176     }
177 
178     @Test
179     @EnabledOnOs(OS.LINUX)
180     void resolveV1ControllerPathSkipsUnifiedLine() {
181         // A pure-v2 cgroup (only the "0::" unified line, empty controllers) never matches -> default path
182         List<String> unifiedOnly = Collections.singletonList("0::/user.slice/user-1000.slice/session-3.scope");
183         assertEquals(SysPath.CGROUP + "cpu/", LinuxCgroupInfo.resolveV1ControllerPath(unifiedOnly, "cpu"));
184     }
