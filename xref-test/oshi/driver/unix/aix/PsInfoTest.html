19  import oshi.SystemInfo;
20  import oshi.jna.platform.unix.AixLibc.AixLwpsInfo;
21  import oshi.jna.platform.unix.AixLibc.AixPsInfo;
22  import oshi.util.Constants;

24  import oshi.util.tuples.Pair;
25  import oshi.util.tuples.Triplet;
26  
27  @EnabledOnOs(OS.AIX)
28  class PsInfoTest {
29      @Test
30      void testQueryPsInfo() {
31          int pid = new SystemInfo().getOperatingSystem().getProcessId();
32          AixPsInfo psinfo = PsInfo.queryPsInfo(pid);
33          assertThat("Process ID in structure should match PID", psinfo.pr_pid, is((long) pid));
34  
35          Triplet<Integer, Long, Long> addrs = PsInfo.queryArgsEnvAddrs(pid, psinfo);
36          assertNotNull(addrs);
37          Pair<List<String>, Map<String, String>> argsEnv = PsInfo.queryArgsEnv(pid, psinfo);
38          assertThat("Arg list size should match argc", argsEnv.getA().size(), is(addrs.getA().intValue()));
39  
40          File directory = new File(String.format("/proc/%d/lwp", pid));
41          File[] numericFiles = directory.listFiles(file -> Constants.DIGITS.matcher(file.getName()).matches());
42          assertNotNull(numericFiles);
43          for (File lwpidFile : numericFiles) {
44              int tid = ParseUtil.parseIntOrDefault(lwpidFile.getName(), 0);
45              AixLwpsInfo lwpsinfo = PsInfo.queryLwpsInfo(pid, tid);
46              assertThat("Thread ID in structure should match TID", lwpsinfo.pr_lwpid, is((long) tid));
47          }
48      }
49  }
