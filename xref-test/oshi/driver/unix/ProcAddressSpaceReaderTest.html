5   package oshi.driver.unix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import org.junit.jupiter.api.Test;
11  import org.junit.jupiter.api.condition.DisabledIfSystemProperty;
12  
13  import com.sun.jna.Memory;
14  
15  /**
16   * Tests the pointer decoding shared by the Solaris and AIX JNA {@code PsInfo} address-space readers.
17   * <p>
18   * Allocating a JNA {@link Memory} requires the JNA native dispatch library, so this test is disabled on the NetBSD CI
19   * runner that intentionally runs without it.
20   */
21  @DisabledIfSystemProperty(named = "os.name", matches = "(?i)netbsd")
22  class ProcAddressSpaceReaderTest {
23  
24      @Test
25      void testDecodePointer64Bit() {
26          try (Memory buffer = new Memory(8)) {
27              buffer.setLong(0, 0x0000_1234_5678_9abcL);
28              assertThat(ProcAddressSpaceReader.decodePointer(buffer, 0, 8), is(0x0000_1234_5678_9abcL));
29          }
30      }
31  
32      @Test
33      void testDecodePointer32BitLowAddress() {
34          try (Memory buffer = new Memory(4)) {
35              buffer.setInt(0, 0x0040_1000);
36              assertThat(ProcAddressSpaceReader.decodePointer(buffer, 0, 4), is(0x0040_1000L));
37          }
38      }
39  
40      @Test
41      void testDecodePointer32BitHighAddressIsUnsigned() {
42          // A 32-bit pointer with the high bit set must be read unsigned, not sign-extended to a negative long.
43          try (Memory buffer = new Memory(4)) {
44              buffer.setInt(0, 0x8000_0000);
45              assertThat(ProcAddressSpaceReader.decodePointer(buffer, 0, 4), is(0x0000_0000_8000_0000L));
46          }
47      }
48  
49      @Test
50      void testDecodePointer32BitMaxAddressIsUnsigned() {
51          try (Memory buffer = new Memory(4)) {
52              buffer.setInt(0, 0xFFFF_FFFF);
53              assertThat(ProcAddressSpaceReader.decodePointer(buffer, 0, 4), is(0x0000_0000_FFFF_FFFFL));
54          }
55      }
56  }
