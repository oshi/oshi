5   package oshi.driver.unix.solaris;

7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.junit.jupiter.api.Assertions.assertNotNull;
10  
11  import java.io.File;
12  
13  import org.junit.jupiter.api.Test;
14  import org.junit.jupiter.api.condition.EnabledOnOs;
15  import org.junit.jupiter.api.condition.OS;
16  
17  import oshi.SystemInfo;
18  import oshi.jna.platform.unix.SolarisLibc.SolarisLwpsInfo;
19  import oshi.jna.platform.unix.SolarisLibc.SolarisPsInfo;
20  import oshi.util.Constants;
21  import oshi.util.ParseUtil;
22  
23  @EnabledOnOs(OS.SOLARIS)
24  class PsInfoTest {
25      @Test
26      void testQueryPsInfo() {
27          int pid = new SystemInfo().getOperatingSystem().getProcessId();
28          SolarisPsInfo psinfo = PsInfo.queryPsInfo(pid);
29          assertThat("Process ID in structure should match PID", psinfo.pr_pid, is(pid));
30          File directory = new File(String.format("/proc/%d/lwp", pid));
31          File[] numericFiles = directory.listFiles(file -> Constants.DIGITS.matcher(file.getName()).matches());
32          assertNotNull(numericFiles);
33          for (File lwpidFile : numericFiles) {
34              int tid = ParseUtil.parseIntOrDefault(lwpidFile.getName(), 0);
35              SolarisLwpsInfo lwpsinfo = PsInfo.queryLwpsInfo(pid, tid);
36              assertThat("Thread ID in structure should match TID", lwpsinfo.pr_lwpid, is(tid));
37          }
38      }
39  }
