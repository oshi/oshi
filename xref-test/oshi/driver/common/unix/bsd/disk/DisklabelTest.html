32  /**
33   * Tests for the shared BSD {@link Disklabel} parser. The {@code parseDiskParams} and {@code parseDfFallback} methods
34   * are exercised with sample command output so they can be covered on any OS; live-system smoke tests are gated to
35   * OpenBSD/NetBSD.
36   */
37  class DisklabelTest {
38  
39      /** Returns deterministic (major,minor) without calling stat(1). */
40      private static final BiFunction<String, String, Pair<Integer, Integer>> ZERO_MAJOR_MINOR = (d, n) -> new Pair<>(0,
41              0);
42  
43      @Test
44      void testParseDiskParamsExtractsHeaderAndPartitions() {
45          // Representative output of `disklabel -n sd0` on OpenBSD/NetBSD.
46          List<String> disklabelOut = Arrays.asList("# /dev/rsd0c:", "type: SCSI", "disk: SCSI disk",
47                  "label: Storage Device", "duid: 0000000000000000", "flags:", "bytes/sector: 512", "sectors/track: 63",
48                  "tracks/cylinder: 255", "sectors/cylinder: 16065", "cylinders: 976", "total sectors: 15693824",
49                  "boundstart: 0", "boundend: 15693824", "drivedata: 0", "", "16 partitions:",
50                  "#                size           offset  fstype [fsize bsize   cpg]",
51                  "  a:          2097152             1024  4.2BSD   2048 16384 12958 # /",
52                  "  b:         17023368          2098176    swap                    # none",
53                  "  c:        500118192                0  unused", "  i:              960               64   MSDOS");
54  
55          Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDiskParams("sd0", disklabelOut,
56                  ZERO_MAJOR_MINOR);
57  
58          assertThat("label", result.getA(), is("Storage Device"));
59          assertThat("duid", result.getB(), is("0000000000000000"));
60          // total sectors x bytes/sector
61          assertThat("size in bytes", result.getC(), is(15693824L * 512L));
62          // Only `a` and `b` keep enough columns to be considered partitions. The single-letter rows for `c` (unused) and
63          // `i` (MSDOS) have <=4 whitespace fields and are deliberately filtered out by `split.length > 4`.
64          assertThat(result.getD(), hasSize(2));
65  
66          HWPartition root = result.getD().get(0);
67          assertThat(root.getIdentification(), is("sd0a"));
68          assertThat(root.getName(), is("a"));
69          assertThat(root.getType(), is("4.2BSD"));
70          assertThat(root.getUuid(), is("0000000000000000.a"));
71          assertThat(root.getSize(), is(2097152L * 512L));
72          assertThat(root.getMountPoint(), is("/"));
73  
74          HWPartition swap = result.getD().get(1);
75          assertThat(swap.getName(), is("b"));
76          assertThat(swap.getType(), is("swap"));
77          assertThat(swap.getMountPoint(), is("none"));
78      }

80      @Test
81      void testParseDiskParamsLargeDiskWithoutIntOverflow() {
82          // 5 TiB disk at 512-byte sectors = 10737418240 sectors, > Integer.MAX_VALUE.
83          long sectors = 10_737_418_240L;
84          List<String> disklabelOut = Arrays.asList("label: Big Disk", "duid: deadbeefdeadbeef", "bytes/sector: 512",
85                  "total sectors: " + sectors);
86  
87          Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDiskParams("sd1", disklabelOut,
88                  ZERO_MAJOR_MINOR);
89  
90          assertThat("size must not overflow", result.getC(), is(sectors * 512L));
91      }

93      @Test
94      void testParseDiskParamsEmptyInputReturnsDefaultsAndEmptyPartitions() {
95          Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDiskParams("sd0",
96                  Collections.emptyList(), ZERO_MAJOR_MINOR);
97  
98          assertThat(result.getA(), is(""));
99          assertThat(result.getB(), is(""));
100         assertThat(result.getC(), is(1L)); // default 1 sector x 1 byte
101         assertThat(result.getD(), is(empty()));
102     }

104     @Test
105     void testParseDiskParamsUsesMajorMinorLookup() {
106         List<String> disklabelOut = Arrays.asList("label: Test", "duid: aaaa", "bytes/sector: 512",
107                 "total sectors: 100", "  a:  100  0  4.2BSD  2048  16384 12958 # /");
108         BiFunction<String, String, Pair<Integer, Integer>> lookup = (disk, name) -> new Pair<>(7, 42);
109 
110         Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDiskParams("sd0", disklabelOut,
111                 lookup);
112 
113         assertThat(result.getD(), hasSize(1));
114         assertThat(result.getD().get(0).getMajor(), is(7));
115         assertThat(result.getD().get(0).getMinor(), is(42));
116     }

118     @Test
119     void testParseDfFallbackBuildsUnknownLabeledQuartet() {
120         // `df` columns: device 1k-blocks used avail capacity mounted
121         List<String> dfOut = Arrays.asList("Filesystem  1K-blocks    Used   Avail Capacity  Mounted on",
122                 "/dev/sd0a     1024000  500000  524000     49%  /",
123                 "/dev/sd0d     2048000 1000000 1048000     49%  /var",
124                 "tmpfs           65536       0   65536      0%  /tmp");
125 
126         Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDfFallback("sd0", dfOut,
127                 ZERO_MAJOR_MINOR);
128 
129         assertThat(result.getA(), is(Constants.UNKNOWN));
130         assertThat(result.getB(), is(Constants.UNKNOWN));
131         assertThat(result.getC(), is(0L));
132         assertThat(result.getD(), hasSize(2));
133         HWPartition rootPart = result.getD().get(0);
134         // df fallback names use the suffix after "/dev/" (e.g. "sd0a"), not just the partition letter.
135         assertThat(rootPart.getName(), is("sd0a"));
136         assertThat(rootPart.getMountPoint(), is("/"));
137         // 1024000 1K blocks x 512 (df fallback assumes 512-byte units)
138         assertThat(rootPart.getSize(), is(1024000L * 512L));
139     }

141     @Test
142     void testParseDfFallbackIgnoresOtherDisks() {
143         List<String> dfOut = Collections.singletonList("/dev/sd1a   1000   500   500   50%  /home");
144         Quartet<String, String, Long, List<HWPartition>> result = Disklabel.parseDfFallback("sd0", dfOut,

149     @Test
150     void testParseDiskParamsRowWithoutFstypeIsSkipped() {
151         // A partition row with fewer than 5 columns (no fstype) is discarded by the parser.
152         List<String> disklabelOut = Arrays.asList("label: X", "duid: x", "bytes/sector: 512", "total sectors: 10",

155                 ZERO_MAJOR_MINOR);
156         assertThat(result.getD(), is(empty()));

161     @Test
162     @EnabledOnOs(OS.OPENBSD)
163     void testGetDiskParamsLiveOpenBsd() {
164         String disknames = ExecutingCommand.getFirstAnswer("sysctl -n hw.disknames");
165         String[] devices = disknames.isEmpty() ? new String[0] : disknames.split(",");
166         boolean elevated = "0".equals(ExecutingCommand.getFirstAnswer("id -u"));
167         for (String device : devices) {
168             String diskName = device.split(":")[0];
169             Quartet<String, String, Long, List<HWPartition>> diskdata = Disklabel.getDiskParams(diskName);
170             if (elevated) {
171                 assertThat("Disk label is not null", diskdata.getA(), is(not(nullValue())));
172                 assertThat("Disk duid is not null", diskdata.getB(), is(not(nullValue())));
173                 assertThat("Disk size is nonnegative", diskdata.getC(), is(greaterThanOrEqualTo(0L)));
174                 for (HWPartition part : diskdata.getD()) {
175                     assertTrue(part.getIdentification().startsWith(diskName), "Partition ID starts with disk");
176                 }
177             }
178         }
179     }

181     @Test
182     @EnabledIfSystemProperty(named = "os.name", matches = "(?i)netbsd")
183     void testGetDiskParamsLiveNetBsd() {
184         String disknames = ExecutingCommand.getFirstAnswer("sysctl -n hw.disknames");
185         String[] devices = disknames.isEmpty() ? new String[0] : disknames.trim().split("\\s+");
186         boolean elevated = "0".equals(ExecutingCommand.getFirstAnswer("id -u"));
187         for (String device : devices) {
188             Quartet<String, String, Long, List<HWPartition>> diskdata = Disklabel.getDiskParams(device);
189             if (elevated) {
190                 assertThat("Disk label is not null", diskdata.getA(), is(not(nullValue())));
191                 assertThat("Disk duid is not null", diskdata.getB(), is(not(nullValue())));
192                 assertThat("Disk size is nonnegative", diskdata.getC(), is(greaterThanOrEqualTo(0L)));
193             }
194         }
