24   * Tests for {@link Systat} parsing methods. Each parse method is exercised with sample {@code systat -ab sensors}
25   * output drawn from real OpenBSD/NetBSD systems so the tests can run on any platform.

29      @Test
30      void testParseSensorsExtractsCpuTempVoltsAndFans() {
31          // CPU rows go to cpuTemps/volts; non-cpu temp rows go to allTemps; "fan" rows to fanRPMs.
32          List<String> systat = Arrays.asList("cpu0.temp0    55.00 degC", "cpu0.volt0    1.25 V DC",
33                  "acpitz0.temp0 40.00 degC", "acpitz1.temp0 42.00 degC", "it0.fan0    1500 RPM", "it0.fan1    2000 RPM");
34  
35          Triplet<Double, int[], Double> r = Systat.parseSensors(systat);
36          // cpuTemps non-empty so the average of cpu temps wins over allTemps.
37          assertThat(r.getA(), is(closeTo(55.0, 1e-9)));
38          assertThat("fan count", r.getB().length, is(2));
39          assertThat(r.getB()[0], is(1500));
40          assertThat(r.getB()[1], is(2000));
41          assertThat(r.getC(), is(closeTo(1.25, 1e-9)));
42      }

44      @Test
45      void testParseSensorsFallsBackToAllTempsWhenNoCpuTemp() {
46          List<String> systat = Arrays.asList("acpitz0.temp0 30.00 degC", "acpitz1.temp0 50.00 degC");
47          Triplet<Double, int[], Double> r = Systat.parseSensors(systat);
48          // Average of acpitz0 and acpitz1.
49          assertThat(r.getA(), is(closeTo(40.0, 1e-9)));
50          assertThat("no fans", r.getB().length, is(0));
51          assertThat(r.getC(), is(closeTo(0.0, 1e-9)));
52      }

57          Triplet<Double, int[], Double> r = Systat.parseSensors(systat);

61      @Test
62      void testParseSensorsIgnoresMalformedRowsAndEmptyInput() {
63          // Single-token rows (no value) and blank lines must be skipped without NaN propagation.
64          List<String> systat = Arrays.asList("", "cpu0.temp0", "junk", "cpu0.temp0 not-a-number");
65          Triplet<Double, int[], Double> r = Systat.parseSensors(systat);
66          // "not-a-number" parses to NaN and is excluded by listAverage; result is 0.
67          assertThat(r.getA(), is(closeTo(0.0, 1e-9)));
68          assertThat("no fans", r.getB().length, is(0));
69  
70          Triplet<Double, int[], Double> empty = Systat.parseSensors(Collections.<String>emptyList());
71          assertThat(empty.getA(), is(closeTo(0.0, 1e-9)));
72          assertThat("no fans on empty", empty.getB().length, is(0));
73          assertThat(empty.getC(), is(closeTo(0.0, 1e-9)));
74      }

77      void testParsePowerSourceNamesExtractsPrefixesAndDedupes() {

90      @Test
91      void testParseBatteryFieldsWatthour() {
92          List<String> systat = Arrays.asList("acpibat0.volt0 11.10 V DC", "acpibat0.current0 0.500 A",
93                  "acpibat0.temp0 30.0 degC", "acpibat0.watthour0 12.00 Wh (remaining capacity)",
94                  "acpibat0.watthour1 24.00 Wh (last full capacity)", "acpibat0.watthour2 36.00 Wh (design capacity)",
95                  "acpibat1.volt0 99.0 V DC"); // different name — ignored
96  
97          BatteryFields b = Systat.parseBatteryFields("acpibat0", systat);
98          assertThat(b.getVoltage(), is(closeTo(11.10, 1e-9)));
99          assertThat(b.getAmperage(), is(closeTo(0.500, 1e-9)));
100         assertThat(b.getTemperature(), is(closeTo(30.0, 1e-9)));
101         assertThat(b.getCapacityUnits(), is(CapacityUnits.MWH));
102         // x1000 conversion to mWh
103         assertThat(b.getCurrentCapacity(), is(12000));
104         assertThat(b.getMaxCapacity(), is(24000));
105         assertThat(b.getDesignCapacity(), is(36000));
106     }

108     @Test
109     void testParseBatteryFieldsAmphour() {
110         List<String> systat = Arrays.asList("acpibat0.amphour0 1.50 Ah (remaining capacity)",
111                 "acpibat0.amphour1 3.00 Ah (last full capacity)", "acpibat0.amphour2 4.00 Ah (new capacity)");
112 
113         BatteryFields b = Systat.parseBatteryFields("acpibat0", systat);
114         assertThat(b.getCapacityUnits(), is(CapacityUnits.MAH));
115         assertThat(b.getCurrentCapacity(), is(1500));
116         assertThat(b.getMaxCapacity(), is(3000));
117         assertThat(b.getDesignCapacity(), is(4000));
118     }

120     @Test
121     void testParseBatteryFieldsDefaultsWhenAbsent() {
122         BatteryFields b = Systat.parseBatteryFields("acpibat0", Collections.<String>emptyList());
123         assertThat(b.getVoltage(), is(closeTo(-1.0, 1e-9)));
124         assertThat(b.getAmperage(), is(closeTo(0.0, 1e-9)));
125         assertThat(b.getTemperature(), is(closeTo(0.0, 1e-9)));
126         assertThat(b.getCapacityUnits(), is(CapacityUnits.RELATIVE));
127         assertThat(b.getCurrentCapacity(), is(0));
128         assertThat(b.getMaxCapacity(), is(1));
129         assertThat(b.getDesignCapacity(), is(1));
130     }

133     void testParseBatteryFieldsVoltMatchesWhenRowDescribesCurrent() {
134         // Some sensors report "volt" without trailing 0 but include "current" in the line text.
135         List<String> systat = Collections.singletonList("acpibat0.volt 12.00 V (current voltage)");

140     @Test
141     void testListAverageSkipsNaNs() {
142         assertThat(Systat.listAverage(Arrays.asList(10.0, Double.NaN, 20.0)), is(closeTo(15.0, 1e-9)));

144         assertThat(Systat.listAverage(Collections.<Double>emptyList()), is(closeTo(0.0, 1e-9)));

147     @Test
148     void testParseSensorsFanZeroDefaultForNonNumeric() {
149         // Fan row with a non-numeric value should default the parsed RPM to 0 rather than throw.
150         List<String> systat = Arrays.asList("it0.fan0 broken RPM");
151         Triplet<Double, int[], Double> r = Systat.parseSensors(systat);
152         assertThat("one fan parsed", r.getB().length, is(1));
