7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.is;

11  import java.util.Arrays;

20  import oshi.hardware.HWPartition;
21  import oshi.util.tuples.Pair;

23  class LspvTest {
24  
25      @Test
26      void testParsePpSize() {
27          // lspv -L: an active volume with a 128 MB physical-partition size, returned in bytes
28          long ppSize = Lspv.parsePpSize(Arrays.asList(//
29                  "PHYSICAL VOLUME:    hdisk0                   VOLUME GROUP:     rootvg", //
30                  "PV STATE:           active", //
31                  "PP SIZE:            128 megabyte(s)          LOGICAL VOLUMES:  12", //
32                  "TOTAL PPs:          271 (34688 megabytes)    VG DESCRIPTORS:   2"));
33          assertThat(ppSize, is(128L << 20));
34      }
35  
36      @Test
37      void testParsePpSizeInactiveOrEmpty() {
38          // A non-active PV yields 0 (no partitions to enumerate)
39          assertThat(
40                  Lspv.parsePpSize(Arrays.asList("PV STATE:           missing", "PP SIZE:            128 megabyte(s)")),
41                  is(0L));
42          assertThat(Lspv.parsePpSize(Collections.emptyList()), is(0L));
43      }
44  
45      @Test
46      void testParsePartitions() {
47          List<String> lspvP = Arrays.asList(//
48                  "hdisk0:", //
49                  "PP RANGE  STATE   REGION        LV NAME             TYPE       MOUNT POINT", //
50                  "1-1     used    outer edge    hd5                 boot       N/A", //
51                  "2-55    free    outer edge", // free rows are skipped
52                  "56-59    used    outer middle  hd6                 paging     N/A", //
53                  "111-112   used    center        hd4                 jfs2       /");
54          Map<String, Pair<Integer, Integer>> majMin = Collections.singletonMap("hd5", new Pair<>(10, 5));
55          long ppSize = 128L << 20;
56          Map<String, HWPartition> byName = Lspv.parsePartitions(lspvP, ppSize, majMin).stream()
57                  .collect(Collectors.toMap(HWPartition::getName, Function.identity()));
58  
59          // hd5: 1 PP, type/mount from the row, major/minor from the supplied map
60          assertThat(byName.get("hd5").getType(), is("boot"));
61          assertThat(byName.get("hd5").getMountPoint(), is("")); // N/A is normalized to empty
62          assertThat(byName.get("hd5").getSize(), is(ppSize));
63          assertThat(byName.get("hd5").getMajor(), is(10));
64          assertThat(byName.get("hd5").getMinor(), is(5));
65          // hd6: 4 PPs (56-59); not in the map, so major/minor fall back to the first int in the name
66          assertThat(byName.get("hd6").getType(), is("paging"));
67          assertThat(byName.get("hd6").getSize(), is(ppSize * 4));
68          assertThat(byName.get("hd6").getMajor(), is(6));
69          // hd4: 2 PPs (111-112), mounted at /
70          assertThat(byName.get("hd4").getMountPoint(), is("/"));
71          assertThat(byName.get("hd4").getSize(), is(ppSize * 2));
72      }
73  
74      @Test
75      void testParsePartitionsEmpty() {
76          assertThat(Lspv.parsePartitions(Collections.emptyList(), 128L << 20, Collections.emptyMap()), is(empty()));
77      }
78  }
