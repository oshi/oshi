5   package oshi.driver.common.mac;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import org.junit.jupiter.api.Test;
11  
12  import oshi.driver.common.mac.ThreadInfo.ThreadStats;
13  import oshi.software.os.OSProcess.State;
14  
15  class ThreadInfoTest {
16  
17      @Test
18      void testThreadStatsStateSleeping() {
19          ThreadStats ts = new ThreadStats(0, 1.0, 'S', 100L, 200L, 31);
20          assertThat(ts.getState(), is(State.SLEEPING));
21          // Also test 'I' (idle/sleeping)
22          ThreadStats ti = new ThreadStats(1, 0.5, 'I', 50L, 100L, 20);
23          assertThat(ti.getState(), is(State.SLEEPING));
24      }
25  
26      @Test
27      void testThreadStatsStateRunning() {
28          ThreadStats ts = new ThreadStats(0, 50.0, 'R', 1000L, 2000L, 31);
29          assertThat(ts.getState(), is(State.RUNNING));
30      }
31  
32      @Test
33      void testThreadStatsStateWaiting() {
34          ThreadStats ts = new ThreadStats(0, 0.1, 'U', 10L, 20L, 15);
35          assertThat(ts.getState(), is(State.WAITING));
36      }
37  
38      @Test
39      void testThreadStatsStateZombie() {
40          ThreadStats ts = new ThreadStats(0, 0.0, 'Z', 0L, 0L, 0);
41          assertThat(ts.getState(), is(State.ZOMBIE));
42      }
43  
44      @Test
45      void testThreadStatsStateStopped() {
46          ThreadStats ts = new ThreadStats(0, 0.0, 'T', 500L, 1000L, 10);
47          assertThat(ts.getState(), is(State.STOPPED));
48      }
49  
50      @Test
51      void testThreadStatsStateOther() {
52          ThreadStats ts = new ThreadStats(0, 0.0, 'X', 0L, 0L, 0);
53          assertThat(ts.getState(), is(State.OTHER));
54      }
55  
56      @Test
57      void testThreadStatsAccessors() {
58          ThreadStats ts = new ThreadStats(5, 25.0, 'R', 3000L, 7000L, 47);
59          assertThat(ts.getThreadId(), is(5));
60          assertThat(ts.getUserTime(), is(7000L));
61          assertThat(ts.getSystemTime(), is(3000L));
62          assertThat(ts.getPriority(), is(47));
63          // upTime = (user+system) / (cpu/100 + 0.0005) = 10000 / 0.2505 ≈ 39920
64          assertThat(ts.getUpTime(), is((long) ((7000L + 3000L) / (25.0 / 100d + 0.0005))));
65      }
66  }
