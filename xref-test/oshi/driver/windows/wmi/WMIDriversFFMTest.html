29  @EnabledForJreRange(min = JRE.JAVA_25)
30  @EnabledOnOs(OS.WINDOWS)
31  class WMIDriversFFMTest {
32  
33      @Test
34      void testQueryWMIWin32Process() {
35          WmiResult<CommandLineProperty> cl = Win32ProcessFFM.queryCommandLines(null);
36          assertThat("Failed Win32Process.queryCommandLines", cl.getResultCount(), is(greaterThan(0)));
37          Set<Integer> clset = IntStream.range(0, cl.getResultCount())
38                  .map(i -> WmiUtil.getUint32(cl, CommandLineProperty.PROCESSID, i)).boxed().collect(Collectors.toSet());
39          assertThat("Failed Win32Process.queryProcesses", Win32ProcessFFM.queryProcesses(clset).getResultCount(),
40                  is(both(greaterThan(0)).and(lessThanOrEqualTo(clset.size()))));
41  
42          Win32ProcessCachedFFM cache = Win32ProcessCachedFFM.getInstance();
43          int processId = new SystemInfo().getOperatingSystem().getProcessId();
44          assertThat("Failed Win32Process.queryCommandLines for current process", clset.contains(processId), is(true));
45          String commandLine = cache.getCommandLine(processId, 0L);
46          assertThat("Failed Win32ProcessCached.getCommandLine", commandLine, is(notNullValue()));
47          assertThat("Failed Win32ProcessCached.getCommandLine cached lookup", cache.getCommandLine(processId, 0L),
48                  is(commandLine));
49      }
50  }
