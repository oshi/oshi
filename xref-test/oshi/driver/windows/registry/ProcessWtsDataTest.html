5   package oshi.driver.windows.registry;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.greaterThan;
9   import static org.hamcrest.Matchers.greaterThanOrEqualTo;
10  import static org.hamcrest.Matchers.is;
11  import static org.hamcrest.Matchers.notNullValue;
12  
13  import java.util.Collections;
14  import java.util.Map;
15  
16  import org.junit.jupiter.api.Test;
17  import org.junit.jupiter.api.condition.EnabledOnOs;
18  import org.junit.jupiter.api.condition.OS;
19  
20  import oshi.SystemInfo;
21  import oshi.driver.common.windows.registry.WtsInfo;
22  
23  @EnabledOnOs(OS.WINDOWS)
24  class ProcessWtsDataTest {
25  
26      @Test
27      void testQueryProcessWtsMapFromPerfMon() {
28          int pid = new SystemInfo().getOperatingSystem().getProcessId();
29          Map<Integer, WtsInfo> wtsMap = ProcessWtsData.queryProcessWtsMapFromPerfMon(Collections.singleton(pid));
30          assertThat("Process WTS map should include the current process", wtsMap.containsKey(pid), is(true));
31          assertWtsInfo(wtsMap.get(pid));
32      }
33  
34      private static void assertWtsInfo(WtsInfo info) {
35          assertThat("WTS info should not be null", info, is(notNullValue()));
36          assertThat("Process name should not be null", info.getName(), is(notNullValue()));
37          assertThat("Thread count should be positive", info.getThreadCount(), is(greaterThan(0)));
38          assertThat("Virtual size should be nonnegative", info.getVirtualSize(), is(greaterThanOrEqualTo(0L)));
39          assertThat("Kernel time should be nonnegative", info.getKernelTime(), is(greaterThanOrEqualTo(0L)));
40          assertThat("User time should be nonnegative", info.getUserTime(), is(greaterThanOrEqualTo(0L)));
41          assertThat("Open files should be nonnegative", info.getOpenFiles(), is(greaterThanOrEqualTo(0L)));
42      }
43  }
