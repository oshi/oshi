21  class XwininfoTest {
22  
23      // Fixture: xprop -root _NET_CLIENT_LIST_STACKING output
24      private static final List<String> STACKING_OUTPUT = Collections
25              .singletonList("_NET_CLIENT_LIST_STACKING(WINDOW): window id # 0x1400003, 0x1600003, 0x1800003");
26  
27      // Fixture: xwininfo -root -tree output with window entries
28      private static final List<String> WINDOW_TREE = Arrays.asList(
29              "xwininfo: Window id: 0x1e7 (the root window) (has no name)",
30              "  Root window id: 0x1e7 (the root window) (has no name)",
31              "     0x1400003 \"Terminal\": (\"gnome-terminal\" \"Gnome-terminal\")  800x600+0+0  +100+200",
32              "     0x1600003 \"Firefox\": (\"navigator\" \"Firefox\")  1024x768+0+0  +50+75",
33              "     0x1800003 (has no name): ()  640x480+0+0  +-10+30");
34  
35      @Test
36      void testParseZOrderValid() {
37          Map<String, Integer> zOrder = Xwininfo.parseZOrder(STACKING_OUTPUT);
38          assertThat(zOrder, aMapWithSize(3));
39          assertThat(zOrder.get("0x1400003"), is(1));
40          assertThat(zOrder.get("0x1600003"), is(2));
41          assertThat(zOrder.get("0x1800003"), is(3));
42      }
43  
44      @Test
45      void testParseZOrderEmpty() {
46          Map<String, Integer> zOrder = Xwininfo.parseZOrder(Collections.emptyList());
47          assertThat(zOrder, is(anEmptyMap()));
48      }
49  
50      @Test
51      void testParseZOrderNoHexId() {
52          List<String> noHex = Collections.singletonList("_NET_CLIENT_LIST_STACKING(WINDOW): window id # ");
53          Map<String, Integer> zOrder = Xwininfo.parseZOrder(noHex);
54          assertThat(zOrder, is(anEmptyMap()));
55      }
56  
57      @Test
58      void testParseWindowTree() {
59          Map<String, Integer> zOrderMap = new HashMap<>();
60          zOrderMap.put("0x1400003", 1);
61          zOrderMap.put("0x1600003", 2);
62          zOrderMap.put("0x1800003", 3);
63  
64          Map<String, String> windowNameMap = new HashMap<>();
65          Map<String, String> windowPathMap = new HashMap<>();
66          Map<String, Rectangle> windowMap = Xwininfo.parseWindowTree(WINDOW_TREE, false, zOrderMap, windowNameMap,
67                  windowPathMap);
68  
69          assertThat(windowMap, aMapWithSize(3));
70  
71          // Verify rectangle bounds (x, y, width, height)
72          Rectangle terminal = windowMap.get("0x1400003");
73          assertThat(terminal.x, is(100));
74          assertThat(terminal.y, is(200));
75          assertThat(terminal.width, is(800));
76          assertThat(terminal.height, is(600));
77  
78          Rectangle firefox = windowMap.get("0x1600003");
79          assertThat(firefox.x, is(50));
80          assertThat(firefox.y, is(75));
81          assertThat(firefox.width, is(1024));
82          assertThat(firefox.height, is(768));
83  
84          // Negative coordinates
85          Rectangle anonymous = windowMap.get("0x1800003");
86          assertThat(anonymous.x, is(-10));
87          assertThat(anonymous.y, is(30));
88          assertThat(anonymous.width, is(640));
89          assertThat(anonymous.height, is(480));
90  
91          // Verify window names
92          assertThat(windowNameMap.get("0x1400003"), is("Terminal"));
93          assertThat(windowNameMap.get("0x1600003"), is("Firefox"));
94          assertThat(windowNameMap.containsKey("0x1800003"), is(false));
95  
96          // Verify window paths
97          assertThat(windowPathMap.get("0x1400003"), is("gnome-terminal"));
98          assertThat(windowPathMap.get("0x1600003"), is("navigator"));
99          assertThat(windowPathMap.containsKey("0x1800003"), is(false));
100     }
101 
102     @Test
103     void testParseWindowTreeVisibleOnly() {
104         // Only 0x1400003 is in the z-order map
105         Map<String, Integer> zOrderMap = new HashMap<>();
106         zOrderMap.put("0x1400003", 1);
107 
108         Map<String, String> windowNameMap = new HashMap<>();
109         Map<String, String> windowPathMap = new HashMap<>();
110         Map<String, Rectangle> windowMap = Xwininfo.parseWindowTree(WINDOW_TREE, true, zOrderMap, windowNameMap,
111                 windowPathMap);
112 
113         // Only the window present in zOrderMap should be included
114         assertThat(windowMap, aMapWithSize(1));
115         assertThat(windowMap.containsKey("0x1400003"), is(true));
116         assertThat(windowMap.containsKey("0x1600003"), is(false));
117         assertThat(windowMap.containsKey("0x1800003"), is(false));
118     }
119 
120     @Test
121     void testParseWindowTreeAnonymousWindow() {
122         // A window line with no quoted name
123         List<String> tree = Collections.singletonList("     0x2000001 (has no name): ()  320x240+0+0  +0+0");
124         Map<String, Integer> zOrderMap = new HashMap<>();
125         zOrderMap.put("0x2000001", 1);
126 
127         Map<String, String> windowNameMap = new HashMap<>();
128         Map<String, String> windowPathMap = new HashMap<>();
129         Map<String, Rectangle> windowMap = Xwininfo.parseWindowTree(tree, false, zOrderMap, windowNameMap,
130                 windowPathMap);
131 
132         assertThat(windowMap, aMapWithSize(1));
133         Rectangle rect = windowMap.get("0x2000001");
134         assertThat(rect.x, is(0));
135         assertThat(rect.y, is(0));
136         assertThat(rect.width, is(320));
137         assertThat(rect.height, is(240));
138 
139         // No name or path should be recorded for anonymous windows
140         assertThat(windowNameMap.containsKey("0x2000001"), is(false));
141         assertThat(windowPathMap.containsKey("0x2000001"), is(false));
142     }
