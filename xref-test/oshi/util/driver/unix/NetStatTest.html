41      // Fixture: netstat -n output
42      private static final List<String> NETSTAT_N = Arrays.asList("Active Internet connections",
43              "Proto Recv-Q Send-Q  Local Address          Foreign Address        (state)",
44              "tcp4       0      0  192.168.1.5.55362      93.184.216.34.443      ESTABLISHED",
45              "udp4       0      0  192.168.1.5.5353       *.* ");

48      private static final List<String> TCP_STATS_LINUX = Arrays.asList("Tcp:", "    1234 active connection openings",

50              "    34 connection resets received", "    56 connections established", "    789012 segments received",
51              "    345678 segments sent out", "    901 segments retransmitted", "    0 bad segments received",

58      @Test
59      void testQueryTcpnetstat() {
60          Pair<Long, Long> result = NetStat.queryTcpnetstat(TCP_NETSTAT);
61          assertThat(result.getA(), is(2L));
62          assertThat(result.getB(), is(1L));
63      }

72      @Test
73      void testQueryNetstat() {
74          List<IPConnection> connections = NetStat.queryNetstat(NETSTAT_N);
75          assertThat(connections, hasSize(2));
76          assertThat(connections.get(0).getType(), is("tcp4"));
77          assertThat(connections.get(0).getState(), is(TcpState.ESTABLISHED));
78          assertThat(connections.get(0).getLocalPort(), is(55362));
79          assertThat(connections.get(0).getForeignPort(), is(443));
80          assertThat(connections.get(1).getType(), is("udp4"));
81          assertThat(connections.get(1).getState(), is(TcpState.NONE));
82          assertThat(connections.get(1).getLocalPort(), is(5353));
83      }
84  
85      @Test
86      void testQueryNetstatEmpty() {
87          assertThat(NetStat.queryNetstat(Collections.emptyList()), is(empty()));
88      }

90      @Test
91      void testQueryTcpStatsLinux() {
92          TcpStats stats = NetStat.queryTcpStats(TCP_STATS_LINUX);
93          assertThat(stats.getConnectionsEstablished(), is(56L));
94          assertThat(stats.getConnectionsActive(), is(1234L));
95          assertThat(stats.getConnectionsPassive(), is(5678L));
96          assertThat(stats.getConnectionFailures(), is(12L));
97          assertThat(stats.getConnectionsReset(), is(34L));
98          assertThat(stats.getSegmentsSent(), is(345678L));
99          assertThat(stats.getSegmentsReceived(), is(789012L));
100         assertThat(stats.getSegmentsRetransmitted(), is(901L));
101         assertThat(stats.getOutResets(), is(45L));
102         assertThat(stats.getInErrors(), is(0L));
103     }

132     @Nested
133     @DisabledOnOs(OS.WINDOWS)
134     class LiveTests {
135         // netstat -n output format varies across platforms; live parsing only verified on FreeBSD/OpenBSD
136         @DisabledOnOs({ OS.WINDOWS, OS.MAC, OS.LINUX })
137         @Test
138         void testQueryNetstat() {
139             for (IPConnection conn : NetStat.queryNetstat()) {
140                 assertThat("Connection is tcp or udp", conn.getType(), anyOf(startsWith("tcp"), startsWith("udp")));
141             }
142         }
143 
144         @EnabledOnOs({ OS.MAC, OS.FREEBSD })
145         @Test
146         void testQueryTcpNetstat() {
147             Pair<Long, Long> tcpConns = NetStat.queryTcpnetstat();
148             assertThat("ipv4 connections must be nonnegative", tcpConns.getA().intValue(), greaterThanOrEqualTo(0));
149             assertThat("ipv6 connections must be nonnegative", tcpConns.getB().intValue(), greaterThanOrEqualTo(0));
150         }
151 
152         @EnabledOnOs(OS.LINUX)
153         @Test
154         void testQueryStatsLinux() {
155             TcpStats tcpStats = NetStat.queryTcpStats("netstat -st4");
156             assertThat("tcp connections must be nonnegative", tcpStats.getConnectionsEstablished(),
157                     greaterThanOrEqualTo(0L));
158             UdpStats udp4Stats = NetStat.queryUdpStats("netstat -su4");
159             assertThat("udp4 datagrams sent must be nonnegative", udp4Stats.getDatagramsSent(),
160                     greaterThanOrEqualTo(0L));
161         }
162 
163         @EnabledOnOs(OS.OPENBSD)
164         @Test
165         void testQueryStatsOpenBSD() {
166             TcpStats tcpStats = NetStat.queryTcpStats("netstat -s -p tcp");
167             assertThat("tcp connections must be nonnegative", tcpStats.getConnectionsEstablished(),
168                     greaterThanOrEqualTo(0L));
169             UdpStats udpStats = NetStat.queryUdpStats("netstat -s -p udp");
170             assertThat("udp datagrams sent must be nonnegative", udpStats.getDatagramsSent(), greaterThanOrEqualTo(0L));
171         }
