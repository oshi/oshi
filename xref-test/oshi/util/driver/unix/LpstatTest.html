5   package oshi.util.driver.unix;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Arrays;
11  import java.util.Collections;
12  import java.util.List;
13  import java.util.Map;
14  
15  import org.junit.jupiter.api.Test;
16  
17  import oshi.hardware.Printer.PrinterStatus;
18  
19  class LpstatTest {
20  
21      // Fixture: lpstat -d
22      private static final List<String> LPSTAT_D = Arrays.asList("system default destination: HP_LaserJet");
23  
24      // Fixture: lpstat -v
25      private static final List<String> LPSTAT_V = Arrays.asList(
26              "device for HP_LaserJet: usb://HP/LaserJet%20Pro?serial=ABC123", "device for PDF_Printer: cups-pdf:/",
27              "device for Network_Printer: ipp://192.168.1.100/ipp/print");
28  
29      // Fixture: lpstat -l -p
30      private static final List<String> LPSTAT_LP = Arrays.asList(
31              "printer HP_LaserJet is idle.  enabled since Mon 01 Jan 2024 12:00:00 AM",
32              "\tDescription: HP LaserJet Pro MFP", "\tLocation: Office",
33              "printer PDF_Printer disabled since Tue 02 Jan 2024 01:00:00 AM -",
34              "\tDescription: CUPS-PDF Virtual Printer");
35  
36      // Fixture: lpoptions -p
37      private static final List<String> LPOPTIONS = Arrays
38              .asList("copies=1 device-uri=usb://HP/LaserJet printer-make-and-model='HP LaserJet Pro' printer-type=2");
39  
40      @Test
41      void testQueryDefaultPrinter() {
42          assertThat(Lpstat.queryDefaultPrinter(LPSTAT_D), is("HP_LaserJet"));
43      }
44  
45      @Test
46      void testQueryDefaultPrinterEmpty() {
47          assertThat(Lpstat.queryDefaultPrinter(Collections.emptyList()), is(""));
48      }
49  
50      @Test
51      void testQueryDefaultPrinterNoDefault() {
52          List<String> noDefault = Arrays.asList("no system default destination");
53          assertThat(Lpstat.queryDefaultPrinter(noDefault), is(""));
54      }
55  
56      @Test
57      void testQueryPortMap() {
58          Map<String, String> map = Lpstat.queryPortMap(LPSTAT_V);
59          assertThat(map.size(), is(3));
60          assertThat(map.get("HP_LaserJet"), is("usb://HP/LaserJet%20Pro?serial=ABC123"));
61          assertThat(map.get("PDF_Printer"), is("cups-pdf:/"));
62          assertThat(map.get("Network_Printer"), is("ipp://192.168.1.100/ipp/print"));
63      }
64  
65      @Test
66      void testQueryPortMapEmpty() {
67          assertThat(Lpstat.queryPortMap(Collections.emptyList()).isEmpty(), is(true));
68      }
69  
70      @Test
71      void testQueryDescriptionMap() {
72          Map<String, String> map = Lpstat.queryDescriptionMap(LPSTAT_LP);
73          assertThat(map.size(), is(2));
74          assertThat(map.get("HP_LaserJet"), is("HP LaserJet Pro MFP"));
75          assertThat(map.get("PDF_Printer"), is("CUPS-PDF Virtual Printer"));
76      }
77  
78      @Test
79      void testQueryDescriptionMapEmpty() {
80          assertThat(Lpstat.queryDescriptionMap(Collections.emptyList()).isEmpty(), is(true));
81      }
82  
83      @Test
84      void testQueryDriver() {
85          assertThat(Lpstat.queryDriver(LPOPTIONS), is("HP LaserJet Pro"));
86      }
87  
88      @Test
89      void testQueryDriverNotFound() {
90          assertThat(Lpstat.queryDriver(Arrays.asList("copies=1 device-uri=usb://HP")), is(""));
91      }
92  
93      @Test
94      void testQueryDriverEmpty() {
95          assertThat(Lpstat.queryDriver(Collections.emptyList()), is(""));
96      }
97  
98      @Test
99      void testParseStatusIdle() {
100         assertThat(Lpstat.parseStatus("printer HP_LaserJet is idle.  enabled since Mon 01 Jan"),
101                 is(PrinterStatus.IDLE));
102     }
103 
104     @Test
105     void testParseStatusDisabled() {
106         assertThat(Lpstat.parseStatus("printer PDF_Printer disabled since Tue 02 Jan"), is(PrinterStatus.OFFLINE));
107     }
108 
109     @Test
110     void testParseStatusPrinting() {
111         assertThat(Lpstat.parseStatus("printer HP_LaserJet now printing HP_LaserJet-42"), is(PrinterStatus.PRINTING));
112     }
113 
114     @Test
115     void testParseStatusError() {
116         assertThat(Lpstat.parseStatus("printer Broken_Printer has error condition"), is(PrinterStatus.ERROR));
117     }
118 
119     @Test
120     void testParseStatusUnknown() {
121         assertThat(Lpstat.parseStatus("printer Mystery_Printer something unexpected"), is(PrinterStatus.UNKNOWN));
122     }
123 
124     @Test
125     void testParseStatusReason() {
126         assertThat(Lpstat.parseStatusReason("printer HP disabled - paused"), is("paused"));
127     }
128 
129     @Test
130     void testParseStatusReasonNone() {
131         assertThat(Lpstat.parseStatusReason("printer HP is idle."), is(""));
132     }
133 
134     @Test
135     void testIsLocalUriUsb() {
136         assertThat(Lpstat.isLocalUri("usb://HP/LaserJet"), is(true));
137     }
138 
139     @Test
140     void testIsLocalUriDev() {
141         assertThat(Lpstat.isLocalUri("/dev/usb/lp0"), is(true));
142     }
143 
144     @Test
145     void testIsLocalUriNetwork() {
146         assertThat(Lpstat.isLocalUri("ipp://192.168.1.100/ipp/print"), is(false));
147     }
148 
149     @Test
150     void testIsLocalUriLocalLpd() {
151         assertThat(Lpstat.isLocalUri("lpd://localhost/queue"), is(true));
152     }
153 
154     @Test
155     void testIsLocalUriEmpty() {
156         assertThat(Lpstat.isLocalUri(""), is(false));
157     }
158 }
