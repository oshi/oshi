20  import org.junit.jupiter.api.Nested;

25  import oshi.TestConstants;
26  import oshi.util.tuples.Triplet;
27  
28  class LshwTest {
29  
30      // Fixture: lshw -C system output
31      private static final List<String> SYSTEM_OUTPUT = Arrays.asList("  *-system", "       description: Computer",
32              "       product: PowerEdge R720", "       vendor: Dell Inc.", "       serial: ABC1234",
33              "       width: 64 bits", "       uuid: 4C4C4544-0044-4810-8031-B4C04F333132");
34  
35      // Fixture: lshw -class processor output
36      private static final List<String> PROCESSOR_OUTPUT = Arrays.asList("  *-cpu",
37              "       product: Intel(R) Xeon(R) CPU E5-2670 v2 @ 2.50GHz", "       vendor: Intel Corp.",
38              "       capacity: 3300MHz", "       width: 64 bits");
39  
40      @Test
41      void testParseSystemInfo() {
42          Triplet<String, String, String> result = Lshw.parseSystemInfo(SYSTEM_OUTPUT);
43          assertThat(result.getA(), is("PowerEdge R720"));
44          assertThat(result.getB(), is("ABC1234"));
45          assertThat(result.getC(), is("4C4C4544-0044-4810-8031-B4C04F333132"));
46      }
47  
48      @Test
49      void testParseSystemInfoEmpty() {
50          Triplet<String, String, String> result = Lshw.parseSystemInfo(Collections.emptyList());
51          assertThat(result.getA(), is(nullValue()));
52          assertThat(result.getB(), is(nullValue()));
53          assertThat(result.getC(), is(nullValue()));
54      }
55  
56      @Test
57      void testParseSystemInfoPartial() {
58          List<String> partial = Arrays.asList("       product: MyServer");
59          Triplet<String, String, String> result = Lshw.parseSystemInfo(partial);
60          assertThat(result.getA(), is("MyServer"));
61          assertThat(result.getB(), is(nullValue()));
62          assertThat(result.getC(), is(nullValue()));
63      }
64  
65      @Test
66      void testQueryCpuCapacityParsing() {
67          assertThat(Lshw.queryCpuCapacity(PROCESSOR_OUTPUT), is(3_300_000_000L));
68      }
69  
70      @Test
71      void testQueryCpuCapacityEmpty() {
72          assertThat(Lshw.queryCpuCapacity(Collections.emptyList()), is(-1L));
73      }
74  
75      @Test
76      void testQueryCpuCapacityNoCapacityLine() {
77          List<String> noCapacity = Arrays.asList("  *-cpu", "       product: ARM Cortex-A72");
78          assertThat(Lshw.queryCpuCapacity(noCapacity), is(-1L));
79      }
80  
81      @Nested
82      @EnabledOnOs(OS.LINUX)
83      class LiveTests {
84          @Test
85          void testQueries() {
86              assertDoesNotThrow(Lshw::queryModel);
87              assertDoesNotThrow(Lshw::querySerialNumber);
88              String uuid = Lshw.queryUUID();
89              if (uuid != null) {
90                  assertThat("Test Lshw queryUUID", uuid, matchesRegex(TestConstants.UUID_REGEX));
91              }
92              assertThat("Test Lshw queryCpuCapacity", Lshw.queryCpuCapacity(), anyOf(greaterThan(0L), equalTo(-1L)));
93          }
94      }
95  }
