167     void testByteArrayToFloat() {
168         byte[] temp = { (byte) 0x12, (byte) 0x34, (byte) 0x56, (byte) 0x78, (byte) 0x9a };

173         f = 0x123 + (float) 0x4 / 0x10;
174         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 2, 4), Float.MIN_VALUE);

344      * Test parse string to array
345      */
346     @Test
347     void testParseStringToLongArray() {
348         int[] indices = { 1, 3 };
349         long now = System.currentTimeMillis();
350 
351         String foo = String.format("The numbers are %d %d %d %d", 123, 456, 789, now);
352         int count = ParseUtil.countStringToLongArray(foo, ' ');
353         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
354         long[] result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
355         assertThat("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(456L));
356         assertThat("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", result[1],
357                 is(now));
358 
359         foo = String.format("The numbers are %d %d %d %d %s", 123, 456, 789, now,
360                 "709af748-5f8e-41b3-b73a-b440ef4406c8");
361         count = ParseUtil.countStringToLongArray(foo, ' ');
362         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
363         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
364         assertThat("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(456L));
365         assertThat("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", result[1],
366                 is(now));
367 
368         foo = String.format("The numbers are %d -%d %d +%d", 123, 456, 789, now);
369         count = ParseUtil.countStringToLongArray(foo, ' ');
370         assertThat("countStringToLongArray should return 4 for \"" + foo + "\"", count, is(4));
371         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
372         assertThat("result[0] should be -4456 using parseStringToLongArray on \"" + foo + "\"", result[0], is(-456L));
373         assertThat("result[1] index should be 456 using parseStringToLongArray on \"" + foo + "\"", result[1], is(now));
374 
375         foo = String.format("NOLOG: Invalid character %d %s %d %d", 123, "4v6", 789, now);
376         count = ParseUtil.countStringToLongArray(foo, ' ');
377         assertThat("countStringToLongArray should return 2 for \"" + foo + "\"", count, is(2));
378         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
379         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));
380 
381         foo = String.format("Exceeds max long %d %d %d 1%d", 123, 456, 789, Long.MAX_VALUE);
382         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
383         assertThat("result[1] index should be " + Long.MAX_VALUE
384                 + " (Long.MAX_VALUE) using parseStringToLongArray on \"" + foo + "\"", result[1], is(Long.MAX_VALUE));
385 
386         foo = String.format("NOLOG: String too short %d %d %d %d", 123, 456, 789, now);
387         result = ParseUtil.parseStringToLongArray(foo, indices, 9, ' ');
388         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));
389 
390         foo = String.format("NOLOG: Array too short %d %d %d %d", 123, 456, 789, now);
391         result = ParseUtil.parseStringToLongArray(foo, indices, 2, ' ');
392         assertThat("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", result[1], is(0L));
393 
394         foo = String.format("%d %d %d %d", 123, 456, 789, now);
395         count = ParseUtil.countStringToLongArray(foo, ' ');
