151      */
152     @Test
153     public void testByteArrayToFloat() {
154         byte[] temp = { (byte) 0x12, (byte) 0x34, (byte) 0x56, (byte) 0x78, (byte) 0x9a };
155         float f = (temp[0] << 22 | temp[1] << 14 | temp[2] << 6 | temp[3] >>> 2) + (float) (temp[3] & 0x3) / 0x4;
156         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 4, 2), Float.MIN_VALUE);
157         f = 0x12345 + (float) 0x6 / 0x10;
158         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 3, 4), Float.MIN_VALUE);
159         f = 0x123 + (float) 0x4 / 0x10;
160         assertEquals(f, ParseUtil.byteArrayToFloat(temp, 2, 4), Float.MIN_VALUE);
161     }

320     /**
321      * Test parse string to array
322      */
323     @Test
324     public void testParseStringToLongArray() {
325         int[] indices = { 1, 3 };
326         long now = System.currentTimeMillis();
327 
328         String foo = String.format("The numbers are %d %d %d %d", 123, 456, 789, now);
329         int count = ParseUtil.countStringToLongArray(foo, ' ');
330         assertEquals(4, count);
331         long[] result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
332         assertEquals(456L, result[0]);
333         assertEquals(now, result[1]);
334 
335         foo = String.format("The numbers are %d %d %d %d %s", 123, 456, 789, now,
336                 "709af748-5f8e-41b3-b73a-b440ef4406c8");
337         count = ParseUtil.countStringToLongArray(foo, ' ');
338         assertEquals(4, count);
339         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
340         assertEquals(456L, result[0]);
341         assertEquals(now, result[1]);
342 
343         foo = String.format("The numbers are %d -%d %d +%d", 123, 456, 789, now);
344         count = ParseUtil.countStringToLongArray(foo, ' ');
345         assertEquals(4, count);
346         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
347         assertEquals(-456L, result[0]);
348         assertEquals(now, result[1]);
349 
350         foo = String.format("Invalid character %d %s %d %d", 123, "4v6", 789, now);
351         count = ParseUtil.countStringToLongArray(foo, ' ');
352         assertEquals(2, count);
353         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
354         assertEquals(0, result[1]);
355 
356         foo = String.format("Exceeds max long %d %d %d 1%d", 123, 456, 789, Long.MAX_VALUE);
357         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
358         assertEquals(Long.MAX_VALUE, result[1]);
359 
360         foo = String.format("String too short %d %d %d %d", 123, 456, 789, now);
361         result = ParseUtil.parseStringToLongArray(foo, indices, 9, ' ');
362         assertEquals(0, result[1]);
363 
364         foo = String.format("Array too short %d %d %d %d", 123, 456, 789, now);
365         result = ParseUtil.parseStringToLongArray(foo, indices, 2, ' ');
366         assertEquals(0, result[1]);
367 
368         foo = String.format("%d %d %d %d", 123, 456, 789, now);
369         count = ParseUtil.countStringToLongArray(foo, ' ');
370         assertEquals(4, count);
371 
372         foo = String.format("%d %d %d %d nonNumeric", 123, 456, 789, now);
373         count = ParseUtil.countStringToLongArray(foo, ' ');
374         assertEquals(4, count);
375 
376         foo = String.format("%d %d %d %d 123-456", 123, 456, 789, now);
377         count = ParseUtil.countStringToLongArray(foo, ' ');
378         assertEquals(4, count);
379     }
