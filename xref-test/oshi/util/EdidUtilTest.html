67      @Test
68      void testGetDescriptors() {
69          byte[][] descs = EdidUtil.getDescriptors(EDID);
70          for (int i = 0; i < 4; i++) {
71              int type = EdidUtil.getDescriptorType(descs[i]);
72              String timing = EdidUtil.getTimingDescriptor(descs[i]);
73              String range = EdidUtil.getDescriptorRangeLimits(descs[i]);
74              switch (i) {
75                  case 0:
76                      assertThat("desc 0 type", type, is(0x565E00A0));
77                      assertThat("desc 0 timing", timing, is("Clock 241MHz, Active Pixels 2560x1440 "));
78                      assertThat("desc 0 range", range,
79                              is("Field Rate -96-41 Hz vertical, 80-48 Hz horizontal, Max clock: 320 MHz"));
80                      break;
81                  case 1:
82                      assertThat("desc 1 type", type, is(0x1A1D0080));
83                      assertThat("desc 1 timing", timing, is("Clock 74MHz, Active Pixels 1280x720 "));
84                      assertThat("desc 1 range", range,
85                              is("Field Rate -48-28 Hz vertical, 32-64 Hz horizontal, Max clock: -1280 MHz"));
86                      break;
87                  case 2:
88                      assertThat("desc 2 type", type, is(0xFF));
89                      assertThat("desc 2 descriptorText", EdidUtil.getDescriptorText(descs[i]), is("C02JM2PFF2GC"));
90                      assertThat("desc 2 descriptorHex", ParseUtil.byteArrayToHexString(descs[i]), is(EDID_DESC3));
91                      break;
92                  case 3:
93                      assertThat("desc 3 type", type, is(0xFC));
94                      assertThat("desc 3 descriptorText", EdidUtil.getDescriptorText(descs[i]), is("Thunderbolt"));
95                      break;
96                  default:
97              }
98          }

124     @Test
125     void testFieldRoundTrip() {
126         // Synthesize an EDID from field values, then confirm each set* encoder inverts its get*/is* parser
127         byte[] edid = EdidUtil.newEdidTemplate();
128         EdidUtil.setManufacturerID(edid, "AUO");
129         EdidUtil.setProductID(edid, "9227");
130         EdidUtil.setSerialNo(edid, "162C0C25");
131         EdidUtil.setWeek(edid, (byte) 44);
132         EdidUtil.setYear(edid, 2012);
133         EdidUtil.setVersion(edid, "1.4");
134         EdidUtil.setDigital(edid, true);
135         EdidUtil.setHcm(edid, 60);
136         EdidUtil.setVcm(edid, 34);
137         EdidUtil.setPreferredResolution(edid, "2560x1440");
138         EdidUtil.setModel(edid, "Thunderbolt");
139         EdidUtil.setProductSerialNumber(edid, "C02JM2PFF2GC");
140         EdidUtil.updateChecksum(edid);
141 
142         assertThat("manufacturerId", EdidUtil.getManufacturerID(edid), is("AUO"));
143         assertThat("productId", EdidUtil.getProductID(edid), is("9227"));
144         assertThat("serialNo", EdidUtil.getSerialNo(edid), is("162C0C25"));
145         assertThat("week", EdidUtil.getWeek(edid), is((byte) 44));
146         assertThat("year", EdidUtil.getYear(edid), is(2012));
147         assertThat("version", EdidUtil.getVersion(edid), is("1.4"));
148         assertThat("digital", EdidUtil.isDigital(edid), is(true));
149         assertThat("hcm", EdidUtil.getHcm(edid), is(60));
150         assertThat("vcm", EdidUtil.getVcm(edid), is(34));
151         assertThat("preferredResolution", EdidUtil.getPreferredResolution(edid), is("2560x1440"));
152         assertThat("model", EdidUtil.getModel(edid), is("Thunderbolt"));
153         assertThat("productSerialNumber", EdidUtil.getProductSerialNumber(edid), is("C02JM2PFF2GC"));
154 
155         // A valid EDID's 128 bytes sum to zero modulo 256
156         int sum = 0;
157         for (byte b : edid) {
158             sum += b & 0xFF;
159         }
160         assertThat("checksum", sum % 256, is(0));
161     }

163     @Test
164     void testByteRoundTrip() {
165         // For injective fields, set*(template, get*(EDID)) reproduces the original EDID bytes exactly
166         byte[] edid = EdidUtil.newEdidTemplate();
167         EdidUtil.setManufacturerID(edid, EdidUtil.getManufacturerID(EDID));
168         EdidUtil.setProductID(edid, EdidUtil.getProductID(EDID));
169         EdidUtil.setSerialNo(edid, EdidUtil.getSerialNo(EDID));
170         EdidUtil.setWeek(edid, EdidUtil.getWeek(EDID));
171         EdidUtil.setYear(edid, EdidUtil.getYear(EDID));
172         EdidUtil.setVersion(edid, EdidUtil.getVersion(EDID));
173         EdidUtil.setHcm(edid, EdidUtil.getHcm(EDID));
174         EdidUtil.setVcm(edid, EdidUtil.getVcm(EDID));
175 
176         assertThat("manufacturer bytes 8-9", Arrays.copyOfRange(edid, 8, 10), is(Arrays.copyOfRange(EDID, 8, 10)));
177         assertThat("product bytes 10-11", Arrays.copyOfRange(edid, 10, 12), is(Arrays.copyOfRange(EDID, 10, 12)));
178         assertThat("serial bytes 12-15", Arrays.copyOfRange(edid, 12, 16), is(Arrays.copyOfRange(EDID, 12, 16)));
179         assertThat("week byte 16", edid[16], is(EDID[16]));
180         assertThat("year byte 17", edid[17], is(EDID[17]));
181         assertThat("version bytes 18-19", Arrays.copyOfRange(edid, 18, 20), is(Arrays.copyOfRange(EDID, 18, 20)));
182         assertThat("hcm byte 21", edid[21], is(EDID[21]));
183         assertThat("vcm byte 22", edid[22], is(EDID[22]));
184     }

186     @Test
187     void testSetDescriptors() {
188         byte[] edid = EdidUtil.newEdidTemplate();
189         // Text descriptors can be written into any of the four slots, including the fourth
190         EdidUtil.setDescriptorText(edid, 3, 0xFE, "Extra Text");
191         byte[][] descs = EdidUtil.getDescriptors(edid);
192         assertThat("slot 3 type", EdidUtil.getDescriptorType(descs[3]), is(0xFE));
193         assertThat("slot 3 text", EdidUtil.getDescriptorText(descs[3]), is("Extra Text"));
194 
195         // A raw 18-byte descriptor round-trips through getDescriptors
196         byte[] raw = descs[3];
197         byte[] edid2 = EdidUtil.newEdidTemplate();
198         EdidUtil.setDescriptor(edid2, 0, raw);
199         assertThat("raw descriptor", EdidUtil.getDescriptors(edid2)[0], is(raw));
200 
201         // Out-of-range slot and wrong-size descriptor are rejected
202         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setDescriptorText(edid, 4, 0xFE, "x"));
203         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setDescriptor(edid, 0, new byte[10]));
204     }

206     @Test
207     void testEncoderEdgeCases() {
208         byte[] edid = EdidUtil.newEdidTemplate();
209         // Version with no minor component defaults the minor to 0
210         EdidUtil.setVersion(edid, "2");
211         assertThat("major-only version", EdidUtil.getVersion(edid), is("2.0"));
212         // Analog (non-digital) display clears the digital bit
213         EdidUtil.setDigital(edid, false);
214         assertThat("analog", EdidUtil.isDigital(edid), is(false));
215         // High unsigned byte values round-trip (guards against signed-byte sign extension)
216         EdidUtil.setHcm(edid, 200);
217         assertThat("hcm 200", EdidUtil.getHcm(edid), is(200));
218         EdidUtil.setVcm(edid, 255);
219         assertThat("vcm 255", EdidUtil.getVcm(edid), is(255));
220         EdidUtil.setYear(edid, 2200);
221         assertThat("year 2200", EdidUtil.getYear(edid), is(2200));
222         // Descriptor text longer than 13 characters is truncated with no terminator/padding
223         EdidUtil.setDescriptorText(edid, 3, 0xFE, "ThisIsAVeryLongName");
224         assertThat("truncated descriptor", EdidUtil.getDescriptorText(EdidUtil.getDescriptors(edid)[3]),
225                 is("ThisIsAVeryLo"));
226         // No serial-number descriptor present returns an empty string (never null), like getSerialNo/getModel
227         assertThat("absent serial", EdidUtil.getProductSerialNumber(EdidUtil.newEdidTemplate()), is(""));
228     }

230     @Test
231     void testEncoderValidation() {
232         byte[] edid = EdidUtil.newEdidTemplate();
233         // Manufacturer ID must be exactly three uppercase A-Z letters
234         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setManufacturerID(edid, "AB"));
235         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setManufacturerID(edid, "auo"));
236         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setManufacturerID(edid, "A1B"));
237         // Product ID must be a hex value within 0-FFFF
238         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setProductID(edid, "10000"));
239         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setProductID(edid, "GG"));
240         // Year must be within 1990-2245
241         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setYear(edid, 1989));
242         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setYear(edid, 2246));
243         // Size in cm must be within 0-255
244         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setHcm(edid, 256));
245         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setHcm(edid, -1));
246         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setVcm(edid, 256));
247         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setVcm(edid, -1));
248         // A negative descriptor slot is rejected
249         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setDescriptor(edid, -1, new byte[18]));
250     }

252     @Test
253     void testSetSerialNo() {
254         // A 4-character all-alphanumeric serial round-trips
255         byte[] edid = EdidUtil.newEdidTemplate();
256         EdidUtil.setSerialNo(edid, "AB12");
257         assertThat("alphanumeric serial", EdidUtil.getSerialNo(edid), is("AB12"));
258         // Wrong length is rejected
259         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setSerialNo(edid, "ABC"));
260         // Hex-looking ASCII whose bytes are letters does not round-trip (getSerialNo would render "ABCD")
261         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setSerialNo(edid, "41424344"));
262         // Non-alphanumeric 4-character value does not round-trip
263         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setSerialNo(edid, "AB-1"));
264     }

266     @Test
267     void testSetSerialNoNumeric() {
268         // The numeric overload writes the raw 32-bit serial little-endian into bytes 12-15
269         byte[] edid = EdidUtil.newEdidTemplate();
270         EdidUtil.setSerialNo(edid, 0x162C0C25L);
271         assertThat("numeric serial", EdidUtil.getSerialNo(edid), is("162C0C25"));
272         // The full unsigned 32-bit range is accepted
273         EdidUtil.setSerialNo(edid, 0xFFFFFFFFL);
274         assertThat("max serial", EdidUtil.getSerialNo(edid), is("FFFFFFFF"));
275         // Values outside the unsigned 32-bit range are rejected rather than truncated
276         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setSerialNo(edid, -1L));
277         assertThrows(IllegalArgumentException.class, () -> EdidUtil.setSerialNo(edid, 0x1_0000_0000L));
278     }
