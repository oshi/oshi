8   import static org.hamcrest.Matchers.emptyString;
9   import static org.hamcrest.Matchers.greaterThanOrEqualTo;

13  import java.lang.management.ManagementFactory;

16  import java.util.List;

18  import org.junit.jupiter.api.Test;
19  import org.junit.jupiter.api.condition.EnabledIfSystemProperty;

21  import oshi.util.ParseUtil;

23  /**
24   * Tests for {@link FstatUtil}. Pure parsing methods are covered with sample {@code fstat} output; the integration test
25   * is gated to NetBSD.
26   */
27  class FstatUtilTest {
28  
29      @Test
30      void testParseCwdFromFstatFindsWdRow() {
31          // Sample NetBSD `fstat -p <pid>` output. The wd row's 5th column (index 4) is the working directory path.
32          List<String> fstat = Arrays.asList("USER     CMD          PID   FD MOUNT      INUM MODE         SZ|DV R/W",
33                  "root     bash        1234   wd /home/dan      12 drwxr-xr-x   512 r",
34                  "root     bash        1234    0 /dev          678 crw-rw-rw-  /dev/tty r",
35                  "root     bash        1234    1 /dev          678 crw-rw-rw-  /dev/tty w");
36  
37          assertThat(FstatUtil.parseCwdFromFstat(fstat), is("/home/dan"));
38      }
39  
40      @Test
41      void testParseCwdFromFstatNoWdRow() {
42          List<String> fstat = Arrays.asList("USER  CMD  PID  FD  MOUNT  INUM  MODE  SZ|DV  R/W",
43                  "root  bash 1234  0  /dev   678   crw   tty    r");
44          assertThat(FstatUtil.parseCwdFromFstat(fstat), is(emptyString()));
45      }
46  
47      @Test
48      void testParseCwdFromFstatEmptyInput() {
49          assertThat(FstatUtil.parseCwdFromFstat(Collections.<String>emptyList()), is(emptyString()));
50      }
51  
52      @Test
53      void testParseOpenFilesSubtractsHeader() {
54          List<String> fstat = Arrays.asList("USER  CMD  PID  FD  MOUNT  INUM  MODE  SZ|DV  R/W",
55                  "root  bash 1234  0  /dev   678   crw   tty    r", "root  bash 1234  1  /dev   678   crw   tty    w",
56                  "root  bash 1234  2  /dev   678   crw   tty    w");
57          // 4 lines - 1 header = 3 open files
58          assertThat(FstatUtil.parseOpenFiles(fstat), is(3L));
59      }
60  
61      @Test
62      void testParseOpenFilesHeaderOnlyOrEmpty() {
63          assertThat(FstatUtil.parseOpenFiles(Collections.singletonList("USER CMD PID FD ...")), is(0L));
64          assertThat(FstatUtil.parseOpenFiles(Collections.<String>emptyList()), is(0L));
65      }
66  
67      @Test
68      @EnabledIfSystemProperty(named = "os.name", matches = "(?i)netbsd")
69      void testFstatLive() {
70          // RuntimeMXBean#getName() returns "<pid>@<host>" — use it to avoid pulling oshi-core's SystemInfo into a
71          // oshi-common test. Mirrors the pattern used in the FreeBSD ProcstatUtilTest.
72          int pid = ParseUtil.parseIntOrDefault(ManagementFactory.getRuntimeMXBean().getName().split("@", 2)[0], -1);
73          assertThat("Number of open files must be nonnegative", FstatUtil.getOpenFiles(pid),
74                  is(greaterThanOrEqualTo(0L)));
75          assertThat("Cwd should not be empty", FstatUtil.getCwd(pid), is(not(emptyString())));
76      }
77  }
