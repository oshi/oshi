5   package oshi.util.common.platform.unix.openbsd;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   
10  import java.util.Arrays;
11  import java.util.Collections;
12  import java.util.List;
13  
14  import org.junit.jupiter.api.Test;
15  
16  /**
17   * Tests for {@link FstatUtil}. Pure parsing methods are covered with sample {@code fstat -sp} output.
18   */
19  class FstatUtilTest {
20  
21      @Test
22      void testParseOpenFilesCountsNonPipeAndNonUnixRows() {
23          List<String> fstat = Arrays.asList("USER CMD PID FD MOUNT INUM MODE RW SZ FLAGS XS",
24                  "dan bash 123 0 /     12345 crw r  64 0 0", "dan bash 123 1 /dev  67890 crw rw 64 0 0",
25                  "dan bash 123 2 /dev  67890 crw rw 64 0 0", "dan bash 123 3 pipe  0     fff rw 0  0 0",
26                  "dan bash 123 4 unix  0     fff rw 0  0 0");
27          assertThat(FstatUtil.parseOpenFiles(fstat), is(3L));
28      }
29  
30      @Test
31      void testParseOpenFilesHeaderOnly() {
32          assertThat(FstatUtil.parseOpenFiles(Collections.singletonList("USER CMD PID FD MOUNT INUM MODE RW SZ FL XS")),
33                  is(0L));
34      }
35  
36      @Test
37      void testParseOpenFilesEmpty() {
38          assertThat(FstatUtil.parseOpenFiles(Collections.<String>emptyList()), is(0L));
39      }
40  
41      @Test
42      void testParseOpenFilesPartialTokenMounts() {
43          List<String> fstat = Arrays.asList("USER CMD PID FD MOUNT INUM MODE RW SZ FLAGS XS",
44                  "dan bash 123 0 pipex 12345 crw r  64 0 0", "dan bash 123 1 unpipe 6789 crw rw 64 0 0",
45                  "dan bash 123 2 unixsocket 0 crw rw 64 0 0", "dan bash 123 3 pipe  0     fff rw 0  0 0",
46                  "dan bash 123 4 unix  0     fff rw 0  0 0");
47          // "pipex", "unpipe", "unixsocket" are NOT exact matches — they count as open files.
48          // "pipe" and "unix" are exact matches — excluded. Header subtracted.
49          assertThat(FstatUtil.parseOpenFiles(fstat), is(3L));
50      }
51  }
