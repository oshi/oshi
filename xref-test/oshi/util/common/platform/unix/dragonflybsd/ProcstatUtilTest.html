24  /**
25   * Tests for {@link ProcstatUtil}. Pure parsing methods are covered with sample {@code fstat} output; the integration
26   * test is gated to DragonFly BSD.
27   */
28  class ProcstatUtilTest {
29  
30      // DragonFly's fstat output positions the FD/type classifier ("wd", "root", "text", or a numeric fd) at the 5th
31      // whitespace-split token (index 4). The parser excludes "wd"/"root"/"text" classifiers when counting open files
32      // and uses the last token of the "wd" row as the working directory path. Sample rows here mimic that layout.
33  
34      @Test
35      void testParseCwdFromFstatFindsWdRow() {
36          List<String> fstat = Arrays.asList("USER     CMD       PID    NLWP   FD MOUNT INUM MODE NAME",
37                  "dan      bash      1234   1      wd /     45   drwx /home/dan",
38                  "dan      bash      1234   1      0  /dev  12   crw  /dev/tty");
39          assertThat(ProcstatUtil.parseCwdFromFstat(fstat), is("/home/dan"));
40      }
41  
42      @Test
43      void testParseCwdFromFstatNoWdRowReturnsEmpty() {
44          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
45                  "dan  bash 1234 1 0  /dev 12 crw /dev/tty");
46          assertThat(ProcstatUtil.parseCwdFromFstat(fstat), is(emptyString()));
47      }
48  
49      @Test
50      void testParseCwdFromFstatEmpty() {
51          assertThat(ProcstatUtil.parseCwdFromFstat(Collections.<String>emptyList()), is(emptyString()));
52      }
53  
54      @Test
55      void testParseOpenFilesExcludesWdRootTextAndHeader() {
56          // 5 rows total. The wd, root, and text rows are excluded; the header (token "FD" at index 4) is not excluded
57          // by the filter but is removed via the -1 adjustment at the end. With 2 countable rows minus the header
58          // adjustment, the result is 1.
59          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
60                  "dan  bash 1234 1 wd   /    45 drwx /home/dan", "dan  bash 1234 1 root /    1  drwx /",
61                  "dan  bash 1234 1 text /    99 -rxr /bin/bash", "dan  bash 1234 1 0    /dev 12 crw  /dev/tty",
62                  "dan  bash 1234 1 1    /dev 12 crw  /dev/tty");
63          assertThat(ProcstatUtil.parseOpenFiles(fstat), is(2L));
64      }
65  
66      @Test
67      void testParseOpenFilesNoCountableRowsReturnsZero() {
68          // Only header + excluded rows → no fd counted → returns 0 (per implementation: fd > 0 ? fd - 1 : 0)
69          List<String> fstat = Arrays.asList("USER CMD PID NLWP FD MOUNT INUM MODE NAME",
70                  "dan  bash 1234 1 wd /home 45 drwx /home/dan");
71          // Header row counts (split[4]="FD" not in {wd,root,text}), then the wd row is excluded.
72          // Count=1 (header) → returns 1-1=0.
73          assertThat(ProcstatUtil.parseOpenFiles(fstat), is(0L));
74      }
75  
76      @Test
77      void testParseOpenFilesEmpty() {
78          assertThat(ProcstatUtil.parseOpenFiles(Collections.<String>emptyList()), is(0L));
79      }
80  
81      @Test
82      @EnabledIfSystemProperty(named = "os.name", matches = "(?i)dragonfly")
83      void testProcstatLive() {
84          // RuntimeMXBean#getName() returns "<pid>@<host>" — use it to avoid pulling oshi-core's SystemInfo into a
85          // oshi-common test. Mirrors the pattern used in the FreeBSD ProcstatUtilTest and NetBSD FstatUtilTest.
86          int pid = ParseUtil.parseIntOrDefault(ManagementFactory.getRuntimeMXBean().getName().split("@", 2)[0], -1);
87          assertThat("RuntimeMXBean#getName() should yield a positive PID, got " + pid, pid, is(greaterThan(0)));
88          assertThat("Open files must be nonnegative", ProcstatUtil.getOpenFiles(pid), is(greaterThanOrEqualTo(0L)));
89          assertThat("Cwd should be nonempty", ProcstatUtil.getCwd(pid), is(not(emptyString())));
90      }
91  }
