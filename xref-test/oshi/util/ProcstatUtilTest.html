24  package oshi.util;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.anEmptyMap;
28  import static org.hamcrest.Matchers.emptyString;
29  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
30  import static org.hamcrest.Matchers.is;
31  import static org.hamcrest.Matchers.not;
32  
33  import org.junit.jupiter.api.Test;
34  
35  import oshi.PlatformEnum;
36  import oshi.SystemInfo;
37  import oshi.util.platform.unix.freebsd.ProcstatUtil;
38  
39  /**
40   * Test general utility methods
41   */
42  class ProcstatUtilTest {
43  
44      @Test
45      void testProcstat() {
46          if (SystemInfo.getCurrentPlatform().equals(PlatformEnum.FREEBSD)) {
47              int pid = new SystemInfo().getOperatingSystem().getProcessId();
48  
49              assertThat("Open files must be nonnegative", ProcstatUtil.getOpenFiles(pid), is(greaterThanOrEqualTo(0L)));
50  
51              assertThat("CwdMap should have at least one element", ProcstatUtil.getCwdMap(-1), is(not(anEmptyMap())));
52  
53              assertThat("CwdMap with pid should have at least one element", ProcstatUtil.getCwdMap(pid),
54                      is(not(anEmptyMap())));
55  
56              assertThat("Cwd should be nonempty", ProcstatUtil.getCwd(pid), is(not(emptyString())));
57          }
58      }
59  }
