8   import static org.hamcrest.Matchers.empty;
9   import static org.hamcrest.Matchers.greaterThan;
10  import static org.hamcrest.Matchers.is;
11  import static org.hamcrest.Matchers.not;
12  import static org.hamcrest.Matchers.notNullValue;
13  import static org.junit.jupiter.api.Assertions.assertTrue;

15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.TestInstance;
17  import org.junit.jupiter.api.TestInstance.Lifecycle;
18  import org.junit.jupiter.api.condition.EnabledForJreRange;
19  import org.junit.jupiter.api.condition.EnabledOnOs;
20  import org.junit.jupiter.api.condition.JRE;
21  import org.junit.jupiter.api.condition.OS;
22  
23  import oshi.software.os.OperatingSystem;
24  import oshi.software.os.mac.MacOperatingSystemFFM;
25  
26  @EnabledForJreRange(min = JRE.JAVA_25)
27  @TestInstance(Lifecycle.PER_CLASS)
28  public class OperatingSystemFFMTest {
29  
30      @Test
31      void testOperatingSystem() {
32          OperatingSystem os = new SystemInfo().getOperatingSystem();
33          assertThat("OS shouldn't be null", os, is(notNullValue()));
34          assertThat("OS family shouldn't be null", os.getFamily(), is(notNullValue()));
35          assertThat("OS should have 1 or more currently running processes", os.getProcessCount(), is(greaterThan(0)));
36          assertThat("OS should have at least 1 currently running process", os.getProcesses(null, null, 0),
37                  is(not(empty())));
38      }
39  
40      @EnabledOnOs(OS.MAC)
41      @Test
42      void testGetMacOperatingSystem() {
43          assertTrue(new SystemInfo().getOperatingSystem() instanceof MacOperatingSystemFFM);
44      }
45  }
