5   package oshi.ffm.util;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.hamcrest.Matchers.oneOf;
10  import static org.junit.jupiter.api.Assertions.assertEquals;
11  
12  import java.nio.ByteBuffer;
13  import java.nio.ByteOrder;
14  
15  import org.junit.jupiter.api.Test;
16  
17  import oshi.ffm.ForeignFunctions;
18  
19  class FileUtilFFMTest {
20  
21      @Test
22      void testNativeSizeConstants() {
23          assertThat("NATIVE_LONG_SIZE should be 4 or 8", ForeignFunctions.NATIVE_LONG_SIZE, is(oneOf(4L, 8L)));
24          assertThat("NATIVE_SIZE_T_SIZE should be 4 or 8", ForeignFunctions.NATIVE_SIZE_T_SIZE, is(oneOf(4L, 8L)));
25          assertThat("NATIVE_POINTER_SIZE should be 4 or 8", ForeignFunctions.NATIVE_POINTER_SIZE, is(oneOf(4L, 8L)));
26      }
27  
28      @Test
29      void testReadNativeLongFromBuffer() {
30          ByteBuffer buff = ByteBuffer.allocate(8).order(ByteOrder.nativeOrder());
31          buff.putLong(0x0000_0000_0000_002AL);
32          buff.flip();
33          long value = FileUtilFFM.readNativeLongFromBuffer(buff);
34          assertEquals(42L, value);
35      }
36  
37      @Test
38      void testReadSizeTFromBuffer() {
39          ByteBuffer buff = ByteBuffer.allocate(8).order(ByteOrder.nativeOrder());
40          buff.putLong(0x0000_0000_0000_0007L);
41          buff.flip();
42          long value = FileUtilFFM.readSizeTFromBuffer(buff);
43          assertEquals(7L, value);
44      }
45  
46      @Test
47      void testReadPointerFromBuffer() {
48          ByteBuffer buff = ByteBuffer.allocate(8).order(ByteOrder.nativeOrder());
49          buff.putLong(0x0000_0000_DEAD_BEEFL);
50          buff.flip();
51          long value = FileUtilFFM.readPointerFromBuffer(buff);
52          if (ForeignFunctions.NATIVE_POINTER_SIZE == 4) {
53              // 32-bit: reads 4 bytes as unsigned
54              assertEquals(0xDEAD_BEEFL, value);
55          } else {
56              assertEquals(0x0000_0000_DEAD_BEEFL, value);
57          }
58      }
59  
60      @Test
61      void testReadPointerFromEmptyBuffer() {
62          ByteBuffer buff = ByteBuffer.allocate(0);
63          assertEquals(0L, FileUtilFFM.readPointerFromBuffer(buff));
64      }
65  }
