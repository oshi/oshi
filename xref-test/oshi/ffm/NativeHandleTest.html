5   package oshi.ffm;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.is;
9   import static org.junit.jupiter.api.Assertions.assertThrows;
10  
11  import java.lang.foreign.MemorySegment;
12  import java.util.concurrent.atomic.AtomicBoolean;
13  import java.util.concurrent.atomic.AtomicInteger;
14  import java.util.concurrent.atomic.AtomicReference;
15  
16  import org.junit.jupiter.api.Test;
17  
18  class NativeHandleTest {
19  
20      @Test
21      void testCloseInvokesCloser() {
22          AtomicBoolean closed = new AtomicBoolean(false);
23          MemorySegment fakeHandle = MemorySegment.ofAddress(0x1234);
24          try (NativeHandle h = NativeHandle.of(fakeHandle, seg -> closed.set(true))) {
25              assertThat(h.get(), is(fakeHandle));
26              assertThat(h.isNull(), is(false));
27          }
28          assertThat(closed.get(), is(true));
29      }
30  
31      @Test
32      void testNullHandleSkipsCloser() {
33          AtomicBoolean closed = new AtomicBoolean(false);
34          try (NativeHandle h = NativeHandle.of(null, seg -> closed.set(true))) {
35              assertThat(h.isNull(), is(true));
36          }
37          assertThat(closed.get(), is(false));
38      }
39  
40      @Test
41      void testZeroAddressSkipsCloser() {
42          AtomicBoolean closed = new AtomicBoolean(false);
43          try (NativeHandle h = NativeHandle.of(MemorySegment.NULL, seg -> closed.set(true))) {
44              assertThat(h.isNull(), is(true));
45          }
46          assertThat(closed.get(), is(false));
47      }
48  
49      @Test
50      void testCloserExceptionIsSuppressed() {
51          AtomicReference<MemorySegment> closedWith = new AtomicReference<>();
52          MemorySegment fakeHandle = MemorySegment.ofAddress(0x5678);
53          try (NativeHandle h = NativeHandle.of(fakeHandle, seg -> {
54              closedWith.set(seg);
55              throw new RuntimeException("simulated close failure");
56          })) {
57              assertThat(h.isNull(), is(false));
58          }
59          // Should not throw, and closer was still called
60          assertThat(closedWith.get(), is(fakeHandle));
61      }
62  
63      @Test
64      void testCloseIsIdempotent() {
65          AtomicInteger closeCount = new AtomicInteger(0);
66          MemorySegment fakeHandle = MemorySegment.ofAddress(0xABCD);
67          NativeHandle h = NativeHandle.of(fakeHandle, seg -> closeCount.incrementAndGet());
68          h.close();
69          h.close();
70          h.close();
71          assertThat(closeCount.get(), is(1));
72      }
73  
74      @Test
75      void testNullCloserThrows() {
76          MemorySegment segment = MemorySegment.ofAddress(0x1);
77          assertThrows(NullPointerException.class, () -> NativeHandle.of(segment, null));
78      }
79  }
