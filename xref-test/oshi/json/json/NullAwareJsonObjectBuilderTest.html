24  package oshi.json.json;
25  
26  import static org.junit.Assert.assertEquals;
27  
28  import java.math.BigDecimal;
29  import java.math.BigInteger;
30  
31  import javax.json.Json;
32  import javax.json.JsonArrayBuilder;
33  import javax.json.JsonBuilderFactory;
34  import javax.json.JsonObject;
35  import javax.json.JsonObjectBuilder;
36  import javax.json.JsonValue;
37  
38  import org.junit.Test;
39  
40  public class NullAwareJsonObjectBuilderTest {
41  
42      @Test(expected = IllegalArgumentException.class)
43      public void testWrapNull() {
44          NullAwareJsonObjectBuilder.wrap(null);
45      }
46  
47      @Test
48      public void testBuilder() {
49          JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
50          JsonObjectBuilder json = NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder());
51  
52          json.addNull("null");
53          json.add("nullValue", JsonValue.NULL);
54          json.add("false", JsonValue.FALSE);
55  
56          String nullString = null;
57          json.add("nullString", nullString);
58          json.add("string", "String");
59  
60          BigInteger nullBigInt = null;
61          json.add("nullBigInt", nullBigInt);
62          json.add("big1", BigInteger.ONE);
63  
64          BigDecimal nullBigDec = null;
65          json.add("nullBigDec", nullBigDec);
66          json.add("big0.42", BigDecimal.valueOf(0.42));
67  
68          json.add("int", 42);
69          json.add("long", 4200L);
70          json.add("double", 42d);
71          json.add("boolean", true);
72          json.add("nan", Double.NaN);
73          json.add("inf", Double.POSITIVE_INFINITY);
74          json.add("-inf", Double.NEGATIVE_INFINITY);
75  
76          JsonArrayBuilder arrayBuilder = null;
77          json.add("nullArray", arrayBuilder);
78  
79          arrayBuilder = jsonFactory.createArrayBuilder();
80          JsonObjectBuilder jsonA = null;
81          json.add("nullBuilder", jsonA);
82  
83          jsonA = NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder());
84          JsonObject zero = null;
85          json.add("nullObject", zero);
86  
87          zero = jsonA.add("zero", 0).build();
88          arrayBuilder.add(zero);
89          json.add("zeroArray", arrayBuilder);
90          json.add("zeroObject", zero);
91  
92          JsonObject obj = json.build();
93  
94          assertEquals(JsonValue.NULL, obj.get("null"));
95          assertEquals(JsonValue.NULL, obj.get("nullValue"));
96          assertEquals(JsonValue.FALSE, obj.get("false"));
97  
98          assertEquals(JsonValue.NULL, obj.get("nullString"));
99          assertEquals(JsonValue.ValueType.STRING, obj.get("string").getValueType());
100         assertEquals("\"String\"", obj.get("string").toString());
101 
102         assertEquals(JsonValue.NULL, obj.get("nullBigInt"));
103         assertEquals(JsonValue.ValueType.NUMBER, obj.get("big1").getValueType());
104         assertEquals("1", obj.get("big1").toString());
105 
106         assertEquals(JsonValue.NULL, obj.get("nullBigDec"));
107         assertEquals(JsonValue.ValueType.NUMBER, obj.get("big0.42").getValueType());
108         assertEquals("0.42", obj.get("big0.42").toString());
109 
110         assertEquals(JsonValue.ValueType.NUMBER, obj.get("int").getValueType());
111         assertEquals("42", obj.get("int").toString());
112         assertEquals(JsonValue.ValueType.NUMBER, obj.get("long").getValueType());
113         assertEquals("4200", obj.get("long").toString());
114         assertEquals(JsonValue.ValueType.NUMBER, obj.get("double").getValueType());
115         assertEquals("42.0", obj.get("double").toString());
116         assertEquals(JsonValue.ValueType.TRUE, obj.get("boolean").getValueType());
117         assertEquals("true", obj.get("boolean").toString());
118 
119         assertEquals(JsonValue.ValueType.STRING, obj.get("nan").getValueType());
120         assertEquals("\"NaN\"", obj.get("nan").toString());
121         assertEquals(JsonValue.ValueType.STRING, obj.get("inf").getValueType());
122         assertEquals("\"Infinity\"", obj.get("inf").toString());
123         assertEquals(JsonValue.ValueType.STRING, obj.get("-inf").getValueType());
124         assertEquals("\"-Infinity\"", obj.get("-inf").toString());
125 
126         assertEquals(JsonValue.ValueType.ARRAY, obj.get("zeroArray").getValueType());
127         assertEquals("[{\"zero\":0}]", obj.get("zeroArray").toString());
128 
129         assertEquals(JsonValue.ValueType.OBJECT, obj.get("zeroObject").getValueType());
130         assertEquals("{\"zero\":0}", obj.get("zeroObject").toString());
131 
132         assertEquals(JsonValue.NULL, obj.get("nullArray"));
133         assertEquals(JsonValue.NULL, obj.get("nullObject"));
134         assertEquals(JsonValue.NULL, obj.get("nullBuilder"));
135     }
136 }
