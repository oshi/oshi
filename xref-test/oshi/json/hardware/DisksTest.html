19  package oshi.json.hardware;
20  
21  import static org.junit.Assert.assertEquals;
22  import static org.junit.Assert.assertNotEquals;
23  import static org.junit.Assert.assertNotNull;
24  import static org.junit.Assert.assertTrue;
25  
26  import java.io.IOException;
27  
28  import org.junit.Test;
29  
30  import oshi.json.SystemInfo;
31  
32  /**
33   * Test Disks
34   */
35  public class DisksTest {
36  
37      /**
38       * Test disks extraction.
39       *
40       * @throws IOException
41       *             Signals that an I/O exception has occurred.
42       */
43      @Test
44      public void testDisks() throws IOException {
45          long timeStamp = System.currentTimeMillis();
46          SystemInfo si = new SystemInfo();
47  
48          HWDiskStore lastDisk = new HWDiskStore();
49          for (HWDiskStore disk : si.getHardware().getDiskStores()) {
50              assertEquals(disk, disk);
51              assertNotEquals(disk, null);
52              assertNotEquals(disk, "A String");
53              assertNotEquals(disk, lastDisk);
54              assertNotEquals(disk.hashCode(), lastDisk.hashCode());
55              HWPartition[] parts = disk.getPartitions();
56              HWPartition[] partArray = new HWPartition[parts.length];
57              for (int i = 0; i < partArray.length; i++) {
58                  partArray[i] = new HWPartition();
59                  partArray[i].setIdentification(parts[i].getIdentification());
60                  partArray[i].setName(parts[i].getName());
61                  partArray[i].setType(parts[i].getType());
62                  partArray[i].setUuid(parts[i].getUuid());
63                  partArray[i].setMountPoint(parts[i].getMountPoint());
64                  partArray[i].setSize(parts[i].getSize());
65                  partArray[i].setMajor(parts[i].getMajor());
66                  partArray[i].setMinor(parts[i].getMinor());
67  
68              }
69  
70              assertNotNull(disk.getName());
71              assertNotNull(disk.getModel());
72              assertNotNull(disk.getSerial());
73              assertTrue(disk.getSize() >= 0);
74              assertTrue(disk.getReads() >= 0);
75              assertTrue(disk.getReadBytes() >= 0);
76              assertTrue(disk.getWrites() >= 0);
77              assertTrue(disk.getWriteBytes() >= 0);
78              assertTrue(disk.getTransferTime() >= 0);
79              assertTrue(disk.getTimeStamp() >= 0);
80  
81              long oldReads = disk.getReads();
82              long oldReadBytes = disk.getReadBytes();
83              assertTrue(disk.updateDiskStats());
84              assertTrue(disk.getReads() >= oldReads);
85              assertTrue(disk.getReadBytes() >= oldReadBytes);
86  
87              lastDisk = new HWDiskStore();
88              assertTrue(disk.compareTo(lastDisk) > 0);
89  
90              lastDisk.setModel(disk.getModel());
91              lastDisk.setName(disk.getName());
92              lastDisk.setSerial(disk.getSerial());
93              lastDisk.setSize(disk.getSize());
94              lastDisk.setPartitions(partArray);
95  
96              assertTrue(disk.equals(lastDisk));
97              assertEquals(disk.hashCode(), lastDisk.hashCode());
98  
99              lastDisk.setModel("model");
100             assertNotEquals(disk, lastDisk);
101             assertNotEquals(lastDisk, disk);
102             lastDisk.setModel(disk.getModel());
103             assertEquals(disk, lastDisk);
104             assertEquals(lastDisk, disk);
105 
106             lastDisk.setName("name");
107             assertNotEquals(disk, lastDisk);
108             assertNotEquals(lastDisk, disk);
109             lastDisk.setName(disk.getName());
110             assertEquals(disk, lastDisk);
111             assertEquals(lastDisk, disk);
112 
113             lastDisk.setSerial("serial");
114             assertNotEquals(lastDisk, disk);
115             assertNotEquals(disk, lastDisk);
116             lastDisk.setSerial(disk.getSerial());
117             assertEquals(disk, lastDisk);
118             assertEquals(lastDisk, disk);
119 
120             lastDisk.setSize(-1);
121             assertNotEquals(lastDisk, disk);
122             assertNotEquals(disk, lastDisk);
123             lastDisk.setSize(disk.getSize());
124             assertEquals(disk, lastDisk);
125             assertEquals(lastDisk, disk);
126 
127             disk.setName("name");
128             disk.setModel("model");
129             disk.setSerial("serial");
130             disk.setSize(123L);
131             disk.setReads(456L);
132             disk.setReadBytes(789L);
133             disk.setWrites(101112L);
134             disk.setWriteBytes(131415L);
135             disk.setTransferTime(161718L);
136             disk.setTimeStamp(timeStamp);
137 
138             assertEquals("name", disk.getName());
139             assertEquals("model", disk.getModel());
140             assertEquals("serial", disk.getSerial());
141             assertEquals(123L, disk.getSize());
142             assertEquals(456L, disk.getReads());
143             assertEquals(789L, disk.getReadBytes());
144             assertEquals(101112L, disk.getWrites());
145             assertEquals(131415L, disk.getWriteBytes());
146             assertEquals(161718L, disk.getTransferTime());
147             assertEquals(timeStamp, disk.getTimeStamp());
148 
149             for (HWPartition partition : disk.getPartitions()) {
150                 assertNotNull(partition.getIdentification());
151                 assertNotNull(partition.getName());
152                 assertNotNull(partition.getType());
153                 assertNotNull(partition.getUuid());
154                 assertNotNull(partition.getMountPoint());
155                 assertTrue(partition.getSize() >= 0);
156                 assertTrue(partition.getMajor() >= 0);
157                 assertTrue(partition.getMinor() >= 0);
158             }
159             HWPartition[] partitions = new HWPartition[2];
160             partitions[0] = new HWPartition();
161             partitions[0].setIdentification("id");
162             partitions[0].setName("name");
163             partitions[0].setType("type");
164             partitions[0].setUuid("uuid");
165             partitions[0].setMountPoint("mount");
166             partitions[0].setSize(123L);
167             partitions[0].setMajor(345);
168             partitions[0].setMinor(456);
169 
170             assertEquals(partitions[0], partitions[0]);
171             assertNotEquals(partitions[0], null);
172             assertNotEquals(partitions[0], "A string");
173 
174             partitions[1] = new HWPartition();
175             assertNotEquals(partitions[0], partitions[1]);
176             partitions[1].setIdentification(null);
177             assertNotEquals(partitions[0], partitions[1]);
178             assertNotEquals(partitions[1], partitions[0]);
179             partitions[1].setIdentification("");
180             assertNotEquals(partitions[0], partitions[1]);
181             assertTrue(partitions[0].compareTo(partitions[1]) > 0);
182             partitions[1].setIdentification("id");
183             assertNotEquals(partitions[0], partitions[1]);
184 
185             partitions[1].setMajor(345);
186             assertNotEquals(partitions[0], partitions[1]);
187             partitions[1].setMinor(456);
188             assertNotEquals(partitions[0], partitions[1]);
189 
190             partitions[1].setMountPoint(null);
191             assertNotEquals(partitions[0], partitions[1]);
192             assertNotEquals(partitions[1], partitions[0]);
193             partitions[1].setMountPoint("");
194             assertNotEquals(partitions[0], partitions[1]);
195             partitions[1].setMountPoint("mount");
196             assertNotEquals(partitions[0], partitions[1]);
197 
198             partitions[1].setName(null);
199             assertNotEquals(partitions[0], partitions[1]);
200             assertNotEquals(partitions[1], partitions[0]);
201             partitions[1].setName("");
202             assertNotEquals(partitions[0], partitions[1]);
203             partitions[1].setName("name");
204             assertNotEquals(partitions[0], partitions[1]);
205 
206             partitions[1].setSize(123L);
207             assertNotEquals(partitions[0], partitions[1]);
208 
209             partitions[1].setType(null);
210             assertNotEquals(partitions[0], partitions[1]);
211             assertNotEquals(partitions[1], partitions[0]);
212             partitions[1].setType("");
213             assertNotEquals(partitions[0], partitions[1]);
214             partitions[1].setType("type");
215             assertNotEquals(partitions[0], partitions[1]);
216 
217             partitions[1].setUuid(null);
218             assertNotEquals(partitions[0], partitions[1]);
219             assertNotEquals(partitions[1], partitions[0]);
220             partitions[1].setUuid("");
221             assertNotEquals(partitions[0], partitions[1]);
222             partitions[1].setUuid("uuid");
223             assertTrue(partitions[0].equals(partitions[1]));
224             assertEquals(partitions[0].hashCode(), partitions[1].hashCode());
225 
226             disk.setPartitions(partitions);
227             partitions = disk.getPartitions();
228             assertEquals("id", partitions[0].getIdentification());
229             assertEquals("name", partitions[0].getName());
230             assertEquals("type", partitions[0].getType());
231             assertEquals("uuid", partitions[0].getUuid());
232             assertEquals("mount", partitions[0].getMountPoint());
233             assertEquals(123L, partitions[0].getSize());
234             assertEquals(345, partitions[0].getMajor());
235         }
236     }
