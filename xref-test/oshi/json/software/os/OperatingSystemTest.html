46       * Test operating system
47       */
48      @Test
49      public void testOperatingSystem() {
50          SystemInfo si = new SystemInfo();
51          OperatingSystem os = si.getOperatingSystem();
52          assertNotNull(os.getFamily());
53          assertNotNull(os.getManufacturer());
54          OperatingSystemVersion version = os.getVersion();
55          assertNotNull(version);
56          assertNotNull(version.getVersion());
57          assertNotNull(version.getCodeName());
58          assertNotNull(version.getBuildNumber());
59  
60          assertTrue(os.getProcessCount() >= 1);
61          assertTrue(os.getThreadCount() >= 1);
62          assertTrue(os.getBitness() == 32 || os.getBitness() == 64);
63          assertTrue(os.getProcessId() > 0);
64  
65          assertTrue(os.getProcesses(0, null).length > 0);
66          OSProcess proc = os.getProcess(os.getProcessId());
67          assertTrue(proc.getName().length() > 0);
68          assertTrue(proc.getPath().length() > 0);
69          assertTrue(proc.getCommandLine().length() > 0);
70          assertNotNull(proc.getCurrentWorkingDirectory());
71          assertNotNull(proc.getUser());
72          assertNotNull(proc.getUserID());
73          assertNotNull(proc.getGroup());
74          assertNotNull(proc.getGroupID());
75          assertNotNull(proc.getState());
76          assertEquals(proc.getProcessID(), os.getProcessId());
77          assertTrue(proc.getParentProcessID() > 0);
78          assertTrue(proc.getThreadCount() > 0);
79          assertTrue(proc.getPriority() >= -20 && proc.getPriority() <= 128);
80          assertTrue(proc.getVirtualSize() >= 0);
81          assertTrue(proc.getResidentSetSize() >= 0);
82          assertTrue(proc.getKernelTime() >= 0);
83          assertTrue(proc.getUserTime() >= 0);
84          assertTrue(proc.getUpTime() >= 0);
85          assertTrue(proc.calculateCpuPercent() >= 0d);
86          assertTrue(proc.getStartTime() >= 0);
87          assertTrue(proc.getBytesRead() >= 0);
88          assertTrue(proc.getBytesWritten() >= 0);
89      }

91      /**
92       * Tests process query by pid list
93       */
94      @Test
95      public void testProcessQueryByList() {
96          SystemInfo si = new SystemInfo();
97          OperatingSystem os = si.getOperatingSystem();
98          assertNotNull(os.getFamily());
99          assertNotNull(os.getManufacturer());
100         OperatingSystemVersion version = os.getVersion();
101         assertNotNull(version);
102         assertNotNull(version.getVersion());
103         assertNotNull(version.getCodeName());
104         assertNotNull(version.getBuildNumber());
105 
106         assertTrue(os.getProcessCount() >= 1);
107         assertTrue(os.getThreadCount() >= 1);
108         assertTrue(os.getProcessId() > 0);
109 
110         OSProcess[] processes = os.getProcesses(5, null);
111         assertNotNull(processes);
112         // every OS should have at least one process running on it
113         assertTrue(processes.length > 0);
114         // the list of pids we want info on
115         List<Integer> pids = new ArrayList<>();
116         for (OSProcess p : processes) {
117             pids.add(p.getProcessID());
118         }
119         // query for just those processes
120         Collection<OSProcess> processes1 = os.getProcesses(pids);
121         // theres a potential for a race condition here, if a process we queried
122         // for initially wasn't running during the second query. In this case,
123         // try again with the shorter list
124         while (processes1.size() < pids.size()) {
125             pids.clear();
126             for (OSProcess p : processes1) {
127                 pids.add(p.getProcessID());
128             }
129             // query for just those processes
130             processes1 = os.getProcesses(pids);
131         }
132         assertEquals(processes1.size(), pids.size());
133     }

135     /**
136      * Tests child process getter
137      */
138     @Test
139     public void testGetChildProcesses() {
140         // Get list of PIDS
141         SystemInfo si = new SystemInfo();
142         OperatingSystem os = si.getOperatingSystem();
143         OSProcess[] processes = os.getProcesses(0, null);
144         Map<Integer, Integer> childMap = new HashMap<>();
145         // First iteration to set all 0's
146         for (OSProcess p : processes) {
147             childMap.put(p.getProcessID(), 0);
148             childMap.put(p.getParentProcessID(), 0);
149         }
150         // Second iteration to count children
151         for (OSProcess p : processes) {
152             childMap.put(p.getParentProcessID(), childMap.get(p.getParentProcessID()) + 1);
153         }
154         // Find a PID with 0, 1, and N>1 children
155         int zeroPid = -1;
156         int onePid = -1;
157         int nPid = -1;
158         int nNum = 0;
159         int mPid = -1;
160         int mNum = 0;
161         for (Integer i : childMap.keySet()) {
162             if (zeroPid < 0 && childMap.get(i) == 0) {
163                 zeroPid = i;
164             } else if (onePid < 0 && childMap.get(i) == 1) {
165                 onePid = i;
166             } else if (nPid < 0 && childMap.get(i) > 1) {
167                 // nPid is probably PID=1 with all PIDs with no other parent
168                 nPid = i;
169                 nNum = childMap.get(i);
170             } else if (mPid < 0 && childMap.get(i) > 1) {
171                 mPid = i;
172                 mNum = childMap.get(i);
173             }
174             if (zeroPid >= 0 && onePid >= 0 && nPid >= 0 && mPid >= 0) {
175                 break;
176             }
177         }
178         if (zeroPid >= 0) {
179             assertEquals(0, os.getChildProcesses(zeroPid, 0, null).length);
180         }
181         // Due to race condition, a process may terminate before we count its
182         // children.
183         if (onePid >= 0) {
184             assertTrue(0 <= os.getChildProcesses(onePid, 0, null).length);
185         }
186         if (nPid >= 0) {
187             assertTrue(0 <= os.getChildProcesses(nPid, 0, null).length);
188         }
189         if (mPid >= 0) {
190             assertTrue(0 <= os.getChildProcesses(mPid, 0, null).length);
191         }
192         // At least one of these tests should work.
193         if (onePid >= 0 && nPid >= 0 && mPid >= 0) {
194             assertTrue(os.getChildProcesses(onePid, 0, null).length == 1
195                     || os.getChildProcesses(nPid, 0, null).length == nNum
196                     || os.getChildProcesses(mPid, 0, null).length == mNum);
197         }
198     }

255     @Test
256     public void testGetCommandLine() {
257         int processesWithNonEmptyCmdLine = 0;
258 
259         SystemInfo si = new SystemInfo();
260         OperatingSystem os = si.getOperatingSystem();
261         for (OSProcess process : os.getProcesses(0, null)) {
262             if (!process.getCommandLine().trim().isEmpty()) {
263                 processesWithNonEmptyCmdLine++;
264             }
265         }
266 
267         assertTrue(processesWithNonEmptyCmdLine >= 1);
268     }
