259     @Test
260     void builderSelectiveRegistration() {
261         MeterRegistry selective = new SimpleMeterRegistry();
262         SystemInfoProvider si = new SystemInfo();
263         OshiMetrics.builder(si).enableCpu(true).enableMemory(false).enablePaging(false).enableDisk(false)
264                 .enableFileSystem(false).enableNetwork(false).enableGeneral(false).enableProcess(false)
265                 .enableContainer(false).build().bindTo(selective);
266         // CPU should be registered
267         assertNotNull(selective.find("system.cpu.time").functionCounter(), "CPU metrics should be registered");
268         // Memory should NOT be registered
269         assertNull(selective.find("system.memory.usage").gauge(), "Memory metrics should not be registered");
270         // Network should NOT be registered
271         assertNull(selective.find("system.network.io").functionCounter(), "Network metrics should not be registered");
272     }
