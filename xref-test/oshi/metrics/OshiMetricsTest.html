43      @Test
44      void systemUptimeRegistered() {
45          Gauge uptime = registry.find("system.uptime").gauge();
46          assertNotNull(uptime, "system.uptime should be registered");
47          // The DragonFly BSD CI VM has a known clock bug that reports a bogus (often future) boot time, which makes the
48          // derived uptime non-positive: https://bugs.dragonflybsd.org/issues/3299
49          if (!(PlatformEnum.getCurrentPlatform() == DRAGONFLYBSD && "true".equals(System.getenv("CI")))) {
50              assertTrue(uptime.value() > 0, "System uptime should be positive");
51          }
52      }

265     @Test
266     void builderSelectiveRegistration() {
267         MeterRegistry selective = new SimpleMeterRegistry();
268         SystemInfoProvider si = new SystemInfo();
269         OshiMetrics.builder(si).enableCpu(true).enableMemory(false).enablePaging(false).enableDisk(false)
270                 .enableFileSystem(false).enableNetwork(false).enableGeneral(false).enableProcess(false)
271                 .enableContainer(false).build().bindTo(selective);
272         // CPU should be registered
273         assertNotNull(selective.find("system.cpu.time").functionCounter(), "CPU metrics should be registered");
274         // Memory should NOT be registered
275         assertNull(selective.find("system.memory.usage").gauge(), "Memory metrics should not be registered");
276         // Network should NOT be registered
277         assertNull(selective.find("system.network.io").functionCounter(), "Network metrics should not be registered");
278     }
