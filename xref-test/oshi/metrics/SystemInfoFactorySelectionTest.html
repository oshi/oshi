5   package oshi.metrics;
6   
7   import static org.junit.jupiter.api.Assertions.assertEquals;
8   
9   import org.junit.jupiter.api.Test;
10  
11  import oshi.spi.SystemInfoFactory;
12  import oshi.spi.SystemInfoProvider;
13  
14  /**
15   * Validates that SystemInfoFactory selects the correct provider based on JDK version when both oshi-core and
16   * oshi-core-ffm are on the classpath.
17   */
18  class SystemInfoFactorySelectionTest {
19  
20      @Test
21      void factorySelectsCorrectProvider() {
22          SystemInfoProvider provider = SystemInfoFactory.create();
23          if (Runtime.version().feature() >= 25) {
24              assertEquals("oshi.ffm.SystemInfo", provider.getClass().getName(),
25                      "JDK 25+: Factory should select FFM provider");
26              assertEquals(20, provider.getPriority());
27          } else {
28              assertEquals("oshi.SystemInfo", provider.getClass().getName(),
29                      "JDK < 25: Factory should select JNA provider");
30              assertEquals(10, provider.getPriority());
31          }
32      }
33  }
