5   package oshi.demo;
6   
7   import java.util.List;
8   import java.util.Locale;
9   
10  import oshi.SystemInfo;
11  import oshi.annotation.SuppressForbidden;
12  import oshi.hardware.GraphicsCard;
13  import oshi.hardware.GpuStats;
14  import oshi.hardware.GpuTicks;
15  import oshi.util.FormatUtil;
16  
17  /**
18   * Demonstrates GPU metrics by polling all graphics cards 10 times at 1-second intervals using the session-based
19   * {@link GpuStats} API. One session per card is held open for the entire polling window so that session-scoped native
20   * state (e.g. IOReport power baselines on macOS) is preserved across iterations.
21   */
22  public final class PollGpuStats {
23  
24      private static final int ITERATIONS = 10;
25      private static final long INTERVAL_MS = 1_000L;
26  
27      private PollGpuStats() {
28      }
29  
30      @SuppressForbidden(reason = "Using System.out in a demo class")
31      public static void main(String[] args) throws InterruptedException {
32          List<GraphicsCard> cards = new SystemInfo().getHardware().getGraphicsCards();
33          if (cards.isEmpty()) {
34              System.out.println("No graphics cards found.");
35              return;
36          }
37  
38          for (GraphicsCard card : cards) {
39              System.out.printf(Locale.ROOT, "GPU   : %s%n", card.getName());
40              System.out.printf(Locale.ROOT, "Vendor: %s%n", card.getVendor());
41              System.out.printf(Locale.ROOT, "VRAM  : %s%n", FormatUtil.formatBytes(card.getVRam()));
42              System.out.println();
43          }
44  
45          int nameWidth = cards.stream().mapToInt(c -> c.getName().length()).max().orElse(10);
46          String hdr = String.format(Locale.ROOT, "%-" + nameWidth + "s  %8s  %10s  %10s  %8s  %8s  %10s", "Card",
47                  "Ticks%", "API Util%", "VRAM Used", "Temp(C)", "Power(W)", "Clock(MHz)");
48          System.out.println(hdr);
49          StringBuilder sep = new StringBuilder(hdr.length());
50          for (int s = 0; s < hdr.length(); s++)
51              sep.append('-');
52          System.out.println(sep.toString());
53  
54          // Open one session per card for the entire polling window
55          GpuStats[] sessions = new GpuStats[cards.size()];
56          GpuTicks[] prev = new GpuTicks[cards.size()];
57          try {
58              for (int i = 0; i < cards.size(); i++) {
59                  sessions[i] = cards.get(i).createStatsSession();
60                  prev[i] = sessions[i].getGpuTicks();
61                  sessions[i].getPowerDraw();
62                  sessions[i].getGpuUtilization();
63              }
64  
65              for (int iter = 0; iter < ITERATIONS; iter++) {
66                  Thread.sleep(INTERVAL_MS);
67                  for (int i = 0; i < cards.size(); i++) {
68                      GraphicsCard card = cards.get(i);
69                      GpuStats stats = sessions[i];
70                      GpuTicks curr = stats.getGpuTicks();
71                      long dActive = curr.getActiveTicks() - prev[i].getActiveTicks();
72                      long dIdle = curr.getIdleTicks() - prev[i].getIdleTicks();
73                      long dTotal = dActive + dIdle;
74                      String tickStr = dTotal > 0 ? String.format(Locale.ROOT, "%5.1f%%", dActive * 100.0 / dTotal)
75                              : "n/a";
76                      prev[i] = curr;
77  
78                      double apiUtil = stats.getGpuUtilization();
79                      long vramUsed = stats.getVramUsed();
80                      double temp = stats.getTemperature();
81                      double power = stats.getPowerDraw();
82                      long clock = stats.getCoreClockMhz();
83  
84                      System.out.printf(Locale.ROOT, "%-" + nameWidth + "s  %8s  %10s  %10s  %8s  %8s  %10s%n",
85                              card.getName(), tickStr,
86                              apiUtil >= 0 ? String.format(Locale.ROOT, "%.1f%%", apiUtil) : "n/a",
87                              vramUsed >= 0 ? FormatUtil.formatBytes(vramUsed) : "n/a",
88                              temp >= 0 ? String.format(Locale.ROOT, "%.1f", temp) : "n/a",
89                              power >= 0 ? String.format(Locale.ROOT, "%.2f", power) : "n/a",
90                              clock >= 0 ? Long.toString(clock) : "n/a");
91                  }
92                  if (cards.size() > 1) {
93                      System.out.println();
94                  }
95              }
96          } finally {
97              for (GpuStats s : sessions) {
98                  if (s != null) {
99                      try {
100                         s.close();
101                     } catch (Exception e) {
102                         // best-effort: attempt to close all sessions even if one throws
103                     }
104                 }
105             }
106         }
107     }
