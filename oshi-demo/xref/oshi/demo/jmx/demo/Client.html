5   package oshi.demo.jmx.demo;
6   
7   import java.io.IOException;
8   import java.util.List;
9   import java.util.Map;
10  
11  import javax.management.AttributeNotFoundException;
12  import javax.management.InstanceNotFoundException;
13  import javax.management.MBeanException;
14  import javax.management.MBeanServerConnection;
15  import javax.management.MBeanServerInvocationHandler;
16  import javax.management.MalformedObjectNameException;
17  import javax.management.ObjectName;
18  import javax.management.ReflectionException;
19  import javax.management.remote.JMXConnector;
20  import javax.management.remote.JMXConnectorFactory;
21  import javax.management.remote.JMXServiceURL;
22  
23  import oshi.annotation.SuppressForbidden;
24  
25  public class Client {
26      @SuppressForbidden(reason = "Using System.out in a demo class")
27      public static void main(String[] args) throws IOException, MalformedObjectNameException, ReflectionException,
28              InstanceNotFoundException, MBeanException, AttributeNotFoundException {
29  
30          // The address of the connector server
31          JMXServiceURL address = new JMXServiceURL("service:jmx:rmi:///jndi/rmi://localhost:8888/server");
32          // Map for custom properties key values
33          Map<String, ?> environment = null;
34          // Create the JMXCconnectorServer
35          JMXConnector cntor = JMXConnectorFactory.connect(address, environment);
36          // Obtain a "stub" for the remote MBeanServer
37          MBeanServerConnection mbsc = cntor.getMBeanServerConnection();
38  
39          // Here we can obtain all the domains and there would be one called oshi, meaning
40          // that is the one that we registered into our MbeanServer related with the
41          // baseboard
42          ObjectName objectQuery = new ObjectName("oshi:component=BaseBoard");
43          PropertiesAvailable mBean = MBeanServerInvocationHandler.newProxyInstance(mbsc, objectQuery,
44                  PropertiesAvailable.class, false);
45  
46          List<String> mBeanInfo = mBean.getProperties();
47  
48          System.out.println(mbsc.getAttribute(objectQuery, mBeanInfo.get(1)));
49          System.out.println(mbsc.getAttribute(objectQuery, mBeanInfo.get(2)));
50          System.out.println(mbsc.getAttribute(objectQuery, mBeanInfo.get(3)));
51          System.out.println(mbsc.getAttribute(objectQuery, mBeanInfo.get(4)));
52  
53      }
54  }
